מִשְׁנָה מַסֶּכֶת מִקְווֹת (קאופמן) מְנֻקֶּדֶת וּמְעֻצֶּבֶת עַל יְדֵי הָרַב דָּן בְּאֵרִי[עריכה]

הניקוד תוקן על פי כללי הדקדוק בנוגע למלא וחסר, ולפי הצליל לגבי קמץ ופתח, צרי וסגול, המתחלפים בכתב יד קאופמן מתוך הגיה ספרדית. שינויי מספור ההלכות צוינו.

שֶׁלַּצֶּמֶר וְשֶׁלַּשֵּׂעָר אֵינָן חוֹצְצִין,
מִפְּנֵי שֶׁהַמַּיִם בָּאִים בָּהֶן.

משנה ב
ב
קִלְקֵי הַלֵּב וְהַזָּקָן,
וּבֵית הַסְּתָרִים בָּאִשָּׁה,
לִפְלוּף שֶׁחוּץ לָעַיִן,
וְגֶלֶד שֶׁחוּץ לַמַּכָּה,
וְהָרְטִיָּה שֶׁעָלֶיהָ,
וְשֶׁרֶף הַיָּבֵשׁ,
וְגִלְדֵי צוֹאָה שֶׁעַל בְּשָׂרוֹ,
וּבָצֵק שֶׁעַל הַצִּפֹּרֶן,
וְהַמַּלְמוּלִין,
וְטִיט הַיָּוֵן,
וְטִיט הַיּוֹצְרִין,
וְגֵץ יְוָנִי.
אֵי זֶה הוּא טִיט הַיָּוֵן?
זֶה טִיט הַבּוֹרוֹת,
שֶׁנֶּאֱמַר: (תְּהִלִּים מ,ג)
"וַיַּעֲלֵנִי מִבּוֹר שָׁאוֹן מִטִּיט הַיָּוֵן". <ויקם על סלע רגלי>

ג
טִיט הַיּוֹצְרִין כִּשְׁמוּעוֹ.
רְבִּי יוֹסֵה מְטַהֵר בְּשֶׁלַּיּוֹצְרִין,
וּמְטַמֵּא בְשֶׁלִּמְרָקָה.
גֵץ יְוָנִי, אֵלּוּ יִתְדוֹת הַדְּרָכִין,
שֶׁאֵין טוֹבְלִין בָּהֶן וְלֹא מַטְבִּילִין אוֹתָן.
וּשְׁאָר כָּל הַטִּיט,
מַטְבִּילִין בּוֹ כְּשֶׁהוּא לַח.
לֹא יִטְבֹּל בָּאָבָק שֶׁעַל רַגְלָיו.
וְלֹא יִטְבֹּל אֶת הַקֻּמְקוּם בַּפֶּחָמִים,
אֶלָּא אִם שִׁפְשֵׁף.

משנה ג
ד
אֵלּוּ שֶׁאֵינָן חוֹצְצִין:
קִלְקֵי הָרֹאשׁ,
וּבֵית הַשֶּׁחִי,
וּבֵית הַסְּתָרִים בָּאִישׁ.
רְבִּי אֶלְעָזָר אוֹמֵר:
אֶחָד הָאִישׁ וְאֶחָד הָאִשָּׁה,
כָּל הַמַּקְפִּיד עָלָיו,
חוֹצֵץ,
וְשֶׁאֵינוּ מַקְפִּיד עָלָיו,
אֵינוּ חוֹצֵץ.

משנה ד
ה
לִפְלוּף שֶׁבָּעַיִן,
וְגֶלֶד שֶׁעַל הַמַּכָּה,
וְשֶׁרֶף הַלַּח,
וְלִכְלוּכֵי צוֹאָה שֶׁעַל בְּשָׂרוֹ,
וְצוֹאָה שֶׁתַּחַת הַצִּפֹּרֶן,
וְצִפֹּרֶן הַמְדֻלְדֶּלֶת,
כְּשׁוּת שֶׁלַּקָּטָן,
לֹא מִטַּמֵּא וְלֹא מְטַמֵּא.
קְרוּם שֶׁעַל הַמַּכָּה,
טָמֵא וּמְטַמֵּא.

משנה ה
ו
אֵלּוּ חוֹצְצִין בַּכֵּלִים:
הַזֶּפֶת וְהַמּוֹר בִּכְלֵי זְכוֹכִית,
בֵּין מִבִּפְנִים בֵּין מִבַּחוּץ,
עַל הַשֻּׁלְחָן וְעַל הַטַּבְלָה וְעַל הַדַּרְגֵּשׁ,
וְעַל הַנְּקִיוֹם,
חוֹצְצִין,
עַל הַבְּלוּסִים, אֵינָן חוֹצְצִין.
עַל מִטּוֹת בַּעַל הַבַּיִת, חוֹצֵץ,
וְעַל שֶׁלֶּעָנִי, אֵינוּ חוֹצֵץ.
עַל אֲכוֹף בַּעַל הַבַּיִת, חוֹצֵץ,
וְעַל שֶׁלְּזַקָּקִין, אֵינוּ חוֹצֵץ.
עַל הַמַּרְדַּעַת חוֹצֵץ.
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר:
עַד כָּאִסָּר הָאִיטַלְקִי.

משנה ו
ז
עַל הַבְּגָדִים מִצַּד אֶחָד, אֵינוּ חוֹצֵץ,
מִשְּׁנֵי צְדָדִין, חוֹצֵץ.
רְבִּי יְהוּדָה אוֹמֵר מִשֵּׁם רְבִּי יִשְׁמָעֵאל:
אַף מִצַּד אֶחָד.
רְבִּי יוֹסֵה אוֹמֵר:
שֶׁל בַּנְּוִים, מִצַּד אֶחָד,
וְשֶׁלַּבּוּר, מִשְּׁנֵי צְדָדִין.

משנה ז
ח
מִטְפַּחַת שֶׁלְּזַפָּתִין וְשֶׁלַּיּוֹצְרִין וְשֶׁלִּמְפַסְּלֵי אִילָנוֹת,
אֵינָן חוֹצְצִין.
רְבִּי יְהוּדָה אוֹמֵר:
אַף שֶׁלְּקַיָּצִים כַּיּוֹצֵא בָהֶן.
זֶה הַכְּלָל:
כָּל הַמַּקְפִּיד עָלָיו, חוֹצֵץ,
וְשֶׁאֵינוּ מַקְפִּיד עָלָיו, אֵינוּ חוֹצֵץ.

מַסֶּכֶת מִקְווֹת פֶּרֶק עֲשִׂירִי[עריכה]

משנה א
כָּל יְדוֹת הַכֵּלִים שֶׁהִכְנִיסָן שֶׁלֹּא כְדַרְכָּן,
אוֹ שֶׁהִכְנִיסָן כְּדַרְכָּן וְלֹא מֵרְקָן,
אוֹ שֶׁמֵּרְקָן וְנִשְׁבָּרוּ,
הֲרֵי אֵלּוּ חוֹצְצִין.
כֶּלִי שֶׁהִטְבִּילוֹ דֶרֶךְ פִּיו,
כְּאִלּוּ לֹא טָבַל.
הִטְבִּילוֹ כְדַרְכּוֹ וְלֹא הַזִּבּוֹרִית,
עַד שֶׁיַּטֶּנּוּ עַל צִדּוֹ.
כֶּלִי שֶׁהוּא צַר מִכָּן וּמִכָּן וְרָחָב בָּאֶמְצַע,
אֵינוּ טָהוֹר, עַד שֶׁיַּטֶּנּוּ עַל צִדּוֹ.
צְלוֹחִית שֶׁפִּיהָ שׁוֹקֵעַ,
אֵינָה טְהוֹרָה, עַד שֶׁיִּקְּבֶנָּה מִצִּדָּהּ.
קַלְמָרִין,
אֵינָה טְהוֹרָה, עַד שֶׁיִּקְּבֶנָּה מִצִּדָּהּ.
קַלְמָרִין שֶׁלְּיוֹסֵף הַכֹּהֵן הָיְתָה נְקוּבָה בְצִדָּהּ.

משנה ב
ב
הַכַּר וְהַכֶּסֶת שֶׁלָּעוֹר,
הֲרֵי אֵלּוּ צְרִיכִין שֶׁיָּבוֹאוּ בָהֶן הַמַּיִם.
כֶּסֶת עֲגֻלָּה, הַכַּדּוּר, וְהָאֵמוֹן,
וְהַקָּמֵיעַ, וּתְפִלָּה,
אֵינָן צְרִיכִין שֶׁיָּבוֹאוּ בָהֶם הַמַּיִם.
זֶה הַכְּלָל:
כָּל שֶׁאֵין דַּרְכּוֹ לְהַכְנִיס וּלְהוֹצִיא,
טוֹבְלִין סְתוּמִין.

משנה ג
ג
אֵלּוּ שֶׁאֵינָן צְרִיכִין שֶׁיָּבוֹאוּ בָהֶם הַמַּיִם:
קִשְׁרֵי הֶעָנִי,
וְהַנּוּמִין,
וַחֲבַט שֶׁלַּסַּנְדָּל,
וּתְפִלָּה שֶׁלָּרֹאשׁ בִּזְמַן שֶׁהִיא חוּצָה,
וְשֶׁלַּזְּרוֹעַ בִּזְמַן שֶׁהִיא עוֹלָה וְיוֹרֶדֶת,
וְאָזְנֵי הַחֵמֶת,
וְאָזְנֵי הַתָּרְמֵל.

משנה ד
ד
אֵלּוּ שֶׁהֵן צְרִיכִין שֶׁיָּבוֹאוּ בָהֶם הַמַּיִם:
הַקֶּשֶׁר שֶׁבַּפִּרְקַרְסִין שֶׁבַּכָּתֵף, <מין גופיה שקושרים על הכתף>
וְשָׂפָה שֶׁלַּסָּדִין,
צָרִיךְ לִמְתַּח,
וּתְפִלָּה שֶׁלָּרֹאשׁ בִּזְמַן שֶׁאֵינָה חוּצָה,
וְשֶׁלַּזְּרוֹעַ בִּזְמַן שֶׁאֵינָה עוֹלָה וְיוֹרֶדֶת,
וְשֶׁנֶץ שֶׁלַּסַּנְדָּל,
וּבְגָדִים שֶׁהִטְבִּילָן מְכֻבָּסִין,
עַד שֶׁיְּבַעְבֵּעוּ. <שֶׁיְּבַעְבִּיעוּ!>
הִטְבִּילָן נְגוּבִים,
עַד שֶׁיְּבַעְבֵּעוּ, וְיָנוּחוּ מִבִּעְבּוּעָן.

משנה ה
ה
כָּל יְדוֹת הַכֵּלִים שֶׁהֵן אֲרֻכִּין וְעָתִיד לְקוֹצָץ,
מַטְבִּילָן עַד מְקוֹם הַמִּדָּה.
רְבִּי יְהוּדָה אוֹמֵר:
עַד שֶׁיַּטְבִּיל אֶת כֻּלּוֹ.
שַׁלְשֶׁלֶת דְּלִי גָדוֹל, אַרְבָּעָה טְפָחִים,
וְשֶׁלַּקָּטָן, עֲשָׂרָה,
מַטְבִּילָן עַד מְקוֹם הַמִּדָּה.
רְבִּי טַרְפוֹן אוֹמֵר:
עַד שֶׁיַּטְבִּיל אֶת כָּל הַטַּבַּעַת.
הַחֶבֶל שֶׁהוּא קָשׁוּר בַּקֻּפָּה,
אֵינוּ חִבּוּר, אֶלָּא אִם תָּפַר. <אֵינָה!>

משנה ו
ו
בֵּית שַׁמַּי אוֹמְרִים:
אֵין מַטְבִּילִין חַמִּים בַּצּוֹנִין,
וְלֹא צוֹנִין בַּחַמִּים,
לֹא יָפִים בָּרָעִים,
וְלֹא רָעִים בַּיָּפִים.
וּבֵית הֶלֵּל אוֹמְרִים:
מַטְבִּילִין.
כֶּלִי שֶׁהוּא מָלֵא מַשְׁקִים,
הִטְבִּילוֹ, כְּאִלּוּ לֹא טָבַל.
מָלֵא מֵי רַגְלַיִם,
רוֹאִין אוֹתָן כְּאִלּוּ הֵן מַיִם.
מָלֵא מֵי חַטָּאת,
עַד שֶׁיִּרְבּוּ הַמַּיִם עַל מֵי חַטָּאת.
רְבִּי יוֹסֵה אוֹמֵר:
אֲפִלּוּ כֶלִי מַחֲזִיק כּוֹר,
וְאֵין בּוֹ אֶלָּא רְבִיעִית,
כְּאִלּוּ לֹא טָבַל.

משנה ז
ז
כָּל הָאֳכָלִין מִצְטָרְפִין לִפְסוֹל אֶת הַגְּוִיָּה בְכַחֲצִי פְרָס.
כָּל הַמַּשְׁקִים מִצְטָרְפִין לִפְסוֹל אֶת הַגְּוִיָּה בָרְבִיעִית.
זֶה חֹמֶר בְּשׁוֹתֶה מַשְׁקִים טְמֵאִים מִבַּמִּקְוֶה,
שֶׁעָשָׁה בוֹ שְׁאָר מַשְׁקִין כַּמַּיִם.

משנה ח
ח
אָכַל אֳכָלִין טְמֵאִין,
וְשָׁתָה מַשְׁקִין טְמֵאִין,
טָבַל וֶהֱקִיאָן,
טְמֵאִין, מִפְּנֵי שֶׁאֵינָן טְהוֹרִין בַּגּוּף.
שָׁתָה מַיִם טְמֵאִין,
טָבַל וֶהֱקִיאָן,
טְהוֹרִין, מִפְּנֵי שֶׁהֵן טְהוֹרִין בַּגּוּף.
בָּלַע טַבַּעַת טְהוֹרָה,
וְנִכְנַס לְאֹהֶל הַמֵּת,
הִזָּה וְשָׁנָה וְטָבַל, הֱקִיאָהּ,
הֲרֵי הִיא כְמוֹת שֶׁהָיָתָה.
בָּלַע טַבַּעַת טְמֵאָה,
טוֹבֵל וְאוֹכֵל בִּתְרוּמָתוֹ;
הֱקִיאָהּ,
טְמֵאָה, וְטִמַּתּוּ.
חֵץ שֶׁהוּא תָּחוּב בָּאָדָם,
בִּזְמַן שֶׁהוּא נִרְאֶה, חוֹצֵץ.
אִם אֵינוּ נִרְאֶה, אֵינוּ חוֹצֵץ,
טוֹבֵל וְאוֹכֵל בִּתְרוּמָתוֹ.
