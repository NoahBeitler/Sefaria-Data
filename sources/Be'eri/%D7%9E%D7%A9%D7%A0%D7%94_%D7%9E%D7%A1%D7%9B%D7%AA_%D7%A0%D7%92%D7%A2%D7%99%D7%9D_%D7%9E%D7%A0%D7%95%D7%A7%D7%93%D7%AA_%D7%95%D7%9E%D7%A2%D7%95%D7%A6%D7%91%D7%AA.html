מִשְׁנָה מַסֶּכֶת נְגָעִים (קאופמן) מְנֻקֶּדֶת וּמְעֻצֶּבֶת עַל יְדֵי הָרַב דָּן בְּאֵרִי[עריכה]

הניקוד תוקן על פי כללי הדקדוק בנוגע למלא וחסר, ולפי הצליל לגבי קמץ ופתח, צרי וסגול, המתחלפים בכתב יד קאופמן מתוך הגיה ספרדית. שינויי מספור ההלכות צוינו.

מַסֶּכֶת נְגָעִים פֶּרֶק רִאשׁוֹן[עריכה]

משנה א
מַרְאוֹת נְגָעִים שְׁנַיִם שֶׁהֵן אַרְבָּעָה:
בַּהֶרֶת עַזָּה כַשֶּׁלֶג,
שְׁנִיָּה לָהּ כְּסִיד הַהֵיכָל. <כַּסִּיד!>
וְהַשְּׂאֵת כִּקְרוּם בֵּיצָה,
וּשְׁנִיָּה לָהּ כְּצֶמֶר לָבָן.
דִּבְרֵי רְבִּי מֵאִיר.
וַחֲכָמִים אוֹמְרִים:
הַשְּׂאֵת כְּצֶמֶר לָבָן,
וּשְׁנִיָּה לָהּ כִּקְרוּם בֵּיצָה.

משנה ב
ב
הַפָּתוֹךְ שֶׁבַּשֶּׁלֶג, כְּיַיִן הַמָּזוּג בְּשֶׁלֶג;
הַפָּתוֹךְ שֶׁבַּסִּיד, כְּדָם הַמָּזוּג בֶּחָלָב.
דִּבְרֵי רְבִּי יִשְׁמָעֵאל.
רְבִּי עֲקִיבָה אוֹמֵר:
אֲדַמְדַּם שֶׁבָּזֶה וְשֶׁבָּזֶה, כְּיַיִן הַמָּזוּג בַּמַּיִם,
אֶלָּא שֶׁלַּשֶּׁלֶג עַזָּה, וְשֶׁלַּסִּיד דְּהֵה מִמֶּנָּה.

משנה ג
ג
אַרְבָּעָה מַרְאוֹת הָאֵלּוּ מִצְטָרְפִין זֶה עִם זֶה,
לִפְטוֹר, לְהַחֲלִיט, וּלְהַסְגִּיר.
לְהַסְגִּיר אֶת הָעוֹמֵד בְּסוֹף שָׁבוּעַ רִאשׁוֹן,
וְלִפְטוֹר אֶת הָעוֹמֵד בְּסוֹף שָׁבוּעַ שֵׁנִי. <לפטור>
לְהַחֲלִיט אֶת שֶׁנּוֹלַד לוֹ מִחְיָה, אוֹ שֵׂעָר לָבָן,
כַּתְּחִלָּה,
בְּסוֹף שָׁבוּעַ רִאשׁוֹן,
וּבְסוֹף שָׁבוּעַ שֵׁנִי, <בסוף>
לְאַחַר הַפְּטוּר.
לְהַחֲלִיט אֶת שֶׁנּוֹלַד לוֹ פִסָּיוֹן,
בְּסוֹף שָׁבוּעַ רִאשׁוֹן,
וּבְסוֹף שָׁבוּעַ שֵׁנִי, <בסוף>
לְאַחַר הַפְּטוּר.
לְהַחֲלִיט אֶת הַהוֹפֵךְ כֻּלּוֹ לָבָן מִתּוֹךְ הַפְּטוּר.
לִפְטוֹר אֶת הַהוֹפֵךְ כֻּלּוֹ לָבָן מִתּוֹךְ הַחְלֵט אוֹ מִתּוֹךְ הַסְגֵּר.
אֵלּוּ מַרְאוֹת נְגָעִים שֶׁכָּל נְגָעִים תְּלוּיִין בָּהֶן.

משנה ד
ד
רְבִּי דוֹסָה בֶן הַרְכִינָס אוֹמֵר:
מַרְאוֹת נְגָעִים שְׁלֹשִׁים וְשִׁשָּׁה. <שְׁלֹשָׁה>
עֲקַבְיָה בֶן מַהֲלַלְאֵל אוֹמֵר:
שִׁבְעִים וּשְׁנַיִם.
רְבִּי חֲנַנְיָה סְגַן הַכֹּהֲנִים אוֹמֵר:
אֵין רוֹאִין אֶת הַנְּגָעִים כַּתְּחִלָּה לְאַחַר הַשַּׁבָּת,
שֶׁהַשָּׁבוּעַ שֶׁלּוֹ חָל לִהְיוֹת בַּשַּׁבָּת;
וְלֹא בַשֵּׁנִי,
שֶׁהַשָּׁבוּעַ שֵׁנִי שֶׁלּוֹ חָל לִהְיוֹת בַּשַּׁבָּת;
וְלֹא בַשְּׁלִישִׁי לַבָּתִּים,
שֶׁהַשָּׁבוּעַ שְׁלִישִׁי שֶׁלּוֹ חָל לִהְיוֹת בַּשַּׁבָּת.
רְבִּי עֲקִיבָה אוֹמֵר:
לְעוֹלָם רוֹאִין אֶת הַנְּגָעִים.
חָל לִהְיוֹת בְּתוֹךְ הַשַּׁבָּת,
מַעֲבִירִין לְאַחַר הַשַּׁבָּת. <וּמַעֲבִירִין!>
וְיֵשׁ בַּדָּבָר לְהָקֵל וּלְהַחְמִיר.

משנה ה
ה
כֵּיצַד לְהָקֵל?
הָיָה בוֹ שֵׂעָר לָבָן, וְהָלַךְ לוֹ שֵׂעָר לָבָן.
הָיוּ לְבָנוֹת, וְהִשְׁחִירוּ.
אַחַת שְׁחוֹרָה וְאַחַת לְבָנָה, וְהִשְׁחִירוּ שְׁתֵּיהֶן.
אֲרֻכּוֹת, וְהִקְצִירוּ.
אַחַת אֲרֻכָּה וְאַחַת קְצָרָה, וְהִקְצִירוּ שְׁתֵּיהֶן.
נִסְמַךְ הַשְּׁחִין לִשְׁתֵּיהֶן, אוֹ לְאַחַת מֵהֶן.
הִקִּיף הַשְּׁחִין אֶת שְׁתֵּיהֶן, אוֹ אֶת אַחַת מֵהֶן,
אוֹ חֲלָקָן,
הַשְּׁחִין, וּמִחְיַת הַשְּׁחִין,
וְהַמִּכְוָה, וּמִחְיַת הַמִּכְוָה,
וְהַבֹּהַק.
הָיְתָה בוֹ מִחְיָה, וְהָלְכָה לָהּ הַמִּחְיָה;
הָיְתָה מְרֻבַּעַת, וְנֶעֱשֵׂית עֲגֻלָּה אוֹ אֲרֻכָּה,
מְבֻצֶּרֶת, וְנֶעֱשֵׂית מִן הַצַּד,
מְכֻנֶּסֶת, וְנִתְפַּזְּרָה,
וּבָא הַשְּׁחִין וְנִכְנַס לְתוֹכָהּ.
הִקִּיפָהּ, חֲלָקָהּ, אוֹ מִעֲטָהּ,
הַשְּׁחִין, אוֹ מִחְיַת הַשְּׁחִין,
וְהַמִּכְוָה, וּמִחְיַת הַמִּכְוָה,
וְהַבֹּהַק.
הָיָה בוֹ פִסָּיוֹן וְהָלַךְ לוֹ פִסָּיוֹן,
אוֹ שֶׁהָלְכָה לָהּ הָאוֹם, אוֹ שֶׁנִּתְמַעֲטָה,
אֵין בָּזֶה וּבָזֶה כַגָּרִיס.
הַשְּׁחִין וּמִחְיַת הַשְּׁחִין,
וְהַמִּכְוָה וּמִחְיַת הַמִּכְוָה,
וְהַבֹּהַק,
וְחוֹלְקִין בֵּין הָאוֹם לַפִּסָּיוֹן.
הֲרֵי אֵלּוּ לְהָקֵל.

משנה ו
ו
כֵּיצַד לְהַחְמִיר?
לֹא הָיָה בוֹ שֵׂעָר לָבָן, וְנוֹלַד לוֹ שֵׂעָר לָבָן.
הָיוּ שְׁחוֹרוֹת, וְהִלְבִּינוּ.
אַחַת שְׁחוֹרָה וְאַחַת לְבָנָה, וְהִלְבִּינוּ שְׁתֵּיהֶן.
קְצָרוֹת, וְהֶאֱרִיכוּ.
אַחַת קְצָרָה וְאַחַת אֲרֻכָּה, וְהֶאֱרִיכוּ שְׁתֵּיהֶן.
נִסְמַךְ הַשְּׁחִין לִשְׁתֵּיהֶן, אוֹ לְאַחַת מֵהֶן.
הִקִּיף הַשְּׁחִין לִשְׁתֵּיהֶן, אוֹ אֶת אַחַת מֵהֶן,
אוֹ חֲלָקָן,
הַשְּׁחִין וּמִחְיַת הַשְּׁחִין,
וְהַמִּכְוָה וּמִחְיַת הַמִּכְוָה,
וְהַבֹּהַק,
וְהָלְכוּ לָהֶן.
לֹא הָיְתָה בוֹ מִחְיָה, וְנוֹלְדָה לּוֹ מִחְיָה.
הָיְתָה עֲגֻלָּה אוֹ אֲרֻכָּה, וְנֶעֱשֵׂית מְרֻבַּעַת,
מִן הַצַּד, וְנֶעֱשֵׂית מְבֻצֶּרֶת,
[מְפֻזֶּרֶת], וְנִתְכַּנָּסָה,
וּבָא הַשְּׁחִין וְנִכְנַס לְתוֹכָהּ.
הִקִּיפָהּ, חֲלָקָהּ, אוֹ מִעֲטָהּ,
הַשְּׁחִין אוֹ מִחְיַת הַשְּׁחִין,
וְהַמִּכְוָה וּמִחְיַת הַמִּכְוָה וְהַבֹּהַק,
וְהָלְכוּ לָהֶן.
לֹא הָיָה בוֹ פִסָּיוֹן, וְנוֹלַד לוֹ פִסָּיוֹן.
הַשְּׁחִין וּמִחְיַת הַשְּׁחִין,
וְהַמִּכְוָה וּמִחְיַת הַמִּכְוָה וְהַבֹּהַק,
חוֹלְקִין בֵּין הָאוֹם לַפִּסָּיוֹן,
וְהָלְכוּ לָהֶן.
הֲרֵי אֵלּוּ לְהַחְמִיר.

מַסֶּכֶת נְגָעִים פֶּרֶק שֵׁנִי[עריכה]

משנה א
בַּהֶרֶת עַזָּה נִרְאֵית בַּגַּרְמְנִי כֵהָה, <בעל עור בהיר כבני גרמניה>
וְהַכֵּהָה בַכּוּשִׁי עַזָּה.
רְבִּי יִשְׁמָעֵאל אוֹמֵר:
בְּנֵי יִשְׂרָאֵל, אֲנִי כַפָּרָתָן, הֲרֵי הֵן כְּאֶשְׁכְּרַע,
לֹא שְׁחוֹרִים, וְלֹא לְבָנִים, אֶלָּא בֵינוֹנִיִּים.
רְבִּי עֲקִיבָה אוֹמֵר:
יֵשׁ לַצַּיָּרִים סַמְמַנִּין,
שֶׁהֵן צָרִים צוּרוֹת שְׁחוֹרוֹת, לְבָנוֹת וּבֵינוֹנִיּוֹת.
מֵבִיא סַם בֵּינוֹנִי וּמַקִּיפוֹ מִבַּחוּץ,
וְתֵרָאֶה כַבֵּינוֹנִי.
רְבִּי יְהוּדָה אוֹמֵר:
מַרְאוֹת נְגָעִים לְהָקֵל, אֲבָל לֹא לְהַחֲמִיר;
יֵרָאֶה הַגַּרְמְנִי בִבְשָׂרוֹ לְהָקֵל,
[וְהַכּוּשִׁי בַבֵּינוֹנִי לְהָקֵל.]
וַחֲכָמִים אוֹמְרִים:
זֶה וָזֶה בַּבֵּינוֹנִי.

משנה ב
ב
אֵין רוֹאִין אֶת הַנְּגָעִין בַּשַׁחְרִית, וּבֵין הָעַרְבַּיִם,

מִצְוָתָן שֶׁיִּהְיוּ שָׁווֹת בְּמַרְאֶה וּבְקוֹמָה וּבְדָמִין,
וּלְקִיחָתָן כְּאַחַת.
אַף עַל פִּי שֶׁאֵינָן שָׁווֹת, כְּשֵׁרוֹת.
שָׁחַט אֶת אַחַת מֵהֶן וְנִמְצֵאת שֶׁלֹּא דְרוֹר,
יִקַּח זוֹג לַשְּׁנִיָּה.
הָרִאשׁוֹנָה מֻתֶּרֶת בַּאֲכִילָה.
שְׁחָטָהּ וְנִמְצֵאת טְרֵפָה,
יִקַּח זוֹג לַשְּׁנִיָּה.
הָרִאשׁוֹנָה מֻתֶּרֶת בַּהֲנָיָה.
נִשְׁפַּךְ הַדָּם, תָּמוּת הַמִּשְׁתַּלַּחַת;
מֵתָה הַמִּשְׁתַּלַּחַת, יִשָּׁפֵךְ הַדָּם.

משנה ו
ז
מִצְוַת עֵץ אֶרֶז,
אָרְכּוֹ אַמָּה, וְעָבְיוֹ כִרְבִיעַ כְּרַע הַמִּטָּה:
אֶחָד לִשְׁנַיִם, וּשְׁנַיִם לְאַרְבָּעָה.
מִצְוַת אֵזוֹב,
לֹא אֵזוֹבְיוֹן,
לֹא אֵזוֹב כּוֹחֶלֶת,
וְלֹא אֵזוֹב רוֹמִי,
וְלֹא אֵזוֹב מִדְבָּרִי,
לֹא כָל אֵזוֹב שֶׁיֶּשׁ לוֹ שֵׁם לָווּי.

משנה ז
ח
בַּיּוֹם הַשְּׁמִינִי,
הָיָה מֵבִיא שָׁלוֹשׁ בְּהֵמוֹת:
חַטָּאת וְאָשָׁם וְעוֹלָה.
וְהַדַּל הָיָה מֵבִיא חַטָּאת הָעוֹף וְעוֹלַת הָעוֹף.

משנה ח
ט
בָּא לוֹ אֵצֶל אָשָׁם,
סָמַךְ שְׁתֵּי יָדָיו עָלָיו, וּשְׁחָטוֹ,
וְקִבְּלוּ שְׁנֵי כֹהֲנִים אֶת דָּמוֹ,
אֶחָד בְּכֶלִי, וְאֶחָד בְּיָד.
זֶה שֶׁקִּבֵּל בְּכֶלִי,
בָּא וּזְרָקוֹ עַל קִיר הַמִּזְבֵּחַ.
וְזֶה שֶׁקִּבֵּל בְּיָד,
בָּא לוֹ אֵצֶל מְצֹרָע.
וּמְצֹרָע טָבַל בְּלִשְׁכַּת הַמְצֹרָעִין,
בָּא וְעָמַד בְּשַׁעַרֵי נִיקָנוֹר.
רְבִּי יְהוּדָה אוֹמֵר:
לֹא הָיָה צָרִיךְ טְבִילָה.

משנה ט
י
הִכְנִיס רֹאשׁוֹ, וְנָתַן עַל תְּנוּךְ אָזְנוֹ.
הִכְנִיס יָדוֹ, וְנָתַן עַל בֹּהֶן יָדוֹ.
הִכְנִיס רַגְלוֹ, וְנָתַן עַל בֹּהֶן רַגְלוֹ.
רְבִּי יְהוּדָה אוֹמֵר:
שְׁלָשְׁתָּם הָיָה מַכְנִיס כְּאַחַת.
אֵין לוֹ בֹהֶן יָד, בֹּהֶן רֶגֶל, אֹזֶן יְמָנִית,
אֵין לוֹ טַהֲרָה עוֹלָמִית.
רְבִּי אֱלִיעֶזֶר אוֹמֵר:
נוֹתֵן הוּא עַל מְקוֹמָן.
רְבִּי שִׁמְעוֹן אוֹמֵר:
אִם נָתַן עַל שֶׁלִּשְׂמאל, יָצָא.

משנה י
יא
נָטַל מִלֹּג הַשֶּׁמֶן וְיָצַק לְתוֹךְ כַּפּוֹ שֶׁלַּחֲבֵרוֹ.
וְאִם יָצַק לְתוֹךְ כַּף עַצְמוֹ, יָצָא.
טָבַל וְהִזָּה שֶׁבַע פְּעָמִים כְּנֶגֶד בֵּית קֹדֶשׁ הַקֳּדָשִׁים.
עַל כָּל הַזָּיָה טְבִילָה.
בָּא לוֹ אֵצֶל מְצֹרָע.
מְקוֹם שֶׁהוּא נוֹתֵן אֶת הַדָּם,
שָׁם הוּא נוֹתֵן אֶת הַשֶּׁמֶן, שֶׁנֶּאֱמַר: (וַיִּקְרָא יד, כח-כט)
"עַל מְקוֹם דַּם הָאָשָׁם". <אל>
"וְהַנּוֹתָר מִן הַשֶּׁמֶן אֲשֶׁר עַל כַּף הַכֹּהֵן, <בשמן>
יִתֵּן עַל רֹאשׁ הַמִּטַּהֵר לְכַפֵּר".
אִם נָתַן, כִּפֵּר.
וְאִם לֹא נָתַן, לֹא כִפֵּר.
דִּבְרֵי רְבִּי עֲקִיבָה.
רְבִּי יוֹחָנָן בֶּן נוּרִי אוֹמֵר:
שִׁירֵי מִצְוָה הֵן,
בֵּין שֶׁנָּתַן בֵּין שֶׁלֹּא נָתַן, כִּפֵּר,
וּמַעֲלִין עָלָיו כְּאִלּוּ לֹא כִפֵּר.

יב
חָסַר הַלֹּג עַד שֶׁלֹּא יָצַק, יְמַלֶּנּוּ;
מִשֶּׁיָּצַק, יָבִיא אַחֵר כַּתְּחִלָּה,
דִּבְרֵי רְבִּי עֲקִיבָה.
רְבִּי שִׁמְעוֹן אוֹמֵר:
חָסַר הַלֹּג עַד שֶׁלֹּא נָתַן, יְמַלֶּנּוּ;
מִשֶּׁנָּתַן, יָבִיא אַחֵר כַּתְּחִלָּה.

משנה יא
יג
מְצֹרָע שֶׁהֵבִיא קָרְבָּנוֹ עָנִי, וְהֶעְשִׁיר,
אוֹ עָשִׁיר, וְהֶעְנִי,
הַכֹּל הוֹלֵךְ אַחַר חַטָּאת.
דִּבְרֵי רְבִּי שִׁמְעוֹן.
רְבִּי יְהוּדָה אוֹמֵר:
אַחַר הָאָשָׁם.

משנה יב
יד
מְצֹרָע עָנִי שֶׁהֵבִיא קָרְבַּן עָשִׁיר, יָצָא;
וְעָשִׁיר שֶׁהֵבִיא קָרְבַּן עָנִי, לֹא יָצָא.
מֵבִיא אָדָם עַל יְדֵי בְנוֹ, עַל יְדֵי בִתּוֹ,
עַל יְדֵי עַבְדּוֹ וְשִׁפְחָתוֹ קָרְבַּן עָנִי,
וּמַאֲכִילָן בִּזְבָחִין.
רְבִּי יְהוּדָה אוֹמֵר:
אַף עַל יְדֵי אִשְׁתּוֹ מֵבִיא קָרְבַּן עָשִׁיר,
וְכֵן כָּל קָרְבָּן שֶׁהִיא חַיֶּבֶת.

משנה יג
יה
שְׁנֵי מְצֹרָעִין שֶׁנִּתְעָרְבוּ קָרְבְּנוֹתֵיהֶן,
קָרַב קָרְבָּנוֹ שֶׁלְּאֶחָד מֵהֶן,
וּמֵת אֶחָד מֵהֶם,
זוֹ שָׁאֲלוּ אַנְשֵׁי אֲלֶכְּסַנְדְּרִיָּה אֶת רְבִּי יְהוֹשֻׁעַ.
אָמַר לָהֶם:
יִכְתֹּב נְכָסָיו לְאַחֵר,
וְיָבִיא קָרְבַּן עָנִי.

חַסְלַת מַסֶּכֶת נְגָעִין.
