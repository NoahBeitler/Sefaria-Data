מִשְׁנָה מַסֶּכֶת נְזִיקִים, בָּבָא קַמָּא (קאופמן) מְנֻקֶּדֶת וּמְעֻצֶּבֶת עַל יְדֵי הָרַב דָּן בְּאֵרִי[עריכה]

הניקוד תוקן על פי כללי הדקדוק בנוגע למלא וחסר, ולפי הצליל לגבי קמץ ופתח, צרי וסגול, המתחלפים בכתב יד קאופמן מתוך הגיה ספרדית. שינויי מספור ההלכות צוינו.

בְּשֵׁם אֵל מַבִּיט עֲמָקִים
נַתְחִיל סֵדֶר נְזִיקִים.‏[1]

הַצַּד הַשָּׁוֶה שֶׁבָּהֶן,

א

הָיְתָה מְבַעֶטֶת,
אוֹ שֶׁהָיוּ צְרוֹרוֹת מְנַתְּזִים מִתַּחַת רַגְלֶיהָ,

ג

וְהַבְּהֵמָה מוּעֶדֶת לֹאכַל פֵּרוֹת וִירָקוֹת.
אָכְלָה כְסוּת אוֹ כֵלִים,
מְשַׁלֵּם חֲצִי נֶזֶק.

בִּרְשׁוּת הַנִּזָּק,
אֲבָל בִּרְשׁוּת הָרַבִּים, פָּטוּר.

ד

מִפֶּתַח הֶחָנוּת, מְשַׁלֵּם מַה שֶּׁנֶּהְנֵית;
מִתּוֹךְ הֶחָנוּת, מְשַׁלֵּם מַה שֶּׁהִזִּיקָה.

אִם הָיָה דָבָר שֶׁיֶּשׁ בּוֹ אַחְרָיוּת,
חַיָּבִים לְשַׁלֵּם.
אֵין פּוֹרְטִין לֹא מִתֵּבַת הַמּוֹכְסִים,
וְלֹא מִכִּיס שֶׁלַּגַּבָּיִים,
וְאֵין נוֹטְלִין מֵהֶן צְדָקָה.
אֲבָל נוֹטֵל הוּא מִתּוֹךְ בֵּיתוֹ אוֹ מִן הַשּׁוּק.

משנה ב
ב
נָטְלוּ מוֹכְסִים אֶת חֲמוֹרוֹ,
וְנָתְנוּ לוֹ חֲמוֹר אַחֵר,
[גָּזְלוּ] הַלֶּסְטִים כְּסוּתוֹ,
וְנָתְנוּ לוֹ כְסוּת אַחֶרֶת,
הֲרֵי אֵלּוּ שֶׁלּוֹ,
מִפְּנֵי שֶׁהַבְּעָלִים מִתְיָאֲשִׁים מֵהֶן.
הַמַּצִּיל מִיַּד הַנָּהָר, מִיַּד הַלֶּסְטִים,
אִם נִתְיָאֲשׁוּ הַבְּעָלִים,
הֲרֵי אֵלּוּ שֶׁלּוֹ.
וְכֵן נְחִיל שֶׁלִּדְבוֹרִים,
אִם נִתְיָאֲשׁוּ הַבְּעָלִים,
הֲרֵי אֵלּוּ שֶׁלּוֹ.
אָמַר רְבִּי יוֹחָנָן בֶּן בְּרוֹקָה:
נֶאֱמֶנֶת אִשָּׁה אוֹ קָטָן לוֹמַר:
"מִכָּן יָצָא נְחִיל זֶה".
וּמְהַלֵּךְ בְּתוֹךְ שְׂדֵהוּ וְנוֹטֵל אֶת נְחִילוֹ,
וְאִם הִזִּיק, מְשַׁלֵּם מַה שֶּׁהִזִּיק.
אֲבָל לֹא יָקֹץ אֶת הַסּוֹכָה עַל מְנָת לִתֵּן דָּמִים.
רְבִּי יִשְׁמָעֵאל בְּנוֹ שֶׁלִּרְבִּי יוֹחָנָן בֶּן בְּרוֹקָה אוֹמֵר:
קוֹצֵץ וְנוֹתֵן דָּמִים.

משנה ג
ג
הַמַּכִּיר כֵּלָיו אוֹ סְפָרָיו בְּיַד אַחֵר,
אִם יָצָא לוֹ שֵׁם גְּנֵבָה בָעִיר,
יִשָּׁבַע כַּמָּה הוֹצִיא, וְיִטֹּל.
וְאִם לָאו, לֹא הַכֹּל מִמֶּנּוּ, שֶׁאֲנִי אוֹמֵר: <שני>
מְכָרָן לְאַחֵר, וּלְקָחָן זֶה מִמֶּנּוּ.

משנה ד
ד
זֶה בָא בַחֲבִיתוֹ שֶׁלַּיַּיִן, <בַחֲבִיתּוֹ>
וְזֶה בָא בְכַדּוֹ שֶׁלִדְבַשׁ;
נִסְדְּקָה חֲבִית שֶׁלַּדְּבַשׁ,
וְשָׁפַךְ זֶה אֶת יֵינוֹ וְהִצִּיל אֶת הַדְּבַשׁ לְתוֹכוֹ,
אֵין לוֹ אֶלָּא שְׂכָרוֹ;
אִם אָמַר לוֹ:
"אַצִּיל אֶת שֶׁלָּךְ, וְאַתָּה נוֹתֵן לִי דְמֵי שֶׁלִּי",
חַיָּב לִתֵּן לוֹ.

ה
שָׁטַף נָהָר חֲמוֹרוֹ וַחֲמוֹר חֲבֵרוֹ,
שֶׁלּוֹ יָפֶה מָנֶה וְשֶׁלַּחֲבֵרוֹ מָאתַיִם,
הִנִּיחַ אֶת שֶׁלּוֹ, וְהִצִּיל אֶת שֶׁלַּחֲבֵרוֹ,
אֵין לוֹ אֶלָּא שְׂכָרוֹ.
אִם אָמַר לוֹ:
"אַצִּיל אֶת שֶׁלָּךְ, וְאַתָּה נוֹתֵן לִי דְמֵי שֶׁלִּי",
חַיָּב לִתֵּן לוֹ.

משנה ה
ו
הַגּוֹזֵל שָׂדֶה, וּנְטָלוּהָ הַמַּסִּיקִים,
אִם מַכַּת מְדִינָה הוּא,
אוֹמֵר לוֹ:
"הֲרֵי שֶׁלָּךְ לְפָנֶיךָ!"
וְאִם מֵחֲמַת הַגָּזְלָן,
חַיָּב לְהַעֲמִיד לוֹ שָׂדֶה [אַחֶרֶת].
שְׁטָפָהּ נָהָר,
אוֹמֵר לוֹ:
"הֲרֵי שֶׁלָּךְ לְפָנֶיךָ!"

משנה ו
ז
הַגּוֹזֵל אֶת חֲבֵרוֹ,
אוֹ שֶׁלָּוָה הֵימֶנּוּ,
אוֹ שֶׁהִפְקִיד לוֹ בַיּוֹשֵׁב,
לֹא יַחְזִיר לוֹ בַמִּדְבָּר;
עַל מְנָת לָצֵאת בַּמִּדְבָּר,
יַחְזִיר לוֹ בַמִּדְבָּר.

משנה ז
ח
הָאוֹמֵר לַחֲבֵרוֹ:
"גְּזַלְתִּיךָ",
וְ"הִלְוִיתַנִי",
"הִפְקַדְתָּ אֶצְלִי,
וְאֵינִי יוֹדֵעַ אִם הֶחֱזַרְתִּי לָךְ, אִם לֹא הֶחֱזַרְתִּי",
חַיָּב לְשַׁלֵּם.
אֲבָל "אֵינִי יוֹדֵעַ אִם גְּזַלְתִּיךָ",
וְ"אִם הִלְוִיתַנִי",
"אִם הִפְקַדְתָּ אֶצְלִי, וְאִם הִפְקַדְתָּ",
פָּטוּר מִלְּשַׁלֵּם.

משנה ח
ט
הַגּוֹנֵב טָלֶה מִן הָעֵדֶר וְהֶחֱזִירוֹ,
מֵת אוֹ נִגְנַב,
חַיָּב בְּאַחְרָיוּתוֹ.
וְאִם לֹא יָדְעוּ הַבְּעָלִים בִּגְנֵבָתוֹ וּבַחֲזִירָתוֹ,
וּמָנוּ אֶת הַצֹּאן וְהִיא שְׁלֵמָה, <מֶן הַצֹּאן>
פָּטוּר מִלְּשַׁלֵּם.

משנה ט
י
אֵין לוֹקְחִין מִן הָרוֹעִים צֶמֶר וְחָלָב וּגְדָיִים,
וְלֹא מִשּׁוֹמְרֵי פֵרוֹת עֵצִים וּפֵרוֹת.
אֲבָל לוֹקְחִין מִן הַנָּשִׁים [כְּלֵי צֶמֶר בִּיהוּדָה],
כְּלֵי פִשְׁתָּן בַּגָּלִיל, וַעֲגָלִים בַּשָּׁרוֹן.
וְכֻלָּם שֶׁאָמְרוּ "הַטְמֵן!"
אָסוּר.
וְלוֹקְחִין בֵּיצִים וְתַּרְנָגְלִים מִכָּל מָקוֹם. <ותורנגלים>

משנה י
יא
מוֹכִים שֶׁהַכּוֹבֵס מוֹצִיא,
הֲרֵי אֵלּוּ שֶׁלּוֹ;
שֶׁהַסָּרוּק מוֹצִיא,
הֲרֵי הֵן שֶׁלְּבַעַל הַבַּיִת.
הַכּוֹבֵס נוֹטֵל שְׁלֹשָׁה חוּטִין, וְהֵן שֶׁלּוֹ.
יָתֵר מִכֵּן, שֶׁלְּבַעַל הַבַּיִת.
וְאִם הָיָה שָׁחוֹר עַל גַּבֵּי לָבָן,
נוֹטֵל אֶת הַכֹּל וְהֵן שֶׁלּוֹ.

יב
הַחַיָּט שֶׁשִּׁיֵּר מִן הַחוּט
כְּדֵי לִתְפּוֹר בּוֹ מַטְלֵית שֶׁהִיא שָׁלוֹשׁ עַל שָׁלוֹשׁ,
חַיָּב לְהַחְזִיר לַבְּעָלִים. <לחזיר>
מַה שֶּׁחָרָשׁ מוֹצִיא בַמַּעְצָד,
הֲרֵי אֵלּוּ שֶׁלּוֹ;
וּבַכָּשִׁיל,
הֲרֵי אֵלּוּ שֶׁלְּבַעַל הַבַּיִת.
אִם הָיָה עוֹשֶׂה אֵצֶל בַּעַל הַבַּיִת,
אַף הַנְּסֹרֶת שֶׁלְּבַעַל הַבַּיִת.
