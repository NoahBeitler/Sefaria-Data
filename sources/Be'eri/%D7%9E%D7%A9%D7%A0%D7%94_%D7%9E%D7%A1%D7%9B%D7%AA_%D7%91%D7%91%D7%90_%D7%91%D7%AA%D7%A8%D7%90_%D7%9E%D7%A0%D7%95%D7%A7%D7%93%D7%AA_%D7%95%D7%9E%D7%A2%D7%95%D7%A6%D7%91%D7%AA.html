מִשְׁנָה מַסֶּכֶת נְזִיקִים, בָּבָא בַּתְרָא (קאופמן) מְנֻקֶּדֶת וּמְעֻצֶּבֶת עַל יְדֵי הָרַב דָּן בְּאֵרִי[עריכה]

הניקוד תוקן על פי כללי הדקדוק בנוגע למלא וחסר, ולפי הצליל לגבי קמץ ופתח, צרי וסגול, המתחלפים בכתב יד קאופמן מתוך הגיה ספרדית. שינויי מספור ההלכות צוינו.

מַסֶּכֶת נְזִיקִים פֶּרֶק אֶחָד וְעֶשְׂרִים (בָּבָא בַּתְרָא פֶּרֶק רִאשׁוֹן)[עריכה]

ח

משנה ט
ט

רְבִּי עֲקִיבָה אוֹמֵר:
לְכָל רוּחַ הוּא עוֹשֶׂה, חוּץ מִן מַעֲרָבָהּ,
וּמַרְחִיק חֲמִשִּׁים אַמָּה.

י

משנה יא
יא

וְאִם הָאִילָן קָדַם, לֹא יָקֹץ.
סָפֵק זֶה קָדַם, וְזֶה קָדַם, לֹא יָקֹץ.
רְבִּי יוֹסֵה אוֹמֵר:
אַף עַל פִּי שֶׁהַבּוֹר קָדַם אֶת הָאִילָן, לֹא יָקֹץ,
זֶה חוֹפֵר בְּתוֹךְ שֶׁלּוֹ וְזֶה נוֹטֵעַ בְּתוֹךְ שֶׁלּוֹ.

משנה יב
יב

אֶחָד גְּפָנִים וְאֶחָד כָּל הָאִילָן.
הָיָה גָדֵר בִּנְתַּיִם,
זֶה סוֹמֵךְ לַגָּדֵר [מִכָּן, וְזֶה סוֹמֵךְ לַגָּדֵר מִכָּן.]
הָיוּ שָׁרָשִׁין יצְאִין לְתוֹךְ שְׂדֵה חֲבֵרוֹ,
מַעֲמִיק שְׁלֹשָׁה טְפָחִין,
כְּדֵי שֶׁלֹּא יְעַכֵּב אֶת הַמַּחֲרֵשָׁה.
הָיָה חוֹפֵר בּוֹר וְשִׁיחַ וּמְעָרָה,
קוֹצֵץ וְיוֹרֵד, וְהָעֵצִים שֶׁלּוֹ.

משנה יג
יג
אִילָן שֶׁהוּא נוֹטֶה לְתוֹךְ שְׂדֵה חֲבֵרוֹ,
קוֹצֵץ מְלֹא מַרְדֵּעַ עַל גַּבֵּי הַמַּחֲרֵשָׁה,
וְהֶחָרוּב וְהַשִּׁקְמָה, כְּנֶגֶד הַמִּשְׁקֹלֶת.
בֵּית הַשְּׁלָחִים וּבֵית הָאִילָן, כְּנֶגֶד הַמִּשְׁקֹלֶת.
אַבָּא שָׁאוּל אוֹמֵר:
כָּל אִילַן סְרָק, כְּנֶגֶד הַמִּשְׁקֹלֶת.

משנה יד
יד
אִילָן שֶׁהוּא נוֹטֶה לִרְשׁוּת הָרַבִּים,
קוֹצֵץ כְּדֵי שֶׁיְּהֵא הַגָּמָל עוֹבֵר בְּרוֹכְבוֹ.
רְבִּי יְהוּדָה אוֹמֵר:
טָעוּן פִּשְׁתָּן אוֹ זְמוֹרוֹת.
רְבִּי שִׁמְעוֹן אוֹמֵר:
כָּל הָאִילָן, כְּנֶגֶד הַמִּשְׁקלֶת, מִפְּנֵי הַטֻּמְאָה.

מַסֶּכֶת נְזִיקִים פֶּרֶק שְׁלוֹשָׁה וְעֶשְׂרִים (בָּבָא בַּתְרָא פֶּרֶק שְׁלִישִׁי)[עריכה]

משנה א
א
חֶזְקַת בָּתִּים, בּוֹרוֹת, שִׁיחִים, וּמְעָרוֹת,
מַרְחָצָיוֹת, שׁוֹבַכּוֹת, בֵּית הַבַּדִּים,
בֵּית הַשְּׁלָחִים, וַעֲבָדִים,
וְכָל שֶׁהוּא עוֹשֶׂה פֵרוֹת תָּדִיר,
חֶזְקָתָן שָׁלוֹשׁ שָׁנִים וּמִיּוֹם לְיוֹם.
שְׂדֵה הַבַּעַל, חֶזְקָתָהּ שָׁלוֹשׁ שָׁנִים,
וְאֵינָה מִיּוֹם לְיוֹם.

ב
רְבִּי יִשְׁמָעֵאל אוֹמֵר:
שְׁלֹשָׁה חֳדָשִׁים בָּרִאשׁוֹנָה,
וּשְׁלֹשָׁה חֳדָשִׁים בָּאַחֲרוֹנָה,
וּשְׁנֵים עָשָׂר חֹדֶשׁ [בָּאֶמְצַע,
שְׁמֹנָה עָשָׂר חֹדֶשׁ.]
רְבִּי עֲקִיבָה אוֹמֵר:
חֹדֶשׁ אֶחָד בָּרִאשֹׁנָה,
וְחֹדֶשׁ אֶחָד בָּאַחֲרוֹנָה
וּשְׁנֵים עָשָׂר חֹדֶשׁ בָּאֶמְצַע,
אַרְבָּעָה עָשָׂר חֹדֶשׁ.
אָמַר רְבִּי יִשְׁמָעֵאל:
בַּמֵּי דְבָרִין אֲמוּרִין? בִּשְׂדֵה הַלָּבָן,
אֲבָל בִּשְׂדֵה הָאִילָן,
כָּנַס אֶת תְּבוּאָתוֹ, וּמָסַק אֶת זֵיתָיו, כָּנַס אֶת קַיְצוֹ,
הֲרֵי אֵלּוּ שָׁלֹשׁ שָׁנִים.

משנה ב
ג
שָׁלוֹשׁ אֲרָצוֹת לַחֲזָקָה:
יְהוּדָה וְעֵבֶר הַיַּרְדֵּן וְגָלִיל.
הָיָה בִיהוּדָה וְהֶחֱזִיק בַּגָּלִיל,
בַּגָּלִיל וְהֶחֱזִיק בִּיהוּדָה,
אֵינָה חֲזָקָה, עַד שֶׁתְּהֵא עִמּוֹ בַמְּדִינָה.
אָמַר רְבִּי יְהוּדָה:
לֹא אָמְרוּ שָׁלוֹשׁ שָׁנִים,
אֶלָּא כְּדֵי שֶׁיְּהֵא בְאִסְפַּנְיָא,
יַחֲזִיק שָׁנָה,
וְיֵלְכוּ וְיוֹדִיעוּהוּ שָׁנָה,
וְיָבֹאוּ שָׁנָה אַחֶרֶת.

משנה ג
ד
כָּל חֲזָקָה שֶׁאֵין עִמָּהּ טְעָנָה אֵינָהּ חֲזָקָה.

"מָה אַתְּ עוֹשֶׂה בְתוֹךְ שֶׁלִּי?"
"שֶׁלֹּא אָמַר לִי אָדָם דָּבָר",
אֵינָה חֲזָקָה.
"שֶׁמָּכַרְתָּ לִי", "שֶׁנָּתַתָּ לִי מַתָּנָה",
הֲרֵי זוֹ חֲזָקָה.
וְהַבָּא מִשֵּׁם יְרוּשָׁה אֵינוּ צָרִיךְ טְעָנָה.

ה

אֵין לָהֶם חֲזָקָה.
אֵין לָאִישׁ חֲזָקָה בְנִכְסֵי אִשְׁתּוֹ,
וְלֹא לָאִשָּׁה חֲזָקָה בְנִכְסֵי בַעְלָהּ,
וְלֹא הָאָב בְּנִכְסֵי הַבֵּן,
וְלֹא הַבֵּן בְּנִכְסֵי הָאָב.
בַּמִּי דְבָרִים אֲמוּרִין? בַּמַּחֲזִיק,
אֲבָל בְּנוֹתֵן מַתָּנָה,
וְהָאַחִים שֶׁיַּחֲלֹקוּ,
וְהַמַּחֲזִיק בְּנִכְסֵי הַגֵּר,
נָעַל גָּדַר, פָּרַץ כָּל שֶׁהוּא,
הֲרֵי זוֹ חֲזָקָה.

משנה ד
ו
שְׁנַיִם מְעִידִים אוֹתוֹ שֶׁאֲכָלָהּ שָׁלוֹשׁ שָׁנִים,
נִמְצְאוּ זוֹמְמִים,
מְשַׁלְּמִים לוֹ אֶת הַכֹּל.
שְׁנַיִם בָּרִאשׁוֹנָה, וּשְׁנַיִם בַּשְּׁנִיָּה, וּשְׁנַיִם בַּשְּׁלִישִׁית,
מְשַׁלִּים בֵּינֵיהֶן.
שְׁלֹשָׁה אָחִין וְאֶחָד מִצְטָרֵף עִמָּהֶן,
הֲרֵי אֵלּוּ שָׁלוֹשׁ עֵדִיּוֹת,
וְהֵן עֵדוּת אַחַת.

ז

וְאֵלּוּ אֵין לָהֶן חֲזָקָה:
הָיָה מַעֲמִיד בְּהֵמָה בֶחָצֵר,
מַעֲמִיד תַּנּוּר וְכִירַיִם וְרֵחַיִם,
וּמְגַדֵּל תַּרְנָגְלִים,

אָמַר לוֹ "הַנַּח לוֹ וַאֲנִי נוֹתֵן לָךְ!"
פָּטוּר, שֶׁלֹּא עַל אֻמָּנוּתוֹ הִלְוָהוּ. <'אוּמָּנוּתוֹ' כתב המגיה על מקום מחוק. אולי היה כתוב 'מנת' כמו להלן>
וְאֵי זֶה הוּא הֶעָרָב שֶׁהוּא חַיָּב לוֹ?
אָמַר לוֹ "הַלְוֵהוּ וַאֲנִי נוֹתֵן לָךְ",
חַיָּב, שֶׁכֵּן עַל מְנַת הִלְוָהוּ.
אָמַר רְבִּי יִשְׁמָעֵאל:
הָרוֹצֶה לְהַחְכִּים, יַעֲסֹק בְּדִינֵי מָמוֹנוֹת,
שֶׁאֵין מִקְצוֹעַ בַּתּוֹרָה גָדוֹל מֵהֶן,
שֶׁהֵן כְּמַעְיָן נוֹבֵעַ.
וְהָרוֹצֶה לַעֲסֹק בְּדִינֵי מָמוֹנוֹת,
יְשַׁמֵּשׁ אֶת שִׁמְעוֹן בֶּן נַנָּס.
