מִשְׁנָה מַסֶּכֶת אֲהִילוֹת (קאופמן) מְנֻקֶּדֶת וּמְעֻצֶּבֶת עַל יְדֵי הָרַב דָּן בְּאֵרִי[עריכה]

הניקוד תוקן על פי כללי הדקדוק בנוגע למלא וחסר, ולפי הצליל לגבי קמץ ופתח, צרי וסגול, המתחלפים בכתב יד קאופמן מתוך הגיה ספרדית. שינויי מספור ההלכות צוינו.

מַסֶּכֶת אֲהִילוֹת פֶּרֶק רִאשׁוֹן[עריכה]

משנה א
שְׁנַיִם טְמֵאִים בַּמֵּת,
אֶחָד טָמֵא טֻמְאַת שִׁבְעָה,
וְאֶחָד טָמֵא טֻמְאַת עֶרֶב.
שְׁלֹשָׁה טְמֵאִים בַּמֵּת,
שְׁנַיִם טְמֵאִים טֻמְאַת שִׁבְעָה,
וְאֶחָד טָמֵא טֻמְאַת עֶרֶב.
אַרְבָּעָה טְמֵאִים בַּמֵּת,
שְׁלֹשָׁה טְמֵאִים טֻמְאַת שִׁבְעָה,
וְאֶחָד טָמֵא טֻמְאַת עֶרֶב.
כֵּיצַד שְׁנַיִם?
אָדָם הַנּוֹגֵעַ בַּמֵּת,
טָמֵא טֻמְאַת שִׁבְעָה;
וְאָדָם הַנּוֹגֵעַ בּוֹ,
טָמֵא טֻמְאַת עֶרֶב.

ב
כֵּיצַד שְׁלֹשָׁה?
כֵּלִים הַנּוֹגְעִים בַּמֵּת,
וְכֵלִים בַּכֵּלִים,

הַשְּׁלִישִׁי, בֵּין אָדָם בֵּין כֵּלִים,
טָמֵא טֻמְאַת עֶרֶב.

ג
כֵּיצַד אַרְבָּעָה?

וְאָדָם בַּכֵּלִים,
וְכֵלִים בָּאָדָם,
טְמֵאִין טֻמְאַת שִׁבְעָה.

בֵּין אָדָם, בֵּין כֵּלִים,
טָמֵא טֻמְאַת עֶרֶב.

ד
אָמַר רְבִּי עֲקִיבָה:
יֶשׁ לִי חֲמִישִׁי:
שְׁפוּד הַתָּחוּב בָּאֹהֶל,
הָאֹהֶל וְהַשְּׁפוּד,
וְאָדָם הַנּוֹגֵעַ בַּשְּׁפוּד,
וְכֵלִים בָּאָדָם,
טְמֵאִין טֻמְאַת שִׁבְעָה.
הַחֲמִישִׁי, בֵּין אָדָם, בֵּין כֵּלִים,
טָמֵא טֻמְאַת עֶרֶב.
אָמְרוּ לוֹ:
אֵין הָאֹהֶל מִתְחַשֵּׁב.

ה
אָדָם וְכֵלִים מִטַּמִּין בַּמֵּת.
חֹמֶר בָּאָדָם מִבַּכֵּלִים,
וּבַכֵּלִים מִבָּאָדָם,
שֶׁהַכֵּלִים שְׁלֹשָׁה,
וְהָאָדָם שְׁנַיִם.
חֹמֶר בָּאָדָם,
שֶׁכָּל זְמַן שֶׁהוּא בְּאֶמְצַע,
הֵן אַרְבָּעָה,
אֵינוּ בְאֶמְצַע,
הֵן שְׁלֹשָׁה.

ו
אָדָם וּבְגָדִים מִטַּמִּין בַּזָּב.
חֹמֶר בָּאָדָם מִבַּבְּגָדִים,

שֶׁהָאָדָם הַנּוֹגֵעַ בַּזָּב מְטַמֵּא בְגָדִים,
וְאֵין בְּגָדִים הַנּוֹגְעִים בַּזָּב מְטַמִּין בְּגָדִים.
חֹמֶר בַּבְּגָדִים,
שֶׁהַבְּגָדִים הַנּוֹשְׂאִים אֶת הַזָּב מְטַמִּין אָדָם,

וַאֲפִלּוּ מְגֻיָּד,

וְכֵן בְּהֵמָה וְחַיָּה,
אֵינָן מְטַמִּין עַד שֶׁתֵּצֵא נַפְשָׁן.
הֻתָּזוּ רָאשֵׁיהֶן,
אַף עַל פִּי מְפַרְכְּסִין, טְמֵאִין,
כְּגוֹן הַזָּנָב שֶׁלַּלְּטָאָה שֶׁהִיא מְפַרְכֶּסֶת.

אֲפִלּוּ פָחוּת מִכַּזַּיִת מִן הַמֵּת,
וּפָחוּת מִכַּזַּיִת מִן הַנְּבֵלָה,

ט
מָאתַיִם וְאַרְבָּעִים וּשְׁמוֹנָה אֵבָרִים בָּאָדָם:
שְׁלֹשִׁים בְּפִסַּת הָרֶגֶל,
שִׁשָּׁה בְכָל אֶצְבַּע,
עֲשָׂרָה בַקַּרְסוּל,
שְׁנַיִם בַּשּׁוֹק,
חֲמִשָּׁה בָאַרְכּוּבָה,
אֶחָד בַּיָּרֵךְ,
שְׁלֹשָׁה בַּקָּטְלִית,
וְאַחַד עֶשְׂרֵה צְלָעוֹת,
שְׁלֹשִׁים בְּפִסַּת הַיָּד,
שִׁשָּׁה בְכָל אֶצְבַּע,
שְׁנַיִם בַּקָּנֶה,
שְׁנַיִם בַּמַּרְפֵּק,
אֶחָד בַּזְּרוֹעַ,
וְאַרְבָּעָה בַכָּתֵף.
מֵאָה וְאֶחָד מִזֶּה,
וּמֵאָה וְאֶחָד מִזֶּה.

י
שְׁמוֹנֶה עֶשְׂרֵה חֲלָיוֹת בַּשִּׁזְרָה,
תִּשְׁעָה בָרֹאשׁ,
שְׁמוֹנָה בַצַּוָּאר,
שִׁשָּׁה בַמַּפְתֵּחַ שֶׁלַּלֵּב,
וַחֲמִשָּׁה בִנְקוּבָיו.
כָּל אֶחָד וְאֶחָד מְטַמֵּא בְמַגָּע וּבְמַשָּׂא וּבְאָהֵל.
אֵמָתַי?
בִּזְמַן שֶׁיֵּשׁ עֲלֵיהֶן בָּשָׂר כָּרָאוּי.
אִם אֵין עֲלֵיהֶן בָּשָׂר כָּרָאוּי,
מְטַמִּין בְּמַגָּע וּבְמַשָּׂא,
וְאֵינָן מְטַמִּין בְּאָהֵל.

מַסֶּכֶת אֲהִילוֹת פֶּרֶק שֵׁנִי[עריכה]

משנה א
אֵלּוּ מְטַמִּין בְּאָהֵל:
הַמֵּת, וְכַזַּיִת מִן הַמֵּת,
וְכַזַּיִת מִן נֶצֶל,
וּמְלֹא תַרְוָד רָקָב,
הַשִּׁזְרָה, וְהַגֻּלְגֹּלֶת,
אֵבֶר מִן הַמֵּת, אֵבֶר מִן הַחַי
שֶׁיֵּשׁ עֲלֵיהֶן בָּשָׂר כָּרָאוּי,
רֹבַע עֲצָמוֹת מֵרֹב הַבִּנְיָן אוֹ מֵרֹב הַמִּנְיָן,
רֹב בִּנְיָנוֹ וְרֹב מִנְיָנוֹ שֶׁלַּמֵּת,
אַף עַל פִּי שֶׁאֵין בָּהֶם רֹבַע,
טְמֵאִין.
וְכַמָּה הוּא רֹב מִנְיָנוֹ?
מֵאָה וְעֶשְׂרִים וַחֲמִשָּׁה.

ב
רְבִיעִית דָּם וּרְבִיעִית דַּם תְּבוּסָה,
מִמֵּת אֶחָד.
רְבִּי עֲקִיבָה אוֹמֵר:
מִשְּׁנֵי מֵתִים.
דַּם קָטָן שֶׁיָּצָא כֻלּוֹ,
רְבִּי עֲקִיבָה אוֹמֵר:
כָּל שֶׁהוּא.
וַחֲכָמִים אוֹמְרִים:
רְבִיעִית.
כַּזַּיִת רִמָּה, בֵין חָיָה, בֵין מֵתָה,
רְבִּי אֱלִיעֶזֶר מְטַמֵּא כִבְשָׂרוֹ,
וַחֲכָמִים מְטַהֲרִין.
אֵפֶר שְׂרוּפִים,
רְבִּי אֱלִיעֶזֶר אוֹמֵר:
שֵׁעוּרוֹ כְּרֹבַע,
וַחֲכָמִים מְטַהֲרִין.
מְלֹא תַרְוָד וָעוֹד עֲפַר קְבָרוֹת, טָמֵא.
רְבִּי שִׁמְעוֹן מְטַהֵר.
וּמְלֹא תַרְוָד רָקָב שֶׁגְּבָלוֹ בַמַּיִם,
אֵינוּ חִבּוּר לַטֻּמְאָה.

משנה ג
ג

וְאֵינָן מְטַמִּין בְּאָהֵל:
עֶצֶם כִּשְׂעוֹרָה,
וְאֶרֶץ הָעַמִּים,
וּבֵית הַפְּרַס,
וְאֵבֶר מִן הַמֵּת,
וְאֵבֶר מִן הַחַי שֶׁאֵין עֲלֵיהֶן בָּשָׂר כָּרָאוּי,
הַשִּׁזְרָה וְהַגֻּלְגֹּלֶת שֶׁחָסָרוּ.
כַּמָּה הוּא חֶסְרוֹנָן?
בֵּית שַׁמַּי אוֹמְרִים:
שְׁתֵּי חֲלָיוֹת;
וּבֵית הֶלֵּל אוֹמְרִים:
אֲפִלּוּ חָלְיָה אַחַת.
וּבַגֻּלְגּלֶת?
בֵּית שַׁמַּי אוֹמְרִים:
מְלֹא מַקְדֵּחַ;
וּבֵית הֶלֵּל אוֹמְרִים:
כְּדֵי שֶׁיִּנָּטֵל מִן הַחַי וְיָמוּת.
בְּאֵיזֶה מַקְדֵּחַ אָמָרוּ?
בַּקָּטָן שֶׁלָּרוֹפְאִין,
דִּבְרֵי רְבִּי מֵאִיר.
וַחֲכָמִים אוֹמְרִים:
בַּגָּדוֹל שֶׁלַּלִּשְׁכָּה.

אִם נִמְצָא שָׁם עֶצֶם כִּשְׂעוֹרָה, טָמֵא.

משנה ה
ה
כֵּיצַד מְטַהֲרִין בֵּית הַפְּרַס?
נוֹטֵל מִמֶּנּוּ שְׁלֹשָׁה טְפָחִים,
וְנוֹתֵן עַל גַּבָּיו שְׁלֹשָׁה טְפָחִים.
אִם נָטַל מֵחֶצְיוֹ אֶחָד שְׁלֹשָׁה טְפָחִים,
וְנָתַן עַל גַּבֵּי חֶצְיוֹ אַחֵר שְׁלֹשָׁה טְפָחִים,
טָהוֹר.
רְבִּי שִׁמְעוֹן אוֹמֵר:
אֲפִלּוּ נָטַל מִמֶּנּוּ טֶפַח וּמַחְצָה,
וְנָתַן עָלָיו טֶפַח וּמַחְצָה מִמָּקוֹם אַחֵר,
טָהוֹר.

ו
הָרוֹצֵף בֵּית הַפְּרַס בַּאֲבָנִים שֶׁאֵינוּ יָכוֹל לַהֲסִיטָן,
טָהוֹר.
רְבִּי שִׁמְעוֹן אוֹמֵר:
אַף הָעוֹזֵק בֵּית הַפְּרַס, טָהוֹר. <עזק: חפר>

משנה ו
ז
הַמְהַלֵּךְ בֵּית הַפְּרַס עַל אֲבָנִים שֶׁאֵינוּ יָכוֹל לַהֲסִיטָן,
עַל הָאָדָם וְעַל הַבְּהֵמָה שֶׁכֹּחָן רַע, טָמֵא.
הַמְהַלֵּךְ בְּאֶרֶץ הָעַמִּים,
בֶּהָרִים וּבַסְּלָעִים, טָמֵא;
בַּיָּם וּבַשּׁוּנִית, טָהוֹר.
וְאֵי זוֹ הִיא הַשּׁוּנִית? <אי זה הוא!>
כָּל מָקוֹם שֶׁהַיָּם עוֹלֶה בְזַעְפּוֹ.

משנה ז
ח
הַקּוֹנֶה שָׂדֶה בְסוּרְיָה, סְמוּכָה לְאֶרֶץ יִשְׂרָאֵל,
אִם יָכוֹל לְהִכָּנֵס לָהּ בְּטַהֲרָה,
טְהוֹרָה, וְחַיֶּבֶת בַּמַּעַשְׂרוֹת וּבַשְּׁבִיעִית.
וְאִם אֵינוּ יָכוֹל לְהִכָּנֵס לָהּ בְּטַהֲרָה,
טְמֵאָה, וְחַיֶּבֶת בַּמַּעַשְׂרוֹת וּבַשְּׁבִיעִית.
מְדוּרוֹת הַגּוֹיִם, טְמֵאִין.
כַּמָּה יִשְׁהֵא בְתוֹכָן וִיהֵא צָרִיךְ בְּדִיקָה?
אַרְבָּעִים יוֹם, אַף עַל פִּי שֶׁאֵין עִמּוֹ אִשָּׁה.
וְאִם הָיָה עֶבֶד אוֹ אִשָּׁה מְשַׁמְּרִין אוֹתוֹ,
אֵינוּ צָרִיךְ בְּדִיקָה. <וְאֵינוּ!>

משנה ח
ט
אֶת מָה הֵם בּוֹדְקִין?
אֶת בִּיבִים הָעֲמֻקִּין,
וְאֶת הַמַּיִם הַסְּרוּחִין.
בֵּית שַׁמַּי אוֹמְרִים:
אַף הָאַשְׁפַּתּוֹת וְעָפָר הַתּוֹחֵחַ.
וּבֵית הֶלֵּל אוֹמְרִים:
כָּל מָקוֹם שֶׁהַחֲזִיר וְהַחֻלְדָּה יְכוּלִים לְהַלְּכוֹ,
אֵינוּ צָרִיךְ בְּדִיקָה.

משנה ט
י
הָאִסְטְוָנוֹת, <stoà: מעבר מכוסה עם עמודים לארכו>
אֵין בָּהֶן מִשֵּׁם מְדוֹר גּוֹיִם.
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר:
עִיר שֶׁחָרָבָה, אֵין בָּהּ מִשֵּׁם מְדוֹר גּוֹיִם.
מִזְרַח קֵסְרִי <קסרין!>
וּמַעֲרָב קֵסֶריוֹן, <קֵסַריוֹן. Caesarea Philippa: עיר ליד פניאס למרגלות החרמון>
קְבָרוֹת.
וּמִזְרַח עַכּוֹ,
הָיָה סָפֵק, וְטִהֲרוּהוּ חֲכָמִים.
רֶבִּי וּבֵית דִּינוֹ נִמְנוּ עַל קֵינֵי וְטִהֲרוּהָ. <קֵינֵי: עיר בדרום>

משנה י
יא
עֲשָׂרָה מְקוֹמוֹת אֵין בָּהֶן מִשֵּׁם מְדוֹר גּוֹיִם:
אָהֳלֵי הָעַרְבִיִּים,
וְהַסֻּכּוֹת,
וְהַצְּרִיפִין,
וְהַבּוֹרְגְּנִין, <purgíon: מגדל, חומה עם מגדלים, מבצר>
וְהַלִּקְטָיוֹת,
וּבֵית שַׁעַר,
וַאֲוֵירָה שֶׁלֶּחָצֵר,
וְהַמַּרְחֵץ,
וּמְקוֹם הַחִצִּים,
וּמְקוֹם הַלִּגְיוֹנוֹת.

חַסְלַת מַסֶּכֶת אֲהִילוֹת.
