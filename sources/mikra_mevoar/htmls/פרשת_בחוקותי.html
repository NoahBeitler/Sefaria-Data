ראשון[עריכה]

פ פָּרָשַׁת הַבְּרָכוֹת
כו ג אִם־בְּחֻקֹּתַ֖י תֵּלֵ֑כוּ וְאֶת־מִצְוֺתַ֣י תִּשְׁמְר֔וּ וַעֲשִׂיתֶ֖ם אֹתָֽם׃ אִם תִּתְנַהֲגוּ לְפִי חֻקַּי וְתִשְׁמְרוּ אֶת מִצְווֹתַי וְתַעֲשׂוּ אוֹתָם.
ד וְנָתַתִּ֥י גִשְׁמֵיכֶ֖ם בְּעִתָּ֑ם וְנָתְנָ֤ה הָאָ֙רֶץ֙ יְבוּלָ֔הּ וְעֵ֥ץ הַשָּׂדֶ֖ה יִתֵּ֥ן פִּרְיֽוֹ׃ אֶתֵּן אֶת גִּשְׁמֵיכֶם בַּזְּמַן הַמַּתְאִים, הָאָרֶץ תִּתֵּן אֶת יְבוּלָהּ וְהָעֵץ בַּשָּׂדֶה יַצְמִיחַ אֶת פִּרְיוֹ.
ה וְהִשִּׂ֨יג לָכֶ֥ם דַּ֙יִשׁ֙ אֶת־בָּצִ֔יר וּבָצִ֖יר יַשִּׂ֣יג אֶת־זָ֑רַע וַאֲכַלְתֶּ֤ם לַחְמְכֶם֙ לָשֹׂ֔בַע וִֽישַׁבְתֶּ֥ם לָבֶ֖טַח בְּאַרְצְכֶֽם׃ עוֹנַת הַדִּישָׁה תִּמָּשֵׁךְ לָכֶם עַד עוֹנַת הַבָּצִיר וְהַבָּצִיר יִמָּשֵׁךְ עַד עוֹנַת הַזְּרִיעָה, וְתֹאכְלוּ אֶת לַחְמְכֶם לָשׂוֹבַע וְתֵשְׁבוּ בְּבִטָּחוֹן בְּאַרְצְכֶם.

שני[עריכה]

ו וְנָתַתִּ֤י שָׁלוֹם֙ בָּאָ֔רֶץ וּשְׁכַבְתֶּ֖ם וְאֵ֣ין מַחֲרִ֑יד וְהִשְׁבַּתִּ֞י חַיָּ֤ה רָעָה֙ מִן־הָאָ֔רֶץ וְחֶ֖רֶב לֹא־תַעֲבֹ֥ר בְּאַרְצְכֶֽם׃ אַשְׁכִּין שָׁלוֹם בָּאָרֶץ, תִּשְׁכְּבוּ לִישׁוֹן וְאַף אֶחָד לֹא יַפְחִיד אֶתְכֶם, אַשְׁמִיד אֶת הַחַיּוֹת הָרָעוֹת מֵהָאָרֶץ, וְלֹא תַּעֲבוֹר מִלְחָמָה בְּאַרְצְכֶם.
ז וּרְדַפְתֶּ֖ם אֶת־אֹיְבֵיכֶ֑ם וְנָפְל֥וּ לִפְנֵיכֶ֖ם לֶחָֽרֶב׃ תִּרְדְּפוּ אֶת אוֹיְבֵיכֶם וְהֵם יִפְּלוּ לִפְנֵיכֶם בַּמִּלְחָמָה.
ח וְרָדְפ֨וּ מִכֶּ֤ם חֲמִשָּׁה֙ מֵאָ֔ה וּמֵאָ֥ה מִכֶּ֖ם רְבָבָ֣ה יִרְדֹּ֑פוּ וְנָפְל֧וּ אֹיְבֵיכֶ֛ם לִפְנֵיכֶ֖ם לֶחָֽרֶב׃ חֲמִשָּׁה מֵאִתְּכֶם יִרְדְּפוּ מֵאָה מֵהָאוֹיֵב, וּמֵאָה מֵאִתְּכֶם יִרְדְּפוּ עֲשֶׂרֶת אֲלָפִים, אוֹיְבֵיכֶם יִפְּלוּ לִפְנֵיכֶם בַּמִּלְחָמָה.
ט וּפָנִ֣יתִי אֲלֵיכֶ֔ם וְהִפְרֵיתִ֣י אֶתְכֶ֔ם וְהִרְבֵּיתִ֖י אֶתְכֶ֑ם וַהֲקִימֹתִ֥י אֶת־בְּרִיתִ֖י אִתְּכֶֽם׃ אֶפְנֶה אֲלֵיכֶם לְהֵטִיב לָכֶם, וְאַרְבֶּה אֶתְכֶם וַאֲגַדֵּל אֶתְכֶם, וַאֲקַיֵּם אֶת הֶסְכֵּמִי אִתְּכֶם.

שלישי (חמישי במחוברין)[עריכה]

י וַאֲכַלְתֶּ֥ם יָשָׁ֖ן נוֹשָׁ֑ן וְיָשָׁ֕ן מִפְּנֵ֥י חָדָ֖שׁ תּוֹצִֽיאוּ׃ תֹּאכְלוּ יָשָׁן נוֹשָׁן מֵרוֹב תְּבוּאָה, תִּצְטָרְכוּ לְהוֹצִיא מֵהַמַּחְסָנִים תְּבוּאָה יְשָׁנָה כְּדֵי לְהַכְנִיס חֲדָשָׁה.
יא וְנָתַתִּ֥י מִשְׁכָּנִ֖י בְּתוֹכְכֶ֑ם וְלֹֽא־תִגְעַ֥ל נַפְשִׁ֖י אֶתְכֶֽם׃ אַשְׁרֶה אֶת שְׁכִינָתִי בְּתוֹכְכֶם וְנַפְשִׁי לֹא תִּדְחֶה אֶתְכֶם.
יב וְהִתְהַלַּכְתִּי֙ בְּת֣וֹכְכֶ֔ם וְהָיִ֥יתִי לָכֶ֖ם לֵֽאלֹהִ֑ים וְאַתֶּ֖ם תִּהְיוּ־לִ֥י לְעָֽם׃ שְׁכִינָתִי תִּהְיֶה מְצוּיָה בְּתוֹכְכֶם וְאֶהְיֶה לָכֶם לֵאלֹהִים וְאַתֶּם תִּהְיוּ לִי לְעָם.
יג אֲנִ֞י יְהֹוָ֣ה אֱלֹֽהֵיכֶ֗ם אֲשֶׁ֨ר הוֹצֵ֤אתִי אֶתְכֶם֙ מֵאֶ֣רֶץ מִצְרַ֔יִם מִֽהְיֹ֥ת לָהֶ֖ם עֲבָדִ֑ים וָאֶשְׁבֹּר֙ מֹטֹ֣ת עֻלְּכֶ֔ם וָאוֹלֵ֥ךְ אֶתְכֶ֖ם קֽוֹמְמִיּֽוּת׃ אֲנִי ה' אֱלֹהֵיכֶם שֶׁהוֹצֵאתִי אֶתְכֶם מֵאֶרֶץ מִצְרַיִם מִלִּהְיוֹת לָהֶם עֲבָדִים, בִּטַּלְתִּי אֶת הָעַבְדּוּת שֶׁלָּכֶם וְהוֹלַכְתִּי אֶתְכֶם בְּקוֹמָה זְקוּפָה.
פ פָּרָשַׁת הַקְּלָלוֹת
יד וְאִם־לֹ֥א תִשְׁמְע֖וּ לִ֑י וְלֹ֣א תַעֲשׂ֔וּ אֵ֥ת כׇּל־הַמִּצְוֺ֖ת הָאֵֽלֶּה׃ וְאִם לֹא תִּשְׁמְעוּ לִי וְלֹא תַּעֲשׂוּ אֶת כָּל הַמִּצְווֹת הָאֵלֶּה.
טו וְאִם־בְּחֻקֹּתַ֣י תִּמְאָ֔סוּ וְאִ֥ם אֶת־מִשְׁפָּטַ֖י תִּגְעַ֣ל נַפְשְׁכֶ֑ם לְבִלְתִּ֤י עֲשׂוֹת֙ אֶת־כׇּל־מִצְוֺתַ֔י לְהַפְרְכֶ֖ם אֶת־בְּרִיתִֽי׃ וְאִם תִּמְאָסוּ בְּחֻקַּי, וְאִם נַפְשְׁכֶם תִּדְחֶה אֶת מִשְׁפָּטַי, שֶׁלֹּא תַּעֲשׂוּ אֶת כָּל מִצְווֹתַי וְתָפֵרוּ אֶת הַהֶסְכֵּם אִתִּי.
טז אַף־אֲנִ֞י אֶֽעֱשֶׂה־זֹּ֣את לָכֶ֗ם וְהִפְקַדְתִּ֨י עֲלֵיכֶ֤ם בֶּֽהָלָה֙ אֶת־הַשַּׁחֶ֣פֶת וְאֶת־הַקַּדַּ֔חַת מְכַלּ֥וֹת עֵינַ֖יִם וּמְדִיבֹ֣ת נָ֑פֶשׁ וּזְרַעְתֶּ֤ם לָרִיק֙ זַרְעֲכֶ֔ם וַאֲכָלֻ֖הוּ אֹיְבֵיכֶֽם׃ גַּם אֲנִי אֶעֱשֶׂה לָכֶם כָּךְ, וְאֶשְׁלַח בָּכֶם פַּחַד שַׁחֶפֶת וְקַדַּחַת – מַחְשִׁיכוֹת עֵינַיִם וּמַעֲצִיבוֹת אֶת הַנֶּפֶשׁ, תִּזְרְעוּ לַשָּׁוְא אֶת זַרְעֲכֶם וְיֹאכְלוּ אוֹתוֹ אוֹיְבֵיכֶם.
יז וְנָתַתִּ֤י פָנַי֙ בָּכֶ֔ם וְנִגַּפְתֶּ֖ם לִפְנֵ֣י אֹיְבֵיכֶ֑ם וְרָד֤וּ בָכֶם֙ שֹֽׂנְאֵיכֶ֔ם וְנַסְתֶּ֖ם וְאֵין־רֹדֵ֥ף אֶתְכֶֽם׃ אַפְנֶה אֶת כַּעֲסִי עֲלֵיכֶם וְתוּבְסוּ לִפְנֵי אוֹיְבֵיכֶם, שׂוֹנְאֵיכֶם יִשְׁלְטוּ בָּכֶם וְתִבְרְחוּ גַּם כְּשֶׁאֵין מִי שֶׁרוֹדֵף אַחֲרֵיכֶם.
יח וְאִ֨ם־עַד־אֵ֔לֶּה לֹ֥א תִשְׁמְע֖וּ לִ֑י וְיָסַפְתִּי֙ לְיַסְּרָ֣ה אֶתְכֶ֔ם שֶׁ֖בַע עַל־חַטֹּאתֵיכֶֽם׃ וְאִם גַּם אַחֲרֵי מַכּוֹת אֵלֶּה לֹא תִּשְׁמְעוּ לִי – אוֹסִיף לְיַסֵּר אֶתְכֶם פִּי שֶׁבַע בִּגְלַל חַטֹּאתֵיכֶם.
יט וְשָׁבַרְתִּ֖י אֶת־גְּא֣וֹן עֻזְּכֶ֑ם וְנָתַתִּ֤י אֶת־שְׁמֵיכֶם֙ כַּבַּרְזֶ֔ל וְאֶֽת־אַרְצְכֶ֖ם כַּנְּחֻשָֽׁה׃ אֶשְׁבּוֹר אֶת גַּאֲוַתְכֶם עַל כּוֹחֲכֶם, אֲיַבֵּשׁ אֶת שְׁמֵיכֶם כַּבַּרְזֶל מִלְּהוֹרִיד גֶּשֶׁם, וְאַקְשֶׁה אֶת אַרְצְכֶם כַּנְּחוֹשֶׁת מִלְּהַצְמִיחַ.
כ וְתַ֥ם לָרִ֖יק כֹּחֲכֶ֑ם וְלֹֽא־תִתֵּ֤ן אַרְצְכֶם֙ אֶת־יְבוּלָ֔הּ וְעֵ֣ץ הָאָ֔רֶץ לֹ֥א יִתֵּ֖ן פִּרְיֽוֹ׃ כּוֹחֲכֶם יִגָּמֵר לַשָּׁוְא בַּעֲבוֹדַת הָאֲדָמָה, אַרְצְכֶם לֹא תִּתֵּן אֶת יְבוּלָהּ וְעֵץ הָאָרֶץ לֹא יִתֵּן אֶת פִּרְיוֹ.
כא וְאִם־תֵּֽלְכ֤וּ עִמִּי֙ קֶ֔רִי וְלֹ֥א תֹאב֖וּ לִשְׁמֹ֣עַֽ לִ֑י וְיָסַפְתִּ֤י עֲלֵיכֶם֙ מַכָּ֔ה שֶׁ֖בַע כְּחַטֹּאתֵיכֶֽם׃ וְאִם תֵּלְכוּ אִתִּי בְּעַקְשָׁנוּת וְלֹא תִּרְצוּ לִשְׁמוֹעַ לִי – אוֹסִיף עֲלֵיכֶם מַכָּה פִּי שֶׁבַע – כָּרָאוּי לְחַטֹּאתֵיכֶם.
כב וְהִשְׁלַחְתִּ֨י בָכֶ֜ם אֶת־חַיַּ֤ת הַשָּׂדֶה֙ וְשִׁכְּלָ֣ה אֶתְכֶ֔ם וְהִכְרִ֙יתָה֙ אֶת־בְּהֶמְתְּכֶ֔ם וְהִמְעִ֖יטָה אֶתְכֶ֑ם וְנָשַׁ֖מּוּ דַּרְכֵיכֶֽם׃ אֲשַׁסֶּה בָּכֶם אֶת חַיּוֹת הַשָּׂדֶה שֶׁיַּעֲשׂוּ אֶתְכֶם שַׁכּוּלִים, הֵם יַשְׁמִידוּ אֶת בֶּהֱמוֹתֵיכֶם וְיַמְעִיטוּ אֶתְכֶם, וְדַרְכֵיכֶם תִּהְיֶינָה שׁוֹמְמוֹת.
כג וְאִ֨ם־בְּאֵ֔לֶּה לֹ֥א תִוָּסְר֖וּ לִ֑י וַהֲלַכְתֶּ֥ם עִמִּ֖י קֶֽרִי׃ וְאִם מִמַּכּוֹת אֵלֶּה לֹא תִּלְמְדוּ מוּסָר לִשְׁמוֹעַ בְּקוֹלִי, וְתֵלְכוּ אִתִּי בְּעַקְשָׁנוּת.
כד וְהָלַכְתִּ֧י אַף־אֲנִ֛י עִמָּכֶ֖ם בְּקֶ֑רִי וְהִכֵּיתִ֤י אֶתְכֶם֙ גַּם־אָ֔נִי שֶׁ֖בַע עַל־חַטֹּאתֵיכֶֽם׃ גַּם אֲנִי אֵלֵךְ אִתְּכֶם בְּעַקְשָׁנוּת וְאַכֶּה אֶתְכֶם גַּם אֲנִי פִּי שֶׁבַע – עוֹנֶשׁ עַל חַטֹּאתֵיכֶם.
כה וְהֵבֵאתִ֨י עֲלֵיכֶ֜ם חֶ֗רֶב נֹקֶ֙מֶת֙ נְקַם־בְּרִ֔ית וְנֶאֱסַפְתֶּ֖ם אֶל־עָרֵיכֶ֑ם וְשִׁלַּ֤חְתִּי דֶ֙בֶר֙ בְּת֣וֹכְכֶ֔ם וְנִתַּתֶּ֖ם בְּיַד־אוֹיֵֽב׃ אָבִיא עֲלֵיכֶם מִלְחָמָה לִנְקוֹם עַל הֲפָרַת הַהֶסְכֵּם אִתִּי וְתִתְכַּנְּסוּ בְּעָרֵיכֶם, וְאָז אֶשְׁלַח בְּתוֹכְכֶם מַחֲלַת דֶּבֶר, וְתִלָּכְדוּ בְּיַד הָאוֹיֵב.
כו בְּשִׁבְרִ֣י לָכֶם֮ מַטֵּה־לֶ֒חֶם֒ וְ֠אָפ֠וּ עֶ֣שֶׂר נָשִׁ֤ים לַחְמְכֶם֙ בְּתַנּ֣וּר אֶחָ֔ד וְהֵשִׁ֥יבוּ לַחְמְכֶ֖ם בַּמִּשְׁקָ֑ל וַאֲכַלְתֶּ֖ם וְלֹ֥א תִשְׂבָּֽעוּ׃ כְּשֶׁאֶשְׁבּוֹר וַאֲמוֹטֵט אֶת פַּרְנָסַתְכֶם, עֶשֶׂר נָשִׁים תֹּאפֶינָה אֶת לַחְמְכֶם הַמּוּעָט בְּתַנּוּר אֶחָד וִיחַלְּקוּ בֵּינֵיהֶן אֶת לַחְמְכֶם בְּמִשְׁקָל, וְתֹאכְלוּ וְלֹא תִּשְׂבְּעוּ.
ס
כז וְאִ֨ם־בְּזֹ֔את לֹ֥א תִשְׁמְע֖וּ לִ֑י וַהֲלַכְתֶּ֥ם עִמִּ֖י בְּקֶֽרִי׃ וְאִם בְּכָל זֹאת לֹא תִּשְׁמְעוּ לִי וְתֵלְכוּ אִתִּי בְּעַקְשָׁנוּת.
כח וְהָלַכְתִּ֥י עִמָּכֶ֖ם בַּחֲמַת־קֶ֑רִי וְיִסַּרְתִּ֤י אֶתְכֶם֙ אַף־אָ֔נִי שֶׁ֖בַע עַל־חַטֹּאתֵיכֶֽם׃ אֵלֵךְ אִתְּכֶם בְּעַקְשָׁנוּת וּבְכַעַס, וַאֲיַסֵּר אֶתְכֶם גַּם אֲנִי פִּי שֶׁבַע, עוֹנֶשׁ עַל חֲטָאֵיכֶם.
כט וַאֲכַלְתֶּ֖ם בְּשַׂ֣ר בְּנֵיכֶ֑ם וּבְשַׂ֥ר בְּנֹתֵיכֶ֖ם תֹּאכֵֽלוּ׃ תֹּאכְלוּ אֶת בְּשַׂר בְּנֵיכֶם וְתֹאכְלוּ אֶת בְּשַׂר בְּנוֹתֵיכֶם.
ל וְהִשְׁמַדְתִּ֞י אֶת־בָּמֹֽתֵיכֶ֗ם וְהִכְרַתִּי֙ אֶת־חַמָּ֣נֵיכֶ֔ם וְנָֽתַתִּי֙ אֶת־פִּגְרֵיכֶ֔ם עַל־פִּגְרֵ֖י גִּלּוּלֵיכֶ֑ם וְגָעֲלָ֥ה נַפְשִׁ֖י אֶתְכֶֽם׃ אַשְׁמִיד אֶת בָּמוֹת אֱלִילֵיכֶם וְאַשְׁמִיד אֶת פִּסְלֵי הַשֶּׁמֶשׁ שֶׁלָּכֶם, אָשִׂים אֶת גּוּפוֹתֵיכֶם עַל שִׁבְרֵי פִּסְלֵיכֶם, וְנַפְשִׁי תִּדְחֶה אֶתְכֶם.
לא וְנָתַתִּ֤י אֶת־עָֽרֵיכֶם֙ חׇרְבָּ֔ה וַהֲשִׁמּוֹתִ֖י אֶת־מִקְדְּשֵׁיכֶ֑ם וְלֹ֣א אָרִ֔יחַ בְּרֵ֖יחַ נִיחֹֽחֲכֶֽם׃ אֶעֱשֶׂה אֶת עָרֵיכֶם חֻרְבּוֹת, וְאֶהֱפוֹךְ אֶת מִקְדְּשֵׁיכֶם לִשְׁמָמָה, וְלֹא אָרִיחַ אֶת נִיחוֹחַ קָרְבְּנוֹתֵיכֶם.
לב וַהֲשִׁמֹּתִ֥י אֲנִ֖י אֶת־הָאָ֑רֶץ וְשָֽׁמְמ֤וּ עָלֶ֙יהָ֙ אֹֽיְבֵיכֶ֔ם הַיֹּשְׁבִ֖ים בָּֽהּ׃ אֲנִי אֶעֱשֶׂה אֶת הָאָרֶץ שְׁמָמָה, אֲפִלּוּ אוֹיְבֵיכֶם שֶׁיִּתְיַשְּׁבוּ בָּהּ יִהְיוּ בִּשְׁמָמָה.
לג וְאֶתְכֶם֙ אֱזָרֶ֣ה בַגּוֹיִ֔ם וַהֲרִיקֹתִ֥י אַחֲרֵיכֶ֖ם חָ֑רֶב וְהָיְתָ֤ה אַרְצְכֶם֙ שְׁמָמָ֔ה וְעָרֵיכֶ֖ם יִהְי֥וּ חׇרְבָּֽה׃ וְאֶתְכֶם אֲפַזֵּר בַּגּוֹיִם, וְאֶשְׁלוֹף עֲלֵיכֶם חֶרֶב – שֶׁיִּרְדְּפוּ אַחֲרֵיכֶם, אַרְצְכֶם תִּהְיֶה שְׁמָמָה וְעָרֵיכֶם תִּהְיֶינָה חֲרֵבוֹת.
לד אָז֩ תִּרְצֶ֨ה הָאָ֜רֶץ אֶת־שַׁבְּתֹתֶ֗יהָ כֹּ֚ל יְמֵ֣י הׇשַּׁמָּ֔הֿ וְאַתֶּ֖ם בְּאֶ֣רֶץ אֹיְבֵיכֶ֑ם אָ֚ז תִּשְׁבַּ֣ת הָאָ֔רֶץ וְהִרְצָ֖ת אֶת־שַׁבְּתֹתֶֽיהָ׃ אָז יְשֻּׁלְמוּ לָאָרֶץ הַשְּׁמִטּוֹת שֶׁהִפְסִידָה – כָּל יְמֵי הַשְּׁמָמָה שֶׁל הַגָּלוּת כְּשֶׁאַתֶּם בְּאֶרֶץ אוֹיְבֵיכֶם, אָז הָאָרֶץ תִּשְׁבּוֹת וּתְקַבֵּל בַּחֲזָרָה אֶת שְׁמִטּוֹתֶיהָ.
לה כׇּל־יְמֵ֥י הׇשַּׁמָּ֖הֿ תִּשְׁבֹּ֑ת אֵ֣ת אֲשֶׁ֧ר לֹֽא־שָׁבְתָ֛ה בְּשַׁבְּתֹתֵיכֶ֖ם בְּשִׁבְתְּכֶ֥ם עָלֶֽיהָ׃ הִיא תִּשְׁבּוֹת כָּל יְמֵי שִׁמְמָתָהּ אֶת מַה שֶּׁלֹּא שָׁבְתָה בִּשְׁנוֹת הַשְּׁמִיטָה בַּזְּמַן שֶׁיְּשַׁבְתֶּם עָלֶיהָ.
לו וְהַנִּשְׁאָרִ֣ים בָּכֶ֔ם וְהֵבֵ֤אתִי מֹ֙רֶךְ֙ בִּלְבָבָ֔ם בְּאַרְצֹ֖ת אֹיְבֵיהֶ֑ם וְרָדַ֣ף אֹתָ֗ם ק֚וֹל עָלֶ֣ה נִדָּ֔ף וְנָס֧וּ מְנֻֽסַת־חֶ֛רֶב וְנָפְל֖וּ וְאֵ֥ין רֹדֵֽף׃ וְהַנִּשְׁאָרִים מִכֶּם, אָבִיא פַּחַד בְּלִבָּם בְּאַרְצוֹת אוֹיְבֵיהֶם, יִרְדּוֹף אוֹתָם אֲפִלּוּ רַעַשׁ שֶׁל עָלֶה מְעוֹפֵף, וְיִבְרְחוּ מִמֶּנּוּ כְּמוֹ מֵחֶרֶב וְיִפְּלוּ גַּם כְּשֶׁאֵין מִי שֶׁרוֹדֵף.
לז וְכָשְׁל֧וּ אִישׁ־בְּאָחִ֛יו כְּמִפְּנֵי־חֶ֖רֶב וְרֹדֵ֣ף אָ֑יִן וְלֹא־תִֽהְיֶ֤ה לָכֶם֙ תְּקוּמָ֔ה לִפְנֵ֖י אֹֽיְבֵיכֶֽם׃ יִתָּקְלוּ אֶחָד בַּשֵּׁנִי כְּמוֹ כְּשֶׁבּוֹרְחִים מֵחֶרֶב וְאֵין מִי שֶׁרוֹדֵף, וְלֹא תַּצְלִיחוּ לְהִתְאוֹשֵׁשׁ מִמַּפָּלַת אוֹיְבֵיכֶם.
לח וַאֲבַדְתֶּ֖ם בַּגּוֹיִ֑ם וְאָכְלָ֣ה אֶתְכֶ֔ם אֶ֖רֶץ אֹיְבֵיכֶֽם׃ תֹּאבְדוּ בַּגּוֹיִם וְאֶרֶץ אוֹיְבֵיכֶם תְּכַלֶּה אֶתְכֶם.
לט וְהַנִּשְׁאָרִ֣ים בָּכֶ֗ם יִמַּ֙קּוּ֙ בַּֽעֲוֺנָ֔ם בְּאַרְצֹ֖ת אֹיְבֵיכֶ֑ם וְאַ֛ף בַּעֲוֺנֹ֥ת אֲבֹתָ֖ם אִתָּ֥ם יִמָּֽקּוּ׃ וְהַנִּשְׁאָרִים מִכֶּם יֵרָקְבוּ בַּעֲווֹנוֹתֵיהֶם בְּאַרְצוֹת אוֹיְבֵיכֶם, וְיֵרָקְבוּ גַּם בַּעֲווֹנוֹת אֲבוֹתֵיהֶם שֶׁהֵם מַמְשִׁיכִים לַעֲשׂוֹת.
מ וְהִתְוַדּ֤וּ אֶת־עֲוֺנָם֙ וְאֶת־עֲוֺ֣ן אֲבֹתָ֔ם בְּמַעֲלָ֖ם אֲשֶׁ֣ר מָֽעֲלוּ־בִ֑י וְאַ֕ף אֲשֶׁר־הָֽלְכ֥וּ עִמִּ֖י בְּקֶֽרִי׃ וְהֵם יִתְוַדּוּ עַל עֲווֹנוֹתֵיהֶם וְעַל עֲווֹנוֹת אֲבוֹתֵיהֶם, בִּבְגִידָתָם שֶׁבָּגְדוּ בִּי, וְגַם עַל שֶׁהָלְכוּ אִתִּי בְּעַקְשָׁנוּת.
מא אַף־אֲנִ֗י אֵלֵ֤ךְ עִמָּם֙ בְּקֶ֔רִי וְהֵבֵאתִ֣י אֹתָ֔ם בְּאֶ֖רֶץ אֹיְבֵיהֶ֑ם אוֹ־אָ֣ז יִכָּנַ֗ע לְבָבָם֙ הֶֽעָרֵ֔ל וְאָ֖ז יִרְצ֥וּ אֶת־עֲוֺנָֽם׃ גַּם אֲנִי אֵלֵךְ אִתָּם בְּעַקְשָׁנוּת וְאָבִיא אוֹתָם לְאֶרֶץ אוֹיְבֵיהֶם, רַק אָז יִכָּנַע לִבָּם הַסָּתוּם וְאָז יְקַבְּלוּ עוֹנֶשׁ עַל עֲווֹנָם.
מב וְזָכַרְתִּ֖י אֶת־בְּרִיתִ֣י יַעֲק֑וֹב וְאַף֩ אֶת־בְּרִיתִ֨י יִצְחָ֜ק וְאַ֨ף אֶת־בְּרִיתִ֧י אַבְרָהָ֛ם אֶזְכֹּ֖ר וְהָאָ֥רֶץ אֶזְכֹּֽר׃ וְאֶזְכּוֹר אֶת הַבְטָחָתִי לְיַעֲקֹב וְגַם אֶת הַבְטָחָתִי לְיִצְחָק וְגַם אֶת הַבְטָחָתִי לְאַבְרָהָם אֶזְכּוֹר, וְאֶזְכּוֹר אֶת אֶרֶץ יִשְׂרָאֵל.
מג וְהָאָ֩רֶץ֩ תֵּעָזֵ֨ב מֵהֶ֜ם וְתִ֣רֶץ אֶת־שַׁבְּתֹתֶ֗יהָ בׇּהְשַׁמָּהֿ֙ מֵהֶ֔ם וְהֵ֖ם יִרְצ֣וּ אֶת־עֲוֺנָ֑ם יַ֣עַן וּבְיַ֔עַן בְּמִשְׁפָּטַ֣י מָאָ֔סוּ וְאֶת־חֻקֹּתַ֖י גָּעֲלָ֥ה נַפְשָֽׁם׃ וְהָאָרֶץ תֵּעָזֵב מֵהֶם וּתְקַבֵּל אֶת הַשְּׁמִטּוֹת שֶׁהִפְסִידָה כְּשֶׁתִּהְיֶה שׁוֹמֵמָה מֵהֶם, וְהֵם יְקַבְּלוּ עוֹנֶשׁ עַל עֲווֹנָם בִּגְלַל שֶׁמָּאָסוּ בְּמִשְׁפָּטַי וְנַפְשָׁם דָּחֲתָה אֶת חֻקַּי.
מד וְאַף־גַּם־זֹ֠את בִּֽהְיוֹתָ֞ם בְּאֶ֣רֶץ אֹֽיְבֵיהֶ֗ם לֹֽא־מְאַסְתִּ֤ים וְלֹֽא־גְעַלְתִּים֙ לְכַלֹּתָ֔ם לְהָפֵ֥ר בְּרִיתִ֖י אִתָּ֑ם כִּ֛י אֲנִ֥י יְהֹוָ֖ה אֱלֹהֵיהֶֽם׃ וְגַם אָז כְּשֶׁיִּהְיוּ בְּאֶרֶץ אוֹיְבֵיהֶם לֹא אֶמְאַס בָּהֶם וְלֹא אֶדְחֶה אוֹתָם לְהַשְׁמִיד אוֹתָם – לְהָפֵר אֶת הֶסְכֵּמִי אִתָּם, כִּי אֲנִי ה' אֱלֹהֵיהֶם.
מה וְזָכַרְתִּ֥י לָהֶ֖ם בְּרִ֣ית רִאשֹׁנִ֑ים אֲשֶׁ֣ר הוֹצֵֽאתִי־אֹתָם֩ מֵאֶ֨רֶץ מִצְרַ֜יִם לְעֵינֵ֣י הַגּוֹיִ֗ם לִהְי֥וֹת לָהֶ֛ם לֵאלֹהִ֖ים אֲנִ֥י יְהֹוָֽה׃ וְאֶזְכּוֹר לָהֶם אֶת הֶסְכֵּמִי עִם אֲבוֹתֵיהֶם הָרִאשׁוֹנִים שֶׁהוֹצֵאתִי אוֹתָם מֵאֶרֶץ מִצְרַיִם לְעֵינֵי הָעַמִּים לִהְיוֹת לָהֶם לֵאלֹהִים, אֲנִי ה'.
מו אֵ֠לֶּה הַֽחֻקִּ֣ים וְהַמִּשְׁפָּטִים֮ וְהַתּוֹרֹת֒ אֲשֶׁר֙ נָתַ֣ן יְהֹוָ֔ה בֵּינ֕וֹ וּבֵ֖ין בְּנֵ֣י יִשְׂרָאֵ֑ל בְּהַ֥ר סִינַ֖י בְּיַד־מֹשֶֽׁה׃ אֵלֶּה הַחֻקִּים וְהַמִּשְׁפָּטִים וְהַהוֹרָאוֹת שֶׁה' הוֹרָה בְּהַר סִינַי בְּאֶמְצָעוּת מֹשֶׁה בַּהֶסְכֵּם שֶׁבֵּינוֹ וּבֵין בְּנֵי יִשְׂרָאֵל.

רביעי (ששי במחוברין)[עריכה]

פ פָּרָשַׁת הָעֲרָכִים
כז א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב דַּבֵּ֞ר אֶל־בְּנֵ֤י יִשְׂרָאֵל֙ וְאָמַרְתָּ֣ אֲלֵהֶ֔ם אִ֕ישׁ כִּ֥י יַפְלִ֖א נֶ֑דֶר בְּעֶרְכְּךָ֥ נְפָשֹׁ֖ת לַֽיהֹוָֽה׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְתֹאמַר לָהֶם: אִישׁ שֶׁיַּפְרִישׁ מִמָּמוֹנוֹ בְּנֶדֶר לְהָבִיא לַה' כֶּסֶף בְּשׁוֹוִי אָדָם.
ג וְהָיָ֤ה עֶרְכְּךָ֙ הַזָּכָ֔ר מִבֶּן֙ עֶשְׂרִ֣ים שָׁנָ֔ה וְעַ֖ד בֶּן־שִׁשִּׁ֣ים שָׁנָ֑ה וְהָיָ֣ה עֶרְכְּךָ֗ חֲמִשִּׁ֛ים שֶׁ֥קֶל כֶּ֖סֶף בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ׃ שׁוֹוִי שֶׁל זָכָר מִגִּיל עֶשְׂרִים שָׁנָה וְעַד בֶּן שִׁשִּׁים שָׁנָה – הַשּׁוֹוִי הוּא כֶּסֶף בְּמִשְׁקַל חֲמִשִּׁים שֶׁקֶל בְּשֶׁקֶל הַקּוֹדֶשׁ (705 גְּרַם).
ד וְאִם־נְקֵבָ֖ה הִ֑וא וְהָיָ֥ה עֶרְכְּךָ֖ שְׁלֹשִׁ֥ים שָֽׁקֶל׃ וְאִם הִיא נְקֵבָה, הַשּׁוֹוִי יִהְיֶה שְׁלוֹשִׁים שֶׁקֶל (423 גְּרַם).
ה וְאִ֨ם מִבֶּן־חָמֵ֜שׁ שָׁנִ֗ים וְעַד֙ בֶּן־עֶשְׂרִ֣ים שָׁנָ֔ה וְהָיָ֧ה עֶרְכְּךָ֛ הַזָּכָ֖ר עֶשְׂרִ֣ים שְׁקָלִ֑ים וְלַנְּקֵבָ֖ה עֲשֶׂ֥רֶת שְׁקָלִֽים׃ וְאִם הוּא מִגִּיל חָמֵשׁ שָׁנִים וְעַד בֶּן עֶשְׂרִים שָׁנָה, הַשּׁוֹוִי יִהְיֶה: הַזָּכָר – עֶשְׂרִים שְׁקָלִים (282 גְּרַם), וְלַנְּקֵבָה – עֲשָׂרָה שְׁקָלִים (141 גְּרַם).
ו וְאִ֣ם מִבֶּן־חֹ֗דֶשׁ וְעַד֙ בֶּן־חָמֵ֣שׁ שָׁנִ֔ים וְהָיָ֤ה עֶרְכְּךָ֙ הַזָּכָ֔ר חֲמִשָּׁ֥ה שְׁקָלִ֖ים כָּ֑סֶף וְלַנְּקֵבָ֣ה עֶרְכְּךָ֔ שְׁלֹ֥שֶׁת שְׁקָלִ֖ים כָּֽסֶף׃ וְאִם מִגִּיל חוֹדֶשׁ וְעַד בֶּן חָמֵשׁ שָׁנִים, הַשּׁוֹוִי יִהְיֶה: הַזָּכָר – חֲמִשָּׁה שְׁקָלִים כֶּסֶף (70.5 גְּרַם), וְלַנְּקֵבָה – הַשּׁוֹוִי שְׁלוֹשָׁה שְׁקָלִים כֶּסֶף (42.3 גְּרַם).
ז וְ֠אִ֠ם מִבֶּן־שִׁשִּׁ֨ים שָׁנָ֤ה וָמַ֙עְלָה֙ אִם־זָכָ֔ר וְהָיָ֣ה עֶרְכְּךָ֔ חֲמִשָּׁ֥ה עָשָׂ֖ר שָׁ֑קֶל וְלַנְּקֵבָ֖ה עֲשָׂרָ֥ה שְׁקָלִֽים׃ וְאִם מִגִּיל שִׁשִּׁים שָׁנָה וָמַעְלָה: אִם זָכָר –הַשּׁוֹוִי יִהְיֶה חֲמִשָּׁה עָשָׂר שֶׁקֶל (211.5 גְּרַם), וְלַנְּקֵבָה – עֲשָׂרָה שְׁקָלִים (141 גְּרַם).
ח וְאִם־מָ֥ךְ הוּא֙ מֵֽעֶרְכֶּ֔ךָ וְהֶֽעֱמִידוֹ֙ לִפְנֵ֣י הַכֹּהֵ֔ן וְהֶעֱרִ֥יךְ אֹת֖וֹ הַכֹּהֵ֑ן עַל־פִּ֗י אֲשֶׁ֤ר תַּשִּׂיג֙ יַ֣ד הַנֹּדֵ֔ר יַעֲרִיכֶ֖נּוּ הַכֹּהֵֽן׃ וְאִם הוּא עָנִי מִלְּהַשִּׂיג אֶת הַשּׁוֹוִי הַמָּלֵא, יַעֲמִידוּ אוֹתוֹ לִפְנֵי הַכֹּהֵן וְהַכֹּהֵן יַעֲרִיךְ אֶת יְכָלְתּוֹ, כְּפִי יְכוֹלֶת הַנּוֹדֵר כָּךְ יַעֲרִיךְ לוֹ הַכֹּהֵן לְשַׁלֵּם.
ס
ט וְאִ֨ם־בְּהֵמָ֔ה אֲשֶׁ֨ר יַקְרִ֧יבוּ מִמֶּ֛נָּה קׇרְבָּ֖ן לַֽיהֹוָ֑ה כֹּל֩ אֲשֶׁ֨ר יִתֵּ֥ן מִמֶּ֛נּוּ לַיהֹוָ֖ה יִֽהְיֶה־קֹּֽדֶשׁ׃ וְאִם נָדַר לָתֵת בְּהֵמָה שֶׁמַּקְרִיבִים מִמֶּנָּה קָרְבָּן לַה', כָּל מַה שֶׁיֻּקְדַּשׁ מִמֶּנָּה לַה' יִהְיֶה מֻקְדָּשׁ לְקָרְבָּן.
י לֹ֣א יַחֲלִיפֶ֗נּוּ וְלֹֽא־יָמִ֥יר אֹת֛וֹ ט֥וֹב בְּרָ֖ע אוֹ־רַ֣ע בְּט֑וֹב וְאִם־הָמֵ֨ר יָמִ֤יר בְּהֵמָה֙ בִּבְהֵמָ֔ה וְהָֽיָה־ה֥וּא וּתְמוּרָת֖וֹ יִֽהְיֶה־קֹּֽדֶשׁ׃ הוּא לֹא יִפְדֶּה אוֹתוֹ וְלֹא יַחֲלִיף אוֹתוֹ לֹא טוֹב בְּרָע וְלֹא רַע בְּטוֹב, וְאִם יַחֲלִיף בְּהֵמָה בִּבְהֵמָה, גַּם הָרִאשׁוֹנָה וְגַם זוֹ שֶׁבִּמְקוֹמָה יִהְיוּ קוֹדֶשׁ.
יא וְאִם֙ כׇּל־בְּהֵמָ֣ה טְמֵאָ֔ה אֲ֠שֶׁ֠ר לֹא־יַקְרִ֧יבוּ מִמֶּ֛נָּה קׇרְבָּ֖ן לַֽיהֹוָ֑ה וְהֶֽעֱמִ֥יד אֶת־הַבְּהֵמָ֖ה לִפְנֵ֥י הַכֹּהֵֽן׃ וְאִם יַקְדִּישׁ בְּהֵמָה טְמֵאָה שֶׁלֹּא מַקְרִיבִים מִמֶּנָּה קָרְבָּן לַה' – יַעֲמִידוּ אֶת הַבְּהֵמָה לִפְנֵי הַכֹּהֵן.
יב וְהֶעֱרִ֤יךְ הַכֹּהֵן֙ אֹתָ֔הּ בֵּ֥ין ט֖וֹב וּבֵ֣ין רָ֑ע כְּעֶרְכְּךָ֥ הַכֹּהֵ֖ן כֵּ֥ן יִהְיֶֽה׃ הַכֹּהֵן יַעֲרִיךְ אֶת שׁוֹוְיָהּ לְפִי מַעֲלוֹתֶיהָ וְחֶסְרוֹנוֹתֶיהָ – כְּהַעֲרָכַת הַכֹּהֵן כָּךְ יִהְיֶה מְחִירָהּ.
יג וְאִם־גָּאֹ֖ל יִגְאָלֶ֑נָּה וְיָסַ֥ף חֲמִישִׁת֖וֹ עַל־עֶרְכֶּֽךָ׃ וְאִם הַמַּקְדִּישׁ בְּעַצְמוֹ יִפְדֶּה אוֹתָהּ, הוּא יוֹסִיף חֲמִישִׁית עַל מְחִירָהּ.
פָּרָשַׁת הַפִּדְיוֹנוֹת
יד וְאִ֗ישׁ כִּֽי־יַקְדִּ֨שׁ אֶת־בֵּית֥וֹ קֹ֙דֶשׁ֙ לַֽיהֹוָ֔ה וְהֶעֱרִיכוֹ֙ הַכֹּהֵ֔ן בֵּ֥ין ט֖וֹב וּבֵ֣ין רָ֑ע כַּאֲשֶׁ֨ר יַעֲרִ֥יךְ אֹת֛וֹ הַכֹּהֵ֖ן כֵּ֥ן יָקֽוּם׃ וְאִם אִישׁ יַקְדִּישׁ אֶת בֵּיתוֹ שֶׁיִּהְיֶה קוֹדֶשׁ לַה', הַכֹּהֵן יַעֲרִיךְ אֶת שׁוֹוְיוֹ לְפִי מַעֲלוֹתָיו וְחֶסְרוֹנוֹתָיו, כְּפִי שֶׁהַכֹּהֵן יַעֲרִיךְ כָּךְ יִהְיֶה מְחִירוֹ.
טו וְאִ֨ם־הַמַּקְדִּ֔ישׁ יִגְאַ֖ל אֶת־בֵּית֑וֹ וְ֠יָסַ֠ף חֲמִישִׁ֧ית כֶּֽסֶף־עֶרְכְּךָ֛ עָלָ֖יו וְהָ֥יָה לֽוֹ׃ וְאִם הַמַּקְדִּישׁ עַצְמוֹ יִפְדֶּה אֶת בֵּיתוֹ, הוּא יוֹסִיף חֲמִישִׁית עַל מְחִיר שׁוֹוְיוֹ, וְיִהְיֶה שֶׁלּוֹ.

חמישי (שביעי במחוברין)[עריכה]

טז וְאִ֣ם ׀ מִשְּׂדֵ֣ה אֲחֻזָּת֗וֹ יַקְדִּ֥ישׁ אִישׁ֙ לַֽיהֹוָ֔ה וְהָיָ֥ה עֶרְכְּךָ֖ לְפִ֣י זַרְע֑וֹ זֶ֚רַע חֹ֣מֶר שְׂעֹרִ֔ים בַּחֲמִשִּׁ֖ים שֶׁ֥קֶל כָּֽסֶף׃ וְאִם אִישׁ יַקְדִּישׁ לַה' מִשְּׂדֵה אֲחֻזָּתוֹ, הַמְּחִיר יִקָּבַע לְפִי כַּמּוּת הַתְּבוּאָה – שֶׁטַח שֶׁזּוֹרְעִים בּוֹ חוֹמֶר שְׂעוֹרִים (כּ216 לִיטֶר) יִפָּדֶה בַּחֲמִשִּׁים שֶׁקֶל כֶּסֶף (705 גְּרַם).
יז אִם־מִשְּׁנַ֥ת הַיֹּבֵ֖ל יַקְדִּ֣ישׁ שָׂדֵ֑הוּ כְּעֶרְכְּךָ֖ יָקֽוּם׃ אִם יַקְדִּישׁ אֶת שָׂדֵהוּ מִיַּד אַחֲרֵי שְׁנַת הַיּוֹבֵל, יִקָּבַע הַמְּחִיר הַזֶּה:
יח וְאִם־אַחַ֣ר הַיֹּבֵל֮ יַקְדִּ֣ישׁ שָׂדֵ֒הוּ֒ וְחִשַּׁב־ל֨וֹ הַכֹּהֵ֜ן אֶת־הַכֶּ֗סֶף עַל־פִּ֤י הַשָּׁנִים֙ הַנּ֣וֹתָרֹ֔ת עַ֖ד שְׁנַ֣ת הַיֹּבֵ֑ל וְנִגְרַ֖ע מֵֽעֶרְכֶּֽךָ׃ וְאִם יַקְדִּישׁ אֶת שָׂדֵהוּ אַחֲרֵי הַיּוֹבֵל הַכֹּהֵן יַפְחִית מֵהַמְּחִיר וִיחַשֵּׁב לוֹ אֶת הַכֶּסֶף רַק עַל פִּי מִסְפַּר הַשָּׁנִים שֶׁנִּשְׁאֲרוּ עַד שְׁנַת הַיּוֹבֵל.
יט וְאִם־גָּאֹ֤ל יִגְאַל֙ אֶת־הַשָּׂדֶ֔ה הַמַּקְדִּ֖ישׁ אֹת֑וֹ וְ֠יָסַ֠ף חֲמִשִׁ֧ית כֶּֽסֶף־עֶרְכְּךָ֛ עָלָ֖יו וְקָ֥ם לֽוֹ׃ וְאִם הַמַּקְדִּישׁ עַצְמוֹ יִפְדֶּה אֶת הַשָּׂדֶה, יוֹסִיף עָלָיו אֶת חֲמִישִׁית מְחִיר שׁוֹוְיוֹ, וְיִהְיֶה שֶׁלּוֹ.
כ וְאִם־לֹ֤א יִגְאַל֙ אֶת־הַשָּׂדֶ֔ה וְאִם־מָכַ֥ר אֶת־הַשָּׂדֶ֖ה לְאִ֣ישׁ אַחֵ֑ר לֹ֥א יִגָּאֵ֖ל עֽוֹד׃ וְאִם הוּא לֹא יִגְאַל אֶת הַשָּׂדֶה, אוֹ אִם הַהֶקְדֵּשׁ מָכַר אֶת הַשָּׂדֶה לְאִישׁ אַחֵר, אִי אֶפְשָׁר לִפְדּוֹתוֹ עוֹד.
כא וְהָיָ֨ה הַשָּׂדֶ֜ה בְּצֵאת֣וֹ בַיֹּבֵ֗ל קֹ֛דֶשׁ לַֽיהֹוָ֖ה כִּשְׂדֵ֣ה הַחֵ֑רֶם לַכֹּהֵ֖ן תִּהְיֶ֥ה אֲחֻזָּתֽוֹ׃ וּכְשֶׁהַשָּׂדֶה תֵּצֵא בַּיּוֹבֵל הִיא תִּהְיֶה קוֹדֶשׁ לַה' כְּמוֹ שָׂדֶה שֶׁהֻקְדְּשָׁה עֲבוּר כֹּהֲנִים, הִיא תִּהְיֶה לַכֹּהֵן לַאֲחֻזָּה.

ששי[עריכה]

כב וְאִם֙ אֶת־שְׂדֵ֣ה מִקְנָת֔וֹ אֲשֶׁ֕ר לֹ֖א מִשְּׂדֵ֣ה אֲחֻזָּת֑וֹ יַקְדִּ֖ישׁ לַֽיהֹוָֽה׃ וְאִם יַקְדִּישׁ לַה' שָׂדֶה קְנוּיָה שֶׁאֵינֶנָּה מִשְּׂדֵה אֲחֻזָּתוֹ,
כג וְחִשַּׁב־ל֣וֹ הַכֹּהֵ֗ן אֵ֚ת מִכְסַ֣ת הָֽעֶרְכְּךָ֔ עַ֖ד שְׁנַ֣ת הַיֹּבֵ֑ל וְנָתַ֤ן אֶת־הָעֶרְכְּךָ֙ בַּיּ֣וֹם הַה֔וּא קֹ֖דֶשׁ לַיהֹוָֽה׃ הַכֹּהֵן יְחַשֵּׁב לוֹ אֶת מְחִיר הַשִּׁמּוּשׁ בַּשָּׂדֶה עַד שְׁנַת הַיּוֹבֵל, וְהָאִישׁ יִתֵּן בַּיּוֹם הַהוּא אֶת הַמְּחִיר, שֶׁיִּהְיֶה קוֹדֶשׁ לַה'.
כד בִּשְׁנַ֤ת הַיּוֹבֵל֙ יָשׁ֣וּב הַשָּׂדֶ֔ה לַאֲשֶׁ֥ר קָנָ֖הוּ מֵאִתּ֑וֹ לַאֲשֶׁר־ל֖וֹ אֲחֻזַּ֥ת הָאָֽרֶץ׃ בִּשְׁנַת הַיּוֹבֵל הַשָּׂדֶה תַּחֲזוֹר לְזֶה שֶׁנִּקְנְתָה מִמֶּנּוּ – לְמִי שֶׁאֲחֻזַּת הָאָרֶץ שֶׁלּוֹ.
כה וְכׇ֨ל־עֶרְכְּךָ֔ יִהְיֶ֖ה בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ עֶשְׂרִ֥ים גֵּרָ֖ה יִהְיֶ֥ה הַשָּֽׁקֶל׃ וְכָל הַהַעֲרָכוֹת יִהְיוּ לְפִי שֶׁקֶל הַקּוֹדֶשׁ – עֶשְׂרִים גֵּרָה יִהְיֶה מִשְׁקַל הַשֶּׁקֶל (14.1 גְּרַם).
כו אַךְ־בְּכ֞וֹר אֲשֶׁר־יְבֻכַּ֤ר לַֽיהֹוָה֙ בִּבְהֵמָ֔ה לֹֽא־יַקְדִּ֥ישׁ אִ֖ישׁ אֹת֑וֹ אִם־שׁ֣וֹר אִם־שֶׂ֔ה לַֽיהֹוָ֖ה הֽוּא׃ אֲבָל בְּכוֹר בְּהֵמָה שֶׁהוּא קוֹדֶשׁ לַה', בֵּין אִם שׁוֹר וּבֵין אִם גְּדִי אוֹ טָלֶה, אִישׁ לֹא יוּכַל לְהַקְדִּישׁ אוֹתוֹ, כִּי הוּא שֶׁל ה'.
כז וְאִ֨ם בַּבְּהֵמָ֤ה הַטְּמֵאָה֙ וּפָדָ֣ה בְעֶרְכֶּ֔ךָ וְיָסַ֥ף חֲמִשִׁת֖וֹ עָלָ֑יו וְאִם־לֹ֥א יִגָּאֵ֖ל וְנִמְכַּ֥ר בְּעֶרְכֶּֽךָ׃ וְאִם יַקְדִּישׁ בְּהֵמָה טְמֵאָה, יִפְדֶּה אוֹתָהּ בְּשׁוֹוְיָהּ וְיוֹסִיף עָלֶיהָ חֲמִישִׁית, וְאִם הַמַּקְדִּישׁ לֹא יִפְדֶּה אוֹתָהּ יִמְכְּרוּ אוֹתָהּ בְּשׁוֹוְיָהּ.
כח אַךְ־כׇּל־חֵ֡רֶם אֲשֶׁ֣ר יַחֲרִם֩ אִ֨ישׁ לַֽיהֹוָ֜ה מִכׇּל־אֲשֶׁר־ל֗וֹ מֵאָדָ֤ם וּבְהֵמָה֙ וּמִשְּׂדֵ֣ה אֲחֻזָּת֔וֹ לֹ֥א יִמָּכֵ֖ר וְלֹ֣א יִגָּאֵ֑ל כׇּל־חֵ֕רֶם קֹֽדֶשׁ־קׇדָשִׁ֥ים ה֖וּא לַיהֹוָֽה׃ אֲבָל כָּל הֶקְדֵּשׁ עֲבוּר כֹּהֲנִים שֶׁאִישׁ יַפְרִישׁ לַה' מִכָּל הַמָּמוֹן שֶׁלּוֹ – מֵאָדָם וּבְהֵמָה וּמִשְּׂדֵה אֲחֻזָּתוֹ – לֹא יִמָּכֵר וְלֹא יִפָּדֶה, כָּל הֶקְדֵּשׁ עֲבוּר כֹּהֲנִים הוּא קוֹדֶשׁ קָדָשִׁים לַה'.

שביעי[עריכה]

כט כׇּל־חֵ֗רֶם אֲשֶׁ֧ר יׇחֳרַ֛ם מִן־הָאָדָ֖ם לֹ֣א יִפָּדֶ֑ה מ֖וֹת יוּמָֽת׃ כָּל אָדָם שֶׁיֻּחְרָם לְמָוֶת לֹא יִפָּדֶה, אֶלָּא יוּמַת.
ל וְכׇל־מַעְשַׂ֨ר הָאָ֜רֶץ מִזֶּ֤רַע הָאָ֙רֶץ֙ מִפְּרִ֣י הָעֵ֔ץ לַיהֹוָ֖ה ה֑וּא קֹ֖דֶשׁ לַֽיהֹוָֽה׃ וְכָל מַעֲשֵׂר שֶׁיֻּפְרַשׁ מִגִּדּוּלֵי הָאָרֶץ – מִזֶּרַע הָאָרֶץ אוֹ מִפֵּרוֹת הָעֵץ, הוּא לַה' – קוֹדֶשׁ לַה'.
לא וְאִם־גָּאֹ֥ל יִגְאַ֛ל אִ֖ישׁ מִמַּֽעַשְׂר֑וֹ חֲמִשִׁית֖וֹ יֹסֵ֥ף עָלָֽיו׃ וְאִם אִישׁ יִפְדֶּה חֵלֶק מִמַּעַשְׂרוֹ, הוּא יוֹסִיף עָלָיו חֲמִישִׁית מִמְּחִירוֹ.

מפטיר[עריכה]

לב וְכׇל־מַעְשַׂ֤ר בָּקָר֙ וָצֹ֔אן כֹּ֥ל אֲשֶׁר־יַעֲבֹ֖ר תַּ֣חַת הַשָּׁ֑בֶט הָֽעֲשִׂירִ֕י יִֽהְיֶה־קֹּ֖דֶשׁ לַֽיהֹוָֽה׃ וְכָל מַעֲשֵׂר שֶׁל בָּקָר וְצֹאן – מִכָּל אֵלֶּה שֶׁיַּעַבְרוּ תַּחַת מַקֵּל הַסְּפִירָה – הָעֲשִׂירִי יִהְיֶה קוֹדֶשׁ לַה'.
לג לֹ֧א יְבַקֵּ֛ר בֵּֽין־ט֥וֹב לָרַ֖ע וְלֹ֣א יְמִירֶ֑נּוּ וְאִם־הָמֵ֣ר יְמִירֶ֔נּוּ וְהָֽיָה־ה֧וּא וּתְמוּרָת֛וֹ יִֽהְיֶה־קֹּ֖דֶשׁ לֹ֥א יִגָּאֵֽל׃ לֹא יִבְדּוֹק לְהַבְחִין אִם הוּא טוֹב אוֹ רַע וְלֹא יַחֲלִיף אוֹתוֹ, וְאִם יַחֲלִיף אוֹתוֹ – הוּא וּפִדְיוֹנוֹ יִהְיוּ קוֹדֶשׁ וְאִי אֶפְשָׁר לִפְדּוֹת אוֹתָם.
לד אֵ֣לֶּה הַמִּצְוֺ֗ת אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֶת־מֹשֶׁ֖ה אֶל־בְּנֵ֣י יִשְׂרָאֵ֑ל בְּהַ֖ר סִינָֽי׃ אֵלֶּה הַמִּצְווֹת שֶׁה' צִוָּה אֶת מֹשֶׁה אֶל בְּנֵי יִשְׂרָאֵל בְּהַר סִינַי.
