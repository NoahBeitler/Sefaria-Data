ראשון[עריכה]

פ פָּרָשַׁת שִׁירַת הַאֲזִינוּ
לב א הַאֲזִ֥ינוּ הַשָּׁמַ֖יִם וַאֲדַבֵּ֑רָה  וְתִשְׁמַ֥ע הָאָ֖רֶץ אִמְרֵי־פִֽי׃ הַאֲזִינוּ הַשָּׁמַיִם וַאֲדַבֵּר, וְהָאָרֶץ תִּשְׁמַע אֶת דִּבְרֵי פִּי.
ב יַעֲרֹ֤ף כַּמָּטָר֙ לִקְחִ֔י  תִּזַּ֥ל כַּטַּ֖ל אִמְרָתִ֑י
כִּשְׂעִירִ֣ם עֲלֵי־דֶ֔שֶׁא  וְכִרְבִיבִ֖ים עֲלֵי־עֵֽשֶׂב׃ 		דְּבַר הַמּוּסָר שֶׁלִּי יִנְטוֹף כְּמוֹ מָטָר מַרְוֶה, אִמְרָתִי תִּנְזוֹל כְּמוֹ הַטַּל, כְּמוֹ טִפּוֹת זְעִירוֹת עַל דֶּשֶׁא, וּכְטִפּוֹת גְּדוֹלוֹת עַל עֵשֶׂב.
ג כִּ֛י שֵׁ֥ם יְהֹוָ֖ה אֶקְרָ֑א  הָב֥וּ גֹ֖דֶל לֵאלֹהֵֽינוּ׃ כְּשֶׁאֶקְרָא בְּשֵׁם ה', תְּנוּ גְּדֻלָּה לֵאלֹהֵינוּ.
ד הַצּוּר֙ תָּמִ֣ים פׇּֽעֳל֔וֹ  כִּ֥י כׇל־דְּרָכָ֖יו מִשְׁפָּ֑ט
אֵ֤ל אֱמוּנָה֙ וְאֵ֣ין עָ֔וֶל  צַדִּ֥יק וְיָשָׁ֖ר הֽוּא׃ 		הָאֵל הֶחָזָק פְּעֻלּוֹתָיו צוֹדְקוֹת כִּי כָּל דְּרָכָיו הֵם בְּיוֹשֶׁר, הוּא אֵל נֶאֱמָן וְאֵין בְּמַעֲשָׂיו עָוֶל, הוּא צַדִּיק וְיָשָׁר.
ה שִׁחֵ֥ת ל֛וֹ לֹ֖א בָּנָ֣יו מוּמָ֑ם  דּ֥וֹר עִקֵּ֖שׁ וּפְתַלְתֹּֽל׃ וְכִי עַם יִשְׂרָאֵל מַפְרִיעַ בַּעֲווֹנוֹ לַה'? לֹא! אֵין זֶה אֶלָּא הַמּוּם שֶׁל בָּנָיו – דּוֹר עַקְשָׁן וּמִתְחַכֵּם.
ו הַ לְיְהֹוָה֙‏[1] תִּגְמְלוּ־זֹ֔את  עַ֥ם נָבָ֖ל וְלֹ֣א חָכָ֑ם
הֲלוֹא־הוּא֙ אָבִ֣יךָ קָּנֶ֔ךָ  ה֥וּא עָשְׂךָ֖ וַֽיְכֹנְנֶֽךָ׃ 		לָמָּה אַתֶּם גּוֹמְלִים כָּךְ לַה', עַם כְּפוּי טוֹבָה וְלֹא חָכָם?! הֲרֵי הוּא אָבִיךָ וְאַתָּה שֶׁלּוֹ, הוּא עָשָׂה אוֹתְךָ וּבִסֵּס אוֹתְךָ!

שני[עריכה]

ז זְכֹר֙ יְמ֣וֹת עוֹלָ֔ם בִּ֖ינוּ שְׁנ֣וֹת דֹּר־וָדֹ֑ר  שְׁאַ֤ל אָבִ֙יךָ֙ וְיַגֵּ֔דְךָ זְקֵנֶ֖יךָ וְיֹ֥אמְרוּ לָֽךְ׃ זְכוֹר אֶת הַיָּמִים שֶׁהָיוּ מֵאָז וּמֵעוֹלָם, הִתְבּוֹנְנוּ בַּשָּׁנִים שֶׁל כָּל הַדּוֹרוֹת, שְׁאַל אֶת אָבִיךָ וְהוּא יַגִּיד לְךָ, שְׁאַל אֶת זְקֵנֶיךָ וְהֵם יֹאמְרוּ לְךָ.
ח בְּהַנְחֵ֤ל עֶלְיוֹן֙ גּוֹיִ֔ם  בְּהַפְרִיד֖וֹ בְּנֵ֣י אָדָ֑ם
יַצֵּב֙ גְּבֻלֹ֣ת עַמִּ֔ים  לְמִסְפַּ֖ר בְּנֵ֥י יִשְׂרָאֵֽל׃ 		בִּזְמַן שֶׁהָאֵל הָעֶלְיוֹן חִלֵּק נַחֲלוֹת לַגּוֹיִם, כְּשֶׁהִפְרִיד בֵּין עַמֵּי בְּנֵי אָדָם, הוּא הִצִּיב אֶת שִׁבְעִים גְּבוּלוֹת הָעַמִּים כְּמִסְפַּר בְּנֵי יִשְׂרָאֵל.
ט כִּ֛י חֵ֥לֶק יְהֹוָ֖ה עַמּ֑וֹ  יַעֲקֹ֖ב חֶ֥בֶל נַחֲלָתֽוֹ׃ כִּי הַחֵלֶק שֶׁל ה' זֶה עַמּוֹ, יַעֲקֹב הוּא כְּמוֹ חֶבֶל הָאָרֶץ שֶׁל נַחֲלָתוֹ.
י יִמְצָאֵ֙הוּ֙ בְּאֶ֣רֶץ מִדְבָּ֔ר  וּבְתֹ֖הוּ יְלֵ֣ל יְשִׁמֹ֑ן
יְסֹבְבֶ֙נְהוּ֙ יְב֣וֹנְנֵ֔הוּ  יִצְּרֶ֖נְהוּ כְּאִישׁ֥וֹן עֵינֽוֹ׃ 		הוּא מָצָא אוֹתוֹ בְּאֶרֶץ מִדְבָּר, בְּמָקוֹם שֶׁל יְלָלוֹת תַּנִּים וּמִדְבַּר שְׁמָמָה, הוּא עָמַד סְבִיבוֹ בַּמִּדְבָּר לְהָגֵן עָלָיו, נָתַן בּוֹ בִּינָה, וְשָׁמַר אוֹתוֹ כְּמוֹ אֶת הָאִישׁוֹן שֶׁל עֵינוֹ.
יא כְּנֶ֙שֶׁר֙ יָעִ֣יר קִנּ֔וֹ עַל־גּוֹזָלָ֖יו יְרַחֵ֑ף  יִפְרֹ֤שׂ כְּנָפָיו֙ יִקָּחֵ֔הוּ יִשָּׂאֵ֖הוּ עַל־אֶבְרָתֽוֹ׃ כְּמוֹ נֶשֶׁר שֶׁמֵּעִיר אֶת קִנּוֹ בַּעֲדִינוּת, מְרַחֵף עַל גּוֹזָלָיו לִשְׁמוֹר עֲלֵיהֶם, פּוֹרֵשׂ אֶת כְּנָפָיו וְלוֹקֵחַ אוֹתָם וְנוֹשֵׂא אוֹתָם עַל כְּנָפָיו.
יב יְהֹוָ֖ה בָּדָ֣ד יַנְחֶ֑נּוּ  וְאֵ֥ין עִמּ֖וֹ אֵ֥ל נֵכָֽר׃ כָּךְ ה' לְבַדּוֹ מוֹבִיל אוֹתוֹ, וְאֵין אִתּוֹ אֵל אַחֵר.

שלישי[עריכה]

יג יַרְכִּבֵ֙הוּ֙ עַל־בָּ֣מֳתֵי במותי אָ֔רֶץ  וַיֹּאכַ֖ל תְּנוּבֹ֣ת שָׂדָ֑י
וַיֵּנִקֵ֤הֽוּ דְבַשׁ֙ מִסֶּ֔לַע  וְשֶׁ֖מֶן מֵחַלְמִ֥ישׁ צֽוּר׃ 		ה' הוֹשִׁיב אוֹתוֹ בְּאֶרֶץ חֲשׁוּבָה, וְהוּא אָכַל יְבוּלֵי שָׂדֶה, הֵנִיק אוֹתוֹ דְּבַשׁ מִסֶּלַע, וְשֶׁמֶן מִסֶּלַע קָשֶׁה.
יד חֶמְאַ֨ת בָּקָ֜ר וַחֲלֵ֣ב צֹ֗אן  עִם־חֵ֨לֶב כָּרִ֜ים וְאֵילִ֤ים
בְּנֵֽי־בָשָׁן֙ וְעַתּוּדִ֔ים  עִם־חֵ֖לֶב כִּלְי֣וֹת חִטָּ֑ה
וְדַם־עֵנָ֖ב תִּשְׁתֶּה־חָֽמֶר׃ 		חֶמְאָה מִבָּקָר וְחָלָב מִצֹּאן, עִם שׁוּמַן כְּבָשִׂים וְאֵילִים מֵהַבָּשָׁן וּתְיָשִׁים, עִם חִטָּה שְׁמֵנָה, וּמֵהַמִּיץ שֶׁל הָעֲנָבִים שָׁתָה יַיִן.
טו וַיִּשְׁמַ֤ן יְשֻׁרוּן֙ וַיִּבְעָ֔ט
שָׁמַ֖נְתָּ עָבִ֣יתָ כָּשִׂ֑יתָ  וַיִּטֹּשׁ֙ אֱל֣וֹהַּ עָשָׂ֔הוּ
וַיְנַבֵּ֖ל צ֥וּר יְשֻׁעָתֽוֹ׃ 		יִשְׂרָאֵל הִשְׁמִין וּבָעַט, הִשְׁמַנְתָּ, נַעֲשֵׂיתָ עָבֶה, וְהִתְכַּסִּיתָ בְּשׁוּמָן, הוּא נָטַשׁ אֶת הָאֱלֹהִים שֶׁבָּרָא אוֹתוֹ, וּבִזָּה אֶת הָאֵל הֶחָזָק שֶׁמַּצִּילוֹ.
טז יַקְנִאֻ֖הוּ בְּזָרִ֑ים
בְּתוֹעֵבֹ֖ת יַכְעִיסֻֽהוּ׃ 		הִרְגִּיזוּ אוֹתוֹ בַּעֲבוֹדַת אֵלִים זָרִים – הִכְעִיסוּ אוֹתוֹ בְּתוֹעֵבוֹת.
יז יִזְבְּח֗וּ לַשֵּׁדִים֙ לֹ֣א אֱלֹ֔הַּ
אֱלֹהִ֖ים לֹ֣א יְדָע֑וּם  חֲדָשִׁים֙ מִקָּרֹ֣ב בָּ֔אוּ
לֹ֥א שְׂעָר֖וּם אֲבֹתֵיכֶֽם׃ 		שָׁחֲטוּ קָרְבָּנוֹת לַשֵּׁדִים – שֶׁאֵינָם אֵל, לְאֵלִים שֶׁלֹּא הִכִּירוּ מֵעוֹלָם, חֲדָשִׁים שֶׁבָּאוּ לָאַחֲרוֹנָה, אֲבוֹתֵיכֶם לֹא הִתְעַסְּקוּ בָּהֶם.
יח צ֥וּר יְלָדְךָ֖ תֶּ֑שִׁי
וַתִּשְׁכַּ֖ח אֵ֥ל מְחֹלְלֶֽךָ׃ 		עָזַבְתָּ אֶת הָאֵל הֶחָזָק שֶׁיָּלַד אוֹתְךָ, שָׁכַחְתָּ אֶת הָאֵל שֶׁעָשָׂה אוֹתְךָ.

רביעי[עריכה]

יט וַיַּ֥רְא יְהֹוָ֖ה וַיִּנְאָ֑ץ
מִכַּ֥עַס בָּנָ֖יו וּבְנֹתָֽיו׃ 		ה' רָאָה וְכָעַס מִמַּה שֶּׁהִכְעִיסוּהוּ בָּנָיו וּבְנוֹתָיו.
כ וַיֹּ֗אמֶר אַסְתִּ֤ירָה פָנַי֙ מֵהֶ֔ם
אֶרְאֶ֖ה מָ֣ה אַחֲרִיתָ֑ם  כִּ֣י ד֤וֹר תַּהְפֻּכֹת֙ הֵ֔מָּה
בָּנִ֖ים לֹא־אֵמֻ֥ן בָּֽם׃ 		אָמַר: אַסְתִּיר אֶת פָּנַי מֵהֶם, אֲנִי צוֹפֶה מַה יִּהְיֶה עֲתִידָם, כִּי הֵם דּוֹר הֲפַכְפָּךְ – בָּנִים שֶׁאֵין בָּהֶם אֵמוּן.
כא הֵ֚ם קִנְא֣וּנִי בְלֹא־אֵ֔ל
כִּעֲס֖וּנִי בְּהַבְלֵיהֶ֑ם  וַֽאֲנִי֙ אַקְנִיאֵ֣ם בְּלֹא־עָ֔ם
בְּג֥וֹי נָבָ֖ל אַכְעִיסֵֽם׃ 		הֵם הִרְגִּיזוּ אוֹתִי בֶּאֱלִיל שֶׁאֵין בּוֹ כֹּחַ – הִכְעִיסוּ אוֹתִי בְּהַבְלֵיהֶם, לָכֵן אֲנִי אַרְגִּיז אוֹתָם בְּשֵׁבֶט שֶׁאֵינוֹ עָם – אַכְעִיס אוֹתָם בְּעַם רַע.
כב כִּי־אֵשׁ֙ קָדְחָ֣ה בְאַפִּ֔י
וַתִּיקַ֖ד עַד־שְׁא֣וֹל תַּחְתִּ֑ית  וַתֹּ֤אכַל אֶ֙רֶץ֙ וִֽיבֻלָ֔הּ
וַתְּלַהֵ֖ט מוֹסְדֵ֥י הָרִֽים׃ 		כִּי מִכַּעֲסִי בָּעֲרָה אֵשׁ וְדָלְקָה עַד עוֹמֶק הַתְּהוֹם, הִשְׁמִידָה אֶרֶץ וִיבוּלָהּ וְשָׂרְפָה יְסוֹדוֹת הָרִים.
כג אַסְפֶּ֥ה עָלֵ֖ימוֹ רָע֑וֹת
חִצַּ֖י אֲכַלֶּה־בָּֽם׃ 		אוֹסִיף עֲלֵיהֶם רָעוֹת, אִירָה בָּהֶם אֶת כָּל חִצַּי.
כד מְזֵ֥י רָעָ֛ב וּלְחֻ֥מֵי רֶ֖שֶׁף
וְקֶ֣טֶב מְרִירִ֑י  וְשֶׁן־בְּהֵמֹת֙ אֲשַׁלַּח־בָּ֔ם
עִם־חֲמַ֖ת זֹחֲלֵ֥י עָפָֽר׃ 		הֵם יִהְיוּ מֻכֵּי רָעָב וְחוֹלִים בְּחוֹם גָּבוֹהַּ וּמָוֶת מַר, וַאֲשַׁסֶּה בָּהֶם שֵׁן שֶׁל חַיּוֹת רָעוֹת עִם אֶרֶס שֶׁל זוֹחֲלֵי עָפָר.
כה מִחוּץ֙ תְּשַׁכֶּל־חֶ֔רֶב וּמֵחֲדָרִ֖ים אֵימָ֑ה גַּם־בָּחוּר֙
גַּם־בְּתוּלָ֔ה יוֹנֵ֖ק עִם־אִ֥ישׁ שֵׂיבָֽה׃ 		מִחוּץ לָעִיר תִּהְיֶה מִלְחָמָה שֶׁתַּהֲרוֹג אֶת בְּנֵיהֶם, וּבַחֲדָרִים הַפְּנִימִיִּים תִּהְיֶה אֵימָה, יָמוּתוּ גַּם בָּחוּר וְגַם בְּתוּלָה, יוֹנֵק עִם אִישׁ זָקֵן.
כו אָמַ֖רְתִּי אַפְאֵיהֶ֑ם
אַשְׁבִּ֥יתָה מֵאֱנ֖וֹשׁ זִכְרָֽם׃ 		אָמַרְתִּי: אַשְׁמִיד אוֹתָם בְּרָגְזִי! לֹא אַשְׁאִיר מֵהֶם זֵכֶר בֵּין הָאֲנָשִׁים.
כז לוּלֵ֗י כַּ֤עַס אוֹיֵב֙ אָג֔וּר
פֶּֽן־יְנַכְּר֖וּ צָרֵ֑ימוֹ  פֶּן־יֹֽאמְרוּ֙ יָדֵ֣נוּ רָ֔מָה
וְלֹ֥א יְהֹוָ֖ה פָּעַ֥ל כׇּל־זֹֽאת׃ 		כָּךְ הָיִיתִי עוֹשֶׂה אִם לֹא הָיִיתִי חוֹשֵׁש שֶׁהָאוֹיֵב יַכְעִיס אוֹתִי, שֶׁמָּא יִתְגָּאוּ צוֹרְרֵיהֶם – שֶׁמָּא יֹאמְרוּ: יָדֵינוּ חֲזָקָה, וְלֹא ה' עָשָׂה אֶת כָּל זֶה.
כח כִּי־ג֛וֹי אֹבַ֥ד עֵצ֖וֹת הֵ֑מָּה
וְאֵ֥ין בָּהֶ֖ם תְּבוּנָֽה׃ 		כִּי הָאוֹיְבִים הֵם עַם חֲסַר חָכְמָה, וְאֵין בָּהֶם תְּבוּנָה.

חמישי[עריכה]

כט ל֥וּ חָכְמ֖וּ יַשְׂכִּ֣ילוּ זֹ֑את
יָבִ֖ינוּ לְאַחֲרִיתָֽם׃ 		אִלּוּ הָיוּ חֲכָמִים הָיוּ יוֹדְעִים זֹאת, הָיוּ מְבִינִים שֶׁאִם יֶחֶטְאוּ יִקְרֶה כָּךְ גַּם לָהֶם בְּסוֹפָם.
ל אֵיכָ֞ה יִרְדֹּ֤ף אֶחָד֙ אֶ֔לֶף
וּשְׁנַ֖יִם יָנִ֣יסוּ רְבָבָ֑ה  אִם־לֹא֙ כִּֽי־צוּרָ֣ם מְכָרָ֔ם
וַֽיהֹוָ֖ה הִסְגִּירָֽם׃ 		אֵיךְ יִתָּכֵן שֶׁאֶחָד מֵהֶם רוֹדֵף אֶלֶף וּשְׁנַיִם מַבְרִיחִים עֲשֶׂרֶת אֲלָפִים, אִם לֹא שֶׁהָאֵל הֶחָזָק שֶׁל יִשְׂרָאֵל מָכַר אוֹתָם, וַה' הִסְגִּיר אוֹתָם?!
לא כִּ֛י לֹ֥א כְצוּרֵ֖נוּ צוּרָ֑ם
וְאֹיְבֵ֖ינוּ פְּלִילִֽים׃ 		הֲרֵי אֱלִילָם אֵינוֹ כְּמוֹ הָאֵל הֶחָזָק שֶׁלָּנוּ, וּבְכָל זֹאת אוֹיְבֵינוּ עוֹשִׂים בָּנוּ שְׁפָטִים.
לב כִּֽי־מִגֶּ֤פֶן סְדֹם֙ גַּפְנָ֔ם
וּמִשַּׁדְמֹ֖ת עֲמֹרָ֑ה  עֲנָבֵ֙מוֹ֙ עִנְּבֵי־ר֔וֹשׁ
אַשְׁכְּלֹ֥ת מְרֹרֹ֖ת לָֽמוֹ׃ 		אֲבָל לְבַסּוֹף הַגֶּפֶן שֶׁלָּהֶם תֵּעָנֵשׁ כְּמוֹ גֶּפֶן סְדוֹם וּשְׂדוֹת עֲמוֹרָה, עִנְּבֵיהֶם יִהְיוּ כְּעִנְּבֵי הָעֵשֶׂב הַמָּר, יִהְיוּ לָהֶם אֶשְׁכּוֹלוֹת מָרִים.
לג חֲמַ֥ת תַּנִּינִ֖ם יֵינָ֑ם
וְרֹ֥אשׁ פְּתָנִ֖ים אַכְזָֽר׃ 		יֵינָם יִהְיֶה כְּמוֹ אֶרֶס נְחָשִׁים, וּכְרַעַל פֶּתֶן אַכְזָר.
לד הֲלֹא־ה֖וּא כָּמֻ֣ס עִמָּדִ֑י
חָת֖וּם בְּאוֹצְרֹתָֽי׃ 		הֲרֵי זֶה שָׁמוּר אֶצְלִי, נָעוּל בַּמַּחְסָנִים שֶׁלִּי –
לה לִ֤י נָקָם֙ וְשִׁלֵּ֔ם
לְעֵ֖ת תָּמ֣וּט רַגְלָ֑ם  כִּ֤י קָרוֹב֙ י֣וֹם אֵידָ֔ם
וְחָ֖שׁ עֲתִדֹ֥ת לָֽמוֹ׃ 		הַנְּקָמָה וְהַתַּגְמוּל מוּכָנִים אֶצְלִי לַזְּמַן שֶׁרַגְלָם תִּתְמוֹטֵט, שֶׁכֵּן קָרוֹב יוֹם שְׁבִירָתָם, וַאֲמַהֵר אֶת הֶעָתִיד לָבוֹא עֲלֵיהֶם.
לו כִּֽי־יָדִ֤ין יְהֹוָה֙ עַמּ֔וֹ
וְעַל־עֲבָדָ֖יו יִתְנֶחָ֑ם  כִּ֤י יִרְאֶה֙ כִּֽי־אָ֣זְלַת יָ֔ד
וְאֶ֖פֶס עָצ֥וּר וְעָזֽוּב׃ 		כְּשֶׁה' יָדוּן אֶת עַמּוֹ וְיִתְחָרֵט מִלְּהַעֲנִישׁ אֶת עֲבָדָיו, כְּשֶׁיִּרְאֶה שֶׁעָזְבָה אוֹתָם הַיָּד הַמְּסַיַּעַת וְאֵין בָּהֶם נֶעֱזָר וְנִתְמָךְ.
לז וְאָמַ֖ר אֵ֣י אֱלֹהֵ֑ימוֹ
צ֖וּר חָסָ֥יוּ בֽוֹ׃ 		ה' יֹאמַר: אֵיפֹה אֱלִילֵיהֶם? – אֱלִילָם הֶחָזָק שֶׁבָּטְחוּ בּוֹ?!
לח אֲשֶׁ֨ר חֵ֤לֶב זְבָחֵ֙ימוֹ֙ יֹאכֵ֔לוּ
יִשְׁתּ֖וּ יֵ֣ין נְסִיכָ֑ם  יָק֙וּמוּ֙ וְיַעְזְרֻכֶ֔ם
יְהִ֥י עֲלֵיכֶ֖ם סִתְרָֽה׃ 		אֱלִילֵיהֶם הָאוֹכְלִים אֶת חֵלֶב זִבְחֵיהֶם וְשׁוֹתִים אֶת הַיַּיִן שֶׁשָּׁפְכוּ לָהֶם? שֶׁיָּקוּמוּ וְיַעַזְרוּ לָכֶם! שֶׁיִּהְיוּ לָכֶם לְמִסְתּוֹר מֵאוֹיֵב!
לט רְא֣וּ ׀ עַתָּ֗ה כִּ֣י אֲנִ֤י אֲנִי֙ ה֔וּא
וְאֵ֥ין אֱלֹהִ֖ים עִמָּדִ֑י  אֲנִ֧י אָמִ֣ית וַאֲחַיֶּ֗ה
מָחַ֙צְתִּי֙ וַאֲנִ֣י אֶרְפָּ֔א  וְאֵ֥ין מִיָּדִ֖י מַצִּֽיל׃ 		רְאוּ עַכְשָׁיו שֶׁאֲנִי אֲנִי הוּא, וְאֵין אֱלֹהִים אֲחֵרִים יַחַד אִתִּי! רַק אֲנִי אָמִית וַאֲחַיֶּה, אֲנִי מָחַצְתִּי וַאֲנִי אֲרַפֵּא, וְאֵין מִי שֶׁיָּכוֹל לְהַצִּיל מִיָּדִי.

ששי[עריכה]

מ כִּֽי־אֶשָּׂ֥א אֶל־שָׁמַ֖יִם יָדִ֑י  וְאָמַ֕רְתִּי חַ֥י אָנֹכִ֖י לְעֹלָֽם׃ כְּשֶׁאָרִים אֶל הַשָּׁמַיִם אֶת יָדִי, וְאֶשָּׁבַע בִּשְׁמִי – 'חַי אֲנִי לְעוֹלָם' שֶׁכָּךְ אֶעֱשֶׂה:
מא אִם־שַׁנּוֹתִי֙ בְּרַ֣ק חַרְבִּ֔י  וְתֹאחֵ֥ז בְּמִשְׁפָּ֖ט יָדִ֑י
אָשִׁ֤יב נָקָם֙ לְצָרָ֔י  וְלִמְשַׂנְאַ֖י אֲשַׁלֵּֽם׃ 		אַשְׁחִיז אֶת חַרְבִּי הַמַּבְרִיקָה וְיָדִי תֹּאחַז בְּמִדַּת הַדִּין, אָשִׁיב נְקָמָה לְצוֹרְרַי וַאֲשַׁלֵּם לְשׂוֹנְאַי.
מב אַשְׁכִּ֤יר חִצַּי֙ מִדָּ֔ם  וְחַרְבִּ֖י תֹּאכַ֣ל בָּשָׂ֑ר
מִדַּ֤ם חָלָל֙ וְשִׁבְיָ֔ה  מֵרֹ֖אשׁ פַּרְע֥וֹת אוֹיֵֽב׃ 		אַרְוֶה אֶת חִצַּי בְּדָם עַד שֶׁיִּשְׁתַּכְּרוּ, וְחַרְבִּי תֹּאכַל בָּשָׂר, מִדַּם הֲרוּגִים וּשְׁבוּיִים, וּמֵהָרָאשִׁים הַשְּׁסוּעִים שֶׁל הָאוֹיֵב.
מג הַרְנִ֤ינוּ גוֹיִם֙ עַמּ֔וֹ  כִּ֥י דַם־עֲבָדָ֖יו יִקּ֑וֹם
וְנָקָם֙ יָשִׁ֣יב לְצָרָ֔יו  וְכִפֶּ֥ר אַדְמָת֖וֹ עַמּֽוֹ׃ 		שַׁבְּחוּ גּוֹיִם אֶת עַמּוֹ, כִּי ה' יִנְקוֹם אֶת דַּם עֲבָדָיו, וְיָשִׁיב נְקָמָה לְצוֹרְרָיו וְהוּא יְכַפֵּר עַל אַדְמָתוֹ וְעַמּוֹ.

שביעי[עריכה]

פ פָּרָשַׁת הַצִּוּוּי עַל הַשִּׁירָה
מד וַיָּבֹ֣א מֹשֶׁ֗ה וַיְדַבֵּ֛ר אֶת־כׇּל־דִּבְרֵ֥י הַשִּׁירָֽה־הַזֹּ֖את בְּאׇזְנֵ֣י הָעָ֑ם ה֖וּא וְהוֹשֵׁ֥עַ בִּן־נֽוּן׃ מֹשֶׁה בָּא, וְדִבֵּר אֶת כָּל דִּבְרֵי הַשִּׁירָה הַזֹּאת בְּאָזְנֵי הָעָם, הוּא וְהוֹשֵׁעַ בִּן נוּן.
מה וַיְכַ֣ל מֹשֶׁ֗ה לְדַבֵּ֛ר אֶת־כׇּל־הַדְּבָרִ֥ים הָאֵ֖לֶּה אֶל־כׇּל־יִשְׂרָאֵֽל׃ מֹשֶׁה גָּמַר לְדַבֵּר אֶת כָּל הַדְּבָרִים הָאֵלֶּה אֶל כָּל יִשְׂרָאֵל.
מו וַיֹּ֤אמֶר אֲלֵהֶם֙ שִׂ֣ימוּ לְבַבְכֶ֔ם לְכׇ֨ל־הַדְּבָרִ֔ים אֲשֶׁ֧ר אָנֹכִ֛י מֵעִ֥יד בָּכֶ֖ם הַיּ֑וֹם אֲשֶׁ֤ר תְּצַוֻּם֙ אֶת־בְּנֵיכֶ֔ם לִשְׁמֹ֣ר לַעֲשׂ֔וֹת אֶת־כׇּל־דִּבְרֵ֖י הַתּוֹרָ֥ה הַזֹּֽאת׃ וְאָמַר לָהֶם: שִׂימוּ לִבְּכֶם לְכָל הַדְּבָרִים שֶׁאֲנִי מַזְהִיר אֶתְכֶם הַיּוֹם, כְּדֵי שֶׁתְּצַוּוּ אוֹתָם אֶת בְּנֵיכֶם, לְהַקְפִּיד לְקַיֵּם אֶת כָּל דִּבְרֵי הַתּוֹרָה הַזֹּאת.
מז כִּ֠י לֹֽא־דָבָ֨ר רֵ֥ק הוּא֙ מִכֶּ֔ם כִּי־ה֖וּא חַיֵּיכֶ֑ם וּבַדָּבָ֣ר הַזֶּ֗ה תַּאֲרִ֤יכוּ יָמִים֙ עַל־הָ֣אֲדָמָ֔ה אֲשֶׁ֨ר אַתֶּ֜ם עֹבְרִ֧ים אֶת־הַיַּרְדֵּ֛ן שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ כִּי הוּא לֹא דָּבָר רֵיק מִתּוֹכֶן עֲבוּרְכֶם, אֶלָּא הוּא חַיֵּיכֶם, וּבִזְכוּת הַדָּבָר הַזֶּה תִּתְקַיְּמוּ זְמַן רָב עַל הָאֲדָמָה שֶׁאַתֶּם עוֹבְרִים אֶת הַיַּרְדֵּן לְשָׁם לִכְבּוֹשׁ אוֹתָהּ.

מפטיר[עריכה]

פ פָּרָשַׁת צִוּוּי הָעֲלִיָּה לְהַר נְבוֹ
מח וַיְדַבֵּ֤ר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה בְּעֶ֛צֶם הַיּ֥וֹם הַזֶּ֖ה לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה בַּיּוֹם הַזֶּה עַצְמוֹ וְאָמַר:
מט עֲלֵ֡ה אֶל־הַר֩ הָעֲבָרִ֨ים הַזֶּ֜ה הַר־נְב֗וֹ אֲשֶׁר֙ בְּאֶ֣רֶץ מוֹאָ֔ב אֲשֶׁ֖ר עַל־פְּנֵ֣י יְרֵח֑וֹ וּרְאֵה֙ אֶת־אֶ֣רֶץ כְּנַ֔עַן אֲשֶׁ֨ר אֲנִ֥י נֹתֵ֛ן לִבְנֵ֥י יִשְׂרָאֵ֖ל לַאֲחֻזָּֽה׃ עֲלֵה עַל הַר הַמַּעֲבָר הַזֶּה – הַר נְבוֹ שֶׁבְּאֶרֶץ מוֹאָב שֶׁמּוּל יְרִיחוֹ, וּרְאֵה אֶת אֶרֶץ כְּנַעַן שֶׁאֲנִי נוֹתֵן לִבְנֵי יִשְׂרָאֵל לַאֲחֻזָּה.
נ וּמֻ֗ת בָּהָר֙ אֲשֶׁ֤ר אַתָּה֙ עֹלֶ֣ה שָׁ֔מָּה וְהֵאָסֵ֖ף אֶל־עַמֶּ֑יךָ כַּֽאֲשֶׁר־מֵ֞ת אַהֲרֹ֤ן אָחִ֙יךָ֙ בְּהֹ֣ר הָהָ֔ר וַיֵּאָ֖סֶף אֶל־עַמָּֽיו׃ וּמוּת בָּהָר שֶׁאַתָּה עוֹלֶה לְשָׁם וְנִשְׁמָתְךָ תֵּאָסֵף אֶל מְקוֹם הַנְּשָׁמוֹת, כְּמוֹ שֶׁאַהֲרֹן אָחִיךָ מֵת בְּהוֹר-הָהָר וְנִשְׁמָתוֹ נֶאֶסְפָה אֶל מְקוֹם הַנְּשָׁמוֹת.
נא עַל֩ אֲשֶׁ֨ר מְעַלְתֶּ֜ם בִּ֗י בְּתוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל בְּמֵֽי־מְרִיבַ֥ת קָדֵ֖שׁ מִדְבַּר־צִ֑ן עַ֣ל אֲשֶׁ֤ר לֹֽא־קִדַּשְׁתֶּם֙ אוֹתִ֔י בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל׃ עַל זֶה שֶׁמְּעַלְתֶּם בִּי בְּכָךְ שֶׁלֹּא קִדַּשְׁתֶּם אֶת שְׁמִי בְּתוֹךְ בְּנֵי יִשְׂרָאֵל בְּפָרָשַׁת מֵי מְרִיבָה בְּקָדֵשׁ בְּמִדְבַּר צִין.
נב כִּ֥י מִנֶּ֖גֶד תִּרְאֶ֣ה אֶת־הָאָ֑רֶץ וְשָׁ֙מָּה֙ לֹ֣א תָב֔וֹא אֶל־הָאָ֕רֶץ אֲשֶׁר־אֲנִ֥י נֹתֵ֖ן לִבְנֵ֥י יִשְׂרָאֵֽל׃ תַּעֲמוֹד רַק מִמּוּל וְתִרְאֶה אֶת הָאָרֶץ, וְלֹא תָּבוֹא לְשָׁם – לָאָרֶץ שֶׁאֲנִי נוֹתֵן לִבְנֵי יִשְׂרָאֵל.
