ראשון[עריכה]

פ פָּרָשַׁת כְּרִיתַת הַבְּרִית
כט ט אַתֶּ֨ם נִצָּבִ֤ים הַיּוֹם֙ כֻּלְּכֶ֔ם לִפְנֵ֖י יְהֹוָ֣ה אֱלֹהֵיכֶ֑ם רָאשֵׁיכֶ֣ם שִׁבְטֵיכֶ֗ם זִקְנֵיכֶם֙ וְשֹׁ֣טְרֵיכֶ֔ם כֹּ֖ל אִ֥ישׁ יִשְׂרָאֵֽל׃ אַתֶּם עוֹמְדִים הַיּוֹם כֻּלְּכֶם לִפְנֵי ה' אֱלֹהֵיכֶם – מַנְהִיגֵי שִׁבְטֵיכֶם זִקְנֵיכֶם וְשׁוֹטְרֵיכֶם, וְכָל אַנְשֵׁי יִשְׂרָאֵל.
י טַפְּכֶ֣ם נְשֵׁיכֶ֔ם וְגֵ֣רְךָ֔ אֲשֶׁ֖ר בְּקֶ֣רֶב מַחֲנֶ֑יךָ מֵחֹטֵ֣ב עֵצֶ֔יךָ עַ֖ד שֹׁאֵ֥ב מֵימֶֽיךָ׃ יַלְדֵיכֶם, נְשֵׁיכֶם, וְהַגֵּרִים שֶׁאִתְּכֶם בְּתוֹךְ מַחֲנֵיכֶם – מֵחוֹטֵב עֵצֶיךָ עַד שׁוֹאֵב מֵימֶיךָ.
יא לְעׇבְרְךָ֗ בִּבְרִ֛ית יְהֹוָ֥ה אֱלֹהֶ֖יךָ וּבְאָלָת֑וֹ אֲשֶׁר֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ כֹּרֵ֥ת עִמְּךָ֖ הַיּֽוֹם׃ כְּדֵי לַעֲשׂוֹת הֶסְכֵּם וּשְׁבוּעָה עִם ה' אֱלֹהֶיךָ, שֶׁה' אֱלֹהֶיךָ עוֹשֶׂה אִתְּךָ הַיּוֹם.

שני[עריכה]

יב לְמַ֣עַן הָקִֽים־אֹתְךָ֩ הַיּ֨וֹם ׀ ל֜וֹ לְעָ֗ם וְה֤וּא יִֽהְיֶה־לְּךָ֙ לֵֽאלֹהִ֔ים כַּאֲשֶׁ֖ר דִּבֶּר־לָ֑ךְ וְכַאֲשֶׁ֤ר נִשְׁבַּע֙ לַאֲבֹתֶ֔יךָ לְאַבְרָהָ֥ם לְיִצְחָ֖ק וּֽלְיַעֲקֹֽב׃ כְּדֵי לַעֲשׂוֹת הַיּוֹם שֶׁאַתָּה תִּהְיֶה עַמּוֹ וְהוּא יִהְיֶה הָאֱלֹהִים שֶׁלְּךָ כְּמוֹ שֶׁהִבְטִיחַ לְךָ, וּכְמוֹ שֶׁנִּשְׁבַּע לַאֲבוֹתֶיךָ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב.
יג וְלֹ֥א אִתְּכֶ֖ם לְבַדְּכֶ֑ם אָנֹכִ֗י כֹּרֵת֙ אֶת־הַבְּרִ֣ית הַזֹּ֔את וְאֶת־הָאָלָ֖ה הַזֹּֽאת׃ וְלֹא אִתְּכֶם בִּלְבַד אֲנִי עוֹשֶׂה אֶת הַהֶסְכֵּם הַזֶּה וְאֶת הַשְּׁבוּעָה הַזֹּאת.
יד כִּי֩ אֶת־אֲשֶׁ֨ר יֶשְׁנ֜וֹ פֹּ֗ה עִמָּ֙נוּ֙ עֹמֵ֣ד הַיּ֔וֹם לִפְנֵ֖י יְהֹוָ֣ה אֱלֹהֵ֑ינוּ וְאֵ֨ת אֲשֶׁ֥ר אֵינֶ֛נּוּ פֹּ֖ה עִמָּ֥נוּ הַיּֽוֹם׃ אֶלָּא, גַּם עִם מִי שֶׁנִּמְצָא פֹּה – שֶׁעוֹמֵד אִתָּנוּ הַיּוֹם לִפְנֵי ה' אֱלֹהֵינוּ, וְגַם עִם צֶאֱצָאֵיכֶם – כָּל מִי שֶׁאֵינֶנּוּ פֹּה אִתָּנוּ הַיּוֹם.

שלישי[עריכה]

טו כִּֽי־אַתֶּ֣ם יְדַעְתֶּ֔ם אֵ֥ת אֲשֶׁר־יָשַׁ֖בְנוּ בְּאֶ֣רֶץ מִצְרָ֑יִם וְאֵ֧ת אֲשֶׁר־עָבַ֛רְנוּ בְּקֶ֥רֶב הַגּוֹיִ֖ם אֲשֶׁ֥ר עֲבַרְתֶּֽם׃ כִּי אַתֶּם יְדַעְתֶּם אֶת כָּל מַה שֶּׁהָיָה כְּשֶׁגַּרְנוּ בְּאֶרֶץ מִצְרַיִם, וּכְשֶׁעָבַרְנוּ בְּתוֹךְ הָעַמִּים שֶׁעֲבַרְתֶּם בְּתוֹכָם.
טז וַתִּרְאוּ֙ אֶת־שִׁקּ֣וּצֵיהֶ֔ם וְאֵ֖ת גִּלֻּלֵיהֶ֑ם עֵ֣ץ וָאֶ֔בֶן כֶּ֥סֶף וְזָהָ֖ב אֲשֶׁ֥ר עִמָּהֶֽם׃ וּרְאִיתֶם אֶת פִּסְלֵיהֶם הַמְּשֻׁקָּצִים וְהַמְּאוּסִים שֶׁיֵּשׁ לָהֶם, שֶׁעֲשׁוּיִים מֵעֵץ וָאֶבֶן אוֹ כֶּסֶף וְזָהָב.
יז פֶּן־יֵ֣שׁ בָּ֠כֶ֠ם אִ֣ישׁ אֽוֹ־אִשָּׁ֞ה א֧וֹ מִשְׁפָּחָ֣ה אוֹ־שֵׁ֗בֶט אֲשֶׁר֩ לְבָב֨וֹ פֹנֶ֤ה הַיּוֹם֙ מֵעִם֙ יְהֹוָ֣ה אֱלֹהֵ֔ינוּ לָלֶ֣כֶת לַעֲבֹ֔ד אֶת־אֱלֹהֵ֖י הַגּוֹיִ֣ם הָהֵ֑ם פֶּן־יֵ֣שׁ בָּכֶ֗ם שֹׁ֛רֶשׁ פֹּרֶ֥ה רֹ֖אשׁ וְלַעֲנָֽה׃ שֶׁמָּא יֵשׁ בָּכֶם אִישׁ אוֹ אִשָּׁה אוֹ מִשְׁפָּחָה אוֹ שֵׁבֶט שֶׁבְּלִבּוֹ הוּא פּוֹנֶה הַיּוֹם מִלַּעֲבוֹד אֶת ה' אֱלֹהֵינוּ לָלֶכֶת לַעֲבוֹד אֶת אֱלֹהֵי הָעַמִּים הָהֵם, שֶׁמָּא יֵשׁ בָּכֶם אִישׁ שֶׁ'מַּצְמִיחַ' חֲטָאִים כְּמוֹ אֶרֶס וְעֵשֶׂב מָר.
יח וְהָיָ֡ה בְּשׇׁמְעוֹ֩ אֶת־דִּבְרֵ֨י הָאָלָ֜ה הַזֹּ֗את וְהִתְבָּרֵ֨ךְ בִּלְבָב֤וֹ לֵאמֹר֙ שָׁל֣וֹם יִֽהְיֶה־לִּ֔י כִּ֛י בִּשְׁרִר֥וּת לִבִּ֖י אֵלֵ֑ךְ לְמַ֛עַן סְפ֥וֹת הָרָוָ֖ה אֶת־הַצְּמֵאָֽה׃ וּכְשֶׁיִּשְׁמַע אֶת דִּבְרֵי הַקְּלָלָה הַזֹּאת יְבָרֵךְ אֶת עַצְמוֹ בְּלִבּוֹ וְיֹאמַר: "יִהְיֶה לִי טוֹב כְּשֶׁאֶתְנַהֵג כְּמוֹ שֶׁיֵּרָאֶה לִי", כְּדֵי לְהוֹסִיף עַל חֲטָאֵי הַתַּאֲווֹת גַּם מְרִידָה לְהַכְעִיס אֶת ה'.
יט לֹא־יֹאבֶ֣ה יְהֹוָה֮ סְלֹ֣חַֽ לוֹ֒ כִּ֣י אָ֠ז יֶעְשַׁ֨ן אַף־יְהֹוָ֤ה וְקִנְאָתוֹ֙ בָּאִ֣ישׁ הַה֔וּא וְרָ֤בְצָה בּוֹ֙ כׇּל־הָ֣אָלָ֔ה הַכְּתוּבָ֖ה בַּסֵּ֣פֶר הַזֶּ֑ה וּמָחָ֤ה יְהֹוָה֙ אֶת־שְׁמ֔וֹ מִתַּ֖חַת הַשָּׁמָֽיִם׃ ה' לֹא יִרְצֶה לִסְלוֹחַ לוֹ, ה' יִכְעַס אָז וְרָגְזוֹ יִפְגַּע בָּאִישׁ הַהוּא וְיִדְבְּקוּ בּוֹ כָּל הַקְּלָלוֹת שֶׁכְּתוּבוֹת בַּסֵּפֶר הַזֶּה, וַה' יִמְחוֹק אֶת זִכְרוֹ מִתַּחַת לַשָּׁמַיִם.
כ וְהִבְדִּיל֤וֹ יְהֹוָה֙ לְרָעָ֔ה מִכֹּ֖ל שִׁבְטֵ֣י יִשְׂרָאֵ֑ל כְּכֹל֙ אָל֣וֹת הַבְּרִ֔ית הַכְּתוּבָ֕ה בְּסֵ֥פֶר הַתּוֹרָ֖ה הַזֶּֽה׃ ה' יַבְדִּיל אוֹתוֹ לְרָעָה מִכָּל שִׁבְטֵי יִשְׂרָאֵל כְּכָל הַקְּלָלוֹת שֶׁבַּהֶסְכֵּם שֶׁכְּתוּבוֹת בְּסֵפֶר הַתּוֹרָה הַזֶּה.
כא וְאָמַ֞ר הַדּ֣וֹר הָאַחֲר֗וֹן בְּנֵיכֶם֙ אֲשֶׁ֤ר יָק֙וּמוּ֙ מֵאַ֣חֲרֵיכֶ֔ם וְהַ֨נׇּכְרִ֔י אֲשֶׁ֥ר יָבֹ֖א מֵאֶ֣רֶץ רְחוֹקָ֑ה וְ֠רָא֠וּ אֶת־מַכּ֞וֹת הָאָ֤רֶץ הַהִוא֙ וְאֶת־תַּ֣חֲלֻאֶ֔יהָ אֲשֶׁר־חִלָּ֥ה יְהֹוָ֖ה בָּֽהּ׃ וְיֹאמְרוּ הַדּוֹר הָאַחֲרוֹן – בְּנֵיכֶם שֶׁיָּבוֹאוּ אַחֲרֵיכֶם, וְהַנָּכְרִי שֶׁיָּבוֹא מֵאֶרֶץ רְחוֹקָה, כְּשֶׁיִּרְאוּ אֶת מַכּוֹת הָאָרֶץ הַהִיא וְאֶת מַחֲלוֹתֶיהָ שֶׁה' יָשִׂים בָּהּ:
כב גׇּפְרִ֣ית וָמֶ֘לַח֮ שְׂרֵפָ֣ה כׇל־אַרְצָהּ֒ לֹ֤א תִזָּרַע֙ וְלֹ֣א תַצְמִ֔חַ וְלֹא־יַעֲלֶ֥ה בָ֖הּ כׇּל־עֵ֑שֶׂב כְּֽמַהְפֵּכַ֞ת סְדֹ֤ם וַעֲמֹרָה֙ אַדְמָ֣ה וצביים וּצְבוֹיִ֔ם אֲשֶׁר֙ הָפַ֣ךְ יְהֹוָ֔ה בְּאַפּ֖וֹ וּבַחֲמָתֽוֹ׃ גָּפְרִית וּמֶלַח – שְׂרֵפָה בְּכָל אַרְצָהּ – לֹא תִּזָּרַע וְלֹא תַּצְמִיחַ וְלֹא יַעֲלֶה בָּהּ שׁוּם עֵשֶׂב, כְּמַהְפֵּכַת סְדוֹם וַעֲמוֹרָה אַדְמָה וּצְבוֹיִם שֶׁה' הָפַךְ בְּרָגְזוֹ וּבְכַעֲסוֹ.
כג וְאָֽמְרוּ֙ כׇּל־הַגּוֹיִ֔ם עַל־מֶ֨ה עָשָׂ֧ה יְהֹוָ֛ה כָּ֖כָה לָאָ֣רֶץ הַזֹּ֑את מֶ֥ה חֳרִ֛י הָאַ֥ף הַגָּד֖וֹל הַזֶּֽה׃ וְכָל הָעַמִּים יֹאמְרוּ: לָמָּה ה' עָשָׂה כָּךְ לָאָרֶץ הַזֹּאת, מָה הַכַּעַס הַגָּדוֹל הַזֶּה?
כד וְאָ֣מְר֔וּ עַ֚ל אֲשֶׁ֣ר עָזְב֔וּ אֶת־בְּרִ֥ית יְהֹוָ֖ה אֱלֹהֵ֣י אֲבֹתָ֑ם אֲשֶׁר֙ כָּרַ֣ת עִמָּ֔ם בְּהוֹצִיא֥וֹ אֹתָ֖ם מֵאֶ֥רֶץ מִצְרָֽיִם׃ וְיֹאמְרוּ: עַל שֶׁעָזְבוּ אֶת הַהֶסְכֵּם עִם ה' אֱלֹהֵי אֲבוֹתֵיהֶם, שֶׁעָשָׂה אִתָּם כְּשֶׁהוֹצִיא אוֹתָם מֵאֶרֶץ מִצְרַיִם.
כה וַיֵּלְכ֗וּ וַיַּֽעַבְדוּ֙ אֱלֹהִ֣ים אֲחֵרִ֔ים וַיִּֽשְׁתַּחֲו֖וּ לָהֶ֑ם אֱלֹהִים֙ אֲשֶׁ֣ר לֹֽא־יְדָע֔וּם וְלֹ֥א חָלַ֖ק לָהֶֽם׃ וְהָלְכוּ וְעָבְדוּ אֱלֹהִים אֲחֵרִים וְהִשְׁתַּחֲווּ לָהֶם – אֱלֹהִים שֶׁלֹּא יָדוּעַ בָּהֶם שׁוּם כֹּחַ, וְה' לֹא חִלֵּק אוֹתָם לָהֶם לַעֲבוֹד אוֹתָם.
כו וַיִּחַר־אַ֥ף יְהֹוָ֖ה בָּאָ֣רֶץ הַהִ֑וא לְהָבִ֤יא עָלֶ֙יהָ֙ אֶת־כׇּל־הַקְּלָלָ֔ה הַכְּתוּבָ֖ה בַּסֵּ֥פֶר הַזֶּֽה׃ ה' כָּעַס עַל הָאָרֶץ הַהִיא וְהֶחְלִיט לְהָבִיא עָלֶיהָ אֶת כָּל הַקְּלָלוֹת שֶׁכְּתוּבוֹת בַּסֵּפֶר הַזֶּה.
כז וַיִּתְּשֵׁ֤ם יְהֹוָה֙ מֵעַ֣ל אַדְמָתָ֔ם בְּאַ֥ף וּבְחֵמָ֖ה וּבְקֶ֣צֶף גָּד֑וֹל וַיַּשְׁלִכֵ֛ם אֶל־אֶ֥רֶץ אַחֶ֖רֶת כַּיּ֥וֹם הַזֶּֽה׃ וַה' הִגְלָה אוֹתָם מֵעַל אַדְמָתָם בְּרוֹגֶז וּבְכַעַס וּבְקֶצֶף גָּדוֹל, וְהִשְׁלִיךְ אוֹתָם לְאֶרֶץ אַחֶרֶת כְּמוֹ הַמַּצָּב הַיּוֹם.
כח הַנִּ֨סְתָּרֹ֔ת לַיהֹוָ֖ה אֱלֹהֵ֑ינוּ וְהַנִּגְלֹ֞ת לָ֤ׄנׄוּׄ וּׄלְׄבָׄנֵ֙ׄיׄנׄוּ֙ׄ עַׄד־עוֹלָ֔ם לַעֲשׂ֕וֹת אֶת־כׇּל־דִּבְרֵ֖י הַתּוֹרָ֥ה הַזֹּֽאת׃ הָעֲווֹנוֹת הַנִּסְתָּרִים מְסוּרִים לַה' אֱלֹהֵינוּ לְהַעֲנִישׁ עֲלֵיהֶם, וְהַגְּלוּיִים נְתוּנִים לָנוּ וּלְבָנֵינוּ עַד עוֹלָם לָדוּן אֶת הַחוֹטְאִים, כְּדֵי לְקַיֵּם אֶת כָּל דִּבְרֵי הַתּוֹרָה הַזֹּאת.

רביעי (שני במחוברין)[עריכה]

ס פָּרָשַׁת הַתְּשׁוּבָה
ל א וְהָיָה֩ כִֽי־יָבֹ֨אוּ עָלֶ֜יךָ כׇּל־הַדְּבָרִ֣ים הָאֵ֗לֶּה הַבְּרָכָה֙ וְהַקְּלָלָ֔ה אֲשֶׁ֥ר נָתַ֖תִּי לְפָנֶ֑יךָ וַהֲשֵׁבֹתָ֙ אֶל־לְבָבֶ֔ךָ בְּכׇ֨ל־הַגּוֹיִ֔ם אֲשֶׁ֧ר הִדִּיחֲךָ֛ יְהֹוָ֥ה אֱלֹהֶ֖יךָ שָֽׁמָּה׃ כְּשֶׁיָּבוֹאוּ עָלֶיךָ כָּל הַדְּבָרִים הָאֵלֶּה – הַבְּרָכָה וְהַקְּלָלָה שֶׁאָמַרְתִּי לְפָנֶיךָ, תַּפְנִים אֶת הַדְּבָרִים, כְּשֶׁתִּהְיֶה בֵּין כָּל הָעַמִּים שֶׁה' אֱלֹהֶיךָ הִגְלָה אוֹתְךָ לְשָׁם.
ב וְשַׁבְתָּ֞ עַד־יְהֹוָ֤ה אֱלֹהֶ֙יךָ֙ וְשָׁמַעְתָּ֣ בְקֹל֔וֹ כְּכֹ֛ל אֲשֶׁר־אָנֹכִ֥י מְצַוְּךָ֖ הַיּ֑וֹם אַתָּ֣ה וּבָנֶ֔יךָ בְּכׇל־לְבָבְךָ֖ וּבְכׇל־נַפְשֶֽׁךָ׃ תַּחֲזוֹר בִּתְשׁוּבָה אֶל ה' אֱלֹהֶיךָ וְתִשְׁמַע בְּקוֹלוֹ, וְתַעֲשֶׂה אֶת כָּל מַה שֶּׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם, אַתָּה וּבָנֶיךָ, בְּכָל לִבְּךָ וּבְכָל רְצוֹנְךָ.
ג וְשָׁ֨ב יְהֹוָ֧ה אֱלֹהֶ֛יךָ אֶת־שְׁבוּתְךָ֖ וְרִחֲמֶ֑ךָ וְשָׁ֗ב וְקִבֶּצְךָ֙ מִכׇּל־הָ֣עַמִּ֔ים אֲשֶׁ֧ר הֱפִֽיצְךָ֛ יְהֹוָ֥ה אֱלֹהֶ֖יךָ שָֽׁמָּה׃ וַה' אֱלֹהֶיךָ יַחֲזִיר אוֹתְךָ חֲזָרָה לָאָרֶץ, וִירַחֵם עָלֶיךָ, וְיָשׁוּב וְיֶאֱסוֹף אוֹתְךָ מִכָּל הָעַמִּים שֶׁה' אֱלֹהֶיךָ פִּזֵּר אוֹתְךָ שָׁם.
ד אִם־יִהְיֶ֥ה נִֽדַּחֲךָ֖ בִּקְצֵ֣ה הַשָּׁמָ֑יִם מִשָּׁ֗ם יְקַבֶּצְךָ֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ וּמִשָּׁ֖ם יִקָּחֶֽךָ׃ אִם אַחַד הַגּוֹלִים יִהְיֶה בִּקְצֵה הָעוֹלָם, גַּם מִשָּׁם ה' אֱלֹהֶיךָ יֶאֱסוֹף אוֹתְךָ וּמִשָּׁם יִקַּח אוֹתְךָ.
ה וֶהֱבִיאֲךָ֞ יְהֹוָ֣ה אֱלֹהֶ֗יךָ אֶל־הָאָ֛רֶץ אֲשֶׁר־יָרְשׁ֥וּ אֲבֹתֶ֖יךָ וִֽירִשְׁתָּ֑הּ וְהֵיטִֽבְךָ֥ וְהִרְבְּךָ֖ מֵאֲבֹתֶֽיךָ׃ ה' אֱלֹהֶיךָ יָבִיא אוֹתְךָ לָאָרֶץ שֶׁאֲבוֹתֶיךָ יָרְשׁוּ וְתִכְבּוֹשׁ אוֹתָהּ, וְיֵיטִיב לְךָ, וְיַרְבֶּה אוֹתְךָ יוֹתֵר מֵאֲבוֹתֶיךָ.
ו וּמָ֨ל יְהֹוָ֧ה אֱלֹהֶ֛יךָ אֶת־לְבָבְךָ֖ וְאֶת־לְבַ֣ב זַרְעֶ֑ךָ לְאַהֲבָ֞ה אֶת־יְהֹוָ֧ה אֱלֹהֶ֛יךָ בְּכׇל־לְבָבְךָ֥ וּבְכׇל־נַפְשְׁךָ֖ לְמַ֥עַן חַיֶּֽיךָ׃ ה' אֱלֹהֶיךָ יָסִיר אֶת הָאֲטִימוּת מִלִּבְּךָ וּמִלֵּב צֶאֱצָאֶיךָ, שֶׁתֹּאהַב אֶת ה' אֱלֹהֶיךָ בְּכָל לִבְּךָ וּבְכָל רְצוֹנְךָ, לְמַעַן חַיֶּיךָ.

חמישי (שלישי במחוברין)[עריכה]

ז וְנָתַן֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ אֵ֥ת כׇּל־הָאָל֖וֹת הָאֵ֑לֶּה עַל־אֹיְבֶ֥יךָ וְעַל־שֹׂנְאֶ֖יךָ אֲשֶׁ֥ר רְדָפֽוּךָ׃ ה' אֱלֹהֶיךָ יָבִיא אֶת כָּל הַקְּלָלוֹת הָאֵלֶּה עַל אוֹיְבֶיךָ וְעַל שׂוֹנְאֶיךָ שֶׁרָדְפוּ אוֹתְךָ.
ח וְאַתָּ֣ה תָשׁ֔וּב וְשָׁמַעְתָּ֖ בְּק֣וֹל יְהֹוָ֑ה וְעָשִׂ֙יתָ֙ אֶת־כׇּל־מִצְוֺתָ֔יו אֲשֶׁ֛ר אָנֹכִ֥י מְצַוְּךָ֖ הַיּֽוֹם׃ וְאַתָּה תַּחֲזוֹר בִּתְשׁוּבָה וְתִשְׁמַע בְּקוֹל ה' וְתַעֲשֶׂה אֶת כָּל מִצְווֹתָיו שֶׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם.
ט וְהוֹתִֽירְךָ֩ יְהֹוָ֨ה אֱלֹהֶ֜יךָ בְּכֹ֣ל ׀ מַעֲשֵׂ֣ה יָדֶ֗ךָ בִּפְרִ֨י בִטְנְךָ֜ וּבִפְרִ֧י בְהֶמְתְּךָ֛ וּבִפְרִ֥י אַדְמָתְךָ֖ לְטֹבָ֑ה כִּ֣י ׀ יָשׁ֣וּב יְהֹוָ֗ה לָשׂ֤וּשׂ עָלֶ֙יךָ֙ לְט֔וֹב כַּאֲשֶׁר־שָׂ֖שׂ עַל־אֲבֹתֶֽיךָ׃ ה' אֱלֹהֶיךָ יוֹסִיף לְךָ טוֹבָה בְּכָל מַעֲשֶׂיךָ – בִּילָדֶיךָ, וּבִוְלָדוֹת בְּהֵמוֹתֶיךָ, וּבִיבוּל אַדְמָתְךָ, כִּי ה' יָשׁוּב לִשְׂמוֹחַ בְּךָ שֶׁיִּהְיֶה לְךָ טוֹב, כְּמוֹ שֶׁשָּׂמַח אֵצֶל אֲבוֹתֶיךָ.
י כִּ֣י תִשְׁמַ֗ע בְּקוֹל֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ לִשְׁמֹ֤ר מִצְוֺתָיו֙ וְחֻקֹּתָ֔יו הַכְּתוּבָ֕ה בְּסֵ֥פֶר הַתּוֹרָ֖ה הַזֶּ֑ה כִּ֤י תָשׁוּב֙ אֶל־יְהֹוָ֣ה אֱלֹהֶ֔יךָ בְּכׇל־לְבָבְךָ֖ וּבְכׇל־נַפְשֶֽׁךָ׃ כִּי תִּשְׁמַע בְּקוֹל ה' אֱלֹהֶיךָ לִשְׁמוֹר אֶת מִצְווֹתָיו וְאֶת חֻקָּיו שֶׁכְּתוּבִים בְּסֵפֶר הַתּוֹרָה הַזֶּה, כִּי תָּשׁוּב אֶל ה' אֱלֹהֶיךָ בְּכָל לִבְּךָ וּבְכָל רְצוֹנְךָ.

ששי[עריכה]

יא כִּ֚י הַמִּצְוָ֣ה הַזֹּ֔את אֲשֶׁ֛ר אָנֹכִ֥י מְצַוְּךָ֖ הַיּ֑וֹם לֹא־נִפְלֵ֥את הִוא֙ מִמְּךָ֔ וְלֹ֥א רְחֹקָ֖ה הִֽוא׃ כִּי הַצִּוּוּי הַזֶּה שֶׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם הוּא לֹא פִּלְאִי לְמַעְלָה מִיְּכָלְתְּךָ, וְהוּא לֹא רָחוֹק.
יב לֹ֥א בַשָּׁמַ֖יִם הִ֑וא לֵאמֹ֗ר מִ֣י יַעֲלֶה־לָּ֤נוּ הַשָּׁמַ֙יְמָה֙ וְיִקָּחֶ֣הָ לָּ֔נוּ וְיַשְׁמִעֵ֥נוּ אֹתָ֖הּ וְנַעֲשֶֽׂנָּה׃ הוּא לֹא בַּשָּׁמַיִם שֶׁתּוּכְלוּ לִטְעוֹן: מִי יַעֲלֶה לָנוּ לַשָּׁמַיִם וְיִקַּח אוֹתוֹ עֲבוּרֵנוּ וְיַשְׁמִיעַ לָנוּ אוֹתוֹ וְנַעֲשֶׂה אוֹתוֹ.
יג וְלֹא־מֵעֵ֥בֶר לַיָּ֖ם הִ֑וא לֵאמֹ֗ר מִ֣י יַעֲבׇר־לָ֜נוּ אֶל־עֵ֤בֶר הַיָּם֙ וְיִקָּחֶ֣הָ לָּ֔נוּ וְיַשְׁמִעֵ֥נוּ אֹתָ֖הּ וְנַעֲשֶֽׂנָּה׃ וְהוּא לֹא מֵעֵבֶר לַיָּם שֶׁתֹּאמְרוּ: מִי יַעֲבוֹר לָנוּ אֶל מֵעֵבֶר לַיָּם וְיִקַּח אוֹתוֹ עֲבוּרֵנוּ וְיַשְׁמִיעַ לָנוּ אוֹתוֹ וְנַעֲשֶׂה אוֹתוֹ.
יד כִּֽי־קָר֥וֹב אֵלֶ֛יךָ הַדָּבָ֖ר מְאֹ֑ד בְּפִ֥יךָ וּבִֽלְבָבְךָ֖ לַעֲשֹׂתֽוֹ׃ אֶלָּא הַדָּבָר קָרוֹב אֵלֶיךָ מְאוֹד – בְּפִיךָ וּבְלִבְּךָ לַעֲשׂוֹתוֹ.

שביעי (רביעי במחוברין)[עריכה]

ס פָּרָשַׁת הַבְּחִירָה הַחָפְשִׁית
טו רְאֵ֨ה נָתַ֤תִּי לְפָנֶ֙יךָ֙ הַיּ֔וֹם אֶת־הַֽחַיִּ֖ים וְאֶת־הַטּ֑וֹב וְאֶת־הַמָּ֖וֶת וְאֶת־הָרָֽע׃ רְאֵה, נָתַתִּי לְפָנֶיךָ הַיּוֹם אֶת הַחַיִּים וְאֶת הַטּוֹב וְאֶת הַמָּוֶת וְאֶת הָרָע –
טז אֲשֶׁ֨ר אָנֹכִ֣י מְצַוְּךָ֮ הַיּוֹם֒ לְאַהֲבָ֞ה אֶת־יְהֹוָ֤ה אֱלֹהֶ֙יךָ֙ לָלֶ֣כֶת בִּדְרָכָ֔יו וְלִשְׁמֹ֛ר מִצְוֺתָ֥יו וְחֻקֹּתָ֖יו וּמִשְׁפָּטָ֑יו וְחָיִ֣יתָ וְרָבִ֔יתָ וּבֵֽרַכְךָ֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ בָּאָ֕רֶץ אֲשֶׁר־אַתָּ֥ה בָא־שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ שֶׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם לֶאֱהוֹב אֶת ה' אֱלֹהֶיךָ, לָלֶכֶת בִּדְרָכָיו וְלִשְׁמוֹר אֶת מִצְווֹתָיו וְחֻקָּיו וּמִשְׁפָּטָיו, וְתִחְיֶה וְתִתְרַבֶּה וַה' אֱלֹהֶיךָ יְבָרֵךְ אוֹתְךָ בָּאָרֶץ שֶׁאַתָּה בָּא לְשָׁם לִכְבּוֹשׁ אוֹתָהּ.
יז וְאִם־יִפְנֶ֥ה לְבָבְךָ֖ וְלֹ֣א תִשְׁמָ֑ע וְנִדַּחְתָּ֗ וְהִֽשְׁתַּחֲוִ֛יתָ לֵאלֹהִ֥ים אֲחֵרִ֖ים וַעֲבַדְתָּֽם׃ וְאִם לֹא תִּרְצֶה, וְלֹא תִּשְׁמַע בְּקוֹלִי, וְתִסְטֶה וְתִשְׁתַּחֲוֶה לֵאלֹהִים אֲחֵרִים וְתַעֲבוֹד אוֹתָם.
יח הִגַּ֤דְתִּי לָכֶם֙ הַיּ֔וֹם כִּ֥י אָבֹ֖ד תֹּאבֵד֑וּן לֹא־תַאֲרִיכֻ֤ן יָמִים֙ עַל־הָ֣אֲדָמָ֔ה אֲשֶׁ֨ר אַתָּ֤ה עֹבֵר֙ אֶת־הַיַּרְדֵּ֔ן לָב֥וֹא שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ אֲנִי אוֹמֵר לָכֶם הַיּוֹם שֶׁתֹּאבְדוּ, לֹא תִּתְקַיְּמוּ זְמַן רָב עַל הָאֲדָמָה שֶׁאַתָּה עוֹבֵר אֶת הַיַּרְדֵּן לָבוֹא לְשָׁם לִכְבּוֹשׁ אוֹתָהּ.
יט הַעִדֹ֨תִי בָכֶ֣ם הַיּוֹם֮ אֶת־הַשָּׁמַ֣יִם וְאֶת־הָאָ֒רֶץ֒ הַחַיִּ֤ים וְהַמָּ֙וֶת֙ נָתַ֣תִּי לְפָנֶ֔יךָ הַבְּרָכָ֖ה וְהַקְּלָלָ֑ה וּבָֽחַרְתָּ֙ בַּחַיִּ֔ים לְמַ֥עַן תִּֽחְיֶ֖ה אַתָּ֥ה וְזַרְעֶֽךָ׃ אֲנִי מֵעִיד בָּכֶם הַיּוֹם אֶת הַשָּׁמַיִם וְאֶת הָאָרֶץ: נָתַתִּי לְפָנֶיךָ אֶת הַחַיִּים וְהַמָּוֶת – הַבְּרָכָה וְהַקְּלָלָה, בְּחַר בַּחַיִּים כְּדֵי שֶׁתִּחְיוּ אַתָּה וְצֶאֱצָאֶיךָ.
כ לְאַֽהֲבָה֙ אֶת־יְהֹוָ֣ה אֱלֹהֶ֔יךָ לִשְׁמֹ֥עַ בְּקֹל֖וֹ וּלְדׇבְקָה־ב֑וֹ כִּ֣י ה֤וּא חַיֶּ֙יךָ֙ וְאֹ֣רֶךְ יָמֶ֔יךָ לָשֶׁ֣בֶת עַל־הָאֲדָמָ֗ה אֲשֶׁר֩ נִשְׁבַּ֨ע יְהֹוָ֧ה לַאֲבֹתֶ֛יךָ לְאַבְרָהָ֛ם לְיִצְחָ֥ק וּֽלְיַעֲקֹ֖ב לָתֵ֥ת לָהֶֽם׃ אֱהוֹב אֶת ה' אֱלֹהֶיךָ, שְׁמַע בְּקוֹלוֹ וְהִדָּבֵק בּוֹ, כִּי הוּא מַטְּרַת חַיֶּיךָ וְרִבּוּי שְׁנוֹתֶיךָ, וְכָךְ תִּתְיַשֵּׁב עַל הָאֲדָמָה שֶׁה' נִשְׁבַּע לַאֲבוֹתֶיךָ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לָתֵת לָהֶם.
