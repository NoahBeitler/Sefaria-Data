ראשון[עריכה]

פ פָּרָשַׁת מִצְווֹת הַקְּדֻשָּׁה
יט א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב דַּבֵּ֞ר אֶל־כׇּל־עֲדַ֧ת בְּנֵי־יִשְׂרָאֵ֛ל וְאָמַרְתָּ֥ אֲלֵהֶ֖ם קְדֹשִׁ֣ים תִּהְי֑וּ כִּ֣י קָד֔וֹשׁ אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ דַּבֵּר אֶל כָּל צִבּוּר בְּנֵי יִשְׂרָאֵל וֶאֱמוֹר לָהֶם: תִּהְיוּ קְדוֹשִׁים כִּי אֲנִי ה' אֱלֹהֵיכֶם קָדוֹשׁ.
ג אִ֣ישׁ אִמּ֤וֹ וְאָבִיו֙ תִּירָ֔אוּ וְאֶת־שַׁבְּתֹתַ֖י תִּשְׁמֹ֑רוּ אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ תְּכַבְּדוּ – כָּל אִישׁ אֶת אִמּוֹ וְאֶת אָבִיו, וְתִשְׁמְרוּ אֶת שַׁבְּתוֹתַי, אֲנִי ה' אֱלֹהֵיכֶם.
ד אַל־תִּפְנוּ֙ אֶל־הָ֣אֱלִילִ֔ם וֵֽאלֹהֵי֙ מַסֵּכָ֔ה לֹ֥א תַעֲשׂ֖וּ לָכֶ֑ם אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ אַל תִּפְנוּ לַעֲבוֹד אֶת הָאֱלִילִים וְאַל תַּעֲשׂוּ לָכֶם פִּסְלֵי מַתָּכוֹת, אֲנִי ה' אֱלֹהֵיכֶם.
ה וְכִ֧י תִזְבְּח֛וּ זֶ֥בַח שְׁלָמִ֖ים לַיהֹוָ֑ה לִֽרְצֹנְכֶ֖ם תִּזְבָּחֻֽהוּ׃ וּכְשֶׁתִּזְבְּחוּ זֶבַח שְׁלָמִים לַה', תִּזְבְּחוּ אוֹתוֹ בְּאוֹפֶן שֶׁה' יִהְיֶה מְרוּצֶה מִמַּעֲשֵׂיכֶם.
ו בְּי֧וֹם זִבְחֲכֶ֛ם יֵאָכֵ֖ל וּמִֽמׇּחֳרָ֑ת וְהַנּוֹתָר֙ עַד־י֣וֹם הַשְּׁלִישִׁ֔י בָּאֵ֖שׁ יִשָּׂרֵֽף׃ הַקָּרְבָּן יֵאָכֵל בְּיוֹם הַשְּׁחִיטָה וּלְמָחֳרָת, וּמַה שֶּׁיִּשָּׁאֵר עַד הַיּוֹם הַשְּׁלִישִׁי יִשָּׂרֵף בָּאֵשׁ.
ז וְאִ֛ם הֵאָכֹ֥ל יֵאָכֵ֖ל בַּיּ֣וֹם הַשְּׁלִישִׁ֑י פִּגּ֥וּל ה֖וּא לֹ֥א יֵרָצֶֽה׃ וְאִם יֵאָכֵל בַּיּוֹם הַשְּׁלִישִׁי, הוּא מָאוּס וְהוּא לֹא יִתְקַבֵּל בְּרָצוֹן.
ח וְאֹֽכְלָיו֙ עֲוֺנ֣וֹ יִשָּׂ֔א כִּֽי־אֶת־קֹ֥דֶשׁ יְהֹוָ֖ה חִלֵּ֑ל וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵעַמֶּֽיהָ׃ וּמִי שֶׁאוֹכֵל אוֹתוֹ יְקַבֵּל אֶת עוֹנֶשׁ עֲווֹנוֹ כִּי הוּא חִלֵּל אֶת קוֹדְשֵׁי ה', הַנֶּפֶשׁ הַהִיא תֹּאבַד מֵעַמָּהּ.
ט וּֽבְקֻצְרְכֶם֙ אֶת־קְצִ֣יר אַרְצְכֶ֔ם לֹ֧א תְכַלֶּ֛ה פְּאַ֥ת שָׂדְךָ֖ לִקְצֹ֑ר וְלֶ֥קֶט קְצִֽירְךָ֖ לֹ֥א תְלַקֵּֽט׃ וּכְשֶׁתִּקְצְרוּ אֶת קְצִיר אַרְצְכֶם, אַל תִּגְמוֹר לִקְצוֹר גַּם אֶת קְצֵה הַשָּׂדֶה, וְאַל תֶּאֱסוֹף אֶת הַשִּׁבֹּלִים שֶׁנָּפְלוּ בִּזְמַן הַקְּצִירָה.
י וְכַרְמְךָ֙ לֹ֣א תְעוֹלֵ֔ל וּפֶ֥רֶט כַּרְמְךָ֖ לֹ֣א תְלַקֵּ֑ט לֶֽעָנִ֤י וְלַגֵּר֙ תַּעֲזֹ֣ב אֹתָ֔ם אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ וְאַל תִּבְצוֹר בְּכַרְמְךָ אֶת הָאֶשְׁכּוֹלוֹת הַקְּטַנִּים, וְאַל תֶּאֱסוֹף אֶת הָעֲנָבִים שֶׁנָּפְלוּ בְּכַרְמְךָ בִּזְמַן הַבָּצִיר, אֶלָּא תַּשְׁאִיר אוֹתָם לֶעָנִי וְלַגֵּר, אֲנִי ה' אֱלֹהֵיכֶם.
יא לֹ֖א תִּגְנֹ֑בוּ וְלֹא־תְכַחֲשׁ֥וּ וְלֹֽא־תְשַׁקְּר֖וּ אִ֥ישׁ בַּעֲמִיתֽוֹ׃ אַל תִּגְנְבוּ, וְאַל תַּכְחִישׁוּ אֶת חוֹבוֹתֵיכֶם וְאַל תְּרַמּוּ אִישׁ אֶת חֲבֵרוֹ.
יב וְלֹֽא־תִשָּׁבְע֥וּ בִשְׁמִ֖י לַשָּׁ֑קֶר וְחִלַּלְתָּ֛ אֶת־שֵׁ֥ם אֱלֹהֶ֖יךָ אֲנִ֥י יְהֹוָֽה׃ וְאַל תִּשָּׁבְעוּ בִּשְׁמִי לַשֶּׁקֶר שֶׁלֹּא תְּחַלֵּל אֶת שֵׁם אֱלֹהֶיךָ, אֲנִי ה'.
יג לֹֽא־תַעֲשֹׁ֥ק אֶת־רֵֽעֲךָ֖ וְלֹ֣א תִגְזֹ֑ל לֹֽא־תָלִ֞ין פְּעֻלַּ֥ת שָׂכִ֛יר אִתְּךָ֖ עַד־בֹּֽקֶר׃ אַל תִּמָּנַע מִלְּשַׁלֵּם לַחֲבֵרְךָ אֶת שְׂכָרוֹ וְאַל תִּגְזוֹל, אַל תַּשְׁאִיר אֶצְלְךָ בַּלַּיְלָה עַד הַבּוֹקֶר אֶת שְׂכַר עֲבוֹדָתוֹ שֶׁל שָׂכִיר.
יד לֹא־תְקַלֵּ֣ל חֵרֵ֔שׁ וְלִפְנֵ֣י עִוֵּ֔ר לֹ֥א תִתֵּ֖ן מִכְשֹׁ֑ל וְיָרֵ֥אתָ מֵּאֱלֹהֶ֖יךָ אֲנִ֥י יְהֹוָֽה׃ אַל תְּקַלֵּל אֲפִלּוּ חֵרֵשׁ, וְאַל תָּשִׂים מִכְשׁוֹל לִפְנֵי עִוֵּר, וּתְכַבֵּד אֶת אֱלֹהֶיךָ, אֲנִי ה'.

שני (חמישי במחוברין)[עריכה]

טו לֹא־תַעֲשׂ֥וּ עָ֙וֶל֙ בַּמִּשְׁפָּ֔ט לֹא־תִשָּׂ֣א פְנֵי־דָ֔ל וְלֹ֥א תֶהְדַּ֖ר פְּנֵ֣י גָד֑וֹל בְּצֶ֖דֶק תִּשְׁפֹּ֥ט עֲמִיתֶֽךָ׃ אַל תַּעֲשׂוּ עָוֶל בַּמִּשְׁפָּט – אַל תִּתֵּן עֲדִיפוּת לְעָנִי וְאַל תְּכַבֵּד אָדָם חָשׁוּב, אֶלָּא תִּשְׁפּוֹט אֶת חֲבֵרְךָ בְּצֶדֶק.
טז לֹא־תֵלֵ֤ךְ רָכִיל֙ בְּעַמֶּ֔יךָ לֹ֥א תַעֲמֹ֖ד עַל־דַּ֣ם רֵעֶ֑ךָ‏[1] אֲנִ֖י יְהֹוָֽה׃ אַל תַּלְשִׁין עַל אָדָם מֵעַמְּךָ, אַל תִּמָּנַע מִלְּהַצִּיל אֶת חַיֵּי חֲבֵרְךָ שֶׁבְּסַכָּנָה, אֲנִי ה'.
יז לֹֽא־תִשְׂנָ֥א אֶת־אָחִ֖יךָ בִּלְבָבֶ֑ךָ הוֹכֵ֤חַ תּוֹכִ֙יחַ֙ אֶת־עֲמִיתֶ֔ךָ וְלֹא־תִשָּׂ֥א עָלָ֖יו חֵֽטְא׃ אַל תִּשְׂנָא אֶת אָחִיךָ בְּלִבְּךָ, הוֹכַח אֶת חֲבֵרְךָ, כְּדֵי שֶׁלֹּא תֵּעָנֵשׁ עַל חֶטְאוֹ.
יח לֹֽא־תִקֹּ֤ם וְלֹֽא־תִטֹּר֙ אֶת־בְּנֵ֣י עַמֶּ֔ךָ וְאָֽהַבְתָּ֥ לְרֵעֲךָ֖ כָּמ֑וֹךָ אֲנִ֖י יְהֹוָֽה׃ אַל תִּנְקוֹם וְאַל תִּשְׁמוֹר בְּלִבְּךָ שִׂנְאָה עַל בְּנֵי עַמְּךָ, אֱהוֹב אֶת חֲבֵרְךָ כָּמוֹךָ, אֲנִי ה'.
יט אֶֽת־חֻקֹּתַי֮ תִּשְׁמֹ֒רוּ֒ בְּהֶמְתְּךָ֙ לֹא־תַרְבִּ֣יעַ כִּלְאַ֔יִם שָׂדְךָ֖ לֹא־תִזְרַ֣ע כִּלְאָ֑יִם וּבֶ֤גֶד כִּלְאַ֙יִם֙ שַֽׁעַטְנֵ֔ז לֹ֥א יַעֲלֶ֖ה עָלֶֽיךָ׃ שִׁמְרוּ אֶת הַחֻקִּים שֶׁלִּי: אַל תַּרְבִּיעַ אֶת בְּהֶמְתְּךָ בְּעִרְבּוּב עִם מִין אַחֵר, אֶת שָׂדְךָ אַל תִּזְרַע בְּעִרְבּוּב שֶׁל מִינֵי זְרָעִים, וְלֹא יַעֲלֶה עָלֶיךָ בֶּגֶד מִצֶּמֶר וּפִשְׁתָּן מְעוֹרָבִים.
כ וְ֠אִ֠ישׁ כִּֽי־יִשְׁכַּ֨ב אֶת־אִשָּׁ֜ה שִׁכְבַת־זֶ֗רַע וְהִ֤וא שִׁפְחָה֙ נֶחֱרֶ֣פֶת לְאִ֔ישׁ וְהׇפְדֵּה֙ לֹ֣א נִפְדָּ֔תָה א֥וֹ חֻפְשָׁ֖הֿ לֹ֣א נִתַּן־לָ֑הּ בִּקֹּ֧רֶת תִּהְיֶ֛ה לֹ֥א יוּמְת֖וּ כִּי־לֹ֥א חֻפָּֽשָׁה׃ וְאִם אִישׁ יִשְׁכַּב עִם אִשָּׁה וְיִתֵּן בָּהּ אֶת נוֹזֵל הַזֶּרַע וְהִיא שִׁפְחָה מְאוֹרֶסֶת לְאִישׁ וְהִיא לֹא נִפְדְּתָה בְּכֶסֶף וְלֹא שִׁחְרְרוּ אוֹתָהּ לַחָפְשִׁי, יִהְיֶה הֶבְדֵּל מֵהַדִּין הָרָגִיל – הֵם לֹא יוּמְתוּ כִּי הִיא לֹא יָצְאָה לַחָפְשִׁי.
כא וְהֵבִ֤יא אֶת־אֲשָׁמוֹ֙ לַֽיהֹוָ֔ה אֶל־פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד אֵ֖יל אָשָֽׁם׃ הוּא יָבִיא אֶת הַקָּרְבָּן עַל אַשְׁמָתוֹ לַה' – אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת, אַיִל לְאָשָׁם.
כב וְכִפֶּר֩ עָלָ֨יו הַכֹּהֵ֜ן בְּאֵ֤יל הָֽאָשָׁם֙ לִפְנֵ֣י יְהֹוָ֔ה עַל־חַטָּאת֖וֹ אֲשֶׁ֣ר חָטָ֑א וְנִסְלַ֣ח ל֔וֹ מֵחַטָּאת֖וֹ אֲשֶׁ֥ר חָטָֽא׃ הַכֹּהֵן יְכַפֵּר עָלָיו בְּאֵיל הָאָשָׁם לִפְנֵי ה' עַל חֶטְאוֹ שֶׁחָטָא, וְיִסָּלַח לוֹ עַל חֶטְאוֹ שֶׁחָטָא.

שלישי[עריכה]

פ
כג וְכִי־תָבֹ֣אוּ אֶל־הָאָ֗רֶץ וּנְטַעְתֶּם֙ כׇּל־עֵ֣ץ מַאֲכָ֔ל וַעֲרַלְתֶּ֥ם עׇרְלָת֖וֹ אֶת־פִּרְי֑וֹ שָׁלֹ֣שׁ שָׁנִ֗ים יִהְיֶ֥ה לָכֶ֛ם עֲרֵלִ֖ים לֹ֥א יֵאָכֵֽל׃ וּכְשֶׁתָּבוֹאוּ לָאָרֶץ וְתִטְּעוּ עֵץ מַאֲכָל מִכָּל מִין, תִּתְרַחֲקוּ מִלֶּאֱכוֹל אֶת פֵּרוֹתָיו שֶׁהֵם עָרְלָה – שָׁלוֹשׁ שָׁנִים יִהְיֶה לָכֶם עָרְלָה, לֹא יֵאָכְלוּ.
כד וּבַשָּׁנָה֙ הָרְבִיעִ֔ת יִהְיֶ֖ה כׇּל־פִּרְי֑וֹ קֹ֥דֶשׁ הִלּוּלִ֖ים לַיהֹוָֽה׃ וּבַשָּׁנָה הָרְבִיעִית יִהְיוּ כָּל פֵּרוֹתָיו קוֹדֶשׁ לְהַלֵּל לַה'.
כה וּבַשָּׁנָ֣ה הַחֲמִישִׁ֗ת תֹּֽאכְלוּ֙ אֶת־פִּרְי֔וֹ לְהוֹסִ֥יף לָכֶ֖ם תְּבוּאָת֑וֹ אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ וּבַשָּׁנָה הַחֲמִישִׁית תֹּאכְלוּ אֶת פֵּרוֹתָיו, וְכָךְ יִתְרַבֶּה הַיְּבוּל שֶׁלּוֹ, אֲנִי ה' אֱלֹהֵיכֶם.
כו לֹ֥א תֹאכְל֖וּ עַל־הַדָּ֑ם לֹ֥א תְנַחֲשׁ֖וּ וְלֹ֥א תְעוֹנֵֽנוּ׃ אַל תֹּאכְלוּ בָּשָׂר בְּעוֹדוֹ עִם הַדָּם, אַל תְּכַשְּׁפוּ וְאַל תֹּאחֲזוּ עֵינַיִם.
כז לֹ֣א תַקִּ֔פוּ פְּאַ֖ת רֹאשְׁכֶ֑ם וְלֹ֣א תַשְׁחִ֔ית אֵ֖ת פְּאַ֥ת זְקָנֶֽךָ׃ אַל תְּגַלְּחוּ אֶת שְׂעַרְכֶם מִסָּבִיב בְּשָׁוֶה וְתוֹרִידוּ אֶת פֵּאוֹת רֹאשְׁכֶם, וְאַל תַּשְׁחִית אֶת קְצָווֹת זְקָנְךָ.
כח וְשֶׂ֣רֶט לָנֶ֗פֶשׁ לֹ֤א תִתְּנוּ֙ בִּבְשַׂרְכֶ֔ם וּכְתֹ֣בֶת קַֽעֲקַ֔ע לֹ֥א תִתְּנ֖וּ בָּכֶ֑ם אֲנִ֖י יְהֹוָֽה׃ אַל תַּעֲשׂוּ בְּגוּפְכֶם שְׂרִיטָה מִתּוֹךְ אֲבֵלוּת עַל נֶפֶשׁ, וְאַל תַּעֲשׂוּ בָּכֶם כְּתוֹבֶת קַעֲקוּעַ, אֲנִי ה'.
כט אַל־תְּחַלֵּ֥ל אֶֽת־בִּתְּךָ֖ לְהַזְנוֹתָ֑הּ וְלֹא־תִזְנֶ֣ה הָאָ֔רֶץ וּמָלְאָ֥ה הָאָ֖רֶץ זִמָּֽה׃ אַל תְּחַלֵּל אֶת בִּתְּךָ לַעֲשׂוֹת אוֹתָהּ זוֹנָה, וְלֹא תִּהְיֶה זְנוּת בָּאָרֶץ – שֶׁתִּתְמַלֵּא הָאָרֶץ בְּמַחְשְׁבוֹת חֵטְא.
ל אֶת־שַׁבְּתֹתַ֣י תִּשְׁמֹ֔רוּ וּמִקְדָּשִׁ֖י תִּירָ֑אוּ אֲנִ֖י יְהֹוָֽה׃ תִּשְׁמְרוּ אֶת שַׁבְּתוֹתַי וּתְכַבְּדוּ אֶת מִקְדָּשִׁי, אֲנִי ה'.
לא אַל־תִּפְנ֤וּ אֶל־הָאֹבֹת֙ וְאֶל־הַיִּדְּעֹנִ֔ים אַל־תְּבַקְשׁ֖וּ לְטׇמְאָ֣ה בָהֶ֑ם אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ אַל תִּפְנוּ אֶל מַעֲלֵי רוּחוֹת מֵתִים בְּכִשּׁוּף וְאֶל מַגִּידֵי עֲתִידוֹת, אַל תְּבַקְשׁוּ לְהִטַּמֵּא בָּהֶם, אֲנִי ה' אֱלֹהֵיכֶם.
לב מִפְּנֵ֤י שֵׂיבָה֙ תָּק֔וּם וְהָדַרְתָּ֖ פְּנֵ֣י זָקֵ֑ן וְיָרֵ֥אתָ מֵּאֱלֹהֶ֖יךָ אֲנִ֥י יְהֹוָֽה׃ קוּם בִּפְנֵי זָקֵן בֶּן שִׁבְעִים, וְכַבֵּד חָכָם בַּתּוֹרָה, כַּבֵּד אֶת אֱלֹהֶיךָ, אֲנִי ה'.

רביעי (ששי במחוברין)[עריכה]

ס פָּרָשַׁת שְׁמִירַת הַצֶּדֶק
לג וְכִֽי־יָג֧וּר אִתְּךָ֛ גֵּ֖ר בְּאַרְצְכֶ֑ם לֹ֥א תוֹנ֖וּ אֹתֽוֹ׃ וְאִם יָגוּר אִתְּכֶם גֵּר בְּאַרְצְכֶם אַל תְּצַעֲרוּ אוֹתוֹ.
לד כְּאֶזְרָ֣ח מִכֶּם֩ יִהְיֶ֨ה לָכֶ֜ם הַגֵּ֣ר ׀ הַגָּ֣ר אִתְּכֶ֗ם וְאָהַבְתָּ֥ לוֹ֙ כָּמ֔וֹךָ כִּֽי־גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֣רֶץ מִצְרָ֑יִם אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ הַגֵּר שֶׁגָּר אִתְּכֶם יִהְיֶה לָכֶם כְּמוֹ אֶזְרָח שֶׁלָּכֶם, אֱהוֹב אוֹתוֹ כְּמוֹ אֶת עַצְמְךָ כִּי הֱיִיתֶם גֵּרִים בְּאֶרֶץ מִצְרַיִם, אֲנִי ה' אֱלֹהֵיכֶם.
לה לֹא־תַעֲשׂ֥וּ עָ֖וֶל בַּמִּשְׁפָּ֑ט בַּמִּדָּ֕ה בַּמִּשְׁקָ֖ל וּבַמְּשׂוּרָֽה׃ אַל תַּעֲשׂוּ עָוֶל בַּמִּשְׁפָּט, וּבְמִדּוֹת נֶפַח שֶׁל יָבֵשׁ, בְּמִשְׁקוֹלוֹת, וּבְמִדּוֹת נוֹזְלִים.
לו מֹ֧אזְנֵי צֶ֣דֶק אַבְנֵי־צֶ֗דֶק אֵ֥יפַת צֶ֛דֶק וְהִ֥ין צֶ֖דֶק יִהְיֶ֣ה לָכֶ֑ם אֲנִי֙ יְהֹוָ֣ה אֱלֹֽהֵיכֶ֔ם אֲשֶׁר־הוֹצֵ֥אתִי אֶתְכֶ֖ם מֵאֶ֥רֶץ מִצְרָֽיִם׃ יִהְיוּ לָכֶם מֹאזְנַיִם, מִשְׁקוֹלוֹת, וּכְלֵי מְדִידַת נֶפַח לְיָבֵשׁ וּלְנוֹזְלִים – מְדֻיָּקוֹת, אֲנִי ה' אֱלֹהֵיכֶם שֶׁהוֹצֵאתִי אֶתְכֶם מֵאֶרֶץ מִצְרַיִם.
לז וּשְׁמַרְתֶּ֤ם אֶת־כׇּל־חֻקֹּתַי֙ וְאֶת־כׇּל־מִשְׁפָּטַ֔י וַעֲשִׂיתֶ֖ם אֹתָ֑ם אֲנִ֖י יְהֹוָֽה׃ שִׁמְרוּ אֶת כָּל חֻקַּי וְאֶת כָּל מִשְׁפָּטַי וַעֲשׂוּ אוֹתָם, אֲנִי ה'.

חמישי[עריכה]

פ פָּרָשַׁת הַמֹּלֶךְ
כ א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב וְאֶל־בְּנֵ֣י יִשְׂרָאֵל֮ תֹּאמַר֒ אִ֣ישׁ אִישׁ֩ מִבְּנֵ֨י יִשְׂרָאֵ֜ל וּמִן־הַגֵּ֣ר ׀ הַגָּ֣ר בְּיִשְׂרָאֵ֗ל אֲשֶׁ֨ר יִתֵּ֧ן מִזַּרְע֛וֹ לַמֹּ֖לֶךְ מ֣וֹת יוּמָ֑ת עַ֥ם הָאָ֖רֶץ יִרְגְּמֻ֥הוּ בָאָֽבֶן׃ אֱמוֹר לִבְנֵי יִשְׂרָאֵל: כָּל אִישׁ מִבְּנֵי יִשְׂרָאֵל וּמֵהַגֵּר שֶׁגָּר בְּיִשְׂרָאֵל שֶׁיִּתֵּן מִצֶּאֱצָאָיו שֶׁיִּשָּׂרֵף בָּאֵשׁ לַעֲבוֹדַת הַמּוֹלֶךְ – יוּמַת; הָעָם יִרְגְּמוּ אוֹתוֹ בָּאֲבָנִים.
ג וַאֲנִ֞י אֶתֵּ֤ן אֶת־פָּנַי֙ בָּאִ֣ישׁ הַה֔וּא וְהִכְרַתִּ֥י אֹת֖וֹ מִקֶּ֣רֶב עַמּ֑וֹ כִּ֤י מִזַּרְעוֹ֙ נָתַ֣ן לַמֹּ֔לֶךְ לְמַ֗עַן טַמֵּא֙ אֶת־מִקְדָּשִׁ֔י וּלְחַלֵּ֖ל אֶת־שֵׁ֥ם קׇדְשִֽׁי׃ וַאֲנִי אֶכְעַס עַל הָאִישׁ הַהוּא – וְאֶעֱשֶׂה שֶׁיֹּאבַד מִתּוֹךְ עַמּוֹ, כִּי הוּא נָתַן מִצֶּאֱצָאָיו לַמּוֹלֶךְ וּבְכָךְ הוּא מְטַמֵּא אֶת עַם קָדְשִׁי וּמְחַלֵּל אֶת שְׁמִי הַקָּדוֹשׁ.
ד וְאִ֡ם הַעְלֵ֣ם יַעְלִ֩ימֽוּ֩ עַ֨ם הָאָ֜רֶץ אֶת־עֵֽינֵיהֶם֙ מִן־הָאִ֣ישׁ הַה֔וּא בְּתִתּ֥וֹ מִזַּרְע֖וֹ לַמֹּ֑לֶךְ לְבִלְתִּ֖י הָמִ֥ית אֹתֽוֹ׃ וְאִם הָעָם יַעֲלִימוּ אֶת עֵינֵיהֶם מֵהָאִישׁ הַהוּא שֶׁנָּתַן מִצֶּאֱצָאָיו לַמּוֹלֶךְ, וְלֹא יַהַרְגוּ אוֹתוֹ.
ה וְשַׂמְתִּ֨י אֲנִ֧י אֶת־פָּנַ֛י בָּאִ֥ישׁ הַה֖וּא וּבְמִשְׁפַּחְתּ֑וֹ וְהִכְרַתִּ֨י אֹת֜וֹ וְאֵ֣ת ׀ כׇּל־הַזֹּנִ֣ים אַחֲרָ֗יו לִזְנ֛וֹת אַחֲרֵ֥י הַמֹּ֖לֶךְ מִקֶּ֥רֶב עַמָּֽם׃ אֲנִי אֶכְעַס עַל הָאִישׁ הַהוּא וְעַל מִשְׁפַּחְתּוֹ, וְאֶעֱשֶׂה שֶׁהוּא וְכָל הַבּוֹגְדִים בִּי בְּעִקְבוֹתָיו וְהוֹלְכִים אַחֲרֵי הַמּוֹלֶךְ יֹאבְדוּ מִתּוֹךְ עַמָּם.
ו וְהַנֶּ֗פֶשׁ אֲשֶׁ֨ר תִּפְנֶ֤ה אֶל־הָֽאֹבֹת֙ וְאֶל־הַיִּדְּעֹנִ֔ים לִזְנֹ֖ת אַחֲרֵיהֶ֑ם וְנָתַתִּ֤י אֶת־פָּנַי֙ בַּנֶּ֣פֶשׁ הַהִ֔וא וְהִכְרַתִּ֥י אֹת֖וֹ מִקֶּ֥רֶב עַמּֽוֹ׃ וְהָאָדָם שֶׁיִּפְנֶה אֶל מַעֲלֵי רוּחוֹת מֵתִים בְּכִשּׁוּף וְאֶל מַגִּידֵי עֲתִידוֹת וְיִבְגּוֹד בִּי וְיֵלֵךְ אַחֲרֵיהֶם, אֶכְעַס עַל הָאָדָם הַהוּא וְאֶעֱשֶׂה שֶׁיֹּאבַד מִתּוֹךְ עַמּוֹ.
ז וְהִ֨תְקַדִּשְׁתֶּ֔ם וִהְיִיתֶ֖ם קְדֹשִׁ֑ים כִּ֛י אֲנִ֥י יְהֹוָ֖ה אֱלֹהֵיכֶֽם׃ קַדְּשׁוּ עַצְמְכֶם וְתִהְיוּ קְדוֹשִׁים, כִּי אֲנִי ה' אֱלֹהֵיכֶם.

ששי (שביעי במחוברין)[עריכה]

ח וּשְׁמַרְתֶּם֙ אֶת־חֻקֹּתַ֔י וַעֲשִׂיתֶ֖ם אֹתָ֑ם אֲנִ֥י יְהֹוָ֖ה מְקַדִּשְׁכֶֽם׃ שִׁמְרוּ אֶת חֻקַּי וַעֲשׂוּ אוֹתָם, אֲנִי ה' שֶׁמְּקַדֵּשׁ אֶתְכֶם.
ט כִּֽי־אִ֣ישׁ אִ֗ישׁ אֲשֶׁ֨ר יְקַלֵּ֧ל אֶת־אָבִ֛יו וְאֶת־אִמּ֖וֹ מ֣וֹת יוּמָ֑ת אָבִ֧יו וְאִמּ֛וֹ קִלֵּ֖ל דָּמָ֥יו בּֽוֹ׃ כָּל אִישׁ שֶׁיְּקַלֵּל אֶת אָבִיו אוֹ אֶת אִמּוֹ יוּמַת, הוּא קִלֵּל אֶת אָבִיו וְאִמּוֹ – הוּא חַיָּב מִיתָה.
פָּרָשַׁת עֹנְשֵׁי הָעֲרָיוֹת
י וְאִ֗ישׁ אֲשֶׁ֤ר יִנְאַף֙ אֶת־אֵ֣שֶׁת אִ֔ישׁ אֲשֶׁ֥ר יִנְאַ֖ף אֶת־אֵ֣שֶׁת רֵעֵ֑הוּ מֽוֹת־יוּמַ֥ת הַנֹּאֵ֖ף וְהַנֹּאָֽפֶת׃ וְאִישׁ שֶׁיִּזְנֶה עִם אֵשֶׁת אִישׁ – שֶׁיִּזְנֶה עִם אֵשֶׁת חֲבֵרוֹ, הַזּוֹנֶה וְהַזּוֹנָה יוּמְתוּ.
יא וְאִ֗ישׁ אֲשֶׁ֤ר יִשְׁכַּב֙ אֶת־אֵ֣שֶׁת אָבִ֔יו עֶרְוַ֥ת אָבִ֖יו גִּלָּ֑ה מֽוֹת־יוּמְת֥וּ שְׁנֵיהֶ֖ם דְּמֵיהֶ֥ם בָּֽם׃ וְאִישׁ שֶׁיִּשְׁכַּב עִם אֵשֶׁת אָבִיו – הוּא גִּלָּה אֶת הַמָּקוֹם הַמֻּצְנָע שֶׁל אָבִיו, שְׁנֵיהֶם יוּמְתוּ כִּי הֵם חַיָּבִים מִיתָה.
יב וְאִ֗ישׁ אֲשֶׁ֤ר יִשְׁכַּב֙ אֶת־כַּלָּת֔וֹ מ֥וֹת יוּמְת֖וּ שְׁנֵיהֶ֑ם תֶּ֥בֶל עָשׂ֖וּ דְּמֵיהֶ֥ם בָּֽם׃ וְאִישׁ שֶׁיִּשְׁכַּב עִם אֵשֶׁת בְּנוֹ, שְׁנֵיהֶם יוּמְתוּ, הֵם עָשׂוּ עִרְבּוּב וּבִלְבּוּל וְהֵם חַיָּבִים מִיתָה.
יג וְאִ֗ישׁ אֲשֶׁ֨ר יִשְׁכַּ֤ב אֶת־זָכָר֙ מִשְׁכְּבֵ֣י אִשָּׁ֔ה תּוֹעֵבָ֥ה עָשׂ֖וּ שְׁנֵיהֶ֑ם מ֥וֹת יוּמָ֖תוּ דְּמֵיהֶ֥ם בָּֽם׃ וְאִישׁ שֶׁיִּשְׁכַּב עִם זָכָר כְּמוֹ בְּמִשְׁכְּבֵי אִשָּׁה – שְׁנֵיהֶם עָשׂוּ תּוֹעֵבָה, הֵם יוּמְתוּ כִּי הֵם חַיָּבִים מִיתָה.
יד וְאִ֗ישׁ אֲשֶׁ֨ר יִקַּ֧ח אֶת־אִשָּׁ֛ה וְאֶת־אִמָּ֖הּ זִמָּ֣ה הִ֑וא בָּאֵ֞שׁ יִשְׂרְפ֤וּ אֹתוֹ֙ וְאֶתְהֶ֔ן וְלֹא־תִהְיֶ֥ה זִמָּ֖ה בְּתוֹכְכֶֽם׃ וְאִישׁ שֶׁיִּשָּׂא אִשָּׁה וְגַם אֶת אִמָּהּ – זוֹ מַחְשֶׁבֶת חֵטְא, יִשְׂרְפוּ אוֹתוֹ וְאוֹתָן בָּאֵשׁ, וְלֹא תִּהְיֶה מַחְשֶׁבֶת חֵטְא בְּתוֹכְכֶם.
טו וְאִ֗ישׁ אֲשֶׁ֨ר יִתֵּ֧ן שְׁכׇבְתּ֛וֹ בִּבְהֵמָ֖ה מ֣וֹת יוּמָ֑ת וְאֶת־הַבְּהֵמָ֖ה תַּהֲרֹֽגוּ׃ וְאִישׁ שֶׁיִּזְנֶה עִם בְּהֵמָה יוּמַת, וְאֶת הַבְּהֵמָה תַּהַרְגוּ.
טז וְאִשָּׁ֗ה אֲשֶׁ֨ר תִּקְרַ֤ב אֶל־כׇּל־בְּהֵמָה֙ לְרִבְעָ֣הֿ אֹתָ֔הּ וְהָרַגְתָּ֥ אֶת־הָאִשָּׁ֖ה וְאֶת־הַבְּהֵמָ֑ה מ֥וֹת יוּמָ֖תוּ דְּמֵיהֶ֥ם בָּֽם׃ וְאִשָּׁה שֶׁתִּגַּשׁ אֶל כָּל בְּהֵמָה שֶׁהִיא שֶׁתִּזְנֶה אִתָּהּ – תַּהֲרוֹג אֶת הָאִשָּׁה וְאֶת הַבְּהֵמָה, הֵם יוּמְתוּ כִּי הֵם חַיָּבִים מִיתָה.
יז וְאִ֣ישׁ אֲשֶׁר־יִקַּ֣ח אֶת־אֲחֹת֡וֹ בַּת־אָבִ֣יו א֣וֹ בַת־אִ֠מּ֠וֹ וְרָאָ֨ה אֶת־עֶרְוָתָ֜הּ וְהִֽיא־תִרְאֶ֤ה אֶת־עֶרְוָתוֹ֙ חֶ֣סֶד ה֔וּא וְנִ֨כְרְת֔וּ לְעֵינֵ֖י בְּנֵ֣י עַמָּ֑ם עֶרְוַ֧ת אֲחֹת֛וֹ גִּלָּ֖ה עֲוֺנ֥וֹ יִשָּֽׂא׃ וְאִישׁ שֶׁיִּשָּׂא אֶת אֲחוֹתוֹ מֵאָבִיו אוֹ מֵאִמּוֹ וְהֵם יִשְׁכְּבוּ יַחַד וְיִזְנוּ – זוֹ בּוּשָׁה, וְהֵם יֻשְׁמְדוּ לְעֵינֵי בְּנֵי עַמָּם, הוּא גִּלָּה אֶת הַמָּקוֹם הַמֻּצְנָע שֶׁל אֲחוֹתוֹ לָכֵן יְקַבֵּל אֶת עוֹנֶשׁ עֲווֹנוֹ.
יח וְ֠אִ֠ישׁ אֲשֶׁר־יִשְׁכַּ֨ב אֶת־אִשָּׁ֜ה דָּוָ֗ה וְגִלָּ֤ה אֶת־עֶרְוָתָהּ֙ אֶת־מְקֹרָ֣הּ הֶֽעֱרָ֔ה וְהִ֕וא גִּלְּתָ֖ה אֶת־מְק֣וֹר דָּמֶ֑יהָ וְנִכְרְת֥וּ שְׁנֵיהֶ֖ם מִקֶּ֥רֶב עַמָּֽם׃ וְאִישׁ שֶׁיִּשְׁכַּב עִם אִשָּׁה זָבָה וִיגַלֶּה אֶת הַמָּקוֹם הַמֻּצְנָע שֶׁלָּהּ – הוּא חָשַׂף אֶת מְקוֹר דָּמֶיהָ, וְהִיא גִּלְּתָה אֶת מְקוֹר דָּמֶיהָ, שְׁנֵיהֶם יֹאבְדוּ מִתּוֹךְ עַמָּם.
יט וְעֶרְוַ֨ת אֲח֧וֹת אִמְּךָ֛ וַאֲח֥וֹת אָבִ֖יךָ לֹ֣א תְגַלֵּ֑ה כִּ֧י אֶת־שְׁאֵר֛וֹ הֶעֱרָ֖ה עֲוֺנָ֥ם יִשָּֽׂאוּ׃ אַל תְּגַלֵּה אֶת הַמָּקוֹם הַמֻּצְנָע שֶׁל אֲחוֹת אִמְּךָ אוֹ אָבִיךָ, הֵם יְקַבְּלוּ עוֹנֶשׁ עַל עֲווֹנָם כִּי הוּא חָשַׂף אֶת הַמָּקוֹם הַמֻּצְנָע שֶׁל קְרוֹבָתוֹ.
כ וְאִ֗ישׁ אֲשֶׁ֤ר יִשְׁכַּב֙ אֶת־דֹּ֣דָת֔וֹ עֶרְוַ֥ת דֹּד֖וֹ גִּלָּ֑ה חֶטְאָ֥ם יִשָּׂ֖אוּ עֲרִירִ֥ים יָמֻֽתוּ׃ וְאִישׁ שֶׁיִּשְׁכַּב עִם דּוֹדָתוֹ – הוּא גִּלָּה אֶת הַמָּקוֹם הַמֻּצְנָע שֶׁל דּוֹדוֹ, הֵם יְקַבְּלוּ אֶת עוֹנֶשׁ חֶטְאָם וְיָמוּתוּ לְלֹא יְלָדִים.
כא וְאִ֗ישׁ אֲשֶׁ֥ר יִקַּ֛ח אֶת־אֵ֥שֶׁת אָחִ֖יו נִדָּ֣ה הִ֑וא עֶרְוַ֥ת אָחִ֛יו גִּלָּ֖ה עֲרִירִ֥ים יִהְיֽוּ׃ וְאִישׁ שֶׁיִּשָּׂא אֶת אֵשֶׁת אָחִיו – זֶהוּ דָּבָר מְרוּחָק, הוּא גִּלָּה אֶת הַמָּקוֹם הַמֻּצְנָע שֶׁל אָחִיו וְהֵם יִהְיוּ לְלֹא יְלָדִים.
כב וּשְׁמַרְתֶּ֤ם אֶת־כׇּל־חֻקֹּתַי֙ וְאֶת־כׇּל־מִשְׁפָּטַ֔י וַעֲשִׂיתֶ֖ם אֹתָ֑ם וְלֹא־תָקִ֤יא אֶתְכֶם֙ הָאָ֔רֶץ אֲשֶׁ֨ר אֲנִ֜י מֵבִ֥יא אֶתְכֶ֛ם שָׁ֖מָּה לָשֶׁ֥בֶת בָּֽהּ׃ שִׁמְרוּ אֶת כָּל חֻקַּי וְאֶת כָּל מִשְׁפָּטַי וַעֲשׂוּ אוֹתָם, וְכָךְ הָאָרֶץ שֶׁאֲנִי מֵבִיא אֶתְכֶם לְשָׁם לָגוּר בָּהּ לֹא תָּקִיא אֶתְכֶם.

שביעי[עריכה]

כג וְלֹ֤א תֵֽלְכוּ֙ בְּחֻקֹּ֣ת הַגּ֔וֹי אֲשֶׁר־אֲנִ֥י מְשַׁלֵּ֖חַ מִפְּנֵיכֶ֑ם כִּ֤י אֶת־כׇּל־אֵ֙לֶּה֙ עָשׂ֔וּ וָאָקֻ֖ץ בָּֽם׃ וְאַל תִּתְנַהֲגוּ עַל פִּי חֻקֵּי הָעָם שֶׁאֲנִי מְגָרֵשׁ מִפְּנֵיכֶם, כִּי הֵם עָשׂוּ אֶת כָּל הַמַּעֲשִׂים הָאֵלֶּה וּמָאַסְתִּי בָּהֶם.
כד וָאֹמַ֣ר לָכֶ֗ם אַתֶּם֮ תִּֽירְשׁ֣וּ אֶת־אַדְמָתָם֒ וַאֲנִ֞י אֶתְּנֶ֤נָּה לָכֶם֙ לָרֶ֣שֶׁת אֹתָ֔הּ אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָ֑שׁ אֲנִי֙ יְהֹוָ֣ה אֱלֹֽהֵיכֶ֔ם אֲשֶׁר־הִבְדַּ֥לְתִּי אֶתְכֶ֖ם מִן־הָֽעַמִּֽים׃ וְאָמַרְתִּי לָכֶם: אַתֶּם תִּכְבְּשׁוּ אֶת אַדְמָתָם, אֲנִי אֶתֵּן אוֹתָהּ לָכֶם לִכְבּוֹשׁ אוֹתָהּ – אֶרֶץ זָבַת חָלָב וּדְבָשׁ, אֲנִי ה' אֱלֹהֵיכֶם שֶׁהִבְדַּלְתִּי אֶתְכֶם מֵהָעַמִּים.

מפטיר[עריכה]

כה וְהִבְדַּלְתֶּ֞ם בֵּֽין־הַבְּהֵמָ֤ה הַטְּהֹרָה֙ לַטְּמֵאָ֔ה וּבֵין־הָע֥וֹף הַטָּמֵ֖א לַטָּהֹ֑ר וְלֹֽא־תְשַׁקְּצ֨וּ אֶת־נַפְשֹֽׁתֵיכֶ֜ם בַּבְּהֵמָ֣ה וּבָע֗וֹף וּבְכֹל֙ אֲשֶׁ֣ר תִּרְמֹ֣שׂ הָֽאֲדָמָ֔ה אֲשֶׁר־הִבְדַּ֥לְתִּי לָכֶ֖ם לְטַמֵּֽא׃ הַבְדִּילוּ בֵּין הַבְּהֵמָה הַטְּהוֹרָה לַטְּמֵאָה, וּבֵין הָעוֹף הַטָּמֵא לַטָּהוֹר, וְלֹא תַּמְאִיסוּ אֶת נַפְשׁוֹתֵיכֶם בַּאֲכִילַת אִסּוּרִים – בַּבְּהֵמָה וּבָעוֹף וּבְכָל הַזּוֹחֲלִים עַל הָאֲדָמָה שֶׁהִבְדַּלְתִּי לָכֶם וְאָמַרְתִּי שֶׁהֵם מְטַמְּאִים.
כו וִהְיִ֤יתֶם לִי֙ קְדֹשִׁ֔ים כִּ֥י קָד֖וֹשׁ אֲנִ֣י יְהֹוָ֑ה וָאַבְדִּ֥ל אֶתְכֶ֛ם מִן־הָֽעַמִּ֖ים לִהְי֥וֹת לִֽי׃ הֱיוּ לִי קְדוֹשִׁים כִּי אֲנִי ה' קָדוֹשׁ, וְהִבְדַּלְתִּי אֶתְכֶם מֵהָעַמִּים כְּדֵי שֶׁתִּהְיוּ שֶׁלִּי.
כז וְאִ֣ישׁ אֽוֹ־אִשָּׁ֗ה כִּֽי־יִהְיֶ֨ה בָהֶ֥ם א֛וֹב א֥וֹ יִדְּעֹנִ֖י מ֣וֹת יוּמָ֑תוּ בָּאֶ֛בֶן יִרְגְּמ֥וּ אֹתָ֖ם דְּמֵיהֶ֥ם בָּֽם׃ וְאִם אִישׁ אוֹ אִשָּׁה יִהְיוּ מַעֲלֵי רוּחוֹת מֵתִים אוֹ מַגִּידֵי עֲתִידוֹת בְּכִשּׁוּף – הֵם יוּמְתוּ, יִרְגְּמוּ אוֹתָם בַּאֲבָנִים כִּי הֵם חַיָּבִים מִיתָה.
