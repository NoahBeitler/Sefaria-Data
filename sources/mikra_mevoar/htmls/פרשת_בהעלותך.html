ראשון[עריכה]

פ פָּרָשַׁת הַדְלָקַת הַמְּנוֹרָה
ח א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב דַּבֵּר֙ אֶֽל־אַהֲרֹ֔ן וְאָמַרְתָּ֖ אֵלָ֑יו בְּהַעֲלֹֽתְךָ֙ אֶת־הַנֵּרֹ֔ת אֶל־מוּל֙ פְּנֵ֣י הַמְּנוֹרָ֔ה יָאִ֖ירוּ שִׁבְעַ֥ת הַנֵּרֽוֹת׃ דַּבֵּר אֶל אַהֲרֹן וֶאֱמוֹר לוֹ: כְּשֶׂתַּדְלִיק אֶת הַנֵּרוֹת, שִׁבְעַת הַנֵּרוֹת יָאִירוּ אֶל מוּל הַמְּנוֹרָה – לְפָנֶיהָ.
ג וַיַּ֤עַשׂ כֵּן֙ אַהֲרֹ֔ן אֶל־מוּל֙ פְּנֵ֣י הַמְּנוֹרָ֔ה הֶעֱלָ֖ה נֵרֹתֶ֑יהָ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ אַהֲרֹן עָשָׂה כָּךְ, הוּא הִדְלִיק אֶת נֵרוֹתֶיהָ לְכִוּוּן קִדְמַת הַמְּנוֹרָה, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.
ד וְזֶ֨ה מַעֲשֵׂ֤ה הַמְּנֹרָה֙ מִקְשָׁ֣ה זָהָ֔ב עַד־יְרֵכָ֥הּ עַד־פִּרְחָ֖הּ מִקְשָׁ֣ה הִ֑וא כַּמַּרְאֶ֗ה אֲשֶׁ֨ר הֶרְאָ֤ה יְהֹוָה֙ אֶת־מֹשֶׁ֔ה כֵּ֥ן עָשָׂ֖ה אֶת־הַמְּנֹרָֽה׃ וְכָךְ צוּרַת עֲשִׂיַּת הַמְּנוֹרָה – מִגּוּשׁ אֶחָד זָהָב, גַּם רַגְלָהּ גַּם פְּרָחֶיהָ – כֻּלָּהּ עֲשׂוּיָה מִגּוּשׁ אֶחָד, כַּמַּרְאֶה שֶׁה' הֶרְאָה לְמֹשֶׁה כָּךְ עָשָׂה אֶת הַמְּנוֹרָה.
פ פָּרָשַׁת הַלְוִיִּים
ה וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ו קַ֚ח אֶת־הַלְוִיִּ֔ם מִתּ֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל וְטִהַרְתָּ֖ אֹתָֽם׃ קַח אֶת הַלְוִיִּים מִתּוֹךְ בְּנֵי יִשְׂרָאֵל וְטַהֵר אוֹתָם.
ז וְכֹֽה־תַעֲשֶׂ֤ה לָהֶם֙ לְטַֽהֲרָ֔ם הַזֵּ֥ה עֲלֵיהֶ֖ם מֵ֣י חַטָּ֑את וְהֶעֱבִ֤ירוּ תַ֙עַר֙ עַל־כׇּל־בְּשָׂרָ֔ם וְכִבְּס֥וּ בִגְדֵיהֶ֖ם וְהִטֶּהָֽרוּ׃ וְכָךְ תַּעֲשֶׂה לָהֶם כְּדֵי לְטַהֵר אוֹתָם: הַזֵּה עֲלֵיהֶם מֵי טָהֳרָה וִיגַלְּחוּ אֶת כָּל גּוּפָם בְּסַכִּין שֶׁל גִּלּוּחַ וִיכַבְּסוּ אֶת בִּגְדֵיהֶם וְיִטָּהֲרוּ.
ח וְלָֽקְחוּ֙ פַּ֣ר בֶּן־בָּקָ֔ר וּמִ֨נְחָת֔וֹ סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן וּפַר־שֵׁנִ֥י בֶן־בָּקָ֖ר תִּקַּ֥ח לְחַטָּֽאת׃ וְיִקְחוּ פַּר בֶּן בָּקָר, וּמִנְחָתוֹ – סוֹלֶת מְעֻרְבֶּבֶת בְּשֶׁמֶן, וּפַר שֵׁנִי תִּקַּח לְקָרְבַּן חַטָּאת.
ט וְהִקְרַבְתָּ֙ אֶת־הַלְוִיִּ֔ם לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וְהִ֨קְהַלְתָּ֔ אֶֽת־כׇּל־עֲדַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ וּתְקָרֵב אֶת הַלְוִיִּים אֶל לִפְנֵי אוֹהֶל הַהִתְכַּנְּסוּת וְתֶאֱסוֹף אֶת כָּל קְהַל בְּנֵי יִשְׂרָאֵל.
י וְהִקְרַבְתָּ֥ אֶת־הַלְוִיִּ֖ם לִפְנֵ֣י יְהֹוָ֑ה וְסָמְכ֧וּ בְנֵי־יִשְׂרָאֵ֛ל אֶת־יְדֵיהֶ֖ם עַל־הַלְוִיִּֽם׃ וּתְקָרֵב אֶת הַלְוִיִּים לִפְנֵי ה', וּבְנֵי יִשְׂרָאֵל יַשְׁעִינוּ אֶת יְדֵיהֶם עַל הַלְוִיִּים.
יא וְהֵנִיף֩ אַהֲרֹ֨ן אֶת־הַלְוִיִּ֤ם תְּנוּפָה֙ לִפְנֵ֣י יְהֹוָ֔ה מֵאֵ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל וְהָי֕וּ לַעֲבֹ֖ד אֶת־עֲבֹדַ֥ת יְהֹוָֽה׃ אַהֲרֹן יָנִיף אֶת הַלְוִיִּים לִפְנֵי ה', שֶׁיִּהְיוּ עוֹבְדֵי עֲבוֹדַת ה'.
יב וְהַלְוִיִּם֙ יִסְמְכ֣וּ אֶת־יְדֵיהֶ֔ם עַ֖ל רֹ֣אשׁ הַפָּרִ֑ים וַ֠עֲשֵׂ֠ה אֶת־הָאֶחָ֨ד חַטָּ֜את וְאֶת־הָאֶחָ֤ד עֹלָה֙ לַֽיהֹוָ֔ה לְכַפֵּ֖ר עַל־הַלְוִיִּֽם׃ וְהַלְוִיִּים יַשְׁעִינוּ אֶת יְדֵיהֶם עַל רָאשֵׁי הַפָּרִים, וַעֲשֵׂה אֶת הָאֶחָד לְקָרְבַּן חַטָּאת וְאֶת הָאֶחָד לְקָרְבַּן עוֹלָה לַה' לְכַפֵּר עַל הַלְוִיִּים.
יג וְהַֽעֲמַדְתָּ֙ אֶת־הַלְוִיִּ֔ם לִפְנֵ֥י אַהֲרֹ֖ן וְלִפְנֵ֣י בָנָ֑יו וְהֵנַפְתָּ֥ אֹתָ֛ם תְּנוּפָ֖ה לַֽיהֹוָֽה׃ וְתַעֲמִיד אֶת הַלְוִיִּים לִפְנֵי אַהֲרֹן וְלִפְנֵי בָּנָיו וְתָנִיף אוֹתָם לַה'.
יד וְהִבְדַּלְתָּ֙ אֶת־הַלְוִיִּ֔ם מִתּ֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל וְהָ֥יוּ לִ֖י הַלְוִיִּֽם׃ וְתַבְדִּיל אֶת הַלְוִיִּים מִתּוֹךְ בְּנֵי יִשְׂרָאֵל, וְהַלְוִיִּים יִהְיוּ שֶׁלִּי.

שני[עריכה]

טו וְאַֽחֲרֵי־כֵן֙ יָבֹ֣אוּ הַלְוִיִּ֔ם לַעֲבֹ֖ד אֶת־אֹ֣הֶל מוֹעֵ֑ד וְטִֽהַרְתָּ֣ אֹתָ֔ם וְהֵנַפְתָּ֥ אֹתָ֖ם תְּנוּפָֽה׃ וְאַחֲרֵי כֵן הַלְוִיִּים יָבוֹאוּ לַעֲבוֹד בְּאוֹהֶל הַהִתְכַּנְּסוּת וּתְטַהֵר אוֹתָם וְתָנִיף אוֹתָם.
טז כִּי֩ נְתֻנִ֨ים נְתֻנִ֥ים הֵ֙מָּה֙ לִ֔י מִתּ֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל תַּ֩חַת֩ פִּטְרַ֨ת כׇּל־רֶ֜חֶם בְּכ֥וֹר כֹּל֙ מִבְּנֵ֣י יִשְׂרָאֵ֔ל לָקַ֥חְתִּי אֹתָ֖ם לִֽי׃ כִּי הֵם נְתוּנִים לִי בְּמַתָּנָה מִתּוֹךְ בְּנֵי יִשְׂרָאֵל, לָקַחְתִּי אוֹתָם לִי בִּמְקוֹם כָּל בְּכוֹר אָדָם מִבְּנֵי יִשְׂרָאֵל – כָּל הַיּוֹצְאִים רִאשׁוֹנִים מִפֶּתַח הָרֶחֶם.
יז כִּ֣י לִ֤י כׇל־בְּכוֹר֙ בִּבְנֵ֣י יִשְׂרָאֵ֔ל בָּאָדָ֖ם וּבַבְּהֵמָ֑ה בְּי֗וֹם הַכֹּתִ֤י כׇל־בְּכוֹר֙ בְּאֶ֣רֶץ מִצְרַ֔יִם הִקְדַּ֥שְׁתִּי אֹתָ֖ם לִֽי׃ כִּי כָּל הַבְּכוֹרוֹת שֶׁל בְּנֵי יִשְׂרָאֵל בָּאָדָם וּבַבְּהֵמָה הֵם שֶׁלִּי, כִּי בַּיּוֹם שֶׁהָרַגְתִּי אֶת כָּל הַבְּכוֹרוֹת שֶׁל אֶרֶץ מִצְרַיִם הִקְדַּשְׁתִּי אוֹתָם לִי.
יח וָאֶקַּ֖ח אֶת־הַלְוִיִּ֑ם תַּ֥חַת כׇּל־בְּכ֖וֹר בִּבְנֵ֥י יִשְׂרָאֵֽל׃ וְעַכְשָׁיו לָקַחְתִּי אֶת הַלְוִיִּים בִּמְקוֹם כָּל הַבְּכוֹרוֹת שֶׁל בְּנֵי יִשְׂרָאֵל.
יט וָאֶתְּנָ֨ה אֶת־הַלְוִיִּ֜ם נְתֻנִ֣ים ׀ לְאַהֲרֹ֣ן וּלְבָנָ֗יו מִתּוֹךְ֮ בְּנֵ֣י יִשְׂרָאֵל֒ לַעֲבֹ֞ד אֶת־עֲבֹדַ֤ת בְּנֵֽי־יִשְׂרָאֵל֙ בְּאֹ֣הֶל מוֹעֵ֔ד וּלְכַפֵּ֖ר עַל־בְּנֵ֣י יִשְׂרָאֵ֑ל וְלֹ֨א יִהְיֶ֜ה בִּבְנֵ֤י יִשְׂרָאֵל֙ נֶ֔גֶף בְּגֶ֥שֶׁת בְּנֵֽי־יִשְׂרָאֵ֖ל אֶל־הַקֹּֽדֶשׁ׃ וְנָתַתִּי אֶת הַלְוִיִּים בְּמַתָּנָה לְאַהֲרֹן וּלְבָנָיו מִתּוֹךְ בְּנֵי יִשְׂרָאֵל כְּדֵי לַעֲבוֹד אֶת עֲבוֹדַת בְּנֵי יִשְׂרָאֵל בְּאוֹהֶל הַהִתְכַּנְּסוּת וּלְכַפֵּר עַל בְּנֵי יִשְׂרָאֵל, כְּדֵי שֶׁלֹּא תִּהְיֶה מַגֵּפָה בִּבְנֵי יִשְׂרָאֵל אִם בְּנֵי יִשְׂרָאֵל בְּעַצְמָם יִגְּשׁוּ אֶל הַקּוֹדֶשׁ.
כ וַיַּ֨עַשׂ מֹשֶׁ֧ה וְאַהֲרֹ֛ן וְכׇל־עֲדַ֥ת בְּנֵי־יִשְׂרָאֵ֖ל לַלְוִיִּ֑ם כְּ֠כֹ֠ל אֲשֶׁר־צִוָּ֨ה יְהֹוָ֤ה אֶת־מֹשֶׁה֙ לַלְוִיִּ֔ם כֵּן־עָשׂ֥וּ לָהֶ֖ם בְּנֵ֥י יִשְׂרָאֵֽל׃ מֹשֶׁה וְאַהֲרֹן וְכָל קְהַל בְּנֵי יִשְׂרָאֵל עָשׂוּ כָּךְ לַלְוִיִּים, כְּכָל מַה שֶּׁה' צִוָּה אֶת מֹשֶׁה בְּעִנְיַן הַלְוִיִּים כָּךְ בְּנֵי יִשְׂרָאֵל עָשׂוּ לָהֶם.
כא וַיִּֽתְחַטְּא֣וּ הַלְוִיִּ֗ם וַֽיְכַבְּסוּ֙ בִּגְדֵיהֶ֔ם וַיָּ֨נֶף אַהֲרֹ֥ן אֹתָ֛ם תְּנוּפָ֖ה לִפְנֵ֣י יְהֹוָ֑ה וַיְכַפֵּ֧ר עֲלֵיהֶ֛ם אַהֲרֹ֖ן לְטַהֲרָֽם׃ הַלְוִיִּים נִטְהֲרוּ וְכִבְּסוּ אֶת בִּגְדֵיהֶם, וְאַהֲרֹן הֵנִיף אוֹתָם לִפְנֵי ה' וְאַהֲרֹן כִּפֵּר עֲלֵיהֶם כְּדֵי לְטַהֵר אוֹתָם.
כב וְאַחֲרֵי־כֵ֞ן בָּ֣אוּ הַלְוִיִּ֗ם לַעֲבֹ֤ד אֶת־עֲבֹֽדָתָם֙ בְּאֹ֣הֶל מוֹעֵ֔ד לִפְנֵ֥י אַהֲרֹ֖ן וְלִפְנֵ֣י בָנָ֑יו כַּאֲשֶׁר֩ צִוָּ֨ה יְהֹוָ֤ה אֶת־מֹשֶׁה֙ עַל־הַלְוִיִּ֔ם כֵּ֖ן עָשׂ֥וּ לָהֶֽם׃ וְאַחֲרֵי כֵן הַלְוִיִּים בָּאוּ לַעֲבוֹד אֶת עֲבוֹדָתָם בְּאוֹהֶל הַהִתְכַּנְּסוּת לְשָׁרֵת תַּחַת פְּקֻדַּת אַהֲרֹן וְתַחַת פְּקֻדַּת בָּנָיו – כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה עַל הַלְוִיִּים כָּךְ עָשׂוּ לָהֶם.
ס פָּרָשַׁת גִּיל עֲבוֹדַת הַלְוִיִּים
כג וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
כד זֹ֖את אֲשֶׁ֣ר לַלְוִיִּ֑ם מִבֶּן֩ חָמֵ֨שׁ וְעֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה יָבוֹא֙ לִצְבֹ֣א צָבָ֔א בַּעֲבֹדַ֖ת אֹ֥הֶל מוֹעֵֽד׃ זֹאת הַמִּצְוָה לַלְוִיִּים: מִגִּיל עֶשְׂרִים וְחָמֵשׁ שָׁנָה וָמַעְלָה יָבוֹאוּ לְהִצְטָרֵף לִקְהִלַּת עֲבוֹדַת אוֹהֶל הַהִתְכַּנְּסוּת.
כה וּמִבֶּן֙ חֲמִשִּׁ֣ים שָׁנָ֔ה יָשׁ֖וּב מִצְּבָ֣א הָעֲבֹדָ֑ה וְלֹ֥א יַעֲבֹ֖ד עֽוֹד׃ וּמִגִּיל חֲמִשִּׁים שָׁנָה יַחֲזוֹר מִקְּהִלַּת הָעֲבוֹדָה וְלֹא יַעֲבוֹד עוֹד.
כו וְשֵׁרֵ֨ת אֶת־אֶחָ֜יו בְּאֹ֤הֶל מוֹעֵד֙ לִשְׁמֹ֣ר מִשְׁמֶ֔רֶת וַעֲבֹדָ֖ה לֹ֣א יַעֲבֹ֑ד כָּ֛כָה תַּעֲשֶׂ֥ה לַלְוִיִּ֖ם בְּמִשְׁמְרֹתָֽם׃ הוּא יְשָׁרֵת עִם אֶחָיו בְּאוֹהֶל הַהִתְכַּנְּסוּת לִשְׁמוֹר בְּמִשְׁמֶרֶת וְלֹא יַעֲבוֹד עֲבוֹדָה, כָּכָה תַּעֲשֶׂה לַלְוִיִּים בְּמִשְׁמְרוֹתֵיהֶם.

שלישי[עריכה]

פ פָּרָשַׁת פֶּסַח שֵׁנִי
ט א וַיְדַבֵּ֣ר יְהֹוָ֣ה אֶל־מֹשֶׁ֣ה בְמִדְבַּר־סִ֠ינַ֠י בַּשָּׁנָ֨ה הַשֵּׁנִ֜ית לְצֵאתָ֨ם מֵאֶ֧רֶץ מִצְרַ֛יִם בַּחֹ֥דֶשׁ הָרִאשׁ֖וֹן לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה בְּמִדְבַּר סִינַי בַּשָּׁנָה הַשְּׁנִיָּה לְצֵאתָם מֵאֶרֶץ מִצְרַיִם בַּחוֹדֶשׁ הָרִאשׁוֹן וְאָמַר:
ב וְיַעֲשׂ֧וּ בְנֵי־יִשְׂרָאֵ֛ל אֶת־הַפָּ֖סַח בְּמוֹעֲדֽוֹ׃ בְּנֵי יִשְׂרָאֵל יַעֲשׂוּ אֶת קָרְבַּן הַפֶּסַח בִּזְמַנּוֹ.
ג בְּאַרְבָּעָ֣ה עָשָֽׂר־י֠וֹם בַּחֹ֨דֶשׁ הַזֶּ֜ה בֵּ֧ין הָֽעַרְבַּ֛יִם תַּעֲשׂ֥וּ אֹת֖וֹ בְּמֹעֲד֑וֹ כְּכׇל־חֻקֹּתָ֥יו וּכְכׇל־מִשְׁפָּטָ֖יו תַּעֲשׂ֥וּ אֹתֽוֹ׃ תַּעֲשׂוּ אוֹתוֹ בִּזְמַנּוֹ בַּיּוֹם הָאַרְבָּעָה עָשָׂר בַּחוֹדֶשׁ הַזֶּה אַחַר הַצָּהֳרַיִם, תַּעֲשׂוּ אוֹתוֹ כְּכָל חֻקָּיו וּכְכָל מִשְׁפָּטָיו.
ד וַיְדַבֵּ֥ר מֹשֶׁ֛ה אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לַעֲשֹׂ֥ת הַפָּֽסַח׃ מֹשֶׁה דִּבֵּר אֶל בְּנֵי יִשְׂרָאֵל שֶׁיַּעֲשׂוּ אֶת קָרְבַּן הַפֶּסַח.
ה וַיַּעֲשׂ֣וּ אֶת־הַפֶּ֡סַח בָּרִאשׁ֡וֹן בְּאַרְבָּעָה֩ עָשָׂ֨ר י֥וֹם לַחֹ֛דֶשׁ בֵּ֥ין הָעַרְבַּ֖יִם בְּמִדְבַּ֣ר סִינָ֑י כְּ֠כֹ֠ל אֲשֶׁ֨ר צִוָּ֤ה יְהֹוָה֙ אֶת־מֹשֶׁ֔ה כֵּ֥ן עָשׂ֖וּ בְּנֵ֥י יִשְׂרָאֵֽל׃ הֵם עָשׂוּ אֶת קָרְבַּן הַפֶּסַח בַּחוֹדֶשׁ הָרִאשׁוֹן בְּאַרְבָּעָה עָשָׂר יוֹם לַחוֹדֶשׁ אַחַר הַצָּהֳרַיִם בְּמִדְבַּר סִינַי, כְּכָל מַה שֶּׁה' צִוָּה אֶת מֹשֶׁה כָּךְ בְּנֵי יִשְׂרָאֵל עָשׂוּ.
ו וַיְהִ֣י אֲנָשִׁ֗ים אֲשֶׁ֨ר הָי֤וּ טְמֵאִים֙ לְנֶ֣פֶשׁ אָדָ֔ם וְלֹא־יָכְל֥וּ לַעֲשֹׂת־הַפֶּ֖סַח בַּיּ֣וֹם הַה֑וּא וַֽיִּקְרְב֞וּ לִפְנֵ֥י מֹשֶׁ֛ה וְלִפְנֵ֥י אַהֲרֹ֖ן בַּיּ֥וֹם הַהֽוּא׃ הָיוּ אֲנָשִׁים שֶׁהָיוּ טְמֵאִים מֵאָדָם מֵת וְלֹא יָכְלוּ לַעֲשׂוֹת אֶת קָרְבַּן הַפֶּסַח בְּאוֹתוֹ הַיּוֹם, הֵם בָּאוּ בְּאוֹתוֹ יוֹם לִפְנֵי מֹשֶׁה וְלִפְנֵי אַהֲרֹן.
ז וַ֠יֹּאמְר֠וּ הָאֲנָשִׁ֤ים הָהֵ֙מָּה֙ אֵלָ֔יו אֲנַ֥חְנוּ טְמֵאִ֖ים לְנֶ֣פֶשׁ אָדָ֑ם לָ֣מָּה נִגָּרַ֗ע לְבִלְתִּ֨י הַקְרִ֜יב אֶת־קׇרְבַּ֤ן יְהֹוָה֙ בְּמֹ֣עֲד֔וֹ בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל׃ הָאֲנָשִׁים הָהֵם אָמְרוּ לוֹ: אֲנַחְנוּ טְמֵאִים מֵאָדָם מֵת, לָמָּה נַפְסִיד שֶׁלֹּא נַקְרִיב אֶת קָרְבַּן ה' בִּזְמַנּוֹ עִם בְּנֵי יִשְׂרָאֵל?!
ח וַיֹּ֥אמֶר אֲלֵהֶ֖ם מֹשֶׁ֑ה עִמְד֣וּ וְאֶשְׁמְעָ֔ה מַה־יְצַוֶּ֥ה יְהֹוָ֖ה לָכֶֽם׃ מֹשֶׁה אָמַר לָהֶם: חַכּוּ וְאֶשְׁמַע מַה ה' יְצַוֶּה עֲלֵיכֶם.
פ
ט וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
י דַּבֵּ֛ר אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר אִ֣ישׁ אִ֣ישׁ כִּי־יִהְיֶֽה־טָמֵ֣א ׀ לָנֶ֡פֶשׁ אוֹ֩ בְדֶ֨רֶךְ רְחֹקָ֜הׄ לָכֶ֗ם א֚וֹ לְדֹרֹ֣תֵיכֶ֔ם וְעָ֥שָׂה פֶ֖סַח לַיהֹוָֽה׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל: כָּל אִישׁ שֶׁיִּהְיֶה טָמֵא לְמֵת אוֹ שֶׁיִּהְיֶה בְּדֶרֶךְ רְחוֹקָה, אֶצְלְכֶם אוֹ בְּדוֹר אַחֵר, הוּא יַעֲשֶׂה קָרְבַּן פֶּסַח לַה'.
יא בַּחֹ֨דֶשׁ הַשֵּׁנִ֜י בְּאַרְבָּעָ֨ה עָשָׂ֥ר י֛וֹם בֵּ֥ין הָעַרְבַּ֖יִם יַעֲשׂ֣וּ אֹת֑וֹ עַל־מַצּ֥וֹת וּמְרֹרִ֖ים יֹאכְלֻֽהוּ׃ יַעֲשׂוּ אוֹתוֹ בַּחוֹדֶשׁ הַשֵּׁנִי בְּאַרְבָּעָה עָשָׂר יוֹם אַחַר הַצָּהֳרַיִם, וְיֹאכְלוּ אוֹתוֹ עִם מַצּוֹת וְעִם יְרָקוֹת מָרִים.
יב לֹֽא־יַשְׁאִ֤ירוּ מִמֶּ֙נּוּ֙ עַד־בֹּ֔קֶר וְעֶ֖צֶם לֹ֣א יִשְׁבְּרוּ־ב֑וֹ כְּכׇל־חֻקַּ֥ת הַפֶּ֖סַח יַעֲשׂ֥וּ אֹתֽוֹ׃ לֹא יַשְׁאִירוּ מִמֶּנּוּ עַד הַבּוֹקֶר, וְלֹא יִשְׁבְּרוּ בּוֹ עֶצֶם, יַעֲשׂוּ אוֹתוֹ כְּכָל חֻקֵּי הַפֶּסַח.
יג וְהָאִישׁ֩ אֲשֶׁר־ה֨וּא טָה֜וֹר וּבְדֶ֣רֶךְ לֹא־הָיָ֗ה וְחָדַל֙ לַעֲשׂ֣וֹת הַפֶּ֔סַח וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵֽעַמֶּ֑יהָ כִּ֣י ׀ קׇרְבַּ֣ן יְהֹוָ֗ה לֹ֤א הִקְרִיב֙ בְּמֹ֣עֲד֔וֹ חֶטְא֥וֹ יִשָּׂ֖א הָאִ֥ישׁ הַהֽוּא׃ וְאִישׁ שֶׁהוּא טָהוֹר וְלֹא הָיָה רָחוֹק וְיִמָּנַע מִלַּעֲשׂוֹת אֶת הַפֶּסַח – הַנֶּפֶשׁ הַהִיא תֹּאבַד מֵעַמָּהּ, הָאִישׁ הַהוּא יְקַבֵּל אֶת עוֹנֶשׁ חֶטְאוֹ כִּי הוּא לֹא הִקְרִיב אֶת קָרְבַּן ה' בִּזְמַנּוֹ.
יד וְכִֽי־יָג֨וּר אִתְּכֶ֜ם גֵּ֗ר וְעָ֤שָֽׂה פֶ֙סַח֙ לַֽיהֹוָ֔ה כְּחֻקַּ֥ת הַפֶּ֛סַח וּכְמִשְׁפָּט֖וֹ כֵּ֣ן יַעֲשֶׂ֑ה חֻקָּ֤ה אַחַת֙ יִהְיֶ֣ה לָכֶ֔ם וְלַגֵּ֖ר וּלְאֶזְרַ֥ח הָאָֽרֶץ׃ וְאִם יָגוּר אִתְּכֶם גֵּר הוּא יַעֲשֶׂה קָרְבַּן פֶּסַח לַה', כְּחֻקֵּי הַפֶּסַח וּכְמִשְׁפָּטוֹ כָּךְ יַעֲשֶׂה, חוֹק אֶחָד יִהְיֶה לָכֶם – לַגֵּר וּלְאֶזְרַח הָאָרֶץ.

רביעי[עריכה]

ס פָּרָשַׁת הֶעָנָן
טו וּבְיוֹם֙ הָקִ֣ים אֶת־הַמִּשְׁכָּ֔ן כִּסָּ֤ה הֶֽעָנָן֙ אֶת־הַמִּשְׁכָּ֔ן לְאֹ֖הֶל הָעֵדֻ֑ת וּבָעֶ֜רֶב יִהְיֶ֧ה עַֽל־הַמִּשְׁכָּ֛ן כְּמַרְאֵה־אֵ֖שׁ עַד־בֹּֽקֶר׃ בַּיּוֹם שֶׁהוּקַם הַמִּשְׁכָּן, הֶעָנָן כִּסָּה אֶת מִשְׁכַּן הָאוֹהֶל שֶׁבּוֹ נִמְצָאִים לוּחוֹת הַבְּרִית, וּבָעֶרֶב הָיָה עַל הַמִּשְׁכָּן כְּמַרְאֵה אֵשׁ עַד הַבּוֹקֶר.
טז כֵּ֚ן יִהְיֶ֣ה תָמִ֔יד הֶעָנָ֖ן יְכַסֶּ֑נּוּ וּמַרְאֵה־אֵ֖שׁ לָֽיְלָה׃ כָּךְ הָיָה תָּמִיד, הֶעָנָן מְכַסֶּה אוֹתוֹ בַּיּוֹם וּמַרְאֶה שֶׁל אֵשׁ בַּלַּיְלָה.
יז וּלְפִ֞י הֵעָל֤וֹת הֶֽעָנָן֙ מֵעַ֣ל הָאֹ֔הֶל וְאַ֣חֲרֵי כֵ֔ן יִסְע֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל וּבִמְק֗וֹם אֲשֶׁ֤ר יִשְׁכׇּן־שָׁם֙ הֶֽעָנָ֔ן שָׁ֥ם יַחֲנ֖וּ בְּנֵ֥י יִשְׂרָאֵֽל׃ וּלְפִי סִימַן עֲלִיַּת הֶעָנָן מֵעַל הָאוֹהֶל, מִיַּד אַחֲרֵי כֵן בְּנֵי יִשְׂרָאֵל יִסְעוּ, וּבַמָּקוֹם שֶׁיַּעֲמוֹד שָׁם הֶעָנָן שָׁם בְּנֵי יִשְׂרָאֵל יַחֲנוּ.
יח עַל־פִּ֣י יְהֹוָ֗ה יִסְעוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וְעַל־פִּ֥י יְהֹוָ֖ה יַחֲנ֑וּ כׇּל־יְמֵ֗י אֲשֶׁ֨ר יִשְׁכֹּ֧ן הֶעָנָ֛ן עַל־הַמִּשְׁכָּ֖ן יַחֲנֽוּ׃ בְּנֵי יִשְׂרָאֵל יִסְעוּ עַל פִּי ה' וְיַחֲנוּ עַל פִּי ה', כָּל הַיָּמִים שֶׁהֶעָנָן יִשְׁכּוֹן עַל הַמִּשְׁכָּן הֵם יַחֲנוּ.
יט וּבְהַאֲרִ֧יךְ הֶֽעָנָ֛ן עַל־הַמִּשְׁכָּ֖ן יָמִ֣ים רַבִּ֑ים וְשָׁמְר֧וּ בְנֵי־יִשְׂרָאֵ֛ל אֶת־מִשְׁמֶ֥רֶת יְהֹוָ֖ה וְלֹ֥א יִסָּֽעוּ׃ וּכְשֶׁהֶעָנָן יַאֲרִיךְ לִשְׁהוֹת עַל הַמִּשְׁכָּן יָמִים רַבִּים בְּנֵי יִשְׂרָאֵל יַקְפִּידוּ עַל מִצְוַת ה' וְלֹא יִסְעוּ.
כ וְיֵ֞שׁ אֲשֶׁ֨ר יִהְיֶ֧ה הֶֽעָנָ֛ן יָמִ֥ים מִסְפָּ֖ר עַל־הַמִּשְׁכָּ֑ן עַל־פִּ֤י יְהֹוָה֙ יַחֲנ֔וּ וְעַל־פִּ֥י יְהֹוָ֖ה יִסָּֽעוּ׃ וְיֵשׁ פְּעָמִים שֶׁהֶעָנָן יִשְׁהֶה מִסְפָּר יָמִים עַל הַמִּשְׁכָּן – הֵם יַחֲנוּ עַל פִּי ה' וְיִסְעוּ עַל פִּי ה'.
כא וְיֵ֞שׁ אֲשֶׁר־יִהְיֶ֤ה הֶֽעָנָן֙ מֵעֶ֣רֶב עַד־בֹּ֔קֶר וְנַעֲלָ֧ה הֶֽעָנָ֛ן בַּבֹּ֖קֶר וְנָסָ֑עוּ א֚וֹ יוֹמָ֣ם וָלַ֔יְלָה וְנַעֲלָ֥ה הֶעָנָ֖ן וְנָסָֽעוּ׃ וְיֵשׁ פְּעָמִים שֶׁהֶעָנָן יִשְׁהֶה רַק מֵעֶרֶב עַד בּוֹקֶר, וְהֶעָנָן יַעֲלֶה בַּבּוֹקֶר וְהֵם יִסְעוּ, אוֹ שֶׁיִּשְׁהֶה יוֹם וְלַיְלָה, וְהֶעָנָן יַעֲלֶה וְהֵם יִסְעוּ.
כב אֽוֹ־יֹמַ֜יִם אוֹ־חֹ֣דֶשׁ אוֹ־יָמִ֗ים בְּהַאֲרִ֨יךְ הֶעָנָ֤ן עַל־הַמִּשְׁכָּן֙ לִשְׁכֹּ֣ן עָלָ֔יו יַחֲנ֥וּ בְנֵֽי־יִשְׂרָאֵ֖ל וְלֹ֣א יִסָּ֑עוּ וּבְהֵעָלֹת֖וֹ יִסָּֽעוּ׃ אוֹ יוֹמַיִם, אוֹ חוֹדֶשׁ, אוֹ שָׁנָה – כְּשֶׁהֶעָנָן יַאֲרִיךְ וְיִשְׁהֶה עַל הַמִּשְׁכָּן הַרְבֶּה זְמָן בְּנֵי יִשְׂרָאֵל יַחֲנוּ וְלֹא יִסְעוּ, וּכְשֶׁיַּעֲלֶה הֵם יִסְעוּ.
כג עַל־פִּ֤י יְהֹוָה֙ יַחֲנ֔וּ וְעַל־פִּ֥י יְהֹוָ֖ה יִסָּ֑עוּ אֶת־מִשְׁמֶ֤רֶת יְהֹוָה֙ שָׁמָ֔רוּ עַל־פִּ֥י יְהֹוָ֖ה בְּיַד־מֹשֶֽׁה׃ עַל פִּי ה' הֵם יַחֲנוּ וְעַל פִּי ה' הֵם יִסְעוּ, הֵם מַקְפִּידִים עַל מִצְוַת ה' כְּדִבְרֵי ה' שֶׁבְּאֶמְצָעוּת מֹשֶׁה.
פ פָּרָשַׁת הַחֲצוֹצְרוֹת
י א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב עֲשֵׂ֣ה לְךָ֗ שְׁתֵּי֙ חֲצֽוֹצְרֹ֣ת כֶּ֔סֶף מִקְשָׁ֖ה תַּעֲשֶׂ֣ה אֹתָ֑ם וְהָי֤וּ לְךָ֙ לְמִקְרָ֣א הָֽעֵדָ֔ה וּלְמַסַּ֖ע אֶת־הַֽמַּחֲנֽוֹת׃ עֲשֵׂה לְךָ שְׁתֵּי חֲצוֹצְרוֹת מִכֶּסֶף, מִגּוּשׁ אֶחָד תַּעֲשֶׂה אוֹתָם, הֵם יִהְיוּ לְךָ כְּדֵי לִקְרֹא לַצִּבּוּר וּלְהַסִּיעַ אֶת הַמַּחֲנוֹת.
ג וְתָקְע֖וּ בָּהֵ֑ן וְנֽוֹעֲד֤וּ אֵלֶ֙יךָ֙ כׇּל־הָ֣עֵדָ֔ה אֶל־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ כְּשֶׁיִּתְקְעוּ בָּהֵן יִתְאַסְּפוּ אֵלֶיךָ כָּל הַקָּהָל אֶל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.
ד וְאִם־בְּאַחַ֖ת יִתְקָ֑עוּ וְנוֹעֲד֤וּ אֵלֶ֙יךָ֙ הַנְּשִׂיאִ֔ים רָאשֵׁ֖י אַלְפֵ֥י יִשְׂרָאֵֽל׃ וְאִם יִתְקְעוּ בַּחֲצוֹצְרָה אַחַת, יִתְאַסְּפוּ אֵלֶיךָ רַק הַנְּשִׂיאִים – רָאשֵׁי אַלְפֵי יִשְׂרָאֵל.
ה וּתְקַעְתֶּ֖ם תְּרוּעָ֑ה וְנָֽסְעוּ֙ הַֽמַּחֲנ֔וֹת הַחֹנִ֖ים קֵֽדְמָה׃ כְּשֶׁתִּתְקְעוּ תְּרוּעוֹת קְצָרוֹת יִסְעוּ הַמַּחֲנוֹת הַחוֹנִים בַּמִּזְרָח.
ו וּתְקַעְתֶּ֤ם תְּרוּעָה֙ שֵׁנִ֔ית וְנָֽסְעוּ֙ הַֽמַּחֲנ֔וֹת הַחֹנִ֖ים תֵּימָ֑נָה תְּרוּעָ֥ה יִתְקְע֖וּ לְמַסְעֵיהֶֽם׃ וּכְשֶׁתִּתְקְעוּ תְּרוּעוֹת קְצָרוֹת שֵׁנִית יִסְעוּ הַמַּחֲנוֹת הַחוֹנִים בַּדָּרוֹם – יִתְקְעוּ תְּרוּעָה לְמַסְעוֹתֵיהֶם.
ז וּבְהַקְהִ֖יל אֶת־הַקָּהָ֑ל תִּתְקְע֖וּ וְלֹ֥א תָרִֽיעוּ׃ וּכְדֵי לֶאֱסוֹף אֶת הַקָּהָל תִּתְקְעוּ וְאַל תָּרִיעוּ תְּרוּעוֹת קְצָרוֹת.
ח וּבְנֵ֤י אַהֲרֹן֙ הַכֹּ֣הֲנִ֔ים יִתְקְע֖וּ בַּחֲצֹֽצְר֑וֹת וְהָי֥וּ לָכֶ֛ם לְחֻקַּ֥ת עוֹלָ֖ם לְדֹרֹתֵיכֶֽם׃ בְּנֵי אַהֲרֹן הַכֹּהֲנִים יִתְקְעוּ בַּחֲצוֹצְרוֹת, זֶה יִהְיֶה לָכֶם חוֹק לְעוֹלָם לְדוֹרוֹת.
ט וְכִֽי־תָבֹ֨אוּ מִלְחָמָ֜ה בְּאַרְצְכֶ֗ם עַל־הַצַּר֙ הַצֹּרֵ֣ר אֶתְכֶ֔ם וַהֲרֵעֹתֶ֖ם בַּחֲצֹצְרֹ֑ת וְנִזְכַּרְתֶּ֗ם לִפְנֵי֙ יְהֹוָ֣ה אֱלֹֽהֵיכֶ֔ם וְנוֹשַׁעְתֶּ֖ם מֵאֹיְבֵיכֶֽם׃ וְאִם תָּבוֹאוּ לְמִלְחָמָה בְּאַרְצְכֶם עַל צוֹרֵר שֶׁעוֹשֶׂה לָכֶם צָרוֹת, תָּרִיעוּ בַּחֲצוֹצְרוֹת תְּרוּעוֹת קְצָרוֹת, כָּךְ תִּזָּכְרוּ לְטוֹב לִפְנֵי ה' אֱלֹהֵיכֶם וְתִנָּצְלוּ מֵאוֹיְבֵיכֶם.
י וּבְי֨וֹם שִׂמְחַתְכֶ֥ם וּֽבְמוֹעֲדֵיכֶם֮ וּבְרָאשֵׁ֣י חׇדְשֵׁיכֶם֒‏[1] וּתְקַעְתֶּ֣ם בַּחֲצֹֽצְרֹ֗ת עַ֚ל עֹלֹ֣תֵיכֶ֔ם וְעַ֖ל זִבְחֵ֣י שַׁלְמֵיכֶ֑ם וְהָי֨וּ לָכֶ֤ם לְזִכָּרוֹן֙ לִפְנֵ֣י אֱלֹֽהֵיכֶ֔ם אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ וּבְיוֹם שִׂמְחַתְכֶם וּבְמוֹעֲדֵיכֶם וּבְרָאשֵׁי חָדְשֵׁיכֶם תִּתְקְעוּ בַּחֲצוֹצְרוֹת עַל עוֹלוֹתֵיכֶם וְעַל זִבְחֵי שַׁלְמֵיכֶם, וְיִהְיוּ עֲבוּרְכֶם לְתִזְכּוֹרֶת לִפְנֵי אֱלֹהֵיכֶם, אֲנִי ה' אֱלֹהֵיכֶם.

חמישי[עריכה]

פ פָּרָשַׁת הַמַּסָּע
יא וַיְהִ֞י בַּשָּׁנָ֧ה הַשֵּׁנִ֛ית בַּחֹ֥דֶשׁ הַשֵּׁנִ֖י בְּעֶשְׂרִ֣ים בַּחֹ֑דֶשׁ נַעֲלָה֙ הֶֽעָנָ֔ן מֵעַ֖ל מִשְׁכַּ֥ן הָעֵדֻֽת׃ בַּשָּׁנָה הַשֵּׁנִית בַּחוֹדֶשׁ הַשֵּׁנִי בְּעֶשְׂרִים בַּחוֹדֶשׁ הֶעָנָן עָלָה מֵעַל מִשְׁכַּן לוּחוֹת הַבְּרִית.
יב וַיִּסְע֧וּ בְנֵֽי־יִשְׂרָאֵ֛ל לְמַסְעֵיהֶ֖ם מִמִּדְבַּ֣ר סִינָ֑י וַיִּשְׁכֹּ֥ן הֶעָנָ֖ן בְּמִדְבַּ֥ר פָּארָֽן׃ בְּנֵי יִשְׂרָאֵל נָסְעוּ לְמַסְעוֹתֵיהֶם מִמִּדְבַּר סִינַי, וְהֶעָנָן שָׁכַן בְּמִדְבַּר פָּארָן.
יג וַיִּסְע֖וּ בָּרִאשֹׁנָ֑ה עַל־פִּ֥י יְהֹוָ֖ה בְּיַד־מֹשֶֽׁה׃ הֵם נָסְעוּ בַּפַּעַם הָרִאשׁוֹנָה עַל פִּי דִּבְרֵי ה' שֶׁבְּאֶמְצָעוּת מֹשֶׁה.
יד וַיִּסַּ֞ע דֶּ֣גֶל מַחֲנֵ֧ה בְנֵֽי־יְהוּדָ֛ה בָּרִאשֹׁנָ֖ה לְצִבְאֹתָ֑ם וְעַ֨ל־צְבָא֔וֹ נַחְשׁ֖וֹן בֶּן־עַמִּינָדָֽב׃ קְהַל דֶּגֶל מַחֲנֵה בְּנֵי יְהוּדָה נָסַע רִאשׁוֹן, וְעַל הַקָּהָל שֶׁלּוֹ מֻפְקָד נַחְשׁוֹן בֶּן עַמִּינָדָב.
טו וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י יִשָּׂשכָ֑ר נְתַנְאֵ֖ל בֶּן־צוּעָֽר׃ וְעַל קְהַל שֵׁבֶט בְּנֵי יִשָּׂשכָר מֻפְקָד נְתַנְאֵל בֶּן צוּעָר.
טז וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י זְבוּלֻ֑ן אֱלִיאָ֖ב בֶּן־חֵלֹֽן׃ וְעַל קְהַל שֵׁבֶט בְּנֵי זְבוּלוּן מֻפְקָד אֱלִיאָב בֶּן חֵלוֹן.
יז וְהוּרַ֖ד הַמִּשְׁכָּ֑ן וְנָסְע֤וּ בְנֵֽי־גֵרְשׁוֹן֙ וּבְנֵ֣י מְרָרִ֔י נֹשְׂאֵ֖י הַמִּשְׁכָּֽן׃ וּפֵרְקוּ אֶת הַמִּשְׁכָּן, וּבְנֵי גֵּרְשׁוֹן וּבְנֵי מְרָרִי נוֹשְׂאֵי הַמִּשְׁכָּן נָסְעוּ.
יח וְנָסַ֗ע דֶּ֛גֶל מַחֲנֵ֥ה רְאוּבֵ֖ן לְצִבְאֹתָ֑ם וְעַ֨ל־צְבָא֔וֹ אֱלִיצ֖וּר בֶּן־שְׁדֵיאֽוּר׃ וְנָסַע קְהַל דֶּגֶל מַחֲנֵה רְאוּבֵן, וְעַל הַקָּהָל שֶׁלּוֹ מֻפְקָד אֱלִיצוּר בֶּן שְׁדֵיאוּר.
יט וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י שִׁמְע֑וֹן שְׁלֻֽמִיאֵ֖ל בֶּן־צוּרִֽישַׁדָּֽי׃ וְעַל קְהַל שֵׁבֶט בְּנֵי שִׁמְעוֹן מֻפְקָד שְׁלוּמִיאֵל בֶּן צוּרִישַׁדָּי.
כ וְעַל־צְבָ֖א מַטֵּ֣ה בְנֵי־גָ֑ד אֶלְיָסָ֖ף בֶּן־דְּעוּאֵֽל׃ וְעַל קְהַל שֵׁבֶט בְּנֵי גָּד מֻפְקָד אֶלְיָסָף בֶּן דְּעוּאֵל.
כא וְנָסְעוּ֙ הַקְּהָתִ֔ים נֹשְׂאֵ֖י הַמִּקְדָּ֑שׁ וְהֵקִ֥ימוּ אֶת־הַמִּשְׁכָּ֖ן עַד־בֹּאָֽם׃ וְהַקְּהָתִים נוֹשְׂאֵי כְּלֵי הַמִּקְדָּשׁ נָסְעוּ, וּבְנֵי גֵּרְשׁוֹן וּבְנֵי מְרָרִי הֵקִימוּ אֶת הַמִּשְׁכָּן עַד לְבוֹאָם.
כב וְנָסַ֗ע דֶּ֛גֶל מַחֲנֵ֥ה בְנֵֽי־אֶפְרַ֖יִם לְצִבְאֹתָ֑ם וְעַ֨ל־צְבָא֔וֹ אֱלִישָׁמָ֖ע בֶּן־עַמִּיהֽוּד׃ וְנָסַע קְהַל דֶּגֶל מַחֲנֵה בְּנֵי אֶפְרַיִם, וְעַל הַקָּהָל שֶׁלּוֹ מֻפְקָד אֱלִישָׁמָע בֶּן עַמִּיהוּד.
כג וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י מְנַשֶּׁ֑ה גַּמְלִיאֵ֖ל בֶּן־פְּדָהצֽוּר׃ וְעַל קְהַל שֵׁבֶט בְּנֵי מְנַשֶּׁה מֻפְקָד גַּמְלִיאֵל בֶּן פְּדָהצוּר.
כד וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י בִנְיָמִ֑ן אֲבִידָ֖ן בֶּן־גִּדְעוֹנִֽי׃ וְעַל קְהַל שֵׁבֶט בְּנֵי בִּנְיָמִין מֻפְקָד אֲבִידָן בֶּן גִּדְעוֹנִי.
כה וְנָסַ֗ע דֶּ֚גֶל מַחֲנֵ֣ה בְנֵי־דָ֔ן מְאַסֵּ֥ף לְכׇל־הַֽמַּחֲנֹ֖ת לְצִבְאֹתָ֑ם וְעַ֨ל־צְבָא֔וֹ אֲחִיעֶ֖זֶר בֶּן־עַמִּישַׁדָּֽי׃ וְנָסַע קְהַל דֶּגֶל מַחֲנֵה בְּנֵי דָּן הַמְּאַסֵּף לְכָל הַמַּחֲנוֹת, וְעַל הַקָּהָל שֶׁלּוֹ מֻפְקָד אֲחִיעֶזֶר בֶּן עַמִּישַׁדָּי.
כו וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י אָשֵׁ֑ר פַּגְעִיאֵ֖ל בֶּן־עׇכְרָֽן׃ וְעַל קְהַל שֵׁבֶט בְּנֵי אָשֵׁר מֻפְקָד פַּגְעִיאֵל בֶּן עָכְרָן.
כז וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י נַפְתָּלִ֑י אֲחִירַ֖ע בֶּן־עֵינָֽן׃ וְעַל קְהַל שֵׁבֶט בְּנֵי נַפְתָּלִי מֻפְקָד אֲחִירַע בֶּן עֵינָן.
כח אֵ֛לֶּה מַסְעֵ֥י בְנֵֽי־יִשְׂרָאֵ֖ל לְצִבְאֹתָ֑ם וַיִּסָּֽעוּ׃ אֵלֶּה מַסְעֵי בְּנֵי יִשְׂרָאֵל לִקְהִלּוֹתֵיהֶם, וְהֵם נָסְעוּ.
ס
כט וַיֹּ֣אמֶר מֹשֶׁ֗ה לְ֠חֹבָ֠ב בֶּן־רְעוּאֵ֣ל הַמִּדְיָנִי֮ חֹתֵ֣ן מֹשֶׁה֒ נֹסְעִ֣ים ׀ אֲנַ֗חְנוּ אֶל־הַמָּקוֹם֙ אֲשֶׁ֣ר אָמַ֣ר יְהֹוָ֔ה אֹת֖וֹ אֶתֵּ֣ן לָכֶ֑ם לְכָ֤ה אִתָּ֙נוּ֙ וְהֵטַ֣בְנוּ לָ֔ךְ כִּֽי־יְהֹוָ֥ה דִּבֶּר־ט֖וֹב עַל־יִשְׂרָאֵֽל׃ מֹשֶׁה אָמַר לְחוֹבָב בֶּן רְעוּאֵל הַמִּדְיָנִי חוֹתֵן מֹשֶׁה: אֲנַחְנוּ נוֹסְעִים לַמָּקוֹם שֶׁה' אָמַר עָלָיו: אֶתֵּן לָכֶם אוֹתוֹ, לֵךְ אִתָּנוּ וְנֵטִיב לְךָ כִּי ה' דִּבֵּר דְּבָרִים טוֹבִים עַל יִשְׂרָאֵל!
ל וַיֹּ֥אמֶר אֵלָ֖יו לֹ֣א אֵלֵ֑ךְ כִּ֧י אִם־אֶל־אַרְצִ֛י וְאֶל־מוֹלַדְתִּ֖י אֵלֵֽךְ׃ אָמַר לוֹ: לֹא אֵלֵךְ אִתְּכֶם, אֶלָּא אֵלֵךְ אֶל אַרְצִי וְאֶל מִשְׁפַּחְתִּי.
לא וַיֹּ֕אמֶר אַל־נָ֖א תַּעֲזֹ֣ב אֹתָ֑נוּ כִּ֣י ׀ עַל־כֵּ֣ן יָדַ֗עְתָּ חֲנֹתֵ֙נוּ֙ בַּמִּדְבָּ֔ר וְהָיִ֥יתָ לָּ֖נוּ לְעֵינָֽיִם׃ אָמַר לוֹ: בְּבַקָּשָׁה אַל תַּעֲזוֹב אוֹתָנוּ, כִּי הֲרֵי אַתָּה מַכִּיר אֶת אוֹפֶן חֲנָיָתֵנוּ בַּמִּדְבָּר וְתוּכַל לִהְיוֹת לָנוּ מַדְרִיךְ.
לב וְהָיָ֖ה כִּי־תֵלֵ֣ךְ עִמָּ֑נוּ וְהָיָ֣ה ׀ הַטּ֣וֹב הַה֗וּא אֲשֶׁ֨ר יֵיטִ֧יב יְהֹוָ֛ה עִמָּ֖נוּ וְהֵטַ֥בְנוּ לָֽךְ׃ אִם תֵּלֵךְ אִתָּנוּ, הַטּוֹב הַהוּא שֶׁה' יֵיטִיב אִתָּנוּ נֵטִיב גַּם לְךָ!
לג וַיִּסְעוּ֙ מֵהַ֣ר יְהֹוָ֔ה דֶּ֖רֶךְ שְׁלֹ֣שֶׁת יָמִ֑ים וַאֲר֨וֹן בְּרִית־יְהֹוָ֜ה נֹסֵ֣עַ לִפְנֵיהֶ֗ם דֶּ֚רֶךְ שְׁלֹ֣שֶׁת יָמִ֔ים לָת֥וּר לָהֶ֖ם מְנוּחָֽה׃ הֵם נָסְעוּ מֵהַר ה' דֶּרֶךְ שֶׁל שְׁלוֹשָׁה יְמֵי הֲלִיכָה, וְהָאֲרוֹן שֶׁבְּתוֹכוֹ לוּחוֹת הַבְּרִית עִם ה' נוֹסֵעַ לִפְנֵיהֶם בְּמֶרְחַק דֶּרֶךְ שֶׁל שְׁלוֹשָׁה יָמִים לִמְצֹא לָהֶם מְקוֹם חֲנָיָה.
לד וַעֲנַ֧ן יְהֹוָ֛ה עֲלֵיהֶ֖ם יוֹמָ֑ם בְּנׇסְעָ֖ם מִן־הַֽמַּחֲנֶֽה׃ וַעֲנַן ה' נוֹסֵעַ מֵעֲלֵיהֶם בַּיּוֹם בִּזְמַן נְסִיעָתָם מִמְּקוֹם הַחֲנָיָה.

ששי[עריכה]

ס
לה ׆  וַיְהִ֛י בִּנְסֹ֥עַ הָאָרֹ֖ן וַיֹּ֣אמֶר מֹשֶׁ֑ה קוּמָ֣ה ׀ יְהֹוָ֗ה וְיָפֻ֙צוּ֙ אֹֽיְבֶ֔יךָ וְיָנֻ֥סוּ מְשַׂנְאֶ֖יךָ מִפָּנֶֽיךָ׃ כְּשֶׁהָאָרוֹן הָיָה נוֹסֵעַ מֹשֶׁה אָמַר: קוּם ה' וְאוֹיְבֶיךָ יִתְפַּזְּרוּ – וְיִבְרְחוּ שׂוֹנְאֶיךָ מִמְּךָ!
לו וּבְנֻחֹ֖ה יֹאמַ֑ר שׁוּבָ֣ה יְהֹוָ֔ה רִֽבְב֖וֹת אַלְפֵ֥י יִשְׂרָאֵֽל׃  ׆ וּכְשֶׁחָנָה הָאָרוֹן אָמַר מֹשֶׁה: ה'! תֵּן מְנוּחָה לְרִבְבוֹת וּלְאַלְפֵי יִשְׂרָאֵל.
פ פָּרָשַׁת הַמִּתְאֹנְנִים
יא א וַיְהִ֤י הָעָם֙ כְּמִתְאֹ֣נְנִ֔ים רַ֖ע בְּאׇזְנֵ֣י יְהֹוָ֑ה וַיִּשְׁמַ֤ע יְהֹוָה֙ וַיִּ֣חַר אַפּ֔וֹ וַתִּבְעַר־בָּם֙ אֵ֣שׁ יְהֹוָ֔ה וַתֹּ֖אכַל בִּקְצֵ֥ה הַֽמַּחֲנֶֽה׃ הָעָם הִתְלוֹנְנוּ וְאָמְרוּ דְּבָרִים רָעִים בְּאָזְנֵי ה', ה' שָׁמַע וְכָעַס – וְאֵשׁ ה' בָּעֲרָה בָּהֶם וְשָׂרְפָה בִּקְצֵה הַמַּחֲנֶה.
ב וַיִּצְעַ֥ק הָעָ֖ם אֶל־מֹשֶׁ֑ה וַיִּתְפַּלֵּ֤ל מֹשֶׁה֙ אֶל־יְהֹוָ֔ה וַתִּשְׁקַ֖ע הָאֵֽשׁ׃ הָעָם צָעַק אֶל מֹשֶׁה, מֹשֶׁה הִתְפַּלֵּל אֶל ה' וְהָאֵשׁ שָׁקְעָה.
ג וַיִּקְרָ֛א שֵֽׁם־הַמָּק֥וֹם הַה֖וּא תַּבְעֵרָ֑ה כִּֽי־בָעֲרָ֥ה בָ֖ם אֵ֥שׁ יְהֹוָֽה׃ שֵׁם הַמָּקוֹם הַהוּא נִקְרָא תַּבְעֵרָה כִּי אֵשׁ ה' בָּעֲרָה בָּהֶם.
ד וְהָֽאסַפְסֻף֙ אֲשֶׁ֣ר בְּקִרְבּ֔וֹ הִתְאַוּ֖וּ תַּאֲוָ֑ה וַיָּשֻׁ֣בוּ וַיִּבְכּ֗וּ גַּ֚ם בְּנֵ֣י יִשְׂרָאֵ֔ל וַיֹּ֣אמְר֔וּ מִ֥י יַאֲכִלֵ֖נוּ בָּשָֽׂר׃ וּבְנֵי הַמִּיעוּטִים שֶׁבְּתוֹךְ יִשְׂרָאֵל הִתְאַוּוּ תַּאֲוָה, וְאַחֲרֵיהֶם בָּכוּ גַּם בְּנֵי יִשְׂרָאֵל וְאָמְרוּ: מִי יַאֲכִיל אוֹתָנוּ בָּשָׂר?
ה זָכַ֙רְנוּ֙ אֶת־הַדָּגָ֔ה אֲשֶׁר־נֹאכַ֥ל בְּמִצְרַ֖יִם חִנָּ֑ם אֵ֣ת הַקִּשֻּׁאִ֗ים וְאֵת֙ הָֽאֲבַטִּחִ֔ים וְאֶת־הֶחָצִ֥יר וְאֶת־הַבְּצָלִ֖ים וְאֶת־הַשּׁוּמִֽים׃ זָכַרְנוּ אֶת הַדָּגִים שֶׁאָכַלְנוּ בְּמִצְרַיִם בְּחִנָּם, אֶת הַקִּשּׁוּאִים וְאֶת הָאֲבַטִּיחִים וְאֶת הַלּוּף וְאֶת הַבְּצָלִים וְאֶת הַשּׁוּמִים.
ו וְעַתָּ֛ה נַפְשֵׁ֥נוּ יְבֵשָׁ֖ה אֵ֣ין כֹּ֑ל בִּלְתִּ֖י אֶל־הַמָּ֥ן עֵינֵֽינוּ׃ וְעַכְשָׁיו נַפְשֵׁנוּ יְבֵשָׁה, אֵין לָנוּ כְּלוּם – אֲנַחְנוּ יְכוֹלִים לְצַפּוֹת רַק לַמָּן.
ז וְהַמָּ֕ן כִּזְרַע־גַּ֖ד ה֑וּא וְעֵינ֖וֹ כְּעֵ֥ין הַבְּדֹֽלַח׃ וְהַמָּן הוּא בְּצוּרַת זֶרַע כֻּסְבְּרָה, וְצִבְעוֹ לָבָן כְּמוֹ פְּנִינָה.
ח שָׁ֩טוּ֩ הָעָ֨ם וְלָֽקְט֜וּ וְטָחֲנ֣וּ בָרֵחַ֗יִם א֤וֹ דָכוּ֙ בַּמְּדֹכָ֔ה וּבִשְּׁלוּ֙ בַּפָּר֔וּר וְעָשׂ֥וּ אֹת֖וֹ עֻג֑וֹת וְהָיָ֣ה טַעְמ֔וֹ כְּטַ֖עַם לְשַׁ֥ד הַשָּֽׁמֶן׃ הָעָם שׁוֹטְטוּ וְאָסְפוּ אוֹתוֹ וְטָחֲנוּ בְּרֵחַיִם אוֹ כָּתְשׁוּ בְּמַכְתֵּשׁ וּבִשְּׁלוּ בְּסִיר וְעָשׂוּ מִמֶּנּוּ פִּתּוֹת עֲגוּלוֹת, וְטַעְמוֹ הָיָה כְּטַעַם מַאֲפֶה שֶׁנִּלּוֹשׁ בְּשֶׁמֶן.
ט וּבְרֶ֧דֶת הַטַּ֛ל עַל־הַֽמַּחֲנֶ֖ה לָ֑יְלָה יֵרֵ֥ד הַמָּ֖ן עָלָֽיו׃ בַּלַּיְלָה כְּשֶׁהַטַּל יָרַד עַל הַמַּחֲנֶה, הַמָּן יָרַד עָלָיו.
י וַיִּשְׁמַ֨ע מֹשֶׁ֜ה אֶת־הָעָ֗ם בֹּכֶה֙ לְמִשְׁפְּחֹתָ֔יו אִ֖ישׁ לְפֶ֣תַח אׇהֳל֑וֹ וַיִּֽחַר־אַ֤ף יְהֹוָה֙ מְאֹ֔ד וּבְעֵינֵ֥י מֹשֶׁ֖ה רָֽע׃ מֹשֶׁה שָׁמַע אֶת הָעָם לְמִשְׁפְּחוֹתָיו בּוֹכֶה – כָּל אֶחָד בְּפֶתַח אָהֳלוֹ, ה' כָּעַס מְאוֹד, וְגַם בְּעֵינֵי מֹשֶׁה זֶה הָיָה רָע.
יא וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶל־יְהֹוָ֗ה לָמָ֤ה הֲרֵעֹ֙תָ֙ לְעַבְדֶּ֔ךָ וְלָ֛מָּה לֹא־מָצָ֥תִי חֵ֖ן בְּעֵינֶ֑יךָ לָשׂ֗וּם אֶת־מַשָּׂ֛א כׇּל־הָעָ֥ם הַזֶּ֖ה עָלָֽי׃ מֹשֶׁה אָמַר לַה' לָמָה הֵרַעְתָּ לִי לְעַבְדְּךָ, וְלָמָּה לֹא מָצָאתִי חֵן בְּעֵינֶיךָ שֶׁהֵטַלְתָּ עָלַי אֶת הָאַחֲרָיוּת עַל כָּל הָעָם הַזֶּה?
יב הֶאָנֹכִ֣י הָרִ֗יתִי אֵ֚ת כׇּל־הָעָ֣ם הַזֶּ֔ה אִם־אָנֹכִ֖י יְלִדְתִּ֑יהוּ כִּֽי־תֹאמַ֨ר אֵלַ֜י שָׂאֵ֣הוּ בְחֵיקֶ֗ךָ כַּאֲשֶׁ֨ר יִשָּׂ֤א הָאֹמֵן֙ אֶת־הַיֹּנֵ֔ק עַ֚ל הָֽאֲדָמָ֔ה אֲשֶׁ֥ר נִשְׁבַּ֖עְתָּ לַאֲבֹתָֽיו׃ וְכִי אֲנִי הַהוֹרֶה שֶׁל כָּל הָעָם הַזֶּה?! הַאִם אֲנִי יָלַדְתִּי אוֹתוֹ שֶׁאַתָּה אוֹמֵר אֵלַי: שָׂא אוֹתוֹ בִּזְרוֹעֲךָ כְּמוֹ שֶׁמְּטַפֵּל נוֹשֵׂא תִּינוֹק – לְהָבִיא אוֹתוֹ לָאָרֶץ שֶׁנִּשְׁבַּעְתָּ לַאֲבוֹתָיו?!
יג מֵאַ֤יִן לִי֙ בָּשָׂ֔ר לָתֵ֖ת לְכׇל־הָעָ֣ם הַזֶּ֑ה כִּֽי־יִבְכּ֤וּ עָלַי֙ לֵאמֹ֔ר תְּנָה־לָּ֥נוּ בָשָׂ֖ר וְנֹאכֵֽלָה׃ מִנַּיִן לִי בָּשָׂר לָתֵת לְכָל הָעָם הַזֶּה שֶׁהֵם בּוֹכִים אֵלַי וְאוֹמְרִים: "תֵּן לָנוּ בָּשָׂר וְנֹאכַל"?!
יד לֹֽא־אוּכַ֤ל אָנֹכִי֙ לְבַדִּ֔י לָשֵׂ֖את אֶת־כׇּל־הָעָ֣ם הַזֶּ֑ה כִּ֥י כָבֵ֖ד מִמֶּֽנִּי׃ לֹא אוּכַל אֲנִי לְבַדִּי לָשֵׂאת בָּאַחֲרָיוּת עַל כָּל הָעָם הַזֶּה כִּי זֶה קָשֶׁה מִדַּי בִּשְׁבִילִי.
טו וְאִם־כָּ֣כָה ׀ אַתְּ־עֹ֣שֶׂה לִּ֗י הׇרְגֵ֤נִי נָא֙ הָרֹ֔ג אִם־מָצָ֥אתִי חֵ֖ן בְּעֵינֶ֑יךָ וְאַל־אֶרְאֶ֖ה בְּרָעָתִֽי׃ וְאִם כָּכָה אַתָּה עוֹשֶׂה לִי, הֲרוֹג אוֹתִי בְּבַקָּשָׁה אִם מָצָאתִי חֵן בְּעֵינֶיךָ, כְּדֵי שֶׁלֹּא אֶרְאֶה בְּרָעָתִי!
פ
טז וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה אֶסְפָה־לִּ֞י שִׁבְעִ֣ים אִישׁ֮ מִזִּקְנֵ֣י יִשְׂרָאֵל֒ אֲשֶׁ֣ר יָדַ֔עְתָּ כִּי־הֵ֛ם זִקְנֵ֥י הָעָ֖ם וְשֹׁטְרָ֑יו וְלָקַחְתָּ֤ אֹתָם֙ אֶל־אֹ֣הֶל מוֹעֵ֔ד וְהִֽתְיַצְּב֥וּ שָׁ֖ם עִמָּֽךְ׃ ה' אָמַר לְמֹשֶׁה: אֱסוֹף לִי שִׁבְעִים אִישׁ מִזִּקְנֵי יִשְׂרָאֵל שֶׁאַתָּה יוֹדֵעַ שֶׁהֵם חַכְמֵי הָעָם וְשׁוֹטְרָיו, וְתִקַּח אוֹתָם לְאוֹהֶל הַהִתְכַּנְּסוּת וְיַעַמְדוּ אִתְּךָ שָׁם.
יז וְיָרַדְתִּ֗י וְדִבַּרְתִּ֣י עִמְּךָ֮ שָׁם֒ וְאָצַלְתִּ֗י מִן־הָר֛וּחַ אֲשֶׁ֥ר עָלֶ֖יךָ וְשַׂמְתִּ֣י עֲלֵיהֶ֑ם וְנָשְׂא֤וּ אִתְּךָ֙ בְּמַשָּׂ֣א הָעָ֔ם וְלֹא־תִשָּׂ֥א אַתָּ֖ה לְבַדֶּֽךָ׃ וְאֵרֵד וַאֲדַבֵּר אִתְּךָ שָׁם, וְאֶקַּח מֵהָרוּחַ שֶׁעָלֶיךָ וְאָשִׂים עֲלֵיהֶם, וְיִשְׁתַּתְּפוּ אִתְּךָ בָּאַחֲרָיוּת עַל הָעָם וְלֹא תִּשָּׂא אוֹתָהּ אַתָּה לְבָד.
יח וְאֶל־הָעָ֨ם תֹּאמַ֜ר הִתְקַדְּשׁ֣וּ לְמָחָר֮ וַאֲכַלְתֶּ֣ם בָּשָׂר֒ כִּ֡י בְּכִיתֶם֩ בְּאׇזְנֵ֨י יְהֹוָ֜ה לֵאמֹ֗ר מִ֤י יַאֲכִלֵ֙נוּ֙ בָּשָׂ֔ר כִּי־ט֥וֹב לָ֖נוּ בְּמִצְרָ֑יִם וְנָתַ֨ן יְהֹוָ֥ה לָכֶ֛ם בָּשָׂ֖ר וַאֲכַלְתֶּֽם׃ וְלָעָם תֹּאמַר: הִתְכּוֹנְנוּ לְמָחָר וְתֹאכְלוּ בָּשָׂר, כִּי בְּכִיתֶם בְּאָזְנֵי ה' וַאֲמַרְתֶּם: מִי יַאֲכִיל אוֹתָנוּ בָּשָׂר הֲרֵי הָיָה לָנוּ טוֹב בְּמִצְרַיִם! ה' יִתֵּן לָכֶם בָּשָׂר וְתֹאכְלוּ.
יט לֹ֣א י֥וֹם אֶחָ֛ד תֹּאכְל֖וּן וְלֹ֣א יוֹמָ֑יִם וְלֹ֣א ׀ חֲמִשָּׁ֣ה יָמִ֗ים וְלֹא֙ עֲשָׂרָ֣ה יָמִ֔ים וְלֹ֖א עֶשְׂרִ֥ים יֽוֹם׃ לֹא יוֹם אֶחָד תֹּאכְלוּ, וְלֹא יוֹמַיִם, וְלֹא חֲמִשָּׁה יָמִים, וְלֹא עֲשָׂרָה יָמִים, וְלֹא עֶשְׂרִים יוֹם.
כ עַ֣ד ׀ חֹ֣דֶשׁ יָמִ֗ים עַ֤ד אֲשֶׁר־יֵצֵא֙ מֵֽאַפְּכֶ֔ם וְהָיָ֥ה לָכֶ֖ם לְזָרָ֑א יַ֗עַן כִּֽי־מְאַסְתֶּ֤ם אֶת־יְהֹוָה֙ אֲשֶׁ֣ר בְּקִרְבְּכֶ֔ם וַתִּבְכּ֤וּ לְפָנָיו֙ לֵאמֹ֔ר לָ֥מָּה זֶּ֖ה יָצָ֥אנוּ מִמִּצְרָֽיִם׃ עַד חוֹדֶשׁ יָמִים – עַד שֶׁיֵּצֵא לָכֶם מֵהָאַף וְיִמָּאֵס לָכֶם, בִּגְלַל שֶׁמְּאַסְתֶּם בַּה' שֶׁשּׁוֹכֵן בְּתוֹכְכֶם וּבְכִיתֶם לְפָנָיו וַאֲמַרְתֶּם: לָמָּה יָצָאנוּ מִמִּצְרַיִם!
כא וַיֹּ֘אמֶר֮ מֹשֶׁה֒ שֵׁשׁ־מֵא֥וֹת אֶ֙לֶף֙ רַגְלִ֔י הָעָ֕ם אֲשֶׁ֥ר אָנֹכִ֖י בְּקִרְבּ֑וֹ וְאַתָּ֣ה אָמַ֗רְתָּ בָּשָׂר֙ אֶתֵּ֣ן לָהֶ֔ם וְאָכְל֖וּ חֹ֥דֶשׁ יָמִֽים׃ מֹשֶׁה אָמַר: שֵׁשׁ מֵאוֹת אֶלֶף הוֹלְכֵי רֶגֶל הָעָם שֶׁאֲנִי בְּתוֹכוֹ וְאַתָּה אָמַרְתָּ שֶׁאֶתֵּן לָהֶם בָּשָׂר וְיֹאכְלוּ חוֹדֶשׁ יָמִים?!
כב הֲצֹ֧אן וּבָקָ֛ר יִשָּׁחֵ֥ט לָהֶ֖ם וּמָצָ֣א לָהֶ֑ם אִ֣ם אֶֽת־כׇּל־דְּגֵ֥י הַיָּ֛ם יֵאָסֵ֥ף לָהֶ֖ם וּמָצָ֥א לָהֶֽם׃ הַאִם צֹאן וּבָקָר יִשָּׁחֵט לָהֶם וְיַסְפִּיק לָהֶם?! הַאִם אֶת כָּל דְּגֵי הַיָּם יַאַסְפוּ לָהֶם וְיַסְפִּיק לָהֶם?!
פ
כג וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה הֲיַ֥ד יְהֹוָ֖ה תִּקְצָ֑ר עַתָּ֥ה תִרְאֶ֛ה הֲיִקְרְךָ֥ דְבָרִ֖י אִם־לֹֽא׃ ה' אָמַר לְמֹשֶׁה: וְכִי אֵין בְּכוֹחוֹ שֶׁל ה' לַעֲשׂוֹת כָּל מַה שֶּׁיִּרְצֶה?! עַכְשָׁיו תִּרְאֶה אִם יִקְרֶה מַה שֶּׁאָמַרְתִּי, אוֹ שֶׁלֹּא.
כד וַיֵּצֵ֣א מֹשֶׁ֔ה וַיְדַבֵּר֙ אֶל־הָעָ֔ם אֵ֖ת דִּבְרֵ֣י יְהֹוָ֑ה וַיֶּאֱסֹ֞ף שִׁבְעִ֥ים אִישׁ֙ מִזִּקְנֵ֣י הָעָ֔ם וַֽיַּעֲמֵ֥ד אֹתָ֖ם סְבִיבֹ֥ת הָאֹֽהֶל׃ מֹשֶׁה יָצָא וְדִבֵּר אֶל הָעָם אֶת דִּבְרֵי ה', הוּא אָסַף שִׁבְעִים אִישׁ מִזִּקְנֵי הָעָם וְהֶעֱמִיד אוֹתָם מִסָּבִיב לְאוֹהֶל הַמִּשְׁכָּן.
כה וַיֵּ֨רֶד יְהֹוָ֥ה ׀ בֶּעָנָן֮ וַיְדַבֵּ֣ר אֵלָיו֒ וַיָּ֗אצֶל מִן־הָר֙וּחַ֙ אֲשֶׁ֣ר עָלָ֔יו וַיִּתֵּ֕ן עַל־שִׁבְעִ֥ים אִ֖ישׁ הַזְּקֵנִ֑ים וַיְהִ֗י כְּנ֤וֹחַ עֲלֵיהֶם֙ הָר֔וּחַ וַיִּֽתְנַבְּא֖וּ וְלֹ֥א יָסָֽפוּ׃ כְּבוֹד ה' יָרַד וְהִתְגַּלָּה בֶּעָנָן וְדִבֵּר אֵלָיו, הוּא לָקַח מֵהָרוּחַ שֶׁעָלָיו וְשָׂם עַל שִׁבְעִים הָאֲנָשִׁים הַזְּקֵנִים, וּכְשֶׁנָּחָה עֲלֵיהֶם הָרוּחַ הֵם הִתְנַבְּאוּ, וּמֵאָז לֹא הִתְנַבְּאוּ יוֹתֵר.
כו וַיִּשָּׁאֲר֣וּ שְׁנֵֽי־אֲנָשִׁ֣ים ׀ בַּֽמַּחֲנֶ֡ה שֵׁ֣ם הָאֶחָ֣ד ׀ אֶלְדָּ֡ד וְשֵׁם֩ הַשֵּׁנִ֨י מֵידָ֜ד וַתָּ֧נַח עֲלֵהֶ֣ם הָר֗וּחַ וְהֵ֙מָּה֙ בַּכְּתֻבִ֔ים וְלֹ֥א יָצְא֖וּ הָאֹ֑הֱלָה וַיִּֽתְנַבְּא֖וּ בַּֽמַּחֲנֶֽה׃ נִשְׁאֲרוּ שְׁנֵי אֲנָשִׁים בַּמַּחֲנֶה שֵׁם הָאֶחָד אֶלְדָּד וְשֵׁם הַשֵּׁנִי מֵידָד, הָרוּחַ נָחָה עֲלֵיהֶם וְהֵם הָיוּ מֵהַמֻּזְמַנִּים אֲבָל הֵם לֹא יָצְאוּ לָאוֹהֶל, וְהֵם הִתְנַבְּאוּ בַּמַּחֲנֶה.
כז וַיָּ֣רׇץ הַנַּ֔עַר וַיַּגֵּ֥ד לְמֹשֶׁ֖ה וַיֹּאמַ֑ר אֶלְדָּ֣ד וּמֵידָ֔ד מִֽתְנַבְּאִ֖ים בַּֽמַּחֲנֶֽה׃ אַחַד הַנְּעָרִים רָץ וְסִפֵּר לְמֹשֶׁה, וְאָמַר: אֶלְדָּד וּמֵידָד מִתְנַבְּאִים בַּמַּחֲנֶה!
כח וַיַּ֜עַן יְהוֹשֻׁ֣עַ בִּן־נ֗וּן מְשָׁרֵ֥ת מֹשֶׁ֛ה מִבְּחֻרָ֖יו וַיֹּאמַ֑ר אֲדֹנִ֥י מֹשֶׁ֖ה כְּלָאֵֽם׃ עָנָה יְהוֹשֻׁעַ בִּן נוּן מֵהַמֻּבְחָרִים שֶׁבִּמְשָׁרְתֵי מֹשֶׁה וְאָמַר: אֲדוֹנִי מֹשֶׁה הַכְנֵס אוֹתָם לַכֶּלֶא!
כט וַיֹּ֤אמֶר לוֹ֙ מֹשֶׁ֔ה הַֽמְקַנֵּ֥א אַתָּ֖ה לִ֑י וּמִ֨י יִתֵּ֜ן כׇּל־עַ֤ם יְהֹוָה֙ נְבִיאִ֔ים כִּי־יִתֵּ֧ן יְהֹוָ֛ה אֶת־רוּח֖וֹ עֲלֵיהֶֽם׃ אָמַר לוֹ מֹשֶׁה: הַאִם אַתָּה דּוֹאֵג לִכְבוֹדִי? מִי יִתֵּן שֶׁכָּל עַם ה' יִהְיוּ נְבִיאִים – שֶׁה' יָנִיחַ אֶת רוּחוֹ עֲלֵיהֶם!

שביעי[עריכה]

ל וַיֵּאָסֵ֥ף מֹשֶׁ֖ה אֶל־הַֽמַּחֲנֶ֑ה ה֖וּא וְזִקְנֵ֥י יִשְׂרָאֵֽל׃ מֹשֶׁה וְזִקְנֵי יִשְׂרָאֵל נִכְנְסוּ לַמַּחֲנֶה.
לא וְר֜וּחַ נָסַ֣ע ׀ מֵאֵ֣ת יְהֹוָ֗ה וַיָּ֣גׇז שַׂלְוִים֮ מִן־הַיָּם֒ וַיִּטֹּ֨שׁ עַל־הַֽמַּחֲנֶ֜ה כְּדֶ֧רֶךְ י֣וֹם כֹּ֗ה וּכְדֶ֤רֶךְ יוֹם֙ כֹּ֔ה סְבִיב֖וֹת הַֽמַּחֲנֶ֑ה וּכְאַמָּתַ֖יִם עַל־פְּנֵ֥י הָאָֽרֶץ׃ וְרוּחַ נָשְׁבָה מֵה' וְהִפְרִיחָה שַׂלְוִים מֵהַיָּם וּפִזְּרָה אוֹתָם עַל הַמַּחֲנֶה בְּמֶרְחָק שֶׁל דֶּרֶךְ יוֹם הֲלִיכָה לְכָאן וְדֶרֶךְ יוֹם לְכָאן מִסָּבִיב לַמַּחֲנֶה וּבְגוֹבָהּ שֶׁל שְׁתֵּי אַמּוֹת (כְּמֶטֶר) עַל פְּנֵי הָאָרֶץ.
לב וַיָּ֣קׇם הָעָ֡ם כׇּל־הַיּוֹם֩ הַה֨וּא וְכׇל־הַלַּ֜יְלָה וְכֹ֣ל ׀ י֣וֹם הַֽמׇּחֳרָ֗ת וַיַּֽאַסְפוּ֙ אֶת־הַשְּׂלָ֔ו הַמַּמְעִ֕יט אָסַ֖ף עֲשָׂרָ֣ה חֳמָרִ֑ים וַיִּשְׁטְח֤וּ לָהֶם֙ שָׁט֔וֹחַ סְבִיב֖וֹת הַֽמַּחֲנֶֽה׃ כָּל הַיּוֹם הַהוּא וְכָל הַלַּיְלָה וְכָל יוֹם הַמָּחֳרָת הָעָם קָמוּ וְאָסְפוּ אֶת הַשְּׂלָו, הַמַּמְעִיט אָסַף עֲשָׂרָה כּוּרִים (כּ2160 לִיטֶר), הֵם שָׁטְחוּ לָהֶם אֶת הַבָּשָׂר מִסָּבִיב לַמַּחֲנֶה כְּדֵי לְיַבֵּשׁ וּלְשַׁמֵּר אוֹתוֹ.
לג הַבָּשָׂ֗ר עוֹדֶ֙נּוּ֙ בֵּ֣ין שִׁנֵּיהֶ֔ם טֶ֖רֶם יִכָּרֵ֑ת וְאַ֤ף יְהֹוָה֙ חָרָ֣ה בָעָ֔ם וַיַּ֤ךְ יְהֹוָה֙ בָּעָ֔ם מַכָּ֖ה רַבָּ֥ה מְאֹֽד׃ הַבָּשָׂר הָיָה עֲדַיִן בֵּין שִׁנֵּיהֶם – עוֹד לֹא הִסְפִּיקוּ לְגָמְרוֹ, וְהַכַּעַס שֶׁל ה' פָּגַע בָּעָם, ה' הִכָּה בָּעָם מַכָּה גְּדוֹלָה מְאוֹד.
לד וַיִּקְרָ֛א אֶת־שֵֽׁם־הַמָּק֥וֹם הַה֖וּא קִבְר֣וֹת הַֽתַּאֲוָ֑ה כִּי־שָׁם֙ קָֽבְר֔וּ אֶת־הָעָ֖ם הַמִּתְאַוִּֽים׃ שֵׁם הַמָּקוֹם הַהוּא נִקְרָא קִבְרוֹת הַתַּאֲוָה כִּי שָׁם קָבְרוּ אֶת אַנְשֵׁי הָעָם שֶׁהִתְאַוּוּ.
לה מִקִּבְר֧וֹת הַֽתַּאֲוָ֛ה נָסְע֥וּ הָעָ֖ם חֲצֵר֑וֹת וַיִּהְי֖וּ בַּחֲצֵרֽוֹת׃ מִקִּבְרוֹת הַתַּאֲוָה הָעָם נָסְעוּ לַחֲצֵרוֹת וְחָנוּ בַּחֲצֵרוֹת.
פ פָּרָשַׁת מִרְיָם
יב א וַתְּדַבֵּ֨ר מִרְיָ֤ם וְאַהֲרֹן֙ בְּמֹשֶׁ֔ה עַל־אֹד֛וֹת הָאִשָּׁ֥ה הַכֻּשִׁ֖ית אֲשֶׁ֣ר לָקָ֑ח כִּֽי־אִשָּׁ֥ה כֻשִׁ֖ית לָקָֽח׃ מִרְיָם דִּבְּרָה עִם אַהֲרֹן עַל מֹשֶׁה – עַל אוֹדוֹת הָאִשָּׁה הַכּוּשִׁית שֶׁלָּקַח לְאִשָּׁה וְעָזַב, שֶׁהֲרֵי אִשָּׁה כּוּשִׁית לָקַח.
ב וַיֹּאמְר֗וּ הֲרַ֤ק אַךְ־בְּמֹשֶׁה֙ דִּבֶּ֣ר יְהֹוָ֔ה הֲלֹ֖א גַּם־בָּ֣נוּ דִבֵּ֑ר וַיִּשְׁמַ֖ע יְהֹוָֽה׃ הֵם אָמְרוּ: וַדָּאי לֹא עָזַב אוֹתָהּ מִפְּנֵי שֶׁהוּא נָבִיא! הַאִם רַק עִם מֹשֶׁה ה' דִּבֵּר בִּנְבוּאָה? הֲרֵי גַּם אִתָּנוּ דִּבֵּר! ה' שָׁמַע אֶת דִּבְרֵיהֶם.
ג וְהָאִ֥ישׁ מֹשֶׁ֖ה עָנָ֣ו מְאֹ֑ד מִכֹּל֙ הָֽאָדָ֔ם אֲשֶׁ֖ר עַל־פְּנֵ֥י הָאֲדָמָֽה׃ וְהָאִישׁ מֹשֶׁה עָנָיו מְאוֹד, יוֹתֵר מִכָּל הָאָדָם שֶׁעַל פְּנֵי הָאֲדָמָה.
ס
ד וַיֹּ֨אמֶר יְהֹוָ֜ה פִּתְאֹ֗ם אֶל־מֹשֶׁ֤ה וְאֶֽל־אַהֲרֹן֙ וְאֶל־מִרְיָ֔ם צְא֥וּ שְׁלׇשְׁתְּכֶ֖ם אֶל־אֹ֣הֶל מוֹעֵ֑ד וַיֵּצְא֖וּ שְׁלׇשְׁתָּֽם׃ ה' אָמַר פִּתְאוֹם לְמֹשֶׁה לְאַהֲרֹן וּלְמִרְיָם: צְאוּ שְׁלָשְׁתְּכֶם לְאוֹהֶל הַהִתְכַּנְּסוּת! שְׁלָשְׁתָּם יָצְאוּ.
ה וַיֵּ֤רֶד יְהֹוָה֙ בְּעַמּ֣וּד עָנָ֔ן וַֽיַּעֲמֹ֖ד פֶּ֣תַח הָאֹ֑הֶל וַיִּקְרָא֙ אַהֲרֹ֣ן וּמִרְיָ֔ם וַיֵּצְא֖וּ שְׁנֵיהֶֽם׃ כְּבוֹד ה' יָרַד וְהִתְגַּלָּה בְּעַמּוּד עָנָן וְעָמַד בְּפֶתַח הָאוֹהֶל, הוּא קָרָא לְאַהֲרֹן וּמִרְיָם וּשְׁנֵיהֶם יָצְאוּ.
ו וַיֹּ֖אמֶר שִׁמְעוּ־נָ֣א דְבָרָ֑י אִם־יִֽהְיֶה֙ נְבִ֣יאֲכֶ֔ם יְהֹוָ֗ה בַּמַּרְאָה֙ אֵלָ֣יו אֶתְוַדָּ֔ע בַּחֲל֖וֹם אֲדַבֶּר־בּֽוֹ׃ אָמַר לָהֶם: שִׁמְעוּ נָא אֶת דְּבָרַי! אִם יִהְיֶה נָבִיא מֵאִתְּכֶם, אֲנִי ה' אֶתְגַּלֶּה אֵלָיו בִּתְמוּנָה, אֲדַבֵּר אִתּוֹ בַּחֲלוֹם.
ז לֹא־כֵ֖ן עַבְדִּ֣י מֹשֶׁ֑ה בְּכׇל־בֵּיתִ֖י נֶאֱמָ֥ן הֽוּא׃ לֹא כָּךְ עַבְדִּי מֹשֶׁה, הוּא רַשָּׁאי לְהִכָּנֵס כִּרְצוֹנוֹ בְּכָל בֵּיתִי.
ח פֶּ֣ה אֶל־פֶּ֞ה אֲדַבֶּר־בּ֗וֹ וּמַרְאֶה֙ וְלֹ֣א בְחִידֹ֔ת וּתְמֻנַ֥ת יְהֹוָ֖ה יַבִּ֑יט וּמַדּ֙וּעַ֙ לֹ֣א יְרֵאתֶ֔ם לְדַבֵּ֖ר בְּעַבְדִּ֥י בְמֹשֶֽׁה׃ אֲנִי מְדַבֵּר אִתּוֹ פָּנִים אֶל פָּנִים וּבְמַרְאֶה בָּרוּר וְלֹא בְּחִידוֹת, וְהוּא מַבִּיט בִּתְמוּנַת כְּבוֹד ה', וְאֵיךְ הֵעַזְתֶּם לְדַבֵּר עַל עַבְדִּי מֹשֶׁה?!
ט וַיִּֽחַר־אַ֧ף יְהֹוָ֛ה בָּ֖ם וַיֵּלַֽךְ׃ ה' כָּעַס עֲלֵיהֶם, וְהָלַךְ.
י וְהֶעָנָ֗ן סָ֚ר מֵעַ֣ל הָאֹ֔הֶל וְהִנֵּ֥ה מִרְיָ֖ם מְצֹרַ֣עַת כַּשָּׁ֑לֶג וַיִּ֧פֶן אַהֲרֹ֛ן אֶל־מִרְיָ֖ם וְהִנֵּ֥ה מְצֹרָֽעַת׃ וְהֶעָנָן סָר מֵעַל הָאוֹהֶל וְהִנֵּה מִרְיָם מְצוֹרַעַת כַּשֶּׁלֶג, אַהֲרֹן פָּנָה לְמִרְיָם וְהִנֵּה הִיא מְצוֹרַעַת.
יא וַיֹּ֥אמֶר אַהֲרֹ֖ן אֶל־מֹשֶׁ֑ה בִּ֣י אֲדֹנִ֔י אַל־נָ֨א תָשֵׁ֤ת עָלֵ֙ינוּ֙ חַטָּ֔את אֲשֶׁ֥ר נוֹאַ֖לְנוּ וַאֲשֶׁ֥ר חָטָֽאנוּ׃ אַהֲרֹן אָמַר לְמֹשֶׁה: בְּבַקָּשָׁה אֲדוֹנִי! בְּבַקָּשָׁה, אַל תַּחְשִׁיב אֶת זֶה כְּחֵטְא שֶׁטָּעִינוּ בְּטִפְּשׁוּתֵנוּ וְשֶׁחָטָאנוּ.
יב אַל־נָ֥א תְהִ֖י כַּמֵּ֑ת אֲשֶׁ֤ר בְּצֵאתוֹ֙ מֵרֶ֣חֶם אִמּ֔וֹ וַיֵּאָכֵ֖ל חֲצִ֥י בְשָׂרֽוֹ׃ בְּבַקָּשָׁה, שֶׁאֲחוֹתֵנוּ לֹא תִּהְיֶה כְּמוֹ מֵת! שְׁלוֹשְׁתֵּנוּ יָצָאנוּ מֵרֶחֶם אִמֵּנוּ, וְעַכְשָׁיו כְִּאִלּוּ נָמַק חֲצִי בְּשָׂרֵנוּ.
יג וַיִּצְעַ֣ק מֹשֶׁ֔ה אֶל־יְהֹוָ֖ה לֵאמֹ֑ר אֵ֕ל נָ֛א רְפָ֥א נָ֖א לָֽהּ׃ מֹשֶׁה צָעַק אֶל ה' וְאָמַר: בְּבַקָּשָׁה אֵל! רַפֵּא אוֹתָהּ בְּבַקָּשָׁה!

מפטיר[עריכה]

פ
יד וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה וְאָבִ֙יהָ֙ יָרֹ֤ק יָרַק֙ בְּפָנֶ֔יהָ הֲלֹ֥א תִכָּלֵ֖ם שִׁבְעַ֣ת יָמִ֑ים תִּסָּגֵ֞ר שִׁבְעַ֤ת יָמִים֙ מִח֣וּץ לַֽמַּחֲנֶ֔ה וְאַחַ֖ר תֵּאָסֵֽף׃ ה' אָמַר לְמֹשֶׁה: אִלּוּ אָבִיהָ הָיָה יוֹרֵק בְּפָנֶיהָ הֲרֵי הָיְתָה מִתְבַּיֶּשֶׁת שִׁבְעָה יָמִים, לָכֵן תִּסְתַּגֵּר שִׁבְעָה יָמִים מִחוּץ לַמַּחֲנֶה וְאַחַר כָּךְ תִּכָּנֵס לַמַּחֲנֶה.
טו וַתִּסָּגֵ֥ר מִרְיָ֛ם מִח֥וּץ לַֽמַּחֲנֶ֖ה שִׁבְעַ֣ת יָמִ֑ים וְהָעָם֙ לֹ֣א נָסַ֔ע עַד־הֵאָסֵ֖ף מִרְיָֽם׃ מִרְיָם הִסְתַּגְּרָה מִחוּץ לַמַּחֲנֶה שִׁבְעָה יָמִים, וְהָעָם לֹא נָסַע עַד שֶׁמִּרְיָם נִכְנְסָה לַמַּחֲנֶה.
טז וְאַחַ֛ר נָסְע֥וּ הָעָ֖ם מֵחֲצֵר֑וֹת וַֽיַּחֲנ֖וּ בְּמִדְבַּ֥ר פָּארָֽן׃ וְאַחַר כָּךְ הָעָם נָסְעוּ מֵחֲצֵרוֹת וְחָנוּ בְּמִדְבַּר פָּארָן.
