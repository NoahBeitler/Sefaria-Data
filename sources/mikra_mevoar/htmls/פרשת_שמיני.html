ראשון[עריכה]

ס פָּרָשַׁת הַיּוֹם הַשְּׁמִינִי לַמִּלּוּאִים
ט א וַיְהִי֙ בַּיּ֣וֹם הַשְּׁמִינִ֔י קָרָ֣א מֹשֶׁ֔ה לְאַהֲרֹ֖ן וּלְבָנָ֑יו וּלְזִקְנֵ֖י יִשְׂרָאֵֽל׃ בַּיּוֹם הַשְּׁמִינִי מֹשֶׁה קָרָא לְאַהֲרֹן וּלְבָנָיו וּלְזִקְנֵי יִשְׂרָאֵל.
ב וַיֹּ֣אמֶר אֶֽל־אַהֲרֹ֗ן קַח־לְ֠ךָ֠ עֵ֣גֶל בֶּן־בָּקָ֧ר לְחַטָּ֛את וְאַ֥יִל לְעֹלָ֖ה תְּמִימִ֑ם וְהַקְרֵ֖ב לִפְנֵ֥י יְהֹוָֽה׃ הוּא אָמַר לְאַהֲרֹן: קַח עֵגֶל מֵהַבָּקָר לְחַטָּאת וְאַיִל לְעוֹלָה – לְלֹא מוּם, וְהַגֵּשׁ אוֹתָם לִפְנֵי ה'.
ג וְאֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל תְּדַבֵּ֣ר לֵאמֹ֑ר קְח֤וּ שְׂעִיר־עִזִּים֙ לְחַטָּ֔את וְעֵ֨גֶל וָכֶ֧בֶשׂ בְּנֵי־שָׁנָ֛ה תְּמִימִ֖ם לְעֹלָֽה׃ וּתְדַבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְתֹאמַר: קְחוּ תַּיִשׁ לְחַטָּאת, וְעֵגֶל וְכֶבֶשׂ שֶׁגִּילָם עַד שָׁנָה לְלֹא מוּם לְעוֹלָה.
ד וְשׁ֨וֹר וָאַ֜יִל לִשְׁלָמִ֗ים לִזְבֹּ֙חַ֙ לִפְנֵ֣י יְהֹוָ֔ה וּמִנְחָ֖ה בְּלוּלָ֣ה בַשָּׁ֑מֶן כִּ֣י הַיּ֔וֹם יְהֹוָ֖ה נִרְאָ֥ה אֲלֵיכֶֽם׃ וְשׁוֹר וְאַיִל לִשְׁלָמִים לִשְׁחוֹט לִפְנֵי ה' וּמִנְחָה מְעֻרְבֶּבֶת בְּשֶׁמֶן, כִּי הַיּוֹם ה' יִתְגַּלֶּה אֲלֵיכֶם.
ה וַיִּקְח֗וּ אֵ֚ת אֲשֶׁ֣ר צִוָּ֣ה מֹשֶׁ֔ה אֶל־פְּנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וַֽיִּקְרְבוּ֙ כׇּל־הָ֣עֵדָ֔ה וַיַּֽעַמְד֖וּ לִפְנֵ֥י יְהֹוָֽה׃ הֵם לָקְחוּ אֶת מַה שֶּׁמֹּשֶׁה צִוָּה אֶל לִפְנֵי אוֹהֶל הַהִתְכַּנְּסוּת, וְכָל הַצִּבּוּר נִגַּשׁ וְעָמַד לִפְנֵי ה'.
ו וַיֹּ֣אמֶר מֹשֶׁ֔ה זֶ֧ה הַדָּבָ֛ר אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה תַּעֲשׂ֑וּ וְיֵרָ֥א אֲלֵיכֶ֖ם כְּב֥וֹד יְהֹוָֽה׃ מֹשֶׁה אָמַר: זֶה הַדָּבָר שֶׁה' צִוָּה שֶׁתַּעֲשׂוּ, וְאָז כְּבוֹד ה' יִתְגַּלֶּה אֲלֵיכֶם:
ז וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶֽל־אַהֲרֹ֗ן קְרַ֤ב אֶל־הַמִּזְבֵּ֙חַ֙ וַעֲשֵׂ֞ה אֶת־חַטָּֽאתְךָ֙ וְאֶת־עֹ֣לָתֶ֔ךָ וְכַפֵּ֥ר בַּֽעַדְךָ֖ וּבְעַ֣ד הָעָ֑ם וַעֲשֵׂ֞ה אֶת־קׇרְבַּ֤ן הָעָם֙ וְכַפֵּ֣ר בַּֽעֲדָ֔ם כַּאֲשֶׁ֖ר צִוָּ֥ה יְהֹוָֽה׃ מֹשֶׁה אָמַר לְאַהֲרֹן: גַּשׁ לַמִּזְבֵּחַ וַעֲשֵׂה אֶת חַטָּאתְךָ וְאֶת עוֹלָתְךָ וְכַפֵּר עַל עַצְמְךָ וְעַל הָעָם, וַעֲשֵׂה אֶת קָרְבַּן הָעָם וְכַפֵּר עֲלֵיהֶם כְּמוֹ שֶׁה' צִוָּה.
ח וַיִּקְרַ֥ב אַהֲרֹ֖ן אֶל־הַמִּזְבֵּ֑חַ וַיִּשְׁחַ֛ט אֶת־עֵ֥גֶל הַחַטָּ֖את אֲשֶׁר־לֽוֹ׃ אַהֲרֹן נִגַּשׁ לַמִּזְבֵּחַ וְשָׁחַט אֶת עֵגֶל הַחַטָּאת שֶׁלּוֹ.
ט וַ֠יַּקְרִ֠בוּ בְּנֵ֨י אַהֲרֹ֣ן אֶת־הַדָּם֮ אֵלָיו֒ וַיִּטְבֹּ֤ל אֶצְבָּעוֹ֙ בַּדָּ֔ם וַיִּתֵּ֖ן עַל־קַרְנ֣וֹת הַמִּזְבֵּ֑חַ וְאֶת־הַדָּ֣ם יָצַ֔ק אֶל־יְס֖וֹד הַמִּזְבֵּֽחַ׃ בְּנֵי אַהֲרֹן הִגִּישׁוּ לוֹ אֶת הַדָּם, הוּא טָבַל אֶת אֶצְבָּעוֹ בַּדָּם וְשָׂם עַל פִּנּוֹת הַמִּזְבֵּחַ, וְאֶת שְׁאָר הַדָּם הוּא שָׁפַךְ עַל יְסוֹד הַמִּזְבֵּחַ.
י וְאֶת־הַחֵ֨לֶב וְאֶת־הַכְּלָיֹ֜ת וְאֶת־הַיֹּתֶ֤רֶת מִן־הַכָּבֵד֙ מִן־הַ֣חַטָּ֔את הִקְטִ֖יר הַמִּזְבֵּ֑חָה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ וְאֶת הַחֵלֶב וְאֶת הַכְּלָיוֹת וְאֶת הַתּוֹסֶפֶת שֶׁעַל הַכָּבֵד מֵהַחַטָּאת הוּא הִקְטִיר עַל הַמִּזְבֵּחַ, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.
יא וְאֶת־הַבָּשָׂ֖ר וְאֶת־הָע֑וֹר שָׂרַ֣ף בָּאֵ֔שׁ מִח֖וּץ לַֽמַּחֲנֶֽה׃ וְאֶת הַבָּשָׂר וְאֶת הָעוֹר הוּא שָׂרַף בָּאֵשׁ מִחוּץ לַמַּחֲנֶה.
יב וַיִּשְׁחַ֖ט אֶת־הָעֹלָ֑ה וַ֠יַּמְצִ֠אוּ בְּנֵ֨י אַהֲרֹ֤ן אֵלָיו֙ אֶת־הַדָּ֔ם וַיִּזְרְקֵ֥הוּ עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ הוּא שָׁחַט אֶת הָעוֹלָה, בְּנֵי אַהֲרֹן הֵבִיאוּ לוֹ אֶת הַדָּם, וְהוּא זָרַק אוֹתוֹ עַל הַמִּזְבֵּחַ מִסָּבִיב.
יג וְאֶת־הָעֹלָ֗ה הִמְצִ֧יאוּ אֵלָ֛יו לִנְתָחֶ֖יהָ וְאֶת־הָרֹ֑אשׁ וַיַּקְטֵ֖ר עַל־הַמִּזְבֵּֽחַ׃ הֵם הֵבִיאוּ אֵלָיו אֶת הָעוֹלָה מְחֻלֶּקֶת לַחֲלָקֶיהָ וְאֶת הָרֹאשׁ, וְהוּא הִקְטִיר עַל הַמִּזְבֵּחַ.
יד וַיִּרְחַ֥ץ אֶת־הַקֶּ֖רֶב וְאֶת־הַכְּרָעָ֑יִם וַיַּקְטֵ֥ר עַל־הָעֹלָ֖ה הַמִּזְבֵּֽחָה׃ הוּא רָחַץ אֶת הַמֵּעַיִם וְאֶת הָרַגְלַיִם, וְהִקְטִיר אוֹתָם עַל הָעוֹלָה עַל הַמִּזְבֵּחַ.
טו וַיַּקְרֵ֕ב אֵ֖ת קׇרְבַּ֣ן הָעָ֑ם וַיִּקַּ֞ח אֶת־שְׂעִ֤יר הַֽחַטָּאת֙ אֲשֶׁ֣ר לָעָ֔ם וַיִּשְׁחָטֵ֥הוּ וַֽיְחַטְּאֵ֖הוּ כָּרִאשֽׁוֹן׃ הוּא הִגִּישׁ אֶת קָרְבַּן הָעָם, הוּא לָקַח אֶת הַתַּיִשׁ לְחַטָּאת שֶׁל הָעָם וְשָׁחַט אוֹתוֹ וְכִפֵּר בְּדָמוֹ כְּמוֹ שֶׁעָשָׂה בַּקָּרְבָּן הָרִאשׁוֹן.
טז וַיַּקְרֵ֖ב אֶת־הָעֹלָ֑ה וַֽיַּעֲשֶׂ֖הָ כַּמִּשְׁפָּֽט׃ הוּא הִגִּישׁ אֶת הָעוֹלָה וְעָשָׂה אוֹתָהּ כַּדִּין.

שני[עריכה]

יז וַיַּקְרֵב֮ אֶת־הַמִּנְחָה֒ וַיְמַלֵּ֤א כַפּוֹ֙ מִמֶּ֔נָּה וַיַּקְטֵ֖ר עַל־הַמִּזְבֵּ֑חַ מִלְּבַ֖ד עֹלַ֥ת הַבֹּֽקֶר׃ הוּא הִגִּישׁ אֶת הַמִּנְחָה וְחָפַן מִמֶּנָּה בְּיָדוֹ וְהִקְטִיר עַל הַמִּזְבֵּחַ, בְּנוֹסָף לַמִּנְחָה שֶׁל עוֹלַת הַבּוֹקֶר.
יח וַיִּשְׁחַ֤ט אֶת־הַשּׁוֹר֙ וְאֶת־הָאַ֔יִל זֶ֥בַח הַשְּׁלָמִ֖ים אֲשֶׁ֣ר לָעָ֑ם וַ֠יַּמְצִ֠אוּ בְּנֵ֨י אַהֲרֹ֤ן אֶת־הַדָּם֙ אֵלָ֔יו וַיִּזְרְקֵ֥הוּ עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ הוּא שָׁחַט אֶת הַשּׁוֹר וְאֶת הָאַיִל – זֶבַח הַשְּׁלָמִים שֶׁל הָעָם, בְּנֵי אַהֲרֹן הֵבִיאוּ לוֹ אֶת הַדָּם וְהוּא זָרַק אוֹתוֹ עַל הַמִּזְבֵּחַ מִסָּבִיב.
יט וְאֶת־הַחֲלָבִ֖ים מִן־הַשּׁ֑וֹר וּמִ֨ן־הָאַ֔יִל הָֽאַלְיָ֤ה וְהַֽמְכַסֶּה֙ וְהַכְּלָיֹ֔ת וְיֹתֶ֖רֶת הַכָּבֵֽד׃ וְאֶת גּוּשֵׁי הַחֵלֶב מֵהַשּׁוֹר, וּמֵהָאַיִל – אֶת שׁוּמַן הַזָּנָב וְהַחֵלֶב שֶׁמְּכַסֶּה אֶת הַמֵּעַיִם וְהַכְּלָיוֹת וְהַתּוֹסֶפֶת לַכָּבֵד.
כ וַיָּשִׂ֥ימוּ אֶת־הַחֲלָבִ֖ים עַל־הֶחָז֑וֹת וַיַּקְטֵ֥ר הַחֲלָבִ֖ים הַמִּזְבֵּֽחָה׃ הֵם שָׂמוּ אֶת גּוּשֵׁי הַחֵלֶב עַל חֲזֵה הַשּׁוֹר וְהָאַיִל, וְהוּא הִקְטִיר אֶת גּוּשֵׁי הַחֵלֶב עַל הַמִּזְבֵּחַ.
כא וְאֵ֣ת הֶחָז֗וֹת וְאֵת֙ שׁ֣וֹק הַיָּמִ֔ין הֵנִ֧יף אַהֲרֹ֛ן תְּנוּפָ֖ה לִפְנֵ֣י יְהֹוָ֑ה כַּאֲשֶׁ֖ר צִוָּ֥ה מֹשֶֽׁה׃ וְאֶת חֲזֵה הַשּׁוֹר וְהָאַיִל וְאֶת הַשּׁוֹק הַיְּמִינִי אַהֲרֹן הֵנִיף לִפְנֵי ה', כְּמוֹ שֶׁמֹּשֶׁה צִוָּה.
כב וַיִּשָּׂ֨א אַהֲרֹ֧ן אֶת־יָדָ֛ו אֶל־הָעָ֖ם וַֽיְבָרְכֵ֑ם וַיֵּ֗רֶד מֵעֲשֹׂ֧ת הַֽחַטָּ֛את וְהָעֹלָ֖ה וְהַשְּׁלָמִֽים׃ אַהֲרֹן נָשָׂא אֶת יָדָיו אֶל הָעָם וּבֵרֵךְ אוֹתָם, וְיָרַד מֵהַמִּזְבֵּחַ אַחֲרֵי שֶׁעָשָׂה אֶת הַחַטָּאת וְאֶת הָעוֹלָה וְאֶת הַשְּׁלָמִים.
כג וַיָּבֹ֨א מֹשֶׁ֤ה וְאַהֲרֹן֙ אֶל־אֹ֣הֶל מוֹעֵ֔ד וַיֵּ֣צְא֔וּ וַֽיְבָרְכ֖וּ אֶת־הָעָ֑ם וַיֵּרָ֥א כְבוֹד־יְהֹוָ֖ה אֶל־כׇּל־הָעָֽם׃ מֹשֶׁה וְאַהֲרֹן בָּאוּ אֶל אוֹהֶל הַהִתְכַּנְּסוּת וְיָצְאוּ וּבֵרְכוּ אֶת הָעָם, וּכְבוֹד ה' הִתְגַּלָּה אֶל כָּל הָעָם –

שלישי[עריכה]

כד וַתֵּ֤צֵא אֵשׁ֙ מִלִּפְנֵ֣י יְהֹוָ֔ה וַתֹּ֙אכַל֙ עַל־הַמִּזְבֵּ֔חַ אֶת־הָעֹלָ֖ה וְאֶת־הַחֲלָבִ֑ים וַיַּ֤רְא כׇּל־הָעָם֙ וַיָּרֹ֔נּוּ וַֽיִּפְּל֖וּ עַל־פְּנֵיהֶֽם׃ – יָצְאָה אֵשׁ מִלִּפְנֵי ה' וְשָׂרְפָה אֶת הָעוֹלָה וְאֶת הַחֲלָבִים עַל הַמִּזְבֵּחַ, כָּל הָעָם רָאָה וְהֵם שִׁבְּחוּ אֶת ה' וְהִשְׁתַּחֲווּ עַל פְּנֵיהֶם.
פָּרָשַׁת נָדָב וַאֲבִיהוּא
י א וַיִּקְח֣וּ בְנֵֽי־אַ֠הֲרֹ֠ן נָדָ֨ב וַאֲבִיה֜וּא אִ֣ישׁ מַחְתָּת֗וֹ וַיִּתְּנ֤וּ בָהֵן֙ אֵ֔שׁ וַיָּשִׂ֥ימוּ עָלֶ֖יהָ קְטֹ֑רֶת וַיַּקְרִ֜יבוּ לִפְנֵ֤י יְהֹוָה֙ אֵ֣שׁ זָרָ֔ה אֲשֶׁ֧ר לֹ֦א צִוָּ֖ה אֹתָֽם׃ בְּנֵי אַהֲרֹן נָדָב וַאֲבִיהוּא לָקְחוּ כָּל אֶחָד אֶת מַחְתָּתוֹ וְשָׂמוּ בָּהֵן גֶּחָלִים וְשָׂמוּ עֲלֵיהֶם קְטוֹרֶת וְהִקְרִיבוּ לִפְנֵי ה' אֵשׁ "זָרָה" – שֶׁהוּא לֹא צִוָּה אוֹתָם.
ב וַתֵּ֥צֵא אֵ֛שׁ מִלִּפְנֵ֥י יְהֹוָ֖ה וַתֹּ֣אכַל אוֹתָ֑ם וַיָּמֻ֖תוּ לִפְנֵ֥י יְהֹוָֽה׃ יָצְאָה אֵשׁ מִלִּפְנֵי ה' וְשָׂרְפָה אוֹתָם, וְהֵם מֵתוּ לִפְנֵי ה'.
ג וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶֽל־אַהֲרֹ֗ן הוּא֩ אֲשֶׁר־דִּבֶּ֨ר יְהֹוָ֤ה ׀ לֵאמֹר֙ בִּקְרֹבַ֣י אֶקָּדֵ֔שׁ וְעַל־פְּנֵ֥י כׇל־הָעָ֖ם אֶכָּבֵ֑ד וַיִּדֹּ֖ם אַהֲרֹֽן׃ מֹשֶׁה אָמַר לְאַהֲרֹן זֶה מַה שֶּׁה' דִּבֵּר: בַּקְּרוֹבִים אֵלַי בְּיוֹתֵר אֲנִי מִתְקַדֵּשׁ, וְכָךְ אֲנִי מִתְכַּבֵּד בִּפְנֵי כָּל הָעָם! אַהֲרֹן שָׁתַק.
ד וַיִּקְרָ֣א מֹשֶׁ֗ה אֶל־מִֽישָׁאֵל֙ וְאֶ֣ל אֶלְצָפָ֔ן בְּנֵ֥י עֻזִּיאֵ֖ל דֹּ֣ד אַהֲרֹ֑ן וַיֹּ֣אמֶר אֲלֵהֶ֗ם קִ֞֠רְב֞֠וּ שְׂא֤וּ אֶת־אֲחֵיכֶם֙ מֵאֵ֣ת פְּנֵי־הַקֹּ֔דֶשׁ אֶל־מִח֖וּץ לַֽמַּחֲנֶֽה׃ מֹשֶׁה קָרָא לְמִישָׁאֵל וּלְאֶלְצָפָן בְּנֵי עֻזִּיאֵל דּוֹדוֹ שֶׁל אַהֲרֹן, וְאָמַר לָהֶם: גְּשׁוּ וּקְחוּ אֶת אֲחֵיכֶם מִלִּפְנֵי הַקּוֹדֶשׁ אֶל מִחוּץ לַמַּחֲנֶה.
ה וַֽיִּקְרְב֗וּ וַיִּשָּׂאֻם֙ בְּכֻתֳּנֹתָ֔ם אֶל־מִח֖וּץ לַֽמַּחֲנֶ֑ה כַּאֲשֶׁ֖ר דִּבֶּ֥ר מֹשֶֽׁה׃ הֵם נִגְּשׁוּ וְהֵרִימוּ אוֹתָם בַּכֻּתּוֹנוֹת שֶׁלָּהֶם אֶל מִחוּץ לַמַּחֲנֶה, כְּמוֹ שֶׁמֹּשֶׁה דִּבֵּר.
ו וַיֹּ֣אמֶר מֹשֶׁ֣ה אֶֽל־אַהֲרֹ֡ן וּלְאֶלְעָזָר֩ וּלְאִֽיתָמָ֨ר ׀ בָּנָ֜יו רָֽאשֵׁיכֶ֥ם אַל־תִּפְרָ֣עוּ ׀ וּבִגְדֵיכֶ֤ם לֹֽא־תִפְרֹ֙מוּ֙ וְלֹ֣א תָמֻ֔תוּ וְעַ֥ל כׇּל־הָעֵדָ֖ה יִקְצֹ֑ף וַאֲחֵיכֶם֙ כׇּל־בֵּ֣ית יִשְׂרָאֵ֔ל יִבְכּוּ֙ אֶת־הַשְּׂרֵפָ֔ה אֲשֶׁ֖ר שָׂרַ֥ף יְהֹוָֽה׃ מֹשֶׁה אָמַר לְאַהֲרֹן וּלְאֶלְעָזָר וְאִיתָמָר בָּנָיו: אַל תְּגַדְּלוּ אֶת שֵׂעָר רֹאשְׁכֶם וְאַל תִּפְרְמוּ אֶת בִּגְדֵיכֶם לַאֲבֵלוּת, כִּי אָז תָּמוּתוּ, וְה' יִכְעַס עַל כָּל הַצִּבּוּר; רַק אֲחֵיכֶם כָּל בֵּית יִשְׂרָאֵל יִבְכּוּ עַל הַשְּׂרֵפָה שֶׁה' שָׂרַף.
ז וּמִפֶּ֩תַח֩ אֹ֨הֶל מוֹעֵ֜ד לֹ֤א תֵֽצְאוּ֙ פֶּן־תָּמֻ֔תוּ כִּי־שֶׁ֛מֶן מִשְׁחַ֥ת יְהֹוָ֖ה עֲלֵיכֶ֑ם וַֽיַּעֲשׂ֖וּ כִּדְבַ֥ר מֹשֶֽׁה׃ וְאַל תֵּצְאוּ מִפֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת כְּדֵי שֶׁלֹּא תָּמוּתוּ כִּי הַשֶּׁמֶן שֶׁל ה' נִמְרַח עֲלֵיכֶם, הֵם עָשׂוּ כִּדְבַר מֹשֶׁה.
פ פָּרָשַׁת אִסּוּר הַיַּיִן
ח וַיְדַבֵּ֣ר יְהֹוָ֔ה אֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ ה' דִּבֵּר אֶל אַהֲרֹן וְאָמַר:
ט יַ֣יִן וְשֵׁכָ֞ר אַל־תֵּ֣שְׁתְּ ׀ אַתָּ֣ה ׀ וּבָנֶ֣יךָ אִתָּ֗ךְ בְּבֹאֲכֶ֛ם אֶל־אֹ֥הֶל מוֹעֵ֖ד וְלֹ֣א תָמֻ֑תוּ חֻקַּ֥ת עוֹלָ֖ם לְדֹרֹתֵיכֶֽם׃ אַל תִּשְׁתֶּה יַיִן וּמַשְׁקֶה מְשַׁכֵּר אַתָּה וּבָנֶיךָ שֶׁאִתְּךָ כְּשֶׁאַתֶּם נִכְנָסִים לְאוֹהֶל הַהִתְכַּנְּסוּת, וְכָךְ לֹא תָּמוּתוּ, חוֹק לְעוֹלָם לְדוֹרוֹת.
י וּֽלְהַבְדִּ֔יל בֵּ֥ין הַקֹּ֖דֶשׁ וּבֵ֣ין הַחֹ֑ל וּבֵ֥ין הַטָּמֵ֖א וּבֵ֥ין הַטָּהֽוֹר׃ וּכְדֵי שֶׁתּוּכְלוּ לְהַבְדִּיל בֵּין הַקּוֹדֶשׁ וּבֵין הַחוֹל וּבֵין הַטָּמֵא וּבֵין הַטָּהוֹר.
יא וּלְהוֹרֹ֖ת אֶת־בְּנֵ֣י יִשְׂרָאֵ֑ל אֵ֚ת כׇּל־הַ֣חֻקִּ֔ים אֲשֶׁ֨ר דִּבֶּ֧ר יְהֹוָ֛ה אֲלֵיהֶ֖ם בְּיַד־מֹשֶֽׁה׃ וּלְהוֹרוֹת לִבְנֵי יִשְׂרָאֵל אֶת כָּל הַחֻקִּים שֶׁה' דִּבֵּר אֲלֵיהֶם בְּאֶמְצָעוּת מֹשֶׁה.

רביעי[עריכה]

פ פָּרָשַׁת אֲכִילַת קָרְבְּנוֹת הַיּוֹם
יב וַיְדַבֵּ֨ר מֹשֶׁ֜ה אֶֽל־אַהֲרֹ֗ן וְאֶ֣ל אֶ֠לְעָזָ֠ר וְאֶל־אִ֨יתָמָ֥ר ׀ בָּנָיו֮ הַנּֽוֹתָרִים֒ קְח֣וּ אֶת־הַמִּנְחָ֗ה הַנּוֹתֶ֙רֶת֙ מֵאִשֵּׁ֣י יְהֹוָ֔ה וְאִכְל֥וּהָ מַצּ֖וֹת אֵ֣צֶל הַמִּזְבֵּ֑חַ כִּ֛י קֹ֥דֶשׁ קׇֽדָשִׁ֖ים הִֽוא׃ מֹשֶׁה דִּבֵּר אֶל אַהֲרֹן וְאֶל אֶלְעָזָר וְאֶל אִיתָמָר בָּנָיו שֶׁנִּשְׁאֲרוּ בַּחַיִּים: קְחוּ אֶת הַמִּנְחָה – הַחֵלֶק שֶׁלֹּא הֻקְטַר בָּאֵשׁ לַה', וְאִכְלוּ אוֹתָהּ בְּלִי שֶׁתַּחְמִיץ לְיַד הַמִּזְבֵּחַ, כִּי הִיא קוֹדֶשׁ קָדָשִׁים.
יג וַאֲכַלְתֶּ֤ם אֹתָהּ֙ בְּמָק֣וֹם קָד֔וֹשׁ כִּ֣י חׇקְךָ֤ וְחׇק־בָּנֶ֙יךָ֙ הִ֔וא מֵאִשֵּׁ֖י יְהֹוָ֑ה כִּי־כֵ֖ן צֻוֵּֽיתִי׃ תֹּאכְלוּ אוֹתָהּ בְּמָקוֹם קָדוֹשׁ כִּי הִיא קִצְבָּתְךָ וְקִצְבַּת בָּנֶיךָ מִקָּרְבְּנוֹת ה', כִּי כָּךְ נִצְטַוֵּיתִי.
יד וְאֵת֩ חֲזֵ֨ה הַתְּנוּפָ֜ה וְאֵ֣ת ׀ שׁ֣וֹק הַתְּרוּמָ֗ה תֹּֽאכְלוּ֙ בְּמָק֣וֹם טָה֔וֹר אַתָּ֕ה וּבָנֶ֥יךָ וּבְנֹתֶ֖יךָ אִתָּ֑ךְ כִּֽי־חׇקְךָ֤ וְחׇק־בָּנֶ֙יךָ֙ נִתְּנ֔וּ מִזִּבְחֵ֥י שַׁלְמֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל׃ וְאֶת הֶחָזֶה הַמּוּנָף וְאֶת הַשּׁוֹק שֶׁהוּרַם תֹּאכְלוּ בְּמָקוֹם טָהוֹר אַתָּה וּבָנֶיךָ וּבְנוֹתֶיךָ אִתְּךָ, כִּי הֵם נִתְּנוּ לְקִצְבָּה לְךָ וּלְבָנֶיךָ מִקָּרְבְּנוֹת הַשְּׁלָמִים שֶׁל בְּנֵי יִשְׂרָאֵל.
טו שׁ֣וֹק הַתְּרוּמָ֞ה וַחֲזֵ֣ה הַתְּנוּפָ֗ה עַ֣ל אִשֵּׁ֤י הַחֲלָבִים֙ יָבִ֔יאוּ לְהָנִ֥יף תְּנוּפָ֖ה לִפְנֵ֣י יְהֹוָ֑ה וְהָיָ֨ה לְךָ֜ וּלְבָנֶ֤יךָ אִתְּךָ֙ לְחׇק־עוֹלָ֔ם כַּאֲשֶׁ֖ר צִוָּ֥ה יְהֹוָֽה׃ אֶת הַשּׁוֹק שֶׁהוּרַם וְהֶחָזֶה הַמּוּנָף יָבִיאוּ כְּשֶׁהוּא מֻנָּח עַל גּוּשֵׁי הַחֵלֶב הַקְּרֵבִים, כְּדֵי לְהָנִיף אוֹתָם לִפְנֵי ה', וְזֶה יִהְיֶה קִצְבָּה לְךָ וּלְבָנֶיךָ אִתְּךָ לְעוֹלָם כְּמוֹ שֶׁה' צִוָּה.

חמישי[עריכה]

טז וְאֵ֣ת ׀ שְׂעִ֣יר הַֽחַטָּ֗את דָּרֹ֥שׁ דָּרַ֛שׁ מֹשֶׁ֖ה וְהִנֵּ֣ה שֹׂרָ֑ף וַ֠יִּקְצֹ֠ף עַל־אֶלְעָזָ֤ר וְעַל־אִֽיתָמָר֙ בְּנֵ֣י אַהֲרֹ֔ן הַנּוֹתָרִ֖ם לֵאמֹֽר׃ מֹשֶׁה חִפֵּשׂ אֶת תַּיִשׁ הַחַטָּאת וְגִלָּה שֶׁהוּא נִשְׂרַף, הוּא גָּעַר בְּאֶלְעָזָר וּבְאִיתָמָר בְּנֵי אַהֲרֹן שֶׁנִּשְׁאֲרוּ בַּחַיִּים, וְאָמַר לָהֶם:
יז מַדּ֗וּעַ לֹֽא־אֲכַלְתֶּ֤ם אֶת־הַחַטָּאת֙ בִּמְק֣וֹם הַקֹּ֔דֶשׁ כִּ֛י קֹ֥דֶשׁ קׇֽדָשִׁ֖ים הִ֑וא וְאֹתָ֣הּ ׀ נָתַ֣ן לָכֶ֗ם לָשֵׂאת֙ אֶת־עֲוֺ֣ן הָעֵדָ֔ה לְכַפֵּ֥ר עֲלֵיהֶ֖ם לִפְנֵ֥י יְהֹוָֽה׃ לָמָּה לֹא אֲכַלְתֶּם אֶת הַחַטָּאת בִּמְקוֹם הַקּוֹדֶשׁ? הֲרֵי הִיא קוֹדֶשׁ קָדָשִׁים! וַה' נָתַן לָכֶם לֶאֱכוֹל אוֹתָהּ כְּדֵי שֶׁיִּסְלַח עַל עֲווֹן הַצִּבּוּר, כְּדֵי לְכַפֵּר עֲלֵיהֶם לִפְנֵי ה'!
יח הֵ֚ן לֹא־הוּבָ֣א אֶת־דָּמָ֔הּ אֶל־הַקֹּ֖דֶשׁ פְּנִ֑ימָה אָכ֨וֹל תֹּאכְל֥וּ אֹתָ֛הּ בַּקֹּ֖דֶשׁ כַּאֲשֶׁ֥ר צִוֵּֽיתִי׃ הֲרֵי לֹא הֵבִיאוּ אֶת דָּמָהּ לַקּוֹדֶשׁ פְּנִימָה, אִכְלוּ אוֹתָהּ בַּקּוֹדֶשׁ כְּמוֹ שֶׁצִּוִּיתִי!
יט וַיְדַבֵּ֨ר אַהֲרֹ֜ן אֶל־מֹשֶׁ֗ה הֵ֣ן הַ֠יּ֠וֹם הִקְרִ֨יבוּ אֶת־חַטָּאתָ֤ם וְאֶת־עֹֽלָתָם֙ לִפְנֵ֣י יְהֹוָ֔ה וַתִּקְרֶ֥אנָה אֹתִ֖י כָּאֵ֑לֶּה וְאָכַ֤לְתִּי חַטָּאת֙ הַיּ֔וֹם הַיִּיטַ֖ב בְּעֵינֵ֥י יְהֹוָֽה׃ אַהֲרֹן דִּבֵּר אֶל מֹשֶׁה: הֲרֵי הַיּוֹם הִקְרִיבוּ אֶת חַטָּאתָם וְאֶת עוֹלָתָם לִפְנֵי ה' וְקָרוּ לִי אֲסוֹנוֹת אֲבֵלוּת אֵלֶּה, אִם אוֹכַל חַטָּאת הַיּוֹם – הַאִם זֶה יִיטַב בְּעֵינֵי ה'?!
כ וַיִּשְׁמַ֣ע מֹשֶׁ֔ה וַיִּיטַ֖ב בְּעֵינָֽיו׃ מֹשֶׁה שָׁמַע וְהִסְכִּים לַתְּשׁוּבָה.

ששי[עריכה]

פ פָּרָשַׁת הַחַיָּה הַטְּהוֹרָה
יא א וַיְדַבֵּ֧ר יְהֹוָ֛ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹ֥ר אֲלֵהֶֽם׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאֶל אַהֲרֹן, וְאָמַר לָהֶם:
ב דַּבְּר֛וּ אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר זֹ֤את הַֽחַיָּה֙ אֲשֶׁ֣ר תֹּאכְל֔וּ מִכׇּל־הַבְּהֵמָ֖ה אֲשֶׁ֥ר עַל־הָאָֽרֶץ׃ דַּבְּרוּ אֶל בְּנֵי יִשְׂרָאֵל וְאִמְרוּ: אֵלֶּה הַחַיּוֹת שֶׁתֹּאכְלוּ מִכָּל הַבְּהֵמוֹת שֶׁעַל הָאָרֶץ.
ג כֹּ֣ל ׀ מַפְרֶ֣סֶת פַּרְסָ֗ה וְשֹׁסַ֤עַת שֶׁ֙סַע֙ פְּרָסֹ֔ת מַעֲלַ֥ת גֵּרָ֖ה בַּבְּהֵמָ֑ה אֹתָ֖הּ תֹּאכֵֽלוּ׃ כָּל בְּהֵמָה שֶׁיֵּשׁ לָהּ פַּרְסוֹת וְלַפַּרְסוֹת שֶׁלָּהּ יֵשׁ שֶׁסַע וְהִיא מַעֲלָה גֵּרָה – אוֹתָהּ תֹּאכְלוּ.
ד אַ֤ךְ אֶת־זֶה֙ לֹ֣א תֹֽאכְל֔וּ מִֽמַּעֲלֵי֙ הַגֵּרָ֔ה וּמִמַּפְרִסֵ֖י הַפַּרְסָ֑ה אֶֽת־הַ֠גָּמָ֠ל כִּֽי־מַעֲלֵ֨ה גֵרָ֜ה ה֗וּא וּפַרְסָה֙ אֵינֶ֣נּוּ מַפְרִ֔יס טָמֵ֥א ה֖וּא לָכֶֽם׃ אֲבָל אֶת זֶה אַל תֹּאכְלוּ מִמַּעֲלֵי הַגֵּרָה וּמִבַּעֲלֵי פַּרְסָה סְדוּקָה: אֶת הַגָּמָל – כִּי הוּא מַעֲלֵה גֵּרָה אֲבָל אֵין לוֹ פַּרְסָה, הוּא טָמֵא עֲבוּרְכֶם.
ה וְאֶת־הַשָּׁפָ֗ן כִּֽי־מַעֲלֵ֤ה גֵרָה֙ ה֔וּא וּפַרְסָ֖ה לֹ֣א יַפְרִ֑יס טָמֵ֥א ה֖וּא לָכֶֽם׃ וְאֶת הַשָּׁפָן – כִּי הוּא מַעֲלֵה גֵּרָה אֲבָל אֵין לוֹ פַּרְסָה, הוּא טָמֵא עֲבוּרְכֶם.
ו וְאֶת־הָאַרְנֶ֗בֶת כִּֽי־מַעֲלַ֤ת גֵּרָה֙ הִ֔וא וּפַרְסָ֖ה לֹ֣א הִפְרִ֑יסָה טְמֵאָ֥ה הִ֖וא לָכֶֽם׃ וְאֶת הָאַרְנֶבֶת – כִּי הִיא מַעֲלָה גֵּרָה אֲבָל אֵין לָהּ פַּרְסָה, הִיא טְמֵאָה עֲבוּרְכֶם.
ז וְאֶת־הַ֠חֲזִ֠יר כִּֽי־מַפְרִ֨יס פַּרְסָ֜ה ה֗וּא וְשֹׁסַ֥ע שֶׁ֙סַע֙ פַּרְסָ֔ה וְה֖וּא גֵּרָ֣ה לֹֽא־יִגָּ֑ר טָמֵ֥א ה֖וּא לָכֶֽם׃ וְאֶת הַחֲזִיר – כִּי יֵשׁ לוֹ פַּרְסָה וְלַפַּרְסָה שֶׁלּוֹ יֵשׁ שֶׁסַע, אֲבָל הוּא לֹא מַעֲלֶה גֵּרָה, הוּא טָמֵא עֲבוּרְכֶם.
ח מִבְּשָׂרָם֙ לֹ֣א תֹאכֵ֔לוּ וּבְנִבְלָתָ֖ם לֹ֣א תִגָּ֑עוּ טְמֵאִ֥ים הֵ֖ם לָכֶֽם׃ אַל תֹּאכְלוּ מִבְּשָׂרָם, וְאַל תִּגְּעוּ בְּנִבְלָתָם כְּשֶׁאָסוּר לְהִטָּמֵא, הֵם טְמֵאִים עֲבוּרְכֶם.
ט אֶת־זֶה֙ תֹּֽאכְל֔וּ מִכֹּ֖ל אֲשֶׁ֣ר בַּמָּ֑יִם כֹּ֣ל אֲשֶׁר־לוֹ֩ סְנַפִּ֨יר וְקַשְׂקֶ֜שֶׂת בַּמַּ֗יִם בַּיַּמִּ֛ים וּבַנְּחָלִ֖ים אֹתָ֥ם תֹּאכֵֽלוּ׃ אֶת זֶה תֹּאכְלוּ מִכָּל מַה שֶּׁבַּמַּיִם: כֹּל דָּג שֶׁיֵּשׁ לוֹ סְנַפִּיר וְקַשְׂקֶשֶׂת, בַּמַּיִם: בַּיַּמִּים וּבַנְּחָלִים, אוֹתָם תֹּאכְלוּ.
י וְכֹל֩ אֲשֶׁ֨ר אֵֽין־ל֜וֹ סְנַפִּ֣יר וְקַשְׂקֶ֗שֶׂת בַּיַּמִּים֙ וּבַנְּחָלִ֔ים מִכֹּל֙ שֶׁ֣רֶץ הַמַּ֔יִם וּמִכֹּ֛ל נֶ֥פֶשׁ הַחַיָּ֖ה אֲשֶׁ֣ר בַּמָּ֑יִם שֶׁ֥קֶץ הֵ֖ם לָכֶֽם׃ וְכָל דָּג שֶׁאֵין לוֹ סְנַפִּיר וְקַשְׂקֶשֶׂת בַּיַּמִּים וּבַנְּחָלִים, מִכָּל הַשְּׁרָצִים שֶׁבַּמַּיִם וּמִכָּל בַּעֲלֵי הַחַיִּים שֶׁבַּמַּיִם – הֵם מְאוּסִים.
יא וְשֶׁ֖קֶץ יִהְי֣וּ לָכֶ֑ם מִבְּשָׂרָם֙ לֹ֣א תֹאכֵ֔לוּ וְאֶת־נִבְלָתָ֖ם תְּשַׁקֵּֽצוּ׃ וְהֵם יִהְיוּ מְאוּסִים עֲבוּרְכֶם – אַל תֹּאכְלוּ מִבְּשָׂרָם וְתַרְחִיקוּ אֶת נִבְלָתָם.
יב כֹּ֣ל אֲשֶׁ֥ר אֵֽין־ל֛וֹ סְנַפִּ֥יר וְקַשְׂקֶ֖שֶׂת בַּמָּ֑יִם שֶׁ֥קֶץ ה֖וּא לָכֶֽם׃ כֹּל בַּעַל חַי בַּמַּיִם שֶׁאֵין לוֹ סְנַפִּיר וְקַשְׂקֶשֶׂת, הוּא מָאוּס עֲבוּרְכֶם.
יג וְאֶת־אֵ֙לֶּה֙ תְּשַׁקְּצ֣וּ מִן־הָע֔וֹף לֹ֥א יֵאָכְל֖וּ שֶׁ֣קֶץ הֵ֑ם אֶת־הַנֶּ֙שֶׁר֙ וְאֶת־הַפֶּ֔רֶס וְאֵ֖ת הָעׇזְנִיָּֽה׃ עוֹפוֹת אֵלֶּה תַּרְחִיקוּ מֵעַצְמְכֶם – הֵם לֹא יֵאָכְלוּ כִּי הֵם מְאוּסִים: אֶת הַנֶּשֶׁר וְאֶת הַפֶּרֶס וְאֶת הָעָזְנִיָּה.
יד וְאֶ֨ת־הַדָּאָ֔ה וְאֶת־הָאַיָּ֖ה לְמִינָֽהּ׃ וְאֶת הַדָּאָה וְאֶת כָּל סוּגֵי הָאַיָּה.
טו אֵ֥ת כׇּל־עֹרֵ֖ב לְמִינֽוֹ׃ אֶת כָּל הָעוֹרְבִים מִכָּל הַסּוּגִים.
טז וְאֵת֙ בַּ֣ת הַֽיַּעֲנָ֔ה וְאֶת־הַתַּחְמָ֖ס וְאֶת־הַשָּׁ֑חַף וְאֶת־הַנֵּ֖ץ לְמִינֵֽהוּ׃ וְאֶת בַּת הַיַּעֲנָה וְאֶת הַתַּחְמָס וְאֶת הַשָּׁחַף וְאֶת כָּל סוּגֵי הַנֵּץ.
יז וְאֶת־הַכּ֥וֹס וְאֶת־הַשָּׁלָ֖ךְ וְאֶת־הַיַּנְשֽׁוּף׃ וְאֶת הַכּוֹס וְאֶת הַשַּׂקְנַאי וְאֶת הַיַּנְשׁוּף.
יח וְאֶת־הַתִּנְשֶׁ֥מֶת וְאֶת־הַקָּאָ֖ת וְאֶת־הָרָחָֽם׃ וְאֶת הַתִּנְשֶׁמֶת וְאֶת הַקָּאָת וְאֶת הָרָחָם.
יט וְאֵת֙ הַחֲסִידָ֔ה הָאֲנָפָ֖ה לְמִינָ֑הּ וְאֶת־הַדּוּכִיפַ֖ת וְאֶת־הָעֲטַלֵּֽף׃ וְאֶת הַחֲסִידָה, כָּל סוּגֵי הָאֲנָפָה, וְאֶת הַדּוּכִיפַת וְאֶת הָעֲטַלֵּף.
כ כֹּ֚ל שֶׁ֣רֶץ הָע֔וֹף הַהֹלֵ֖ךְ עַל־אַרְבַּ֑ע שֶׁ֥קֶץ ה֖וּא לָכֶֽם׃ כֹּל שֶׁרֶץ עוֹף שֶׁהוֹלֵךְ עַל אַרְבַּע, הוּא מָאוּס עֲבוּרְכֶם.
כא אַ֤ךְ אֶת־זֶה֙ תֹּֽאכְל֔וּ מִכֹּל֙ שֶׁ֣רֶץ הָע֔וֹף הַהֹלֵ֖ךְ עַל־אַרְבַּ֑ע אֲשֶׁר־ל֤וֹ לא כְרָעַ֙יִם֙ מִמַּ֣עַל לְרַגְלָ֔יו לְנַתֵּ֥ר בָּהֵ֖ן עַל־הָאָֽרֶץ׃ אֲבָל אֶת זֶה תֹּאכְלוּ מִכָּל שִׁרְצֵי הָעוֹף שֶׁהוֹלְכִים עַל אַרְבַּע – אֶת שֶׁיֵּשׁ לוֹ קַרְסֻלַּיִם מֵעַל לְרַגְלָיו כְּדֵי לְנַתֵּר בָּהֵן עַל הָאָרֶץ.
כב אֶת־אֵ֤לֶּה מֵהֶם֙ תֹּאכֵ֔לוּ אֶת־הָֽאַרְבֶּ֣ה לְמִינ֔וֹ וְאֶת־הַסׇּלְעָ֖ם לְמִינֵ֑הוּ וְאֶת־הַחַרְגֹּ֣ל לְמִינֵ֔הוּ וְאֶת־הֶחָגָ֖ב לְמִינֵֽהוּ׃ אֶת אֵלֶּה מֵהֶם תֹּאכְלוּ: אֶת כָּל סוּגֵי הָאַרְבֶּה, וְאֶת כָּל סוּגֵי הַסָּלְעָם, וְאֶת כָּל סוּגֵי הַחַרְגּוֹל וְאֶת כָּל סוּגֵי הֶחָגָב.
כג וְכֹל֙ שֶׁ֣רֶץ הָע֔וֹף אֲשֶׁר־ל֖וֹ אַרְבַּ֣ע רַגְלָ֑יִם שֶׁ֥קֶץ ה֖וּא לָכֶֽם׃ וְכָל שֶׁרֶץ עוֹף שֶׁיֵּשׁ לוֹ אַרְבַּע רַגְלַיִם הוּא מָאוּס עֲבוּרְכֶם.
כד וּלְאֵ֖לֶּה תִּטַּמָּ֑אוּ כׇּל־הַנֹּגֵ֥עַ בְּנִבְלָתָ֖ם יִטְמָ֥א עַד־הָעָֽרֶב׃ וּלְאֵלֶּה תִּטַּמְּאוּ – כָּל מִי שֶׁנּוֹגֵעַ בְּנִבְלָתָם יִטְמָא עַד הָעֶרֶב.
כה וְכׇל־הַנֹּשֵׂ֖א מִנִּבְלָתָ֑ם יְכַבֵּ֥ס בְּגָדָ֖יו וְטָמֵ֥א עַד־הָעָֽרֶב׃ וְכָל מִי שֶׁמֵּרִים מִנִּבְלָתָם יַטְבִּיל אֶת בְּגָדָיו וְיִטְמָא עַד הָעֶרֶב.
כו לְֽכׇל־הַבְּהֵמָ֡ה אֲשֶׁ֣ר הִוא֩ מַפְרֶ֨סֶת פַּרְסָ֜ה וְשֶׁ֣סַע ׀ אֵינֶ֣נָּה שֹׁסַ֗עַת וְגֵרָה֙ אֵינֶ֣נָּה מַעֲלָ֔ה טְמֵאִ֥ים הֵ֖ם לָכֶ֑ם כׇּל־הַנֹּגֵ֥עַ בָּהֶ֖ם יִטְמָֽא׃ כָּל מִינֵי בְּהֵמָה שֶׁיֵּשׁ לָהּ פְּרָסוֹת אֲבָל אֵין לַפְּרָסוֹת סֶדֶק אוֹ שֶׁאֵינָהּ מַעֲלָה גֵּרָה – הֵם טְמֵאִים עֲבוּרְכֶם, כָּל מִי שֶׁנּוֹגֵעַ בָּהֶם יִטְמָא.
כז וְכֹ֣ל ׀ הוֹלֵ֣ךְ עַל־כַּפָּ֗יו בְּכׇל־הַֽחַיָּה֙ הַהֹלֶ֣כֶת עַל־אַרְבַּ֔ע טְמֵאִ֥ים הֵ֖ם לָכֶ֑ם כׇּל־הַנֹּגֵ֥עַ בְּנִבְלָתָ֖ם יִטְמָ֥א עַד־הָעָֽרֶב׃ וּבַחַיּוֹת שֶׁהוֹלְכוֹת עַל אַרְבַּע – כָּל הַהוֹלְכִים עַל כַּפּוֹת רַגְלֵיהֶם בְּלִי פַּרְסָה הֵם טְמֵאִים עֲבוּרְכֶם, כָּל מִי שֶׁנּוֹגֵעַ בְּנִבְלָתָם יִטְמָא עַד הָעֶרֶב.
כח וְהַנֹּשֵׂא֙ אֶת־נִבְלָתָ֔ם יְכַבֵּ֥ס בְּגָדָ֖יו וְטָמֵ֣א עַד־הָעָ֑רֶב טְמֵאִ֥ים הֵ֖מָּה לָכֶֽם׃ וּמִי שֶׁמֵּרִים אֶת נִבְלָתָם יַטְבִּיל אֶת בְּגָדָיו וְיִטְמָא עַד הָעֶרֶב – הֵם טְמֵאִים עֲבוּרְכֶם.
ס פָּרָשַׁת טֻמְאַת הַשֶּׁרֶץ
כט וְזֶ֤ה לָכֶם֙ הַטָּמֵ֔א בַּשֶּׁ֖רֶץ הַשֹּׁרֵ֣ץ עַל־הָאָ֑רֶץ הַחֹ֥לֶד וְהָעַכְבָּ֖ר וְהַצָּ֥ב לְמִינֵֽהוּ׃ וְזֶה לָכֶם הַטָּמֵא מֵהַשְּׁרָצִים שֶׁשּׁוֹרְצִים עַל הָאָרֶץ: הַחוֹלֶד וְהָעַכְבָּר, וְכָל סוּגֵי הַחַרְדּוֹן.
ל וְהָאֲנָקָ֥ה וְהַכֹּ֖חַ וְהַלְּטָאָ֑ה וְהַחֹ֖מֶט וְהַתִּנְשָֽׁמֶת׃ וְהָאֲנָקָה וְהַכּוֹחַ וְהַלְּטָאָה וְהַחוֹמֶט וְהַתִּנְשֶׁמֶת.
לא אֵ֛לֶּה הַטְּמֵאִ֥ים לָכֶ֖ם בְּכׇל־הַשָּׁ֑רֶץ כׇּל־הַנֹּגֵ֧עַ בָּהֶ֛ם בְּמֹתָ֖ם יִטְמָ֥א עַד־הָעָֽרֶב׃ אֵלֶּה הַשְּׁרָצִים שֶׁטְּמֵאִים לָכֶם מִכָּל הַשְּׁרָצִים – כָּל מִי שֶׁנּוֹגֵעַ בָּהֶם בְּמוֹתָם יִטְמָא עַד הָעֶרֶב.
לב וְכֹ֣ל אֲשֶׁר־יִפֹּל־עָלָיו֩ מֵהֶ֨ם ׀ בְּמֹתָ֜ם יִטְמָ֗א מִכׇּל־כְּלִי־עֵץ֙ א֣וֹ בֶ֤גֶד אוֹ־עוֹר֙ א֣וֹ שָׂ֔ק כׇּל־כְּלִ֕י אֲשֶׁר־יֵעָשֶׂ֥ה מְלָאכָ֖ה בָּהֶ֑ם בַּמַּ֧יִם יוּבָ֛א וְטָמֵ֥א עַד־הָעֶ֖רֶב וְטָהֵֽר׃ וְכָל דָּבָר שֶׁיִּפּוֹל עָלָיו מֵהֶם בְּמוֹתָם – יִטְמָא: מִכָּל כְּלִי שֶׁעָשׂוּי מֵעֵץ אוֹ בֶּגֶד אוֹ עוֹר אוֹ שָׂק שֶׁעוֹשִׂים בּוֹ מְלָאכָה – יֻטְבַּל בְּמַיִם וְיִהְיֶה טָמֵא עַד הָעֶרֶב וְיִטְהַר.

שביעי[עריכה]

לג וְכׇ֨ל־כְּלִי־חֶ֔רֶשׂ אֲשֶׁר־יִפֹּ֥ל מֵהֶ֖ם אֶל־תּוֹכ֑וֹ כֹּ֣ל אֲשֶׁ֧ר בְּתוֹכ֛וֹ יִטְמָ֖א וְאֹת֥וֹ תִשְׁבֹּֽרוּ׃ וְכָל כְּלִי חֶרֶשׂ שֶׁיִּפּוֹל מֵהֶם לְתוֹכוֹ – כָּל מַה שֶּׁבְּתוֹכוֹ יִטְמָא וְאוֹתוֹ תִּשְׁבְּרוּ.
לד מִכׇּל־הָאֹ֜כֶל אֲשֶׁ֣ר יֵאָכֵ֗ל אֲשֶׁ֨ר יָב֥וֹא עָלָ֛יו מַ֖יִם יִטְמָ֑א וְכׇל־מַשְׁקֶה֙ אֲשֶׁ֣ר יִשָּׁתֶ֔ה בְּכׇל־כְּלִ֖י יִטְמָֽא׃ מִכָּל הָאוֹכֶל שֶׁיֵּאָכֵל שֶׁיָּבוֹא עָלָיו מַיִם – יִטְמָא, וְכָל מַשְׁקֶה שֶׁיִּשָּׁתֶה בְּכָל כְּלִי – יִטְמָא.
לה וְ֠כֹ֠ל אֲשֶׁר־יִפֹּ֨ל מִנִּבְלָתָ֥ם ׀ עָלָיו֮ יִטְמָא֒ תַּנּ֧וּר וְכִירַ֛יִם יֻתָּ֖ץ טְמֵאִ֣ים הֵ֑ם וּטְמֵאִ֖ים יִהְי֥וּ לָכֶֽם׃ וְכָל דָּבָר שֶׁיִּפּוֹל מִנִּבְלָתָם עָלָיו – יִטְמָא, תַּנּוּר וְכִירַיִם טְמֵאִים יִנָּתְצוּ, וְהֵם יִהְיוּ טְמֵאִים עֲבוּרְכֶם.
לו אַ֣ךְ מַעְיָ֥ן וּב֛וֹר מִקְוֵה־מַ֖יִם יִהְיֶ֣ה טָה֑וֹר וְנֹגֵ֥עַ בְּנִבְלָתָ֖ם יִטְמָֽא׃ רַק מַעְיָן וּבוֹר וּמַיִם מְכֻנָּסִים יִהְיוּ טְהוֹרִים, וּמִי שֶׁנּוֹגֵעַ בְּנִבְלָתָם כְּשֶׁהוּא בַּמַּיִם – יִטְמָא.
לז וְכִ֤י יִפֹּל֙ מִנִּבְלָתָ֔ם עַל־כׇּל־זֶ֥רַע זֵר֖וּעַ אֲשֶׁ֣ר יִזָּרֵ֑עַ טָה֖וֹר הֽוּא׃ וְאִם יִפּוֹל מִנִּבְלָתָם עַל כָּל זֶרַע שֶׁיִּזָּרַע – הוּא טָהוֹר.
לח וְכִ֤י יֻתַּן־מַ֙יִם֙ עַל־זֶ֔רַע וְנָפַ֥ל מִנִּבְלָתָ֖ם עָלָ֑יו טָמֵ֥א ה֖וּא לָכֶֽם׃ אֲבָל אִם יִשָּׁפֵךְ מַיִם עַל זֶרַע וְאָז יִפּוֹל מִנִּבְלָתָם עָלָיו – הוּא טָמֵא עֲבוּרְכֶם.
ס
לט וְכִ֤י יָמוּת֙ מִן־הַבְּהֵמָ֔ה אֲשֶׁר־הִ֥יא לָכֶ֖ם לְאׇכְלָ֑ה הַנֹּגֵ֥עַ בְּנִבְלָתָ֖הּ יִטְמָ֥א עַד־הָעָֽרֶב׃ וְאִם תָּמוּת אַחַת מֵהַבְּהֵמוֹת שֶׁמֻּתָּרוֹת לָכֶם בַּאֲכִילָה – מִי שֶׁנּוֹגֵעַ בְּנִבְלָתָהּ יִטְמָא עַד הָעֶרֶב.
מ וְהָֽאֹכֵל֙ מִנִּבְלָתָ֔הּ יְכַבֵּ֥ס בְּגָדָ֖יו וְטָמֵ֣א עַד־הָעָ֑רֶב וְהַנֹּשֵׂא֙ אֶת־נִבְלָתָ֔הּ יְכַבֵּ֥ס בְּגָדָ֖יו וְטָמֵ֥א עַד־הָעָֽרֶב׃ וּמִי שֶׁאוֹכֵל מִנִּבְלָתָהּ יַטְבִּיל אֶת בְּגָדָיו וְיִטְמָא עַד הָעֶרֶב, וּמִי שֶׁמֵּרִים אֶת נִבְלָתָהּ יַטְבִּיל אֶת בְּגָדָיו וְיִטְמָא עַד הָעֶרֶב.
מא וְכׇל־הַשֶּׁ֖רֶץ הַשֹּׁרֵ֣ץ עַל־הָאָ֑רֶץ שֶׁ֥קֶץ ה֖וּא לֹ֥א יֵאָכֵֽל׃ וְכָל הַשְּׁרָצִים שֶׁשּׁוֹרְצִים עַל הָאָרֶץ, הֵם מְאוּסִים וְאָסוּר לֶאֱכוֹל אוֹתָם.
מב כֹּל֩ הוֹלֵ֨ךְ עַל־גָּח֜וֹן וְכֹ֣ל ׀ הוֹלֵ֣ךְ עַל־אַרְבַּ֗ע עַ֚ד כׇּל־מַרְבֵּ֣ה רַגְלַ֔יִם לְכׇל־הַשֶּׁ֖רֶץ הַשֹּׁרֵ֣ץ עַל־הָאָ֑רֶץ לֹ֥א תֹאכְל֖וּם כִּי־שֶׁ֥קֶץ הֵֽם׃ מִכָּל הַשְּׁרָצִים שֶׁשּׁוֹרְצִים עַל הָאֲדָמָה – כָּל שֶׁזּוֹחֵל עַל בִּטְנוֹ וְכָל שֶׁהוֹלֵךְ עַל אַרְבַּע רַגְלַיִם אוֹ יוֹתֵר, עַד כָּל מַרְבֵּה רַגְלַיִם – אַל תֹּאכְלוּ אוֹתָם כִּי הֵם מְאוּסִים.
מג אַל־תְּשַׁקְּצוּ֙ אֶת־נַפְשֹׁ֣תֵיכֶ֔ם בְּכׇל־הַשֶּׁ֖רֶץ הַשֹּׁרֵ֑ץ וְלֹ֤א תִֽטַּמְּאוּ֙ בָּהֶ֔ם וְנִטְמֵתֶ֖ם בָּֽם׃ אַל תְּטַנְּפוּ אֶת נַפְשׁוֹתֵיכֶם בְּכָל הַשְּׁרָצִים שֶׁשּׁוֹרְצִים, וְאַל תִּטַּמְּאוּ בָּהֶם וְתִהְיוּ טְמֵאִים מֵהֶם.
מד כִּ֣י אֲנִ֣י יְהֹוָה֮ אֱלֹֽהֵיכֶם֒ וְהִתְקַדִּשְׁתֶּם֙ וִהְיִיתֶ֣ם קְדֹשִׁ֔ים כִּ֥י קָד֖וֹשׁ אָ֑נִי וְלֹ֤א תְטַמְּאוּ֙ אֶת־נַפְשֹׁ֣תֵיכֶ֔ם בְּכׇל־הַשֶּׁ֖רֶץ הָרֹמֵ֥שׂ עַל־הָאָֽרֶץ׃ כִּי אֲנִי ה' אֱלֹהֵיכֶם, תִּתְקַדְּשׁוּ וְתִהְיוּ קְדוֹשִׁים כִּי אֲנִי קָדוֹשׁ, וְאַל תְּטַמְּאוּ אֶת נַפְשׁוֹתֵיכֶם בְּכָל הַשְּׁרָצִים שֶׁזּוֹחֲלִים עַל הָאָרֶץ.

מפטיר[עריכה]

מה כִּ֣י ׀ אֲנִ֣י יְהֹוָ֗ה הַֽמַּעֲלֶ֤ה אֶתְכֶם֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לִהְיֹ֥ת לָכֶ֖ם לֵאלֹהִ֑ים וִהְיִיתֶ֣ם קְדֹשִׁ֔ים כִּ֥י קָד֖וֹשׁ אָֽנִי׃ כִּי אֲנִי ה' שֶׁהֶעֱלֵיתִי אֶתְכֶם מֵאֶרֶץ מִצְרַיִם לִהְיוֹת לָכֶם לֵאלֹהִים, הֱיוּ קְדוֹשִׁים כִּי אֲנִי קָדוֹשׁ.
מו זֹ֣את תּוֹרַ֤ת הַבְּהֵמָה֙ וְהָע֔וֹף וְכֹל֙ נֶ֣פֶשׁ הַֽחַיָּ֔ה הָרֹמֶ֖שֶׂת בַּמָּ֑יִם וּלְכׇל־נֶ֖פֶשׁ הַשֹּׁרֶ֥צֶת עַל־הָאָֽרֶץ׃ אֵלֶּה דִּינֵי הַבְּהֵמָה וְהָעוֹף וְכָל בַּעֲלֵי הַחַיִּים שֶׁנָּעִים לְאִטָּם בַּמַּיִם, וּלְכָל בַּעֲלֵי הַחַיִּים שֶׁשּׁוֹרְצִים עַל הָאֲדָמָה.
מז לְהַבְדִּ֕יל בֵּ֥ין הַטָּמֵ֖א וּבֵ֣ין הַטָּהֹ֑ר וּבֵ֤ין הַֽחַיָּה֙ הַֽנֶּאֱכֶ֔לֶת וּבֵין֙ הַֽחַיָּ֔ה אֲשֶׁ֖ר לֹ֥א תֵאָכֵֽל׃ לְהַבְדִּיל בֵּין הַטָּמֵא לְבֵין הַטָּהוֹר, וּבֵין חַיָּה שֶׁמֻּתֶּרֶת בַּאֲכִילָה לְבֵין חַיָּה שֶׁאֲסוּרָה בַּאֲכִילָה.
