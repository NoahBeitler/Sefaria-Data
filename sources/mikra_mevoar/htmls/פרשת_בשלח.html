ראשון[עריכה]

ס פָּרָשַׁת הַהַסַּבָּה לְיַם סוּף
יג יז וַיְהִ֗י בְּשַׁלַּ֣ח פַּרְעֹה֮ אֶת־הָעָם֒ וְלֹא־נָחָ֣ם אֱלֹהִ֗ים דֶּ֚רֶךְ אֶ֣רֶץ פְּלִשְׁתִּ֔ים כִּ֥י קָר֖וֹב ה֑וּא כִּ֣י ׀ אָמַ֣ר אֱלֹהִ֗ים פֶּֽן־יִנָּחֵ֥ם הָעָ֛ם בִּרְאֹתָ֥ם מִלְחָמָ֖ה וְשָׁ֥בוּ מִצְרָֽיְמָה׃ כְּשֶׁפַּרְעֹה שִׁחְרֵר אֶת הָעָם, אֱלֹהִים לֹא הוֹבִיל אוֹתָם דֶּרֶךְ אֶרֶץ פְּלִשְׁתִּים כִּי זֶה קָרוֹב, כִּי אֱלֹהִים חָשַׁשׁ שֶׁהָעָם יִתְחָרְטוּ כְּשֶׁיִּרְאוּ מִלְחָמָה וְיַחְזְרוּ לְמִצְרַיִם.
יח וַיַּסֵּ֨ב אֱלֹהִ֧ים ׀ אֶת־הָעָ֛ם דֶּ֥רֶךְ הַמִּדְבָּ֖ר יַם־ס֑וּף וַחֲמֻשִׁ֛ים עָל֥וּ בְנֵי־יִשְׂרָאֵ֖ל מֵאֶ֥רֶץ מִצְרָֽיִם׃ אֱלֹהִים סוֹבֵב אֶת הָעָם דֶּרֶךְ הַמִּדְבָּר לְיַם סוּף, בְּנֵי יִשְׂרָאֵל עָלוּ מֵאֶרֶץ מִצְרַיִם חֲמוּשִׁים בְּנֶשֶׁק.
יט וַיִּקַּ֥ח מֹשֶׁ֛ה אֶת־עַצְמ֥וֹת יוֹסֵ֖ף עִמּ֑וֹ כִּי֩ הַשְׁבֵּ֨עַ הִשְׁבִּ֜יעַ אֶת־בְּנֵ֤י יִשְׂרָאֵל֙ לֵאמֹ֔ר פָּקֹ֨ד יִפְקֹ֤ד אֱלֹהִים֙ אֶתְכֶ֔ם וְהַעֲלִיתֶ֧ם אֶת־עַצְמֹתַ֛י מִזֶּ֖ה אִתְּכֶֽם׃ מֹשֶׁה לָקַח אֶת עַצְמוֹת יוֹסֵף אִתּוֹ, כִּי יוֹסֵף הִשְׁבִּיעַ אֶת בְּנֵי יִשְׂרָאֵל וְאָמַר: כְּשֶׁאֱלֹהִים יִזְכּוֹר אֶתְכֶם וְיוֹצִיא אֶתְכֶם תַּעֲלוּ אֶת עַצְמוֹתַי מִכָּאן אִתְּכֶם.
כ וַיִּסְע֖וּ מִסֻּכֹּ֑ת וַיַּחֲנ֣וּ בְאֵתָ֔ם בִּקְצֵ֖ה הַמִּדְבָּֽר׃ הֵם נָסְעוּ מִסֻּכּוֹת וְחָנוּ בְּאֵתָם שֶׁבִּקְצֵה הַמִּדְבָּר.
כא וַֽיהֹוָ֡ה הֹלֵךְ֩ לִפְנֵיהֶ֨ם יוֹמָ֜ם בְּעַמּ֤וּד עָנָן֙ לַנְחֹתָ֣ם הַדֶּ֔רֶךְ וְלַ֛יְלָה בְּעַמּ֥וּד אֵ֖שׁ לְהָאִ֣יר לָהֶ֑ם לָלֶ֖כֶת יוֹמָ֥ם וָלָֽיְלָה׃ וַה' הוֹלֵךְ לִפְנֵיהֶם בַּיּוֹם בְּעַמּוּד עָנָן לְהַדְרִיךְ אוֹתָם בַּדֶּרֶךְ, וּבַלַּיְלָה בְּעַמּוּד אֵשׁ לְהָאִיר לָהֶם, כְּדֵי שֶׁיּוּכְלוּ לָלֶכֶת בַּיּוֹם וּבַלַּיְלָה.
כב לֹֽא־יָמִ֞ישׁ עַמּ֤וּד הֶֽעָנָן֙ יוֹמָ֔ם וְעַמּ֥וּד הָאֵ֖שׁ לָ֑יְלָה לִפְנֵ֖י הָעָֽם׃ עַמּוּד הֶעָנָן בַּיּוֹם וְעַמּוּד הָאֵשׁ בַּלַּיְלָה לֹא עָזְבוּ אֶת מְקוֹמָם לִפְנֵי הָעָם.
פ פָּרָשַׁת מִרְדַּף פַּרְעֹה
יד א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב דַּבֵּר֮ אֶל־בְּנֵ֣י יִשְׂרָאֵל֒ וְיָשֻׁ֗בוּ וְיַחֲנוּ֙ לִפְנֵי֙ פִּ֣י הַחִירֹ֔ת בֵּ֥ין מִגְדֹּ֖ל וּבֵ֣ין הַיָּ֑ם לִפְנֵי֙ בַּ֣עַל צְפֹ֔ן נִכְח֥וֹ תַחֲנ֖וּ עַל־הַיָּֽם׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל שֶׁיַּחְזְרוּ וְיַחֲנוּ לִפְנֵי 'פֶּתַח הַחִירוֹת' שֶׁבֵּין מִגְדּוֹל וּבֵין הַיָּם, לִפְנֵי 'בַּעַל צְפוֹן' – תַּחֲנוּ מוּלוֹ לְיַד הַיָּם.
ג וְאָמַ֤ר פַּרְעֹה֙ לִבְנֵ֣י יִשְׂרָאֵ֔ל נְבֻכִ֥ים הֵ֖ם בָּאָ֑רֶץ סָגַ֥ר עֲלֵיהֶ֖ם הַמִּדְבָּֽר׃ וּפַרְעֹה יֹאמַר עַל בְּנֵי יִשְׂרָאֵל: "הֵם תּוֹעִים בַּדֶּרֶךְ בָּאָרֶץ – הַמִּדְבָּר מַקִּיף אוֹתָם"!
ד וְחִזַּקְתִּ֣י אֶת־לֵב־פַּרְעֹה֮ וְרָדַ֣ף אַחֲרֵיהֶם֒ וְאִכָּבְדָ֤ה בְּפַרְעֹה֙ וּבְכׇל־חֵיל֔וֹ וְיָדְע֥וּ מִצְרַ֖יִם כִּֽי־אֲנִ֣י יְהֹוָ֑ה וַיַּֽעֲשׂוּ־כֵֽן׃ וְאַקְשִׁיחַ אֶת לֵב פַּרְעֹה וְהוּא יִרְדּוֹף אַחֲרֵיהֶם, וְאֶתְכַּבֵּד בְּכָךְ שֶׁאֶהֱרוֹג אֶת פַּרְעֹה וְאֶת כָּל צְבָאוֹ, וּמִצְרַיִם יֵדְעוּ שֶׁאֲנִי ה', בְּנֵי יִשְׂרָאֵל עָשׂוּ כָּךְ.
ה וַיֻּגַּד֙ לְמֶ֣לֶךְ מִצְרַ֔יִם כִּ֥י בָרַ֖ח הָעָ֑ם וַ֠יֵּהָפֵ֠ךְ לְבַ֨ב פַּרְעֹ֤ה וַעֲבָדָיו֙ אֶל־הָעָ֔ם וַיֹּֽאמְרוּ֙ מַה־זֹּ֣את עָשִׂ֔ינוּ כִּֽי־שִׁלַּ֥חְנוּ אֶת־יִשְׂרָאֵ֖ל מֵעׇבְדֵֽנוּ׃ סִפְּרוּ לְמֶלֶךְ מִצְרַיִם שֶׁהָעָם בָּרְחוּ, יַחֲסָם שֶׁל פַּרְעֹה וַעֲבָדָיו לָעָם הִתְהַפֵּךְ, הֵם אָמְרוּ: מַה עָשִׂינוּ שֶׁשִּׁחְרַרְנוּ אֶת יִשְׂרָאֵל מִלַּעֲבוֹד עֲבוּרֵנוּ!
ו וַיֶּאְסֹ֖ר אֶת־רִכְבּ֑וֹ וְאֶת־עַמּ֖וֹ לָקַ֥ח עִמּֽוֹ׃ הוּא רָתַם אֶת מֶרְכַּבְתּוֹ, וְלָקַח אִתּוֹ אֶת עַמּוֹ.
ז וַיִּקַּ֗ח שֵׁשׁ־מֵא֥וֹת רֶ֙כֶב֙ בָּח֔וּר וְכֹ֖ל רֶ֣כֶב מִצְרָ֑יִם וְשָׁלִשִׁ֖ם עַל־כֻּלּֽוֹ׃ הוּא לָקַח שֵׁשׁ מֵאוֹת מֶרְכָּבוֹת מֻבְחָרוֹת וְאֶת כָּל הַמֶּרְכָּבוֹת שֶׁל מִצְרַיִם וּמִנָּה מְפַקְּדִים עַל כֻּלָּם.
ח וַיְחַזֵּ֣ק יְהֹוָ֗ה אֶת־לֵ֤ב פַּרְעֹה֙ מֶ֣לֶךְ מִצְרַ֔יִם וַיִּרְדֹּ֕ף אַחֲרֵ֖י בְּנֵ֣י יִשְׂרָאֵ֑ל וּבְנֵ֣י יִשְׂרָאֵ֔ל יֹצְאִ֖ים בְּיָ֥ד רָמָֽה׃ ה' הִקְשִׁיחַ אֶת לֵב פַּרְעֹה מֶלֶךְ מִצְרַיִם וְהוּא רָדַף אַחֲרֵי בְּנֵי יִשְׂרָאֵל, וּבְנֵי יִשְׂרָאֵל יוֹצְאִים בְּיָד מוּרֶמֶת בְּגַאֲוָה.

שני[עריכה]

ט וַיִּרְדְּפ֨וּ מִצְרַ֜יִם אַחֲרֵיהֶ֗ם וַיַּשִּׂ֤יגוּ אוֹתָם֙ חֹנִ֣ים עַל־הַיָּ֔ם כׇּל־סוּס֙ רֶ֣כֶב פַּרְעֹ֔ה וּפָרָשָׁ֖יו וְחֵיל֑וֹ עַל־פִּי֙ הַֽחִירֹ֔ת לִפְנֵ֖י בַּ֥עַל צְפֹֽן׃ מִצְרַיִם רָדְפוּ אַחֲרֵיהֶם וְהִשִּׂיגוּ אוֹתָם חוֹנִים עַל יַד הַיָּם, כָּל סוּסֵי מַרְכְּבוֹת פַּרְעֹה וּפָרָשָׁיו וּצְבָאוֹ, עַל יַד 'פֶּתַח הַחִירוֹת' לִפְנֵי 'בַּעַל צְפוֹן'.
י וּפַרְעֹ֖ה הִקְרִ֑יב וַיִּשְׂאוּ֩ בְנֵֽי־יִשְׂרָאֵ֨ל אֶת־עֵינֵיהֶ֜ם וְהִנֵּ֥ה מִצְרַ֣יִם ׀ נֹסֵ֣עַ אַחֲרֵיהֶ֗ם וַיִּֽירְאוּ֙ מְאֹ֔ד וַיִּצְעֲק֥וּ בְנֵֽי־יִשְׂרָאֵ֖ל אֶל־יְהֹוָֽה׃ וּפַרְעֹה הִתְקָרֵב, בְּנֵי יִשְׂרָאֵל הֵרִימוּ אֶת עֵינֵיהֶם וְהִנֵּה מִצְרַיִם נוֹסְעִים אַחֲרֵיהֶם, הֵם פָּחֲדוּ מְאוֹד, וּבְנֵי יִשְׂרָאֵל צָעֲקוּ אֶל ה'.
יא וַיֹּאמְרוּ֮ אֶל־מֹשֶׁה֒ הֲֽמִבְּלִ֤י אֵין־קְבָרִים֙ בְּמִצְרַ֔יִם לְקַחְתָּ֖נוּ לָמ֣וּת בַּמִּדְבָּ֑ר מַה־זֹּאת֙ עָשִׂ֣יתָ לָּ֔נוּ לְהוֹצִיאָ֖נוּ מִמִּצְרָֽיִם׃ הֵם אָמְרוּ לְמֹשֶׁה: וְכִי בִּגְלַל מַחְסוֹר בִּקְבָרִים בְּמִצְרַיִם לָקַחְתָּ אוֹתָנוּ לָמוּת בַּמִּדְבָּר?! מַה זֶּה עָשִׂיתָ לָנוּ לְהוֹצִיא אוֹתָנוּ מִמִּצְרַיִם!
יב הֲלֹא־זֶ֣ה הַדָּבָ֗ר אֲשֶׁר֩ דִּבַּ֨רְנוּ אֵלֶ֤יךָ בְמִצְרַ֙יִם֙ לֵאמֹ֔ר חֲדַ֥ל מִמֶּ֖נּוּ וְנַֽעַבְדָ֣ה אֶת־מִצְרָ֑יִם כִּ֣י ט֥וֹב לָ֙נוּ֙ עֲבֹ֣ד אֶת־מִצְרַ֔יִם מִמֻּתֵ֖נוּ בַּמִּדְבָּֽר׃ הֲרֵי זֶה הַדָּבָר שֶׁדִּבַּרְנוּ אֵלֶיךָ בְּמִצְרַיִם, אָמַרְנוּ: עֲזוֹב אוֹתָנוּ וְנַעֲבוֹד עֲבוּר מִצְרַיִם, כִּי עָדִיף לָנוּ לַעֲבוֹד עֲבוּר מִצְרַיִם מֵאֲשֶׁר לָמוּת בַּמִּדְבָּר!
יג וַיֹּ֨אמֶר מֹשֶׁ֣ה אֶל־הָעָם֮ אַל־תִּירָ֒אוּ֒ הִֽתְיַצְּב֗וּ וּרְאוּ֙ אֶת־יְשׁוּעַ֣ת יְהֹוָ֔ה אֲשֶׁר־יַעֲשֶׂ֥ה לָכֶ֖ם הַיּ֑וֹם כִּ֗י אֲשֶׁ֨ר רְאִיתֶ֤ם אֶת־מִצְרַ֙יִם֙ הַיּ֔וֹם לֹ֥א תֹסִ֛פוּ לִרְאֹתָ֥ם ע֖וֹד עַד־עוֹלָֽם׃ מֹשֶׁה אָמַר לָעָם: אַל תִּפְחֲדוּ! עִמְדוּ וּרְאוּ אֶת הַיְּשׁוּעָה שֶׁה' יַצִּיל אֶתְכֶם הַיּוֹם, כִּי מַה שֶּׁרְאִיתֶם אֶת מִצְרַיִם הַיּוֹם – לֹא תִּרְאוּ אוֹתָם עוֹד שׁוּב לְעוֹלָם!
יד יְהֹוָ֖ה יִלָּחֵ֣ם לָכֶ֑ם וְאַתֶּ֖ם תַּחֲרִשֽׁוּן׃ ה' יִלָּחֵם עֲבוּרְכֶם, וְאַתֶּם תִּשְׁתְּקוּ.

שלישי[עריכה]

פ פָּרָשַׁת בְּקִיעַת הַיָּם
טו וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה מַה־תִּצְעַ֖ק אֵלָ֑י דַּבֵּ֥ר אֶל־בְּנֵי־יִשְׂרָאֵ֖ל וְיִסָּֽעוּ׃ ה' אָמַר לְמֹשֶׁה: מַה אַתָּה צוֹעֵק אֵלַי, דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְיִסְעוּ!
טז וְאַתָּ֞ה הָרֵ֣ם אֶֽת־מַטְּךָ֗ וּנְטֵ֧ה אֶת־יָדְךָ֛ עַל־הַיָּ֖ם וּבְקָעֵ֑הוּ וְיָבֹ֧אוּ בְנֵֽי־יִשְׂרָאֵ֛ל בְּת֥וֹךְ הַיָּ֖ם בַּיַּבָּשָֽׁה׃ וְאַתָּה, הָנֵף אֶת הַמַקֵּל שֶׁלְּךָ וְהָנֵף אֶת יָדְךָ אֶל הַיָּם וּבְקַע אוֹתוֹ, וּבְנֵי יִשְׂרָאֵל יָבוֹאוּ לְתוֹךְ הַיָּם בַּיַּבָּשָׁה.
יז וַאֲנִ֗י הִנְנִ֤י מְחַזֵּק֙ אֶת־לֵ֣ב מִצְרַ֔יִם וְיָבֹ֖אוּ אַחֲרֵיהֶ֑ם וְאִכָּבְדָ֤ה בְּפַרְעֹה֙ וּבְכׇל־חֵיל֔וֹ בְּרִכְבּ֖וֹ וּבְפָרָשָֽׁיו׃ וַאֲנִי הִנֵּה אֲנִי מַקְשִׁיחַ אֶת לֵב הַמִּצְרִים וְהֵם יָבוֹאוּ אַחֲרֵיהֶם, וְאֶתְכַּבֵּד בְּכָךְ שֶׁאֶהֱרוֹג אֶת פַּרְעֹה וְאֶת כָּל צְבָאוֹ עִם רִכְבּוֹ וּפָרָשָׁיו.
יח וְיָדְע֥וּ מִצְרַ֖יִם כִּי־אֲנִ֣י יְהֹוָ֑ה בְּהִכָּבְדִ֣י בְּפַרְעֹ֔ה בְּרִכְבּ֖וֹ וּבְפָרָשָֽׁיו׃ וּמִצְרַיִם יֵדְעוּ שֶׁאֲנִי ה', כְּשֶׁאֶתְכַּבֵּד בְּנִקְמָתִי בְּפַרְעֹה בְּרִכְבּוֹ וּבְפָרָשָׁיו.
יט וַיִּסַּ֞ע מַלְאַ֣ךְ הָאֱלֹהִ֗ים הַהֹלֵךְ֙ לִפְנֵי֙ מַחֲנֵ֣ה יִשְׂרָאֵ֔ל וַיֵּ֖לֶךְ מֵאַחֲרֵיהֶ֑ם וַיִּסַּ֞ע עַמּ֤וּד הֶֽעָנָן֙ מִפְּנֵיהֶ֔ם וַיַּֽעֲמֹ֖ד מֵאַחֲרֵיהֶֽם׃ שְׁלִיחַ הָאֱלֹהִים שֶׁהוֹלֵךְ לִפְנֵי מַחֲנֵה יִשְׂרָאֵל נָסַע וְהָלַךְ מֵאֲחוֹרֵיהֶם לִשְׁמוֹר אוֹתָם – עַמּוּד הֶעָנָן נָסַע מִלִּפְנֵיהֶם וְעָמַד מֵאַחֲרֵיהֶם.
כ וַיָּבֹ֞א בֵּ֣ין ׀ מַחֲנֵ֣ה מִצְרַ֗יִם וּבֵין֙ מַחֲנֵ֣ה יִשְׂרָאֵ֔ל וַיְהִ֤י הֶֽעָנָן֙ וְהַחֹ֔שֶׁךְ וַיָּ֖אֶר אֶת־הַלָּ֑יְלָה וְלֹא־קָרַ֥ב זֶ֛ה אֶל־זֶ֖ה כׇּל־הַלָּֽיְלָה׃ הוּא בָּא בֵּין מַחֲנֵה מִצְרַיִם וּבֵין מַחֲנֵה יִשְׂרָאֵל, הֶעָנָן הֶחְשִׁיךְ לְמִצְרַיִם – שֶׁעַמּוּד הָאֵשׁ שֶׁהֵאִיר אֶת הַלַּיְלָה לֹא יָאִיר לָהֶם, וְלֹא הִתְקָרֵב מַחֲנֶה זֶה לְמַחֲנֶה זֶה כָּל הַלַּיְלָה.
כא וַיֵּ֨ט מֹשֶׁ֣ה אֶת־יָדוֹ֮ עַל־הַיָּם֒ וַיּ֣וֹלֶךְ יְהֹוָ֣ה ׀ אֶת־הַ֠יָּ֠ם בְּר֨וּחַ קָדִ֤ים עַזָּה֙ כׇּל־הַלַּ֔יְלָה וַיָּ֥שֶׂם אֶת־הַיָּ֖ם לֶחָרָבָ֑ה וַיִּבָּקְע֖וּ הַמָּֽיִם׃ מֹשֶׁה הֵנִיף אֶת יָדוֹ עַל הַיָּם, וַה' הֶעֱבִיר עַל הַיָּם רוּחַ מִזְרָחִית חֲזָקָה כָּל הַלַּיְלָה וְהָפַךְ אֶת הַיָּם לְיַבָּשָׁה, וְהַמַּיִם נִבְקְעוּ.
כב וַיָּבֹ֧אוּ בְנֵֽי־יִשְׂרָאֵ֛ל בְּת֥וֹךְ הַיָּ֖ם בַּיַּבָּשָׁ֑ה וְהַמַּ֤יִם לָהֶם֙ חוֹמָ֔ה מִֽימִינָ֖ם וּמִשְּׂמֹאלָֽם׃ בְּנֵי יִשְׂרָאֵל הָלְכוּ בְּתוֹךְ הַיָּם בַּיַּבָּשָׁה, וְהַמַּיִם הָיוּ לָהֶם כְּמוֹ חוֹמָה מִיְּמִינָם וּמִשְּׂמֹאלָם.
כג וַיִּרְדְּפ֤וּ מִצְרַ֙יִם֙ וַיָּבֹ֣אוּ אַחֲרֵיהֶ֔ם כֹּ֚ל ס֣וּס פַּרְעֹ֔ה רִכְבּ֖וֹ וּפָרָשָׁ֑יו אֶל־תּ֖וֹךְ הַיָּֽם׃ מִצְרַיִם רָדְפוּ וּבָאוּ אַחֲרֵיהֶם – כָּל סוּסֵי פַּרְעֹה מַרְכְּבוֹתָיו וּפָרָשָׁיו, אֶל תּוֹךְ הַיָּם.
כד וַֽיְהִי֙ בְּאַשְׁמֹ֣רֶת הַבֹּ֔קֶר וַיַּשְׁקֵ֤ף יְהֹוָה֙ אֶל־מַחֲנֵ֣ה מִצְרַ֔יִם בְּעַמּ֥וּד אֵ֖שׁ וְעָנָ֑ן וַיָּ֕הׇם אֵ֖ת מַחֲנֵ֥ה מִצְרָֽיִם׃ לִפְנוֹת בּוֹקֶר, ה' פָּנָה אֶל מַחֲנֵה מִצְרַיִם בְּעַמּוּד אֵשׁ וְעָנָן וְעָשָׂה מְהוּמָה בְּמַחֲנֵה מִצְרַיִם.
כה וַיָּ֗סַר אֵ֚ת אֹפַ֣ן מַרְכְּבֹתָ֔יו וַֽיְנַהֲגֵ֖הוּ בִּכְבֵדֻ֑ת וַיֹּ֣אמֶר מִצְרַ֗יִם אָנ֙וּסָה֙ מִפְּנֵ֣י יִשְׂרָאֵ֔ל כִּ֣י יְהֹוָ֔ה נִלְחָ֥ם לָהֶ֖ם בְּמִצְרָֽיִם׃ הוּא הֵסִיר אֶת גַּלְגְּלֵי מַרְכְּבוֹתָיו וְהוֹבִיל אוֹתוֹ בִּכְבֵדוּת, מִצְרַיִם אָמְרוּ: נִבְרַח מִבְּנֵי יִשְׂרָאֵל כִּי ה' נִלְחָם עֲבוּרָם בְּמִצְרַיִם.

רביעי[עריכה]

פ פָּרָשַׁת טְבִיעַת מִצְרַיִם
כו וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה נְטֵ֥ה אֶת־יָדְךָ֖ עַל־הַיָּ֑ם וְיָשֻׁ֤בוּ הַמַּ֙יִם֙ עַל־מִצְרַ֔יִם עַל־רִכְבּ֖וֹ וְעַל־פָּרָשָֽׁיו׃ ה' אָמַר לְמֹשֶׁה: הָנֵף אֶת יָדְךָ עַל הַיָּם, וְהַמַּיִם יַחְזְרוּ עַל מִצְרַיִם עַל רִכְבּוֹ וְעַל פָּרָשָׁיו.
כז וַיֵּט֩ מֹשֶׁ֨ה אֶת־יָד֜וֹ עַל־הַיָּ֗ם וַיָּ֨שׇׁב הַיָּ֜ם לִפְנ֥וֹת בֹּ֙קֶר֙ לְאֵ֣יתָנ֔וֹ וּמִצְרַ֖יִם נָסִ֣ים לִקְרָאת֑וֹ וַיְנַעֵ֧ר יְהֹוָ֛ה אֶת־מִצְרַ֖יִם בְּת֥וֹךְ הַיָּֽם׃ מֹשֶׁה הֵנִיף אֶת יָדוֹ אֶל עֵבֶר הַיָּם וְהַיָּם חָזַר לִפְנוֹת בּוֹקֶר לְמַצָּבוֹ הַקָּבוּעַ, וּמִצְרַיִם בּוֹרְחִים לִקְרָאתוֹ, וַה' נִעֵר אֶת מִצְרַיִם בְּתוֹךְ הַיָּם.
כח וַיָּשֻׁ֣בוּ הַמַּ֗יִם וַיְכַסּ֤וּ אֶת־הָרֶ֙כֶב֙ וְאֶת־הַפָּ֣רָשִׁ֔ים לְכֹל֙ חֵ֣יל פַּרְעֹ֔ה הַבָּאִ֥ים אַחֲרֵיהֶ֖ם בַּיָּ֑ם לֹֽא־נִשְׁאַ֥ר בָּהֶ֖ם עַד־אֶחָֽד׃ הַמַּיִם חָזְרוּ וְכִסּוּ אֶת הָרֶכֶב וְאֶת הַפָּרָשִׁים – כָּל צְבָא פַּרְעֹה שֶׁבָּאוּ אַחֲרֵיהֶם בַּיָּם – לֹא נִשְׁאַר מֵהֶם אַף אֶחָד.
כט וּבְנֵ֧י יִשְׂרָאֵ֛ל הָלְכ֥וּ בַיַּבָּשָׁ֖ה בְּת֣וֹךְ הַיָּ֑ם וְהַמַּ֤יִם לָהֶם֙ חֹמָ֔ה מִֽימִינָ֖ם וּמִשְּׂמֹאלָֽם׃ וּבְנֵי יִשְׂרָאֵל הָלְכוּ בַּיַּבָּשָׁה בְּתוֹךְ הַיָּם, וְהַמַּיִם הָיוּ לָהֶם כְּמוֹ חוֹמָה מִיְּמִינָם וּמִשְּׂמֹאלָם.
ל וַיּ֨וֹשַׁע יְהֹוָ֜ה בַּיּ֥וֹם הַה֛וּא אֶת־יִשְׂרָאֵ֖ל מִיַּ֣ד מִצְרָ֑יִם וַיַּ֤רְא יִשְׂרָאֵל֙ אֶת־מִצְרַ֔יִם מֵ֖ת עַל־שְׂפַ֥ת הַיָּֽם׃ ה' הוֹשִׁיעַ בַּיּוֹם הַהוּא אֶת יִשְׂרָאֵל מִיַּד מִצְרַיִם, יִשְׂרָאֵל רָאוּ אֶת מִצְרַיִם מֵת עַל שְׂפַת הַיָּם.
לא וַיַּ֨רְא יִשְׂרָאֵ֜ל אֶת־הַיָּ֣ד הַגְּדֹלָ֗ה אֲשֶׁ֨ר עָשָׂ֤ה יְהֹוָה֙ בְּמִצְרַ֔יִם וַיִּֽירְא֥וּ הָעָ֖ם אֶת־יְהֹוָ֑ה וַיַּֽאֲמִ֙ינוּ֙ בַּֽיהֹוָ֔ה וּבְמֹשֶׁ֖ה עַבְדּֽוֹ׃ יִשְׂרָאֵל רָאוּ אֶת הַכֹּחַ הַגָּדוֹל שֶׁהִכָּה בּוֹ ה' בְּמִצְרַיִם, וְהָעָם הֶעֱרִיצוּ אֶת ה', וְהֶאֱמִינוּ בַּה' וּבְמֹשֶׁה עַבְדּוֹ.
פ פָּרָשַׁת שִׁירַת הַיָּם
טו א אָ֣ז יָשִֽׁיר־מֹשֶׁה֩ וּבְנֵ֨י יִשְׂרָאֵ֜ל אֶת־הַשִּׁירָ֤ה הַזֹּאת֙ לַֽיהֹוָ֔ה וַיֹּאמְר֖וּ לֵאמֹ֑ר  אָשִׁ֤ירָה לַֽיהֹוָה֙ כִּֽי־גָאֹ֣ה גָּאָ֔ה  ס֥וּס וְרֹכְב֖וֹ רָמָ֥ה בַיָּֽם׃ אָז מֹשֶׁה וּבְנֵי יִשְׂרָאֵל שָׁרוּ אֶת הַשִּׁירָה הַזֹּאת לַה', וְאָמְרוּ כָּךְ: אָשִׁיר לַה' כִּי הוּא הֶרְאָה אֶת גְּדֻלָּתוֹ – הוּא הִשְׁלִיךְ בַּיָּם סוּס וְרוֹכְבוֹ.
ב עׇזִּ֤י וְזִמְרָת֙ יָ֔הּ וַֽיְהִי־לִ֖י לִֽישׁוּעָ֑ה  זֶ֤ה אֵלִי֙ וְאַנְוֵ֔הוּ  אֱלֹהֵ֥י אָבִ֖י וַאֲרֹמְמֶֽנְהוּ׃ ה' הוּא הָעוֹז שֶׁלִּי וְהַזִּמְרָה שֶׁלִּי וְהוּא הוֹשִׁיעַ אוֹתִי, זֶה אֵלִי וַאֲסַפֵּר אֶת שִׁבְחוֹ, הוּא אֱלֹהֵי אָבִי וַאֲרוֹמֵם אוֹתוֹ.
ג יְהֹוָ֖ה אִ֣ישׁ מִלְחָמָ֑ה יְהֹוָ֖ה שְׁמֽוֹ׃ ה' לוֹחֵם מוּמְחֶה – ה' שְׁמוֹ.
ד מַרְכְּבֹ֥ת פַּרְעֹ֛ה וְחֵיל֖וֹ יָרָ֣ה בַיָּ֑ם  וּמִבְחַ֥ר שָֽׁלִשָׁ֖יו טֻבְּע֥וּ בְיַם־סֽוּף׃ הוּא הִשְׁלִיךְ בַּיָּם אֶת מַרְכְּבוֹת פַּרְעֹה וְאֶת צְבָאוֹ, הַטּוֹבִים שֶׁבִּמְפַקְּדָיו הֻטְבְּעוּ בְּיַם סוּף.
ה תְּהֹמֹ֖ת יְכַסְיֻ֑מוּ יָרְד֥וּ בִמְצוֹלֹ֖ת כְּמוֹ־אָֽבֶן׃ הַתְּהוֹמוֹת כִּסּוּ אוֹתָם, הֵם שָׁקְעוּ כְּמוֹ אֶבֶן בְּמַעֲמַקֵּי הַיָּם.
ו יְמִֽינְךָ֣ יְהֹוָ֔ה נֶאְדָּרִ֖י בַּכֹּ֑חַ  יְמִֽינְךָ֥ יְהֹוָ֖ה תִּרְעַ֥ץ אוֹיֵֽב׃ יְמִינְךָ ה' אַדִּירָה בַּכֹּחַ, יְמִינְךָ ה' תִּשְׁבּוֹר אוֹיֵב.
ז וּבְרֹ֥ב גְּאוֹנְךָ֖ תַּהֲרֹ֣ס קָמֶ֑יךָ  תְּשַׁלַּח֙ חֲרֹ֣נְךָ֔ יֹאכְלֵ֖מוֹ כַּקַּֽשׁ׃ וּבְרוֹב רוֹמְמוּתְךָ תַּהֲרוֹס אֶת הַקָּמִים עַל עַמְּךָ, תִּשְׁלַח אֶת כַּעַסְךָ שֶׁיֹּאכַל אוֹתָם כְּמוֹ אֵשׁ שֶׁשּׂוֹרֶפֶת קַשׁ.
ח וּבְר֤וּחַ אַפֶּ֙יךָ֙ נֶ֣עֶרְמוּ מַ֔יִם  נִצְּב֥וּ כְמוֹ־נֵ֖ד נֹזְלִ֑ים  קָֽפְא֥וּ תְהֹמֹ֖ת בְּלֶב־יָֽם׃ וּמֵהָרוּחַ שֶׁנָּשַׁבְתָּ נַעֲשׂוּ עֲרֵמוֹת שֶׁל מַיִם, וְהַנּוֹזְלִים עָמְדוּ כְּמוֹ חוֹמָה, הַתְּהוֹמוֹת צָפוּ בְּלֵב הַיָּם לִבְלוֹעַ אוֹתָם.
ט אָמַ֥ר אוֹיֵ֛ב אֶרְדֹּ֥ף אַשִּׂ֖יג  אֲחַלֵּ֣ק שָׁלָ֑ל תִּמְלָאֵ֣מוֹ נַפְשִׁ֔י  אָרִ֣יק חַרְבִּ֔י תּוֹרִישֵׁ֖מוֹ יָדִֽי׃ הָאוֹיֵב אָמַר: אֶרְדּוֹף, אַשִּׂיג אוֹתָם, אֲחַלֵּק שָׁלָל, נַפְשִׁי תִּתְמַלֵּא מֵהַשָּׁלָל, אֶשְׁלוֹף אֶת חַרְבִּי, אֶכְבּוֹשׁ אוֹתָם בְּכוֹחִי.
י נָשַׁ֥פְתָּ בְרוּחֲךָ֖ כִּסָּ֣מוֹ יָ֑ם  צָֽלְלוּ֙ כַּֽעוֹפֶ֔רֶת בְּמַ֖יִם אַדִּירִֽים׃ נָשַׁפְתָּ בָּהֶם בָּרוּח שֶׁלְּךָ וְהַיָּם כִּסָּה אוֹתָם, צָלְלוּ כְּמוֹ עוֹפֶרֶת בְּמַיִם אַדִּירִים.
יא מִֽי־כָמֹ֤כָה בָּֽאֵלִם֙ יְהֹוָ֔ה  מִ֥י כָּמֹ֖כָה נֶאְדָּ֣ר בַּקֹּ֑דֶשׁ  נוֹרָ֥א תְהִלֹּ֖ת עֹ֥שֵׂה פֶֽלֶא׃ ה'! אֵין עוֹד אֵל אֲמִתִּי כָּמוֹךָ בֵּין הָאֵלִים, אֵין עוֹד כָּמוֹךָ שֶׁקְּדֻשָּׁתוֹ אַדִּירָה, שֶׁמְּהַלְּלִים אֶת נוֹרְאוֹתָיו, עוֹשֵׂה נִפְלָאוֹת!
יב נָטִ֙יתָ֙ יְמִ֣ינְךָ֔ תִּבְלָעֵ֖מוֹ אָֽרֶץ׃ הֵנַפְתָּ אֶת יַד יְמִינְךָ – בָּלְעָה אוֹתָם הָאָרֶץ.
יג נָחִ֥יתָ בְחַסְדְּךָ֖ עַם־ז֣וּ גָּאָ֑לְתָּ  נֵהַ֥לְתָּ בְעׇזְּךָ֖ אֶל־נְוֵ֥ה קׇדְשֶֽׁךָ׃ הִדְרַכְתָּ בְּחַסְדְּךָ אֶת הָעָם שֶׁגָּאַלְתָּ, הוֹבַלְתָּ אוֹתָם בָּעוֹז שֶׁלְּךָ אֶל בֵּית מִקְדָּשְׁךָ.
יד שָֽׁמְע֥וּ עַמִּ֖ים יִרְגָּז֑וּן  חִ֣יל אָחַ֔ז יֹשְׁבֵ֖י פְּלָֽשֶׁת׃ שָׁמְעוּ עַמִּים וְרָעֲדוּ, פַּחַד אָחַז אֶת יוֹשְׁבֵי אֶרֶץ פְּלִשְׁתִּים.
טו אָ֤ז נִבְהֲלוּ֙ אַלּוּפֵ֣י אֱד֔וֹם  אֵילֵ֣י מוֹאָ֔ב יֹֽאחֲזֵ֖מוֹ רָ֑עַד  נָמֹ֕גוּ כֹּ֖ל יֹשְׁבֵ֥י כְנָֽעַן׃ אָז נִבְהֲלוּ שַׁלִּיטֵי אֱדוֹם, יֶאֱחוֹז רָעַד אֶת גִּבּוֹרֵי מוֹאָב, נָמַס לִבָּם שֶׁל כָּל יוֹשְׁבֵי כְּנַעַן.
טז תִּפֹּ֨ל עֲלֵיהֶ֤ם אֵימָ֙תָה֙ וָפַ֔חַד  בִּגְדֹ֥ל זְרוֹעֲךָ֖ יִדְּמ֣וּ כָּאָ֑בֶן  עַד־יַעֲבֹ֤ר עַמְּךָ֙ יְהֹוָ֔ה  עַֽד־יַעֲבֹ֖ר עַם־ז֥וּ קָנִֽיתָ׃ תִּפּוֹל עֲלֵיהֶם אֵימָה וָפַחַד, בְּחוֹזֶק הַכֹּח שֶׁלְּךָ הֵם יִשְׁתְּקוּ כְּמוֹ אֶבֶן, עַד שֶׁעַמְּךָ יַעֲבוֹר, ה', עַד שֶׁיַּעֲבוֹר הָעָם שֶׁשַּׁיָּךְ לְךָ.
יז תְּבִאֵ֗מוֹ וְתִטָּעֵ֙מוֹ֙ בְּהַ֣ר נַחֲלָֽתְךָ֔  מָכ֧וֹן לְשִׁבְתְּךָ֛ פָּעַ֖לְתָּ יְהֹוָ֑ה  מִקְּדָ֕שׁ אֲדֹנָ֖י כּוֹנְנ֥וּ יָדֶֽיךָ׃ תָּבִיא אוֹתָם וְתִטַּע אוֹתָם בְּהַר נַחֲלָתְךָ – בַּמָּקוֹם שֶׁהֵכַנְתָּ לְעַצְמְךָ לִשְׁכּוֹן בּוֹ, ה', בַּמִּקְדָשׁ שֶׁיָּדֶיךָ הֵכִינוּ, ה'.
יח יְהֹוָ֥ה ׀ יִמְלֹ֖ךְ לְעֹלָ֥ם וָעֶֽד׃ ה' יִמְלוֹךְ לְעוֹלָם וָעֶד.
יט כִּ֣י בָא֩ ס֨וּס פַּרְעֹ֜ה בְּרִכְבּ֤וֹ וּבְפָרָשָׁיו֙ בַּיָּ֔ם  וַיָּ֧שֶׁב יְהֹוָ֛ה עֲלֵהֶ֖ם אֶת־מֵ֣י הַיָּ֑ם  וּבְנֵ֧י יִשְׂרָאֵ֛ל הָלְכ֥וּ בַיַּבָּשָׁ֖ה בְּת֥וֹךְ הַיָּֽם׃ כְּשֶׁהַסּוּס שֶׁל פַּרְעֹה נִכְנַס עִם רִכְבּוֹ וּפָרָשָׁיו בַּיָּם, ה' הֶחְזִיר עֲלֵהֶם אֶת מֵי הַיָּם, וּבְנֵי יִשְׂרָאֵל עָבְרוּ בַּיַּבָּשָׁה בְּתוֹךְ הַיָּם.
פ
כ וַתִּקַּח֩ מִרְיָ֨ם הַנְּבִיאָ֜ה אֲח֧וֹת אַהֲרֹ֛ן אֶת־הַתֹּ֖ף בְּיָדָ֑הּ וַתֵּצֶ֤אןָ כׇֽל־הַנָּשִׁים֙ אַחֲרֶ֔יהָ בְּתֻפִּ֖ים וּבִמְחֹלֹֽת׃ מִרְיָם הַנְּבִיאָה אֲחוֹת אַהֲרֹן לָקְחָה אֶת הַתּוֹף בְּיָדָהּ, וְכָל הַנָּשִׁים יָצְאוּ אַחֲרֶיהָ בְּתֻפִּים וּבְרִקּוּדִים לִצְלִילֵי חָלִיל.
כא וַתַּ֥עַן לָהֶ֖ם מִרְיָ֑ם שִׁ֤ירוּ לַֽיהֹוָה֙ כִּֽי־גָאֹ֣ה גָּאָ֔ה ס֥וּס וְרֹכְב֖וֹ רָמָ֥ה בַיָּֽם׃ עָנְתָה לָהֶם מִרְיָם: שִׁירוּ לַה' כִּי הוּא הֶרְאָה אֶת גְּדֻלָּתוֹ – הוּא הִשְׁלִיךְ בַּיָּם סוּס וְרוֹכְבוֹ!
ס פָּרָשַׁת מָרָה
כב וַיַּסַּ֨ע מֹשֶׁ֤ה אֶת־יִשְׂרָאֵל֙ מִיַּם־ס֔וּף וַיֵּצְא֖וּ אֶל־מִדְבַּר־שׁ֑וּר וַיֵּלְכ֧וּ שְׁלֹֽשֶׁת־יָמִ֛ים בַּמִּדְבָּ֖ר וְלֹא־מָ֥צְאוּ מָֽיִם׃ מֹשֶׁה הִסִּיעַ אֶת יִשְׂרָאֵל מִיַּם סוּף וְהֵם יָצְאוּ לְמִדְבַּר שׁוּר, הֵם הָלְכוּ שְׁלוֹשָׁה יָמִים בַּמִּדְבָּר וְלֹא מָצְאוּ מַיִם.
כג וַיָּבֹ֣אוּ מָרָ֔תָה וְלֹ֣א יָֽכְל֗וּ לִשְׁתֹּ֥ת מַ֙יִם֙ מִמָּרָ֔ה כִּ֥י מָרִ֖ים הֵ֑ם עַל־כֵּ֥ן קָרָֽא־שְׁמָ֖הּ מָרָֽה׃ הֵם בָּאוּ לְמָרָה וְלֹא יָכְלוּ לִשְׁתּוֹת מַיִם מִמָּרָה כִּי הֵם מָרִים, לָכֵן נִקְרָא שְׁמָהּ מָרָה.
כד וַיִּלֹּ֧נוּ הָעָ֛ם עַל־מֹשֶׁ֥ה לֵּאמֹ֖ר מַה־נִּשְׁתֶּֽה׃ הָעָם הִתְלוֹנְנוּ עַל מֹשֶׁה, וְאָמְרוּ: מַה נִּשְׁתֶּה?
כה וַיִּצְעַ֣ק אֶל־יְהֹוָ֗ה וַיּוֹרֵ֤הוּ יְהֹוָה֙ עֵ֔ץ וַיַּשְׁלֵךְ֙ אֶל־הַמַּ֔יִם וַֽיִּמְתְּק֖וּ הַמָּ֑יִם שָׁ֣ם שָׂ֥ם ל֛וֹ חֹ֥ק וּמִשְׁפָּ֖ט וְשָׁ֥ם נִסָּֽהוּ׃ הוּא צָעַק אֶל ה', ה' הֶרְאָה לוֹ עֵץ, הוּא הִשְׁלִיךְ אוֹתוֹ לַמַּיִם וְהַמַּיִם הִתְמַתְּקוּ, שָׁם ה' חָקַק לוֹ חוֹק וּמִשְׁפָּט וְשָׁם נִסָּה אֶת הָעָם.
כו וַיֹּ֩אמֶר֩ אִם־שָׁמ֨וֹעַ תִּשְׁמַ֜ע לְק֣וֹל ׀ יְהֹוָ֣ה אֱלֹהֶ֗יךָ וְהַיָּשָׁ֤ר בְּעֵינָיו֙ תַּעֲשֶׂ֔ה וְהַֽאֲזַנְתָּ֙ לְמִצְוֺתָ֔יו וְשָׁמַרְתָּ֖ כׇּל־חֻקָּ֑יו כׇּֽל־הַמַּחֲלָ֞ה אֲשֶׁר־שַׂ֤מְתִּי בְמִצְרַ֙יִם֙ לֹא־אָשִׂ֣ים עָלֶ֔יךָ כִּ֛י אֲנִ֥י יְהֹוָ֖ה רֹפְאֶֽךָ׃ אָמַר: אִם תִּשְׁמַע בְּקוֹל ה' אֱלֹהֶיךָ וְתַעֲשֶׂה אֶת הַיָּשָׁר בְּעֵינָיו וְתַאֲזִין לַמִּצְווֹת שֶׁלּוֹ וְתִשְׁמוֹר אֶת כָּל הַחֻקִּים שֶׁלּוֹ, לֹא אָשִׂים עָלֶיךָ אֶת כָּל הַמַּחֲלוֹת שֶׁשַּׂמְתִּי בְּמִצְרַיִם כִּי אֲנִי ה' שֶׁמְּרַפֵּא אוֹתְךָ.

חמישי[עריכה]

ס פָּרָשַׁת הַמָּן
כז וַיָּבֹ֣אוּ אֵילִ֔מָה וְשָׁ֗ם שְׁתֵּ֥ים עֶשְׂרֵ֛ה עֵינֹ֥ת מַ֖יִם וְשִׁבְעִ֣ים תְּמָרִ֑ים וַיַּחֲנוּ־שָׁ֖ם עַל־הַמָּֽיִם׃ הֵם בָּאוּ לְאֵילִים וְהָיוּ שָׁם שְׁתֵּים עֶשְׂרֵה מַעְיָנוֹת מַיִם וְשִׁבְעִים עֲצֵי תָּמָר, הֵם חָנוּ שָׁם לְיַד הַמַּיִם.
טז א וַיִּסְעוּ֙ מֵֽאֵילִ֔ם וַיָּבֹ֜אוּ כׇּל־עֲדַ֤ת בְּנֵֽי־יִשְׂרָאֵל֙ אֶל־מִדְבַּר־סִ֔ין אֲשֶׁ֥ר בֵּין־אֵילִ֖ם וּבֵ֣ין סִינָ֑י בַּחֲמִשָּׁ֨ה עָשָׂ֥ר יוֹם֙ לַחֹ֣דֶשׁ הַשֵּׁנִ֔י לְצֵאתָ֖ם מֵאֶ֥רֶץ מִצְרָֽיִם׃ הֵם נָסְעוּ מֵאֵילִים וְכָל צִבּוּר בְּנֵי יִשְׂרָאֵל בָּאוּ לְמִדְבַּר סִין שֶׁבֵּין אֵילִים וּבֵין סִינַי, בַּיּוֹם הַחֲמִשָּׁה עָשָׂר בַּחוֹדֶשׁ הַשֵּׁנִי מִצֵּאתָם מֵאֶרֶץ מִצְרַיִם.
ב וילינו וַיִּלּ֜וֹנוּ כׇּל־עֲדַ֧ת בְּנֵי־יִשְׂרָאֵ֛ל עַל־מֹשֶׁ֥ה וְעַֽל־אַהֲרֹ֖ן בַּמִּדְבָּֽר׃ כָּל צִבּוּר בְּנֵי יִשְׂרָאֵל הִתְלוֹנְנוּ עַל מֹשֶׁה וְעַל אַהֲרֹן בַּמִּדְבָּר.
ג וַיֹּאמְר֨וּ אֲלֵהֶ֜ם בְּנֵ֣י יִשְׂרָאֵ֗ל מִֽי־יִתֵּ֨ן מוּתֵ֤נוּ בְיַד־יְהֹוָה֙ בְּאֶ֣רֶץ מִצְרַ֔יִם בְּשִׁבְתֵּ֙נוּ֙ עַל־סִ֣יר הַבָּשָׂ֔ר בְּאׇכְלֵ֥נוּ לֶ֖חֶם לָשֹׂ֑בַע כִּֽי־הוֹצֵאתֶ֤ם אֹתָ֙נוּ֙ אֶל־הַמִּדְבָּ֣ר הַזֶּ֔ה לְהָמִ֛ית אֶת־כׇּל־הַקָּהָ֥ל הַזֶּ֖ה בָּרָעָֽב׃ אָמְרוּ לָהֶם בְּנֵי יִשְׂרָאֵל: הַלְוַאי שֶׁהָיִינוּ מֵתִים בִּידֵי ה' בְּאֶרֶץ מִצְרַיִם כְּשֶׁיָּשַׁבְנוּ סָמוּךְ לְסִיר הַבָּשָׂר, כְּשֶׁאָכַלְנוּ לֶחֶם לָשׂוֹבַע, הֲרֵי הוֹצֵאתֶם אוֹתָנוּ לַמִּדְבָּר הַזֶּה לְהָמִית אֶת כָּל הַקָּהָל הַזֶּה בָּרָעָב!
ס
ד וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה הִנְנִ֨י מַמְטִ֥יר לָכֶ֛ם לֶ֖חֶם מִן־הַשָּׁמָ֑יִם וְיָצָ֨א הָעָ֤ם וְלָֽקְטוּ֙ דְּבַר־י֣וֹם בְּיוֹמ֔וֹ לְמַ֧עַן אֲנַסֶּ֛נּוּ הֲיֵלֵ֥ךְ בְּתוֹרָתִ֖י אִם־לֹֽא׃ ה' אָמַר לְמֹשֶׁה: הִנֵה אֲנִי מַמְטִיר לָכֶם לֶחֶם מֵהַשָּׁמַיִם וְהָעָם יֵצְאוּ וְיֶאֶסְפוּ בְּכָל יוֹם לְאוֹתוֹ הַיּוֹם, כְּדֵי שֶׁאֲנַסֶּה אוֹתָם אִם יֵלְכוּ בְּדֶרֶךְ תּוֹרָתִי וְיִבְטְחוּ בִּי שֶׁאֶתֵּן לָהֶם לֶחֶם, אוֹ לֹא.
ה וְהָיָה֙ בַּיּ֣וֹם הַשִּׁשִּׁ֔י וְהֵכִ֖ינוּ אֵ֣ת אֲשֶׁר־יָבִ֑יאוּ וְהָיָ֣ה מִשְׁנֶ֔ה עַ֥ל אֲשֶֽׁר־יִלְקְט֖וּ י֥וֹם ׀ יֽוֹם׃ בַּיּוֹם הַשִּׁשִּׁי יָכִינוּ אֶת הַלֶּחֶם שֶׁיָּבִיאוּ, וְיִהְיֶה פִּי שְׁנַיִם מִמַּה שֶּׁיֶּאֶסְפוּ בְּכָל יוֹם.
ו וַיֹּ֤אמֶר מֹשֶׁה֙ וְאַהֲרֹ֔ן אֶֽל־כׇּל־בְּנֵ֖י יִשְׂרָאֵ֑ל עֶ֕רֶב וִֽידַעְתֶּ֕ם כִּ֧י יְהֹוָ֛ה הוֹצִ֥יא אֶתְכֶ֖ם מֵאֶ֥רֶץ מִצְרָֽיִם׃ מֹשֶׁה וְאַהֲרֹן אָמְרוּ לְכָל בְּנֵי יִשְׂרָאֵל: הָעֶרֶב יִתְבָּרֵר לָכֶם שֶׁה' הוֹצִיא אֶתְכֶם מֵאֶרֶץ מִצְרַיִם וְלֹא אֲנַחְנוּ.
ז וּבֹ֗קֶר וּרְאִיתֶם֙ אֶת־כְּב֣וֹד יְהֹוָ֔ה בְּשׇׁמְע֥וֹ אֶת־תְּלֻנֹּתֵיכֶ֖ם עַל־יְהֹוָ֑ה וְנַ֣חְנוּ מָ֔ה כִּ֥י תלונו תַלִּ֖ינוּ עָלֵֽינוּ׃ וּבַבּוֹקֶר תִּרְאוּ אֶת כְּבוֹד ה' שֶׁשָּׁמַע אֶת תְּלֻנּוֹתֵיכֶם עַל ה', וַאֲנַחְנוּ, מִי אֲנַחְנוּ שֶׁתִּתְלוֹנְנוּ עָלֵינוּ.
ח וַיֹּ֣אמֶר מֹשֶׁ֗ה בְּתֵ֣ת יְהֹוָה֩ לָכֶ֨ם בָּעֶ֜רֶב בָּשָׂ֣ר לֶאֱכֹ֗ל וְלֶ֤חֶם בַּבֹּ֙קֶר֙ לִשְׂבֹּ֔עַ בִּשְׁמֹ֤עַ יְהֹוָה֙ אֶת־תְּלֻנֹּ֣תֵיכֶ֔ם אֲשֶׁר־אַתֶּ֥ם מַלִּינִ֖ם עָלָ֑יו וְנַ֣חְנוּ מָ֔ה לֹא־עָלֵ֥ינוּ תְלֻנֹּתֵיכֶ֖ם כִּ֥י עַל־יְהֹוָֽה׃ אָמַר מֹשֶׁה: זֶה יִהְיֶה כְּשֶׁה' יִתֵּן לָכֶם בָּעֶרֶב בָּשָׂר לֶאֱכוֹל וּבַבּוֹקֶר לֶחֶם לִשְׂבּוֹעַ, כִּי ה' שָׁמַע אֶת תְּלֻנּוֹתֵיכֶם שֶׁאַתֶּם מִתְלוֹנְנִים עָלָיו, וַאֲנַחְנוּ מִי אֲנַחְנוּ! לֹא עָלֵינוּ תְּלֻנּוֹתֵיכֶם אֶלָּא עַל ה'.
ט וַיֹּ֤אמֶר מֹשֶׁה֙ אֶֽל־אַהֲרֹ֔ן אֱמֹ֗ר אֶֽל־כׇּל־עֲדַת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל קִרְב֖וּ לִפְנֵ֣י יְהֹוָ֑ה כִּ֣י שָׁמַ֔ע אֵ֖ת תְּלֻנֹּתֵיכֶֽם׃ מֹשֶׁה אָמַר לְאַהֲרֹן: אֱמוֹר לְכָל קְהַל בְּנֵי יִשְׂרָאֵל: גְּשׁוּ לִפְנֵי גִּילוּי שֶׁל כְּבוֹד ה', כִּי הוּא שָׁמַע אֶת תְּלֻנּוֹתֵיכֶם.
י וַיְהִ֗י כְּדַבֵּ֤ר אַהֲרֹן֙ אֶל־כׇּל־עֲדַ֣ת בְּנֵֽי־יִשְׂרָאֵ֔ל וַיִּפְנ֖וּ אֶל־הַמִּדְבָּ֑ר וְהִנֵּה֙ כְּב֣וֹד יְהֹוָ֔ה נִרְאָ֖ה בֶּעָנָֽן׃ וּכְשֶׁאַהֲרֹן דִּבֵּר אֶל כָּל צִבּוּר בְּנֵי יִשְׂרָאֵל הֵם פָּנוּ אֶל הַמִּדְבָּר, וְהִנֵּה כְּבוֹד ה' נִרְאָה בֶּעָנָן.

ששי[עריכה]

פ
יא וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יב שָׁמַ֗עְתִּי אֶת־תְּלוּנֹּת֮ בְּנֵ֣י יִשְׂרָאֵל֒ דַּבֵּ֨ר אֲלֵהֶ֜ם לֵאמֹ֗ר בֵּ֤ין הָֽעַרְבַּ֙יִם֙ תֹּאכְל֣וּ בָשָׂ֔ר וּבַבֹּ֖קֶר תִּשְׂבְּעוּ־לָ֑חֶם וִֽידַעְתֶּ֕ם כִּ֛י אֲנִ֥י יְהֹוָ֖ה אֱלֹהֵיכֶֽם׃ שָׁמַעְתִּי אֶת תְּלוּנּוֹת בְּנֵי יִשְׂרָאֵל, דַּבֵּר אֲלֵהֶם וֶאֱמוֹר: אַחַר הַצָּהֳרַיִם תֹּאכְלוּ בָּשָׂר וּבַבּוֹקֶר תִּשְׂבְּעוּ מִלֶּחֶם, וְתֵדְעוּ שֶׁאֲנִי ה' אֱלֹהֵיכֶם.
יג וַיְהִ֣י בָעֶ֔רֶב וַתַּ֣עַל הַשְּׂלָ֔ו וַתְּכַ֖ס אֶת־הַֽמַּחֲנֶ֑ה וּבַבֹּ֗קֶר הָֽיְתָה֙ שִׁכְבַ֣ת הַטַּ֔ל סָבִ֖יב לַֽמַּחֲנֶֽה׃ בָּעֶרֶב עָלָה הָעוֹף הַנִּקְרָא שְׂלָו וְכִסָּה אֶת הַמַּחֲנֶה, וּבַבּוֹקֶר הָיְתָה שִׁכְבַת טַל סָבִיב לַמַּחֲנֶה.
יד וַתַּ֖עַל שִׁכְבַ֣ת הַטָּ֑ל וְהִנֵּ֞ה עַל־פְּנֵ֤י הַמִּדְבָּר֙ דַּ֣ק מְחֻסְפָּ֔ס דַּ֥ק כַּכְּפֹ֖ר עַל־הָאָֽרֶץ׃ שִׁכְבַת הַטַּל עָלְתָה, וְהִנֵּה רָאוּ עַל פְּנֵי הַמִּדְבָּר עַל הָאָרֶץ דָּבָר דַּק וְחָשׂוּף, דַּק כְּמוֹ בָּרָד.
טו וַיִּרְא֣וּ בְנֵֽי־יִשְׂרָאֵ֗ל וַיֹּ֨אמְר֜וּ אִ֤ישׁ אֶל־אָחִיו֙ מָ֣ן ה֔וּא כִּ֛י לֹ֥א יָדְע֖וּ מַה־ה֑וּא וַיֹּ֤אמֶר מֹשֶׁה֙ אֲלֵהֶ֔ם ה֣וּא הַלֶּ֔חֶם אֲשֶׁ֨ר נָתַ֧ן יְהֹוָ֛ה לָכֶ֖ם לְאׇכְלָֽה׃ בְּנֵי יִשְׂרָאֵל רָאוּ וְאָמְרוּ אֶחָד לַשֵּׁנִי: זוֹהִי מָנָה! כִּי לֹא יָדְעוּ מַה הוּא, מֹשֶׁה אָמַר לָהֶם: זֶה הַלֶּחֶם שֶׁה' נָתַן לָכֶם לֶאֱכוֹל.
טז זֶ֤ה הַדָּבָר֙ אֲשֶׁ֣ר צִוָּ֣ה יְהֹוָ֔ה לִקְט֣וּ מִמֶּ֔נּוּ אִ֖ישׁ לְפִ֣י אׇכְל֑וֹ עֹ֣מֶר לַגֻּלְגֹּ֗לֶת מִסְפַּר֙ נַפְשֹׁ֣תֵיכֶ֔ם אִ֛ישׁ לַאֲשֶׁ֥ר בְּאׇהֳל֖וֹ תִּקָּֽחוּ׃ זֶה הַדָּבָר שֶׁה' צִוָּה: אִסְפוּ מִמֶּנּוּ כָּל אֶחָד לְפִי צוֹרֶךְ אֲכִילָתוֹ – עוֹמֶר (כִּ2.16 לִיטֶר) לְכָל גֻּלְגּוֹלֶת כְּמִסְפַּר אַנְשֵׁיכֶם, כָּל אֶחָד יִקַּח לְאֵלֶּה שֶׁבְּאָהֳלוֹ.
יז וַיַּעֲשׂוּ־כֵ֖ן בְּנֵ֣י יִשְׂרָאֵ֑ל וַֽיִּלְקְט֔וּ הַמַּרְבֶּ֖ה וְהַמַּמְעִֽיט׃ בְּנֵי יִשְׂרָאֵל עָשׂוּ כָּךְ, הֵם אָסְפוּ – הַמַּרְבֶּה וְהַמַּמְעִיט.
יח וַיָּמֹ֣דּוּ בָעֹ֔מֶר וְלֹ֤א הֶעְדִּיף֙ הַמַּרְבֶּ֔ה וְהַמַּמְעִ֖יט לֹ֣א הֶחְסִ֑יר אִ֥ישׁ לְפִֽי־אׇכְל֖וֹ לָקָֽטוּ׃ הֵם מָדְדוּ בִּכְלִי שֶׁמֵכִיל מִדַּת עוֹמֶר, וְהַמַּרְבֶּה לֹא הוֹסִיף, וְהַמַּמְעִיט לֹא הֶחְסִיר, אֶלָּא כָּל אִישׁ אָסַף לְפִי יְכוֹלֶת אֲכִילָתוֹ.
יט וַיֹּ֥אמֶר מֹשֶׁ֖ה אֲלֵהֶ֑ם אִ֕ישׁ אַל־יוֹתֵ֥ר מִמֶּ֖נּוּ עַד־בֹּֽקֶר׃ מֹשֶׁה אָמַר לָהֶם: שׁוּם אִישׁ שֶׁלֹּא יַשְׁאִיר מִמֶּנּוּ עַד הַבּוֹקֶר!
כ וְלֹא־שָׁמְע֣וּ אֶל־מֹשֶׁ֗ה וַיּוֹתִ֨רוּ אֲנָשִׁ֤ים מִמֶּ֙נּוּ֙ עַד־בֹּ֔קֶר וַיָּ֥רֻם תּוֹלָעִ֖ים וַיִּבְאַ֑שׁ וַיִּקְצֹ֥ף עֲלֵהֶ֖ם מֹשֶֽׁה׃ וְלֹא שָׁמְעוּ בְּקוֹל מֹשֶׁה, וַאֲנָשִׁים הִשְׁאִירוּ מִמֶּנּוּ עַד הַבּוֹקֶר, וְנִהְיוּ בּוֹ תּוֹלָעִים וְהִסְרִיחַ, מֹשֶׁה כָּעַס עֲלֵיהֶם.
כא וַיִּלְקְט֤וּ אֹתוֹ֙ בַּבֹּ֣קֶר בַּבֹּ֔קֶר אִ֖ישׁ כְּפִ֣י אׇכְל֑וֹ וְחַ֥ם הַשֶּׁ֖מֶשׁ וְנָמָֽס׃ הֵם אָסְפוּ אוֹתוֹ מוּקְדָם בַּבּוֹקֶר – כָּל אֶחָד לְפִי אֲכִילָתוֹ, וּכְשֶׁהַשֶּׁמֶשׁ הִתְחַמְּמָה – הוּא נָמַס.
כב וַיְהִ֣י ׀ בַּיּ֣וֹם הַשִּׁשִּׁ֗י לָֽקְט֥וּ לֶ֙חֶם֙ מִשְׁנֶ֔ה שְׁנֵ֥י הָעֹ֖מֶר לָאֶחָ֑ד וַיָּבֹ֙אוּ֙ כׇּל־נְשִׂיאֵ֣י הָֽעֵדָ֔ה וַיַּגִּ֖ידוּ לְמֹשֶֽׁה׃ בַּיּוֹם הַשִּׁשִּׁי אָסְפוּ פִּי שְׁנַיִם לֶחֶם – שְׁנֵי עֳמָרִים לְכָל אֶחָד, כָּל נְשִׂיאֵי הַצִּבּוּר בָּאוּ וְסִפְּרוּ לְמֹשֶׁה.
כג וַיֹּ֣אמֶר אֲלֵהֶ֗ם ה֚וּא אֲשֶׁ֣ר דִּבֶּ֣ר יְהֹוָ֔ה שַׁבָּת֧וֹן שַׁבַּת־קֹ֛דֶשׁ לַֽיהֹוָ֖ה מָחָ֑ר אֵ֣ת אֲשֶׁר־תֹּאפ֞וּ אֵפ֗וּ וְאֵ֤ת אֲשֶֽׁר־תְּבַשְּׁלוּ֙ בַּשֵּׁ֔לוּ וְאֵת֙ כׇּל־הָ֣עֹדֵ֔ף הַנִּ֧יחוּ לָכֶ֛ם לְמִשְׁמֶ֖רֶת עַד־הַבֹּֽקֶר׃ אָמַר לָהֶם: זֶה מַה שֶּׁה' דִּבֵּר – מָחָר שַׁבָּתוֹן שַׁבַּת קוֹדֶשׁ לַה', אֶת מַה שֶּׁתֹּאפוּ – אֵפוּ, וְאֶת מַה שֶּׁתְּבַשְּׁלוּ – בַּשְּׁלוּ, וְאֶת כָּל מַה שֶּׁיִּשָּׁאֵר – הַנִּיחוּ לָכֶם לִשְׁמִירָה עַד הַבּוֹקֶר.
כד וַיַּנִּ֤יחוּ אֹתוֹ֙ עַד־הַבֹּ֔קֶר כַּאֲשֶׁ֖ר צִוָּ֣ה מֹשֶׁ֑ה וְלֹ֣א הִבְאִ֔ישׁ וְרִמָּ֖ה לֹא־הָ֥יְתָה בּֽוֹ׃ הֵם הִנִּיחוּ אוֹתוֹ עַד הַבּוֹקֶר כְּמוֹ שֶׁמֹּשֶׁה צִוָּה, וְהוּא לֹא הִסְרִיח, וְלֹא הָיוּ בּוֹ תּוֹלָעִים.
כה וַיֹּ֤אמֶר מֹשֶׁה֙ אִכְלֻ֣הוּ הַיּ֔וֹם כִּֽי־שַׁבָּ֥ת הַיּ֖וֹם לַיהֹוָ֑ה הַיּ֕וֹם לֹ֥א תִמְצָאֻ֖הוּ בַּשָּׂדֶֽה׃ אָמַר מֹשֶׁה: אִכְלוּ אוֹתוֹ הַיּוֹם כִּי הַיּוֹם יֵשׁ שְׁבִיתָה לַה', הַיּוֹם לֹא תִּמְצְאוּ אוֹתוֹ בַּשָּׂדֶה.
כו שֵׁ֥שֶׁת יָמִ֖ים תִּלְקְטֻ֑הוּ וּבַיּ֧וֹם הַשְּׁבִיעִ֛י שַׁבָּ֖ת לֹ֥א יִֽהְיֶה־בּֽוֹ׃ שִׁשָּׁה יָמִים תַּאַסְפוּ אוֹתוֹ, וּבַיּוֹם הַשְּׁבִיעִי שְׁבִיתָה – לֹא יִהְיֶה בּוֹ מָן.
כז וַֽיְהִי֙ בַּיּ֣וֹם הַשְּׁבִיעִ֔י יָצְא֥וּ מִן־הָעָ֖ם לִלְקֹ֑ט וְלֹ֖א מָצָֽאוּ׃ בַּיּוֹם הַשְּׁבִיעִי יָצְאוּ אֲנָשִׁים מֵהָעָם לֶאֱסוֹף וְלֹא מָצְאוּ.
ס
כח וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל־מֹשֶׁ֑ה עַד־אָ֙נָה֙ מֵֽאַנְתֶּ֔ם לִשְׁמֹ֥ר מִצְוֺתַ֖י וְתוֹרֹתָֽי׃ ה' אָמַר לְמֹשֶׁה: עַד מָתַי אַתֶּם מְסָרְבִים לִשְׁמוֹר אֶת מִצְווֹתַי וְהוֹרָאוֹתַי?
כט רְא֗וּ כִּֽי־יְהֹוָה֮ נָתַ֣ן לָכֶ֣ם הַשַּׁבָּת֒ עַל־כֵּ֠ן ה֣וּא נֹתֵ֥ן לָכֶ֛ם בַּיּ֥וֹם הַשִּׁשִּׁ֖י לֶ֣חֶם יוֹמָ֑יִם שְׁב֣וּ ׀ אִ֣ישׁ תַּחְתָּ֗יו אַל־יֵ֥צֵא אִ֛ישׁ מִמְּקֹמ֖וֹ בַּיּ֥וֹם הַשְּׁבִיעִֽי׃ רְאוּ שֶׁה' נָתַן לָכֶם אֶת הַשַּׁבָּת לָכֵן הוּא נוֹתֵן לָכֶם בַּיּוֹם הַשִּׁשִּׁי לֶחֶם בְּכַמּוּת שֶׁל יוֹמַיִם, שְׁבוּ אִישׁ בְּבֵיתוֹ – שֶׁלֹּא יֵצֵא אִישׁ מִמְּקוֹמוֹ בַּיּוֹם הַשְּׁבִיעִי.
ל וַיִּשְׁבְּת֥וּ הָעָ֖ם בַּיּ֥וֹם הַשְּׁבִעִֽי׃ הָעָם שָׁבְתוּ בַּיּוֹם הַשְּׁבִיעִי.
לא וַיִּקְרְא֧וּ בֵֽית־יִשְׂרָאֵ֛ל אֶת־שְׁמ֖וֹ מָ֑ן וְה֗וּא כְּזֶ֤רַע גַּד֙ לָבָ֔ן וְטַעְמ֖וֹ כְּצַפִּיחִ֥ת בִּדְבָֽשׁ׃ בֵּית יִשְׂרָאֵל קָרְאוּ אֶת שְׁמוֹ מָן, וְהוּא בְּצוּרַת זֶרַע כֻּסְבְּרָה, וְלָבָן, וְטַעְמוֹ כְּבָצֵק מְטֻגָּן בִּדְבָשׁ.
לב וַיֹּ֣אמֶר מֹשֶׁ֗ה זֶ֤ה הַדָּבָר֙ אֲשֶׁ֣ר צִוָּ֣ה יְהֹוָ֔ה מְלֹ֤א הָעֹ֙מֶר֙ מִמֶּ֔נּוּ לְמִשְׁמֶ֖רֶת לְדֹרֹתֵיכֶ֑ם לְמַ֣עַן ׀ יִרְא֣וּ אֶת־הַלֶּ֗חֶם אֲשֶׁ֨ר הֶאֱכַ֤לְתִּי אֶתְכֶם֙ בַּמִּדְבָּ֔ר בְּהוֹצִיאִ֥י אֶתְכֶ֖ם מֵאֶ֥רֶץ מִצְרָֽיִם׃ מֹשֶׁה אָמַר: זֶה הַדָּבָר שֶׁה' צִוָּה – מְלֹא הָעוֹמֶר מִמֶּנּוּ יִהְיֶה לִשְׁמִירָה לְדוֹרוֹת, כְּדֵי שֶׁיִּרְאוּ אֶת הַלֶּחֶם שֶׁהֶאֱכַלְתִּי אֶתְכֶם בַּמִּדְבָּר כְּשֶׁהוֹצֵאתִי אֶתְכֶם מֵאֶרֶץ מִצְרַיִם.
לג וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶֽל־אַהֲרֹ֗ן קַ֚ח צִנְצֶ֣נֶת אַחַ֔ת וְתֶן־שָׁ֥מָּה מְלֹֽא־הָעֹ֖מֶר מָ֑ן וְהַנַּ֤ח אֹתוֹ֙ לִפְנֵ֣י יְהֹוָ֔ה לְמִשְׁמֶ֖רֶת לְדֹרֹתֵיכֶֽם׃ מֹשֶׁה אָמַר לְאַהֲרֹן: קַח צִנְצֶנֶת אַחַת וְשִׂים שָׁם מְלֹא הָעוֹמֶר מָן, וְהַנַּח אוֹתוֹ לִפְנֵי ה' לִשְׁמִירָה לַדּוֹרוֹת הַבָּאִים.
לד כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶל־מֹשֶׁ֑ה וַיַּנִּיחֵ֧הוּ אַהֲרֹ֛ן לִפְנֵ֥י הָעֵדֻ֖ת לְמִשְׁמָֽרֶת׃ כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה – אַהֲרֹן הִנִּיחַ אוֹתוֹ לִפְנֵי לוּחוֹת הַבְּרִית, לְמִשְׁמֶרֶת.
לה וּבְנֵ֣י יִשְׂרָאֵ֗ל אָֽכְל֤וּ אֶת־הַמָּן֙ אַרְבָּעִ֣ים שָׁנָ֔ה עַד־בֹּאָ֖ם אֶל־אֶ֣רֶץ נוֹשָׁ֑בֶת אֶת־הַמָּן֙ אָֽכְל֔וּ עַד־בֹּאָ֕ם אֶל־קְצֵ֖ה אֶ֥רֶץ כְּנָֽעַן׃ וּבְנֵי יִשְׂרָאֵל אָכְלוּ אֶת הַמָּן אַרְבָּעִים שָׁנָה עַד שֶׁבָּאוּ לְאֶרֶץ מְיֻשֶּׁבֶת – אֶת הַמָּן אָכְלוּ עַד שֶׁבָּאוּ לִקְצֵה אֶרֶץ כְּנַעַן.
לו וְהָעֹ֕מֶר עֲשִׂרִ֥ית הָאֵיפָ֖ה הֽוּא׃ וְהָעוֹמֶר הוּא נֶפַח שֶׁל עֲשִׂירִית אֵיפָה (כִּ2.16 לִיטֶר).

שביעי[עריכה]

פ פָּרָשַׁת מֵימֵי הַצּוּר
יז א וַ֠יִּסְע֠וּ כׇּל־עֲדַ֨ת בְּנֵֽי־יִשְׂרָאֵ֧ל מִמִּדְבַּר־סִ֛ין לְמַסְעֵיהֶ֖ם עַל־פִּ֣י יְהֹוָ֑ה וַֽיַּחֲנוּ֙ בִּרְפִידִ֔ים וְאֵ֥ין מַ֖יִם לִשְׁתֹּ֥ת הָעָֽם׃ כָּל צִבּוּר בְּנֵי יִשְׂרָאֵל נָסְעוּ מִמִּדְבַּר סִין לְמַסְעוֹתֵיהֶם עַל פִּי ה', הֵם חָנוּ בִּרְפִידִים וְלֹא הָיָה מַיִם לָעָם לִשְׁתּוֹת.
ב וַיָּ֤רֶב הָעָם֙ עִם־מֹשֶׁ֔ה וַיֹּ֣אמְר֔וּ תְּנוּ־לָ֥נוּ מַ֖יִם וְנִשְׁתֶּ֑ה וַיֹּ֤אמֶר לָהֶם֙ מֹשֶׁ֔ה מַה־תְּרִיבוּן֙ עִמָּדִ֔י מַה־תְּנַסּ֖וּן אֶת־יְהֹוָֽה׃ הָעָם רָבוּ עִם מֹשֶׁה, וְאָמְרוּ: תְּנוּ לָנוּ מַיִם וְנִשְׁתֶּה! אָמַר לָהֶם מֹשֶׁה: לָמָּה תָּרִיבוּ אִתִּי? לָמָּה תְּנַסּוּ אֶת ה'?
ג וַיִּצְמָ֨א שָׁ֤ם הָעָם֙ לַמַּ֔יִם וַיָּ֥לֶן הָעָ֖ם עַל־מֹשֶׁ֑ה וַיֹּ֗אמֶר לָ֤מָּה זֶּה֙ הֶעֱלִיתָ֣נוּ מִמִּצְרַ֔יִם לְהָמִ֥ית אֹתִ֛י וְאֶת־בָּנַ֥י וְאֶת־מִקְנַ֖י בַּצָּמָֽא׃ הָעָם צָמְאוּ שָׁם לְמַיִם, הָעָם הִתְלוֹנְנוּ עַל מֹשֶׁה, וְאָמְרוּ: לָמָּה הֶעֱלִיתָ אוֹתָנוּ מִמִּצְרַיִם לַהֲרוֹג אוֹתָנוּ וְאֶת בָּנֵינוּ וְאֶת בְּהֵמוֹתֵנוּ בַּצָּמָא?
ד וַיִּצְעַ֤ק מֹשֶׁה֙ אֶל־יְהֹוָ֣ה לֵאמֹ֔ר מָ֥ה אֶעֱשֶׂ֖ה לָעָ֣ם הַזֶּ֑ה ע֥וֹד מְעַ֖ט וּסְקָלֻֽנִי׃ מֹשֶׁה צָעַק אֶל ה' וְאָמַר: מָה אֶעֱשֶׂה לָעָם הַזֶּה? עוֹד מְעַט יִסְקְלוּ אוֹתִי!
ה וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה עֲבֹר֙ לִפְנֵ֣י הָעָ֔ם וְקַ֥ח אִתְּךָ֖ מִזִּקְנֵ֣י יִשְׂרָאֵ֑ל וּמַטְּךָ֗ אֲשֶׁ֨ר הִכִּ֤יתָ בּוֹ֙ אֶת־הַיְאֹ֔ר קַ֥ח בְּיָדְךָ֖ וְהָלָֽכְתָּ׃ ה' אָמַר לְמֹשֶׁה: עֲבוֹר לִפְנֵי הָעָם וְקַח אִתְּךָ מִזִּקְנֵי יִשְׂרָאֵל, וְקַח בְּיָדְךָ אֶת הַמַּקֵּל שֶׁלְּךָ שֶׁהִכִּיתָ בּוֹ אֶת הַיְּאוֹר וְלֵךְ.
ו הִנְנִ֣י עֹמֵד֩ לְפָנֶ֨יךָ שָּׁ֥ם ׀ עַֽל־הַצּוּר֮ בְּחֹרֵב֒ וְהִכִּ֣יתָ בַצּ֗וּר וְיָצְא֥וּ מִמֶּ֛נּוּ מַ֖יִם וְשָׁתָ֣ה הָעָ֑ם וַיַּ֤עַשׂ כֵּן֙ מֹשֶׁ֔ה לְעֵינֵ֖י זִקְנֵ֥י יִשְׂרָאֵֽל׃ הִנֵּה אֲנִי עוֹמֵד לְפָנֶיךָ שָׁם עַל יַד הַסֶּלַע בְּחוֹרֵב, הַכֶּה בַּסֶּלַע וְיֵצְאוּ מִמֶּנּוּ מַיִם וְהָעָם יִשְׁתֶּה, מֹשֶׁה עָשָׂה כָּךְ לְעֵינֵי זִקְנֵי יִשְׂרָאֵל.
ז וַיִּקְרָא֙ שֵׁ֣ם הַמָּק֔וֹם מַסָּ֖ה וּמְרִיבָ֑ה עַל־רִ֣יב ׀ בְּנֵ֣י יִשְׂרָאֵ֗ל וְעַ֨ל נַסֹּתָ֤ם אֶת־יְהֹוָה֙ לֵאמֹ֔ר הֲיֵ֧שׁ יְהֹוָ֛ה בְּקִרְבֵּ֖נוּ אִם־אָֽיִן׃ הוּא קָרָא אֶת שֵׁם הַמָּקוֹם נִסָּיוֹן וּמְרִיבָה – עַל רִיב בְּנֵי יִשְׂרָאֵל וְעַל שֶׁנִּסּוּ אֶת ה' בְּאָמְרָם: הַאִם ה' נִמְצָא בְּתוֹכֵנוּ אוֹ לֹא.
פ פָּרָשַׁת עֲמָלֵק
ח וַיָּבֹ֖א עֲמָלֵ֑ק וַיִּלָּ֥חֶם עִם־יִשְׂרָאֵ֖ל בִּרְפִידִֽם׃ עֲמָלֵק בָּא וְנִלְחַם עִם יִשְׂרָאֵל בִּרְפִידִים.
ט וַיֹּ֨אמֶר מֹשֶׁ֤ה אֶל־יְהוֹשֻׁ֙עַ֙ בְּחַר־לָ֣נוּ אֲנָשִׁ֔ים וְצֵ֖א הִלָּחֵ֣ם בַּעֲמָלֵ֑ק מָחָ֗ר אָנֹכִ֤י נִצָּב֙ עַל־רֹ֣אשׁ הַגִּבְעָ֔ה וּמַטֵּ֥ה הָאֱלֹהִ֖ים בְּיָדִֽי׃ מֹשֶׁה אָמַר לִיהוֹשֻׁעַ: בְּחַר לָנוּ אֲנָשִׁים וְצֵא הִלָּחֵם בַּעֲמָלֵק, מָחָר אֲנִי עוֹמֵד עַל רֹאשׁ הַגִּבְעָה וּמַקֵּל הָאֱלֹהִים בְּיָדִי.
י וַיַּ֣עַשׂ יְהוֹשֻׁ֗עַ כַּאֲשֶׁ֤ר אָֽמַר־לוֹ֙ מֹשֶׁ֔ה לְהִלָּחֵ֖ם בַּעֲמָלֵ֑ק וּמֹשֶׁה֙ אַהֲרֹ֣ן וְח֔וּר עָל֖וּ רֹ֥אשׁ הַגִּבְעָֽה׃ יְהוֹשֻׁעַ עָשָׂה כְּמוֹ שֶׁאָמַר לוֹ מֹשֶׁה – לְהִלָּחֵם בַּעֲמָלֵק, וּמֹשֶׁה אַהֲרֹן וְחוּר עָלוּ לְרֹאשׁ הַגִּבְעָה.
יא וְהָיָ֗ה כַּאֲשֶׁ֨ר יָרִ֥ים מֹשֶׁ֛ה יָד֖וֹ וְגָבַ֣ר יִשְׂרָאֵ֑ל וְכַאֲשֶׁ֥ר יָנִ֛יחַ יָד֖וֹ וְגָבַ֥ר עֲמָלֵֽק׃ כְּשֶׁמֹּשֶׁה הָיָה מֵרִים אֶת יָדוֹ הָיָה יִשְׂרָאֵל גּוֹבֵר, וּכְשֶׁהָיָה מֵנִיחַ אֶת יָדוֹ הָיָה עֲמָלֵק גּוֹבֵר.
יב וִידֵ֤י מֹשֶׁה֙ כְּבֵדִ֔ים וַיִּקְחוּ־אֶ֛בֶן וַיָּשִׂ֥ימוּ תַחְתָּ֖יו וַיֵּ֣שֶׁב עָלֶ֑יהָ וְאַהֲרֹ֨ן וְח֜וּר תָּֽמְכ֣וּ בְיָדָ֗יו מִזֶּ֤ה אֶחָד֙ וּמִזֶּ֣ה אֶחָ֔ד וַיְהִ֥י יָדָ֛יו אֱמוּנָ֖ה עַד־בֹּ֥א הַשָּֽׁמֶשׁ׃ יְדֵי מֹשֶׁה הִתְעַיְּפוּ, לָקְחוּ אֶבֶן וְשָׂמוּ תַּחְתָּיו וְהוּא יָשַׁב עָלֶיהָ, וְאַהֲרֹן וְחוּר תָּמְכוּ בְּיָדָיו אֶחָד מִצַּד זֶה וְאֶחָד מִצַּד זֶה, וְיָדָיו הָיוּ פְּרוּשׂוֹת בִּתְפִילָה עַד לִשְׁקִיעַת הַשֶּׁמֶשׁ.
יג וַיַּחֲלֹ֧שׁ יְהוֹשֻׁ֛עַ אֶת־עֲמָלֵ֥ק וְאֶת־עַמּ֖וֹ לְפִי־חָֽרֶב׃ יְהוֹשֻׁעַ הִכְנִיעַ אֶת עֲמָלֵק וְאֶת עַמּוֹ בְּלַהַב הַחֶרֶב.

מפטיר[עריכה]

פ
יד וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה כְּתֹ֨ב זֹ֤את זִכָּרוֹן֙ בַּסֵּ֔פֶר וְשִׂ֖ים בְּאׇזְנֵ֣י יְהוֹשֻׁ֑עַ כִּֽי־מָחֹ֤ה אֶמְחֶה֙ אֶת־זֵ֣כֶר עֲמָלֵ֔ק מִתַּ֖חַת הַשָּׁמָֽיִם׃ ה' אָמַר לְמֹשֶׁה: כְּתוֹב זֹאת לְזִכָּרוֹן בְּסֵפֶר וְתַשְׁמִיעַ בְּאָזְנֵי יְהוֹשֻׁעַ, שֶׁאֶמְחֶה אֶת זֵכֶר עֲמָלֵק מִתַּחַת הַשָּׁמַיִם.
טו וַיִּ֥בֶן מֹשֶׁ֖ה מִזְבֵּ֑חַ וַיִּקְרָ֥א שְׁמ֖וֹ יְהֹוָ֥ה ׀ נִסִּֽי׃ מֹשֶׁה בָּנָה מִזְבֵּחַ וְקָרָא אֶת שְׁמוֹ: 'ה' עָשָׂה לִי נֵס'.
טז וַיֹּ֗אמֶר כִּֽי־יָד֙ עַל־כֵּ֣ס יָ֔הּ‏[1] מִלְחָמָ֥ה לַיהֹוָ֖ה בַּֽעֲמָלֵ֑ק מִדֹּ֖ר דֹּֽר׃ אָמַר: יָדוֹ שֶׁל ה' מוּרֶמֶת בִּשְׁבוּעָה בְּכִסְאוֹ, שֶׁתִּהְיֶה מִלְחָמָה לַה' בַּעֲמָלֵק, בְּכָל דּוֹר וָדוֹר.
