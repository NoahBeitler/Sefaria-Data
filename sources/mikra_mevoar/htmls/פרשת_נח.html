ראשון[עריכה]

פ פָּרָשַׁת צִוּוּי הַתֵּיבָה
ו ט אֵ֚לֶּה תּוֹלְדֹ֣ת נֹ֔חַ נֹ֗חַ אִ֥ישׁ צַדִּ֛יק תָּמִ֥ים הָיָ֖ה בְּדֹֽרֹתָ֑יו אֶת־הָֽאֱלֹהִ֖ים הִֽתְהַלֶּךְ־נֹֽחַ׃ אֵלֶּה צֶאֱצָאֵי נֹחַ, נֹחַ הָיָה אִישׁ צַדִּיק תָּמִים בְּדוֹרוֹ, נֹחַ הִתְהַלֵּךְ בְּדַרְכֵי הָאֱלֹהִים.
י וַיּ֥וֹלֶד נֹ֖חַ שְׁלֹשָׁ֣ה בָנִ֑ים אֶת־שֵׁ֖ם אֶת־חָ֥ם וְאֶת־יָֽפֶת׃ נֹחַ הוֹלִיד שְׁלוֹשָׁה בָּנִים – אֶת שֵׁם אֶת חָם וְאֶת יֶפֶת.
יא וַתִּשָּׁחֵ֥ת הָאָ֖רֶץ לִפְנֵ֣י הָֽאֱלֹהִ֑ים וַתִּמָּלֵ֥א הָאָ֖רֶץ חָמָֽס׃ הָאָרֶץ הִתְקַלְקְלָה בְּעֵינֵי הָאֱלֹהִים, הָאָרֶץ הִתְמַלְּאָה בְּגַנָּבִים.
יב וַיַּ֧רְא אֱלֹהִ֛ים אֶת־הָאָ֖רֶץ וְהִנֵּ֣ה נִשְׁחָ֑תָה כִּֽי־הִשְׁחִ֧ית כׇּל־בָּשָׂ֛ר אֶת־דַּרְכּ֖וֹ עַל־הָאָֽרֶץ׃ אֱלֹהִים רָאָה אֶת הָאָרֶץ וְהִנֵּה הָפְכָה לְמוּשְׁחֶתֶת שֶׁכָּל אִישׁ בָּשָׂר וָדָם שֶׁעַל הָאָרֶץ הִשְׁחִית אֶת הִתְנַהֲגוּתוֹ.
ס
יג וַיֹּ֨אמֶר אֱלֹהִ֜ים לְנֹ֗חַ קֵ֤ץ כׇּל־בָּשָׂר֙ בָּ֣א לְפָנַ֔י כִּֽי־מָלְאָ֥ה הָאָ֛רֶץ חָמָ֖ס מִפְּנֵיהֶ֑ם וְהִנְנִ֥י מַשְׁחִיתָ֖ם אֶת־הָאָֽרֶץ׃ אֱלֹהִים אָמַר לְנֹחַ: הֶחְלַטְתִּי שֶׁהִגִּיעַ הַסּוֹף שֶׁל כָּל בָּשָׂר וָדָם כִּי בִּגְלָלָם הָעוֹלָם הִתְמַלֵּא בִּשְׁחִיתוּת, וַאֲנִי עוֹמֵד לְהַשְׁמִיד אוֹתָם וְאֶת הָעוֹלָם.
יד עֲשֵׂ֤ה לְךָ֙ תֵּבַ֣ת עֲצֵי־גֹ֔פֶר קִנִּ֖ים תַּֽעֲשֶׂ֣ה אֶת־הַתֵּבָ֑ה וְכָֽפַרְתָּ֥ אֹתָ֛הּ מִבַּ֥יִת וּמִח֖וּץ בַּכֹּֽפֶר׃ עֲשֵׂה לְךָ תֵּבָה מֵעֵץ אֶדֶר, תַּעֲשֶׂה אֶת הַתֵּבָה תָּאִים תָּאִים, וּתְצַפֶּה אוֹתָהּ מִבִּפְנִים וּמִבַּחוּץ בְּזֶפֶת.
טו וְזֶ֕ה אֲשֶׁ֥ר תַּֽעֲשֶׂ֖ה אֹתָ֑הּ שְׁלֹ֧שׁ מֵא֣וֹת אַמָּ֗ה אֹ֚רֶךְ הַתֵּבָ֔ה חֲמִשִּׁ֤ים אַמָּה֙ רׇחְבָּ֔הּ וּשְׁלֹשִׁ֥ים אַמָּ֖ה קוֹמָתָֽהּ׃ וְכָךְ תַּעֲשֶׂה אוֹתָהּ: שְׁלוֹשׁ מֵאוֹת אַמָּה אוֹרֶךְ הַתֵּבָה (כְּ140 מֶטֶר), חֲמִשִּׁים אַמָּה רָחְבָּהּ (כְּ⅓23 מֶטֶר), וּשְׁלוֹשִׁים אַמָּה גּוֹבְהָהּ (כְּ14 מֶטֶר).
טז צֹ֣הַר ׀ תַּעֲשֶׂ֣ה לַתֵּבָ֗ה וְאֶל־אַמָּה֙ תְּכַלֶּ֣נָּה מִלְמַ֔עְלָה וּפֶ֥תַח הַתֵּבָ֖ה בְּצִדָּ֣הּ תָּשִׂ֑ים תַּחְתִּיִּ֛ם שְׁנִיִּ֥ם וּשְׁלִשִׁ֖ים תַּֽעֲשֶֽׂהָ׃ תַּעֲשֶׂה חַלּוֹן לַתֵּבָה, וְהַקָּצֶה שֶׁלָּהּ לְמַעְלָה יִסְתַּיֵּם בְּרוֹחַב אַמָּה (כְּ47 ס"מ), וְאֶת פֶּתַח הַתֵּבָה תַּעֲשֶׂה בְּצִדָּהּ, תַּעֲשֶׂה אוֹתָהּ שָׁלֹשׁ קוֹמוֹת – תַּחְתּוֹנָה, שְׁנִיָּה, וּשְׁלִישִׁית.
יז וַאֲנִ֗י הִנְנִי֩ מֵבִ֨יא אֶת־הַמַּבּ֥וּל מַ֙יִם֙ עַל־הָאָ֔רֶץ לְשַׁחֵ֣ת כׇּל־בָּשָׂ֗ר אֲשֶׁר־בּוֹ֙ ר֣וּחַ חַיִּ֔ים מִתַּ֖חַת הַשָּׁמָ֑יִם כֹּ֥ל אֲשֶׁר־בָּאָ֖רֶץ יִגְוָֽע׃ וְהִנֵּה אֲנִי מֵבִיא אֶת הַמַּבּוּל מַיִם עַל הָאָרֶץ, לְהַשְׁמִיד כָּל בָּשָׂר וָדָם שֶׁיֵּשׁ בּוֹ רוּחַ חַיִּים, שֶׁמִּתַּחַת לַשָּׁמַיִם – כָּל מִי שֶׁבָּאָרֶץ יָמוּת.
יח וַהֲקִמֹתִ֥י אֶת־בְּרִיתִ֖י אִתָּ֑ךְ וּבָאתָ֙ אֶל־הַתֵּבָ֔ה אַתָּ֕ה וּבָנֶ֛יךָ וְאִשְׁתְּךָ֥ וּנְשֵֽׁי־בָנֶ֖יךָ אִתָּֽךְ׃ וַאֲקַיֵּם אֶת הִתְחַיְּבוּתִי לְךָ לְהַצִּילְךָ, וְתָבוֹא אֶל הַתֵּבָה אַתָּה וּבָנֶיךָ וְאִשְׁתְּךָ וּנְשֵׁי בָּנֶיךָ אִתְּךָ.
יט וּמִכׇּל־הָ֠חַ֠י מִֽכׇּל־בָּשָׂ֞ר שְׁנַ֧יִם מִכֹּ֛ל תָּבִ֥יא אֶל־הַתֵּבָ֖ה לְהַחֲיֹ֣ת אִתָּ֑ךְ זָכָ֥ר וּנְקֵבָ֖ה יִֽהְיֽוּ׃ וּמִכָּל בַּעַל חַי – מִכָּל בָּשָׂר וָדָם תָּבִיא אֶל הַתֵּבָה שְׁנַיִם מִכָּל מִין שֶׁיִּחְיוּ אִתְּךָ, וְהֵם יִהְיוּ זָכָר וּנְקֵבָה.
כ מֵהָע֣וֹף לְמִינֵ֗הוּ וּמִן־הַבְּהֵמָה֙ לְמִינָ֔הּ מִכֹּ֛ל רֶ֥מֶשׂ הָֽאֲדָמָ֖ה לְמִינֵ֑הוּ שְׁנַ֧יִם מִכֹּ֛ל יָבֹ֥אוּ אֵלֶ֖יךָ לְהַֽחֲיֽוֹת׃ מֵהָעוֹף לְמִינָיו וּמֵהַבְּהֵמָה לְמִינֶיהָ מִכָּל הַחֲרָקִים שֶׁעַל הָאֲדָמָה לְמִינֵיהֶם שְׁנַיִם מִכָּל מִין יָבוֹאוּ אֵלֶיךָ שֶׁיִּחְיוּ.
כא וְאַתָּ֣ה קַח־לְךָ֗ מִכׇּל־מַֽאֲכָל֙ אֲשֶׁ֣ר יֵֽאָכֵ֔ל וְאָסַפְתָּ֖ אֵלֶ֑יךָ וְהָיָ֥ה לְךָ֛ וְלָהֶ֖ם לְאׇכְלָֽה׃ וְאַתָּה קַח לְךָ מִכָּל מַאֲכָל שֶׁנֶּאֱכָל וְתֶאֱסוֹף אוֹתוֹ אֵלֶיךָ, וְיִהְיֶה לְךָ וְלָהֶם לַאֲכִילָה.
כב וַיַּ֖עַשׂ נֹ֑חַ כְּ֠כֹ֠ל אֲשֶׁ֨ר צִוָּ֥ה אֹת֛וֹ אֱלֹהִ֖ים כֵּ֥ן עָשָֽׂה׃ נֹחַ עָשָׂה, כְּכָל מַה שֶּׁאֱלֹהִים צִוָּה אוֹתוֹ כָּךְ הוּא עָשָׂה.

שני[עריכה]

פָּרָשַׁת הַכְּנִיסָה לַתֵּיבָה
ז א וַיֹּ֤אמֶר יְהֹוָה֙ לְנֹ֔חַ בֹּֽא־אַתָּ֥ה וְכׇל־בֵּיתְךָ֖ אֶל־הַתֵּבָ֑ה כִּֽי־אֹתְךָ֥ רָאִ֛יתִי צַדִּ֥יק לְפָנַ֖י בַּדּ֥וֹר הַזֶּֽה׃ ה' אָמַר לְנֹחַ: בֹּא אַתָּה וְכָל מִשְׁפַּחְתְּךָ אֶל הַתֵּבָה, כִּי רַק אוֹתְךָ רָאִיתִי כְּזַכַּאי לְפָנַי בַּדּוֹר הַזֶּה.
ב מִכֹּ֣ל ׀ הַבְּהֵמָ֣ה הַטְּהוֹרָ֗ה תִּֽקַּח־לְךָ֛ שִׁבְעָ֥ה שִׁבְעָ֖ה אִ֣ישׁ וְאִשְׁתּ֑וֹ וּמִן־הַבְּהֵמָ֡ה אֲ֠שֶׁ֠ר לֹ֣א טְהֹרָ֥ה הִ֛וא שְׁנַ֖יִם אִ֥ישׁ וְאִשְׁתּֽוֹ׃ קַח לְךָ מִכָּל הַבְּהֵמוֹת הַטְּהוֹרוֹת שִׁבְעָה שִׁבְעָה זוּגוֹת – אִישׁ וְאִשְׁתּוֹ, וּמֵהַבְּהֵמוֹת שֶׁהֵן לֹא טְהוֹרוֹת – שְׁנַיִם, אִישׁ וְאִשְׁתּוֹ.
ג גַּ֣ם מֵע֧וֹף הַשָּׁמַ֛יִם שִׁבְעָ֥ה שִׁבְעָ֖ה זָכָ֣ר וּנְקֵבָ֑ה לְחַיּ֥וֹת זֶ֖רַע עַל־פְּנֵ֥י כׇל־הָאָֽרֶץ׃ גַּם מֵעוֹף הַשָּׁמַיִם הַטָּהוֹר תִּקַּח שִׁבְעָה שִׁבְעָה זוּגוֹת שֶׁל זָכָר וּנְקֵבָה, כְּדֵי לְהַחֲיוֹת צֶאֱצָאִים עַל פְּנֵי כָּל הָאָרֶץ.
ד כִּי֩ לְיָמִ֨ים ע֜וֹד שִׁבְעָ֗ה אָֽנֹכִי֙ מַמְטִ֣יר עַל־הָאָ֔רֶץ אַרְבָּעִ֣ים י֔וֹם וְאַרְבָּעִ֖ים לָ֑יְלָה וּמָחִ֗יתִי אֶֽת־כׇּל־הַיְקוּם֙ אֲשֶׁ֣ר עָשִׂ֔יתִי מֵעַ֖ל פְּנֵ֥י הָֽאֲדָמָֽה׃ כִּי עוֹד שִׁבְעָה יָמִים אֲנִי מַמְטִיר עַל הָאָרֶץ אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה, וְאֶמְחֶה מֵעַל פְּנֵי הָאֲדָמָה אֶת כָּל בַּעֲלֵי הַחַיִּים שֶׁעָשִׂיתִי.
ה וַיַּ֖עַשׂ נֹ֑חַ כְּכֹ֥ל אֲשֶׁר־צִוָּ֖הוּ יְהֹוָֽה׃ נֹחַ עָשָׂה אֶת כָּל מַה שֶּׁה' צִוָּה אוֹתוֹ.
ו וְנֹ֕חַ בֶּן־שֵׁ֥שׁ מֵא֖וֹת שָׁנָ֑ה וְהַמַּבּ֣וּל הָיָ֔ה מַ֖יִם עַל־הָאָֽרֶץ׃ וְנֹחַ הָיָה בֶּן שֵׁשׁ מֵאוֹת שָׁנָה וְהָיָה מַבּוּל שֶׁל מַיִם עַל הָאָרֶץ.
ז וַיָּ֣בֹא נֹ֗חַ וּ֠בָנָ֠יו וְאִשְׁתּ֧וֹ וּנְשֵֽׁי־בָנָ֛יו אִתּ֖וֹ אֶל־הַתֵּבָ֑ה מִפְּנֵ֖י מֵ֥י הַמַּבּֽוּל׃ נֹחַ וּבָנָיו וְאִשְׁתּוֹ וּנְשֵׁי בָּנָיו אִתּוֹ בָּאוּ אֶל הַתֵּבָה מִפְּנֵי מֵי הַמַּבּוּל.
ח מִן־הַבְּהֵמָה֙ הַטְּהוֹרָ֔ה וּמִ֨ן־הַבְּהֵמָ֔ה אֲשֶׁ֥ר אֵינֶ֖נָּה טְהֹרָ֑ה וּמִ֨ן־הָע֔וֹף וְכֹ֥ל אֲשֶׁר־רֹמֵ֖שׂ עַל־הָֽאֲדָמָֽה׃ מֵהַבְּהֵמָה הַטְּהוֹרָה וּמֵהַבְּהֵמָה שֶׁאֵינֶנָּה טְהוֹרָה וּמֵהָעוֹף וְכָל הַחֲרָקִים שֶׁזּוֹחֲלִים עַל הָאֲדָמָה,
ט שְׁנַ֨יִם שְׁנַ֜יִם בָּ֧אוּ אֶל־נֹ֛חַ אֶל־הַתֵּבָ֖ה זָכָ֣ר וּנְקֵבָ֑ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה אֱלֹהִ֖ים אֶת־נֹֽחַ׃ בָּאוּ אֶל נֹחַ שְׁנַיִם שְׁנַיִם אֶל הַתֵּבָה – זָכָר וּנְקֵבָה, כְּמוֹ שֶׁאֱלֹהִים צִוָּה אֶת נֹחַ.
י וַיְהִ֖י לְשִׁבְעַ֣ת הַיָּמִ֑ים וּמֵ֣י הַמַּבּ֔וּל הָי֖וּ עַל־הָאָֽרֶץ׃ לְאַחַר שִׁבְעַת הַיָּמִים מֵי הַמַּבּוּל הָיוּ עַל הָאָרֶץ.
פָּרָשַׁת הַמַּבּוּל
יא בִּשְׁנַ֨ת שֵׁשׁ־מֵא֤וֹת שָׁנָה֙ לְחַיֵּי־נֹ֔חַ בַּחֹ֙דֶשׁ֙ הַשֵּׁנִ֔י בְּשִׁבְעָֽה־עָשָׂ֥ר י֖וֹם לַחֹ֑דֶשׁ בַּיּ֣וֹם הַזֶּ֗ה נִבְקְעוּ֙ כׇּֽל־מַעְיְנֹת֙‏[1] תְּה֣וֹם רַבָּ֔ה וַאֲרֻבֹּ֥ת הַשָּׁמַ֖יִם נִפְתָּֽחוּ׃ בִּשְׁנַת שֵׁשׁ מֵאוֹת שָׁנָה לְחַיֵּי נֹחַ בַּחוֹדֶשׁ הַשֵּׁנִי בְּשִׁבְעָה עָשָׂר יוֹם לַחוֹדֶשׁ, בְּאוֹתוֹ הַיּוֹם נִבְקְעוּ כָּל מַעַיְנוֹת הַתְּהוֹם הַגָּדוֹל וְנִפְתְּחוּ אֲרֻבּוֹת הַשָּׁמַיִם.
יב וַיְהִ֥י הַגֶּ֖שֶׁם עַל־הָאָ֑רֶץ אַרְבָּעִ֣ים י֔וֹם וְאַרְבָּעִ֖ים לָֽיְלָה׃ יָרַד גֶּשֶׁם עַל הָאָרֶץ אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה.
יג בְּעֶ֨צֶם הַיּ֤וֹם הַזֶּה֙ בָּ֣א נֹ֔חַ וְשֵׁם־וְחָ֥ם וָיֶ֖פֶת בְּנֵי־נֹ֑חַ וְאֵ֣שֶׁת נֹ֗חַ וּשְׁלֹ֧שֶׁת נְשֵֽׁי־בָנָ֛יו אִתָּ֖ם אֶל־הַתֵּבָֽה׃ בַּיּוֹם הַזֶּה עַצְמוֹ בָּאוּ נֹחַ, וְשֵׁם חָם וְיֶפֶת בְּנֵי נֹחַ, וְאֵשֶׁת נֹחַ וּשְׁלוֹשֶׁת נְשֵׁי בָּנָיו אִתָּם, אֶל הַתֵּבָה.
יד הֵ֜מָּה וְכׇל־הַֽחַיָּ֣ה לְמִינָ֗הּ וְכׇל־הַבְּהֵמָה֙ לְמִינָ֔הּ וְכׇל־הָרֶ֛מֶשׂ הָרֹמֵ֥שׂ עַל־הָאָ֖רֶץ לְמִינֵ֑הוּ וְכׇל־הָע֣וֹף לְמִינֵ֔הוּ כֹּ֖ל צִפּ֥וֹר כׇּל־כָּנָֽף׃ הֵם וְכָל הַחַיּוֹת לְמִינֵיהֶן וְכָל הַבְּהֵמוֹת לְמִינֵיהֶן וְכָל הַחֲרָקִים שֶׁזּוֹחֲלִים עַל הָאֲדָמָה לְמִינֵיהֶם וְכָל הָעוֹפוֹת לְמִינֵיהֶם, כָּל צִפּוֹר – כָּל בַּעַל כָּנָף.
טו וַיָּבֹ֥אוּ אֶל־נֹ֖חַ אֶל־הַתֵּבָ֑ה שְׁנַ֤יִם שְׁנַ֙יִם֙ מִכׇּל־הַבָּשָׂ֔ר אֲשֶׁר־בּ֖וֹ ר֥וּחַ חַיִּֽים׃ הֵם בָּאוּ אֶל נֹחַ אֶל הַתֵּבָה שְׁנַיִם שְׁנַיִם מִכָּל בָּשָׂר וָדָם שֶׁבּוֹ רוּחַ חַיִּים.
טז וְהַבָּאִ֗ים זָכָ֨ר וּנְקֵבָ֤ה מִכׇּל־בָּשָׂר֙ בָּ֔אוּ כַּֽאֲשֶׁ֛ר צִוָּ֥ה אֹת֖וֹ אֱלֹהִ֑ים וַיִּסְגֹּ֥ר יְהֹוָ֖ה בַּֽעֲדֽוֹ׃ וְהַבָּאִים בָּאוּ זָכָר וּנְקֵבָה מִכָּל בָּשָׂר וָדָם, כְּמוֹ שֶׁאֱלֹהִים צִוָּה אוֹתוֹ, וַה' סָגַר עָלָיו לִשְׁמוֹר אוֹתוֹ מִמֵּי הַמַּבּוּל.

שלישי[עריכה]

יז וַֽיְהִ֧י הַמַּבּ֛וּל אַרְבָּעִ֥ים י֖וֹם עַל־הָאָ֑רֶץ וַיִּרְבּ֣וּ הַמַּ֗יִם וַיִּשְׂאוּ֙ אֶת־הַתֵּבָ֔ה וַתָּ֖רׇם מֵעַ֥ל הָאָֽרֶץ׃ הַמַּבּוּל הָיָה אַרְבָּעִים יוֹם עַל הָאָרֶץ, הַמַּיִם הִתְרַבּוּ וְנָשְׂאוּ אֶת הַתֵּבָה וְהִיא הִתְרוֹמְמָה מֵעַל הָאָרֶץ.
יח וַיִּגְבְּר֥וּ הַמַּ֛יִם וַיִּרְבּ֥וּ מְאֹ֖ד עַל־הָאָ֑רֶץ וַתֵּ֥לֶךְ הַתֵּבָ֖ה עַל־פְּנֵ֥י הַמָּֽיִם׃ הַמַּיִם הִתְגַּבְּרוּ וְהִתְרַבּוּ מְאוֹד עַל הָאָרֶץ וְהַתֵּבָה שָׁטָה עַל פְּנֵי הַמַּיִם.
יט וְהַמַּ֗יִם גָּ֥בְר֛וּ מְאֹ֥ד מְאֹ֖ד עַל־הָאָ֑רֶץ וַיְכֻסּ֗וּ כׇּל־הֶֽהָרִים֙ הַגְּבֹהִ֔ים אֲשֶׁר־תַּ֖חַת כׇּל־הַשָּׁמָֽיִם׃ וְהַמַּיִם גָּבְרוּ מְאוֹד מְאוֹד מֵעַל הָאָרֶץ וְכָל הֶהָרִים הַגְּבוֹהִים שֶׁמִּתַּחַת לַשָּׁמַיִם הִתְכַּסּוּ.
כ חֲמֵ֨שׁ עֶשְׂרֵ֤ה אַמָּה֙ מִלְמַ֔עְלָה גָּבְר֖וּ הַמָּ֑יִם וַיְכֻסּ֖וּ הֶהָרִֽים׃ הֶהָרִים הִתְכַּסּוּ וְהַמַּיִם הִתְגַּבְּרוּ עוֹד חֲמֵשׁ עֶשְׂרֵה אַמּוֹת מִלְמַעְלָה.
כא וַיִּגְוַ֞ע כׇּל־בָּשָׂ֣ר ׀ הָרֹמֵ֣שׂ עַל־הָאָ֗רֶץ בָּע֤וֹף וּבַבְּהֵמָה֙ וּבַ֣חַיָּ֔ה וּבְכׇל־הַשֶּׁ֖רֶץ הַשֹּׁרֵ֣ץ עַל־הָאָ֑רֶץ וְכֹ֖ל הָאָדָֽם׃ מֵתוּ כָּל בַּעֲלֵי הַחַיִּים שֶׁזּוֹחֲלִים עַל הָאָרֶץ – בָּעוֹף וּבַבְּהֵמָה וּבַחַיָּה וּבְכָל הַשְּׁרָצִים שֶׁשּׁוֹרְצִים עַל הָאָרֶץ, וְכָל בְּנֵי הָאָדָם מֵתוּ.
כב כֹּ֡ל אֲשֶׁר֩ נִשְׁמַת־ר֨וּחַ חַיִּ֜ים בְּאַפָּ֗יו מִכֹּ֛ל אֲשֶׁ֥ר בֶּחָֽרָבָ֖ה מֵֽתוּ׃ כָּל מִי שֶׁיֵּשׁ בְּאַפּוֹ נְשִׁימָה שֶׁל רוּחַ חַיִּים מִכָּל אֵלֶּה שֶׁחַיִּים בַּיַּבָּשָׁה – מֵתוּ.
כג וַיִּ֜מַח אֶֽת־כׇּל־הַיְק֣וּם ׀ אֲשֶׁ֣ר ׀ עַל־פְּנֵ֣י הָֽאֲדָמָ֗ה מֵאָדָ֤ם עַד־בְּהֵמָה֙ עַד־רֶ֙מֶשׂ֙ וְעַד־ע֣וֹף הַשָּׁמַ֔יִם וַיִּמָּח֖וּ מִן־הָאָ֑רֶץ וַיִּשָּׁ֧אֶר אַךְ־נֹ֛חַ וַֽאֲשֶׁ֥ר אִתּ֖וֹ בַּתֵּבָֽה׃ ה' מָחָה אֶת כָּל הַחַיִּים שֶׁעַל פְּנֵי הָאֲדָמָה מֵאָדָם עַד בְּהֵמָה עַד חָרָק וְעַד עוֹף הַשָּׁמַיִם, וְהֵם נִמְחוּ מֵהָאָרֶץ, וְנִשְׁאֲרוּ רַק נֹחַ וְאֵלֶּה שֶׁאִתּוֹ בַּתֵּבָה.
כד וַיִּגְבְּר֥וּ הַמַּ֖יִם עַל־הָאָ֑רֶץ חֲמִשִּׁ֥ים וּמְאַ֖ת יֽוֹם׃ הַמַּיִם גָּבְרוּ עַל הָאָרֶץ מֵאָה וַחֲמִשִּׁים יוֹם.
פָּרָשַׁת סִיּוּם הַמַּבּוּל
ח א וַיִּזְכֹּ֤ר אֱלֹהִים֙ אֶת־נֹ֔חַ וְאֵ֤ת כׇּל־הַֽחַיָּה֙ וְאֶת־כׇּל־הַבְּהֵמָ֔ה אֲשֶׁ֥ר אִתּ֖וֹ בַּתֵּבָ֑ה וַיַּעֲבֵ֨ר אֱלֹהִ֥ים ר֙וּחַ֙ עַל־הָאָ֔רֶץ וַיָּשֹׁ֖כּוּ הַמָּֽיִם׃ אֱלֹהִים זָכַר אֶת נֹחַ וְאֶת כָּל הַחַיָּה וְאֶת כָּל הַבְּהֵמָה שֶׁאִתּוֹ בַּתֵּבָה, אֱלֹהִים הֶעֱבִיר רוּחַ עַל הָאָרֶץ וְהַמַּיִם נָחוּ.
ב וַיִּסָּֽכְרוּ֙ מַעְיְנֹ֣ת תְּה֔וֹם וַֽאֲרֻבֹּ֖ת הַשָּׁמָ֑יִם וַיִּכָּלֵ֥א הַגֶּ֖שֶׁם מִן־הַשָּׁמָֽיִם׃ מַעַיְנוֹת הַתְּהוֹם וַאֲרֻבּוֹת הַשָּׁמַיִם נִסְגְּרוּ, וְהַגֶּשֶׁם מֵהַשָּׁמַיִם פָּסַק.
ג וַיָּשֻׁ֧בוּ הַמַּ֛יִם מֵעַ֥ל הָאָ֖רֶץ הָל֣וֹךְ וָשׁ֑וֹב וַיַּחְסְר֣וּ הַמַּ֔יִם מִקְצֵ֕ה חֲמִשִּׁ֥ים וּמְאַ֖ת יֽוֹם׃ הַמַּיִם הָלְכוּ וּפָחֲתוּ מֵעַל הָאָרֶץ בְּהַדְרָגָה, הַמַּיִם הִתְחִילוּ לְהֵחָסֵר לְאַחַר מֵאָה וַחֲמִשִּׁים יָמִים.
ד וַתָּ֤נַח הַתֵּבָה֙ בַּחֹ֣דֶשׁ הַשְּׁבִיעִ֔י בְּשִׁבְעָה־עָשָׂ֥ר י֖וֹם לַחֹ֑דֶשׁ עַ֖ל הָרֵ֥י אֲרָרָֽט׃ הַתֵּבָה נָחָה בַּחוֹדֶשׁ הַשְּׁבִיעִי בְּשִׁבְעָה עָשָׂר יוֹם לַחוֹדֶשׁ עַל הָרֵי אֲרָרָט.
ה וְהַמַּ֗יִם הָיוּ֙ הָל֣וֹךְ וְחָס֔וֹר עַ֖ד הַחֹ֣דֶשׁ הָֽעֲשִׂירִ֑י בָּֽעֲשִׂירִי֙ בְּאֶחָ֣ד לַחֹ֔דֶשׁ נִרְא֖וּ רָאשֵׁ֥י הֶֽהָרִֽים׃ וְהַמַּיִם הָלְכוּ וּפָחֲתוּ עַד הַחוֹדֶשׁ הָעֲשִׂירִי, בַּיּוֹם הָרִאשׁוֹן שֶׁל הַחוֹדֶשׁ הָעֲשִׂירִי נִרְאוּ רָאשֵׁי הֶהָרִים.
ו וַיְהִ֕י מִקֵּ֖ץ אַרְבָּעִ֣ים י֑וֹם וַיִּפְתַּ֣ח נֹ֔חַ אֶת־חַלּ֥וֹן הַתֵּבָ֖ה אֲשֶׁ֥ר עָשָֽׂה׃ בְּסוֹף אַרְבָּעִים יוֹם נֹחַ פָּתַח אֶת חַלּוֹן הַתֵּבָה שֶׁעָשָׂה.
ז וַיְשַׁלַּ֖ח אֶת־הָֽעֹרֵ֑ב וַיֵּצֵ֤א יָצוֹא֙ וָשׁ֔וֹב עַד־יְבֹ֥שֶׁת הַמַּ֖יִם מֵעַ֥ל הָאָֽרֶץ׃ הוּא הוֹצִיא אֶת הָעוֹרֵב, וְהוּא יָצָא הָלוֹךְ וָשׁוֹב, עַד שֶׁהַמַּיִם הִתְיַבְּשׁוּ מֵעַל הָאָרֶץ.
ח וַיְשַׁלַּ֥ח אֶת־הַיּוֹנָ֖ה מֵאִתּ֑וֹ לִרְאוֹת֙ הֲקַ֣לּוּ הַמַּ֔יִם מֵעַ֖ל פְּנֵ֥י הָֽאֲדָמָֽה׃ הוּא הוֹצִיא אֶת הַיּוֹנָה מֵאִתּוֹ לִרְאוֹת אִם נִתְמַעֲטוּ הַמַּיִם מֵעַל פְּנֵי הָאֲדָמָה.
ט וְלֹֽא־מָצְאָה֩ הַיּוֹנָ֨ה מָנ֜וֹחַ לְכַף־רַגְלָ֗הּ וַתָּ֤שׇׁב אֵלָיו֙ אֶל־הַתֵּבָ֔ה כִּי־מַ֖יִם עַל־פְּנֵ֣י כׇל־הָאָ֑רֶץ וַיִּשְׁלַ֤ח יָדוֹ֙ וַיִּקָּחֶ֔הָ וַיָּבֵ֥א אֹתָ֛הּ אֵלָ֖יו אֶל־הַתֵּבָֽה׃ הַיּוֹנָה לֹא מָצְאָה מָקוֹם לְהָנִיחַ אֶת כַּף רַגְלָהּ, הִיא חָזְרָה אֵלָיו אֶל הַתֵּבָה כִּי הָיוּ מַיִם עַל פְּנֵי כָּל הָאָרֶץ, הוּא הוֹשִׁיט אֶת יָדוֹ וְלָקַח אוֹתָהּ וְהֵבִיא אוֹתָהּ אֵלָיו לַתֵּבָה.
י וַיָּ֣חֶל ע֔וֹד שִׁבְעַ֥ת יָמִ֖ים אֲחֵרִ֑ים וַיֹּ֛סֶף שַׁלַּ֥ח אֶת־הַיּוֹנָ֖ה מִן־הַתֵּבָֽה׃ הוּא הִמְתִּין עוֹד שִׁבְעָה יָמִים אֲחֵרִים, וְשׁוּב הוֹצִיא אֶת הַיּוֹנָה מֵהַתֵּבָה.
יא וַתָּבֹ֨א אֵלָ֤יו הַיּוֹנָה֙ לְעֵ֣ת עֶ֔רֶב וְהִנֵּ֥ה עֲלֵה־זַ֖יִת טָרָ֣ף בְּפִ֑יהָ וַיֵּ֣דַע נֹ֔חַ כִּי־קַ֥לּוּ הַמַּ֖יִם מֵעַ֥ל הָאָֽרֶץ׃ הַיּוֹנָה בָּאָה אֵלָיו לְעֵת עֶרֶב וְהִנֵּה עֲלֵה זַיִת שֶׁחָטְפָה בְּפִיהָ, נֹחַ יָדַע שֶׁהַמַּיִם הִתְמַעֲטוּ מֵעַל הָאָרֶץ.
יב וַיִּיָּ֣חֶל ע֔וֹד שִׁבְעַ֥ת יָמִ֖ים אֲחֵרִ֑ים וַיְשַׁלַּח֙ אֶת־הַיּוֹנָ֔ה וְלֹֽא־יָסְפָ֥ה שׁוּב־אֵלָ֖יו עֽוֹד׃ הוּא הִמְתִּין עוֹד שִׁבְעָה יָמִים נוֹסָפִים וְהוֹצִיא אֶת הַיּוֹנָה, וְהִיא לֹא הוֹסִיפָה לָשׁוּב אֵלָיו עוֹד.
יג וַ֠יְהִ֠י בְּאַחַ֨ת וְשֵׁשׁ־מֵא֜וֹת שָׁנָ֗ה בָּֽרִאשׁוֹן֙ בְּאֶחָ֣ד לַחֹ֔דֶשׁ חָֽרְב֥וּ הַמַּ֖יִם מֵעַ֣ל הָאָ֑רֶץ וַיָּ֤סַר נֹ֙חַ֙ אֶת־מִכְסֵ֣ה הַתֵּבָ֔ה וַיַּ֕רְא וְהִנֵּ֥ה חָֽרְב֖וּ פְּנֵ֥י הָֽאֲדָמָֽה׃ וּבִשְׁנַת שֵׁשׁ מֵאוֹת וְאַחַת לְחַיֵּי נֹחַ בַּיּוֹם הָרִאשׁוֹן שֶׁל הַחוֹדֶשׁ הָרִאשׁוֹן, הַמַּיִם הִתְיַבְּשׁוּ מֵעַל הָאָרֶץ, נֹחַ הֵסִיר אֶת מִכְסֵה הַתֵּבָה וְרָאָה שֶׁהִנֵּה יָבְשׁוּ פְּנֵי הָאֲדָמָה הָעֶלְיוֹנִים.
יד וּבַחֹ֙דֶשׁ֙ הַשֵּׁנִ֔י בְּשִׁבְעָ֧ה וְעֶשְׂרִ֛ים י֖וֹם לַחֹ֑דֶשׁ יָבְשָׁ֖ה הָאָֽרֶץ׃ וּבַיּוֹם הָעֶשְׂרִים וְשִׁבְעָה שֶׁל הַחוֹדֶשׁ הַשֵּׁנִי יָבְשָׁה הָאָרֶץ לְגַמְרֵי.

רביעי[עריכה]

ס פָּרָשַׁת הַיְּצִיאָה מֵהַתֵּיבָה
טו וַיְדַבֵּ֥ר אֱלֹהִ֖ים אֶל־נֹ֥חַ לֵאמֹֽר׃ אֱלֹהִים דִּבֵּר אֶל נֹחַ וְאָמַר:
טז צֵ֖א מִן־הַתֵּבָ֑ה אַתָּ֕ה וְאִשְׁתְּךָ֛ וּבָנֶ֥יךָ וּנְשֵֽׁי־בָנֶ֖יךָ אִתָּֽךְ׃ צֵא מֵהַתֵּבָה אַתָּה וְאִשְׁתְּךָ וּבָנֶיךָ וּנְשֵׁי בָּנֶיךָ אִתְּךָ.
יז כׇּל־הַחַיָּ֨ה אֲשֶֽׁר־אִתְּךָ֜ מִכׇּל־בָּשָׂ֗ר בָּע֧וֹף וּבַבְּהֵמָ֛ה וּבְכׇל־הָרֶ֛מֶשׂ הָרֹמֵ֥שׂ עַל־הָאָ֖רֶץ הוצא הַיְצֵ֣א אִתָּ֑ךְ וְשָֽׁרְצ֣וּ בָאָ֔רֶץ וּפָר֥וּ וְרָב֖וּ עַל־הָאָֽרֶץ׃ כָּל הַחַיָּה שֶׁאִתְּךָ מִכָּל הַחַי: הָעוֹף וְהַבְּהֵמָה וְכָל הַחֲרָקִים שֶׁזּוֹחֲלִים עַל הָאֲדָמָה – הוֹצֵא אִתְּךָ, וְיִשְׁרְצוּ בָּאָרֶץ וְיוֹלִידוּ וְלָדוֹת וְיִתְרַבּוּ עַל הָאָרֶץ.
יח וַיֵּ֖צֵא־נֹ֑חַ וּבָנָ֛יו וְאִשְׁתּ֥וֹ וּנְשֵֽׁי־בָנָ֖יו אִתּֽוֹ׃ נֹחַ יָצָא, וּבָנָיו וְאִשְׁתּוֹ וּנְשֵׁי בָּנָיו יָצְאוּ אִתּוֹ.
יט כׇּל־הַֽחַיָּ֗ה כׇּל־הָרֶ֙מֶשׂ֙ וְכׇל־הָע֔וֹף כֹּ֖ל רוֹמֵ֣שׂ עַל־הָאָ֑רֶץ לְמִשְׁפְּחֹ֣תֵיהֶ֔ם יָצְא֖וּ מִן־הַתֵּבָֽה׃ כָּל הַחַיּוֹת כָּל הַחֲרָקִים וְכָל הָעוֹפוֹת – כָּל זוֹחֵל עַל הָאָרֶץ, יָצְאוּ מֵהַתֵּבָה לְמִשְׁפְּחוֹתֵיהֶם.
כ וַיִּ֥בֶן נֹ֛חַ מִזְבֵּ֖חַ לַֽיהֹוָ֑ה וַיִּקַּ֞ח מִכֹּ֣ל ׀ הַבְּהֵמָ֣ה הַטְּהֹרָ֗ה וּמִכֹּל֙ הָע֣וֹף הַטָּה֔וֹר וַיַּ֥עַל עֹלֹ֖ת בַּמִּזְבֵּֽחַ׃ נֹחַ בָּנָה מִזְבֵּחַ לַה', וְלָקַח מִכָּל הַבְּהֵמָה הַטְּהוֹרָה וּמִכָּל הָעוֹף הַטָּהוֹר וְהֶעֱלָה עוֹלוֹת בַּמִּזְבֵּחַ.
כא וַיָּ֣רַח יְהֹוָה֮ אֶת־רֵ֣יחַ הַנִּיחֹ֒חַ֒ וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־לִבּ֗וֹ לֹֽא־אֹ֠סִ֠ף לְקַלֵּ֨ל ע֤וֹד אֶת־הָֽאֲדָמָה֙ בַּעֲב֣וּר הָֽאָדָ֔ם כִּ֠י יֵ֣צֶר לֵ֧ב הָאָדָ֛ם רַ֖ע מִנְּעֻרָ֑יו וְלֹֽא־אֹסִ֥ף ע֛וֹד לְהַכּ֥וֹת אֶת־כׇּל־חַ֖י כַּֽאֲשֶׁ֥ר עָשִֽׂיתִי׃ ה' הֵרִיחַ אֶת רֵיחַ הַנִּיחוֹחַ, ה' אָמַר בְּלִבּוֹ: לֹא אוֹסִיף עוֹד לְקַלֵּל אֶת הָאֲדָמָה בִּגְלַל הָאָדָם כִּי כְּבָר מִנְּעֻרָיו יֵשׁ לָאָדָם יֵצֶר-לֵב רַע, וְלֹא אוֹסִיף עוֹד לְהַכּוֹת אֶת כָּל הַחַי כְּמוֹ שֶׁעָשִׂיתִי.
כב עֹ֖ד כׇּל־יְמֵ֣י הָאָ֑רֶץ זֶ֡רַע וְ֠קָצִ֠יר וְקֹ֨ר וָחֹ֜ם וְקַ֧יִץ וָחֹ֛רֶף וְי֥וֹם וָלַ֖יְלָה לֹ֥א יִשְׁבֹּֽתוּ׃ כָּל עוֹנוֹת הָאָרֶץ – עוֹנַת הַזְּרִיעָה וְהַקְּצִירָה וְקוֹר וָחוֹם וְקַיִץ וָחוֹרֶף וְיוֹם וָלַיְלָה, לֹא יִתְבַּטְּלוּ יוֹתֵר.
ט א וַיְבָ֣רֶךְ אֱלֹהִ֔ים אֶת־נֹ֖חַ וְאֶת־בָּנָ֑יו וַיֹּ֧אמֶר לָהֶ֛ם פְּר֥וּ וּרְב֖וּ וּמִלְא֥וּ אֶת־הָאָֽרֶץ׃ אֱלֹהִים בֵּרֵךְ אֶת נֹחַ וְאֶת בָּנָיו וְאָמַר לָהֶם: הוֹלִידוּ יְלָדִים וְהִתְרַבּוּ וּמִלְּאוּ אֶת הָאָרֶץ.
ב וּמוֹרַאֲכֶ֤ם וְחִתְּכֶם֙ יִֽהְיֶ֔ה עַ֚ל כׇּל־חַיַּ֣ת הָאָ֔רֶץ וְעַ֖ל כׇּל־ע֣וֹף הַשָּׁמָ֑יִם בְּכֹל֩ אֲשֶׁ֨ר תִּרְמֹ֧שׂ הָֽאֲדָמָ֛ה וּֽבְכׇל־דְּגֵ֥י הַיָּ֖ם בְּיֶדְכֶ֥ם נִתָּֽנוּ׃ וּכְבוֹדְכֶם וּפַחְדְּכֶם יִהְיֶה עַל כָּל חַיַּת הָאָרֶץ וְעַל כָּל עוֹף הַשָּׁמַיִם, וְעַל הַחֲרָקִים שֶׁעַל הָאֲדָמָה וְעַל כָּל דְּגֵי הַיָּם, כָּל אֵלֶּה נִתְּנוּ בְּיֶדְכֶם.
ג כׇּל־רֶ֙מֶשׂ֙ אֲשֶׁ֣ר הוּא־חַ֔י לָכֶ֥ם יִהְיֶ֖ה לְאׇכְלָ֑ה כְּיֶ֣רֶק עֵ֔שֶׂב נָתַ֥תִּי לָכֶ֖ם אֶת־כֹּֽל׃ כָּל חָרָק חַי יִהְיֶה מֻתָּר לָכֶם לַאֲכִילָה, הִתַּרְתִּי לָכֶם אֶת הַכֹּל שֶׁיִּהְיֶה כְּמוֹ יֶרֶק וְעֵשֶׂב.
ד אַךְ־בָּשָׂ֕ר בְּנַפְשׁ֥וֹ דָמ֖וֹ לֹ֥א תֹאכֵֽלוּ׃ רַק אַל תֹּאכְלוּ בָּשָׂר בְּעוֹדוֹ חַי עִם נַפְשׁוֹ וְדָמוֹ.
ה וְאַ֨ךְ אֶת־דִּמְכֶ֤ם לְנַפְשֹֽׁתֵיכֶם֙ אֶדְרֹ֔שׁ מִיַּ֥ד כׇּל־חַיָּ֖ה אֶדְרְשֶׁ֑נּוּ וּמִיַּ֣ד הָֽאָדָ֗ם מִיַּד֙ אִ֣ישׁ אָחִ֔יו אֶדְרֹ֖שׁ אֶת־נֶ֥פֶשׁ הָֽאָדָֽם׃ לֹא אַעֲנִישׁ אֶתְכֶם עַל הֲרִיגַת בַּעַל חַי, רַק אֶת שְׁפִיכוּת דַּמְכֶם אֶתְבַּע, מִיַּד כָּל חַיָּה שֶׁתַּהֲרוֹג אִישׁ אֶתְבַּע אוֹתוֹ, וּמִיַּד הָאָדָם – מִיַּד אִישׁ שֶׁיַּהֲרוֹג אֶת אָחִיו אֶתְבַּע אֶת נֶפֶשׁ הָאָדָם.
ו שֹׁפֵךְ֙ דַּ֣ם הָֽאָדָ֔ם בָּֽאָדָ֖ם דָּמ֣וֹ יִשָּׁפֵ֑ךְ כִּ֚י בְּצֶ֣לֶם אֱלֹהִ֔ים עָשָׂ֖ה אֶת־הָאָדָֽם׃ שׁוֹפֵךְ דַּם שֶׁל אָדָם, דָּמוֹ יִשָּׁפֵךְ בִּידֵי שׁוֹפְטִים וְעֵדִים, כִּי הָאָדָם נַעֲשָׂה בְּצֶלֶם אֱלֹהִים.
ז וְאַתֶּ֖ם פְּר֣וּ וּרְב֑וּ שִׁרְצ֥וּ בָאָ֖רֶץ וּרְבוּ־בָֽהּ׃ וְאַתֶּם הוֹלִידוּ יְלָדִים וְהִתְרַבּוּ, הוֹלִידוּ בָּאָרֶץ וְהִתְרַבּוּ בָּהּ.

חמישי[עריכה]

ס פָּרָשַׁת בְּרִית בְּנֵי נֹחַ
ח וַיֹּ֤אמֶר אֱלֹהִים֙ אֶל־נֹ֔חַ וְאֶל־בָּנָ֥יו אִתּ֖וֹ לֵאמֹֽר׃ אֱלֹהִים אָמַר לְנֹחַ וּלְבָנָיו אִתּוֹ:
ט וַאֲנִ֕י הִנְנִ֥י מֵקִ֛ים אֶת־בְּרִיתִ֖י אִתְּכֶ֑ם וְאֶֽת־זַרְעֲכֶ֖ם אַֽחֲרֵיכֶֽם׃ וְהִנֵּה אֲנִי עוֹשֶׂה הֶסְכֵּם אִתְּכֶם וְעִם צֶאֱצָאֵיכֶם אַחֲרֵיכֶם.
י וְאֵ֨ת כׇּל־נֶ֤פֶשׁ הַֽחַיָּה֙ אֲשֶׁ֣ר אִתְּכֶ֔ם בָּע֧וֹף בַּבְּהֵמָ֛ה וּֽבְכׇל־חַיַּ֥ת הָאָ֖רֶץ אִתְּכֶ֑ם מִכֹּל֙ יֹצְאֵ֣י הַתֵּבָ֔ה לְכֹ֖ל חַיַּ֥ת הָאָֽרֶץ׃ וְעִם כָּל נֶפֶשׁ הַחַיָּה שֶׁאִתְּכֶם, הָעוֹף הַבְּהֵמָה וְכָל חַיַּת הָאָרֶץ שֶׁאִתְּכֶם, מִכָּל יוֹצְאֵי הַתֵּבָה – כָּל חַיּוֹת הָאָרֶץ.
יא וַהֲקִמֹתִ֤י אֶת־בְּרִיתִי֙ אִתְּכֶ֔ם וְלֹֽא־יִכָּרֵ֧ת כׇּל־בָּשָׂ֛ר ע֖וֹד מִמֵּ֣י הַמַּבּ֑וּל וְלֹֽא־יִהְיֶ֥ה ע֛וֹד מַבּ֖וּל לְשַׁחֵ֥ת הָאָֽרֶץ׃ וַאֲקַיֵּם אֶת הֶסְכֵּמִי אִתְּכֶם וְלֹא יֻשְׁמַד שׁוּב כָּל בָּשָׂר וָדָם בְּמֵי הַמַּבּוּל וְלֹא יִהְיֶה עוֹד מַבּוּל שֶׁיַּהֲרוֹס אֶת הָאָרֶץ.
יב וַיֹּ֣אמֶר אֱלֹהִ֗ים זֹ֤את אֽוֹת־הַבְּרִית֙ אֲשֶׁר־אֲנִ֣י נֹתֵ֗ן בֵּינִי֙ וּבֵ֣ינֵיכֶ֔ם וּבֵ֛ין כׇּל־נֶ֥פֶשׁ חַיָּ֖ה אֲשֶׁ֣ר אִתְּכֶ֑ם לְדֹרֹ֖ת עוֹלָֽם׃ אֱלֹהִים אָמַר: זֶה הַסִּימַן עַל הַהֶסְכֵּם שֶׁאֲנִי עוֹשֶׂה בֵּינִי וּבֵינֵיכֶם וּבֵין כָּל נֶפֶשׁ חַיָּה שֶׁאִתְּכֶם לְדוֹרוֹת הָעוֹלָם –
יג אֶת־קַשְׁתִּ֕י נָתַ֖תִּי בֶּֽעָנָ֑ן וְהָֽיְתָה֙ לְא֣וֹת בְּרִ֔ית בֵּינִ֖י וּבֵ֥ין הָאָֽרֶץ׃ שַׂמְתִּי אֶת הַקֶּשֶׁת שֶׁלִּי בֶּעָנָן וְהִיא תִּהְיֶה לְסִימַן הַהֶסְכֵּם שֶׁבֵּינִי וּבֵין הָאָרֶץ.
יד וְהָיָ֕ה בְּעַֽנְנִ֥י עָנָ֖ן עַל־הָאָ֑רֶץ וְנִרְאֲתָ֥ה הַקֶּ֖שֶׁת בֶּעָנָֽן׃ וּכְשֶׁאָבִיא עֲנָנִים עַל הָאָרֶץ תֵּרָאֶה הַקֶּשֶׁת בֶּעָנָן.
טו וְזָכַרְתִּ֣י אֶת־בְּרִיתִ֗י אֲשֶׁ֤ר בֵּינִי֙ וּבֵ֣ינֵיכֶ֔ם וּבֵ֛ין כׇּל־נֶ֥פֶשׁ חַיָּ֖ה בְּכׇל־בָּשָׂ֑ר וְלֹֽא־יִֽהְיֶ֨ה ע֤וֹד הַמַּ֙יִם֙ לְמַבּ֔וּל לְשַׁחֵ֖ת כׇּל־בָּשָֽׂר׃ וְאֶזְכּוֹר אֶת הֶסְכֵּמִי שֶׁבֵּינִי וּבֵינֵיכֶם וּבֵין כָּל נֶפֶשׁ שֶׁבְּכָל גּוּף שֶׁל בַּעַל חַיִּים, וְהַמַּיִם לֹא יְשַׁמְּשׁוּ עוֹד לְמַבּוּל לְהַשְׁמִיד כָּל בָּשָׂר וָדָם.
טז וְהָיְתָ֥ה הַקֶּ֖שֶׁת בֶּֽעָנָ֑ן וּרְאִיתִ֗יהָ לִזְכֹּר֙ בְּרִ֣ית עוֹלָ֔ם בֵּ֣ין אֱלֹהִ֔ים וּבֵין֙ כׇּל־נֶ֣פֶשׁ חַיָּ֔ה בְּכׇל־בָּשָׂ֖ר אֲשֶׁ֥ר עַל־הָאָֽרֶץ׃ הַקֶּשֶׁת תִּהְיֶה בֶּעָנָן, וְאֶרְאֶה אוֹתָהּ כְּדֵי לִזְכּוֹר אֶת הַהֶסְכֵּם הַתְּמִידִי שֶׁבֵּין אֱלֹהִים וּבֵין כָּל נֶפֶשׁ שֶׁבְּכָל גּוּף שֶׁל בַּעַל חַיִּים שֶׁעַל הָאָרֶץ.
יז וַיֹּ֥אמֶר אֱלֹהִ֖ים אֶל־נֹ֑חַ זֹ֤את אֽוֹת־הַבְּרִית֙ אֲשֶׁ֣ר הֲקִמֹ֔תִי בֵּינִ֕י וּבֵ֥ין כׇּל־בָּשָׂ֖ר אֲשֶׁ֥ר עַל־הָאָֽרֶץ׃ אֱלֹהִים אָמַר לְנֹחַ זֶה הַסִּימַן שֶׁהֶעֱמַדְתִּי עַל הַהֶסְכֵּם שֶׁבֵּינִי וּבֵין כָּל בָּשָׂר וָדָם שֶׁעַל הָאָרֶץ.

ששי[עריכה]

פ פָּרָשַׁת קִלְלַת כְּנַעַן
יח וַיִּֽהְי֣וּ בְנֵי־נֹ֗חַ הַיֹּֽצְאִים֙ מִן־הַתֵּבָ֔ה שֵׁ֖ם וְחָ֣ם וָיָ֑פֶת וְחָ֕ם ה֖וּא אֲבִ֥י כְנָֽעַן׃ בְּנֵי נֹחַ שֶׁיָּצְאוּ מֵהַתֵּבָה הָיוּ שֵׁם וְחָם וְיֶפֶת, וְחָם הוּא אֲבִיו שֶׁל כְּנַעַן.
יט שְׁלֹשָׁ֥ה אֵ֖לֶּה בְּנֵי־נֹ֑חַ וּמֵאֵ֖לֶּה נָֽפְצָ֥ה כׇל־הָאָֽרֶץ׃ שְׁלוֹשָׁה אֵלֶּה הֵם בְּנֵי נֹחַ וּמֵאֵלֶּה הִתְפַּזְּרוּ כָּל יוֹשְׁבֵי הָאָרֶץ.
כ וַיָּ֥חֶל נֹ֖חַ אִ֣ישׁ הָֽאֲדָמָ֑ה וַיִּטַּ֖ע כָּֽרֶם׃ נֹחַ הִתְחִיל לִהְיוֹת אִישׁ אֲדָמָה וְנָטַע כֶּרֶם.
כא וַיֵּ֥שְׁתְּ מִן־הַיַּ֖יִן וַיִּשְׁכָּ֑ר וַיִּתְגַּ֖ל בְּת֥וֹךְ אׇהֳלֹֽה׃ הוּא שָׁתָה מֵהַיַּיִן וְהִשְׁתַּכֵּר וְהוּא הִתְפַּשֵּׁט בְּתוֹךְ אָהֳלוֹ.
כב וַיַּ֗רְא חָ֚ם אֲבִ֣י כְנַ֔עַן אֵ֖ת עֶרְוַ֣ת אָבִ֑יו וַיַּגֵּ֥ד לִשְׁנֵֽי־אֶחָ֖יו בַּחֽוּץ׃ חָם אֲבִיו שֶׁל כְּנַעַן רָאָה אֶת עֶרְוַת אָבִיו וְסִפֵּר לִשְׁנֵי אֶחָיו בַּחוּץ.
כג וַיִּקַּח֩ שֵׁ֨ם וָיֶ֜פֶת אֶת־הַשִּׂמְלָ֗ה וַיָּשִׂ֙ימוּ֙ עַל־שְׁכֶ֣ם שְׁנֵיהֶ֔ם וַיֵּֽלְכוּ֙ אֲחֹ֣רַנִּ֔ית וַיְכַסּ֕וּ אֵ֖ת עֶרְוַ֣ת אֲבִיהֶ֑ם וּפְנֵיהֶם֙ אֲחֹ֣רַנִּ֔ית וְעֶרְוַ֥ת אֲבִיהֶ֖ם לֹ֥א רָאֽוּ׃ שֵׁם וְיֶפֶת לָקְחוּ אֶת הַבֶּגֶד וְשָׂמוּ עַל כִּתְפֵי שְׁנֵיהֶם וְהָלְכוּ אֲחוֹרַנִּית וְכִסּוּ אֶת עֶרְוַת אֲבִיהֶם, וּפְנֵיהֶם הָיוּ אֲחוֹרַנִּית וְהֵם לֹא רָאוּ אֶת עֶרְוַת אֲבִיהֶם.
כד וַיִּ֥יקֶץ נֹ֖חַ מִיֵּינ֑וֹ וַיֵּ֕דַע אֵ֛ת אֲשֶׁר־עָ֥שָׂה ל֖וֹ בְּנ֥וֹ הַקָּטָֽן׃ נֹחַ הִתְפַּקֵּחַ מִשִּׁכְרוּתוֹ וְיָדַע אֶת מַה שֶּׁעָשָׂה לוֹ בְּנוֹ הַקָּטָן.
כה וַיֹּ֖אמֶר אָר֣וּר כְּנָ֑עַן עֶ֥בֶד עֲבָדִ֖ים יִֽהְיֶ֥ה לְאֶחָֽיו׃ הוּא אָמַר: אָרוּר כְּנַעַן, הוּא יִהְיֶה עֶבֶד עֲבָדִים לְאֶחָיו!
כו וַיֹּ֕אמֶר בָּר֥וּךְ יְהֹוָ֖ה אֱלֹ֣הֵי שֵׁ֑ם וִיהִ֥י כְנַ֖עַן עֶ֥בֶד לָֽמוֹ׃ וְאָמַר: בָּרוּךְ ה' אֱלֹהֵי שֵׁם, וּכְנַעַן יִהְיֶה לוֹ עֶבֶד.
כז יַ֤פְתְּ אֱלֹהִים֙ לְיֶ֔פֶת וְיִשְׁכֹּ֖ן בְּאׇֽהֳלֵי־שֵׁ֑ם וִיהִ֥י כְנַ֖עַן עֶ֥בֶד לָֽמוֹ׃ אֱלֹהִים יַרְחִיב אֶת גְּבוּלוֹ שֶׁל יֶפֶת וְהוּא יִשְׁכּוֹן בְּאָהֳלֵי שֵׁם, וּכְנַעַן יִהְיֶה לָהֶם לְעֶבֶד.
כח וַֽיְחִי־נֹ֖חַ אַחַ֣ר הַמַּבּ֑וּל שְׁלֹ֤שׁ מֵאוֹת֙ שָׁנָ֔ה וַֽחֲמִשִּׁ֖ים שָׁנָֽה׃ נֹחַ חַי אַחֲרֵי הַמַּבּוּל שְׁלוֹשׁ מֵאוֹת וַחֲמִשִּׁים שָׁנָה.
כט וַיִּֽהְיוּ֙‏[2] כׇּל־יְמֵי־נֹ֔חַ תְּשַׁ֤ע מֵאוֹת֙ שָׁנָ֔ה וַחֲמִשִּׁ֖ים שָׁנָ֑ה וַיָּמֹֽת׃ כָּל חַיֵּי נֹחַ הָיוּ תְּשַׁע מֵאוֹת וַחֲמִשִּׁים שָׁנָה, וְהוּא מֵת.
פ פָּרָשַׁת תּוֹלְדֹת בְּנֵי נֹחַ
י א וְאֵ֙לֶּה֙ תּוֹלְדֹ֣ת בְּנֵי־נֹ֔חַ שֵׁ֖ם חָ֣ם וָיָ֑פֶת וַיִּוָּלְד֥וּ לָהֶ֛ם בָּנִ֖ים אַחַ֥ר הַמַּבּֽוּל׃ וְאֵלֶּה צֶאֱצָאֵי בְּנֵי נֹחַ: שֵׁם חָם וְיֶפֶת, וְנוֹלְדוּ לָהֶם בָּנִים אַחֲרֵי הַמַּבּוּל.
ב בְּנֵ֣י יֶ֔פֶת גֹּ֣מֶר וּמָג֔וֹג וּמָדַ֖י וְיָוָ֣ן וְתֻבָ֑ל וּמֶ֖שֶׁךְ וְתִירָֽס׃ בְּנֵי יֶפֶת: גּוֹמֶר וּמָגוֹג וּמָדַי וְיָוָן וְתֻבָל וּמֶשֶׁךְ וְתִירָס.
ג וּבְנֵ֖י גֹּ֑מֶר אַשְׁכְּﬞנַ֥ז וְרִיפַ֖ת וְתֹגַרְמָֽה׃ וּבְנֵי גּוֹמֶר: אַשְׁכְּנַז וְרִיפַת וְתוֹגַרְמָה.
ד וּבְנֵ֥י יָוָ֖ן אֱלִישָׁ֣ה וְתַרְשִׁ֑ישׁ כִּתִּ֖ים וְדֹדָנִֽים׃ וּבְנֵי יָוָן: אֱלִישָׁה וְתַרְשִׁישׁ כִּתִּים וְדוֹדָנִים.
ה מֵ֠אֵ֠לֶּה נִפְרְד֞וּ אִיֵּ֤י הַגּוֹיִם֙ בְּאַרְצֹתָ֔ם אִ֖ישׁ לִלְשֹׁנ֑וֹ לְמִשְׁפְּחֹתָ֖ם בְּגוֹיֵהֶֽם׃ מֵאֵלֶּה נִפְרְדוּ תּוֹשְׁבֵי אִיֵּי הָעַמִּים בְּאַרְצוֹתֵיהֶם כָּל אֶחָד בִּשְׂפָתוֹ לְמִשְׁפְּחוֹתֵיהֶם בְּעַמֵּיהֶם.
ו וּבְנֵ֖י חָ֑ם כּ֥וּשׁ וּמִצְרַ֖יִם וּפ֥וּט וּכְנָֽעַן׃ וּבְנֵי חָם: כּוּשׁ וּמִצְרַיִם וּפוּט וּכְנָעַן.
ז וּבְנֵ֣י כ֔וּשׁ סְבָא֙ וַֽחֲוִילָ֔ה וְסַבְתָּ֥ה וְרַעְמָ֖ה וְסַבְתְּכָ֑א וּבְנֵ֥י רַעְמָ֖ה שְׁבָ֥א וּדְדָֽן׃ וּבְנֵי כּוּשׁ: סְבָא וַחֲוִילָה וְסַבְתָּה וְרַעְמָה וְסַבְתְּכָא וּבְנֵי רַעְמָה שְׁבָא וּדְדָן.
ח וְכ֖וּשׁ יָלַ֣ד אֶת־נִמְרֹ֑ד ה֣וּא הֵחֵ֔ל לִֽהְי֥וֹת גִּבֹּ֖ר בָּאָֽרֶץ׃ וְכוּשׁ יָלַד אֶת נִמְרוֹד – הוּא הִתְחִיל לִהְיוֹת גֶּבֶר תַּקִּיף בָּאָרֶץ.
ט הֽוּא־הָיָ֥ה גִבֹּֽר־צַ֖יִד לִפְנֵ֣י יְהֹוָ֑ה עַל־כֵּן֙ יֵֽאָמַ֔ר כְּנִמְרֹ֛ד גִּבּ֥וֹר צַ֖יִד לִפְנֵ֥י יְהֹוָֽה׃ הוּא הָיָה גֶּבֶר שֶׁמּוֹרֵד בַּעֲבוֹדַת ה', לָכֵן אוֹמְרִים: כְּנִמְרוֹד שֶׁהָיָה גִּבּוֹר צַיִד לִפְנֵי ה'.
י וַתְּהִ֨י רֵאשִׁ֤ית מַמְלַכְתּוֹ֙ בָּבֶ֔ל וְאֶ֖רֶךְ וְאַכַּ֣ד וְכַלְנֵ֑ה בְּאֶ֖רֶץ שִׁנְעָֽר׃ רֵאשִׁית מַמְלַכְתּוֹ הָיְתָה בְּבָבֶל וְאֶרֶךְ וְאַכַּד וְכַלְנֵה, בְּאֶרֶץ עִירָק.
יא מִן־הָאָ֥רֶץ הַהִ֖וא יָצָ֣א אַשּׁ֑וּר וַיִּ֙בֶן֙ אֶת־נִ֣ינְוֵ֔ה וְאֶת־רְחֹבֹ֥ת עִ֖יר וְאֶת־כָּֽלַח׃ מֵהָאָרֶץ הַהִיא יָצָא אַשּׁוּר, הוּא בָּנָה אֶת נִינְוֵה וְאֶת רְחוֹבוֹת עִיר וְאֶת כָּלַח.
יב וְֽאֶת־רֶ֔סֶן בֵּ֥ין נִֽינְוֵ֖ה וּבֵ֣ין כָּ֑לַח הִ֖וא הָעִ֥יר הַגְּדֹלָֽה׃ וְאֶת רֶסֶן בֵּין נִינְוֵה וּבֵין כָּלַח הִיא הָעִיר הַגְּדוֹלָה.
יג וּמִצְרַ֡יִם יָלַ֞ד אֶת־לוּדִ֧ים וְאֶת־עֲנָמִ֛ים וְאֶת־לְהָבִ֖ים וְאֶת־נַפְתֻּחִֽים׃ וּמִצְרַיִם יָלַד אֶת לוּדִים וְאֶת עֲנָמִים וְאֶת לְהָבִים וְאֶת נַפְתּוּחִים.
יד וְֽאֶת־פַּתְרֻסִ֞ים וְאֶת־כַּסְלֻחִ֗ים אֲשֶׁ֨ר יָצְא֥וּ מִשָּׁ֛ם פְּלִשְׁתִּ֖ים וְאֶת־כַּפְתֹּרִֽים׃ וְאֶת פַּתְרוּסִים וְאֶת כַּסְלוּחִים שֶׁנּוֹלְדוּ מֵהֶם הַפְּלִשְׁתִּים, וְאֶת כַּפְתּוֹרִים.
ס
טו וּכְנַ֗עַן יָלַ֛ד אֶת־צִידֹ֥ן בְּכֹר֖וֹ וְאֶת־חֵֽת׃ וּכְנַעַן יָלַד אֶת צִידוֹן בְּכוֹרוֹ וְאֶת חֵת.
טז וְאֶת־הַיְבוּסִי֙ וְאֶת־הָ֣אֱמֹרִ֔י וְאֵ֖ת הַגִּרְגָּשִֽׁי׃ וְאֶת הַיְּבוּסִי וְאֶת הָאֱמוֹרִי וְאֶת הַגִּרְגָּשִׁי.
יז וְאֶת־הַֽחִוִּ֥י וְאֶת־הַֽעַרְקִ֖י וְאֶת־הַסִּינִֽי׃ וְאֶת הַחִוִּי וְאֶת הַעַרְקִי וְאֶת הַסִּינִי.
יח וְאֶת־הָֽאַרְוָדִ֥י וְאֶת־הַצְּמָרִ֖י וְאֶת־הַֽחֲמָתִ֑י וְאַחַ֣ר נָפֹ֔צוּ מִשְׁפְּח֖וֹת הַֽכְּנַעֲנִֽי׃ וְאֶת הָאַרְוָדִי וְאֶת הַצְּמָרִי וְאֶת הַחֲמָתִי, וְאַחַר כָּךְ הִתְפַּשְּׁטוּ מֵהֶם כָּל מִשְׁפְּחוֹת הַכְּנַעֲנִי.
יט וַיְהִ֞י גְּב֤וּל הַֽכְּנַעֲנִי֙ מִצִּידֹ֔ן בֹּאֲכָ֥ה גְרָ֖רָה עַד־עַזָּ֑ה בֹּאֲכָ֞ה סְדֹ֧מָה וַעֲמֹרָ֛ה וְאַדְמָ֥ה וּצְבֹיִ֖ם עַד־לָֽשַׁע׃ גְּבוּל הַכְּנַעֲנִי הָיָה מִצִּידוֹן – הוֹלֵךְ דָּרוֹמָה לִגְרָר עַד עַזָּה, וְהוֹלֵךְ מִזְרָחָה לִסְדוֹם וַעֲמוֹרָה וְאַדְמָה וּצְבוֹיִים עַד לָשַׁע.
כ אֵ֣לֶּה בְנֵי־חָ֔ם לְמִשְׁפְּחֹתָ֖ם לִלְשֹֽׁנֹתָ֑ם בְּאַרְצֹתָ֖ם בְּגוֹיֵהֶֽם׃ אֵלֶּה בְּנֵי חָם לְמִשְׁפְּחוֹתֵיהֶם לַשָּׂפוֹת שֶׁלָּהֶם בְּאַרְצוֹתֵיהֶם בְּעַמֵּיהֶם.
ס
כא וּלְשֵׁ֥ם יֻלַּ֖ד גַּם־ה֑וּא אֲבִי֙ כׇּל־בְּנֵי־עֵ֔בֶר אֲחִ֖י יֶ֥פֶת הַגָּדֽוֹל׃ וְגַם לְשֵׁם נוֹלְדוּ, שֵׁם הוּא אֲבִיהֶם שֶׁל כָּל בְּנֵי עֵבֶר, הוּא אֲחִיו שֶׁל יֶפֶת – הָאַח הַגָּדוֹל.
כב בְּנֵ֥י שֵׁ֖ם עֵילָ֣ם וְאַשּׁ֑וּר וְאַרְפַּכְשַׁ֖ד וְל֥וּד וַֽאֲרָֽם׃ בְּנֵי שֵׁם: עֵילָם וְאַשּׁוּר וְאַרְפַּכְשַׁד וְלוּד וַאֲרָם.
כג וּבְנֵ֖י אֲרָ֑ם ע֥וּץ וְח֖וּל וְגֶ֥תֶר וָמַֽשׁ׃ וּבְנֵי אֲרָם: עוּץ וְחוּל וְגֶתֶר וּמַשׁ.
כד וְאַרְפַּכְשַׁ֖ד יָלַ֣ד אֶת־שָׁ֑לַח וְשֶׁ֖לַח יָלַ֥ד אֶת־עֵֽבֶר׃ וְאַרְפַּכְשַׁד יָלַד אֶת שֶׁלַח, וְשֶׁלַח יָלַד אֶת עֵבֶר.
כה וּלְעֵ֥בֶר יֻלַּ֖ד שְׁנֵ֣י בָנִ֑ים שֵׁ֣ם הָֽאֶחָ֞ד פֶּ֗לֶג כִּ֤י בְיָמָיו֙ נִפְלְגָ֣ה הָאָ֔רֶץ וְשֵׁ֥ם אָחִ֖יו יׇקְטָֽן׃ וּלְעֵבֶר נוֹלְדוּ שְׁנֵי בָּנִים שֵׁם הָאֶחָד פֶּלֶג כִּי בְּיָמָיו הִתְחַלְּקוּ עַמֵּי הָעוֹלָם לְשָׂפוֹת שׁוֹנוֹת, וְשֵׁם אָחִיו יָקְטָן.
כו וְיׇקְטָ֣ן יָלַ֔ד אֶת־אַלְמוֹדָ֖ד וְאֶת־שָׁ֑לֶף וְאֶת־חֲצַרְמָ֖וֶת וְאֶת־יָֽרַח׃ וְיָקְטָן יָלַד אֶת אַלְמוֹדָד וְאֶת שָׁלֶף וְאֶת חֲצַרְמָוֶת וְאֶת יָרַח.
כז וְאֶת־הֲדוֹרָ֥ם וְאֶת־אוּזָ֖ל וְאֶת־דִּקְלָֽה׃ וְאֶת הֲדוֹרָם וְאֶת אוּזָל וְאֶת דִּקְלָה.
כח וְאֶת־עוֹבָ֥ל וְאֶת־אֲבִֽימָאֵ֖ל וְאֶת־שְׁבָֽא׃ וְאֶת עוֹבָל וְאֶת אֲבִימָאֵל וְאֶת שְׁבָא.
כט וְאֶת־אוֹפִ֥ר וְאֶת־חֲוִילָ֖ה וְאֶת־יוֹבָ֑ב כׇּל־אֵ֖לֶּה בְּנֵ֥י יׇקְטָֽן׃ וְאֶת אוֹפִיר וְאֶת חֲוִילָה וְאֶת יוֹבָב כָּל אֵלֶּה בְּנֵי יָקְטָן.
ל וַיְהִ֥י מוֹשָׁבָ֖ם מִמֵּשָׁ֑א בֹּאֲכָ֥ה סְפָ֖רָה הַ֥ר הַקֶּֽדֶם׃ מוֹשָׁבָם הָיָה מִמֵּשָׁא עַד לָעִיר הַנִּקְרֵאת סְפָר שֶׁבְּהַר הַמִּזְרָח.
לא אֵ֣לֶּה בְנֵי־שֵׁ֔ם לְמִשְׁפְּחֹתָ֖ם לִלְשֹׁנֹתָ֑ם בְּאַרְצֹתָ֖ם לְגוֹיֵהֶֽם׃ אֵלֶּה בְּנֵי שֵׁם לְמִשְׁפְּחוֹתֵיהֶם לַשָּׂפוֹת שֶׁלָּהֶם בְּאַרְצוֹתֵיהֶם לְעַמֵּיהֶם.
לב אֵ֣לֶּה מִשְׁפְּחֹ֧ת בְּנֵי־נֹ֛חַ לְתוֹלְדֹתָ֖ם בְּגוֹיֵהֶ֑ם וּמֵאֵ֜לֶּה נִפְרְד֧וּ הַגּוֹיִ֛ם בָּאָ֖רֶץ אַחַ֥ר הַמַּבּֽוּל׃ אֵלֶּה צֶאֱצָאֵי מִשְׁפְּחוֹת בְּנֵי נֹחַ בְּעַמֵּיהֶם, וּמֵאֵלֶּה נִפְרְדוּ הָעַמִּים בָּאָרֶץ אַחֲרֵי הַמַּבּוּל.

שביעי[עריכה]

פ פָּרָשַׁת מִגְדַּל בָּבֶל
יא א וַיְהִ֥י כׇל־הָאָ֖רֶץ שָׂפָ֣ה אֶחָ֑ת וּדְבָרִ֖ים אֲחָדִֽים׃ כָּל יוֹשְׁבֵי הָאָרֶץ הָיוּ מְדַבְּרִים בְּשָׂפָה אַחַת וּבְאוֹתָם מֻשָּׂגִים.
ב וַיְהִ֖י בְּנׇסְעָ֣ם מִקֶּ֑דֶם וַֽיִּמְצְא֥וּ בִקְעָ֛ה בְּאֶ֥רֶץ שִׁנְעָ֖ר וַיֵּ֥שְׁבוּ שָֽׁם׃ בְּנָסְעָם מִמִּזְרָח, הֵם מָצְאוּ בִּקְעָה בְּעִירָק וְגָרוּ שָׁם.
ג וַיֹּאמְר֞וּ אִ֣ישׁ אֶל־רֵעֵ֗הוּ הָ֚בָה נִלְבְּנָ֣ה לְבֵנִ֔ים וְנִשְׂרְפָ֖ה לִשְׂרֵפָ֑ה וַתְּהִ֨י לָהֶ֤ם הַלְּבֵנָה֙ לְאָ֔בֶן וְהַ֣חֵמָ֔ר הָיָ֥ה לָהֶ֖ם לַחֹֽמֶר׃ אָמְרוּ אִישׁ לַחֲבֵרוֹ: בּוֹאוּ נַעֲשֶׂה לְבֵנִים וְנִשְׂרוֹף אוֹתָם בְּכִבְשַׁן כְּדֵי לְחַזֵּק אוֹתָם, הַלְּבֵנָה שִׁמְּשָׁה לָהֶם לְאֶבֶן וְהַחֵמָר שִׁמֵּשׁ לָהֶם לְמֶלֶט.
ד וַיֹּאמְר֞וּ הָ֣בָה ׀ נִבְנֶה־לָּ֣נוּ עִ֗יר וּמִגְדָּל֙ וְרֹאשׁ֣וֹ בַשָּׁמַ֔יִם וְנַֽעֲשֶׂה־לָּ֖נוּ שֵׁ֑ם פֶּן־נָפ֖וּץ עַל־פְּנֵ֥י כׇל־הָאָֽרֶץ׃ אָמְרוּ: בּוֹאוּ נִבְנֶה לָנוּ עִיר וּמִגְדָּל שֶׁרֹאשׁוֹ בַּשָּׁמַיִם וְנַעֲשֶׂה לָנוּ מֶרְכַּז תַּרְבּוּתִי, שֶׁלֹּא נִתְפַּזֵּר לְתַרְבֻּיּוֹת שׁוֹנוֹת עַל פְּנֵי כָּל הָאָרֶץ.
ה וַיֵּ֣רֶד יְהֹוָ֔ה לִרְאֹ֥ת אֶת־הָעִ֖יר וְאֶת־הַמִּגְדָּ֑ל אֲשֶׁ֥ר בָּנ֖וּ בְּנֵ֥י הָאָדָֽם׃ ה' יָרַד לִרְאוֹת אֶת הָעִיר וְאֶת הַמִּגְדָּל שֶׁבָּנוּ בְּנֵי הָאָדָם.
ו וַיֹּ֣אמֶר יְהֹוָ֗ה הֵ֣ן עַ֤ם אֶחָד֙ וְשָׂפָ֤ה אַחַת֙ לְכֻלָּ֔ם וְזֶ֖ה הַחִלָּ֣ם לַעֲשׂ֑וֹת וְעַתָּה֙ לֹֽא־יִבָּצֵ֣ר מֵהֶ֔ם כֹּ֛ל אֲשֶׁ֥ר יָזְמ֖וּ לַֽעֲשֽׂוֹת׃ ה' אָמַר: הֲרֵי הֵם עַם אֶחָד, וּלְכֻלָּם יֵשׁ שָׂפָה אַחַת, לָכֵן הֵם הִתְחִילוּ לַעֲשׂוֹת כָּךְ, וְעַכְשָׁיו לֹא יִמָּנַע מֵהֶם כָּל מַה שֶּׁתִּכְנְנוּ לַעֲשׂוֹת.
ז הָ֚בָה נֵֽרְדָ֔ה וְנָבְלָ֥ה שָׁ֖ם שְׂפָתָ֑ם אֲשֶׁר֙ לֹ֣א יִשְׁמְע֔וּ אִ֖ישׁ שְׂפַ֥ת רֵעֵֽהוּ׃ בֹּא נֵרֶד וּנְבַלְבֵּל שָׁם אֶת שְׂפָתָם כְּדֵי שֶׁלֹּא יָבִינוּ אִישׁ אֶת שְׂפַת חֲבֵרוֹ.
ח וַיָּ֨פֶץ יְהֹוָ֥ה אֹתָ֛ם מִשָּׁ֖ם עַל־פְּנֵ֣י כׇל־הָאָ֑רֶץ וַֽיַּחְדְּל֖וּ לִבְנֹ֥ת הָעִֽיר׃ ה' פִּזֵּר אוֹתָם מִשָּׁם עַל פְּנֵי כָּל הָעוֹלָם וְהֵם הִפְסִיקוּ לִבְנוֹת אֶת הָעִיר.
ט עַל־כֵּ֞ן קָרָ֤א שְׁמָהּ֙ בָּבֶ֔ל כִּי־שָׁ֛ם בָּלַ֥ל יְהֹוָ֖ה שְׂפַ֣ת כׇּל־הָאָ֑רֶץ וּמִשָּׁם֙ הֱפִיצָ֣ם יְהֹוָ֔ה עַל־פְּנֵ֖י כׇּל־הָאָֽרֶץ׃ לָכֵן נִקְרָא שְׁמָהּ בָּבֶל כִּי שָׁם ה' בִּלְבֵּל אֶת שְׂפַת כָּל הָעוֹלָם וּמִשָּׁם ה' הֱפִיץ אוֹתָם עַל פְּנֵי כָּל הָעוֹלָם.
פ פָּרָשַׁת תּוֹלְדֹת בְּנֵי שֵׁם
י אֵ֚לֶּה תּוֹלְדֹ֣ת שֵׁ֔ם שֵׁ֚ם בֶּן־מְאַ֣ת שָׁנָ֔ה וַיּ֖וֹלֶד אֶת־אַרְפַּכְשָׁ֑ד שְׁנָתַ֖יִם אַחַ֥ר הַמַּבּֽוּל׃ אֵלֶּה צֶאֱצָאֵי שֵׁם: שֵׁם הָיָה בֶּן מֵאָה שָׁנָה, וְהוֹלִיד אֶת אַרְפַּכְשָׁד שְׁנָתַיִם אַחֲרֵי הַמַּבּוּל.
יא וַֽיְחִי־שֵׁ֗ם אַֽחֲרֵי֙ הוֹלִיד֣וֹ אֶת־אַרְפַּכְשָׁ֔ד חֲמֵ֥שׁ מֵא֖וֹת שָׁנָ֑ה וַיּ֥וֹלֶד בָּנִ֖ים וּבָנֽוֹת׃ שֵׁם חַי אַחֲרֵי שֶׁהוֹלִיד אֶת אַרְפַּכְשָׁד חֲמֵשׁ מֵאוֹת שָׁנָה וְהוֹלִיד בָּנִים וּבָנוֹת.
ס
יב וְאַרְפַּכְשַׁ֣ד חַ֔י חָמֵ֥שׁ וּשְׁלֹשִׁ֖ים שָׁנָ֑ה וַיּ֖וֹלֶד אֶת־שָֽׁלַח׃ וְאַרְפַּכְשַׁד חַי שְׁלוֹשִׁים וְחָמֵשׁ שָׁנִים וְהוֹלִיד אֶת שֶׁלַח.
יג וַיְחִ֣י אַרְפַּכְשַׁ֗ד אַֽחֲרֵי֙ הוֹלִיד֣וֹ אֶת־שֶׁ֔לַח שָׁלֹ֣שׁ שָׁנִ֔ים וְאַרְבַּ֥ע מֵא֖וֹת שָׁנָ֑ה וַיּ֥וֹלֶד בָּנִ֖ים וּבָנֽוֹת׃ אַרְפַּכְשַׁד חַי אַחֲרֵי שֶׁהוֹלִיד אֶת שֶׁלַח אַרְבַּע מֵאוֹת וְשָׁלוֹשׁ שָׁנִים וְהוֹלִיד בָּנִים וּבָנוֹת.
ס
יד וְשֶׁ֥לַח חַ֖י שְׁלֹשִׁ֣ים שָׁנָ֑ה וַיּ֖וֹלֶד אֶת־עֵֽבֶר׃ וְשֶׁלַח חַי שְׁלוֹשִׁים שָׁנָה וְהוֹלִיד אֶת עֵבֶר.
טו וַֽיְחִי־שֶׁ֗לַח אַחֲרֵי֙ הוֹלִיד֣וֹ אֶת־עֵ֔בֶר שָׁלֹ֣שׁ שָׁנִ֔ים וְאַרְבַּ֥ע מֵא֖וֹת שָׁנָ֑ה וַיּ֥וֹלֶד בָּנִ֖ים וּבָנֽוֹת׃ שֶׁלַח חַי אַחֲרֵי שֶׁהוֹלִיד אֶת עֵבֶר אַרְבַּע מֵאוֹת וְשָׁלוֹשׁ שָׁנִים וְהוֹלִיד בָּנִים וּבָנוֹת.
ס
טז וַֽיְחִי־עֵ֕בֶר אַרְבַּ֥ע וּשְׁלֹשִׁ֖ים שָׁנָ֑ה וַיּ֖וֹלֶד אֶת־פָּֽלֶג׃ עֵבֶר חַי שְׁלוֹשִׁים וְאַרְבַּע שָׁנִים וְהוֹלִיד אֶת פֶּלֶג.
יז וַֽיְחִי־עֵ֗בֶר אַחֲרֵי֙ הוֹלִיד֣וֹ אֶת־פֶּ֔לֶג שְׁלֹשִׁ֣ים שָׁנָ֔ה וְאַרְבַּ֥ע מֵא֖וֹת שָׁנָ֑ה וַיּ֥וֹלֶד בָּנִ֖ים וּבָנֽוֹת׃ עֵבֶר חַי אַחֲרֵי שֶׁהוֹלִיד אֶת פֶּלֶג אַרְבַּע מֵאוֹת וּשְׁלוֹשִׁים שָׁנָה וְהוֹלִיד בָּנִים וּבָנוֹת.

יח וַֽיְחִי־פֶ֖לֶג שְׁלֹשִׁ֣ים שָׁנָ֑ה וַיּ֖וֹלֶד אֶת־רְעֽוּ׃ פֶּלֶג חַי שְׁלוֹשִׁים שָׁנָה וְהוֹלִיד אֶת רְעוּ.
יט וַֽיְחִי־פֶ֗לֶג אַחֲרֵי֙ הוֹלִיד֣וֹ אֶת־רְע֔וּ תֵּ֥שַׁע שָׁנִ֖ים וּמָאתַ֣יִם שָׁנָ֑ה וַיּ֥וֹלֶד בָּנִ֖ים וּבָנֽוֹת׃ פֶּלֶג חַי אַחֲרֵי שֶׁהוֹלִיד אֶת רְעוּ מָאתַיִם וְתֵשַׁע שָׁנִים וְהוֹלִיד בָּנִים וּבָנוֹת.

כ וַיְחִ֣י רְע֔וּ שְׁתַּ֥יִם וּשְׁלֹשִׁ֖ים שָׁנָ֑ה וַיּ֖וֹלֶד אֶת־שְׂרֽוּג׃ רְעוּ חַי שְׁלוֹשִׁים וּשְׁתַּיִם שָׁנִים וְהוֹלִיד אֶת שְׂרוּג.
כא וַיְחִ֣י רְע֗וּ אַחֲרֵי֙ הוֹלִיד֣וֹ אֶת־שְׂר֔וּג שֶׁ֥בַע שָׁנִ֖ים וּמָאתַ֣יִם שָׁנָ֑ה וַיּ֥וֹלֶד בָּנִ֖ים וּבָנֽוֹת׃ רְעוּ חַי אַחֲרֵי שֶׁהוֹלִיד אֶת שְׂרוּג מָאתַיִם וְשֶׁבַע שָׁנִים וְהוֹלִיד בָּנִים וּבָנוֹת.
ס
כב וַיְחִ֥י שְׂר֖וּג שְׁלֹשִׁ֣ים שָׁנָ֑ה וַיּ֖וֹלֶד אֶת־נָחֽוֹר׃ שְׂרוּג חַי שְׁלוֹשִׁים שָׁנָה וְהוֹלִיד אֶת נָחוֹר.
כג וַיְחִ֣י שְׂר֗וּג אַחֲרֵ֛י הוֹלִיד֥וֹ אֶת־נָח֖וֹר מָאתַ֣יִם שָׁנָ֑ה וַיּ֥וֹלֶד בָּנִ֖ים וּבָנֽוֹת׃ שְׂרוּג חַי אַחֲרֵי שֶׁהוֹלִיד אֶת נָחוֹר מָאתַיִם שָׁנָה וְהוֹלִיד בָּנִים וּבָנוֹת.

כד וַיְחִ֣י נָח֔וֹר תֵּ֥שַׁע וְעֶשְׂרִ֖ים שָׁנָ֑ה וַיּ֖וֹלֶד אֶת־תָּֽרַח׃ נָחוֹר חַי עֶשְׂרִים וְתֵשַׁע שָׁנִים וְהוֹלִיד אֶת תֶּרַח.
כה וַיְחִ֣י נָח֗וֹר אַחֲרֵי֙ הוֹלִיד֣וֹ אֶת־תֶּ֔רַח תְּשַֽׁע־עֶשְׂרֵ֥ה שָׁנָ֖ה וּמְאַ֣ת שָׁנָ֑ה וַיּ֥וֹלֶד בָּנִ֖ים וּבָנֽוֹת׃ נָחוֹר חַי אַחֲרֵי שֶׁהוֹלִיד אֶת תֶּרַח מֵאָה וּתְשַׁע עֶשְׂרֵה שָׁנָה וְהוֹלִיד בָּנִים וּבָנוֹת.
ס פָּרָשַׁת תּוֹלְדוֹת בְּנֵי תֶּרַח
כו וַֽיְחִי־תֶ֖רַח שִׁבְעִ֣ים שָׁנָ֑ה וַיּ֙וֹלֶד֙ אֶת־אַבְרָ֔ם אֶת־נָח֖וֹר וְאֶת־הָרָֽן׃ תֶּרַח חַי שִׁבְעִים שָׁנָה וְהוֹלִיד אֶת אַבְרָם אֶת נָחוֹר וְאֶת הָרָן.
כז וְאֵ֙לֶּה֙ תּוֹלְדֹ֣ת תֶּ֔רַח תֶּ֚רַח הוֹלִ֣יד אֶת־אַבְרָ֔ם אֶת־נָח֖וֹר וְאֶת־הָרָ֑ן וְהָרָ֖ן הוֹלִ֥יד אֶת־לֽוֹט׃ וְאֵלֶּה צֶאֱצָאֵי תֶּרַח: תֶּרַח הוֹלִיד אֶת אַבְרָם אֶת נָחוֹר וְאֶת הָרָן וְהָרָן הוֹלִיד אֶת לוֹט.
כח וַיָּ֣מׇת הָרָ֔ן עַל־פְּנֵ֖י תֶּ֣רַח אָבִ֑יו בְּאֶ֥רֶץ מוֹלַדְתּ֖וֹ בְּא֥וּר כַּשְׂדִּֽים׃ הָרָן מֵת בְּחַיֵּי תֶּרַח אָבִיו בָּאָרֶץ שֶׁנּוֹלַד בָּהּ – בְּאוּר כַּשְׂדִּים.

מפטיר[עריכה]

כט וַיִּקַּ֨ח אַבְרָ֧ם וְנָח֛וֹר לָהֶ֖ם נָשִׁ֑ים שֵׁ֤ם אֵֽשֶׁת־אַבְרָם֙ שָׂרָ֔י וְשֵׁ֤ם אֵֽשֶׁת־נָחוֹר֙ מִלְכָּ֔ה בַּת־הָרָ֥ן אֲבִֽי־מִלְכָּ֖ה וַֽאֲבִ֥י יִסְכָּֽה׃ אַבְרָם וְנָחוֹר לָקְחוּ לָהֶם נָשִׁים, שֵׁם אֵשֶׁת אַבְרָם – שָׂרַי, וְשֵׁם אֵשֶׁת נָחוֹר – מִלְכָּה בַּת הָרָן שֶׁהָיָה אֲבִיהָ שֶׁל מִלְכָּה וַאֲבִיהָ שֶׁל יִסְכָּה.
ל וַתְּהִ֥י שָׂרַ֖י עֲקָרָ֑ה אֵ֥ין לָ֖הּ וָלָֽד׃ שָׂרַי הָיְתָה עֲקָרָה – אֵין לָהּ יֶלֶד.
לא וַיִּקַּ֨ח תֶּ֜רַח אֶת־אַבְרָ֣ם בְּנ֗וֹ וְאֶת־ל֤וֹט בֶּן־הָרָן֙ בֶּן־בְּנ֔וֹ וְאֵת֙ שָׂרַ֣י כַּלָּת֔וֹ אֵ֖שֶׁת אַבְרָ֣ם בְּנ֑וֹ וַיֵּצְא֨וּ אִתָּ֜ם מֵא֣וּר כַּשְׂדִּ֗ים לָלֶ֙כֶת֙ אַ֣רְצָה כְּנַ֔עַן וַיָּבֹ֥אוּ עַד־חָרָ֖ן וַיֵּ֥שְׁבוּ שָֽׁם׃ תֶּרַח לָקַח אֶת אַבְרָם בְּנוֹ וְאֶת לוֹט בֶּן הָרָן בֶּן בְּנוֹ וְאֶת שָׂרַי כַּלָּתוֹ אֵשֶׁת אַבְרָם בְּנוֹ, תֶּרַח וְאַבְרָם יָצְאוּ אִתָּם מֵאוּר כַּשְׂדִּים לָלֶכֶת לְאֶרֶץ כְּנַעַן וּבָאוּ עַד חָרָן וְהִתְיַשְּׁבוּ שָׁם.
לב וַיִּהְי֣וּ יְמֵי־תֶ֔רַח חָמֵ֥שׁ שָׁנִ֖ים וּמָאתַ֣יִם שָׁנָ֑ה וַיָּ֥מׇת תֶּ֖רַח בְּחָרָֽן׃ יְמֵי תֶּרַח הָיוּ מָאתַיִם וְחָמֵשׁ שָׁנִים, וְתֶרַח מֵת בְּחָרָן.
