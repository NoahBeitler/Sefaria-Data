קפיצה לניווט קפיצה לחיפוש
א א וַיְהִ֗י בִּימֵי֙ שְׁפֹ֣ט הַשֹּׁפְטִ֔ים וַיְהִ֥י רָעָ֖ב בָּאָ֑רֶץ וַיֵּ֨לֶךְ אִ֜ישׁ מִבֵּ֧ית לֶ֣חֶם יְהוּדָ֗ה לָגוּר֙ בִּשְׂדֵ֣י מוֹאָ֔ב ה֥וּא וְאִשְׁתּ֖וֹ וּשְׁנֵ֥י בָנָֽיו׃ בִּתְקוּפַת הַשּׁוֹפְטִים הָיָה רָעָב בְּאֶרֶץ יִשְׂרָאֵל, וְאִישׁ אֶחָד מֵהָעִיר בֵּית לֶחֶם שֶׁבְּנַחֲלַת שֵׁבֶט יְהוּדָה הָלַךְ לָגוּר בְּאֶרֶץ מוֹאָב וְאִתּוֹ אִשְׁתּוֹ וּשְׁנֵי בָּנָיו.
ב וְשֵׁ֣ם הָאִ֣ישׁ אֱֽלִימֶ֡לֶךְ וְשֵׁם֩ אִשְׁתּ֨וֹ נׇעֳמִ֜י וְשֵׁ֥ם שְׁנֵֽי־בָנָ֣יו ׀ מַחְל֤וֹן וְכִלְיוֹן֙ אֶפְרָתִ֔ים מִבֵּ֥ית לֶ֖חֶם יְהוּדָ֑ה וַיָּבֹ֥אוּ שְׂדֵֽי־מוֹאָ֖ב וַיִּֽהְיוּ־שָֽׁם׃ שֵׁם הָאִישׁ הָיָה אֱלִימֶלֶךְ וְשֵׁם אִשְׁתּוֹ נָעֳמִי וְשֵׁם שְׁנֵי בָּנָיו מַחְלוֹן וְכִלְיוֹן, הֵם הָיוּ מֵהָעִיר אֶפְרָת הִיא בֵּית לֶחֶם שֶׁבִּיהוּדָה, הֵם בָּאוּ לְאֶרֶץ מוֹאָב וְנִשְׁאֲרוּ שָׁם.
ג וַיָּ֥מׇת אֱלִימֶ֖לֶךְ אִ֣ישׁ נׇעֳמִ֑י וַתִּשָּׁאֵ֥ר הִ֖יא וּשְׁנֵ֥י בָנֶֽיהָ׃ אֱלִימֶלֶךְ בַּעֲלָהּ שֶׁל נָעֳמִי מֵת, וְהִיא נִשְׁאֲרָה עִם שְׁנֵי בָּנֶיהָ.
ד וַיִּשְׂא֣וּ לָהֶ֗ם נָשִׁים֙ מֹֽאֲבִיּ֔וֹת שֵׁ֤ם הָאַחַת֙ עׇרְפָּ֔ה וְשֵׁ֥ם הַשֵּׁנִ֖ית ר֑וּת וַיֵּ֥שְׁבוּ שָׁ֖ם כְּעֶ֥שֶׂר שָׁנִֽים׃ הֵם נָשְׂאוּ לָהֶם נָשִׁים מוֹאֲבִיּוֹת, שֵׁם הָאַחַת עָרְפָּה וְשֵׁם הַשְּׁנִיָּה רוּת, וְהֵם גָּרוּ שָׁם כְּעֶשֶׂר שָׁנִים.
ה וַיָּמֻ֥תוּ גַם־שְׁנֵיהֶ֖ם מַחְל֣וֹן וְכִלְי֑וֹן וַתִּשָּׁאֵר֙ הָֽאִשָּׁ֔ה מִשְּׁנֵ֥י יְלָדֶ֖יהָ וּמֵאִישָֽׁהּ׃ גַּם מַחְלוֹן וְכִלְיוֹן שְׁנֵיהֶם מֵתוּ, וְהָאִשָּׁה נִשְׁאֲרָה לְבַדָּהּ מִשְּׁנֵי יְלָדֶיהָ וּמִבַּעֲלָהּ.
ו וַתָּ֤קׇם הִיא֙ וְכַלֹּתֶ֔יהָ וַתָּ֖שׇׁב מִשְּׂדֵ֣י מוֹאָ֑ב כִּ֤י שָֽׁמְעָה֙ בִּשְׂדֵ֣ה מוֹאָ֔ב כִּֽי־פָקַ֤ד יְהֹוָה֙ אֶת־עַמּ֔וֹ לָתֵ֥ת לָהֶ֖ם לָֽחֶם׃ הִיא קָמָה עִם כַּלֹּתֶיהָ וְחָזְרָה מֵאֶרֶץ מוֹאָב כִּי הִיא שָׁמְעָה בְּאֶרֶץ מוֹאָב שֶׁה' זָכַר אֶת עַמּוֹ וְנָתַן לָהֶם מָזוֹן.
ז וַתֵּצֵ֗א מִן־הַמָּקוֹם֙ אֲשֶׁ֣ר הָֽיְתָה־שָּׁ֔מָּה וּשְׁתֵּ֥י כַלּוֹתֶ֖יהָ עִמָּ֑הּ וַתֵּלַ֣כְנָה בַדֶּ֔רֶךְ לָשׁ֖וּב אֶל־אֶ֥רֶץ יְהוּדָֽה׃ הִיא יָצְאָה מֵהַמָּקוֹם שֶׁהָיְתָה שָׁם וּשְׁתֵּי כַּלֹּתֶיהָ יָצְאוּ אִתָּהּ, וְהֵן הָלְכוּ בַּדֶּרֶךְ לַחֲזוֹר לְאֶרֶץ יְהוּדָה.
ח וַתֹּ֤אמֶר נׇעֳמִי֙ לִשְׁתֵּ֣י כַלֹּתֶ֔יהָ לֵ֣כְנָה שֹּׁ֔בְנָה אִשָּׁ֖ה לְבֵ֣ית אִמָּ֑הּ יעשה יַ֣עַשׂ יְהֹוָ֤ה עִמָּכֶם֙ חֶ֔סֶד כַּאֲשֶׁ֧ר עֲשִׂיתֶ֛ם עִם־הַמֵּתִ֖ים וְעִמָּדִֽי׃ אָמְרָה נָעֳמִי לִשְׁתֵּי כַּלֹּתֶיהָ: לְכוּ וְשׁוּבוּ כָּל אַחַת לַבַּיִת שֶׁל אִמָּא שֶׁלָּהּ, ה' יַעֲשֶׂה אִתְּכֶם חֶסֶד כְּמוֹ שֶׁעֲשִׂיתֶן עִם הַנִּפְטָרִים וְאִתִּי.
ט יִתֵּ֤ן יְהֹוָה֙ לָכֶ֔ם וּמְצֶ֣אןָ מְנוּחָ֔ה אִשָּׁ֖ה בֵּ֣ית אִישָׁ֑הּ וַתִּשַּׁ֣ק לָהֶ֔ן וַתִּשֶּׂ֥אנָה קוֹלָ֖ן וַתִּבְכֶּֽינָה׃ ה' יְבָרֵךְ אֶתְכֶן שֶׁתִּמְצְאוּ כָּל אַחַת שַׁלְוָה בְּבֵית בַּעֲלָה הֶחָדָשׁ, הִיא נָשְׁקָה לָהֶן וְהֵן בָּכוּ בְּקוֹל.
י וַתֹּאמַ֖רְנָה־לָּ֑הּ כִּֽי־אִתָּ֥ךְ נָשׁ֖וּב לְעַמֵּֽךְ׃ הֵן אָמְרוּ לָהּ: לֹא, אֶלָּא נַחֲזוֹר אִתָּךְ, לְעַמֵּךְ!
יא וַתֹּ֤אמֶר נׇעֳמִי֙ שֹׁ֣בְנָה בְנֹתַ֔י לָ֥מָּה תֵלַ֖כְנָה עִמִּ֑י הַֽעֽוֹד־לִ֤י בָנִים֙ בְּֽמֵעַ֔י וְהָי֥וּ לָכֶ֖ם לַאֲנָשִֽׁים׃ אָמְרָה נָעֳמִי: חִזְרוּ בְּנוֹתַי, לָמָּה לָכֵן לָלֶכֶת אִתִּי, וְכִי יְכוֹלָה אֲנִי עֲדַיִן לָלֶדֶת בָּנִים שֶׁיּוּכְלוּ לִהְיוֹת בְּעָלִים שֶׁלָּכֵן?!
יב שֹׁ֤בְנָה בְנֹתַי֙ לֵ֔כְןָ כִּ֥י זָקַ֖נְתִּי מִהְי֣וֹת לְאִ֑ישׁ כִּ֤י אָמַ֙רְתִּי֙ יֶשׁ־לִ֣י תִקְוָ֔ה גַּ֣ם הָיִ֤יתִי הַלַּ֙יְלָה֙ לְאִ֔ישׁ וְגַ֖ם יָלַ֥דְתִּי בָנִֽים׃ חִזְרוּ בְּנוֹתַי, לְכוּ, כִּי אֲנִי זְקֵנָה מִדַּי מִלְּהִנָּשֵׂא לְאִישׁ, כִּי גַּם אִם אוֹמַר שֶׁיֵּשׁ לִי סִכּוּי, גַּם אִם אֶנָּשֵׂא הַלַּיְלָה לְאִישׁ וְגַם אֵלֵד בָּנִים.
יג הֲלָהֵ֣ן ׀ תְּשַׂבֵּ֗רְנָה עַ֚ד אֲשֶׁ֣ר יִגְדָּ֔לוּ הֲלָהֵן֙ תֵּֽעָגֵ֔נָה לְבִלְתִּ֖י הֱי֣וֹת לְאִ֑ישׁ אַ֣ל בְּנֹתַ֗י כִּֽי־מַר־לִ֤י מְאֹד֙ מִכֶּ֔ם כִּֽי־יָצְאָ֥ה בִ֖י יַד־יְהֹוָֽה׃ וְכִי תְּחַכּוּ לָהֶם עַד שֶׁיִּגְדְּלוּ? וְכִי בִּשְׁבִילָם תִּשָּׁאֲרוּ עֲגוּנוֹת וְלֹא תִּנָּשְׂאוּ לְאִישׁ?! אַל תַּעֲשׂוּ כָּךְ בְּנוֹתַי, כִּי אֲנִי מִצְטַעֶרֶת מְאוֹד עֲבוּרְכֶן, כִּי הִכְּתָה בִּי הַיָּד שֶׁל ה'.
יד וַתִּשֶּׂ֣נָה קוֹלָ֔ן וַתִּבְכֶּ֖ינָה ע֑וֹד וַתִּשַּׁ֤ק עׇרְפָּה֙ לַחֲמוֹתָ֔הּ וְר֖וּת דָּ֥בְקָה בָּֽהּ׃ הֵן בָּכוּ עוֹד בְּקוֹל, עָרְפָּה נָשְׁקָה לַחֲמוֹתָהּ וְנִפְרְדָה, וְרוּת נִשְׁאֲרָה צְמוּדָה אֵלֶיהָ.
טו וַתֹּ֗אמֶר הִנֵּה֙ שָׁ֣בָה יְבִמְתֵּ֔ךְ אֶל־עַמָּ֖הּ וְאֶל־אֱלֹהֶ֑יהָ שׁ֖וּבִי אַחֲרֵ֥י יְבִמְתֵּֽךְ׃ אָמְרָה נָעֳמִי: הִנֵּה גִּיסָתֵךְ חָזְרָה לָעָם וְלַדָּת שֶׁלָּהּ, חִזְרִי גַּם אַתְּ כְּמוֹ גִּיסָתֵךְ!
טז וַתֹּ֤אמֶר רוּת֙ אַל־תִּפְגְּעִי־בִ֔י לְעׇזְבֵ֖ךְ לָשׁ֣וּב מֵאַחֲרָ֑יִךְ כִּ֠י אֶל־אֲשֶׁ֨ר תֵּלְכִ֜י אֵלֵ֗ךְ וּבַאֲשֶׁ֤ר תָּלִ֙ינִי֙ אָלִ֔ין עַמֵּ֣ךְ עַמִּ֔י וֵאלֹהַ֖יִךְ אֱלֹהָֽי׃ אָמְרָה רוּת: אַל תַּפְצִירִי בִּי לַעֲזוֹב אוֹתָךְ וְלַחֲזוֹר מֵאִתָּךְ, כִּי אֶל כָּל מָקוֹם שֶׁתֵּלְכִי אֵלֵךְ וּבַמָּקוֹם שֶׁתִּשְׁהִי בַּלַּיְלָה אֶשְׁהֶה, עַמֵּךְ הוּא עַמִּי וֶאֱלוֹהַיִךְ הוּא אֱלוֹהַי!
יז בַּאֲשֶׁ֤ר תָּמ֙וּתִי֙ אָמ֔וּת וְשָׁ֖ם אֶקָּבֵ֑ר כֹּה֩ יַעֲשֶׂ֨ה יְהֹוָ֥ה לִי֙ וְכֹ֣ה יוֹסִ֔יף כִּ֣י הַמָּ֔וֶת יַפְרִ֖יד בֵּינִ֥י וּבֵינֵֽךְ׃ בַּמָּקוֹם שֶׁתָּמוּתִי אָמוּת וְשָׁם אֶקָּבֵר, שֶׁה' יַעֲנִישׁ אוֹתִי עוֹד וָעוֹד אִם לֹא אֲקַיֵּם אֶת שְׁבוּעָתִי זוֹ, כִּי רַק הַמָּוֶת יַפְרִיד בֵּינִי לְבֵינֵךְ.
יח וַתֵּ֕רֶא כִּֽי־מִתְאַמֶּ֥צֶת הִ֖יא לָלֶ֣כֶת אִתָּ֑הּ וַתֶּחְדַּ֖ל לְדַבֵּ֥ר אֵלֶֽיהָ׃ נָעֳמִי רָאֲתָה שֶׁהִיא כָּל כָּךְ מִתְאַמֶּצֶת לָלֶכֶת אִתָּהּ, וְהִיא הִפְסִיקָה לְדַבֵּר אֵלֶיהָ.
יט וַתֵּלַ֣כְנָה שְׁתֵּיהֶ֔ם עַד־בּוֹאָ֖נָה בֵּ֣ית לָ֑חֶם וַיְהִ֗י כְּבוֹאָ֙נָה֙ בֵּ֣ית לֶ֔חֶם וַתֵּהֹ֤ם כׇּל־הָעִיר֙ עֲלֵיהֶ֔ן וַתֹּאמַ֖רְנָה הֲזֹ֥את נׇעֳמִֽי׃ שְׁתֵּיהֶן הָלְכוּ עַד שֶׁהִגִּיעוּ לְבֵית לֶחֶם, כְּשֶׁהֵן נִכְנְסוּ לְבֵית לֶחֶם כָּל הָעִיר תָּמְהָה עֲלֵיהֶן, וְאָמְרוּ: הֲאִם זֹאת נָעֳמִי?!
כ וַתֹּ֣אמֶר אֲלֵיהֶ֔ן אַל־תִּקְרֶ֥אנָה לִ֖י נׇעֳמִ֑י קְרֶ֤אןָ לִי֙ מָרָ֔א כִּי־הֵמַ֥ר שַׁדַּ֛י לִ֖י מְאֹֽד׃ אָמְרָה לָהֶן: אַל תִּקְרְאוּ לִי נָעֳמִי, קִרְאוּ לִי מָרָה כִּי ה' נָתַן לִי מְרִירוּת רַבָּה.
כא אֲנִי֙ מְלֵאָ֣ה הָלַ֔כְתִּי וְרֵיקָ֖ם הֱשִׁיבַ֣נִי יְהֹוָ֑ה לָ֣מָּה תִקְרֶ֤אנָה לִי֙ נׇעֳמִ֔י וַֽיהֹוָה֙ עָ֣נָה בִ֔י וְשַׁדַּ֖י הֵ֥רַֽע לִֽי׃ הָלַכְתִּי מִכָּאן כְּשֶׁאֲנִי מְלֵאָה וַה' הֶחְזִיר אוֹתִי רֵיקָה, לָמָּה תִּקְרְאוּ לִי נָעֳמִי? וַה' הֵשִׁיב לִי כִּגְמוּלִי וְהֵרַע לִי!
כב וַתָּ֣שׇׁב נׇעֳמִ֗י וְר֨וּת הַמּוֹאֲבִיָּ֤ה כַלָּתָהּ֙ עִמָּ֔הּ הַשָּׁ֖בָה מִשְּׂדֵ֣י מוֹאָ֑ב וְהֵ֗מָּה בָּ֚אוּ בֵּ֣ית לֶ֔חֶם בִּתְחִלַּ֖ת קְצִ֥יר שְׂעֹרִֽים׃ נָעֳמִי חָזְרָה וְאִתָּהּ רוּת הַמּוֹאֲבִית כַּלָּתָהּ שֶׁחָזְרָה אִתָּהּ מֵאֶרֶץ מוֹאָב, וְהֵן הִגִּיעוּ לְבֵית לֶחֶם בִּתְחִלַּת קְצִיר הַשְּׂעוֹרִים.
ב א וּֽלְנׇעֳמִ֞י מידע מוֹדָ֣ע לְאִישָׁ֗הּ אִ֚ישׁ גִּבּ֣וֹר חַ֔יִל מִמִּשְׁפַּ֖חַת אֱלִימֶ֑לֶךְ וּשְׁמ֖וֹ בֹּֽעַז׃ לְנָעֳמִי הָיָה קְרוֹב מִשְׁפָּחָה שֶׁל בַּעֲלָהּ, לוֹחֵם גִּבּוֹר, מִמִּשְׁפַּחַת אֱלִימֶלֶךְ, וּשְׁמוֹ בּוֹעַז.
ב וַתֹּ֩אמֶר֩ ר֨וּת הַמּוֹאֲבִיָּ֜ה אֶֽל־נׇעֳמִ֗י אֵֽלְﬞכָה־נָּ֤א הַשָּׂדֶה֙ וַאֲלַקֳּטָ֣ה בַֽשִּׁבֳּלִ֔ים אַחַ֕ר אֲשֶׁ֥ר אֶמְצָא־חֵ֖ן בְּעֵינָ֑יו וַתֹּ֥אמֶר לָ֖הּ לְכִ֥י בִתִּֽי׃ אָמְרָה רוּת הַמּוֹאֲבִית לְנָעֳמִי: אוּלַי אֵלֵךְ לַשָּׂדֶה וַאֲלַקֵּט שִׁבֳּלִים בְּעִקְבוֹת מִי שֶׁאֶמְצָא חֵן בְּעֵינָיו? אָמְרָה לָהּ: לְכִי בִּתִּי.
ג וַתֵּ֤לֶךְ וַתָּבוֹא֙ וַתְּלַקֵּ֣ט בַּשָּׂדֶ֔ה אַחֲרֵ֖י הַקֹּצְרִ֑ים וַיִּ֣קֶר מִקְרֶ֔הָ חֶלְקַ֤ת הַשָּׂדֶה֙ לְבֹ֔עַז אֲשֶׁ֖ר מִמִּשְׁפַּ֥חַת אֱלִימֶֽלֶךְ׃ הִיא הָלְכָה וְחָזְרָה וְלִקְּטָה בַּשָּׂדֶה אַחֲרֵי הַקּוֹצְרִים, קָרָה הַמִּקְרֶה וְנִזְדַמְּנָה לָהּ חֶלְקַת הַשָּׂדֶה שֶׁל בּוֹעַז שֶׁמִּמִּשְׁפַּחַת אֱלִימֶלֶךְ.
ד וְהִנֵּה־בֹ֗עַז בָּ֚א מִבֵּ֣ית לֶ֔חֶם וַיֹּ֥אמֶר לַקּוֹצְרִ֖ים יְהֹוָ֣ה עִמָּכֶ֑ם וַיֹּ֥אמְרוּ ל֖וֹ יְבָרֶכְךָ֥ יְהֹוָֽה׃ וְאָז בּוֹעַז הִגִּיעַ מִבֵּית לֶחֶם, הוּא אָמַר לַקּוֹצְרִים: ה' עִמָּכֶם! עָנוּ לוֹ: ה' יְבָרֵךְ אוֹתְךָ!
ה וַיֹּ֤אמֶר בֹּ֙עַז֙ לְֽנַעֲר֔וֹ הַנִּצָּ֖ב עַל־הַקּוֹצְרִ֑ים לְמִ֖י הַנַּעֲרָ֥ה הַזֹּֽאת׃ אָמַר בּוֹעַז לְעוֹזְרוֹ שֶׁמַּשְׁגִּיחַ עַל הַקּוֹצְרִים: שֶׁל מִי הַנַּעֲרָה הַזֹּאת?
ו וַיַּ֗עַן הַנַּ֛עַר הַנִּצָּ֥ב עַל־הַקּוֹצְרִ֖ים וַיֹּאמַ֑ר נַעֲרָ֤ה מֽוֹאֲבִיָּה֙ הִ֔יא הַשָּׁ֥בָה עִֽם־נׇעֳמִ֖י מִשְּׂדֵ֥י מוֹאָֽב׃ עָנָה הַנַּעַר שֶׁמַּשְׁגִּיחַ עַל הַקּוֹצְרִים וְאָמַר: הִיא נַעֲרָה מוֹאֲבִית שֶׁחָזְרָה עִם נָעֳמִי מֵאֶרֶץ מוֹאָב.
ז וַתֹּ֗אמֶר אֲלַקֳּטָה־נָּא֙ וְאָסַפְתִּ֣י בָעֳמָרִ֔ים אַחֲרֵ֖י הַקּוֹצְרִ֑ים וַתָּב֣וֹא וַֽתַּעֲמ֗וֹד מֵאָ֤ז הַבֹּ֙קֶר֙ וְעַד־עַ֔תָּה זֶ֛ה שִׁבְתָּ֥הּ הַבַּ֖יִת מְעָֽט׃ הִיא אָמְרָה: אֲלַקֵּט וְאֶאֱסוֹף אַחֲרֵי הַקּוֹצְרִים בֵּין הָעוֹמָרִים, וְהִיא בָּאָה וְהָיְתָה פֹּה מֵאָז הַבּוֹקֶר וְעַד עַכְשָׁיו, וְזֶה הַמְּעַט שֶׁאָסְפָה לְהָבִיא הַבַּיְתָה.
ח וַיֹּ֩אמֶר֩ בֹּ֨עַז אֶל־ר֜וּת הֲל֧וֹא שָׁמַ֣עַתְּ בִּתִּ֗י אַל־תֵּֽלְכִי֙ לִלְקֹט֙ בְּשָׂדֶ֣ה אַחֵ֔ר וְגַ֛ם לֹ֥א תַעֲבוּרִ֖י מִזֶּ֑ה וְכֹ֥ה תִדְבָּקִ֖ין עִם־נַעֲרֹתָֽי׃ אָמַר בּוֹעַז לְרוּת: שִׁמְעִי בִּתִּי, אַל תֵּלְכִי לְלַקֵּט בְּשָׂדֶה אַחֵר וְאַל תַּעַבְרִי מִכָּאן, תַּמְשִׁיכִי כָּכָה לְהִצְטָרֵף לְנַעֲרוֹתַי.
ט עֵינַ֜יִךְ בַּשָּׂדֶ֤ה אֲשֶׁר־יִקְצֹרוּן֙ וְהָלַ֣כְתְּ אַחֲרֵיהֶ֔ן הֲל֥וֹא צִוִּ֛יתִי אֶת־הַנְּעָרִ֖ים לְבִלְתִּ֣י נׇגְעֵ֑ךְ וְצָמִ֗ת וְהָלַכְתְּ֙ אֶל־הַכֵּלִ֔ים וְשָׁתִ֕ית מֵאֲשֶׁ֥ר יִשְׁאֲב֖וּן הַנְּעָרִֽים׃ חַפְּשִׂי בַּשָּׂדֶה שֶׁיִּקְצְרוּ וְתֵלְכִי אַחֲרֵיהֶן, הֲרֵי צִוִּיתִי אֶת הַנְּעָרִים שֶׁלֹּא יִגְּעוּ בָּךְ, וּכְשֶׁתִּהְיִי צְמֵאָה תֵּלְכִי לִכְלֵי הַמַּיִם וְתִשְׁתִּי מִמַּה שֶּׁהַנְּעָרִים שׁוֹאֲבִים.
י וַתִּפֹּל֙ עַל־פָּנֶ֔יהָ וַתִּשְׁתַּ֖חוּ אָ֑רְצָה וַתֹּ֣אמֶר אֵלָ֗יו מַדּ֩וּעַ֩ מָצָ֨אתִי חֵ֤ן בְּעֵינֶ֙יךָ֙ לְהַכִּירֵ֔נִי וְאָנֹכִ֖י נׇכְרִיָּֽה׃ הִיא הִשְׁתַּטְּחָה לְפָנֶיהָ וְהִשְׁתַּחַוְתָה עַל הָאָרֶץ, וְאָמְרָה לוֹ: אֵיךְ מָצָאתִי חֵן בְּעֵינֶיךָ כְּאִלּוּ אַתָּה מַכִּיר אוֹתִי, הֲרֵי אֲנִי זָרָה?
יא וַיַּ֤עַן בֹּ֙עַז֙ וַיֹּ֣אמֶר לָ֔הּ הֻגֵּ֨ד הֻגַּ֜ד לִ֗י כֹּ֤ל אֲשֶׁר־עָשִׂית֙ אֶת־חֲמוֹתֵ֔ךְ אַחֲרֵ֖י מ֣וֹת אִישֵׁ֑ךְ וַתַּֽעַזְבִ֞י אָבִ֣יךְ וְאִמֵּ֗ךְ וְאֶ֙רֶץ֙ מֽוֹלַדְתֵּ֔ךְ וַתֵּ֣לְכִ֔י אֶל־עַ֕ם אֲשֶׁ֥ר לֹא־יָדַ֖עַתְּ תְּמ֥וֹל שִׁלְשֽׁוֹם׃ עָנָה בּוֹעַז וְאָמַר לָהּ: סִפְּרוּ לִי עַל כָּל מַה שֶּׁעָשִׂית עִם חֲמוֹתֵךְ אַחֲרֵי שֶׁבַּעֲלֵךְ מֵת, שֶׁעָזַבְתְּ אֶת אָבִיךְ וְאִמֵּךְ וְאֶת אֶרֶץ מוֹלַדְתֵּךְ וְהָלַכְתְּ אֶל עַם שֶׁלֹּא הִכַּרְתְּ אֶתְמוֹל וְשִׁלְשׁוֹם.
יב יְשַׁלֵּ֥ם יְהֹוָ֖ה פׇּעֳלֵ֑ךְ וּתְהִ֨י מַשְׂכֻּרְתֵּ֜ךְ שְׁלֵמָ֗ה מֵעִ֤ם יְהֹוָה֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁר־בָּ֖את לַחֲס֥וֹת תַּֽחַת־כְּנָפָֽיו׃ ה' יְשַׁלֵּם לָךְ עַל פְּעֻלָּתֵךְ וְיִהְיֶה לָךְ שָׂכָר שָׁלֵם מֵה' אֱלֹהֵי יִשְׂרָאֵל שֶׁבָּאת לַחֲסוֹת תַּחַת חֲסוּתוֹ.
יג וַ֠תֹּ֠אמֶר אֶמְצָא־חֵ֨ן בְּעֵינֶ֤יךָ אֲדֹנִי֙ כִּ֣י נִֽחַמְתָּ֔נִי וְכִ֥י דִבַּ֖רְתָּ עַל־לֵ֣ב שִׁפְחָתֶ֑ךָ וְאָֽנֹכִי֙ לֹ֣א אֶהְיֶ֔ה כְּאַחַ֖ת שִׁפְחֹתֶֽיךָ׃ הִיא אָמְרָה: אֶמְצָא חֵן בְּעֵינֶיךָ אֲדוֹנִי כִּי נִחַמְתָּ אוֹתִי וְנָגַעְתָּ בִּדְבָרֶיךָ לַלֵּב שֶׁל שִׁפְחָתְךָ, וְאֵינֶנִּי אֲפִלּוּ כְּמוֹ אַחַת מִשִּׁפְחוֹתֶיךָ.
יד וַיֹּ֩אמֶר֩ לָ֨הֿ בֹ֜עַז לְעֵ֣ת הָאֹ֗כֶל גֹּ֤שִֽׁי הֲלֹם֙ וְאָכַ֣לְתְּ מִן־הַלֶּ֔חֶם וְטָבַ֥לְתְּ פִּתֵּ֖ךְ בַּחֹ֑מֶץ וַתֵּ֙שֶׁב֙ מִצַּ֣ד הַקֹּצְרִ֔ים וַיִּצְבׇּט־לָ֣הּ קָלִ֔י וַתֹּ֥אכַל וַתִּשְׂבַּ֖ע וַתֹּתַֽר׃ אָמַר לָהּ בּוֹעַז בִּזְמַן הָאוֹכֶל: גְּשִׁי לְכָאן וְתֹאכְלִי מֵהָאוֹכֶל וְתִטְבְּלִי אֶת לַחְמֵךְ בַּחוֹמֶץ. הִיא יָשְׁבָה לְצַד הַקּוֹצְרִים, הוּא הֵבִיא לָהּ חוֹפֶן שְׂעוֹרִים קְלוּיִים, הִיא אָכְלָה וְשָׂבְעָה וְגַם נִשְׁאַר לָהּ.
טו וַתָּ֖קׇם לְלַקֵּ֑ט וַיְצַו֩ בֹּ֨עַז אֶת־נְעָרָ֜יו לֵאמֹ֗ר גַּ֣ם בֵּ֧ין הָעֳמָרִ֛ים תְּלַקֵּ֖ט וְלֹ֥א תַכְלִימֽוּהָ׃ הִיא קָמָה לְהַמְשִׁיךְ לְלַקֵּט, וּבוֹעַז צִוָּה אֶת עוֹבְדָיו: תְּנוּ לָהּ לְלַקֵּט גַּם בֵּין הָעֳמָרִים, וְאַל תְּבַיְּשׁוּ אוֹתָהּ.
טז וְגַ֛ם שֹׁל־תָּשֹׁ֥לּוּ לָ֖הּ מִן־הַצְּבָתִ֑ים וַעֲזַבְתֶּ֥ם וְלִקְּטָ֖ה וְלֹ֥א תִגְעֲרוּ־בָֽהּ׃ וְגַם תַּפִּילוּ בִּשְׁבִילָהּ מֵהַחֲבִילוֹת וְתַשְׁאִירוּ, וְהִיא תֶּאֱסוֹף וְלֹא תִּצְעֲקוּ עָלֶיהָ!
יז וַתְּלַקֵּ֥ט בַּשָּׂדֶ֖ה עַד־הָעָ֑רֶב וַתַּחְבֹּט֙ אֵ֣ת אֲשֶׁר־לִקֵּ֔טָה וַיְהִ֖י כְּאֵיפָ֥ה שְׂעֹרִֽים׃ הִיא לִקְּטָה בַּשָּׂדֶה עַד הָעֶרֶב, הִיא חָבְטָה אֶת הַשִּׁבֳּלִים שֶׁלִּקְּטָה וְהָיָה לָהּ כְּאֵיפָה (כְּ22 לִיטֶר) שְׂעוֹרִים.
יח וַתִּשָּׂא֙ וַתָּב֣וֹא הָעִ֔יר וַתֵּ֥רֶא חֲמוֹתָ֖הּ אֵ֣ת אֲשֶׁר־לִקֵּ֑טָה וַתּוֹצֵא֙ וַתִּתֶּן־לָ֔הּ אֵ֥ת אֲשֶׁר־הוֹתִ֖רָה מִשׇּׂבְעָֽהּ׃ הִיא נָשְׂאָה אוֹתָם וּבָאָה לָעִיר, חֲמוֹתָהּ רָאֲתָה אֶת מַה שֶּׁלִּקְּטָה, וְרוּת הוֹצִיאָה וְנָתְנָה לָהּ אֶת הָאוֹכֶל שֶׁנִּשְׁאַר לָהּ אַחֲרֵי שֶׁשָּׂבְעָה.
יט וַתֹּ֩אמֶר֩ לָ֨הּ חֲמוֹתָ֜הּ אֵיפֹ֨ה לִקַּ֤טְתְּ הַיּוֹם֙ וְאָ֣נָה עָשִׂ֔ית יְהִ֥י מַכִּירֵ֖ךְ בָּר֑וּךְ וַתַּגֵּ֣ד לַחֲמוֹתָ֗הּ אֵ֤ת אֲשֶׁר־עָֽשְׂתָה֙ עִמּ֔וֹ וַתֹּ֗אמֶר שֵׁ֤ם הָאִישׁ֙ אֲשֶׁ֨ר עָשִׂ֧יתִי עִמּ֛וֹ הַיּ֖וֹם בֹּֽעַז׃ אָמְרָה לָהּ חֲמוֹתָהּ: אֵיפֹה לִקַּטְתְּ הַיּוֹם? לְאָן הָלַכְתְּ לַעֲשׂוֹת? שֶׁיִּהְיֶה זֶה שֶׁהִכִּיר אוֹתָךְ מְבוֹרָךְ! הִיא סִפְּרָה לַחֲמוֹתָהּ אֵצֶל מִי הִיא עָשְׂתָה, וְאָמְרָה: הַשֵּׁם שֶׁל הָאִישׁ שֶׁעָשִׂיתִי אֶצְלוֹ הַיּוֹם הוּא בּוֹעַז.
כ וַתֹּ֨אמֶר נׇעֳמִ֜י לְכַלָּתָ֗הּ בָּר֥וּךְ הוּא֙ לַיהֹוָ֔ה אֲשֶׁר֙ לֹא־עָזַ֣ב חַסְדּ֔וֹ אֶת־הַחַיִּ֖ים וְאֶת־הַמֵּתִ֑ים וַתֹּ֧אמֶר לָ֣הּ נׇעֳמִ֗י קָר֥וֹב לָ֙נוּ֙ הָאִ֔ישׁ מִֽגֹּאֲלֵ֖נוּ הֽוּא׃ אָמְרָה נָעֳמִי לְכַלָּתָהּ: מְבוֹרָךְ הוּא לַה' עַל שֶׁבְּחַסְדּוֹ לֹא עָזַב לֹא אֶת הַחַיִּים וְלֹא אֶת הַמֵּתִים! אָמְרָה לָהּ נָעֳמִי: הָאִישׁ הוּא קָרוֹב שֶׁלָּנוּ, הוּא מִגּוֹאֲלֵנוּ!
כא וַתֹּ֖אמֶר ר֣וּת הַמּוֹאֲבִיָּ֑ה גַּ֣ם ׀ כִּי־אָמַ֣ר אֵלַ֗י עִם־הַנְּעָרִ֤ים אֲשֶׁר־לִי֙ תִּדְבָּקִ֔ין עַ֣ד אִם־כִּלּ֔וּ אֵ֥ת כׇּל־הַקָּצִ֖יר אֲשֶׁר־לִֽי׃ אָמְרָה רוּת הַמּוֹאֲבִית: הוּא גַּם אָמַר לִי: תִּצְטָרְפִי לַנְּעָרִים הָעוֹבְדִים שֶׁלִּי עַד שֶׁיְּסַיְּמוּ אֶת כָּל הַקָּצִיר שֶׁלִּי.
כב וַתֹּ֥אמֶר נׇעֳמִ֖י אֶל־ר֣וּת כַּלָּתָ֑הּ ט֣וֹב בִּתִּ֗י כִּ֤י תֵֽצְאִי֙ עִם־נַ֣עֲרוֹתָ֔יו וְלֹ֥א יִפְגְּעוּ־בָ֖ךְ בְּשָׂדֶ֥ה אַחֵֽר׃ אָמְרָה נָעֳמִי אֶל רוּת כַּלָּתָהּ: עָדִיף בִּתִּי שֶׁתֵּצְאִי עִם נַעֲרוֹתָיו, וְכָךְ לֹא יִפְגְּעוּ בָּךְ בְּשָׂדֶה אַחֵר.
כג וַתִּדְבַּ֞ק בְּנַעֲר֥וֹת בֹּ֙עַז֙ לְלַקֵּ֔ט עַד־כְּל֥וֹת קְצִֽיר־הַשְּׂעֹרִ֖ים וּקְצִ֣יר הַֽחִטִּ֑ים וַתֵּ֖שֶׁב אֶת־חֲמוֹתָֽהּ׃ הִיא הִצְטָרְפָה לַנַּעֲרוֹת שֶׁל בּוֹעַז לְלַקֵּט, עַד שֶׁנִּגְמַר קְצִיר הַשְּׂעוֹרִים וּקְצִיר הַחִטִּים, וְהִיא הָלְכָה לָגוּר עִם חֲמוֹתָהּ.
ג א וַתֹּ֥אמֶר לָ֖הּ נׇעֳמִ֣י חֲמוֹתָ֑הּ בִּתִּ֕י הֲלֹ֧א אֲבַקֶּשׁ־לָ֛ךְ מָנ֖וֹחַ אֲשֶׁ֥ר יִֽיטַב־לָֽךְ׃ אָמְרָה לָהּ נָעֳמִי חֲמוֹתָהּ: בִּתִּי, הֲרֵי אֲנִי רוֹצָה שֶׁיִּהְיֶה לָךְ שַׁלְוָה – שֶׁיִּיטַב לָךְ.
ב וְעַתָּ֗ה הֲלֹ֥א בֹ֙עַז֙ מֹֽדַעְתָּ֔נוּ אֲשֶׁ֥ר הָיִ֖ית אֶת־נַעֲרוֹתָ֑יו הִנֵּה־ה֗וּא זֹרֶ֛ה אֶת־גֹּ֥רֶן הַשְּׂעֹרִ֖ים הַלָּֽיְלָה׃ וְעַכְשָׁיו, הֲרֵי בּוֹעַז קְרוֹב מִשְׁפַּחְתֵּנוּ שֶׁהָיִית עִם נַעֲרוֹתָיו, הוּא זוֹרֶה אֶת גּוֹרֶן הַשְּׂעוֹרִים הַלַּיְלָה.
ג וְרָחַ֣צְתְּ ׀ וָסַ֗כְתְּ וְשַׂ֧מְתְּ שמלתך שִׂמְלֹתַ֛יִךְ עָלַ֖יִךְ וירדתי וְיָרַ֣דְתְּ הַגֹּ֑רֶן אַל־תִּוָּדְעִ֣י לָאִ֔ישׁ עַ֥ד כַּלֹּת֖וֹ לֶאֱכֹ֥ל וְלִשְׁתּֽוֹת׃ תִּתְרַחֲצִי, תְּטַפְּחִי אֶת עַצְמֵךְ, תִּלְבְּשִׁי אֶת שִׂמְלוֹתַיִךְ, וְתֵרְדִי לַגּוֹרֶן; אַל תְּגַלִּי אֶת עַצְמֵךְ לָאִישׁ עַד שֶׁיְּסַיֵּם לֶאֱכוֹל וְלִשְׁתּוֹת.
ד וִיהִ֣י בְשׇׁכְב֗וֹ וְיָדַ֙עַתְּ֙ אֶת־הַמָּקוֹם֙ אֲשֶׁ֣ר יִשְׁכַּב־שָׁ֔ם וּבָ֛את וְגִלִּ֥ית מַרְגְּלֹתָ֖יו ושכבתי וְשָׁכָ֑בְתְּ וְהוּא֙ יַגִּ֣יד לָ֔ךְ אֵ֖ת אֲשֶׁ֥ר תַּעֲשִֽׂין׃ וּכְשֶׁהוּא יֵלֵךְ לִשְׁכַּב, תְּוַדְּאִי מָה הַמָּקוֹם שֶׁהוּא שׁוֹכֵב שָׁם, וְתָבוֹאִי וְתָרִימִי אֶת הַכִּסּוּי לְצַד רַגְלָיו וְתִשְׁכְּבִי שָׁם, וְהוּא כְּבָר יַגִּיד לָךְ מַה לַּעֲשׂוֹת.
ה וַתֹּ֖אמֶר אֵלֶ֑יהָ כֹּ֛ל אֲשֶׁר־תֹּאמְרִ֥י [אֵלַ֖י] אֶֽעֱשֶֽׂה׃ הִיא אָמְרָה לָהּ: כָּל מַה שֶּׁתֹּאמְרִי לִי אֶעֱשֶׂה.
ו וַתֵּ֖רֶד הַגֹּ֑רֶן וַתַּ֕עַשׂ כְּכֹ֥ל אֲשֶׁר־צִוַּ֖תָּה חֲמוֹתָֽהּ׃ הִיא יָרְדָה לַגּוֹרֶן וְעָשְׂתָה אֶת כָּל מַה שֶׁצִּוְּתָה אוֹתָהּ חֲמוֹתָהּ.
ז וַיֹּ֨אכַל בֹּ֤עַז וַיֵּשְׁתְּ֙ וַיִּיטַ֣ב לִבּ֔וֹ וַיָּבֹ֕א לִשְׁכַּ֖ב בִּקְצֵ֣ה הָעֲרֵמָ֑ה וַתָּבֹ֣א בַלָּ֔ט וַתְּגַ֥ל מַרְגְּלֹתָ֖יו וַתִּשְׁכָּֽב׃ בּוֹעַז אָכַל וְשָׁתָה וְהִרְגִּישׁ טוֹב וְהוּא בָּא לִשְׁכַּב בִּקְצֵה הָעֲרֵמָה, הִיא בָּאָה בַּסֵּתֶר הֵרִימָה אֶת הַכִּסּוּי לְצַד רַגְלָיו וְשָׁכְבָה שָׁם.
ח וַֽיְהִי֙ בַּחֲצִ֣י הַלַּ֔יְלָה וַיֶּחֱרַ֥ד הָאִ֖ישׁ וַיִּלָּפֵ֑ת וְהִנֵּ֣ה אִשָּׁ֔ה שֹׁכֶ֖בֶת מַרְגְּלֹתָֽיו׃ בְּאֶמְצַע הַלַּיְלָה הָאִישׁ נִבְהַל וְהִתְכַּוֵּץ, וְהוּא גִּלָּה שֶׁאִשָּׁה שׁוֹכֶבֶת לְצַד רַגְלָיו.
ט וַיֹּ֖אמֶר מִי־אָ֑תְּ וַתֹּ֗אמֶר אָֽנֹכִי֙ ר֣וּת אֲמָתֶ֔ךָ וּפָרַשְׂתָּ֤ כְנָפֶ֙ךָ֙ עַל־אֲמָ֣תְךָ֔ כִּ֥י גֹאֵ֖ל אָֽתָּה׃ אָמַר לָהּ: מִי אַתְּ? אָמְרָה: אֲנִי רוּת שִׁפְחָתְךָ, פְּרוֹשׂ אֶת חֲסוּתְךָ עַל שִׁפְחָתְךָ כִּי אַתָּה קְרוֹב הַמִּשְׁפָּחָה הַגּוֹאֵל.
י וַיֹּ֗אמֶר בְּרוּכָ֨ה אַ֤תְּ לַֽיהֹוָה֙ בִּתִּ֔י הֵיטַ֛בְתְּ חַסְדֵּ֥ךְ הָאַחֲר֖וֹן מִן־הָרִאשׁ֑וֹן לְבִלְתִּי־לֶ֗כֶת אַֽחֲרֵי֙ הַבַּ֣חוּרִ֔ים אִם־דַּ֖ל וְאִם־עָשִֽׁיר׃ אָמַר לָהּ: בְּרוּכָה אַתְּ לַה' בִּתִּי! הֵיטַבְתְּ בְּחַסְדֵּךְ הָאַחֲרוֹן יוֹתֵר מִבָּרִאשׁוֹן – שֶׁלֹּא הָלַכְתְּ אַחֲרֵי הַבַּחוּרִים הַצְּעִירִים, בֵּין עָנִי וּבֵין עָשִׁיר.
יא וְעַתָּ֗ה בִּתִּי֙ אַל־תִּ֣ירְאִ֔י כֹּ֥ל אֲשֶׁר־תֹּאמְרִ֖י אֶעֱשֶׂה־לָּ֑ךְ כִּ֤י יוֹדֵ֙עַ֙ כׇּל־שַׁ֣עַר עַמִּ֔י כִּ֛י אֵ֥שֶׁת חַ֖יִל אָֽתְּ׃ וְעַכְשָׁיו בִּתִּי אַל תַּחְשְׁשִׁי, כָּל מַה שֶּׁתֹּאמְרִי אֶעֱשֶׂה עֲבוּרֵךְ, כִּי כָּל תּוֹשָׁבֵי עִירִי יוֹדְעִים שֶׁאַתְּ אֵשֶׁת חַיִל.
יב וְעַתָּה֙ כִּ֣י אׇמְנָ֔ם כִּ֥י (אם) גֹאֵ֖ל אָנֹ֑כִי וְגַ֛ם יֵ֥שׁ גֹּאֵ֖ל קָר֥וֹב מִמֶּֽנִּי׃ וְעַכְשָׁיו, אָמְנָם נָכוֹן שֶׁאֲנִי גּוֹאֵל, אֲבָל יֵשׁ גַּם גּוֹאֵל קָרוֹב יוֹתֵר מִמֶּנִּי.
יג לִ֣ינִי ׀ הַלַּ֗יְלָה וְהָיָ֤ה בַבֹּ֙קֶר֙ אִם־יִגְאָלֵ֥ךְ טוֹב֙ יִגְאָ֔ל וְאִם־לֹ֨א יַחְפֹּ֧ץ לְגׇאֳלֵ֛ךְ וּגְאַלְתִּ֥יךְ אָנֹ֖כִי חַי־יְהֹוָ֑ה שִׁכְבִ֖י עַד־הַבֹּֽקֶר׃ תִּשְׁנִי הַלַּיְלָה וְנִרְאֶה בַּבּוֹקֶר אִם הוּא יִגְאַל אוֹתָךְ – טוֹב, שֶׁיִּגְאַל, וְאִם הוּא לֹא יִרְצֶה לִגְאוֹל אוֹתָךְ, אֲנִי נִשְׁבַּע בְּחַיֵּי ה' שֶׁאֶגְאַל אוֹתָךְ, שִׁכְבִי פֹּה עַד הַבּוֹקֶר.
יד וַתִּשְׁכַּ֤ב מַרְגְּלוֹתָו֙ עַד־הַבֹּ֔קֶר וַתָּ֕קׇם בטרום בְּטֶ֛רֶם יַכִּ֥יר אִ֖ישׁ אֶת־רֵעֵ֑הוּ וַיֹּ֙אמֶר֙ אַל־יִוָּדַ֔ע כִּי־בָ֥אָה הָאִשָּׁ֖ה הַגֹּֽרֶן׃ הִיא שָׁכְבָה לְצַד רַגְלָיו עַד הַבּוֹקֶר וְקָמָה לִפְנֵי שֶׁאָדָם יָכוֹל לְזַהוֹת אֶת חֲבֵרוֹ כִּי הוּא רָצָה שֶׁלֹּא יֵדְעוּ שֶׁהָאִשָּׁה בָּאָה לַגּוֹרֶן.
טו וַיֹּ֗אמֶר הָ֠בִ֠י הַמִּטְפַּ֧חַת אֲשֶׁר־עָלַ֛יִךְ וְאֶחֳזִי־בָ֖הּ וַתֹּ֣אחֶז בָּ֑הּ וַיָּ֤מׇד שֵׁשׁ־שְׂעֹרִים֙ וַיָּ֣שֶׁת עָלֶ֔יהָ וַיָּבֹ֖א הָעִֽיר׃ הוּא אָמַר: תְּנִי אֶת הַמִּטְפַּחַת שֶׁעָלַיִךְ וְתַחֲזִיקִי בָּהּ, הִיא הֶחֱזִיקָה בָּהּ, הוּא מָדַד כַּמּוּת שֶׁל שֵׁשׁ סְאִים שֶׁל גַּרְעִינֵי שְׂעוֹרִים (כְּ7 לִיטֶר) וְשָׂם עָלֶיהָ וְהוּא בָּא לָעִיר.
טז וַתָּבוֹא֙ אֶל־חֲמוֹתָ֔הּ וַתֹּ֖אמֶר מִי־אַ֣תְּ בִּתִּ֑י וַתַּ֨גֶּד־לָ֔הּ אֵ֛ת כׇּל־אֲשֶׁ֥ר עָשָׂה־לָ֖הּ הָאִֽישׁ׃ הִיא בָּאָה אֶל חֲמוֹתָהּ, אָמְרָה לָהּ: שֶׁל מִי אַתְּ עַכְשָׁיו בִּתִּי? רוּת סִפְּרָה לָהּ אֶת כָּל מַה שֶּׁעָשָׂה לָהּ הָאִישׁ.
יז וַתֹּ֕אמֶר שֵׁשׁ־הַשְּׂעֹרִ֥ים הָאֵ֖לֶּה נָ֣תַן לִ֑י כִּ֚י אָמַ֣ר [אֵלַ֔י] אַל־תָּב֥וֹאִי רֵיקָ֖ם אֶל־חֲמוֹתֵֽךְ׃ וְאָמְרָה: אֶת שֵׁשׁ הַסְּאִים שֶׁל שְׂעוֹרִים הָאֵלֶּה הוּא נָתַן לִי כִּי הוּא אָמַר לִי: אַל תָּבוֹאִי אֶל חֲמוֹתֵךְ בְּלִי כְּלוּם.
יח וַתֹּ֙אמֶר֙ שְׁבִ֣י בִתִּ֔י עַ֚ד אֲשֶׁ֣ר תֵּֽדְעִ֔ין אֵ֖יךְ יִפֹּ֣ל דָּבָ֑ר כִּ֣י לֹ֤א יִשְׁקֹט֙ הָאִ֔ישׁ כִּֽי־אִם־כִּלָּ֥ה הַדָּבָ֖ר הַיּֽוֹם׃ אָמְרָה לָהּ: שְׁבִי בִּתִּי עַד שֶׁתֵּדְעִי אֵיךְ יִקְרוּ הַדְּבָרִים, כִּי הָאִישׁ לֹא יָנוּחַ אֶלָּא אִם הוּא יְסַיֵּם אֶת הָעִנְיָן הַיּוֹם.
ד א וּבֹ֨עַז עָלָ֣ה הַשַּׁ֘עַר֮ וַיֵּ֣שֶׁב שָׁם֒ וְהִנֵּ֨ה הַגֹּאֵ֤ל עֹבֵר֙ אֲשֶׁ֣ר דִּבֶּר־בֹּ֔עַז וַיֹּ֛אמֶר ס֥וּרָה שְׁבָה־פֹּ֖ה פְּלֹנִ֣י אַלְמֹנִ֑י וַיָּ֖סַר וַיֵּשֵֽׁב׃ וּבוֹעַז עָלָה לְשַּׁעַר הָעִיר וְיָשַׁב שָׁם עַד שֶׁרָאָה אֶת הַגּוֹאֵל שֶׁבּוֹעַז דִּבֵּר עָלָיו עוֹבֵר, אָמַר לוֹ: סוּר לְכָאן וְשֵׁב פֹּה פְּלוֹנִי אַלְמוֹנִי, הוּא סָר וְיָשַׁב.
ב וַיִּקַּ֞ח עֲשָׂרָ֧ה אֲנָשִׁ֛ים מִזִּקְנֵ֥י הָעִ֖יר וַיֹּ֣אמֶר שְׁבוּ־פֹ֑ה וַיֵּשֵֽׁבוּ׃ הוּא אָסַף עֲשָׂרָה אֲנָשִׁים מֵחַכְמֵי הָעִיר וְאָמַר: שְׁבוּ פֹּה, וְהֵם יָשְׁבוּ.
ג וַיֹּ֙אמֶר֙ לַגֹּאֵ֔ל חֶלְקַת֙ הַשָּׂדֶ֔ה אֲשֶׁ֥ר לְאָחִ֖ינוּ לֶאֱלִימֶ֑לֶךְ מָכְרָ֣ה נׇעֳמִ֔י הַשָּׁ֖בָה מִשְּׂדֵ֥ה מוֹאָֽב׃ אָמַר לַגּוֹאֵל: נָעֳמִי שֶׁחָזְרָה מֵאֶרֶץ מוֹאָב מָכְרָה אֶת חֶלְקַת הַשָּׂדֶה שֶׁל אֱלִימֶלֶךְ אָחִינוּ.
ד וַאֲנִ֨י אָמַ֜רְתִּי אֶגְלֶ֧ה אׇזְנְךָ֣ לֵאמֹ֗ר קְ֠נֵ֠ה נֶ֥גֶד הַיֹּשְׁבִים֮ וְנֶ֣גֶד זִקְנֵ֣י עַמִּי֒ אִם־תִּגְאַל֙ גְּאָ֔ל וְאִם־לֹ֨א יִגְאַ֜ל הַגִּ֣ידָה לִּ֗י ואדע וְאֵֽדְעָה֙ כִּ֣י אֵ֤ין זוּלָֽתְךָ֙ לִגְא֔וֹל וְאָנֹכִ֖י אַחֲרֶ֑יךָ וַיֹּ֖אמֶר אָנֹכִ֥י אֶגְאָֽל׃ וַאֲנִי גִּלִּיתִי לְךָ כְּדֵי שֶׁתִּקְנֶה אוֹתָהּ בִּפְנֵי הַיּוֹשְׁבִים פֹּה וּבִפְנֵי חַכְמֵי עִירֵנוּ. אִם תִּפְדֶּה – תִּפְדֶּה, וְאִם לֹא תִּפְדֶּה סַפֵּר לִי, וְאֵדַע – כִּי חוּץ מִמְּךָ אֵין מִי שֶׁיִּפְדֶּה וַאֲנִי אַחֲרֶיךָ. אָמַר לוֹ אֲנִי אֶפְדֶּה.
ה וַיֹּ֣אמֶר בֹּ֔עַז בְּיוֹם־קְנוֹתְךָ֥ הַשָּׂדֶ֖ה מִיַּ֣ד נׇעֳמִ֑י וּ֠מֵאֵ֠ת ר֣וּת הַמּוֹאֲבִיָּ֤ה אֵֽשֶׁת־הַמֵּת֙ קניתי קָנִ֔יתָ לְהָקִ֥ים שֵׁם־הַמֵּ֖ת עַל־נַחֲלָתֽוֹ׃ אָמַר בּוֹעַז: בַּיּוֹם שֶׁתִּקְנֶה אֶת הַשָּׂדֶה מִנָּעֳמִי, גַּם מֵרוּת הַמּוֹאֲבִית אֵשֶׁת הַמֵּת תִּקְנֶה אֶת זִכְרוֹ כְּדֵי לְהָקִים אֶת שֵׁם הַמֵּת עַל נַחֲלָתוֹ.
ו וַיֹּ֣אמֶר הַגֹּאֵ֗ל לֹ֤א אוּכַל֙ לגאול לִגְאׇל־לִ֔י פֶּן־אַשְׁחִ֖ית אֶת־נַחֲלָתִ֑י גְּאַל־לְךָ֤ אַתָּה֙ אֶת־גְּאֻלָּתִ֔י כִּ֥י לֹא־אוּכַ֖ל לִגְאֹֽל׃ אָמַר הַגּוֹאֵל: לֹא אוּכַל לִפְדּוֹת לִי כְּדֵי שֶׁלֹּא אַשְׁחִית אֶת מִשְׁפַּחְתִּי בְּאִשָּׁה מוֹאֲבִית, פְּדֵה לְךָ אַתָּה בַּזְּכוּת שֶׁלִּי לִפְדּוֹת, כִּי אֲנִי לֹא אוּכַל לִפְדּוֹת.
ז וְזֹאת֩ לְפָנִ֨ים בְּיִשְׂרָאֵ֜ל עַל־הַגְּאֻלָּ֤ה וְעַל־הַתְּמוּרָה֙ לְקַיֵּ֣ם כׇּל־דָּבָ֔ר שָׁלַ֥ף אִ֛ישׁ נַעֲל֖וֹ וְנָתַ֣ן לְרֵעֵ֑הוּ וְזֹ֥את הַתְּעוּדָ֖ה בְּיִשְׂרָאֵֽל׃ וְכָךְ הָיָה נָהוּג מִקֶּדֶם בְּיִשְׂרָאֵל, בְּפִדְיוֹן וּבְהַחְלָפָה – לְאַשְׁרֵר כָּל דָּבָר, אִישׁ חוֹלֵץ אֶת נַעֲלוֹ וְנוֹתֵן לַחֲבֵרוֹ, זֶה הַחוֹזֶה בְּיִשְׂרָאֵל.
ח וַיֹּ֧אמֶר הַגֹּאֵ֛ל לְבֹ֖עַז קְנֵה־לָ֑ךְ וַיִּשְׁלֹ֖ף נַעֲלֽוֹ׃ אָמַר הַגּוֹאֵל לְבוֹעַז קְנֵה לְךְ, וְהוּא חָלַץ אֶת נַעֲלוֹ.
ט וַיֹּ֩אמֶר֩ בֹּ֨עַז לַזְּקֵנִ֜ים וְכׇל־הָעָ֗ם עֵדִ֤ים אַתֶּם֙ הַיּ֔וֹם כִּ֤י קָנִ֙יתִי֙ אֶת־כׇּל־אֲשֶׁ֣ר לֶאֱלִימֶ֔לֶךְ וְאֵ֛ת כׇּל־אֲשֶׁ֥ר לְכִלְי֖וֹן וּמַחְל֑וֹן מִיַּ֖ד נׇעֳמִֽי׃ אָמַר בּוֹעַז לַזְּקֵנִים וּלְכָל הָעָם: אַתֶּם הַיּוֹם עֵדִים שֶׁקָּנִיתִי מִנָּעֳמִי אֶת כָּל הָרְכוּשׁ שֶׁל אֱלִימֶלֶךְ וְאֶת כָּל הָרְכוּשׁ שֶׁל כִּלְיוֹן וּמַחְלוֹן.
י וְגַ֣ם אֶת־ר֣וּת הַמֹּאֲבִיָּה֩ אֵ֨שֶׁת מַחְל֜וֹן קָנִ֧יתִי לִ֣י לְאִשָּׁ֗ה לְהָקִ֤ים שֵׁם־הַמֵּת֙ עַל־נַ֣חֲלָת֔וֹ וְלֹא־יִכָּרֵ֧ת שֵׁם־הַמֵּ֛ת מֵעִ֥ם אֶחָ֖יו וּמִשַּׁ֣עַר מְקוֹמ֑וֹ עֵדִ֥ים אַתֶּ֖ם הַיּֽוֹם׃ וְגַם אֶת רוּת הַמּוֹאֲבִית אֵשֶׁת מַחְלוֹן קָנִיתִי לִי לְאִשָּׁה לְהָקִים אֶת שֵׁם הַמֵּת עַל נַחֲלָתוֹ, וְלֹא יִכָּרֵת שֵׁם זֵכֶר הַמֵּת מִלִּהְיוֹת עִם אֶחָיו וּמֵעִיר נַחֲלָתוֹ – אַתֶּם עֵדִים הַיּוֹם!
יא וַיֹּ֨אמְר֜וּ כׇּל־הָעָ֧ם אֲשֶׁר־בַּשַּׁ֛עַר וְהַזְּקֵנִ֖ים עֵדִ֑ים יִתֵּן֩ יְהֹוָ֨ה אֶֽת־הָאִשָּׁ֜ה הַבָּאָ֣ה אֶל־בֵּיתֶ֗ךָ כְּרָחֵ֤ל ׀ וּכְלֵאָה֙ אֲשֶׁ֨ר בָּנ֤וּ שְׁתֵּיהֶם֙ אֶת־בֵּ֣ית יִשְׂרָאֵ֔ל וַעֲשֵׂה־חַ֣יִל בְּאֶפְרָ֔תָה וּקְרָא־שֵׁ֖ם בְּבֵ֥ית לָֽחֶם׃ אָמְרוּ כָּל הָעָם שֶׁבַּשַּׁעַר וְהַזְּקֵנִים: אֲנַחְנוּ עֵדִים! יִתֵּן ה' אֶת הָאִשָּׁה הַבָּאָה אֶל בֵּיתֶךָ שֶׁתִּהְיֶה כְּרָחֵל וּכְלֵאָה שֶׁבָּנוּ שְׁתֵּיהֶן אֶת בֵּית יִשְׂרָאֵל, וְתַצְלִיחַ בְּאֶפְרָת וְיִתְפַּרְסֵם שִׁמְךָ בְּבֵית לֶחֶם.
יב וִיהִ֤י בֵֽיתְךָ֙ כְּבֵ֣ית פֶּ֔רֶץ אֲשֶׁר־יָלְדָ֥ה תָמָ֖ר לִיהוּדָ֑ה מִן־הַזֶּ֗רַע אֲשֶׁ֨ר יִתֵּ֤ן יְהֹוָה֙ לְךָ֔ מִן־הַֽנַּעֲרָ֖ה הַזֹּֽאת׃ וּמִשְׁפַּחְתְּךָ מֵהַצֶּאֱצָאִים שֶׁה' יִתֵּן לְךָ מֵהַנַּעֲרָה הַזֹּאת תִּהְיֶה כְּמִשְׁפַּחְתּוֹ שֶׁל פֶּרֶץ שֶׁתָּמָר יָלְדָה לִיהוּדָה.
יג וַיִּקַּ֨ח בֹּ֤עַז אֶת־רוּת֙ וַתְּהִי־ל֣וֹ לְאִשָּׁ֔ה וַיָּבֹ֖א אֵלֶ֑יהָ וַיִּתֵּ֨ן יְהֹוָ֥ה לָ֛הּ הֵרָי֖וֹן וַתֵּ֥לֶד בֵּֽן׃ בּוֹעַז לָקַח אֶת רוּת וְהִיא נִשְּׂאָה לוֹ וְהוּא בָּא אֵלֶיהָ, ה' זִכָּה אוֹתָהּ בְּהֵרָיוֹן וְהִיא יָלְדָה בֵּן.
יד וַתֹּאמַ֤רְנָה הַנָּשִׁים֙ אֶֽל־נׇעֳמִ֔י בָּר֣וּךְ יְהֹוָ֔ה אֲ֠שֶׁ֠ר לֹ֣א הִשְׁבִּ֥ית לָ֛ךְ גֹּאֵ֖ל הַיּ֑וֹם וְיִקָּרֵ֥א שְׁמ֖וֹ בְּיִשְׂרָאֵֽל׃ אָמְרוּ הַנָּשִׁים לְנָעֳמִי: בָּרוּךְ ה' שֶׁלֹּא מָנַע מִמֵּךְ גּוֹאֵל הַיּוֹם, וְיִתְפַּרְסֵם שְׁמוֹ בְּיִשְׂרָאֵל.
טו וְהָ֤יָה לָךְ֙ לְמֵשִׁ֣יב נֶ֔פֶשׁ וּלְכַלְכֵּ֖ל אֶת־שֵׂיבָתֵ֑ךְ כִּ֣י כַלָּתֵ֤ךְ אֲֽשֶׁר־אֲהֵבַ֙תֶךְ֙ יְלָדַ֔תּוּ אֲשֶׁר־הִיא֙ ט֣וֹבָה לָ֔ךְ מִשִּׁבְעָ֖ה בָּנִֽים׃ וְשֶׁיִּהְיֶה עֲבוּרֵךְ מֵבִיא נַחַת וְדוֹאֵג לָךְ בְּזִקְנוּתֵךְ, כִּי כַּלָּתֵךְ שֶׁאוֹהֶבֶת אוֹתָךְ יָלְדָה אוֹתוֹ – שֶׁהִיא טוֹבָה לָךְ יוֹתֵר מִשִּׁבְעָה בָּנִים.
טז וַתִּקַּ֨ח נׇעֳמִ֤י אֶת־הַיֶּ֙לֶד֙ וַתְּשִׁתֵ֣הוּ בְחֵיקָ֔הּ וַתְּהִי־ל֖וֹ לְאֹמֶֽנֶת׃ נָעֳמִי לָקְחָה אֶת הַיֶּלֶד וְשָׂמָה אוֹתוֹ בִּזְרוֹעוֹתֶיהָ, וְהִיא הָיְתָה הַמְּטַפֶּלֶת שֶׁלּוֹ.
יז וַתִּקְרֶ֩אנָה֩ ל֨וֹ הַשְּׁכֵנ֥וֹת שֵׁם֙ לֵאמֹ֔ר יֻלַּד־בֵּ֖ן לְנׇעֳמִ֑י וַתִּקְרֶ֤אנָֽה שְׁמוֹ֙ עוֹבֵ֔ד ה֥וּא אֲבִֽי־יִשַׁ֖י אֲבִ֥י דָוִֽד׃ הַשְּׁכֵנוֹת קָרְאוּ לוֹ שֵׁם וְלֹא רוּת, לְהַרְאוֹת שֶׁהַבֵּן נוֹלַד לְנָעֳמִי, הֵן קָרְאוּ אֶת שְׁמוֹ עוֹבֵד, הוּא אֲבִיו שֶׁל יִשַׁי שֶׁהָיָה אֲבִיו שֶׁל דָּוִד הַמֶּלֶךְ.
פ
יח וְאֵ֙לֶּה֙ תּוֹלְד֣וֹת פָּ֔רֶץ פֶּ֖רֶץ הוֹלִ֥יד אֶת־חֶצְרֽוֹן׃ וְאֵלֶּה צֶאֱצָאֵי פֶּרֶץ: פֶּרֶץ הוֹלִיד אֶת חֶצְרוֹן.
יט וְחֶצְרוֹן֙ הוֹלִ֣יד אֶת־רָ֔ם וְרָ֖ם הוֹלִ֥יד אֶת־עַמִּֽינָדָֽב׃ חֶצְרוֹן הוֹלִיד אֶת רָם, וְרָם הוֹלִיד אֶת עַמִּינָדָב.
כ וְעַמִּֽינָדָב֙ הוֹלִ֣יד אֶת־נַחְשׁ֔וֹן וְנַחְשׁ֖וֹן הוֹלִ֥יד אֶת־שַׂלְמָֽה׃ עַמִּינָדָב הוֹלִיד אֶת נַחְשׁוֹן, וְנַחְשׁוֹן הוֹלִיד אֶת שַׂלְמָה.
כא וְשַׂלְמוֹן֙ הוֹלִ֣יד אֶת־בֹּ֔עַז וּבֹ֖עַז הוֹלִ֥יד אֶת־עוֹבֵֽד׃ שַׂלְמוֹן הוֹלִיד אֶת בּוֹעַז, וּבוֹעַז הוֹלִיד אֶת עוֹבֵד.
כב וְעֹבֵד֙ הוֹלִ֣יד אֶת־יִשָׁ֔י וְיִשַׁ֖י הוֹלִ֥יד אֶת־דָּוִֽד׃ עוֹבֵד הוֹלִיד אֶת יִשַׁי, וְיִשַׁי הוֹלִיד אֶת דָּוִד הַמֶּלֶךְ.
