ראשון[עריכה]

פ פָּרָשַׁת פָּרָה אֲדֻמָּה
יט א וַיְדַבֵּ֣ר יְהֹוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאֶל אַהֲרֹן וְאָמַר:
ב זֹ֚את חֻקַּ֣ת הַתּוֹרָ֔ה אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה לֵאמֹ֑ר דַּבֵּ֣ר ׀ אֶל־בְּנֵ֣י יִשְׂרָאֵ֗ל וְיִקְח֣וּ אֵלֶ֩יךָ֩ פָרָ֨ה אֲדֻמָּ֜ה תְּמִימָ֗ה אֲשֶׁ֤ר אֵֽין־בָּהּ֙ מ֔וּם אֲשֶׁ֛ר לֹא־עָלָ֥ה עָלֶ֖יהָ עֹֽל׃ זֹאת חֻקַּת הַתּוֹרָה שֶׁה' צִוָּה: דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְיִקְחוּ אֵלֶיךָ פָּרָה אֲדֻמָּה לְגַמְרֵי, שֶׁאֵין בָּהּ מוּם וְלֹא עָשְׂתָה מְלָאכָה.
ג וּנְתַתֶּ֣ם אֹתָ֔הּ אֶל־אֶלְעָזָ֖ר הַכֹּהֵ֑ן וְהוֹצִ֤יא אֹתָהּ֙ אֶל־מִח֣וּץ לַֽמַּחֲנֶ֔ה וְשָׁחַ֥ט אֹתָ֖הּ לְפָנָֽיו׃ וְתִתְּנוּ אוֹתָהּ לְאֶלְעָזָר הַכֹּהֵן, וְהוּא יוֹצִיא אוֹתָהּ אֶל מִחוּץ לַמַּחֲנֶה וְיִשְׁחֲטוּ אוֹתָהּ לְפָנָיו.
ד וְלָקַ֞ח אֶלְעָזָ֧ר הַכֹּהֵ֛ן מִדָּמָ֖הּ בְּאֶצְבָּע֑וֹ וְהִזָּ֞ה אֶל־נֹ֨כַח פְּנֵ֧י אֹֽהֶל־מוֹעֵ֛ד מִדָּמָ֖הּ שֶׁ֥בַע פְּעָמִֽים׃ וְאֶלְעָזָר הַכֹּהֵן יִקַּח מִדָּמָהּ בְּאֶצְבָּעוֹ וְיַתִּיז שֶׁבַע פְּעָמִים אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.
ה וְשָׂרַ֥ף אֶת־הַפָּרָ֖ה לְעֵינָ֑יו אֶת־עֹרָ֤הּ וְאֶת־בְּשָׂרָהּ֙ וְאֶת־דָּמָ֔הּ עַל־פִּרְשָׁ֖הּ יִשְׂרֹֽף׃ וְיִשְׂרְפוּ אֶת הַפָּרָה לְעֵינָיו, אֶת הָעוֹר וְהַבָּשָׂר וְהַדָּם שֶׁלָּהּ עִם הָאוֹכֶל הַמְּעֻכָּל שֶׁבָּהּ – אֶת הַכֹּל יִשְׂרוֹף.
ו וְלָקַ֣ח הַכֹּהֵ֗ן עֵ֥ץ אֶ֛רֶז וְאֵז֖וֹב וּשְׁנִ֣י תוֹלָ֑עַת וְהִשְׁלִ֕יךְ אֶל־תּ֖וֹךְ שְׂרֵפַ֥ת הַפָּרָֽה׃ וְהַכֹּהֵן יִקַּח עֵץ אֶרֶז וְאֵזוֹב וְצֶבַע אָדוֹם מִתּוֹלַעַת, וְיַשְׁלִיךְ לְתוֹךְ שְׂרֵפַת הַפָּרָה.
ז וְכִבֶּ֨ס בְּגָדָ֜יו הַכֹּהֵ֗ן וְרָחַ֤ץ בְּשָׂרוֹ֙ בַּמַּ֔יִם וְאַחַ֖ר יָבֹ֣א אֶל־הַֽמַּחֲנֶ֑ה וְטָמֵ֥א הַכֹּהֵ֖ן עַד־הָעָֽרֶב׃ הַכֹּהֵן יַטְבִּיל אֶת בְּגָדָיו וְיַטְבִּיל אֶת גּוּפוֹ בְּמַיִם, וְאַחַר כָּךְ יָבוֹא לַמַּחֲנֶה, הַכֹּהֵן יִהְיֶה טָמֵא עַד הָעֶרֶב.
ח וְהַשֹּׂרֵ֣ף אֹתָ֔הּ יְכַבֵּ֤ס בְּגָדָיו֙ בַּמַּ֔יִם וְרָחַ֥ץ בְּשָׂר֖וֹ בַּמָּ֑יִם וְטָמֵ֖א עַד־הָעָֽרֶב׃ וְהָאִישׁ שֶׁשּׂוֹרֵף אוֹתָהּ יַטְבִּיל אֶת בְּגָדָיו בְּמַיִם, וְיַטְבִּיל אֶת גּוּפוֹ בְּמַיִם, וְיִהְיֶה טָמֵא עַד הָעֶרֶב.
ט וְאָסַ֣ף ׀ אִ֣ישׁ טָה֗וֹר אֵ֚ת אֵ֣פֶר הַפָּרָ֔ה וְהִנִּ֛יחַ מִח֥וּץ לַֽמַּחֲנֶ֖ה בְּמָק֣וֹם טָה֑וֹר וְ֠הָיְתָ֠ה לַעֲדַ֨ת בְּנֵֽי־יִשְׂרָאֵ֧ל לְמִשְׁמֶ֛רֶת לְמֵ֥י נִדָּ֖ה חַטָּ֥את הִֽוא׃ אִישׁ טָהוֹר יֶאֱסוֹף אֶת אֵפֶר הַפָּרָה, וְיַנִּיחַ מִחוּץ לַמַּחֲנֶה בְּמָקוֹם טָהוֹר, וְצִבּוּר בְּנֵי יִשְׂרָאֵל יִשְׁמְרוּ אוֹתָהּ לְצוֹרֶךְ מֵי הַהַזָּאָה, כְּדֵי לְהִטַּהֵר.
י וְ֠כִבֶּ֠ס הָאֹסֵ֨ף אֶת־אֵ֤פֶר הַפָּרָה֙ אֶת־בְּגָדָ֔יו וְטָמֵ֖א עַד־הָעָ֑רֶב וְֽהָיְתָ֞ה לִבְנֵ֣י יִשְׂרָאֵ֗ל וְלַגֵּ֛ר הַגָּ֥ר בְּתוֹכָ֖ם לְחֻקַּ֥ת עוֹלָֽם׃ הָאִישׁ שֶׁאוֹסֵף אֶת אֵפֶר הַפָּרָה יַטְבִּיל אֶת בְּגָדָיו וְיִהְיֶה טָמֵא עַד הָעֶרֶב, זֶה יִהְיֶה חוֹק לְעוֹלָם לִבְנֵי יִשְׂרָאֵל וְלַגֵּר שֶׁגָּר אִתָּם.
פָּרָשַׁת דִּינֵי טֻמְאָה
יא הַנֹּגֵ֥עַ בְּמֵ֖ת לְכׇל־נֶ֣פֶשׁ אָדָ֑ם וְטָמֵ֖א שִׁבְעַ֥ת יָמִֽים׃ מִי שֶׁנּוֹגֵעַ בְּגוּפָה שֶׁל כָּל אָדָם מֵת יִהְיֶה טָמֵא שִׁבְעָה יָמִים.
יב ה֣וּא יִתְחַטָּא־ב֞וֹ בַּיּ֧וֹם הַשְּׁלִישִׁ֛י וּבַיּ֥וֹם הַשְּׁבִיעִ֖י יִטְהָ֑ר וְאִם־לֹ֨א יִתְחַטָּ֜א בַּיּ֧וֹם הַשְּׁלִישִׁ֛י וּבַיּ֥וֹם הַשְּׁבִיעִ֖י לֹ֥א יִטְהָֽר׃ הוּא יִטַּהֵר בְּאֵפֶר הַפָּרָה בַּיּוֹם הַשְּׁלִישִׁי וּבַיּוֹם הַשְּׁבִיעִי, וְאָז יִהְיֶה טָהוֹר, וְאִם לֹא יִטַּהֵר בַּיּוֹם הַשְּׁלִישִׁי וּבַיּוֹם הַשְּׁבִיעִי, הוּא לֹא יִהְיֶה טָהוֹר.
יג כׇּֽל־הַנֹּגֵ֡עַ בְּמֵ֣ת בְּנֶ֩פֶשׁ֩ הָאָדָ֨ם אֲשֶׁר־יָמ֜וּת וְלֹ֣א יִתְחַטָּ֗א אֶת־מִשְׁכַּ֤ן יְהֹוָה֙ טִמֵּ֔א וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מִיִּשְׂרָאֵ֑ל כִּי֩ מֵ֨י נִדָּ֜ה לֹא־זֹרַ֤ק עָלָיו֙ טָמֵ֣א יִהְיֶ֔ה ע֖וֹד טֻמְאָת֥וֹ בֽוֹ׃ כָּל הַנּוֹגֵעַ בְּמֵת – בְּנֶפֶשׁ הָאָדָם שֶׁיָּמוּת, וְלֹא יִטַּהֵר וְיִכָּנֵס לַמִּקְדָּשׁ, הוּא טִמֵּא אֶת מִשְׁכַּן ה', הַנֶּפֶשׁ הַהִיא תֹּאבַד מִיִּשְׂרָאֵל, כִּי לֹא זָרְקוּ עָלָיו מֵי הַזָּאָה, הוּא יִהְיֶה טָמֵא – טֻמְאָתוֹ עֲדַיִן בּוֹ.
יד זֹ֚את הַתּוֹרָ֔ה אָדָ֖ם כִּֽי־יָמ֣וּת בְּאֹ֑הֶל כׇּל־הַבָּ֤א אֶל־הָאֹ֙הֶל֙ וְכׇל־אֲשֶׁ֣ר בָּאֹ֔הֶל יִטְמָ֖א שִׁבְעַ֥ת יָמִֽים׃ זֹאת הַהוֹרָאָה: אִם אָדָם יָמוּת בְּאוֹהֶל, כָּל מִי שֶׁבָּא לָאוֹהֶל וְכָל מַה שֶּׁבָּאוֹהֶל יִטְמָא שִׁבְעָה יָמִים.
טו וְכֹל֙ כְּלִ֣י פָת֔וּחַ אֲשֶׁ֛ר אֵין־צָמִ֥יד פָּתִ֖יל עָלָ֑יו טָמֵ֖א הֽוּא׃ וְכָל כְּלִי פָּתוּחַ שֶׁלֹּא מְלֻפָּף עָלָיו כִּסּוּי צָמוּד – הוּא טָמֵא.
טז וְכֹ֨ל אֲשֶׁר־יִגַּ֜ע עַל־פְּנֵ֣י הַשָּׂדֶ֗ה בַּֽחֲלַל־חֶ֙רֶב֙ א֣וֹ בְמֵ֔ת אֽוֹ־בְעֶ֥צֶם אָדָ֖ם א֣וֹ בְקָ֑בֶר יִטְמָ֖א שִׁבְעַ֥ת יָמִֽים׃ וְכָל מַה שֶּׁיִּגַּע עַל פְּנֵי הַשָּׂדֶה בְּמִי שֶׁנֶּהֱרַג בְּחֶרֶב, אוֹ בְּכָל מֵת, אוֹ בְּעֶצֶם שֶׁל אָדָם, אוֹ בְּקֶבֶר, יִטְמָא שִׁבְעָה יָמִים.
יז וְלָֽקְחוּ֙ לַטָּמֵ֔א מֵעֲפַ֖ר שְׂרֵפַ֣ת הַֽחַטָּ֑את וְנָתַ֥ן עָלָ֛יו מַ֥יִם חַיִּ֖ים אֶל־כֶּֽלִי׃ וְיִקְחוּ בִּשְׁבִיל הַטָּמֵא מֵעֲפַר שְׂרֵפַת הַפָּרָה לְהַזָּאָה, וְיָשִׂימוּ עַל הֶעָפָר מַיִם מִמַּעְיָן לְתוֹךְ כְּלִי.

שני[עריכה]

יח וְלָקַ֨ח אֵז֜וֹב וְטָבַ֣ל בַּמַּ֘יִם֮ אִ֣ישׁ טָהוֹר֒ וְהִזָּ֤ה עַל־הָאֹ֙הֶל֙ וְעַל־כׇּל־הַכֵּלִ֔ים וְעַל־הַנְּפָשׁ֖וֹת אֲשֶׁ֣ר הָֽיוּ־שָׁ֑ם וְעַל־הַנֹּגֵ֗עַ בַּעֶ֙צֶם֙ א֣וֹ בֶֽחָלָ֔ל א֥וֹ בַמֵּ֖ת א֥וֹ בַקָּֽבֶר׃ אִישׁ טָהוֹר יִקַּח אֵזוֹב וְיַטְבִּיל בַּמַּיִם, וְיַזֶּה עַל הָאוֹהֶל וְעַל כָּל הַכֵּלִים וְעַל הַנְּפָשׁוֹת שֶׁהָיוּ שָׁם, וְעַל הַנּוֹגֵעַ בָּעֶצֶם אוֹ בַּנִּרְצָח אוֹ בַּמֵּת אוֹ בַּקֶּבֶר.
יט וְהִזָּ֤ה הַטָּהֹר֙ עַל־הַטָּמֵ֔א בַּיּ֥וֹם הַשְּׁלִישִׁ֖י וּבַיּ֣וֹם הַשְּׁבִיעִ֑י וְחִטְּאוֹ֙ בַּיּ֣וֹם הַשְּׁבִיעִ֔י וְכִבֶּ֧ס בְּגָדָ֛יו וְרָחַ֥ץ בַּמַּ֖יִם וְטָהֵ֥ר בָּעָֽרֶב׃ הָאִישׁ הַטָּהוֹר יַזֶּה עַל הַטָּמֵא בַּיּוֹם הַשְּׁלִישִׁי וּבַיּוֹם הַשְּׁבִיעִי, וִיטַהֵר אוֹתוֹ בַּיּוֹם הַשְּׁבִיעִי, וְהוּא יַטְבִּיל אֶת בְּגָדָיו וְיִטְבּוֹל בְּמַיִם וְיִהְיֶה טָהוֹר בָּעֶרֶב.
כ וְאִ֤ישׁ אֲשֶׁר־יִטְמָא֙ וְלֹ֣א יִתְחַטָּ֔א וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מִתּ֣וֹךְ הַקָּהָ֑ל כִּי֩ אֶת־מִקְדַּ֨שׁ יְהֹוָ֜ה טִמֵּ֗א מֵ֥י נִדָּ֛ה לֹא־זֹרַ֥ק עָלָ֖יו טָמֵ֥א הֽוּא׃ וְאִישׁ שֶׁיִּטְמָא וְלֹא יִטַּהֵר וְיִכָּנֵס לַמִּקְדָּשׁ –הַנֶּפֶשׁ הַהִיא תֹּאבַד מִתּוֹךְ הַקָּהָל, כִּי הוּא טִמֵּא אֶת קָדְשֵׁי ה', לֹא זָרְקוּ עָלָיו מֵי הַזָּאָה – הוּא טָמֵא.
כא וְהָיְתָ֥ה לָהֶ֖ם לְחֻקַּ֣ת עוֹלָ֑ם וּמַזֵּ֤ה מֵֽי־הַנִּדָּה֙ יְכַבֵּ֣ס בְּגָדָ֔יו וְהַנֹּגֵ֙עַ֙ בְּמֵ֣י הַנִּדָּ֔ה יִטְמָ֖א עַד־הָעָֽרֶב׃ זֶה יִהְיֶה לָהֶם חוֹק לְעוֹלָם, וְהַמַּזֵּה אֶת מֵי הַהַזָּאָה יַטְבִּיל אֶת בְּגָדָיו, וְהַנּוֹגֵעַ בְּמֵי הַהַזָּאָה יִהְיֶה טָמֵא עַד הָעֶרֶב.
כב וְכֹ֛ל אֲשֶׁר־יִגַּע־בּ֥וֹ הַטָּמֵ֖א יִטְמָ֑א וְהַנֶּ֥פֶשׁ הַנֹּגַ֖עַת תִּטְמָ֥א עַד־הָעָֽרֶב׃ וְכָל מַה שֶּׁיִּגַּע בּוֹ הַטָּמֵא – יִטְמָא, וְהָאָדָם שֶׁיִּגַּע בּוֹ – יִטְמָא עַד הָעֶרֶב.
פ פָּרָשַׁת מֵי הַמְּרִיבָה
כ א וַיָּבֹ֣אוּ בְנֵֽי־יִ֠שְׂרָאֵ֠ל כׇּל־הָ֨עֵדָ֤ה מִדְבַּר־צִן֙ בַּחֹ֣דֶשׁ הָֽרִאשׁ֔וֹן וַיֵּ֥שֶׁב הָעָ֖ם בְּקָדֵ֑שׁ וַתָּ֤מׇת שָׁם֙ מִרְיָ֔ם וַתִּקָּבֵ֖ר שָֽׁם׃ כָּל צִבּוּר בְּנֵי יִשְׂרָאֵל בָּאוּ לְמִדְבַּר צִין בַּחוֹדֶשׁ הָרִאשׁוֹן, הָעָם הִתְיַשֵּׁב בְּקָדֵשׁ, וּמִרְיָם מֵתָה שָׁם וְנִקְבְּרָה שָׁם.
ב וְלֹא־הָ֥יָה מַ֖יִם לָעֵדָ֑ה וַיִּקָּ֣הֲל֔וּ עַל־מֹשֶׁ֖ה וְעַֽל־אַהֲרֹֽן׃ וְלֹא הָיָה מַיִם לַצִּבּוּר, הֵם הִתְקָהֲלוּ לִפְנֵי מֹשֶׁה וְאַהֲרֹן.
ג וַיָּ֥רֶב הָעָ֖ם עִם־מֹשֶׁ֑ה וַיֹּאמְר֣וּ לֵאמֹ֔ר וְל֥וּ גָוַ֛עְנוּ בִּגְוַ֥ע אַחֵ֖ינוּ לִפְנֵ֥י יְהֹוָֽה׃ הָעָם רָבוּ עִם מֹשֶׁה וְאָמְרוּ: הַלְוַאי שֶׁהָיִינוּ נֶהֱרָגִים כְּשֶׁאַחֵינוּ מֵתוּ בְּחֵטְא הַמְּרַגְּלִים שֶׁה' הֶעֱנִישָׁם!
ד וְלָמָ֤ה הֲבֵאתֶם֙ אֶת־קְהַ֣ל יְהֹוָ֔ה אֶל־הַמִּדְבָּ֖ר הַזֶּ֑ה לָמ֣וּת שָׁ֔ם אֲנַ֖חְנוּ וּבְעִירֵֽנוּ׃ וְלָמָה הֲבֵאתֶם אֶת קְהַל ה' לַמִּדְבָּר הַזֶּה, לָמוּת שָׁם – אֲנַחְנוּ וְצֹאנֵנוּ?!
ה וְלָמָ֤ה הֶֽעֱלִיתֻ֙נוּ֙ מִמִּצְרַ֔יִם לְהָבִ֣יא אֹתָ֔נוּ אֶל־הַמָּק֥וֹם הָרָ֖ע הַזֶּ֑ה לֹ֣א ׀ מְק֣וֹם זֶ֗רַע וּתְאֵנָ֤ה וְגֶ֙פֶן֙ וְרִמּ֔וֹן וּמַ֥יִם אַ֖יִן לִשְׁתּֽוֹת׃ וְלָמָה הֶעֱלִיתֶם אוֹתָנוּ מִמִּצְרַיִם לְהָבִיא אוֹתָנוּ לַמָּקוֹם הָרָע הַזֶּה – לֹא מָקוֹם שֶׁל זֶרַע וּתְאֵנָה וְגֶפֶן וְרִמּוֹן, וְאֵין מַיִם לִשְׁתּוֹת?!
ו וַיָּבֹא֩ מֹשֶׁ֨ה וְאַהֲרֹ֜ן מִפְּנֵ֣י הַקָּהָ֗ל אֶל־פֶּ֙תַח֙ אֹ֣הֶל מוֹעֵ֔ד וַֽיִּפְּל֖וּ עַל־פְּנֵיהֶ֑ם וַיֵּרָ֥א כְבוֹד־יְהֹוָ֖ה אֲלֵיהֶֽם׃ מֹשֶׁה וְאַהֲרֹן בָּאוּ לְפֶתַח אוֹהֶל הַהִתְכַּנְּסוּת מִפְּנֵי הַקָּהָל, וְהִשְׁתַּטְּחוּ עַל פְּנֵיהֶם, וּכְבוֹד ה' נִגְלָה אֲלֵיהֶם.

שלישי (שני במחוברין)[עריכה]

פ
ז וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ח קַ֣ח אֶת־הַמַּטֶּ֗ה וְהַקְהֵ֤ל אֶת־הָעֵדָה֙ אַתָּה֙ וְאַהֲרֹ֣ן אָחִ֔יךָ וְדִבַּרְתֶּ֧ם אֶל־הַסֶּ֛לַע לְעֵינֵיהֶ֖ם וְנָתַ֣ן מֵימָ֑יו וְהוֹצֵאתָ֨ לָהֶ֥ם מַ֙יִם֙ מִן־הַסֶּ֔לַע וְהִשְׁקִיתָ֥ אֶת־הָעֵדָ֖ה וְאֶת־בְּעִירָֽם׃ קַח אֶת הַמַּקֵּל וְהַקְהֵל אֶת הַקָּהָל אַתָּה וְאַהֲרֹן אָחִיךָ, וּתְדַבְּרוּ אֶל הַסֶּלַע לְעֵינֵיהֶם וְיִתֵּן אֶת מֵימָיו, וְתוֹצִיא לָהֶם מַיִם מֵהַסֶּלַע וְתַשְׁקֶה אֶת הַצִּבּוּר וְאֶת בֶּהֱמוֹתֵיהֶם.
ט וַיִּקַּ֥ח מֹשֶׁ֛ה אֶת־הַמַּטֶּ֖ה מִלִּפְנֵ֣י יְהֹוָ֑ה כַּאֲשֶׁ֖ר צִוָּֽהוּ׃ מֹשֶׁה לָקַח אֶת הַמַּקֵּל וְיָצָא מִלִּפְנֵי ה' כְּמוֹ שֶׁצִּוָּה אוֹתוֹ.
י וַיַּקְהִ֜לוּ מֹשֶׁ֧ה וְאַהֲרֹ֛ן אֶת־הַקָּהָ֖ל אֶל־פְּנֵ֣י הַסָּ֑לַע וַיֹּ֣אמֶר לָהֶ֗ם שִׁמְעוּ־נָא֙ הַמֹּרִ֔ים הֲמִן־הַסֶּ֣לַע הַזֶּ֔ה נוֹצִ֥יא לָכֶ֖ם מָֽיִם׃ מֹשֶׁה וְאַהֲרֹן הִקְהִילוּ אֶת הַקָּהָל לִפְנֵי הַסֶּלַע, וְאָמַר לָהֶם: שִׁמְעוּ בְּבַקָּשָׁה מַרְדָּנִים! הַאִם מֵהַסֶּלַע הַזֶּה נוֹצִיא לָכֶם מַיִם?
יא וַיָּ֨רֶם מֹשֶׁ֜ה אֶת־יָד֗וֹ וַיַּ֧ךְ אֶת־הַסֶּ֛לַע בְּמַטֵּ֖הוּ פַּעֲמָ֑יִם וַיֵּצְאוּ֙ מַ֣יִם רַבִּ֔ים וַתֵּ֥שְׁתְּ הָעֵדָ֖ה וּבְעִירָֽם׃ מֹשֶׁה הֵנִיף אֶת יָדוֹ וְהִכָּה אֶת הַסֶּלַע בְּמַקְלוֹ פַּעֲמַיִם, מַיִם רַבִּים יָצְאוּ וְהַצִּבּוּר וְצֹאנָם שָׁתוּ.
ס
יב וַיֹּ֣אמֶר יְהֹוָה֮ אֶל־מֹשֶׁ֣ה וְאֶֽל־אַהֲרֹן֒ יַ֚עַן לֹא־הֶאֱמַנְתֶּ֣ם בִּ֔י לְהַ֨קְדִּישֵׁ֔נִי לְעֵינֵ֖י בְּנֵ֣י יִשְׂרָאֵ֑ל לָכֵ֗ן לֹ֤א תָבִ֙יאוּ֙ אֶת־הַקָּהָ֣ל הַזֶּ֔ה אֶל־הָאָ֖רֶץ אֲשֶׁר־נָתַ֥תִּי לָהֶֽם׃ ה' אָמַר לְמֹשֶׁה וּלְאַהֲרֹן: כֵּיוָן שֶׁלֹּא הֶאֱמַנְתֶּם בִּי לְהַרְאוֹת אֶת קְדֻשָּׁתִי לְעֵינֵי בְּנֵי יִשְׂרָאֵל, לָכֵן לֹא תָּבִיאוּ אֶת הַקָּהָל הַזֶּה לָאָרֶץ שֶׁנָּתַתִּי לָהֶם.
יג הֵ֚מָּה מֵ֣י מְרִיבָ֔ה אֲשֶׁר־רָב֥וּ בְנֵֽי־יִשְׂרָאֵ֖ל אֶת־יְהֹוָ֑ה וַיִּקָּדֵ֖שׁ בָּֽם׃ זֶהוּ מַעֲשֶׂה 'מֵי מְרִיבָה', שֶׁבְּנֵי יִשְׂרָאֵל רָבוּ עִם ה', וּשְׁמוֹ הִתְקַדֵּשׁ בָּהֶם.

רביעי[עריכה]

ס פָּרָשַׁת אֱדוֹם
יד וַיִּשְׁלַ֨ח מֹשֶׁ֧ה מַלְאָכִ֛ים מִקָּדֵ֖שׁ אֶל־מֶ֣לֶךְ אֱד֑וֹם כֹּ֤ה אָמַר֙ אָחִ֣יךָ יִשְׂרָאֵ֔ל אַתָּ֣ה יָדַ֔עְתָּ אֵ֥ת כׇּל־הַתְּלָאָ֖ה אֲשֶׁ֥ר מְצָאָֽתְנוּ׃ מֹשֶׁה שָׁלַח שְׁלִיחִים מִקָּדֵשׁ אֶל מֶלֶךְ אֱדוֹם שֶׁיֹּאמְרוּ לוֹ: כָּךְ אָמַר אָחִיךָ יִשְׂרָאֵל – אַתָּה יָדַעְתָּ אֶת כָּל הַצָּרוֹת שֶׁקָּרוּ לָנוּ –
טו וַיֵּרְד֤וּ אֲבֹתֵ֙ינוּ֙ מִצְרַ֔יְמָה וַנֵּ֥שֶׁב בְּמִצְרַ֖יִם יָמִ֣ים רַבִּ֑ים וַיָּרֵ֥עוּ לָ֛נוּ מִצְרַ֖יִם וְלַאֲבֹתֵֽינוּ׃ אֲבוֹתֵינוּ יָרְדוּ לְמִצְרַיִם, וְיָשַׁבְנוּ בְּמִצְרַיִם זְמַן רָב, וְהַמִּצְרִים עָשׂוּ רָע לָנוּ וְלַאֲבוֹתֵינוּ.
טז וַנִּצְעַ֤ק אֶל־יְהֹוָה֙ וַיִּשְׁמַ֣ע קֹלֵ֔נוּ וַיִּשְׁלַ֣ח מַלְאָ֔ךְ וַיֹּצִאֵ֖נוּ מִמִּצְרָ֑יִם וְהִנֵּה֙ אֲנַ֣חְנוּ בְקָדֵ֔שׁ עִ֖יר קְצֵ֥ה גְבוּלֶֽךָ׃ צָעַקְנוּ אֶל ה' וְהוּא שָׁמַע אֶת קוֹלֵנוּ, הוּא שָׁלַח שָׁלִיחַ וְהוֹצִיא אוֹתָנוּ מִמִּצְרַיִם, וְהִנֵּה אֲנַחְנוּ בְּקָדֵשׁ – הָעִיר שֶׁבִּקְצֵה גְּבוּלְךָ.
יז נַעְבְּרָה־נָּ֣א בְאַרְצֶ֗ךָ לֹ֤א נַעֲבֹר֙ בְּשָׂדֶ֣ה וּבְכֶ֔רֶם וְלֹ֥א נִשְׁתֶּ֖ה מֵ֣י בְאֵ֑ר דֶּ֧רֶךְ הַמֶּ֣לֶךְ נֵלֵ֗ךְ לֹ֤א נִטֶּה֙ יָמִ֣ין וּשְׂמֹ֔אול עַ֥ד אֲשֶֽׁר־נַעֲבֹ֖ר גְּבֻלֶֽךָ׃ נַעֲבוֹר בְּבַקָּשָׁה בְּאַרְצְךָ, לֹא נַעֲבוֹר בְּשָׂדֶה וּבְכֶרֶם וְלֹא נִשְׁתֶּה מַיִם מִבְּאֵר, נֵלֵךְ רַק בַּדֶּרֶךְ הָרָאשִׁית – לֹא נִסְטֶה יָמִינָה אוֹ שְׂמֹאלָה, עַד שֶׁנַּעֲבוֹר אֶת הַשֶּׁטַח שֶׁלְּךָ.
יח וַיֹּ֤אמֶר אֵלָיו֙ אֱד֔וֹם לֹ֥א תַעֲבֹ֖ר בִּ֑י פֶּן־בַּחֶ֖רֶב אֵצֵ֥א לִקְרָאתֶֽךָ׃ אָמַר לוֹ אֱדוֹם: אַל תַּעֲבוֹר בְּאַרְצִי, אַחֶרֶת – אֵצֵא לְמִלְחָמָה לִקְרָאתְךָ!
יט וַיֹּאמְר֨וּ אֵלָ֥יו בְּנֵֽי־יִשְׂרָאֵל֮ בַּֽמְסִלָּ֣ה נַעֲלֶה֒ וְאִם־מֵימֶ֤יךָ נִשְׁתֶּה֙ אֲנִ֣י וּמִקְנַ֔י וְנָתַתִּ֖י מִכְרָ֑ם רַ֥ק אֵין־דָּבָ֖ר בְּרַגְלַ֥י אֶֽעֱבֹֽרָה׃ אָמְרוּ לוֹ בְּנֵי יִשְׂרָאֵל: נַעֲבוֹר רַק בַּדֶּרֶךְ הַסְּלוּלָה, וְאִם אֲנִי וּבֶהֱמוֹתַי נִשְׁתֶּה מֵהַמַּיִם שֶׁלְּךָ אֲשַׁלֵּם אֶת מְחִירָם, לֹא אֶעֱשֶׂה דָּבָר רָע, רַק אֶעֱבוֹר בְּרַגְלַי.
כ וַיֹּ֖אמֶר לֹ֣א תַעֲבֹ֑ר וַיֵּצֵ֤א אֱדוֹם֙ לִקְרָאת֔וֹ בְּעַ֥ם כָּבֵ֖ד וּבְיָ֥ד חֲזָקָֽה׃ אָמַר לוֹ: אַל תַּעֲבוֹר! אֱדוֹם יָצָא לִקְרָאתוֹ בְּעַם רָב וּבְיָד חֲזָקָה.
כא וַיְמָאֵ֣ן ׀ אֱד֗וֹם נְתֹן֙ אֶת־יִשְׂרָאֵ֔ל עֲבֹ֖ר בִּגְבֻל֑וֹ וַיֵּ֥ט יִשְׂרָאֵ֖ל מֵעָלָֽיו׃ אֱדוֹם סֵרֵב לָתֵת לְיִשְׂרָאֵל לַעֲבוֹר בִּגְבוּלוֹ, וְעַם יִשְׂרָאֵל סָטוּ מִמֶּנּוּ.

חמישי (שלישי במחוברין)[עריכה]

פ פָּרָשַׁת מִיתַת אַהֲרֹן
כב וַיִּסְע֖וּ מִקָּדֵ֑שׁ וַיָּבֹ֧אוּ בְנֵֽי־יִשְׂרָאֵ֛ל כׇּל־הָעֵדָ֖ה הֹ֥ר הָהָֽר׃ הֵם נָסְעוּ מִקָּדֵשׁ, וּבְנֵי יִשְׂרָאֵל – כָּל הַקָּהָל, בָּאוּ לְהוֹר-הָהָר.
כג וַיֹּ֧אמֶר יְהֹוָ֛ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן בְּהֹ֣ר הָהָ֑ר עַל־גְּב֥וּל אֶֽרֶץ־אֱד֖וֹם לֵאמֹֽר׃ ה' אָמַר לְמֹשֶׁה וּלְאַהֲרֹן בְּהוֹר-הָהָר עַל גְּבוּל אֶרֶץ אֱדוֹם:
כד יֵאָסֵ֤ף אַהֲרֹן֙ אֶל־עַמָּ֔יו כִּ֣י לֹ֤א יָבֹא֙ אֶל־הָאָ֔רֶץ אֲשֶׁ֥ר נָתַ֖תִּי לִבְנֵ֣י יִשְׂרָאֵ֑ל עַ֛ל אֲשֶׁר־מְרִיתֶ֥ם אֶת־פִּ֖י לְמֵ֥י מְרִיבָֽה׃ נִשְׁמָתוֹ שֶׁל אַהֲרֹן תֵּאָסֵף אֶל מְקוֹם הַנְּשָׁמוֹת, כִּי הוּא לֹא יָבוֹא לָאָרֶץ שֶׁנָּתַתִּי לִבְנֵי יִשְׂרָאֵל, עוֹנֶשׁ עַל שֶׁסֵּרַבְתֶּם לְקַיֵּם אֶת דְּבָרַי בְּמֵי מְרִיבָה.
כה קַ֚ח אֶֽת־אַהֲרֹ֔ן וְאֶת־אֶלְעָזָ֖ר בְּנ֑וֹ וְהַ֥עַל אֹתָ֖ם הֹ֥ר הָהָֽר׃ קַח אֶת אַהֲרֹן וְאֶת אֶלְעָזָר בְּנוֹ וְהַעֲלֶה אוֹתָם לְהוֹר-הָהָר.
כו וְהַפְשֵׁ֤ט אֶֽת־אַהֲרֹן֙ אֶת־בְּגָדָ֔יו וְהִלְבַּשְׁתָּ֖ם אֶת־אֶלְעָזָ֣ר בְּנ֑וֹ וְאַהֲרֹ֥ן יֵאָסֵ֖ף וּמֵ֥ת שָֽׁם׃ וְהַפְשֵׁט אֶת אַהֲרֹן מִבְּגָדָיו וְתַלְבִּישׁ אוֹתָם אֶת אֶלְעָזָר בְּנוֹ, וְנִשְׁמַת אַהֲרֹן תֵּאָסֵף אֶל מְקוֹם הַנְּשָׁמוֹת, וְהוּא יָמוּת שָׁם.
כז וַיַּ֣עַשׂ מֹשֶׁ֔ה כַּאֲשֶׁ֖ר צִוָּ֣ה יְהֹוָ֑ה וַֽיַּעֲלוּ֙ אֶל־הֹ֣ר הָהָ֔ר לְעֵינֵ֖י כׇּל־הָעֵדָֽה׃ מֹשֶׁה עָשָׂה כְּמוֹ שֶׁה' צִוָּה, הֵם עָלוּ לְהוֹר-הָהָר לְעֵינֵי כָּל הַצִּבּוּר.
כח וַיַּפְשֵׁט֩ מֹשֶׁ֨ה אֶֽת־אַהֲרֹ֜ן אֶת־בְּגָדָ֗יו וַיַּלְבֵּ֤שׁ אֹתָם֙ אֶת־אֶלְעָזָ֣ר בְּנ֔וֹ וַיָּ֧מׇת אַהֲרֹ֛ן שָׁ֖ם בְּרֹ֣אשׁ הָהָ֑ר וַיֵּ֧רֶד מֹשֶׁ֛ה וְאֶלְעָזָ֖ר מִן־הָהָֽר׃ מֹשֶׁה הִפְשִׁיט אֶת אַהֲרֹן מִבְּגָדָיו וְהִלְבִּישׁ בָּהֶם אֶת אֶלְעָזָר בְּנוֹ, וְאַהֲרֹן מֵת שָׁם בְּרֹאשׁ הָהָר, וּמֹשֶׁה וְאֶלְעָזָר יָרְדוּ מֵהָהָר.
כט וַיִּרְאוּ֙ כׇּל־הָ֣עֵדָ֔ה כִּ֥י גָוַ֖ע אַהֲרֹ֑ן וַיִּבְכּ֤וּ אֶֽת־אַהֲרֹן֙ שְׁלֹשִׁ֣ים י֔וֹם כֹּ֖ל בֵּ֥ית יִשְׂרָאֵֽל׃ כָּל הַצִּבּוּר רָאוּ שֶׁאַהֲרֹן מֵת, וְכָל בֵּית יִשְׂרָאֵל בָּכוּ עַל אַהֲרֹן שְׁלוֹשִׁים יוֹם.
ס פָּרָשַׁת הַחְרָמַת הַכְּנַעֲנִי
כא א וַיִּשְׁמַ֞ע הַכְּנַעֲנִ֤י מֶֽלֶךְ־עֲרָד֙ יֹשֵׁ֣ב הַנֶּ֔גֶב כִּ֚י בָּ֣א יִשְׂרָאֵ֔ל דֶּ֖רֶךְ הָאֲתָרִ֑ים וַיִּלָּ֙חֶם֙ בְּיִשְׂרָאֵ֔ל וַיִּ֥שְׁבְּ ׀ מִמֶּ֖נּוּ שֶֽׁבִי׃ מֶלֶךְ עֲרָד הַכְּנַעֲנִי שֶׁיּוֹשֵׁב בַּנֶּגֶב שָׁמַע שֶׁיִּשְׂרָאֵל בָּא בְּדֶרֶךְ הַתַּיָּרִים, וְהוּא נִלְחַם בְּיִשְׂרָאֵל וְשָׁבָה מִמֶּנּוּ שֶׁבִי.
ב וַיִּדַּ֨ר יִשְׂרָאֵ֥ל נֶ֛דֶר לַֽיהֹוָ֖ה וַיֹּאמַ֑ר אִם־נָתֹ֨ן תִּתֵּ֜ן אֶת־הָעָ֤ם הַזֶּה֙ בְּיָדִ֔י וְהַֽחֲרַמְתִּ֖י אֶת־עָרֵיהֶֽם׃ יִשְׂרָאֵל נָדַר נֶדֶר לַה' וְאָמַר: אִם תִּתֵּן אֶת הָעָם הַזֶּה בְּיָדִי אַשְׁמִיד אֶת עָרֵיהֶם.
ג וַיִּשְׁמַ֨ע יְהֹוָ֜ה בְּק֣וֹל יִשְׂרָאֵ֗ל וַיִּתֵּן֙ אֶת־הַֽכְּנַעֲנִ֔י וַיַּחֲרֵ֥ם אֶתְהֶ֖ם וְאֶת־עָרֵיהֶ֑ם וַיִּקְרָ֥א שֵׁם־הַמָּק֖וֹם חׇרְמָֽה׃ ה' שָׁמַע בְּקוֹל יִשְׂרָאֵל וְנָתַן בְּיָדָם אֶת הַכְּנַעֲנִי וְהוּא הִשְׁמִיד אוֹתָם וְאֶת עָרֵיהֶם, וְקָרָא אֶת שֵׁם הַמָּקוֹם חָרְמָה, כְּלוֹמַר הַשְׁמָדָה.
פ פָּרָשַׁת נְחַשׁ הַנְּחוֹשֶׁת
ד וַיִּסְע֞וּ מֵהֹ֤ר הָהָר֙ דֶּ֣רֶךְ יַם־ס֔וּף לִסְבֹ֖ב אֶת־אֶ֣רֶץ אֱד֑וֹם וַתִּקְצַ֥ר נֶֽפֶשׁ־הָעָ֖ם בַּדָּֽרֶךְ׃ הֵם נָסְעוּ מֵהוֹר-הָהָר דֶּרֶךְ יַם סוּף לְהַקִּיף אֶת אֶרֶץ אֱדוֹם, וְנִמְאַס לָעָם בַּדֶּרֶךְ.
ה וַיְדַבֵּ֣ר הָעָ֗ם בֵּֽאלֹהִים֮ וּבְמֹשֶׁה֒ לָמָ֤ה הֶֽעֱלִיתֻ֙נוּ֙ מִמִּצְרַ֔יִם לָמ֖וּת בַּמִּדְבָּ֑ר כִּ֣י אֵ֥ין לֶ֙חֶם֙ וְאֵ֣ין מַ֔יִם וְנַפְשֵׁ֣נוּ קָ֔צָה בַּלֶּ֖חֶם הַקְּלֹקֵֽל׃ הָעָם דִּבֵּר נֶגֶד אֱלֹהִים וּמֹשֶׁה: לָמָה הֶעֱלִיתֶם אוֹתָנוּ מִמִּצְרַיִם לָמוּת בַּמִּדְבָּר, הֲרֵי אֵין כָּאן לֶחֶם וְאֵין מַיִם, וְנִמְאַס לָנוּ מֵהַלֶּחֶם הַמִּתְקַלְקֵל בְּקַלּוּת.
ו וַיְשַׁלַּ֨ח יְהֹוָ֜ה בָּעָ֗ם אֵ֚ת הַנְּחָשִׁ֣ים הַשְּׂרָפִ֔ים וַֽיְנַשְּׁכ֖וּ אֶת־הָעָ֑ם וַיָּ֥מׇת עַם־רָ֖ב מִיִּשְׂרָאֵֽל׃ ה' שִׁלַּח בָּעָם אֶת הַנְּחָשִׁים הָאַרְסִיִּים וְהֵם נָשְׁכוּ אֶת הָעָם, וּמֵתוּ אֲנָשִׁים רַבִּים מֵעַם יִשְׂרָאֵל.
ז וַיָּבֹא֩ הָעָ֨ם אֶל־מֹשֶׁ֜ה וַיֹּאמְר֣וּ חָטָ֗אנוּ כִּֽי־דִבַּ֤רְנוּ בַֽיהֹוָה֙ וָבָ֔ךְ הִתְפַּלֵּל֙ אֶל־יְהֹוָ֔ה וְיָסֵ֥ר מֵעָלֵ֖ינוּ אֶת־הַנָּחָ֑שׁ וַיִּתְפַּלֵּ֥ל מֹשֶׁ֖ה בְּעַ֥ד הָעָֽם׃ הָעָם בָּאוּ לְמֹשֶׁה וְאָמְרוּ: חָטָאנוּ כְּשֶׁדִּבַּרְנוּ נֶגֶד ה' וְנֶגְדְּךָ, הִתְפַּלֵּל אֶל ה' שֶׁיָּסִיר מֵעָלֵינוּ אֶת הַנְּחָשִׁים! מֹשֶׁה הִתְפַּלֵּל עַל הָעָם.
ח וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה עֲשֵׂ֤ה לְךָ֙ שָׂרָ֔ף וְשִׂ֥ים אֹת֖וֹ עַל־נֵ֑ס וְהָיָה֙ כׇּל־הַנָּשׁ֔וּךְ וְרָאָ֥ה אֹת֖וֹ וָחָֽי׃ ה' אָמַר לְמֹשֶׁה: עֲשֵׂה לְךָ דְּמוּת נָחָשׁ אַרְסִי, וְשִׂים אוֹתוֹ עַל עַמּוּד גָּבוֹהַּ, וְכָל נָשׁוּךְ יִרְאֶה אוֹתוֹ וְיִחְיֶה.
ט וַיַּ֤עַשׂ מֹשֶׁה֙ נְחַ֣שׁ נְחֹ֔שֶׁת וַיְשִׂמֵ֖הוּ עַל־הַנֵּ֑ס וְהָיָ֗ה אִם־נָשַׁ֤ךְ הַנָּחָשׁ֙ אֶת־אִ֔ישׁ וְהִבִּ֛יט אֶל־נְחַ֥שׁ הַנְּחֹ֖שֶׁת וָחָֽי׃ מֹשֶׁה עָשָׂה נָחָשׁ מִנְּחוֹשֶׁת שָׂם אוֹתוֹ עַל עַמּוּד גָּבוֹהַּ, וְאִם נָחָשׁ נָשַׁךְ אִישׁ, הוּא הִבִּיט אֶל נְחַשׁ הַנְּחוֹשֶׁת וְנִשְׁאַר חַי.

ששי[עריכה]

פָּרָשַׁת הַמַּסָּעוֹת וְהַבְּאֵר
י וַיִּסְע֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל וַֽיַּחֲנ֖וּ בְּאֹבֹֽת׃ בְּנֵי יִשְׂרָאֵל נָסְעוּ וְחָנוּ בְּאוֹבוֹת.
יא וַיִּסְע֖וּ מֵאֹבֹ֑ת וַֽיַּחֲנ֞וּ בְּעִיֵּ֣י הָֽעֲבָרִ֗ים בַּמִּדְבָּר֙ אֲשֶׁר֙ עַל־פְּנֵ֣י מוֹאָ֔ב מִמִּזְרַ֖ח הַשָּֽׁמֶשׁ׃ הֵם נָסְעוּ מֵאוֹבוֹת וְחָנוּ בַּחֳרָבוֹת שֶׁל הַמַּעֲבָרִים, בַּמִּדְבָּר שֶׁמִּמּוּל לְמוֹאָב מִצַּד מִזְרָח.
יב מִשָּׁ֖ם נָסָ֑עוּ וַֽיַּחֲנ֖וּ בְּנַ֥חַל זָֽרֶד׃ מִשָּׁם נָסְעוּ וְחָנוּ בְּנַחַל זֶרֶד.
יג מִשָּׁם֮ נָסָ֒עוּ֒ וַֽיַּחֲנ֗וּ מֵעֵ֤בֶר אַרְנוֹן֙ אֲשֶׁ֣ר בַּמִּדְבָּ֔ר הַיֹּצֵ֖א מִגְּבֻ֣ל הָֽאֱמֹרִ֑י כִּ֤י אַרְנוֹן֙ גְּב֣וּל מוֹאָ֔ב בֵּ֥ין מוֹאָ֖ב וּבֵ֥ין הָאֱמֹרִֽי׃ מִשָּׁם נָסְעוּ וְחָנוּ מֵעֵבֶר לְנַחַל אַרְנוֹן שֶׁבַּמִּדְבָּר, שֶׁזּוֹרֵם מִשֶּׁטַח הָאֱמוֹרִי, כִּי נַחַל אַרְנוֹן הוּא בִּתְחוּם מוֹאָב, הֵם חָנוּ בֵּין מוֹאָב וּבֵין הָאֱמוֹרִי.
יד עַל־כֵּן֙ יֵֽאָמַ֔ר בְּסֵ֖פֶר מִלְחֲמֹ֣ת יְהֹוָ֑ה אֶת־וָהֵ֣ב בְּסוּפָ֔ה וְאֶת־הַנְּחָלִ֖ים אַרְנֽוֹן׃ לָכֵן נֶאֱמַר בְּסֵפֶר מִלְחֲמוֹת ה' עַל הַנִּסִּים שֶׁנַּעֲשׂוּ בְּיַם סוּף וּבְנַחַל אַרְנוֹן.
טו וְאֶ֙שֶׁד֙ הַנְּחָלִ֔ים אֲשֶׁ֥ר נָטָ֖ה לְשֶׁ֣בֶת עָ֑ר וְנִשְׁעַ֖ן לִגְב֥וּל מוֹאָֽב׃ וְחָנוּ לְיַד מַפַּל הַנַּחַל שֶׁנּוֹטֶה לְמוּל עָר – סָמוּךְ לִגְבוּל מוֹאָב.
טז וּמִשָּׁ֖ם בְּאֵ֑רָה הִ֣וא הַבְּאֵ֗ר אֲשֶׁ֨ר אָמַ֤ר יְהֹוָה֙ לְמֹשֶׁ֔ה אֱסֹף֙ אֶת־הָעָ֔ם וְאֶתְּנָ֥ה לָהֶ֖ם מָֽיִם׃ וּמִשָּׁם לַבְּאֵר, הִיא הַבְּאֵר שֶׁה' אָמַר לְמֹשֶׁה אֱסוֹף אֶת הָעָם וְאֶתֵּן לָהֶם מַיִם.
ס
יז אָ֚ז יָשִׁ֣יר יִשְׂרָאֵ֔ל אֶת־הַשִּׁירָ֖ה הַזֹּ֑את עֲלִ֥י בְאֵ֖ר עֱנוּ־לָֽהּ׃ אָז שָׁרוּ יִשְׂרָאֵל אֶת הַשִּׁירָה הַזֹּאת: עֲלִי בְּאֵר! שַׁבְּחוּ לָהּ!
יח בְּאֵ֞ר חֲפָר֣וּהָ שָׂרִ֗ים כָּר֙וּהָ֙ נְדִיבֵ֣י הָעָ֔ם בִּמְחֹקֵ֖ק בְּמִשְׁעֲנֹתָ֑ם וּמִמִּדְבָּ֖ר מַתָּנָֽה׃ בְּאֵר שֶׁחָפְרוּ אוֹתָהּ שָׂרִים, כָּרוּ אוֹתָהּ רָאשֵׁי הָעָם מֹשֶׁה וְאַהֲרֹן בְּמַטֶּה – בַּמִּשְׁעֶנֶת שֶׁלָּהֶם, הִיא מַתָּנָה מִמִּדְבָּר.
יט וּמִמַּתָּנָ֖ה נַחֲלִיאֵ֑ל וּמִנַּחֲלִיאֵ֖ל בָּמֽוֹת׃ מֵהַמָּקוֹם שֶׁנִּתְּנָה בְּמַתָּנָה נָסְעָה אִתָּם הַבְּאֵר לַנַּחַל, וּמֵהַנַּחַל לָרָמוֹת.
כ וּמִבָּמ֗וֹת הַגַּיְא֙ אֲשֶׁר֙ בִּשְׂדֵ֣ה מוֹאָ֔ב רֹ֖אשׁ הַפִּסְגָּ֑ה וְנִשְׁקָ֖פָה עַל־פְּנֵ֥י הַיְשִׁימֹֽן׃ וּמֵהָרָמוֹת לְרֹאשׁ הַפִּסְגָּה שֶׁמַּשְׁקִיף אֶל מוּל הַשְּׁמָמָה – לָעֵמֶק שֶׁבִּשְׂדֵה מוֹאָב.

שביעי (רביעי במחוברין)[עריכה]

פ פָּרָשַׁת כִּבּוּשׁ הָאֱמוֹרִי
כא וַיִּשְׁלַ֤ח יִשְׂרָאֵל֙ מַלְאָכִ֔ים אֶל־סִיחֹ֥ן מֶֽלֶךְ־הָאֱמֹרִ֖י לֵאמֹֽר׃ יִשְׂרָאֵל שָׁלַח שְׁלִיחִים לְסִיחוֹן מֶלֶךְ הָאֱמוֹרִי לוֹמַר לוֹ:
כב אֶעְבְּרָ֣ה בְאַרְצֶ֗ךָ לֹ֤א נִטֶּה֙ בְּשָׂדֶ֣ה וּבְכֶ֔רֶם לֹ֥א נִשְׁתֶּ֖ה מֵ֣י בְאֵ֑ר בְּדֶ֤רֶךְ הַמֶּ֙לֶךְ֙ נֵלֵ֔ךְ עַ֥ד אֲשֶֽׁר־נַעֲבֹ֖ר גְּבֻלֶֽךָ׃ נַעֲבוֹר בְּאַרְצְךָ לֹא נִסְטֶה לַעֲבוֹר בְּשָׂדֶה אוֹ בְּכֶרֶם וְלֹא נִשְׁתֶּה מֵי בְּאֵר, נֵלֵךְ רַק בַּדֶּרֶךְ הָרָאשִׁית עַד שֶׁנַּעֲבוֹר אֶת גְּבוּלְךָ.
כג וְלֹא־נָתַ֨ן סִיחֹ֣ן אֶת־יִשְׂרָאֵל֮ עֲבֹ֣ר בִּגְבֻלוֹ֒ וַיֶּאֱסֹ֨ף סִיחֹ֜ן אֶת־כׇּל־עַמּ֗וֹ וַיֵּצֵ֞א לִקְרַ֤את יִשְׂרָאֵל֙ הַמִּדְבָּ֔רָה וַיָּבֹ֖א יָ֑הְצָה וַיִּלָּ֖חֶם בְּיִשְׂרָאֵֽל׃ סִיחוֹן לֹא נָתַן לְעַם יִשְׂרָאֵל לַעֲבוֹר בִּגְבוּלוֹ – סִיחוֹן אָסַף אֶת כָּל עַמּוֹ וְיָצָא לִקְרַאת יִשְׂרָאֵל לַמִּדְבָּר, הוּא בָּא לְיַהַץ וְנִלְחַם בְּיִשְׂרָאֵל.
כד וַיַּכֵּ֥הוּ יִשְׂרָאֵ֖ל לְפִי־חָ֑רֶב וַיִּירַ֨שׁ אֶת־אַרְצ֜וֹ מֵֽאַרְנֹ֗ן עַד־יַבֹּק֙ עַד־בְּנֵ֣י עַמּ֔וֹן כִּ֣י עַ֔ז גְּב֖וּל בְּנֵ֥י עַמּֽוֹן׃ יִשְׂרָאֵל הִכָּה אוֹתוֹ בְּלַהַב הַחֶרֶב, וְכָבַשׁ אֶת אַרְצוֹ מֵאַרְנוֹן עַד יַבּוֹק – עַד בְּנֵי עַמּוֹן, כִּי גְּבוּל בְּנֵי עַמּוֹן הָיָה חָזָק.
כה וַיִּקַּח֙ יִשְׂרָאֵ֔ל אֵ֥ת כׇּל־הֶעָרִ֖ים הָאֵ֑לֶּה וַיֵּ֤שֶׁב יִשְׂרָאֵל֙ בְּכׇל־עָרֵ֣י הָֽאֱמֹרִ֔י בְּחֶשְׁבּ֖וֹן וּבְכׇל־בְּנֹתֶֽיהָ׃ יִשְׂרָאֵל לָקַח אֶת כָּל הֶעָרִים הָאֵלֶּה, יִשְׂרָאֵל הִתְיַשֵּׁב בְּכָל עָרֵי הָאֱמוֹרִי – בְּחֶשְׁבּוֹן וּבְכָל הַכְּפָרִים שֶׁסְּבִיבָהּ.
כו כִּ֣י חֶשְׁבּ֔וֹן עִ֗יר סִיחֹ֛ן מֶ֥לֶךְ הָאֱמֹרִ֖י הִ֑וא וְה֣וּא נִלְחַ֗ם בְּמֶ֤לֶךְ מוֹאָב֙ הָֽרִאשׁ֔וֹן וַיִּקַּ֧ח אֶת־כׇּל־אַרְצ֛וֹ מִיָּד֖וֹ עַד־אַרְנֹֽן׃ כִּי חֶשְׁבּוֹן כְּבָר לֹא הָיְתָה שַׁיֶּכֶת לְמוֹאָב, אֶלָּא הָיְתָה הָעִיר שֶׁל סִיחוֹן מֶלֶךְ הָאֱמוֹרִי, וְהוּא נִלְחַם בְּמֶלֶךְ מוֹאָב הָרִאשׁוֹן וְלָקַח אֶת כָּל אַרְצוֹ מִיָּדוֹ עַד אַרְנוֹן.
כז עַל־כֵּ֛ן יֹאמְר֥וּ הַמֹּשְׁלִ֖ים בֹּ֣אוּ חֶשְׁבּ֑וֹן תִּבָּנֶ֥ה וְתִכּוֹנֵ֖ן עִ֥יר סִיחֽוֹן׃ עַל כֵּן אָמְרוּ אוֹמְרֵי הַמְּשָׁלִים: בּוֹאוּ לְחֶשְׁבּוֹן! הָעִיר שֶׁל סִיחוֹן תִּבָּנֶה וְתִתְבַּסֵּס.
כח כִּי־אֵשׁ֙ יָֽצְאָ֣ה מֵֽחֶשְׁבּ֔וֹן לֶהָבָ֖ה מִקִּרְיַ֣ת סִיחֹ֑ן אָֽכְלָה֙ עָ֣ר מוֹאָ֔ב בַּעֲלֵ֖י בָּמ֥וֹת אַרְנֹֽן׃ כִּי יָצְאָה אֵשׁ מֵחֶשְׁבּוֹן – לֶהָבָה מִקִּרְיַת סִיחוֹן, וְהִשְׁמִידָה אֶת הָעִיר עָר שֶׁל מוֹאָב וְאֶת הַמִּישׁוֹרִים שֶׁלְּיַד רָמוֹת אַרְנוֹן.
כט אוֹי־לְךָ֣ מוֹאָ֔ב אָבַ֖דְתָּ עַם־כְּמ֑וֹשׁ נָתַ֨ן בָּנָ֤יו פְּלֵיטִם֙ וּבְנֹתָ֣יו בַּשְּׁבִ֔ית לְמֶ֥לֶךְ אֱמֹרִ֖י סִיחֽוֹן׃ אוֹי לְךָ מוֹאָב, אָבַדְתָּ הָעָם הָעוֹבֵד לָאֱלִיל כְּמוֹשׁ, בָּנָיו נַעֲשׂוּ פְּלִיטִים וּבְנוֹתָיו נִלְקְחוּ בַּשֶּׁבִי לְסִיחוֹן מֶלֶךְ הָאֱמוֹרִי.
ל וַנִּירָ֛ם אָבַ֥ד חֶשְׁבּ֖וֹן עַד־דִּיבֹ֑ן וַנַּשִּׁ֣ים עַד־נֹ֔פַח אֲשֶׁ֖רׄ עַד־מֵֽידְבָֽא׃ וַאֲנַחְנוּ גֵּרַשְׁנוּ אוֹתָם – אָבַד חֶשְׁבּוֹן עַד דִּיבוֹן, וְעָשִׂינוּ שְׁמָמָה עַד נוֹפַח הַקְּרוֹבָה לְמֵידְבָא.
לא וַיֵּ֙שֶׁב֙ יִשְׂרָאֵ֔ל בְּאֶ֖רֶץ הָאֱמֹרִֽי׃ יִשְׂרָאֵל הִתְיַשֵּׁב בְּאֶרֶץ הָאֱמוֹרִי.
לב וַיִּשְׁלַ֤ח מֹשֶׁה֙ לְרַגֵּ֣ל אֶת־יַעְזֵ֔ר וַֽיִּלְכְּד֖וּ בְּנֹתֶ֑יהָ ויירש וַיּ֖וֹרֶשׁ אֶת־הָאֱמֹרִ֥י אֲשֶׁר־שָֽׁם׃ מֹשֶׁה שָׁלַח אֲנָשִׁים לְרַגֵּל אֶת הָעִיר יַעְזֵר וְיִשְׂרָאֵל לָכְדוּ אֶת הַיִּשּׁוּבִים שֶׁסְּבִיבָהּ וְגֵרְשׁוּ אֶת הָאֱמוֹרִי שֶׁשָּׁם.
פָּרָשַׁת כִּבּוּשׁ הַבָּשָׁן
לג וַיִּפְנוּ֙ וַֽיַּעֲל֔וּ דֶּ֖רֶךְ הַבָּשָׁ֑ן וַיֵּצֵ֣א עוֹג֩ מֶֽלֶךְ־הַבָּשָׁ֨ן לִקְרָאתָ֜ם ה֧וּא וְכׇל־עַמּ֛וֹ לַמִּלְחָמָ֖ה אֶדְרֶֽעִי׃ הֵם פָּנוּ וְעָלוּ דֶּרֶךְ הַבָּשָׁן, עוֹג מֶלֶךְ הַבָּשָׁן יָצָא לִקְרָאתָם לַמִּלְחָמָה בְּאֶדְרֶעִי הוּא וְכָל עַמּוֹ.

מפטיר[עריכה]

לד וַיֹּ֨אמֶר יְהֹוָ֤ה אֶל־מֹשֶׁה֙ אַל־תִּירָ֣א אֹת֔וֹ כִּ֣י בְיָדְךָ֞ נָתַ֧תִּי אֹת֛וֹ וְאֶת־כׇּל־עַמּ֖וֹ וְאֶת־אַרְצ֑וֹ וְעָשִׂ֣יתָ לּ֔וֹ כַּאֲשֶׁ֣ר עָשִׂ֗יתָ לְסִיחֹן֙ מֶ֣לֶךְ הָֽאֱמֹרִ֔י אֲשֶׁ֥ר יוֹשֵׁ֖ב בְּחֶשְׁבּֽוֹן׃ ה' אָמַר לְמֹשֶׁה: אַל תִּפְחַד מִמֶּנּוּ כִּי אֶמְסוֹר לְיָדְךָ אוֹתוֹ וְאֶת כָּל עַמּוֹ וְאֶת אַרְצוֹ, וְתַעֲשֶׂה לוֹ כְּמוֹ שֶׁעָשִׂיתָ לְסִיחוֹן מֶלֶךְ הָאֱמוֹרִי שֶׁגָּר בְּחֶשְׁבּוֹן.
לה וַיַּכּ֨וּ אֹת֤וֹ וְאֶת־בָּנָיו֙ וְאֶת־כׇּל־עַמּ֔וֹ עַד־בִּלְתִּ֥י הִשְׁאִֽיר־ל֖וֹ שָׂרִ֑יד וַיִּֽירְשׁ֖וּ אֶת־אַרְצֽוֹ׃ הֵם הִכּוּ אוֹתוֹ וְאֶת בָּנָיו וְאֶת כָּל עַמּוֹ עַד שֶׁלֹּא נִשְׁאַר לוֹ שָׁרִיד, וְהֵם כָּבְשׁוּ אֶת אַרְצוֹ.
כב א וַיִּסְע֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל וַֽיַּחֲנוּ֙ בְּעַֽרְב֣וֹת מוֹאָ֔ב מֵעֵ֖בֶר לְיַרְדֵּ֥ן יְרֵחֽוֹ׃ בְּנֵי יִשְׂרָאֵל נָסְעוּ וְחָנוּ בְּעַרְבוֹת מוֹאָב, מֵעֵבֶר לַיַּרְדֵּן מוּל יְרִיחוֹ.
