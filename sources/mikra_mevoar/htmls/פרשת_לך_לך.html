ראשון[עריכה]

פ פָּרָשַׁת הָעֲלִיָּה לָאָרֶץ
יב א וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־אַבְרָ֔ם לֶךְ־לְךָ֛ מֵאַרְצְךָ֥ וּמִמּֽוֹלַדְתְּךָ֖ וּמִבֵּ֣ית אָבִ֑יךָ אֶל־הָאָ֖רֶץ אֲשֶׁ֥ר אַרְאֶֽךָּ׃ ה' אָמַר לְאַבְרָם לֵךְ לְךָ מֵאַרְצְךָ, מִמִּשְׁפַּחְתְּךָ וּמֵהַבַּיִת שֶׁל אָבִיךָ אֶל הָאָרֶץ שֶׁאַרְאֶה לְךָ.
ב וְאֶֽעֶשְׂךָ֙ לְג֣וֹי גָּד֔וֹל וַאֲבָ֣רֶכְךָ֔ וַאֲגַדְּלָ֖ה שְׁמֶ֑ךָ וֶהְיֵ֖ה בְּרָכָֽה׃ וְאֶעֱשֶׂה מִמְּךָ עַם גָּדוֹל וַאֲבָרֵךְ אוֹתְךָ וַאֲפַרְסֵם אֶת שִׁמְךָ וְתִהְיֶה מְבוֹרָךְ.
ג וַאֲבָֽרְﬞכָה֙ מְבָ֣רְכֶ֔יךָ וּמְקַלֶּלְךָ֖ אָאֹ֑ר וְנִבְרְכ֣וּ בְךָ֔ כֹּ֖ל מִשְׁפְּחֹ֥ת הָאֲדָמָֽה׃ וַאֲבָרֵךְ אֶת הַמְּבָרְכִים אוֹתְךָ וַאֲקַלֵּל אֶת הַמְּקַלְּלִים אוֹתְךָ, וְכָל מִשְׁפְּחוֹת הָאֲדָמָה יִרְצוּ לְהִתְבָּרֵךְ לִהְיוֹת כָּמוֹךָ.
ד וַיֵּ֣לֶךְ אַבְרָ֗ם כַּאֲשֶׁ֨ר דִּבֶּ֤ר אֵלָיו֙ יְהֹוָ֔ה וַיֵּ֥לֶךְ אִתּ֖וֹ ל֑וֹט וְאַבְרָ֗ם בֶּן־חָמֵ֤שׁ שָׁנִים֙ וְשִׁבְעִ֣ים שָׁנָ֔ה בְּצֵאת֖וֹ מֵחָרָֽן׃ אַבְרָם הָלַךְ כְּמוֹ שֶׁה' דִּבֵּר אִתּוֹ וְלוֹט הָלַךְ אִתּוֹ, אַבְרָם הָיָה בֶּן שִׁבְעִים וְחָמֵשׁ שָׁנִים כְּשֶׁיָּצָא מֵחָרָן.
ה וַיִּקַּ֣ח אַבְרָם֩ אֶת־שָׂרַ֨י אִשְׁתּ֜וֹ וְאֶת־ל֣וֹט בֶּן־אָחִ֗יו וְאֶת־כׇּל־רְכוּשָׁם֙ אֲשֶׁ֣ר רָכָ֔שׁוּ וְאֶת־הַנֶּ֖פֶשׁ אֲשֶׁר־עָשׂ֣וּ בְחָרָ֑ן וַיֵּצְא֗וּ לָלֶ֙כֶת֙ אַ֣רְצָה כְּנַ֔עַן וַיָּבֹ֖אוּ אַ֥רְצָה כְּנָֽעַן׃ אַבְרָם לָקַח אֶת שָׂרַי אִשְׁתּוֹ וְאֶת לוֹט בֶּן אָחִיו וְאֶת כָּל רְכוּשָׁם שֶׁרָכְשׁוּ וְאֶת הָעֲבָדִים שֶׁקָּנוּ בְּחָרָן, הֵם יָצְאוּ לָלֶכֶת לְאֶרֶץ כְּנַעַן וּבָאוּ לְאֶרֶץ כְּנַעַן.
ו וַיַּעֲבֹ֤ר אַבְרָם֙ בָּאָ֔רֶץ עַ֚ד מְק֣וֹם שְׁכֶ֔ם עַ֖ד אֵל֣וֹן מוֹרֶ֑ה וְהַֽכְּנַעֲנִ֖י אָ֥ז בָּאָֽרֶץ׃ אַבְרָם עָבַר בָּאָרֶץ עַד הַמָּקוֹם שֶׁל הָאִישׁ שְׁכֶם – עַד אֵלוֹן מוֹרֶה, וְהַכְּנַעֲנִי הָיָה אָז בָּאָרֶץ.
ז וַיֵּרָ֤א יְהֹוָה֙ אֶל־אַבְרָ֔ם וַיֹּ֕אמֶר לְזַ֨רְעֲךָ֔ אֶתֵּ֖ן אֶת־הָאָ֣רֶץ הַזֹּ֑את וַיִּ֤בֶן שָׁם֙ מִזְבֵּ֔חַ לַיהֹוָ֖ה הַנִּרְאֶ֥ה אֵלָֽיו׃ ה' הִתְגַּלָּה לְאַבְרָם, וְאָמַר לוֹ: אֶתֵּן לְצֶאֱצָאֶיךָ אֶת הָאָרֶץ הַזֹּאת, הוּא בָּנָה שָׁם מִזְבֵּחַ לַה' שֶׁנִּגְלָה אֵלָיו.
ח וַיַּעְתֵּ֨ק מִשָּׁ֜ם הָהָ֗רָה מִקֶּ֛דֶם לְבֵֽית־אֵ֖ל וַיֵּ֣ט אׇהֳלֹ֑ה בֵּֽית־אֵ֤ל מִיָּם֙ וְהָעַ֣י מִקֶּ֔דֶם וַיִּֽבֶן־שָׁ֤ם מִזְבֵּ֙חַ֙ לַֽיהֹוָ֔ה וַיִּקְרָ֖א בְּשֵׁ֥ם יְהֹוָֽה׃ הוּא הֶעֱבִיר מִשָּׁם אֶת אָהֳלוֹ לָהָר מִזְרָחִית לְבֵית אֵל, וְהֵקִים שָׁם אֶת אָהֳלוֹ, בַּמָּקוֹם שֶׁבֵּית אֵל מִמַּעֲרָב וְהָעַי מִמִּזְרָח, הוּא בָּנָה שָׁם מִזְבֵּחַ לַה' וְהִתְפַּלֵּל לַה'.
ט וַיִּסַּ֣ע אַבְרָ֔ם הָל֥וֹךְ וְנָס֖וֹעַ הַנֶּֽגְבָּה׃ אַבְרָם נָסַע וְהִתְקַדֵּם לְעֵבֶר הַנֶּגֶב.
פ פָּרָשַׁת מִצְרַיִם
י וַיְהִ֥י רָעָ֖ב בָּאָ֑רֶץ וַיֵּ֨רֶד אַבְרָ֤ם מִצְרַ֙יְמָה֙ לָג֣וּר שָׁ֔ם כִּֽי־כָבֵ֥ד הָרָעָ֖ב בָּאָֽרֶץ׃ הָיָה רָעָב בָּאָרֶץ, אַבְרָם יָרַד לְמִצְרַיִם לָגוּר שָׁם כִּי הָרָעָב בָּאָרֶץ הָיָה חָמוּר.
יא וַיְהִ֕י כַּאֲשֶׁ֥ר הִקְרִ֖יב לָב֣וֹא מִצְרָ֑יְמָה וַיֹּ֙אמֶר֙ אֶל־שָׂרַ֣י אִשְׁתּ֔וֹ הִנֵּה־נָ֣א יָדַ֔עְתִּי כִּ֛י אִשָּׁ֥ה יְפַת־מַרְאֶ֖ה אָֽתְּ׃ כְּשֶׁהִתְקָרֵב לָבוֹא לְמִצְרַיִם אָמַר לְשָׂרַי אִשְׁתּוֹ: הֲרֵי יָדַעְתִּי שֶׁאַתְּ אִשָּׁה יָפָה.
יב וְהָיָ֗ה כִּֽי־יִרְא֤וּ אֹתָךְ֙ הַמִּצְרִ֔ים וְאָמְר֖וּ אִשְׁתּ֣וֹ זֹ֑את וְהָרְג֥וּ אֹתִ֖י וְאֹתָ֥ךְ יְחַיּֽוּ׃ וּכְשֶׁיִּרְאוּ אוֹתָךְ הַמִּצְרִים, יֹאמְרוּ: זֹאת אִשְׁתּוֹ! וְיַהַרְגוּ אוֹתִי, וְאוֹתָךְ יְחַיּוּ כְּדֵי לָקַחַת אוֹתָךְ.
יג אִמְרִי־נָ֖א אֲחֹ֣תִי אָ֑תְּ לְמַ֙עַן֙ יִֽיטַב־לִ֣י בַעֲבוּרֵ֔ךְ וְחָיְתָ֥ה נַפְשִׁ֖י בִּגְלָלֵֽךְ׃ אִמְרִי בְּבַקָּשָׁה שֶׁאַתְּ אֲחוֹתִי, כְּדֵי שֶׁיֵּטִיבוּ לִי תְּמוּרָתֵךְ, וְאֶחְיֵה בִּזְכוּתֵךְ.

שני[עריכה]

יד וַיְהִ֕י כְּב֥וֹא אַבְרָ֖ם מִצְרָ֑יְמָה וַיִּרְא֤וּ הַמִּצְרִים֙ אֶת־הָ֣אִשָּׁ֔ה כִּֽי־יָפָ֥ה הִ֖וא מְאֹֽד׃ כְּשֶׁאַבְרָם בָּא לְמִצְרַיִם, הַמִּצְרִים רָאוּ אֶת הָאִשָּׁה שֶׁהִיא יָפָה מְאוֹד.
טו וַיִּרְא֤וּ אֹתָהּ֙ שָׂרֵ֣י פַרְעֹ֔ה וַיְהַֽלְל֥וּ אֹתָ֖הּ אֶל־פַּרְעֹ֑ה וַתֻּקַּ֥ח הָאִשָּׁ֖ה בֵּ֥ית פַּרְעֹֽה׃ שָׂרֵי פַּרְעֹה רָאוּ אוֹתָהּ וְשִׁבְּחוּ אוֹתָהּ שֶׁהִיא רְאוּיָה לְפַרְעֹה, וְהָאִשָּׁה נִלְקְחָה לְבֵית פַּרְעֹה.
טז וּלְאַבְרָ֥ם הֵיטִ֖יב בַּעֲבוּרָ֑הּ וַֽיְהִי־ל֤וֹ צֹאן־וּבָקָר֙ וַחֲמֹרִ֔ים וַעֲבָדִים֙ וּשְׁפָחֹ֔ת וַאֲתֹנֹ֖ת וּגְמַלִּֽים׃ פַּרְעֹה הֵיטִיב לְאַבְרָם תְּמוּרָתָהּ, וְנָתַן לוֹ צֹאן וּבָקָר וַחֲמוֹרִים וַעֲבָדִים וּשְׁפָחוֹת וַאֲתוֹנוֹת וּגְמַלִּים.
יז וַיְנַגַּ֨ע יְהֹוָ֧ה ׀ אֶת־פַּרְעֹ֛ה נְגָעִ֥ים גְּדֹלִ֖ים וְאֶת־בֵּית֑וֹ עַל־דְּבַ֥ר שָׂרַ֖י אֵ֥שֶׁת אַבְרָֽם׃ ה' הִלְקָה אֶת פַּרְעֹה וְאֶת מִשְׁפַּחְתּוֹ בְּנִגְעֵי צָרַעַת גְּדוֹלִים מִפְּנֵי שֶׁלָּקַח אֶת שָׂרַי אֵשֶׁת אַבְרָם.
יח וַיִּקְרָ֤א פַרְעֹה֙ לְאַבְרָ֔ם וַיֹּ֕אמֶר מַה־זֹּ֖את עָשִׂ֣יתָ לִּ֑י לָ֚מָּה לֹא־הִגַּ֣דְתָּ לִּ֔י כִּ֥י אִשְׁתְּךָ֖ הִֽוא׃ פַּרְעֹה קָרָא לְאַבְרָם, וְאָמַר: מַה זֶּה עָשִׂיתָ לִי?! לָמָּה לֹא סִפַּרְתָּ לִי שֶׁהִיא אִשְׁתְּךָ?
יט לָמָ֤ה אָמַ֙רְתָּ֙ אֲחֹ֣תִי הִ֔וא וָאֶקַּ֥ח אֹתָ֛הּ לִ֖י לְאִשָּׁ֑ה וְעַתָּ֕ה הִנֵּ֥ה אִשְׁתְּךָ֖ קַ֥ח וָלֵֽךְ׃ לָמָּה אָמַרְתָּ: אֲחוֹתִי הִיא, וְלָקַחְתִּי אוֹתָהּ לִי לְאִשָּׁה? וְעַכְשָׁיו הִנֵּה אִשְׁתְּךָ – קַח וְלֵךְ!
כ וַיְצַ֥ו עָלָ֛יו פַּרְעֹ֖ה אֲנָשִׁ֑ים וַֽיְשַׁלְּח֥וּ אֹת֛וֹ וְאֶת־אִשְׁתּ֖וֹ וְאֶת־כׇּל־אֲשֶׁר־לֽוֹ׃ פַּרְעֹה צִוָּה עַל אֲנָשִׁים לְלַוּוֹתוֹ, וְהֵם שָׁלְחוּ אוֹתוֹ וְאֶת אִשְׁתּוֹ וְאֶת כָּל הַשַּׁיָּךְ לוֹ.
יג א וַיַּ֩עַל֩ אַבְרָ֨ם מִמִּצְרַ֜יִם ה֠וּא וְאִשְׁתּ֧וֹ וְכׇל־אֲשֶׁר־ל֛וֹ וְל֥וֹט עִמּ֖וֹ הַנֶּֽגְבָּה׃ אַבְרָם עָלָה מִמִּצְרַיִם, הוּא וְאִשְׁתּוֹ וְכָל הַשַּׁיָּךְ לוֹ וְלוֹט אִתּוֹ, לַנֶּגֶב.
ב וְאַבְרָ֖ם כָּבֵ֣ד מְאֹ֑ד בַּמִּקְנֶ֕ה בַּכֶּ֖סֶף וּבַזָּהָֽב׃ וְאַבְרָם עָשִׁיר מְאוֹד, בַּעַל בְּהֵמוֹת כֶּסֶף וְזָהָב.
ג וַיֵּ֙לֶךְ֙ לְמַסָּעָ֔יו מִנֶּ֖גֶב וְעַד־בֵּֽית־אֵ֑ל עַד־הַמָּק֗וֹם אֲשֶׁר־הָ֨יָה שָׁ֤ם אׇֽהֳלֹה֙ בַּתְּחִלָּ֔ה בֵּ֥ין בֵּֽית־אֵ֖ל וּבֵ֥ין הָעָֽי׃ הוּא הָלַךְ בְּמַסְּעוֹתָיו – מֵהַנֶּגֶב וְעַד בֵּית אֵל עַד הַמָּקוֹם שֶׁהָיָה שָׁם אָהֳלוֹ בַּהַתְחָלָה – בֵּין בֵּית אֵל וּבֵין הָעַי.
ד אֶל־מְקוֹם֙ הַמִּזְבֵּ֔חַ אֲשֶׁר־עָ֥שָׂה שָׁ֖ם בָּרִאשֹׁנָ֑ה וַיִּקְרָ֥א שָׁ֛ם אַבְרָ֖ם בְּשֵׁ֥ם יְהֹוָֽה׃ לִמְקוֹם הַמִּזְבֵּחַ שֶׁעָשָׂה שָׁם בַּהַתְחָלָה, וְאַבְרָם הִתְפַּלֵּל שָׁם לַה'.

שלישי[עריכה]

פָּרָשַׁת הַפְּרֵדָה מִלּוֹט
ה וְגַ֨ם־לְל֔וֹט הַהֹלֵ֖ךְ אֶת־אַבְרָ֑ם הָיָ֥ה צֹאן־וּבָקָ֖ר וְאֹהָלִֽים׃ וְגַם לְלוֹט שֶׁהוֹלֵךְ עִם אַבְרָם הָיָה צֹאן וּבָקָר וְאוֹהָלִים.
ו וְלֹא־נָשָׂ֥א אֹתָ֛ם הָאָ֖רֶץ לָשֶׁ֣בֶת יַחְדָּ֑ו כִּֽי־הָיָ֤ה רְכוּשָׁם֙ רָ֔ב וְלֹ֥א יָֽכְל֖וּ לָשֶׁ֥בֶת יַחְדָּֽו׃ וְלֹא הָיָה מַסְפִּיק מָקוֹם בָּאָרֶץ שֶׁיָּגוּרוּ יַחַד, כִּי הָיָה לָהֶם רְכוּשׁ רָב וְלֹא יָכְלוּ לָגוּר יַחַד.
ז וַֽיְהִי־רִ֗יב בֵּ֚ין רֹעֵ֣י מִקְנֵֽה־אַבְרָ֔ם וּבֵ֖ין רֹעֵ֣י מִקְנֵה־ל֑וֹט וְהַֽכְּנַעֲנִי֙ וְהַפְּרִזִּ֔י אָ֖ז יֹשֵׁ֥ב בָּאָֽרֶץ׃ וְהָיָה רִיב בֵּין רוֹעֵי הַבְּהֵמוֹת שֶׁל אַבְרָם לְבֵין רוֹעֵי הַבְּהֵמוֹת שֶׁל לוֹט, וְהַכְּנַעֲנִי וְהַפְּרִזִּי הָיוּ אָז הַתּוֹשָׁבִים בְּרוֹב הַמְּקוֹמוֹת בָּאָרֶץ.
ח וַיֹּ֨אמֶר אַבְרָ֜ם אֶל־ל֗וֹט אַל־נָ֨א תְהִ֤י מְרִיבָה֙ בֵּינִ֣י וּבֵינֶ֔ךָ וּבֵ֥ין רֹעַ֖י וּבֵ֣ין רֹעֶ֑יךָ כִּֽי־אֲנָשִׁ֥ים אַחִ֖ים אֲנָֽחְנוּ׃ אַבְרָם אָמַר לְלוֹט: בְּבַקָּשָׁה, שֶׁלֹּא תִּהְיֶה מְרִיבָה בֵּינִי לְבֵינְךָ וּבֵין רוֹעַי לְבֵין רוֹעֶיךָ, הֲרֵי אֲנָחְנוּ אַחִים!
ט הֲלֹ֤א כׇל־הָאָ֙רֶץ֙ לְפָנֶ֔יךָ הִפָּ֥רֶד נָ֖א מֵעָלָ֑י אִם־הַשְּׂמֹ֣אל וְאֵימִ֔נָה וְאִם־הַיָּמִ֖ין וְאַשְׂמְאִֽילָה׃ הֲרֵי כָּל הָאָרֶץ לְפָנֶיךָ, הִפָּרֵד בְּבַקָּשָׁה מֵעָלַי, אִם תֵּלֵךְ שְׂמֹאלָה אֵלֵךְ יְמִינָה, וְאִם תֵּלֵךְ יְמִינָה אֵלֵךְ שְׂמֹאלָה.
י וַיִּשָּׂא־ל֣וֹט אֶת־עֵינָ֗יו וַיַּרְא֙ אֶת־כׇּל־כִּכַּ֣ר הַיַּרְדֵּ֔ן כִּ֥י כֻלָּ֖הּ מַשְׁקֶ֑ה לִפְנֵ֣י ׀ שַׁחֵ֣ת יְהֹוָ֗ה אֶת־סְדֹם֙ וְאֶת־עֲמֹרָ֔ה כְּגַן־יְהֹוָה֙ כְּאֶ֣רֶץ מִצְרַ֔יִם בֹּאֲכָ֖ה צֹֽעַר׃ לוֹט הֵרִים אֶת מַבָּטוֹ, וְרָאָה אֶת כָּל בִּקְעַת הַיַּרְדֵּן שֶׁכֻּלָּהּ מֻשְׁקֵית בְּמַיִם, לִפְנֵי שֶׁה' הִשְׁחִית אֶת סְדוֹם וְאֶת עֲמוֹרָה הָיְתָה כָּל הַבִּקְעָה עַד צוֹעַר, כְּמוֹ גַּן ה' וּכְאֶרֶץ מִצְרַיִם שֶׁמֻּשְׁקֵית מֵהַיְּאוֹר.
יא וַיִּבְחַר־ל֣וֹ ל֗וֹט אֵ֚ת כׇּל־כִּכַּ֣ר הַיַּרְדֵּ֔ן וַיִּסַּ֥ע ל֖וֹט מִקֶּ֑דֶם וַיִּפָּ֣רְד֔וּ אִ֖ישׁ מֵעַ֥ל אָחִֽיו׃ לוֹט בָּחַר לוֹ אֶת כָּל בִּקְעַת הַיַּרְדֵּן, וְהוּא נָסַע מִמִּזְרַח בֵּית אֵל, הֵם נִפְרְדוּ אִישׁ מֵאָחִיו.
יב אַבְרָ֖ם יָשַׁ֣ב בְּאֶֽרֶץ־כְּנָ֑עַן וְל֗וֹט יָשַׁב֙ בְּעָרֵ֣י הַכִּכָּ֔ר וַיֶּאֱהַ֖ל עַד־סְדֹֽם׃ אַבְרָם הִתְיַשֵּׁב בְּאֶרֶץ כְּנַעַן וְלוֹט הִתְיַשֵּׁב בְּעָרֵי הַבִּקְעָה וְאוֹהֲלָיו הִגִּיעוּ עַד סְדוֹם.
יג וְאַנְשֵׁ֣י סְדֹ֔ם רָעִ֖ים וְחַטָּאִ֑ים לַיהֹוָ֖ה מְאֹֽד׃ וְאַנְשֵׁי סְדוֹם הָיוּ רָעִים וְחוֹטְאִים לַה' מְאוֹד.
יד וַֽיהֹוָ֞ה אָמַ֣ר אֶל־אַבְרָ֗ם אַחֲרֵי֙ הִפָּֽרֶד־ל֣וֹט מֵֽעִמּ֔וֹ שָׂ֣א נָ֤א עֵינֶ֙יךָ֙ וּרְאֵ֔ה מִן־הַמָּק֖וֹם אֲשֶׁר־אַתָּ֣ה שָׁ֑ם צָפֹ֥נָה וָנֶ֖גְבָּה וָקֵ֥דְמָה וָיָֽמָּה׃ וַה' אָמַר לְאַבְרָם אַחֲרֵי שֶׁלּוֹט נִפְרַד מִמֶּנּוּ: הָרֵם בְּבַקָּשָׁה אֶת מַבָּטְךָ, וּרְאֵה מֵהַמָּקוֹם שֶׁאַתָּה שָׁם, צָפוֹנָה דָּרוֹמָה מִזְרָחָה וּמַעֲרָבָה.
טו כִּ֧י אֶת־כׇּל־הָאָ֛רֶץ אֲשֶׁר־אַתָּ֥ה רֹאֶ֖ה לְךָ֣ אֶתְּנֶ֑נָּה וּֽלְזַרְעֲךָ֖ עַד־עוֹלָֽם׃ כִּי אֶת כָּל הָאָרֶץ שֶׁאַתָּה רוֹאֶה – אֶתֵּן אוֹתָהּ לְךָ וּלְצֶאֱצָאֶיךָ עַד עוֹלָם.
טז וְשַׂמְתִּ֥י אֶֽת־זַרְעֲךָ֖ כַּעֲפַ֣ר הָאָ֑רֶץ אֲשֶׁ֣ר ׀ אִם־יוּכַ֣ל אִ֗ישׁ לִמְנוֹת֙ אֶת־עֲפַ֣ר הָאָ֔רֶץ גַּֽם־זַרְעֲךָ֖ יִמָּנֶֽה׃ אַרְבֶּה אֶת צֶאֱצָאֶיךָ כַּעֲפַר הָאָרֶץ – שֶׁאִם יוּכַל אִישׁ לִסְפּוֹר אֶת עֲפַר הָאָרֶץ גַּם צֶאֱצָאֶיךָ יִסָּפְרוּ.
יז ק֚וּם הִתְהַלֵּ֣ךְ בָּאָ֔רֶץ לְאׇרְכָּ֖הּ וּלְרׇחְבָּ֑הּ כִּ֥י לְךָ֖ אֶתְּנֶֽנָּה׃ קוּם הִתְהַלֵּךְ בָּאָרֶץ לְאָרְכָּהּ וּלְרָחְבָּהּ כִּי לְךָ אֶתֵּן אוֹתָהּ.
יח וַיֶּאֱהַ֣ל אַבְרָ֗ם וַיָּבֹ֛א וַיֵּ֛שֶׁב בְּאֵלֹנֵ֥י מַמְרֵ֖א אֲשֶׁ֣ר בְּחֶבְר֑וֹן וַיִּֽבֶן־שָׁ֥ם מִזְבֵּ֖חַ לַֽיהֹוָֽה׃ אַבְרָם הֵקִים אֶת אוֹהֲלָיו וּבָא וְיָשַׁב בַּמִּישׁוֹר שֶׁל מַמְרֵא שֶׁבְּחֶבְרוֹן, וּבָנָה שָׁם מִזְבֵּחַ לַה'.

רביעי[עריכה]

פ פָּרָשַׁת מִלְחֶמֶת הַמְּלָכִים
יד א וַיְהִ֗י בִּימֵי֙ אַמְרָפֶ֣ל מֶֽלֶךְ־שִׁנְעָ֔ר אַרְי֖וֹךְ מֶ֣לֶךְ אֶלָּסָ֑ר כְּדׇרְלָעֹ֙מֶר֙ מֶ֣לֶךְ עֵילָ֔ם וְתִדְעָ֖ל מֶ֥לֶךְ גּוֹיִֽם׃ בִּימֵי אַמְרָפֶל מֶלֶךְ עִירָק אַרְיוֹךְ מֶלֶךְ אֶלָּסָר כְּדָרְלָעוֹמֶר מֶלֶךְ עֵילָם שֶׁבְּמַעֲרַב אִירָאן וְתִדְעָל מֶלֶךְ גּוֹיִם.
ב עָשׂ֣וּ מִלְחָמָ֗ה אֶת־בֶּ֙רַע֙ מֶ֣לֶךְ סְדֹ֔ם וְאֶת־בִּרְשַׁ֖ע מֶ֣לֶךְ עֲמֹרָ֑ה שִׁנְאָ֣ב ׀ מֶ֣לֶךְ אַדְמָ֗ה וְשֶׁמְאֵ֙בֶר֙ מֶ֣לֶךְ צביים צְבוֹיִ֔ם וּמֶ֥לֶךְ בֶּ֖לַע הִיא־צֹֽעַר׃ הֵם עָשׂוּ מִלְחָמָה נֶגֶד בֶּרַע מֶלֶךְ סְדוֹם וְנֶגֶד בִּרְשַׁע מֶלֶךְ עֲמוֹרָה שִׁנְאָב מֶלֶךְ אַדְמָה וְשֶׁמְאֵבֶר מֶלֶךְ צְבוֹיִים וּמֶלֶךְ בֶּלַע שֶׁהִיא צוֹעַר.
ג כׇּל־אֵ֙לֶּה֙ חָֽבְר֔וּ אֶל־עֵ֖מֶק הַשִּׂדִּ֑ים ה֖וּא יָ֥ם הַמֶּֽלַח׃ כָּל אֵלֶּה חָבְרוּ יַחַד וְהָלְכוּ אֶל עֵמֶק הַשָּׂדוֹת שֶׁהוּא הַיּוֹם יָם הַמֶּלַח.
ד שְׁתֵּ֤ים עֶשְׂרֵה֙ שָׁנָ֔ה עָבְד֖וּ אֶת־כְּדׇרְלָעֹ֑מֶר וּשְׁלֹשׁ־עֶשְׂרֵ֥ה שָׁנָ֖ה מָרָֽדוּ׃ שְׁתֵּים עֶשְׂרֵה שָׁנָה עָבְדוּ אֶת כְּדָרְלָעוֹמֶר, וּשְׁלוֹשׁ עֶשְׂרֵה שָׁנָה מָרְדוּ בּוֹ.
ה וּבְאַרְבַּע֩ עֶשְׂרֵ֨ה שָׁנָ֜ה בָּ֣א כְדׇרְלָעֹ֗מֶר וְהַמְּלָכִים֙ אֲשֶׁ֣ר אִתּ֔וֹ וַיַּכּ֤וּ אֶת־רְפָאִים֙ בְּעַשְׁתְּרֹ֣ת קַרְנַ֔יִם וְאֶת־הַזּוּזִ֖ים בְּהָ֑ם וְאֵת֙ הָֽאֵימִ֔ים בְּשָׁוֵ֖ה קִרְיָתָֽיִם׃ וּבַשָּׁנָה הָאַרְבַּע עֶשְׂרֵה בָּא כְּדָרְלָעוֹמֶר וְהַמְּלָכִים שֶׁאִתּוֹ וְהָרְגוּ אֶת הָעָם רְפָאִים שֶׁבְּעַשְׁתְּרוֹת קַרְנַיִם, וְאֶת עַם הַזּוּזִים שֶׁבְּהָם, וְאֶת עַם הָאֶמְתָּנִיִּים שֶׁבְּשָׁוֵה קִרְיָתָיִם.
ו וְאֶת־הַחֹרִ֖י בְּהַרְרָ֣ם שֵׂעִ֑יר עַ֚ד אֵ֣יל פָּארָ֔ן אֲשֶׁ֖ר עַל־הַמִּדְבָּֽר׃ וְאֶת הַחוֹרִי בָּהָר שֶׁלָּהֶם – שֵׂעִיר, וְעַד מִישׁוֹר פָּארָן שֶׁעַל יַד הַמִּדְבָּר.
ז וַ֠יָּשֻׁ֠בוּ וַיָּבֹ֜אוּ אֶל־עֵ֤ין מִשְׁפָּט֙ הִ֣וא קָדֵ֔שׁ וַיַּכּ֕וּ אֶֽת־כׇּל־שְׂדֵ֖ה הָעֲמָלֵקִ֑י וְגַם֙ אֶת־הָ֣אֱמֹרִ֔י הַיֹּשֵׁ֖ב בְּחַֽצְצֹ֥ן תָּמָֽר׃ הֵם חָזְרוּ וּבָאוּ לְעֵין מִשְׁפָּט הִיא קָדֵשׁ, וְהָרְגוּ אֶת כָּל תּוֹשְׁבֵי שְׂדֵה הָעֲמָלֵקִי וְגַם אֶת הָאֱמוֹרִי שֶׁיּוֹשֵׁב בְּחַצְצוֹן תָּמָר.
ח וַיֵּצֵ֨א מֶֽלֶךְ־סְדֹ֜ם וּמֶ֣לֶךְ עֲמֹרָ֗ה וּמֶ֤לֶךְ אַדְמָה֙ וּמֶ֣לֶךְ צביים צְבוֹיִ֔ם וּמֶ֥לֶךְ בֶּ֖לַע הִוא־צֹ֑עַר וַיַּֽעַרְכ֤וּ אִתָּם֙ מִלְחָמָ֔ה בְּעֵ֖מֶק הַשִּׂדִּֽים׃ מֶלֶךְ סְדוֹם וּמֶלֶךְ עֲמוֹרָה וּמֶלֶךְ אַדְמָה וּמֶלֶךְ צְבוֹיִם וּמֶלֶךְ בֶּלַע הִיא צוֹעַר יָצְאוּ וְעָרְכוּ מוּלָם מִלְחָמָה בְּעֵמֶק הַשָּׂדוֹת.
ט אֵ֣ת כְּדׇרְלָעֹ֜מֶר מֶ֣לֶךְ עֵילָ֗ם וְתִדְעָל֙ מֶ֣לֶךְ גּוֹיִ֔ם וְאַמְרָפֶל֙ מֶ֣לֶךְ שִׁנְעָ֔ר וְאַרְי֖וֹךְ מֶ֣לֶךְ אֶלָּסָ֑ר אַרְבָּעָ֥ה מְלָכִ֖ים אֶת־הַחֲמִשָּֽׁה׃ מוּל כְּדָרְלָעוֹמֶר מֶלֶךְ עֵילָם שֶׁבְּמַעֲרַב אִירָאן וְתִדְעָל מֶלֶךְ גּוֹיִם וְאַמְרָפֶל מֶלֶךְ עִירָק וְאַרְיוֹךְ מֶלֶךְ אֶלָּסָר – אַרְבָּעָה מְלָכִים נֶגֶד הַחֲמִשָּׁה.
י וְעֵ֣מֶק הַשִּׂדִּ֗ים בֶּֽאֱרֹ֤ת בֶּאֱרֹת֙ חֵמָ֔ר וַיָּנֻ֛סוּ מֶֽלֶךְ־סְדֹ֥ם וַעֲמֹרָ֖ה וַיִּפְּלוּ־שָׁ֑מָּה וְהַנִּשְׁאָרִ֖ים הֶ֥רָה נָּֽסוּ׃ וְעֵמֶק הַשָּׂדוֹת הָיָה מָלֵא בִּבְאֵרוֹת מְלֵאוֹת בּוֹץ, מֶלֶךְ סְדוֹם וַעֲמוֹרָה בָּרְחוּ וְנָפְלוּ שָׁמָּה, וְהַנִּשְׁאָרִים בָּרְחוּ לָהָר.
יא וַ֠יִּקְח֠וּ אֶת־כׇּל־רְכֻ֨שׁ סְדֹ֧ם וַעֲמֹרָ֛ה וְאֶת־כׇּל־אׇכְלָ֖ם וַיֵּלֵֽכוּ׃ הֵם לָקְחוּ אֶת כָּל הָרְכוּשׁ שֶׁל סְדוֹם וַעֲמוֹרָה וְאֶת כָּל הָאוֹכֶל שֶׁלָּהֶם וְהָלְכוּ.
יב וַיִּקְח֨וּ אֶת־ל֧וֹט וְאֶת־רְכֻשׁ֛וֹ בֶּן־אֲחִ֥י אַבְרָ֖ם וַיֵּלֵ֑כוּ וְה֥וּא יֹשֵׁ֖ב בִּסְדֹֽם׃ הֵם לָקְחוּ אֶת לוֹט בֶּן אֲחִי אַבְרָם וְאֶת רְכוּשׁוֹ וְהָלְכוּ, וְהוּא הָיָה גָּר בִּסְדוֹם.
יג וַיָּבֹא֙ הַפָּלִ֔יט וַיַּגֵּ֖ד לְאַבְרָ֣ם הָעִבְרִ֑י וְהוּא֩ שֹׁכֵ֨ן בְּאֵֽלֹנֵ֜י מַמְרֵ֣א הָאֱמֹרִ֗י אֲחִ֤י אֶשְׁכֹּל֙ וַאֲחִ֣י עָנֵ֔ר וְהֵ֖ם בַּעֲלֵ֥י בְרִית־אַבְרָֽם׃ הַפָּלִיט בָּא וְהוֹדִיעַ לְאַבְרָם הָעִבְרִי שֶׁהָיָה גָּר בַּמִּישׁוֹר שֶׁל מַמְרֵא הָאֱמוֹרִי אֲחִיהֶם שֶׁל אֶשְׁכּוֹל וְשֶׁל עָנֵר, שְׁלָשְׁתָּם הָיוּ בַּעֲלֵי בְּרִית שֶׁל אַבְרָם.
יד וַיִּשְׁמַ֣ע אַבְרָ֔ם כִּ֥י נִשְׁבָּ֖ה אָחִ֑יו וַיָּ֨רֶק אֶת־חֲנִיכָ֜יו יְלִידֵ֣י בֵית֗וֹ שְׁמֹנָ֤ה עָשָׂר֙ וּשְׁלֹ֣שׁ מֵא֔וֹת וַיִּרְדֹּ֖ף עַד־דָּֽן׃ אַבְרָם שָׁמַע שֶׁקְּרוֹבוֹ נָפַל בַּשֶּׁבִי, הוּא גִּיֵּס אֶת שְׁלֹשׁ מֵאוֹת וּשְׁמוֹנָה עָשָׂר בְּנֵי עֲבָדָיו שֶׁנּוֹלְדוּ וְגָדְלוּ בְּבֵיתוֹ, וְרָדַף עַד לַמָּקוֹם שֶׁנִּקְרָא דָּן.
טו וַיֵּחָלֵ֨ק עֲלֵיהֶ֧ם ׀ לַ֛יְלָה ה֥וּא וַעֲבָדָ֖יו וַיַּכֵּ֑ם וַֽיִּרְדְּפֵם֙ עַד־חוֹבָ֔ה אֲשֶׁ֥ר מִשְּׂמֹ֖אל לְדַמָּֽשֶׂק׃ בַּלַּיְלָה הוּא חִלֵּק אֵת צְבָאוֹ – הוּא וַעֲבָדָיו, לִשְׁתֵּי זְרוֹעוֹת, וְהִכָּה אוֹתָם, וְרָדַף אַחֲרֵיהֶם עַד לְחוֹבָה שֶׁמִּצְּפוֹן לְדַמֶּשֶׂק.
טז וַיָּ֕שֶׁב אֵ֖ת כׇּל־הָרְכֻ֑שׁ וְגַם֩ אֶת־ל֨וֹט אָחִ֤יו וּרְכֻשׁוֹ֙ הֵשִׁ֔יב וְגַ֥ם אֶת־הַנָּשִׁ֖ים וְאֶת־הָעָֽם׃ הוּא הֶחֱזִיר אֶת כָּל הָרְכוּשׁ, וְגַם הֶחֱזִיר אֶת לוֹט קְרוֹבוֹ וּרְכוּשׁוֹ, וְגַם אֶת הַנָּשִׁים וְאֶת הָעָם.
יז וַיֵּצֵ֣א מֶֽלֶךְ־סְדֹם֮ לִקְרָאתוֹ֒ אַחֲרֵ֣י שׁוּב֗וֹ מֵֽהַכּוֹת֙ אֶת־כְּדׇרְלָעֹ֔מֶר וְאֶת־הַמְּלָכִ֖ים אֲשֶׁ֣ר אִתּ֑וֹ אֶל־עֵ֣מֶק שָׁוֵ֔ה ה֖וּא עֵ֥מֶק הַמֶּֽלֶךְ׃ אַחֲרֵי שֶׁשָּׁב מִלְּהַכּוֹת אֶת כְּדָרְלָעוֹמֶר וְאֶת הַמְּלָכִים שֶׁאִתּוֹ, יָצָא לִקְרָאתוֹ מֶלֶךְ סְדוֹם אֶל הָעֵמֶק הַנִּקְרָא שָׁוֵה – הוּא עֵמֶק הַמֶּלֶךְ.
יח וּמַלְכִּי־צֶ֙דֶק֙ מֶ֣לֶךְ שָׁלֵ֔ם הוֹצִ֖יא לֶ֣חֶם וָיָ֑יִן וְה֥וּא כֹהֵ֖ן לְאֵ֥ל עֶלְיֽוֹן׃ וּמַלְכִּי צֶדֶק מֶלֶךְ שָׁלֵם הוֹצִיא לִקְרַאת אַבְרָם לֶחֶם וְיָיִן, וְהוּא הָיָה כֹּהֵן לַעֲבוֹדַת הָאֵל הָעֶלְיוֹן.
יט וַֽיְבָרְכֵ֖הוּ וַיֹּאמַ֑ר בָּר֤וּךְ אַבְרָם֙ לְאֵ֣ל עֶלְי֔וֹן קֹנֵ֖ה שָׁמַ֥יִם וָאָֽרֶץ׃ הוּא בֵּרַךְ אוֹתוֹ וְאָמַר: בָּרוּךְ אַבְרָם בְּבִרְכַּת אֵל עֶלְיוֹן אֲדוֹן הַשָּׁמַיִם וְהָאָרֶץ.
כ וּבָרוּךְ֙ אֵ֣ל עֶלְי֔וֹן אֲשֶׁר־מִגֵּ֥ן צָרֶ֖יךָ בְּיָדֶ֑ךָ וַיִּתֶּן־ל֥וֹ מַעֲשֵׂ֖ר מִכֹּֽל׃ וּבָרוּךְ אֵל עֶלְיוֹן שֶׁמָּסַר בְּיָדְךָ אֶת צוֹרְרֶיךָ, אַבְרָם נָתַן לוֹ מַעֲשֵׂר מִכָּל רְכוּשׁוֹ.

חמישי[עריכה]

כא וַיֹּ֥אמֶר מֶֽלֶךְ־סְדֹ֖ם אֶל־אַבְרָ֑ם תֶּן־לִ֣י הַנֶּ֔פֶשׁ וְהָרְכֻ֖שׁ קַֽח־לָֽךְ׃ מֶלֶךְ סְדוֹם אָמַר לְאַבְרָם: תֵּן לִי אֶת הָאֲנָשִׁים, וְקַח לְךָ אֶת הָרְכוּשׁ.
כב וַיֹּ֥אמֶר אַבְרָ֖ם אֶל־מֶ֣לֶךְ סְדֹ֑ם הֲרִמֹ֨תִי יָדִ֤י אֶל־יְהֹוָה֙ אֵ֣ל עֶלְי֔וֹן קֹנֵ֖ה שָׁמַ֥יִם וָאָֽרֶץ׃ אָמַר אַבְרָם לְמֶלֶךְ סְדוֹם: אֲנִי מֵרִים אֶת יָדִי בִּשְׁבוּעָה אֶל ה' הָאֵל הָעֶלְיוֹן אֲדוֹן הַשָּׁמַיִם וְהָאָרֶץ.
כג אִם־מִחוּט֙ וְעַ֣ד שְׂרֽוֹךְ־נַ֔עַל וְאִם־אֶקַּ֖ח מִכׇּל־אֲשֶׁר־לָ֑ךְ וְלֹ֣א תֹאמַ֔ר אֲנִ֖י הֶעֱשַׁ֥רְתִּי אֶת־אַבְרָֽם׃ שֶׁלֹּא אֶקַּח אֲפִלּוּ מִחוּט וְעַד שְׂרוֹךְ נַעַל, וְלֹא אֶקַּח שׁוּם דָּבָר מִכָּל הַשַּׁיָּךְ לְךָ, כְּדֵי שֶׁלֹּא תֹּאמַר: אֲנִי הֶעֱשַׁרְתִּי אֶת אַבְרָם.
כד בִּלְעָדַ֗י רַ֚ק אֲשֶׁ֣ר אָֽכְל֣וּ הַנְּעָרִ֔ים וְחֵ֙לֶק֙ הָֽאֲנָשִׁ֔ים אֲשֶׁ֥ר הָלְכ֖וּ אִתִּ֑י עָנֵר֙ אֶשְׁכֹּ֣ל וּמַמְרֵ֔א הֵ֖ם יִקְח֥וּ חֶלְקָֽם׃ חוּץ מִמַּה שֶׁאָכְלוּ הַנְּעָרִים כְּשֶׁנִּלְחֲמוּ, וְחוּץ מֵחֵלֶק הַשָּׁלָל שֶׁל הָאֲנָשִׁים שֶׁהָלְכוּ אִתִּי – עָנֵר אֶשְׁכֹּל וּמַמְרֵא, הֵם יִקְחוּ אֶת חֶלְקָם.
ס פָּרָשַׁת בְּרִית בֵּין הַבְּתָרִים
טו א אַחַ֣ר ׀ הַדְּבָרִ֣ים הָאֵ֗לֶּה הָיָ֤ה דְבַר־יְהֹוָה֙ אֶל־אַבְרָ֔ם בַּֽמַּחֲזֶ֖ה לֵאמֹ֑ר אַל־תִּירָ֣א אַבְרָ֗ם אָנֹכִי֙ מָגֵ֣ן לָ֔ךְ שְׂכָרְךָ֖ הַרְבֵּ֥ה מְאֹֽד׃ אַחֲרֵי הַדְּבָרִים הָאֵלֶּה, נֶאֱמַר דְּבַר ה' לְאַבְרָם בְּמַרְאֵה נְבוּאָה: אַל תִּפְחַד אַבְרָם, אֲנִי מָגֵן עָלֶיךָ, וְיִהְיֶה לְךָ שָׂכָר רָב.
ב וַיֹּ֣אמֶר אַבְרָ֗ם אֲדֹנָ֤י יֱהֹוִה֙ מַה־תִּתֶּן־לִ֔י וְאָנֹכִ֖י הוֹלֵ֣ךְ עֲרִירִ֑י וּבֶן־מֶ֣שֶׁק בֵּיתִ֔י ה֖וּא דַּמֶּ֥שֶׂק אֱלִיעֶֽזֶר׃ אַבְרָם אָמַר: ה' אֱלֹהִים מַה תִּתֶּן לִי?! הֲרֵי אֲנִי הוֹלֵךְ לָמוּת בְּלִי בֵּן שֶׁיִּירַשׁ אוֹתִי, וְהָאַחְרַאי עַל מֶשֶׁק בֵּיתִי הוּא אֱלִיעֶזֶר מִדַּמֶּשֶׂק.
ג וַיֹּ֣אמֶר אַבְרָ֔ם הֵ֣ן לִ֔י לֹ֥א נָתַ֖תָּה זָ֑רַע וְהִנֵּ֥ה בֶן־בֵּיתִ֖י יוֹרֵ֥שׁ אֹתִֽי׃ אַבְרָם אָמַר: הֲרֵי לֹא נָתַתָּ לִי בֵּן, וְהִנֵּה הָאַחְרַאי עַל בֵּיתִי יוֹרֵשׁ אוֹתִי.
ד וְהִנֵּ֨ה דְבַר־יְהֹוָ֤ה אֵלָיו֙ לֵאמֹ֔ר לֹ֥א יִֽירָשְׁךָ֖ זֶ֑ה כִּי־אִם֙ אֲשֶׁ֣ר יֵצֵ֣א מִמֵּעֶ֔יךָ ה֖וּא יִֽירָשֶֽׁךָ׃ וְהִנֵּה נֶאֱמַר לוֹ דְּבַר ה': הָאִישׁ הַזֶּה לֹא יִירַשׁ אוֹתְךָ, אֶלָּא בִּנְךָ שֶׁיֵּצֵא מִמְּךָ הוּא יִירַשׁ אוֹתְךָ.
ה וַיּוֹצֵ֨א אֹת֜וֹ הַח֗וּצָה וַיֹּ֙אמֶר֙ הַבֶּט־נָ֣א הַשָּׁמַ֗יְמָה וּסְפֹר֙ הַכּ֣וֹכָבִ֔ים אִם־תּוּכַ֖ל לִסְפֹּ֣ר אֹתָ֑ם וַיֹּ֣אמֶר ל֔וֹ כֹּ֥ה יִהְיֶ֖ה זַרְעֶֽךָ׃ הוּא הוֹצִיא אוֹתוֹ הַחוּצָה וְאָמַר: הַבֵּט בְּבַקָּשָׁה לַשָּׁמַיִם וּסְפוֹר אֶת הַכּוֹכָבִים – אִם תּוּכַל לִסְפּוֹר אוֹתָם! אָמַר לוֹ: כָּךְ יִהְיוּ צֶאֱצָאֶיךָ!
ו וְהֶאֱמִ֖ן בַּֽיהֹוָ֑ה וַיַּחְשְׁבֶ֥הָ לּ֖וֹ צְדָקָֽה׃ אַבְרָם הֶאֱמִין בַּה' שֶׁיִּתֵּן לוֹ בֵּן, וַה' הֶחְשִׁיב לוֹ אֶת הַדָּבָר לִזְכוּת.

ששי[עריכה]

ז וַיֹּ֖אמֶר אֵלָ֑יו אֲנִ֣י יְהֹוָ֗ה אֲשֶׁ֤ר הוֹצֵאתִ֙יךָ֙ מֵא֣וּר כַּשְׂדִּ֔ים לָ֧תֶת לְךָ֛ אֶת־הָאָ֥רֶץ הַזֹּ֖את לְרִשְׁתָּֽהּ׃ אָמַר לוֹ ה': אֲנִי ה' שֶׁהוֹצֵאתִי אוֹתְךָ מֵאוּר כַּשְׂדִּים לָתֵת לְךָ אֶת הָאָרֶץ הַזֹּאת לִכְבּוֹשׁ אוֹתָהּ.
ח וַיֹּאמַ֑ר אֲדֹנָ֣י יֱהֹוִ֔ה בַּמָּ֥ה אֵדַ֖ע כִּ֥י אִֽירָשֶֽׁנָּה׃ אָמַר אַבְרָם: ה' אֱלֹהִים, אֵיךְ אֵדַע וְאֶשְׁתַּכְנֵעַ שֶׁאֶכְבּוֹשׁ אוֹתָהּ?
ט וַיֹּ֣אמֶר אֵלָ֗יו קְחָ֥ה לִי֙ עֶגְלָ֣ה מְשֻׁלֶּ֔שֶׁת וְעֵ֥ז מְשֻׁלֶּ֖שֶׁת וְאַ֣יִל מְשֻׁלָּ֑שׁ וְתֹ֖ר וְגוֹזָֽל׃ אָמַר לוֹ: הָבֵא לִי עֶגְלָה שְׁמֵנָה וּבְרִיאָה וְעֵז שְׁמֵנָה וְאַיִל שָׁמֵן וְתוֹר וְגוֹזָל בֶּן יוֹנָה.
י וַיִּֽקַּֽח־ל֣וֹ אֶת־כׇּל־אֵ֗לֶּה וַיְבַתֵּ֤ר אֹתָם֙ בַּתָּ֔וֶךְ וַיִּתֵּ֥ן אִישׁ־בִּתְר֖וֹ לִקְרַ֣את רֵעֵ֑הוּ וְאֶת־הַצִּפֹּ֖ר לֹ֥א בָתָֽר׃ הוּא הֵבִיא לוֹ אֶת כָּל אֵלֶּה וְהוּא חָתַךְ אוֹתָם בְּאֶמְצָעָם וְהֵנִיחַ כָּל חֵצִי מוּל הַחֵצִי הַשֵּׁנִי, וְאֶת הַצִּפּוֹרִים הוּא לֹא חָתַךְ.
יא וַיֵּ֥רֶד הָעַ֖יִט עַל־הַפְּגָרִ֑ים וַיַּשֵּׁ֥ב אֹתָ֖ם אַבְרָֽם׃ עוֹפוֹת עַיִט יָרְדוּ עַל הַפְּגָרִים, וְאַבְרָם הֵעִיף אוֹתָם.
יב וַיְהִ֤י הַשֶּׁ֙מֶשׁ֙ לָב֔וֹא וְתַרְדֵּמָ֖ה נָפְלָ֣ה עַל־אַבְרָ֑ם וְהִנֵּ֥ה אֵימָ֛ה חֲשֵׁכָ֥ה גְדֹלָ֖ה נֹפֶ֥לֶת עָלָֽיו׃ הַשֶּׁמֶשׁ עָמְדָה לִשְׁקוֹעַ וְתַרְדֵּמָה נָפְלָה עַל אַבְרָם, וְהִנֵּה פַּחַד וְחוֹשֶׁךְ גָּדוֹל נוֹפֵל עָלָיו.
יג וַיֹּ֣אמֶר לְאַבְרָ֗ם יָדֹ֨עַ תֵּדַ֜ע כִּי־גֵ֣ר ׀ יִהְיֶ֣ה זַרְעֲךָ֗ בְּאֶ֙רֶץ֙ לֹ֣א לָהֶ֔ם וַעֲבָד֖וּם וְעִנּ֣וּ אֹתָ֑ם אַרְבַּ֥ע מֵא֖וֹת שָׁנָֽה׃ ה' אָמַר לְאַבְרָם: דַּע שֶׁצֶּאֱצָאֶיךָ יִהְיוּ גֵּרִים בְּאֶרֶץ שֶׁאֵינָהּ שֶׁלָּהֶם וְיִהְיוּ לָהֶם לַעֲבָדִים וִיעַנּוּ אוֹתָם – אַרְבַּע מֵאוֹת שָׁנָה.
יד וְגַ֧ם אֶת־הַגּ֛וֹי אֲשֶׁ֥ר יַעֲבֹ֖דוּ דָּ֣ן אָנֹ֑כִי וְאַחֲרֵי־כֵ֥ן יֵצְא֖וּ בִּרְכֻ֥שׁ גָּדֽוֹל׃ וְגַם אֶשְׁפּוֹט אֶת הָעָם שֶׁהֵם יִהְיוּ לוֹ לַעֲבָדִים, וְאַחֲרֵי כֵן הֵם יֵצְאוּ בִּרְכוּשׁ גָּדוֹל.
טו וְאַתָּ֛ה תָּב֥וֹא אֶל־אֲבֹתֶ֖יךָ בְּשָׁל֑וֹם תִּקָּבֵ֖ר בְּשֵׂיבָ֥ה טוֹבָֽה׃ וְאַתָּה תָּבוֹא בְּמוֹתְךָ בְּשָׁלוֹם אֶל אֲבוֹתֶיךָ שֶׁמֵּתוּ – תִּקָּבֵר אַחֲרֵי זִקְנָה נְעִימָה.
טז וְד֥וֹר רְבִיעִ֖י יָשׁ֣וּבוּ הֵ֑נָּה כִּ֧י לֹא־שָׁלֵ֛ם עֲוֺ֥ן הָאֱמֹרִ֖י עַד־הֵֽנָּה׃ וְרַק הַדּוֹר הָרְבִיעִי לַגָּלוּת יָשׁוּבוּ הֵנָּה, כִּי עֲדַיִן לֹא נִשְׁלְמוּ עֲווֹנוֹת הָאֱמוֹרִי שֶׁאַעֲנִישׁ אוֹתָם וְאֶקַּח אֶת אַרְצָם.
יז וַיְהִ֤י הַשֶּׁ֙מֶשׁ֙ בָּ֔אָה וַעֲלָטָ֖ה הָיָ֑ה וְהִנֵּ֨ה תַנּ֤וּר עָשָׁן֙ וְלַפִּ֣יד אֵ֔שׁ אֲשֶׁ֣ר עָבַ֔ר בֵּ֖ין הַגְּזָרִ֥ים הָאֵֽלֶּה׃ הַשֶּׁמֶשׁ שָׁקְעָה וְנִהְיָה חוֹשֶׁךְ, וְהִנֵּה תַּנּוּר מַעֲלֶה עָשָׁן וְלַפִּיד אֵשׁ עָבְרוּ בֵּין חֶלְקֵי הַבְּהֵמוֹת הָאֵלֶּה.
יח בַּיּ֣וֹם הַה֗וּא כָּרַ֧ת יְהֹוָ֛ה אֶת־אַבְרָ֖ם בְּרִ֣ית לֵאמֹ֑ר לְזַרְעֲךָ֗ נָתַ֙תִּי֙ אֶת־הָאָ֣רֶץ הַזֹּ֔את מִנְּהַ֣ר מִצְרַ֔יִם עַד־הַנָּהָ֥ר הַגָּדֹ֖ל נְהַר־פְּרָֽת׃ בַּיּוֹם הַהוּא ה' עָשָׂה עִם אַבְרָם הֶסְכֵּם, וְהִתְחַיֵּב: לְצֶאֱצָאֶיךָ נָתַתִּי אֶת הָאָרֶץ הַזֹּאת מִנְּהַר מִצְרַיִם עַד הַנָּהָר הַגָּדוֹל נְהַר פְּרָת.
יט אֶת־הַקֵּינִי֙ וְאֶת־הַקְּנִזִּ֔י וְאֵ֖ת הַקַּדְמֹנִֽי׃ אֶת אֶרֶץ הַקֵּינִי וְאֶת הַקְּנִזִּי וְאֶת הַקַּדְמוֹנִי,
כ וְאֶת־הַחִתִּ֥י וְאֶת־הַפְּרִזִּ֖י וְאֶת־הָרְפָאִֽים׃ וְאֶת הַחִתִּי וְאֶת הַפְּרִזִּי וְאֶת הָרְפָאִים,
כא וְאֶת־הָֽאֱמֹרִי֙ וְאֶת־הַֽכְּנַעֲנִ֔י וְאֶת־הַגִּרְגָּשִׁ֖י וְאֶת־הַיְבוּסִֽי׃ וְאֶת הָאֱמוֹרִי וְאֶת הַכְּנַעֲנִי וְאֶת הַגִּרְגָּשִׁי וְאֶת הַיְּבוּסִי.
ס פָּרָשַׁת הָגָר
טז א וְשָׂרַי֙ אֵ֣שֶׁת אַבְרָ֔ם לֹ֥א יָלְדָ֖ה ל֑וֹ וְלָ֛הּ שִׁפְחָ֥ה מִצְרִ֖ית וּשְׁמָ֥הּ הָגָֽר׃ וְשָׂרַי אֵשֶׁת אַבְרָם לֹא יָלְדָה לוֹ, וְהָיְתָה לָהּ שִׁפְחָה מִצְרִית וּשְׁמָהּ הָגָר.
ב וַתֹּ֨אמֶר שָׂרַ֜י אֶל־אַבְרָ֗ם הִנֵּה־נָ֞א עֲצָרַ֤נִי יְהֹוָה֙ מִלֶּ֔דֶת בֹּא־נָא֙ אֶל־שִׁפְחָתִ֔י אוּלַ֥י אִבָּנֶ֖ה מִמֶּ֑נָּה וַיִּשְׁמַ֥ע אַבְרָ֖ם לְק֥וֹל שָׂרָֽי׃ שָׂרַי אָמְרָה לְאַבְרָם: בְּבַקָּשָׁה, הִנֵּה ה' מָנַע מִמֶּנִּי לָלֶדֶת, בֹּא בְּבַקָּשָׁה אֶל שִׁפְחָתִי, אוּלַי יִהְיֶה לִי בֵּן מִמֶּנָּה! אַבְרָם שָׁמַע לְקוֹל שָׂרַי.
ג וַתִּקַּ֞ח שָׂרַ֣י אֵֽשֶׁת־אַבְרָ֗ם אֶת־הָגָ֤ר הַמִּצְרִית֙ שִׁפְחָתָ֔הּ מִקֵּץ֙ עֶ֣שֶׂר שָׁנִ֔ים לְשֶׁ֥בֶת אַבְרָ֖ם בְּאֶ֣רֶץ כְּנָ֑עַן וַתִּתֵּ֥ן אֹתָ֛הּ לְאַבְרָ֥ם אִישָׁ֖הּ ל֥וֹ לְאִשָּֽׁה׃ שָׂרַי אֵשֶׁת אַבְרָם לָקְחָה אֶת הָגָר הַמִּצְרִית שִׁפְחָתָהּ אַחֲרֵי עֶשֶׂר שָׁנִים לְהִתְיַשְּׁבוּת אַבְרָם בְּאֶרֶץ כְּנַעַן, וְנָתְנָה אוֹתָהּ לְאַבְרָם בַּעֲלָהּ שֶׁתִּהְיֶה לוֹ לְאִשָּׁה.
ד וַיָּבֹ֥א אֶל־הָגָ֖ר וַתַּ֑הַר וַתֵּ֙רֶא֙ כִּ֣י הָרָ֔תָה וַתֵּקַ֥ל גְּבִרְתָּ֖הּ בְּעֵינֶֽיהָ׃ הוּא בָּא אֶל הָגָר וְהִיא נִכְנְסָה לְהֵרָיוֹן, כֵּיוָן שֶׁרָאֲתָה שֶׁהִיא נִכְנְסָה לְהֵרָיוֹן גְּבִרְתָּהּ הִתְבַּזְּתָה בְּעֵינֶיהָ.
ה וַתֹּ֨אמֶר שָׂרַ֣י אֶל־אַבְרָם֮ חֲמָסִ֣י עָלֶ֒יךָ֒ אָנֹכִ֗י נָתַ֤תִּי שִׁפְחָתִי֙ בְּחֵיקֶ֔ךָ וַתֵּ֙רֶא֙ כִּ֣י הָרָ֔תָה וָאֵקַ֖ל בְּעֵינֶ֑יהָ יִשְׁפֹּ֥ט יְהֹוָ֖ה בֵּינִ֥י וּבֵינֶֽיׄךָ׃ אָמְרָה שָׂרַי לְאַבְרָם: הַפְּגִיעָה בִּכְבוֹדִי הִיא בְּאַשְׁמָתְךָ! אֲנִי נָתַתִּי אֶת שִׁפְחָתִי בְּחֵיקְךָ, הִיא רָאֲתָה שֶׁהִיא נִכְנְסָה לְהֵרָיוֹן וְהִתְבַּזֵּיתִי בְּעֵינֶיהָ, ה' יִשְׁפּוֹט בֵּינִי וּבֵינְךָ.
ו וַיֹּ֨אמֶר אַבְרָ֜ם אֶל־שָׂרַ֗י הִנֵּ֤ה שִׁפְחָתֵךְ֙ בְּיָדֵ֔ךְ עֲשִׂי־לָ֖הּ הַטּ֣וֹב בְּעֵינָ֑יִךְ וַתְּעַנֶּ֣הָ שָׂרַ֔י וַתִּבְרַ֖ח מִפָּנֶֽיהָ׃ אָמַר אַבְרָם לְשָׂרַי: הִנֵּה שִׁפְחָתֵךְ בְּיָדֵךְ עֲשִׂי לָהּ כַּטּוֹב בְּעֵינַיִךְ, שָׂרַי הִשְׁפִּילָה אוֹתָהּ וְהִיא בָּרְחָה מִמֶּנָּה.
ז וַֽיִּמְצָאָ֞הּ מַלְאַ֧ךְ יְהֹוָ֛ה עַל־עֵ֥ין הַמַּ֖יִם בַּמִּדְבָּ֑ר עַל־הָעַ֖יִן בְּדֶ֥רֶךְ שֽׁוּר׃ מַלְאַךְ ה' מָצָא אוֹתָהּ עַל יַד מַעְיַן הַמַּיִם בַּמִּדְבָּר, עַל יַד הַמַּעְיַן בְּחוֹלוֹת חֲלוּצָה בַּדֶּרֶךְ לְמִדְבַּר שׁוּר.
ח וַיֹּאמַ֗ר הָגָ֞ר שִׁפְחַ֥ת שָׂרַ֛י אֵֽי־מִזֶּ֥ה בָ֖את וְאָ֣נָה תֵלֵ֑כִי וַתֹּ֕אמֶר מִפְּנֵי֙ שָׂרַ֣י גְּבִרְתִּ֔י אָנֹכִ֖י בֹּרַֽחַת׃ אָמַר לָהּ: הָגָר שִׁפְחַת שָׂרַי, מֵאֵיפֹה בָּאת וּלְאָן אַתְּ הוֹלֶכֶת? הִיא אָמְרָה: אֲנִי בּוֹרַחַת מִשָּׂרַי גְּבִרְתִּי.
ט וַיֹּ֤אמֶר לָהּ֙ מַלְאַ֣ךְ יְהֹוָ֔ה שׁ֖וּבִי אֶל־גְּבִרְתֵּ֑ךְ וְהִתְעַנִּ֖י תַּ֥חַת יָדֶֽיהָ׃ אָמַר לָהּ מַלְאַךְ ה': חִזְרִי אֶל גְּבִרְתֵּךְ וֶהֱיִי כְּפוּפָה תַּחְתֶּיהָ.
י וַיֹּ֤אמֶר לָהּ֙ מַלְאַ֣ךְ יְהֹוָ֔ה הַרְבָּ֥ה אַרְבֶּ֖ה אֶת־זַרְעֵ֑ךְ וְלֹ֥א יִסָּפֵ֖ר מֵרֹֽב׃ אָמַר לָהּ מַלְאַךְ ה': אַרְבֶּה מְאוֹד אֶת צֶאֱצָאַיִךְ וְלֹא יִסָּפְרוּ מֵרוֹב.
יא וַיֹּ֤אמֶר לָהּ֙ מַלְאַ֣ךְ יְהֹוָ֔ה הִנָּ֥ךְ הָרָ֖ה וְיֹלַ֣דְתְּ בֵּ֑ן וְקָרָ֤את שְׁמוֹ֙ יִשְׁמָעֵ֔אל כִּֽי־שָׁמַ֥ע יְהֹוָ֖ה אֶל־עׇנְיֵֽךְ׃ אָמַר לָהּ מַלְאַךְ ה': אַתְּ תִּכַּנְסִי לְהֵרָיוֹן, וְתֵלְדִי בֵּן, וְתִקְרְאִי אֶת שְׁמוֹ יִשְׁמָעֵאל כִּי ה' שָׁמַע אוֹתָךְ מִשּׁוּם צַעֲרֵךְ.
יב וְה֤וּא יִהְיֶה֙ פֶּ֣רֶא אָדָ֔ם יָד֣וֹ בַכֹּ֔ל וְיַ֥ד כֹּ֖ל בּ֑וֹ וְעַל־פְּנֵ֥י כׇל־אֶחָ֖יו יִשְׁכֹּֽן׃ וְהוּא יִהְיֶה אָדָם דּוֹמֶה לְחַיַּת פֶּרֶא, הוּא יָרִיב עִם כֻּלָּם וְכֻלָּם יָרִיבוּ אִתּוֹ, וְצֶאֱצָאָיו יָגוּרוּ לְיַד הַצֶּאֱצָאִים שֶׁל כָּל אֶחָיו.
יג וַתִּקְרָ֤א שֵׁם־יְהֹוָה֙ הַדֹּבֵ֣ר אֵלֶ֔יהָ אַתָּ֖ה אֵ֣ל רֳאִ֑י כִּ֣י אָֽמְרָ֗ה הֲגַ֥ם הֲלֹ֛ם רָאִ֖יתִי אַחֲרֵ֥י רֹאִֽי׃ הִיא הוֹדְתָה לַה' הַמְּדַבֵּר אֵלֶיהָ וְאָמְרָה: אַתָּה אֵל רוֹאֶה וּמַשְׁגִּיחַ! כִּי אָמְרָה: גַּם כָּאן רָאִיתִי אֶת הַמַּלְאָךְ שֶׁלְּךָ, אַחֲרֵי שֶׁרָאִיתָ אֶת צַעֲרִי.
יד עַל־כֵּן֙ קָרָ֣א לַבְּאֵ֔ר בְּאֵ֥ר לַחַ֖י רֹאִ֑י הִנֵּ֥ה בֵין־קָדֵ֖שׁ וּבֵ֥ין בָּֽרֶד׃ לָכֵן נִקְרְאָה הַבְּאֵר 'בְּאֵר לַחַי רוֹאִי' – 'הַבְּאֵר לָאֵל הַחַי שֶׁרוֹאֶה אוֹתִי', וְהִיא נִמְצֵאת בֵּין רֶקֶם לְבֵין צֶאֱלִים.
טו וַתֵּ֧לֶד הָגָ֛ר לְאַבְרָ֖ם בֵּ֑ן וַיִּקְרָ֨א אַבְרָ֧ם שֶׁם־בְּנ֛וֹ אֲשֶׁר־יָלְדָ֥ה הָגָ֖ר יִשְׁמָעֵֽאל׃ הָגָר יָלְדָה לְאַבְרָם בֵּן, אֶת שֵׁם בְּנוֹ שֶׁיָּלְדָה הָגָר קָרָא אַבְרָם יִשְׁמָעֵאל.
טז וְאַבְרָ֕ם בֶּן־שְׁמֹנִ֥ים שָׁנָ֖ה וְשֵׁ֣שׁ שָׁנִ֑ים בְּלֶֽדֶת־הָגָ֥ר אֶת־יִשְׁמָעֵ֖אל לְאַבְרָֽם׃ וְאַבְרָם הָיָה בֶּן שְׁמוֹנִים וְשֵׁשׁ שָׁנִים כְּשֶׁהָגָר יָלְדָה אֶת יִשְׁמָעֵאל לְאַבְרָם.
ס פָּרָשַׁת בְּרִית הַמִּילָה
יז א וַיְהִ֣י אַבְרָ֔ם בֶּן־תִּשְׁעִ֥ים שָׁנָ֖ה וְתֵ֣שַׁע שָׁנִ֑ים וַיֵּרָ֨א יְהֹוָ֜ה אֶל־אַבְרָ֗ם וַיֹּ֤אמֶר אֵלָיו֙ אֲנִי־אֵ֣ל שַׁדַּ֔י הִתְהַלֵּ֥ךְ לְפָנַ֖י וֶהְיֵ֥ה תָמִֽים׃ אַבְרָם הָיָה בֶּן תִּשְׁעִים וְתֵשַׁע שָׁנִים, ה' הִתְגַּלָּה לְאַבְרָם וְאָמַר לוֹ: אֲנִי אֵל כָּל יָכוֹל, לֵךְ בִּדְרָכַי בִּשְׁלֵמוּת!
ב וְאֶתְּנָ֥ה בְרִיתִ֖י בֵּינִ֣י וּבֵינֶ֑ךָ וְאַרְבֶּ֥ה אוֹתְךָ֖ בִּמְאֹ֥ד מְאֹֽד׃ וְאֶעֱשֶׂה בֵּינִי וּבֵינְךָ אֶת הֶסְכֵּם הַבְּרִית שֶׁלִּי, וְאַרְבֶּה אֶת צֶאֱצָאֶיךָ מְאוֹד מְאוֹד.
ג וַיִּפֹּ֥ל אַבְרָ֖ם עַל־פָּנָ֑יו וַיְדַבֵּ֥ר אִתּ֛וֹ אֱלֹהִ֖ים לֵאמֹֽר׃ אַבְרָם נָפַל עַל פָּנָיו וְהִשְׁתַּחֲוָה, אֱלֹהִים דִּבֵּר אִתּוֹ וְאָמַר:
ד אֲנִ֕י הִנֵּ֥ה בְרִיתִ֖י אִתָּ֑ךְ וְהָיִ֕יתָ לְאַ֖ב הֲמ֥וֹן גּוֹיִֽם׃ הִנֵּה הֶסְכֵּם הַבְּרִית שֶׁלִּי אִתְּךָ, תִּהְיֶה לְאָב שֶׁל עַמִּים רַבִּים.
ה וְלֹא־יִקָּרֵ֥א ע֛וֹד אֶת־שִׁמְךָ֖ אַבְרָ֑ם וְהָיָ֤ה שִׁמְךָ֙ אַבְרָהָ֔ם כִּ֛י אַב־הֲמ֥וֹן גּוֹיִ֖ם נְתַתִּֽיךָ׃ וְלֹא יִקָּרֵא עוֹד שִׁמְךָ אַבְרָם, שִׁמְךָ יִהְיֶה אַבְרָהָם, כִּי אֲנִי עוֹשֶׂה אוֹתְךָ אָב לְעַמִּים רַבִּים.
ו וְהִפְרֵתִ֤י אֹֽתְךָ֙ בִּמְאֹ֣ד מְאֹ֔ד וּנְתַתִּ֖יךָ לְגוֹיִ֑ם וּמְלָכִ֖ים מִמְּךָ֥ יֵצֵֽאוּ׃ אַרְבֶּה אֶת צֶאֱצָאֶיךָ מְאוֹד מְאוֹד, וְאֶעֱשֶׂה מֵהֶם עַמִּים, וְיֵצְאוּ מִמְּךָ מְלָכִים.

שביעי[עריכה]

ז וַהֲקִמֹתִ֨י אֶת־בְּרִיתִ֜י בֵּינִ֣י וּבֵינֶ֗ךָ וּבֵ֨ין זַרְעֲךָ֧ אַחֲרֶ֛יךָ לְדֹרֹתָ֖ם לִבְרִ֣ית עוֹלָ֑ם לִהְי֤וֹת לְךָ֙ לֵֽאלֹהִ֔ים וּֽלְזַרְעֲךָ֖ אַחֲרֶֽיךָ׃ וְאֶעֱשֶׂה אֶת הֶסְכֵּם הַבְּרִית שֶׁלִּי בֵּינִי וּבֵינְךָ וּבֵין צֶאֱצָאֶיךָ אַחֲרֶיךָ בְּכָל הַדּוֹרוֹת – בְּרִית נִצְחִית, שֶׁאֶהְיֶה לְךָ לֵאלֹהִים וּלְצֶאֱצָאֶיךָ אַחֲרֶיךָ.
ח וְנָתַתִּ֣י לְ֠ךָ֠ וּלְזַרְעֲךָ֨ אַחֲרֶ֜יךָ אֵ֣ת ׀ אֶ֣רֶץ מְגֻרֶ֗יךָ אֵ֚ת כׇּל־אֶ֣רֶץ כְּנַ֔עַן לַאֲחֻזַּ֖ת עוֹלָ֑ם וְהָיִ֥יתִי לָהֶ֖ם לֵאלֹהִֽים׃ אֶתֵּן לְךָ וּלְצֶאֱצָאֶיךָ אַחֲרֶיךָ אֶת הָאָרֶץ שֶׁאַתָּה גָּר בָּהּ – אֶת כָּל אֶרֶץ כְּנַעַן שֶׁתִּהְיֶה אֲחֻזָתְךָ לְעוֹלָם, וְאֶהְיֶה לָהֶם לֵאלֹהִים.
ט וַיֹּ֤אמֶר אֱלֹהִים֙ אֶל־אַבְרָהָ֔ם וְאַתָּ֖ה אֶת־בְּרִיתִ֣י תִשְׁמֹ֑ר אַתָּ֛ה וְזַרְעֲךָ֥ אַֽחֲרֶ֖יךָ לְדֹרֹתָֽם׃ אֱלֹהִים אָמַר לְאַבְרָהָם: אַתָּה תִּשְׁמוֹר עַל הַהֶסְכֵּם אִתִּי, אַתָּה וְצֶאֱצָאֶיךָ אַחֲרֶיךָ בְּכָל הַדּוֹרוֹת.
י זֹ֣את בְּרִיתִ֞י אֲשֶׁ֣ר תִּשְׁמְר֗וּ בֵּינִי֙ וּבֵ֣ינֵיכֶ֔ם וּבֵ֥ין זַרְעֲךָ֖ אַחֲרֶ֑יךָ הִמּ֥וֹל לָכֶ֖ם כׇּל־זָכָֽר׃ זֶה הַהֶסְכֵּם בֵּינִי וּבֵינֵיכֶם וּבֵין צֶאֱצָאֶיךָ אַחֲרֶיךָ שֶׁתִּשְׁמְרוּ עָלָיו – לָמוּל לָכֶם כָּל זָכָר.
יא וּנְמַלְתֶּ֕ם אֵ֖ת בְּשַׂ֣ר עׇרְלַתְכֶ֑ם וְהָיָה֙ לְא֣וֹת בְּרִ֔ית בֵּינִ֖י וּבֵינֵיכֶֽם׃ תָּמוּלוּ אֶת בְּשַׂר עָרְלַתְכֶם וְזֶה יִהְיֶה סִימַן עַל הַהֶסְכֵּם שֶׁבֵּינִי וּבֵינֵיכֶם.
יב וּבֶן־שְׁמֹנַ֣ת יָמִ֗ים יִמּ֥וֹל לָכֶ֛ם כׇּל־זָכָ֖ר לְדֹרֹתֵיכֶ֑ם יְלִ֣יד בָּ֔יִת וּמִקְנַת־כֶּ֙סֶף֙ מִכֹּ֣ל בֶּן־נֵכָ֔ר אֲשֶׁ֛ר לֹ֥א מִֽזַּרְעֲךָ֖ הֽוּא׃ יָמוּלוּ כָּל זָכָר בֶּן שְׁמוֹנַת יָמִים שֶׁיִּוָּלֵד לָכֶם בְּכָל הַדּוֹרוֹת, גַּם בֵּן שֶׁל שִׁפְחָתְךָ וְגַם עֶבֶד שֶׁנִּקְנָה בְּכֶסֶף מִכָּל עַם אַחֵר שֶׁאֵינוֹ מִצֶּאֱצָאֶיךָ.
יג הִמּ֧וֹל ׀ יִמּ֛וֹל יְלִ֥יד בֵּֽיתְךָ֖ וּמִקְנַ֣ת כַּסְפֶּ֑ךָ וְהָיְתָ֧ה בְרִיתִ֛י בִּבְשַׂרְכֶ֖ם לִבְרִ֥ית עוֹלָֽם׃ יָמוּלוּ אֶת בֶּן הַשִּׁפְחָה וְאֶת הָעֶבֶד שֶׁנִּקְנָה בְּכֶסֶף, וְסִימַן הַהֶסְכֵּם אִתִּי יִהְיֶה בִּבְשַׂרְכֶם לְהֶסְכֵּם בְּרִית נִצְחִית.
יד וְעָרֵ֣ל ׀ זָכָ֗ר אֲשֶׁ֤ר לֹֽא־יִמּוֹל֙ אֶת־בְּשַׂ֣ר עׇרְלָת֔וֹ וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵעַמֶּ֑יהָ אֶת־בְּרִיתִ֖י הֵפַֽר׃ וְזָכָר עָרֵל – שֶׁלֹּא יִמּוֹל אֶת בְּשַׂר עָרְלָתוֹ – הַנֶּפֶשׁ הַהִיא תֹּאבַד מִתּוֹךְ עַמָּהּ כִּי הוּא הֵפֵר אֶת הֶסְכֵּמִי.
ס פָּרָשַׁת בְּשׂוֹרַת לֵידַת יִצְחָק
טו וַיֹּ֤אמֶר אֱלֹהִים֙ אֶל־אַבְרָהָ֔ם שָׂרַ֣י אִשְׁתְּךָ֔ לֹא־תִקְרָ֥א אֶת־שְׁמָ֖הּ שָׂרָ֑י כִּ֥י שָׂרָ֖ה שְׁמָֽהּ׃ אֱלֹהִים אָמַר לְאַבְרָהָם: שָׂרַי אִשְׁתְּךָ אַל תִּקְרָא אֶת שְׁמָהּ שָׂרַי כִּי שְׁמָהּ שָׂרָה.
טז וּבֵרַכְתִּ֣י אֹתָ֔הּ וְגַ֨ם נָתַ֧תִּי מִמֶּ֛נָּה לְךָ֖ בֵּ֑ן וּבֵֽרַכְתִּ֙יהָ֙ וְהָֽיְתָ֣ה לְגוֹיִ֔ם מַלְכֵ֥י עַמִּ֖ים מִמֶּ֥נָּה יִהְיֽוּ׃ אֲבָרֵךְ אוֹתָהּ וְגַם אֶתֵּן לְךָ בֵּן מִמֶּנָּה, וַאֲבָרֵךְ אוֹתָהּ שֶׁיִּהְיוּ מִצֶּאֱצָאֶיהָ עַמִּים – מַלְכֵי עַמִּים יֵצְאוּ מִמֶּנָּה.
יז וַיִּפֹּ֧ל אַבְרָהָ֛ם עַל־פָּנָ֖יו וַיִּצְחָ֑ק וַיֹּ֣אמֶר בְּלִבּ֗וֹ הַלְּבֶ֤ן מֵאָֽה־שָׁנָה֙ יִוָּלֵ֔ד וְאִ֨ם־שָׂרָ֔ה הֲבַת־תִּשְׁעִ֥ים שָׁנָ֖ה תֵּלֵֽד׃ אַבְרָהָם נָפַל עַל פָּנָיו וְהִשְׁתַּחֲוָה וְתָמַהּ וְאָמַר בְּלִבּוֹ: הַאִם לְבֶן מֵאָה שָׁנָה יִוָּלֵד בֵּן?! וְהַאִם שָׂרָה שֶׁהִיא בַּת תִּשְׁעִים שָׁנָה תֵּלֵד?!
יח וַיֹּ֥אמֶר אַבְרָהָ֖ם אֶל־הָֽאֱלֹהִ֑ים ל֥וּ יִשְׁמָעֵ֖אל יִחְיֶ֥ה לְפָנֶֽיךָ׃ אַבְרָהָם אָמַר לָאֱלֹהִים: הַלְוַאי שֶׁצֶּאֱצָאֵי יִשְׁמָעֵאל יִחְיוּ לְפָנֶיךָ!
יט וַיֹּ֣אמֶר אֱלֹהִ֗ים אֲבָל֙ שָׂרָ֣ה אִשְׁתְּךָ֗ יֹלֶ֤דֶת לְךָ֙ בֵּ֔ן וְקָרָ֥אתָ אֶת־שְׁמ֖וֹ יִצְחָ֑ק וַהֲקִמֹתִ֨י אֶת־בְּרִיתִ֥י אִתּ֛וֹ לִבְרִ֥ית עוֹלָ֖ם לְזַרְע֥וֹ אַחֲרָֽיו׃ אֱלֹהִים אָמַר: אֲבָל שָׂרָה שֶׁהִיא אִשְׁתְּךָ יוֹלֶדֶת לְךָ בֵּן, וְתִקְרָא אֶת שְׁמוֹ יִצְחָק! וְאֶעֱשֶׂה אֶת הֶסְכֵּם הַבְּרִית שֶׁלִּי דַּוְקָא אִתּוֹ לִבְרִית נִצְחִית לְצֶאֱצָאָיו אַחֲרָיו.
כ וּֽלְיִשְׁמָעֵאל֮ שְׁמַעְתִּ֒יךָ֒ הִנֵּ֣ה ׀ בֵּרַ֣כְתִּי אֹת֗וֹ וְהִפְרֵיתִ֥י אֹת֛וֹ וְהִרְבֵּיתִ֥י אֹת֖וֹ בִּמְאֹ֣ד מְאֹ֑ד שְׁנֵים־עָשָׂ֤ר נְשִׂיאִם֙ יוֹלִ֔יד וּנְתַתִּ֖יו לְג֥וֹי גָּדֽוֹל׃ וּלְגַבֵּי יִשְׁמָעֵאל אֲנִי מְקַבֵּל אֶת בַּקָּשְׁתְךָ, הִנֵּה בֵּרַכְתִּי אוֹתוֹ וְאֶתֵּן לוֹ צֶאֱצָאִים וְאַרְבֶּה אֶת צֶאֱצָאָיו מְאוֹד מְאוֹד, שְׁנֵים עָשָׂר נְשִׂיאִים יוֹלִיד, וְאֶעֱשֶׂה מִמֶּנּוּ עַם גָּדוֹל.
כא וְאֶת־בְּרִיתִ֖י אָקִ֣ים אֶת־יִצְחָ֑ק אֲשֶׁר֩ תֵּלֵ֨ד לְךָ֤ שָׂרָה֙ לַמּוֹעֵ֣ד הַזֶּ֔ה בַּשָּׁנָ֖ה הָאַחֶֽרֶת׃ אֲבָל אֶת הֶסְכֵּם הַבְּרִית שֶׁלִּי אֶעֱשֶׂה עִם יִצְחָק שֶׁשָּׂרָה תֵּלֵד לְךָ בַּתַּאֲרִיךְ וּבַזְּמַן הַזֶּה בַּשָּׁנָה הַבָּאָה.
כב וַיְכַ֖ל לְדַבֵּ֣ר אִתּ֑וֹ וַיַּ֣עַל אֱלֹהִ֔ים מֵעַ֖ל אַבְרָהָֽם׃ כְּשֶׁסִּיֵּם לְדַבֵּר אִתּוֹ, אֱלֹהִים הִסְתַּלֵּק מֵעַל אַבְרָהָם.
פָּרָשַׁת מִלַּת אַנְשֵׁי אַבְרָהָם
כג וַיִּקַּ֨ח אַבְרָהָ֜ם אֶת־יִשְׁמָעֵ֣אל בְּנ֗וֹ וְאֵ֨ת כׇּל־יְלִידֵ֤י בֵיתוֹ֙ וְאֵת֙ כׇּל־מִקְנַ֣ת כַּסְפּ֔וֹ כׇּל־זָכָ֕ר בְּאַנְשֵׁ֖י בֵּ֣ית אַבְרָהָ֑ם וַיָּ֜מׇל אֶת־בְּשַׂ֣ר עׇרְלָתָ֗ם בְּעֶ֙צֶם֙ הַיּ֣וֹם הַזֶּ֔ה כַּאֲשֶׁ֛ר דִּבֶּ֥ר אִתּ֖וֹ אֱלֹהִֽים׃ אַבְרָהָם לָקַח אֶת יִשְׁמָעֵאל בְּנוֹ וְאֶת כָּל בְּנֵי שִׁפְחוֹתָיו וְאֶת כָּל הָעֲבָדִים שֶׁקָּנָה בְּכַסְפּוֹ, כָּל זָכָר בְּאַנְשֵׁי בֵּית אַבְרָהָם, וּמָל אֶת עוֹר עָרְלָתָם בַּיּוֹם הַזֶּה עַצְמוֹ כְּמוֹ שֶׁאֱלֹהִים דִּבֵּר אִתּוֹ.

מפטיר[עריכה]

כד וְאַ֨בְרָהָ֔ם בֶּן־תִּשְׁעִ֥ים וָתֵ֖שַׁע שָׁנָ֑ה בְּהִמֹּל֖וֹ בְּשַׂ֥ר עׇרְלָתֽוֹ׃ וְאַבְרָהָם הָיָה בֶּן תִּשְׁעִים וָתֵשַׁע שָׁנָה כְּשֶׁמָּל אֶת עוֹר עָרְלָתוֹ.
כה וְיִשְׁמָעֵ֣אל בְּנ֔וֹ בֶּן־שְׁלֹ֥שׁ עֶשְׂרֵ֖ה שָׁנָ֑ה בְּהִ֨מֹּל֔וֹ אֵ֖ת בְּשַׂ֥ר עׇרְלָתֽוֹ׃ וְיִשְׁמָעֵאל בְּנוֹ הָיָה בֶּן שְׁלֹשׁ עֶשְׂרֵה שָׁנָה כְּשֶׁמָּלוּ אֶת עוֹר עָרְלָתוֹ.
כו בְּעֶ֙צֶם֙ הַיּ֣וֹם הַזֶּ֔ה נִמּ֖וֹל אַבְרָהָ֑ם וְיִשְׁמָעֵ֖אל בְּנֽוֹ׃ בַּיּוֹם הַזֶּה עַצְמוֹ נִמּוֹלוּ אַבְרָהָם וְיִשְׁמָעֵאל בְּנוֹ.
כז וְכׇל־אַנְשֵׁ֤י בֵיתוֹ֙ יְלִ֣יד בָּ֔יִת וּמִקְנַת־כֶּ֖סֶף מֵאֵ֣ת בֶּן־נֵכָ֑ר נִמֹּ֖לוּ אִתּֽוֹ׃ וְכָל אַנְשֵׁי בֵּיתוֹ – בְּנֵי שִׁפְחוֹתָיו וְאֵלֶּה שֶׁנִּקְנוּ בְּכֶסֶף מִבֶּן עַם אַחֵר – נִמּוֹלוּ אִתּוֹ.
