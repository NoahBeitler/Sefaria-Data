ראשון[עריכה]

ס פָּרָשַׁת דִּבְרֵי יְהוּדָה
מד יח וַיִּגַּ֨שׁ אֵלָ֜יו יְהוּדָ֗ה וַיֹּ֘אמֶר֮ בִּ֣י אֲדֹנִי֒ יְדַבֶּר־נָ֨א עַבְדְּךָ֤ דָבָר֙ בְּאׇזְנֵ֣י אֲדֹנִ֔י וְאַל־יִ֥חַר אַפְּךָ֖ בְּעַבְדֶּ֑ךָ כִּ֥י כָמ֖וֹךָ כְּפַרְעֹֽה׃ יְהוּדָה נִגַּשׁ אֵלָיו וְאָמַר: בְּבַקָּשָׁה אֲדוֹנִי, עַבְדְּךָ יְדַבֵּר דָּבָר בְּאָזְנֵי אֲדוֹנִי בְּבַקָּשָׁה וְאַל תִּכְעַס עַל עַבְדְּךָ, שֶׁהֲרֵי אַתָּה חָשׁוּב כְּמוֹ פַּרְעֹה.
יט אֲדֹנִ֣י שָׁאַ֔ל אֶת־עֲבָדָ֖יו לֵאמֹ֑ר הֲיֵשׁ־לָכֶ֥ם אָ֖ב אוֹ־אָֽח׃ אֲדוֹנִי שָׁאַל אֶת עֲבָדָיו וְאָמַר: הַאִם יֵשׁ לָכֶם אָב אוֹ אָח?
כ וַנֹּ֙אמֶר֙ אֶל־אֲדֹנִ֔י יֶשׁ־לָ֙נוּ֙ אָ֣ב זָקֵ֔ן וְיֶ֥לֶד זְקֻנִ֖ים קָטָ֑ן וְאָחִ֣יו מֵ֔ת וַיִּוָּתֵ֨ר ה֧וּא לְבַדּ֛וֹ לְאִמּ֖וֹ וְאָבִ֥יו אֲהֵבֽוֹ׃ אָמַרְנוּ לַאֲדוֹנִי: יֵשׁ לָנוּ אָב זָקֵן וְיֵשׁ לוֹ יֶלֶד קָטָן שֶׁנּוֹלַד לְעֵת זִקְנוּתוֹ, וְאָחִיו מֵאִמּוֹ מֵת וְהוּא נִשְׁאַר לְבַדּוֹ מִבָּנֶיהָ, וְאָבִיו אוֹהֵב אוֹתוֹ.
כא וַתֹּ֙אמֶר֙ אֶל־עֲבָדֶ֔יךָ הוֹרִדֻ֖הוּ אֵלָ֑י וְאָשִׂ֥ימָה עֵינִ֖י עָלָֽיו׃ אָמַרְתָּ לָנוּ – לַעֲבָדֶיךָ: הוֹרִידוּ אוֹתוֹ אֵלַי וְאֶרְאֶה אוֹתוֹ.
כב וַנֹּ֙אמֶר֙ אֶל־אֲדֹנִ֔י לֹא־יוּכַ֥ל הַנַּ֖עַר לַעֲזֹ֣ב אֶת־אָבִ֑יו וְעָזַ֥ב אֶת־אָבִ֖יו וָמֵֽת׃ אָמַרְנוּ לַאֲדוֹנִי: הַנַּעַר לֹא יוּכַל לַעֲזוֹב אֶת אָבִיו, אִם יַעֲזוֹב אֶת אָבִיו, אָבִיו עָלוּל לָמוּת.
כג וַתֹּ֙אמֶר֙ אֶל־עֲבָדֶ֔יךָ אִם־לֹ֥א יֵרֵ֛ד אֲחִיכֶ֥ם הַקָּטֹ֖ן אִתְּכֶ֑ם לֹ֥א תֹסִפ֖וּן לִרְא֥וֹת פָּנָֽי׃ אָמַרְתָּ לַעֲבָדֶיךָ: אִם אֲחִיכֶם הַקָּטָן לֹא יֵרֵד אִתְּכֶם – לֹא תּוּכְלוּ יוֹתֵר לִפְגוֹשׁ אוֹתִי.
כד וַֽיְהִי֙ כִּ֣י עָלִ֔ינוּ אֶֽל־עַבְדְּךָ֖ אָבִ֑י וַנַּ֨גֶּד־ל֔וֹ אֵ֖ת דִּבְרֵ֥י אֲדֹנִֽי׃ כְּשֶׁעָלִינוּ אֶל עַבְדְּךָ – אָבִי, אָמַרְנוּ לוֹ אֶת דִּבְרֵי אֲדוֹנִי.
כה וַיֹּ֖אמֶר אָבִ֑ינוּ שֻׁ֖בוּ שִׁבְרוּ־לָ֥נוּ מְעַט־אֹֽכֶל׃ אָמַר אָבִינוּ: שׁוּבוּ וּקְנוּ לָנוּ מְעַט אוֹכֶל.
כו וַנֹּ֕אמֶר לֹ֥א נוּכַ֖ל לָרֶ֑דֶת אִם־יֵשׁ֩ אָחִ֨ינוּ הַקָּטֹ֤ן אִתָּ֙נוּ֙ וְיָרַ֔דְנוּ כִּי־לֹ֣א נוּכַ֗ל לִרְאוֹת֙ פְּנֵ֣י הָאִ֔ישׁ וְאָחִ֥ינוּ הַקָּטֹ֖ן אֵינֶ֥נּוּ אִתָּֽנוּ׃ אָמַרְנוּ: לֹא נוּכַל לָרֶדֶת, אֶלָּא רַק אִם אָחִינוּ הַקָּטָן יִהְיֶה אִתָּנוּ נֵרֵד, כִּי לֹא נוּכַל לִפְגוֹשׁ אֶת הָאִישׁ כְּשֶׁאָחִינוּ הַקָּטָן אֵינֶנּוּ אִתָּנוּ.
כז וַיֹּ֛אמֶר עַבְדְּךָ֥ אָבִ֖י אֵלֵ֑ינוּ אַתֶּ֣ם יְדַעְתֶּ֔ם כִּ֥י שְׁנַ֖יִם יָֽלְדָה־לִּ֥י אִשְׁתִּֽי׃ עַבְדְּךָ אָבִי אָמַר לָנוּ: הֲרֵי יְדַעְתֶּם שֶׁאִשְׁתִּי יָלְדָה לִי שְׁנֵי בָּנִים.
כח וַיֵּצֵ֤א הָֽאֶחָד֙ מֵֽאִתִּ֔י וָאֹמַ֕ר אַ֖ךְ טָרֹ֣ף טֹרָ֑ף וְלֹ֥א רְאִיתִ֖יו עַד־הֵֽנָּה׃ אֶחָד מֵהֶם הָלַךְ מִמֶּנִּי, וְאָמַרְתִּי שֶׁכַּנִּרְאֶה נִטְרַף, וְאָכֵן לֹא רָאִיתִי אוֹתוֹ עַד עַכְשָׁיו.
כט וּלְקַחְתֶּ֧ם גַּם־אֶת־זֶ֛ה מֵעִ֥ם פָּנַ֖י וְקָרָ֣הוּ אָס֑וֹן וְהֽוֹרַדְתֶּ֧ם אֶת־שֵׂיבָתִ֛י בְּרָעָ֖ה שְׁאֹֽלָה׃ וְאִם תִּקְחוּ גַּם אוֹתוֹ מִמֶּנִּי וְיִקְרֶה לוֹ אָסוֹן – תּוֹרִידוּ אוֹתִי בְּזִקְנוּתִי בְּצַעַר לַקֶּבֶר.
ל וְעַתָּ֗ה כְּבֹאִי֙ אֶל־עַבְדְּךָ֣ אָבִ֔י וְהַנַּ֖עַר אֵינֶ֣נּוּ אִתָּ֑נוּ וְנַפְשׁ֖וֹ קְשׁוּרָ֥ה בְנַפְשֽׁוֹ׃ וְעַכְשָׁיו כְּשֶׁאָבוֹא אֶל עַבְדְּךָ אָבִי, וְהַנַּעַר אֵינֶנּוּ אִתָּנוּ, וְנַפְשׁוֹ קְשׁוּרָה בְּנַפְשׁוֹ –

שני[עריכה]

לא וְהָיָ֗ה כִּרְאוֹת֛וֹ כִּי־אֵ֥ין הַנַּ֖עַר וָמֵ֑ת וְהוֹרִ֨ידוּ עֲבָדֶ֜יךָ אֶת־שֵׂיבַ֨ת עַבְדְּךָ֥ אָבִ֛ינוּ בְּיָג֖וֹן שְׁאֹֽלָה׃ כְּשֶׁיִּרְאֶה שֶׁהַנַּעַר אֵינֶנּוּ, הוּא יָמוּת! וַעֲבָדֶיךָ יוֹרִידוּ אֶת עַבְדְּךָ אָבִינוּ בְּזִקְנוּתוֹ בְּצַעַר לַקֶּבֶר.
לב כִּ֤י עַבְדְּךָ֙ עָרַ֣ב אֶת־הַנַּ֔עַר מֵעִ֥ם אָבִ֖י לֵאמֹ֑ר אִם־לֹ֤א אֲבִיאֶ֙נּוּ֙ אֵלֶ֔יךָ וְחָטָ֥אתִי לְאָבִ֖י כׇּל־הַיָּמִֽים׃ הֲרֵי אֲנִי עַבְדְּךָ נַעֲשֵׂיתִי עָרֵב עַל הַנַּעַר כְּשֶׁלָּקַחְתִּי אוֹתוֹ מֵאָבִי, וְאָמַרְתִּי לוֹ: אִם לֹא אֲבִיא אוֹתוֹ אֵלֶיךָ אֵחָשֵׁב חוֹטֵא לְאָבִי לְעוֹלָם.
לג וְעַתָּ֗ה יֵֽשֶׁב־נָ֤א עַבְדְּךָ֙ תַּ֣חַת הַנַּ֔עַר עֶ֖בֶד לַֽאדֹנִ֑י וְהַנַּ֖עַר יַ֥עַל עִם־אֶחָֽיו׃ וְעַכְשָׁיו עַבְדְּךָ יִשָּׁאֵר בְּבַקָּשָׁה בִּמְקוֹם הַנַּעַר כְּעֶבֶד לַאדוֹנִי, וְהַנַּעַר יַעֲלֶה עִם אֶחָיו.
לד כִּי־אֵיךְ֙ אֶֽעֱלֶ֣ה אֶל־אָבִ֔י וְהַנַּ֖עַר אֵינֶ֣נּוּ אִתִּ֑י פֶּ֚ן אֶרְאֶ֣ה בָרָ֔ע אֲשֶׁ֥ר יִמְצָ֖א אֶת־אָבִֽי׃ כִּי אֵיךְ אֶעֱלֶה אֶל אָבִי וְהַנַּעַר אֵינֶנּוּ אִתִּי, שֶׁלֹּא אֶרְאֶה בָּרָע שֶׁיִּקְרֶה לְאָבִי.
פָּרָשַׁת הִתְוַדְּעוּת יוֹסֵף לְאֶחָיו
מה א וְלֹֽא־יָכֹ֨ל יוֹסֵ֜ף לְהִתְאַפֵּ֗ק לְכֹ֤ל הַנִּצָּבִים֙ עָלָ֔יו וַיִּקְרָ֕א הוֹצִ֥יאוּ כׇל־אִ֖ישׁ מֵעָלָ֑י וְלֹא־עָ֤מַד אִישׁ֙ אִתּ֔וֹ בְּהִתְוַדַּ֥ע יוֹסֵ֖ף אֶל־אֶחָֽיו׃ יוֹסֵף לֹא הָיָה יָכוֹל לְהִתְאַפֵּק מִלִּבְכּוֹת לִפְנֵי כָּל הָעוֹמְדִים לְפָנָיו, וְקָרָא: הוֹצִיאוּ אֶת כָּל הָאֲנָשִׁים מִלְּפָנַי! וְאִישׁ לֹא עָמַד אִתּוֹ כְּשֶׁיּוֹסֵף גִּלָּה לְאֶחָיו אֶת זֶהוּתוֹ.
ב וַיִּתֵּ֥ן אֶת־קֹל֖וֹ בִּבְכִ֑י וַיִּשְׁמְע֣וּ מִצְרַ֔יִם וַיִּשְׁמַ֖ע בֵּ֥ית פַּרְעֹֽה׃ הוּא הִשְׁמִיעַ אֶת קוֹלוֹ בְּבֶכִי, אַנְשֵׁי מִצְרַיִם שָׁמְעוּ וְאַנְשֵׁי בֵּית פַּרְעֹה שָׁמְעוּ.
ג וַיֹּ֨אמֶר יוֹסֵ֤ף אֶל־אֶחָיו֙ אֲנִ֣י יוֹסֵ֔ף הַע֥וֹד אָבִ֖י חָ֑י וְלֹֽא־יָכְל֤וּ אֶחָיו֙ לַעֲנ֣וֹת אֹת֔וֹ כִּ֥י נִבְהֲל֖וּ מִפָּנָֽיו׃ יוֹסֵף אָמַר לְאֶחָיו: אֲנִי יוֹסֵף! הַאִם אָבִי עֲדַיִן חַי? אֶחָיו לֹא יָכְלוּ לַעֲנוֹת לוֹ כִּי הֵם נִבְהֲלוּ מִמֶּנּוּ.
ד וַיֹּ֨אמֶר יוֹסֵ֧ף אֶל־אֶחָ֛יו גְּשׁוּ־נָ֥א אֵלַ֖י וַיִּגָּ֑שׁוּ וַיֹּ֗אמֶר אֲנִי֙ יוֹסֵ֣ף אֲחִיכֶ֔ם אֲשֶׁר־מְכַרְתֶּ֥ם אֹתִ֖י מִצְרָֽיְמָה׃ יוֹסֵף אָמַר לְאֶחָיו: גְּשׁוּ אֵלַי בְּבַקָּשָׁה! הֵם נִגְּשׁוּ. אָמַר לָהֶם: אֲנִי יוֹסֵף אֲחִיכֶם שֶׁמְּכַרְתֶּם אוֹתִי לְמִצְרַיִם.
ה וְעַתָּ֣ה ׀ אַל־תֵּעָ֣צְב֗וּ וְאַל־יִ֙חַר֙ בְּעֵ֣ינֵיכֶ֔ם כִּֽי־מְכַרְתֶּ֥ם אֹתִ֖י הֵ֑נָּה כִּ֣י לְמִֽחְיָ֔ה שְׁלָחַ֥נִי אֱלֹהִ֖ים לִפְנֵיכֶֽם׃ וְעַכְשָׁיו אַל תִּצְטַעֲרוּ וְאַל יִהְיֶה רַע בְּעֵינֵיכֶם שֶׁמְּכַרְתֶּם אוֹתִי לְכָאן, כִּי אֱלֹהִים שָׁלַח אוֹתִי כְּדֵי שֶׁיִּהְיֶה מָזוֹן לְקִיּוּמְכֶם.
ו כִּי־זֶ֛ה שְׁנָתַ֥יִם הָרָעָ֖ב בְּקֶ֣רֶב הָאָ֑רֶץ וְעוֹד֙ חָמֵ֣שׁ שָׁנִ֔ים אֲשֶׁ֥ר אֵין־חָרִ֖ישׁ וְקָצִֽיר׃ כִּי כְּבָר שְׁנָתַיִם שֶׁיֵּשׁ רָעָב בָּאֵזוֹר, וְנוֹתְרוּ עוֹד חָמֵשׁ שָׁנִים לְלֹא חָרִישׁ וְקָצִיר.
ז וַיִּשְׁלָחֵ֤נִי אֱלֹהִים֙ לִפְנֵיכֶ֔ם לָשׂ֥וּם לָכֶ֛ם שְׁאֵרִ֖ית בָּאָ֑רֶץ וּלְהַחֲי֣וֹת לָכֶ֔ם לִפְלֵיטָ֖ה גְּדֹלָֽה׃ אֱלֹהִים שָׁלַח אוֹתִי לִפְנֵיכֶם כְּדֵי שֶׁתִּשְׂרְדוּ וְתִשָּׁאֲרוּ בָּאָרֶץ, לָתֵת לָכֶם מָזוֹן וּמִחְיָה לְקִיּוּמְכֶם, לְהַצָּלָה גְּדוֹלָה.

שלישי[עריכה]

ח וְעַתָּ֗ה לֹֽא־אַתֶּ֞ם שְׁלַחְתֶּ֤ם אֹתִי֙ הֵ֔נָּה כִּ֖י הָאֱלֹהִ֑ים וַיְשִׂימֵ֨נִֽי לְאָ֜ב לְפַרְעֹ֗ה וּלְאָדוֹן֙ לְכׇל־בֵּית֔וֹ וּמֹשֵׁ֖ל בְּכׇל־אֶ֥רֶץ מִצְרָֽיִם׃ וְעַכְשָׁיו מִתְבָּרֵר שֶׁלֹּא אַתֶּם שְׁלַחְתֶּם אוֹתִי לְכָאן אֶלָּא הָאֱלֹהִים, וְהוּא עָשָׂה אוֹתִי מַדְרִיךְ לְפַרְעֹה וְאָדוֹן לְכָל אַנְשֵׁי בֵּיתוֹ וּמוֹשֵׁל בְּכָל אֶרֶץ מִצְרַיִם.
ט מַהֲרוּ֮ וַעֲל֣וּ אֶל־אָבִי֒ וַאֲמַרְתֶּ֣ם אֵלָ֗יו כֹּ֤ה אָמַר֙ בִּנְךָ֣ יוֹסֵ֔ף שָׂמַ֧נִי אֱלֹהִ֛ים לְאָד֖וֹן לְכׇל־מִצְרָ֑יִם רְדָ֥ה אֵלַ֖י אַֽל־תַּעֲמֹֽד׃ מַהֲרוּ וַעֲלוּ אֶל אָבִי וְאִמְרוּ לוֹ: כָּךְ אָמַר בִּנְךָ יוֹסֵף: אֱלֹהִים עָשָׂה אוֹתִי אָדוֹן שֶׁל כָּל מִצְרַיִם, רֵד אֵלַי, אַל תִּתְעַכֵּב!
י וְיָשַׁבְתָּ֣ בְאֶֽרֶץ־גֹּ֗שֶׁן וְהָיִ֤יתָ קָרוֹב֙ אֵלַ֔י אַתָּ֕ה וּבָנֶ֖יךָ וּבְנֵ֣י בָנֶ֑יךָ וְצֹאנְךָ֥ וּבְקָרְךָ֖ וְכׇל־אֲשֶׁר־לָֽךְ׃ תָּגוּר בְּאֶרֶץ גּוֹשֶׁן וְתִהְיֶה קָרוֹב אֵלַי, אַתָּה וּבָנֶיךָ וּבְנֵי בָּנֶיךָ וְצֹאנְךָ וּבְקָרְךָ וְכָל מַה שֶּׁשַּׁיָּךְ לְךָ.
יא וְכִלְכַּלְתִּ֤י אֹֽתְךָ֙ שָׁ֔ם כִּי־ע֛וֹד חָמֵ֥שׁ שָׁנִ֖ים רָעָ֑ב פֶּן־תִּוָּרֵ֛שׁ אַתָּ֥ה וּבֵֽיתְךָ֖ וְכׇל־אֲשֶׁר־לָֽךְ׃ וַאֲפַרְנֵס אוֹתְךָ שָׁם כִּי נוֹתְרוּ עוֹד חָמֵשׁ שָׁנִים שֶׁל רָעָב, כְּדֵי שֶׁלֹּא תַּהֲפוֹךְ לְעָנִי אַתָּה וּמִשְׁפַּחְתְּךָ וְכָל מַה שֶּׁשַּׁיָּךְ לְךָ.
יב וְהִנֵּ֤ה עֵֽינֵיכֶם֙ רֹא֔וֹת וְעֵינֵ֖י אָחִ֣י בִנְיָמִ֑ין כִּי־פִ֖י הַֽמְדַבֵּ֥ר אֲלֵיכֶֽם׃ וְהִנֵּה עֵינֵיכֶם רוֹאוֹת וְעֵינֵי אָחִי בִּנְיָמִין שֶׁאֲנִי מְדַבֵּר אֲלֵיכֶם בְּפִי בִּשְׂפַתְכֶם.
יג וְהִגַּדְתֶּ֣ם לְאָבִ֗י אֶת־כׇּל־כְּבוֹדִי֙ בְּמִצְרַ֔יִם וְאֵ֖ת כׇּל־אֲשֶׁ֣ר רְאִיתֶ֑ם וּמִֽהַרְתֶּ֛ם וְהוֹרַדְתֶּ֥ם אֶת־אָבִ֖י הֵֽנָּה׃ סַפְּרוּ לְאָבִי עַל כָּל כְּבוֹדִי בְּמִצְרַיִם, וְאֶת כָּל מַה שֶּׁרְאִיתֶם, וּמַהֲרוּ וְהוֹרִידוּ אֶת אָבִי לְכָאן.
יד וַיִּפֹּ֛ל עַל־צַוְּארֵ֥י בִנְיָמִֽן־אָחִ֖יו וַיֵּ֑בְךְּ וּבִ֨נְיָמִ֔ן בָּכָ֖ה עַל־צַוָּארָֽיו׃ הוּא נָפַל עַל צַוָּארוֹ שֶׁל בִּנְיָמִין אָחִיו וּבָכָה, וּבִנְיָמִין בָּכָה עַל צַוָּארוֹ.
טו וַיְנַשֵּׁ֥ק לְכׇל־אֶחָ֖יו וַיֵּ֣בְךְּ עֲלֵהֶ֑ם וְאַ֣חֲרֵי כֵ֔ן דִּבְּר֥וּ אֶחָ֖יו אִתּֽוֹ׃ הוּא נִשֵּׁק אֶת כָּל אֶחָיו וּבָכָה עֲלֵיהֶם, וְרַק אַחַר כָּךְ אֶחָיו דִּבְּרוּ אִתּוֹ.
פָּרָשַׁת הַזְמָנַת יִשְׂרָאֵל
טז וְהַקֹּ֣ל נִשְׁמַ֗ע בֵּ֤ית פַּרְעֹה֙ לֵאמֹ֔ר בָּ֖אוּ אֲחֵ֣י יוֹסֵ֑ף וַיִּיטַב֙ בְּעֵינֵ֣י פַרְעֹ֔ה וּבְעֵינֵ֖י עֲבָדָֽיו׃ הַשְּׁמוּעָה נִשְׁמְעָה בְּבֵית פַּרְעֹה שֶׁאֲחֵי יוֹסֵף בָּאוּ וְהַדָּבָר מָצָא חֵן בְּעֵינֵי פַּרְעֹה וּבְעֵינֵי עֲבָדָיו.
יז וַיֹּ֤אמֶר פַּרְעֹה֙ אֶל־יוֹסֵ֔ף אֱמֹ֥ר אֶל־אַחֶ֖יךָ זֹ֣את עֲשׂ֑וּ טַֽעֲנוּ֙ אֶת־בְּעִ֣ירְכֶ֔ם וּלְכוּ־בֹ֖אוּ אַ֥רְצָה כְּנָֽעַן׃ פַּרְעֹה אָמַר לְיוֹסֵף: אֱמוֹר לְאַחֶיךָ: כָּךְ עֲשׂוּ – הַעֲמִיסוּ אֶת בֶּהֱמוֹתֵיכֶם בִּתְבוּאָה וּלְכוּ וּבוֹאוּ לְאֶרֶץ כְּנַעַן.
יח וּקְח֧וּ אֶת־אֲבִיכֶ֛ם וְאֶת־בָּתֵּיכֶ֖ם וּבֹ֣אוּ אֵלָ֑י וְאֶתְּנָ֣ה לָכֶ֗ם אֶת־טוּב֙ אֶ֣רֶץ מִצְרַ֔יִם וְאִכְל֖וּ אֶת־חֵ֥לֶב הָאָֽרֶץ׃ וּקְחוּ אֶת אֲבִיכֶם וְאֶת בְּנֵי מִשְׁפְּחוֹתֵכֶם, וּבוֹאוּ אֵלַי, וְאֶתֵּן לָכֶם אֶת טוּב אֶרֶץ מִצְרַיִם וְאִכְלוּ מֵהַדְּבָרִים הַמְּעֻלִּים שֶׁבָּאָרֶץ.

רביעי[עריכה]

יט וְאַתָּ֥ה צֻוֵּ֖יתָה זֹ֣את עֲשׂ֑וּ קְחוּ־לָכֶם֩ מֵאֶ֨רֶץ מִצְרַ֜יִם עֲגָל֗וֹת לְטַפְּכֶם֙ וְלִנְשֵׁיכֶ֔ם וּנְשָׂאתֶ֥ם אֶת־אֲבִיכֶ֖ם וּבָאתֶֽם׃ וַאֲנִי מְצַוֶּה אוֹתְךָ: כָּךְ עֲשׂוּ – קְחוּ לָכֶם מֵאֶרֶץ מִצְרַיִם עֲגָלוֹת לְיַלְדֵיכֶם וְלִנְשֵׁיכֶם, וּקְחוּ אֶת אֲבִיכֶם וּבוֹאוּ.
כ וְעֵ֣ינְכֶ֔ם אַל־תָּחֹ֖ס עַל־כְּלֵיכֶ֑ם כִּי־ט֛וּב כׇּל־אֶ֥רֶץ מִצְרַ֖יִם לָכֶ֥ם הֽוּא׃ וְאַל תִּצְטַעֲרוּ עַל כְּלֵיכֶם שֶׁקָּשֶׁה לַהֲבִיאָם, כִּי טוּב כָּל אֶרֶץ מִצְרַיִם שֶׁלָּכֶם הוּא.
כא וַיַּֽעֲשׂוּ־כֵן֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַיִּתֵּ֨ן לָהֶ֥ם יוֹסֵ֛ף עֲגָל֖וֹת עַל־פִּ֣י פַרְעֹ֑ה וַיִּתֵּ֥ן לָהֶ֛ם צֵדָ֖ה לַדָּֽרֶךְ׃ כָּךְ עָשׂוּ בְּנֵי יִשְׂרָאֵל, יוֹסֵף נָתַן לָהֶם עֲגָלוֹת עַל פִּי פְּקֻדַּת פַּרְעֹה, וְנָתַן לָהֶם צֵדָה לַדֶּרֶךְ.
כב לְכֻלָּ֥ם נָתַ֛ן לָאִ֖ישׁ חֲלִפ֣וֹת שְׂמָלֹ֑ת וּלְבִנְיָמִ֤ן נָתַן֙ שְׁלֹ֣שׁ מֵא֣וֹת כֶּ֔סֶף וְחָמֵ֖שׁ חֲלִפֹ֥ת שְׂמָלֹֽת׃ לְכֻלָּם נָתַן לְכָל אִישׁ בְּגָדִים לְהַחְלָפָה וּלְבִנְיָמִין נָתַן כֶּסֶף בְּמִשְׁקַל שְׁלוֹשׁ מֵאוֹת שְׁקָלִים (4.230 קִילוֹ) וְחָמֵשׁ בְּגָדִים לְהַחְלָפָה.
כג וּלְאָבִ֞יו שָׁלַ֤ח כְּזֹאת֙ עֲשָׂרָ֣ה חֲמֹרִ֔ים נֹשְׂאִ֖ים מִטּ֣וּב מִצְרָ֑יִם וְעֶ֣שֶׂר אֲתֹנֹ֡ת נֹֽ֠שְׂאֹ֠ת בָּ֣ר וָלֶ֧חֶם וּמָז֛וֹן לְאָבִ֖יו לַדָּֽרֶךְ׃ וּלְאָבִיו שָׁלַח מִטּוּב מִצְרַיִם כַּמּוּת שֶׁעֲשָׂרָה חֲמוֹרִים נוֹשְׂאִים, וּלְאָבִיו לַדֶּרֶךְ – כַּמּוּת שֶׁל תְּבוּאָה וְלֶחֶם וּמָזוֹן שֶׁעֶשֶׂר אֲתוֹנוֹת נוֹשְׂאוֹת.
כד וַיְשַׁלַּ֥ח אֶת־אֶחָ֖יו וַיֵּלֵ֑כוּ וַיֹּ֣אמֶר אֲלֵהֶ֔ם אַֽל־תִּרְגְּז֖וּ בַּדָּֽרֶךְ׃ הוּא שָׁלַח אֶת אֶחָיו וְהֵם הָלְכוּ, אָמַר לָהֶם: אַל תָּרִיבוּ בַּדֶּרֶךְ!
כה וַֽיַּעֲל֖וּ מִמִּצְרָ֑יִם וַיָּבֹ֙אוּ֙ אֶ֣רֶץ כְּנַ֔עַן אֶֽל־יַעֲקֹ֖ב אֲבִיהֶֽם׃ הֵם עָלוּ מִמִּצְרַיִם וּבָאוּ לְאֶרֶץ כְּנַעַן לְיַעֲקֹב אֲבִיהֶם.
כו וַיַּגִּ֨דוּ ל֜וֹ לֵאמֹ֗ר ע֚וֹד יוֹסֵ֣ף חַ֔י וְכִֽי־ה֥וּא מֹשֵׁ֖ל בְּכׇל־אֶ֣רֶץ מִצְרָ֑יִם וַיָּ֣פׇג לִבּ֔וֹ כִּ֥י לֹא־הֶאֱמִ֖ין לָהֶֽם׃ סִפְּרוּ לוֹ: יוֹסֵף עֲדַיִן חַי, וְהוּא מוֹשֵׁל בְּכָל אֶרֶץ מִצְרַיִם! פְּעִימַת לִבּוֹ פָּסְקָה כִּי הוּא לֹא הֶאֱמִין לָהֶם.
כז וַיְדַבְּר֣וּ אֵלָ֗יו אֵ֣ת כׇּל־דִּבְרֵ֤י יוֹסֵף֙ אֲשֶׁ֣ר דִּבֶּ֣ר אֲלֵהֶ֔ם וַיַּרְא֙ אֶת־הָ֣עֲגָל֔וֹת אֲשֶׁר־שָׁלַ֥ח יוֹסֵ֖ף לָשֵׂ֣את אֹת֑וֹ וַתְּחִ֕י ר֖וּחַ יַעֲקֹ֥ב אֲבִיהֶֽם׃ הֵם אָמְרוּ לוֹ אֶת כָּל דִּבְרֵי יוֹסֵף שֶׁהוּא דִּבֵּר אִתָּם, הוּא רָאָה אֶת הָעֲגָלוֹת שֶׁיּוֹסֵף שָׁלַח כְּדֵי לְהָבִיא אוֹתוֹ, וּמַצָּב רוּחוֹ שֶׁל יַעֲקֹב אֲבִיהֶם קָם לִתְחִיָּה.

חמישי[עריכה]

כח וַיֹּ֙אמֶר֙ יִשְׂרָאֵ֔ל רַ֛ב עוֹד־יוֹסֵ֥ף בְּנִ֖י חָ֑י אֵֽלְכָ֥ה וְאֶרְאֶ֖נּוּ בְּטֶ֥רֶם אָמֽוּת׃ יִשְׂרָאֵל אָמַר: שִׂמְחָה גְּדוֹלָה! – יוֹסֵף בְּנִי עֲדַיִן חַי! אֵלֵךְ וְאֶרְאֶה אוֹתוֹ לִפְנֵי שֶׁאָמוּת.
פָּרָשַׁת יְרִידַת יִשְׂרָאֵל
מו א וַיִּסַּ֤ע יִשְׂרָאֵל֙ וְכׇל־אֲשֶׁר־ל֔וֹ וַיָּבֹ֖א בְּאֵ֣רָה שָּׁ֑בַע וַיִּזְבַּ֣ח זְבָחִ֔ים לֵאלֹהֵ֖י אָבִ֥יו יִצְחָֽק׃ יִשְׂרָאֵל נָסַע עִם כָּל מַה שֶּׁשַּׁיָּךְ לוֹ וּבָא לִבְאֵר שֶׁבַע, וְהִקְרִיב זְבָחִים לֵאלֹהֵי אָבִיו יִצְחָק.
ב וַיֹּ֨אמֶר אֱלֹהִ֤ים ׀ לְיִשְׂרָאֵל֙ בְּמַרְאֹ֣ת הַלַּ֔יְלָה וַיֹּ֖אמֶר יַעֲקֹ֣ב ׀ יַעֲקֹ֑ב וַיֹּ֖אמֶר הִנֵּֽנִי׃ אֱלֹהִים אָמַר לְיִשְׂרָאֵל בִּנְבוּאָה בַּחֲלוֹם הַלַּיְלָה – אָמַר: יַעֲקֹב יַעֲקֹב! אָמַר לוֹ: הִנֵּה אֲנִי!
ג וַיֹּ֕אמֶר אָנֹכִ֥י הָאֵ֖ל אֱלֹהֵ֣י אָבִ֑יךָ אַל־תִּירָא֙ מֵרְדָ֣ה מִצְרַ֔יְמָה כִּֽי־לְג֥וֹי גָּד֖וֹל אֲשִֽׂימְךָ֥ שָֽׁם׃ אָמַר: אֲנִי הָאֵל – אֱלֹהֵי אָבִיךָ, אַל תִּפְחַד מִלָּרֶדֶת לְמִצְרַיִם, כִּי שָׁם אֶעֱשֶׂה מִמְּךָ עָם גָּדוֹל.
ד אָנֹכִ֗י אֵרֵ֤ד עִמְּךָ֙ מִצְרַ֔יְמָה וְאָנֹכִ֖י אַֽעַלְךָ֣ גַם־עָלֹ֑ה וְיוֹסֵ֕ף יָשִׁ֥ית יָד֖וֹ עַל־עֵינֶֽיךָ׃ אֲנִי אֵרֵד אִתְּךָ לְמִצְרַיִם וַאֲנִי גַּם אַעֲלֶה אוֹתְךָ מִשָּׁם, וְיוֹסֵף יִדְאַג לְכָל צְרָכֶיךָ.
ה וַיָּ֥קׇם יַעֲקֹ֖ב מִבְּאֵ֣ר שָׁ֑בַע וַיִּשְׂא֨וּ בְנֵֽי־יִשְׂרָאֵ֜ל אֶת־יַעֲקֹ֣ב אֲבִיהֶ֗ם וְאֶת־טַפָּם֙ וְאֶת־נְשֵׁיהֶ֔ם בָּעֲגָל֕וֹת אֲשֶׁר־שָׁלַ֥ח פַּרְעֹ֖ה לָשֵׂ֥את אֹתֽוֹ׃ יַעֲקֹב קָם מִבְּאֵר שֶׁבַע, וּבְנֵי יִשְׂרָאֵל לָקְחוּ אֶת יַעֲקֹב אֲבִיהֶם וְאֶת יַלְדֵיהֶם וְאֶת נְשֵׁיהֶם בָּעֲגָלוֹת שֶׁפַּרְעֹה שָׁלַח כְּדֵי לָקַחַת אוֹתוֹ.
ו וַיִּקְח֣וּ אֶת־מִקְנֵיהֶ֗ם וְאֶת־רְכוּשָׁם֙ אֲשֶׁ֤ר רָֽכְשׁוּ֙ בְּאֶ֣רֶץ כְּנַ֔עַן וַיָּבֹ֖אוּ מִצְרָ֑יְמָה יַעֲקֹ֖ב וְכׇל־זַרְע֥וֹ אִתּֽוֹ׃ הֵם לָקְחוּ אֶת עֶדְרֵיהֶם וְאֶת רְכוּשָׁם שֶׁרָכְשׁוּ בְּאֶרֶץ כְּנַעַן וּבָאוּ לְמִצְרַיִם – יַעֲקֹב וְכָל צֶאֱצָאָיו אִתּוֹ.
ז בָּנָ֞יו וּבְנֵ֤י בָנָיו֙ אִתּ֔וֹ בְּנֹתָ֛יו וּבְנ֥וֹת בָּנָ֖יו וְכׇל־זַרְע֑וֹ הֵבִ֥יא אִתּ֖וֹ מִצְרָֽיְמָה׃ אֶת בָּנָיו וּבְנֵי בָּנָיו אִתּוֹ בְּנוֹתָיו וּבְנוֹת בָּנָיו וְכָל צֶאֱצָאָיו, הֵבִיא אִתּוֹ לְמִצְרַיִם.
ס פָּרָשַׁת שְׁמוֹת בְּנֵי יִשְׂרָאֵל
ח וְאֵ֨לֶּה שְׁמ֧וֹת בְּנֵֽי־יִשְׂרָאֵ֛ל הַבָּאִ֥ים מִצְרַ֖יְמָה יַעֲקֹ֣ב וּבָנָ֑יו בְּכֹ֥ר יַעֲקֹ֖ב רְאוּבֵֽן׃ וְאֵלֶּה הַשֵּׁמוֹת שֶׁל בְּנֵי יִשְׂרָאֵל שֶׁבָּאִים לְמִצְרַיִם, יַעֲקֹב וּבָנָיו: בְּכוֹר יַעֲקֹב רְאוּבֵן.
ט וּבְנֵ֖י רְאוּבֵ֑ן חֲנ֥וֹךְ וּפַלּ֖וּא וְחֶצְרֹ֥ן וְכַרְמִֽי׃ וּבְנֵי רְאוּבֵן: חֲנוֹךְ וּפַלּוּא וְחֶצְרוֹן וְכַרְמִי.
י וּבְנֵ֣י שִׁמְע֗וֹן יְמוּאֵ֧ל וְיָמִ֛ין וְאֹ֖הַד וְיָכִ֣ין וְצֹ֑חַר וְשָׁא֖וּל בֶּן־הַֽכְּנַעֲנִֽית׃ וּבְנֵי שִׁמְעוֹן: יְמוּאֵל וְיָמִין וְאוֹהַד וְיָכִין וְצוֹחַר וְשָׁאוּל שֶׁהָיָה בֶּן אִשָּׁה כְּנַעֲנִית.
יא וּבְנֵ֖י לֵוִ֑י גֵּרְשׁ֕וֹן קְהָ֖ת וּמְרָרִֽי׃ וּבְנֵי לֵוִי: גֵּרְשׁוֹן קְהָת וּמְרָרִי.
יב וּבְנֵ֣י יְהוּדָ֗ה עֵ֧ר וְאוֹנָ֛ן וְשֵׁלָ֖ה וָפֶ֣רֶץ וָזָ֑רַח וַיָּ֨מׇת עֵ֤ר וְאוֹנָן֙ בְּאֶ֣רֶץ כְּנַ֔עַן וַיִּהְי֥וּ בְנֵי־פֶ֖רֶץ חֶצְרֹ֥ן וְחָמֽוּל׃ וּבְנֵי יְהוּדָה: עֵר וְאוֹנָן וְשֵׁלָה וּפֶרֶץ וְזֶרַח, עֵר וְאוֹנָן מֵתוּ בְּאֶרֶץ כְּנַעַן, בְּנֵי פֶּרֶץ הָיוּ חֶצְרוֹן וְחָמוּל.
יג וּבְנֵ֖י יִשָּׂשכָ֑ר תּוֹלָ֥ע וּפֻוָ֖ה וְי֥וֹב וְשִׁמְרֹֽן׃ וּבְנֵי יִשָּׂשכָר: תּוֹלָע וּפֻוָה וְיוֹב וְשִׁמְרוֹן.
יד וּבְנֵ֖י זְבֻל֑וּן סֶ֥רֶד וְאֵל֖וֹן וְיַחְלְאֵֽל׃ וּבְנֵי זְבוּלוּן: סֶרֶד וְאֵלוֹן וְיַחְלְאֵל.
טו אֵ֣לֶּה ׀ בְּנֵ֣י לֵאָ֗ה אֲשֶׁ֨ר יָֽלְדָ֤ה לְיַעֲקֹב֙ בְּפַדַּ֣ן אֲרָ֔ם וְאֵ֖ת דִּינָ֣ה בִתּ֑וֹ כׇּל־נֶ֧פֶשׁ בָּנָ֛יו וּבְנוֹתָ֖יו שְׁלֹשִׁ֥ים וְשָׁלֹֽשׁ׃ אֵלֶּה בְּנֵי לֵאָה שֶׁיָּלְדָה לְיַעֲקֹב בְּפַדַּן אֲרָם, וְיַחַד עִם דִּינָה בִּתּוֹ, מִסְפַּר נַפְשׁוֹת בָּנָיו וּבְנוֹתָיו שְׁלוֹשִׁים וְשָׁלוֹשׁ.
טז וּבְנֵ֣י גָ֔ד צִפְי֥וֹן וְחַגִּ֖י שׁוּנִ֣י וְאֶצְבֹּ֑ן עֵרִ֥י וַֽאֲרוֹדִ֖י וְאַרְאֵלִֽי׃ וּבְנֵי גָּד: צִפְיוֹן וְחַגִּי שׁוּנִי וְאֶצְבּוֹן עֵרִי וַאֲרוֹדִי וְאַרְאֵלִי.
יז וּבְנֵ֣י אָשֵׁ֗ר יִמְנָ֧ה וְיִשְׁוָ֛ה וְיִשְׁוִ֥י וּבְרִיעָ֖ה וְשֶׂ֣רַח אֲחֹתָ֑ם וּבְנֵ֣י בְרִיעָ֔ה חֶ֖בֶר וּמַלְכִּיאֵֽל׃ וּבְנֵי אָשֵׁר: יִמְנָה וְיִשְׁוָה וְיִשְׁוִי וּבְרִיעָה וְשֶׂרַח אֲחוֹתָם וּבְנֵי בְּרִיעָה חֶבֶר וּמַלְכִּיאֵל.
יח אֵ֚לֶּה בְּנֵ֣י זִלְפָּ֔ה אֲשֶׁר־נָתַ֥ן לָבָ֖ן לְלֵאָ֣ה בִתּ֑וֹ וַתֵּ֤לֶד אֶת־אֵ֙לֶּה֙ לְיַעֲקֹ֔ב שֵׁ֥שׁ עֶשְׂרֵ֖ה נָֽפֶשׁ׃ אֵלֶּה בְּנֵי זִלְפָּה שֶׁנָּתַן לָבָן לְלֵאָה בִּתּוֹ, הִיא יָלְדָה אֶת אֵלֶּה לְיַעֲקֹב, שֵׁשׁ עֶשְׂרֵה נֶפֶשׁ.
יט בְּנֵ֤י רָחֵל֙ אֵ֣שֶׁת יַעֲקֹ֔ב יוֹסֵ֖ף וּבִנְיָמִֽן׃ בְּנֵי רָחֵל אֵשֶׁת יַעֲקֹב: יוֹסֵף וּבִנְיָמִין.
כ וַיִּוָּלֵ֣ד לְיוֹסֵף֮ בְּאֶ֣רֶץ מִצְרַ֒יִם֒ אֲשֶׁ֤ר יָֽלְדָה־לּוֹ֙ אָֽסְנַ֔ת בַּת־פּ֥וֹטִי פֶ֖רַע‏[1] כֹּהֵ֣ן אֹ֑ן אֶת־מְנַשֶּׁ֖ה וְאֶת־אֶפְרָֽיִם׃ לְיוֹסֵף נוֹלְדוּ מְנַשֶּׁה וְאֶפְרַיִם בְּאֶרֶץ מִצְרַיִם שֶׁיָּלְדָה לוֹ אָסְנַת בַּת פּוֹטִי-פֶרַע הַכֹּהֵן שֶׁל הָעִיר אוֹן.
כא וּבְנֵ֣י בִנְיָמִ֗ן בֶּ֤לַע וָבֶ֙כֶר֙ וְאַשְׁבֵּ֔ל גֵּרָ֥א וְנַעֲמָ֖ן אֵחִ֣י וָרֹ֑אשׁ מֻפִּ֥ים וְחֻפִּ֖ים וָאָֽרְדְּ׃ וּבְנֵי בִּנְיָמִין: בֶּלַע וּבֶכֶר וְאַשְׁבֵּל גֵּרָא וְנַעֲמָן אֵחִי וְרֹאשׁ מֻפִּים וְחֻפִּים וְאָרְדְּ.
כב אֵ֚לֶּה בְּנֵ֣י רָחֵ֔ל אֲשֶׁ֥ר יֻלַּ֖ד לְיַעֲקֹ֑ב כׇּל־נֶ֖פֶשׁ אַרְבָּעָ֥ה עָשָֽׂר׃ אֵלֶּה בְּנֵי רָחֵל שֶׁנּוֹלְדוּ לְיַעֲקֹב, מִסְפַּר הַנְּפָשׁוֹת אַרְבָּעָה עָשָׂר.
כג וּבְנֵי־דָ֖ן חֻשִֽׁים׃ וּבְנֵי דָּן: חֻשִׁים.
כד וּבְנֵ֖י נַפְתָּלִ֑י יַחְצְאֵ֥ל וְגוּנִ֖י וְיֵ֥צֶר וְשִׁלֵּֽם׃ וּבְנֵי נַפְתָּלִי: יַחְצְאֵל וְגוּנִי וְיֵצֶר וְשִׁלֵּם.
כה אֵ֚לֶּה בְּנֵ֣י בִלְהָ֔ה אֲשֶׁר־נָתַ֥ן לָבָ֖ן לְרָחֵ֣ל בִּתּ֑וֹ וַתֵּ֧לֶד אֶת־אֵ֛לֶּה לְיַעֲקֹ֖ב כׇּל־נֶ֥פֶשׁ שִׁבְעָֽה׃ אֵלֶּה בְּנֵי בִּלְהָה שֶׁנָּתַן לָבָן לְרָחֵל בִּתּוֹ, הִיא יָלְדָה אֶת אֵלֶּה לְיַעֲקֹב, מִסְפַּר הַנְּפָשׁוֹת – שִׁבְעָה.
כו כׇּל־הַ֠נֶּ֠פֶשׁ הַבָּאָ֨ה לְיַעֲקֹ֤ב מִצְרַ֙יְמָה֙ יֹצְאֵ֣י יְרֵכ֔וֹ מִלְּבַ֖ד נְשֵׁ֣י בְנֵי־יַעֲקֹ֑ב כׇּל־נֶ֖פֶשׁ שִׁשִּׁ֥ים וָשֵֽׁשׁ׃ כָּל הַנְּפָשׁוֹת שֶׁל יַעֲקֹב שֶׁבָּאִים לְמִצְרַיִם, צֶאֱצָאָיו, חוּץ מֵהַנָשִׁים שֶׁל בְּנֵי יַעֲקֹב – מִסְפַּר הַנְּפָשׁוֹת – שִׁשִּׁים וָשֵׁשׁ.
כז וּבְנֵ֥י יוֹסֵ֛ף אֲשֶׁר־יֻלַּד־ל֥וֹ בְמִצְרַ֖יִם נֶ֣פֶשׁ שְׁנָ֑יִם כׇּל־הַנֶּ֧פֶשׁ לְבֵֽית־יַעֲקֹ֛ב הַבָּ֥אָה מִצְרַ֖יְמָה שִׁבְעִֽים׃ וּבְנֵי יוֹסֵף שֶׁנּוֹלְדוּ לוֹ בְּמִצְרַיִם – שְׁתֵּי נְפָשׁוֹת, מִסְפַּר נְפָשׁוֹת מִשְׁפַּחַת יַעֲקֹב שֶׁבָּאוּ לְמִצְרַיִם – שִׁבְעִים.

ששי[עריכה]

ס פָּרָשַׁת בִּיאַת יַעֲקֹב לְמִצְרַיִם
כח וְאֶת־יְהוּדָ֞ה שָׁלַ֤ח לְפָנָיו֙ אֶל־יוֹסֵ֔ף לְהוֹרֹ֥ת לְפָנָ֖יו גֹּ֑שְׁנָה וַיָּבֹ֖אוּ אַ֥רְצָה גֹּֽשֶׁן׃ וְיַעֲקֹב שָׁלַח לְפָנָיו אֶת יְהוּדָה אֶל יוֹסֵף לְהַרְאוֹת לְפָנָיו אֶת הַדֶּרֶךְ לְגוֹשֶׁן, וְהֵם בָּאוּ לְאֶרֶץ גּוֹשֶׁן.
כט וַיֶּאְסֹ֤ר יוֹסֵף֙ מֶרְכַּבְתּ֔וֹ וַיַּ֛עַל לִקְרַֽאת־יִשְׂרָאֵ֥ל אָבִ֖יו גֹּ֑שְׁנָה וַיֵּרָ֣א אֵלָ֗יו וַיִּפֹּל֙ עַל־צַוָּארָ֔יו וַיֵּ֥בְךְּ עַל־צַוָּארָ֖יו עֽוֹד׃ יוֹסֵף קָשַׁר אֶת הַסּוּסִים לְמֶרְכַּבְתּוֹ וְעָלָה לִקְרַאת יִשְׂרָאֵל אָבִיו לְאֶרֶץ גּוֹשֶׁן, הוּא הוֹפִיעַ לְפָנָיו, נָפַל עַל צַוָּארוֹ וּבָכָה עוֹד עַל צַוָּארוֹ.
ל וַיֹּ֧אמֶר יִשְׂרָאֵ֛ל אֶל־יוֹסֵ֖ף אָמ֣וּתָה הַפָּ֑עַם אַחֲרֵי֙ רְאוֹתִ֣י אֶת־פָּנֶ֔יךָ כִּ֥י עוֹדְךָ֖ חָֽי׃ יִשְׂרָאֵל אָמַר לְיוֹסֵף: עַכְשָׁיו אֲנִי מוּכָן לָמוּת אַחֲרֵי שֶׁרָאִיתִי אֶת פָּנֶיךָ שֶׁאַתָּה עֲדַיִן חַי!
לא וַיֹּ֨אמֶר יוֹסֵ֤ף אֶל־אֶחָיו֙ וְאֶל־בֵּ֣ית אָבִ֔יו אֶעֱלֶ֖ה וְאַגִּ֣ידָה לְפַרְעֹ֑ה וְאֹֽמְרָ֣ה אֵלָ֔יו אַחַ֧י וּבֵית־אָבִ֛י אֲשֶׁ֥ר בְּאֶֽרֶץ־כְּנַ֖עַן בָּ֥אוּ אֵלָֽי׃ יוֹסֵף אָמַר לְאֶחָיו וּלְמִשְׁפַּחַת אָבִיו: אֶעֱלֶה וְאַגִּיד לְפַרְעֹה וְאוֹמַר לוֹ שֶׁאַחַי וּמִשְׁפַּחַת אָבִי שֶׁבְּאֶרֶץ כְּנַעַן בָּאוּ אֵלַי.
לב וְהָאֲנָשִׁים֙ רֹ֣עֵי צֹ֔אן כִּֽי־אַנְשֵׁ֥י מִקְנֶ֖ה הָי֑וּ וְצֹאנָ֧ם וּבְקָרָ֛ם וְכׇל־אֲשֶׁ֥ר לָהֶ֖ם הֵבִֽיאוּ׃ וְשֶׁהָאֲנָשִׁים רוֹעֵי צֹאן כִּי הָיוּ בַּעֲלֵי עֲדָרִים, וְהֵם הֵבִיאוּ אֶת צֹאנָם וּבְקָרָם וְכָל מַה שֶּׁיֵּשׁ לָהֶם.
לג וְהָיָ֕ה כִּֽי־יִקְרָ֥א לָכֶ֖ם פַּרְעֹ֑ה וְאָמַ֖ר מַה־מַּעֲשֵׂיכֶֽם׃ וּכְשֶׁיִּקְרָא לָכֶם פַּרְעֹה, וְיֹאמַר: מַה מַּעֲשֵׂיכֶם?
לד וַאֲמַרְתֶּ֗ם אַנְשֵׁ֨י מִקְנֶ֜ה הָי֤וּ עֲבָדֶ֙יךָ֙ מִנְּעוּרֵ֣ינוּ וְעַד־עַ֔תָּה גַּם־אֲנַ֖חְנוּ גַּם־אֲבֹתֵ֑ינוּ בַּעֲב֗וּר תֵּשְׁבוּ֙ בְּאֶ֣רֶץ גֹּ֔שֶׁן כִּֽי־תוֹעֲבַ֥ת מִצְרַ֖יִם כׇּל־רֹ֥עֵה צֹֽאן׃ תֹּאמְרוּ: עֲבָדֶיךָ הָיוּ אֲנָשִׁים בַּעֲלֵי עֲדָרִים מִנְּעוּרֵינוּ וְעַד עַכְשָׁיו, גַּם אֲנַחְנוּ גַּם אֲבוֹתֵינוּ, אִמְרוּ כָּךְ כְּדֵי שֶׁתֵּשְׁבוּ בְּאֶרֶץ גּוֹשֶׁן, כִּי מִצְרַיִם שׂוֹנְאִים כָּל רוֹעֵה צֹאן.
מז א וַיָּבֹ֣א יוֹסֵף֮ וַיַּגֵּ֣ד לְפַרְעֹה֒ וַיֹּ֗אמֶר אָבִ֨י וְאַחַ֜י וְצֹאנָ֤ם וּבְקָרָם֙ וְכׇל־אֲשֶׁ֣ר לָהֶ֔ם בָּ֖אוּ מֵאֶ֣רֶץ כְּנָ֑עַן וְהִנָּ֖ם בְּאֶ֥רֶץ גֹּֽשֶׁן׃ יוֹסֵף בָּא וְסִפֵּר לְפַרְעֹה, וְאָמַר: אָבִי וְאַחַי וְצֹאנָם וּבְקָרָם וְכָל מַה שֶּׁשַּׁיָּךְ לָהֶם בָּאוּ מֵאֶרֶץ כְּנַעַן, וְהֵם בְּאֶרֶץ גּוֹשֶׁן.
ב וּמִקְצֵ֣ה אֶחָ֔יו לָקַ֖ח חֲמִשָּׁ֣ה אֲנָשִׁ֑ים וַיַּצִּגֵ֖ם לִפְנֵ֥י פַרְעֹֽה׃ וּמְעַט מֵאֶחָיו לָקַח – חֲמִשָּׁה אֲנָשִׁים, וְהֶעֱמִיד אוֹתָם לִפְנֵי פַּרְעֹה.
ג וַיֹּ֧אמֶר פַּרְעֹ֛ה אֶל־אֶחָ֖יו מַה־מַּעֲשֵׂיכֶ֑ם וַיֹּאמְר֣וּ אֶל־פַּרְעֹ֗ה רֹעֵ֥ה צֹאן֙ עֲבָדֶ֔יךָ גַּם־אֲנַ֖חְנוּ גַּם־אֲבוֹתֵֽינוּ׃ פַּרְעֹה אָמַר לְאֶחָיו: מַה מַּעֲשֵׂיכֶם? אָמְרוּ לְפַרְעֹה: עֲבָדֶיךָ רוֹעֵי צֹאן, גַּם אֲנַחְנוּ גַּם אֲבוֹתֵינוּ.
ד וַיֹּאמְר֣וּ אֶל־פַּרְעֹ֗ה לָג֣וּר בָּאָ֘רֶץ֮ בָּ֒אנוּ֒ כִּי־אֵ֣ין מִרְעֶ֗ה לַצֹּאן֙ אֲשֶׁ֣ר לַעֲבָדֶ֔יךָ כִּֽי־כָבֵ֥ד הָרָעָ֖ב בְּאֶ֣רֶץ כְּנָ֑עַן וְעַתָּ֛ה יֵֽשְׁבוּ־נָ֥א עֲבָדֶ֖יךָ בְּאֶ֥רֶץ גֹּֽשֶׁן׃ הֵם אָמְרוּ לְפַרְעֹה: בָּאנוּ לָגוּר כְּתוֹשָׁבִים זְמַנִּיִּים בָּאָרֶץ, כִּי אֵין מִרְעֶה לַצֹּאן שֶׁשַּׁיָּךְ לַעֲבָדֶיךָ, כִּי הָרָעָב בְּאֶרֶץ כְּנַעַן קָשֶׁה, וְעַכְשָׁיו עֲבָדֶיךָ יִתְיַשְּׁבוּ בְּבַקָּשָׁה בְּאֶרֶץ גּוֹשֶׁן.
ה וַיֹּ֣אמֶר פַּרְעֹ֔ה אֶל־יוֹסֵ֖ף לֵאמֹ֑ר אָבִ֥יךָ וְאַחֶ֖יךָ בָּ֥אוּ אֵלֶֽיךָ׃ פַּרְעֹה אָמַר לְיוֹסֵף: אָבִיךָ וְאַחֶיךָ בָּאוּ אֵלֶיךָ.
ו אֶ֤רֶץ מִצְרַ֙יִם֙ לְפָנֶ֣יךָ הִ֔וא בְּמֵיטַ֣ב הָאָ֔רֶץ הוֹשֵׁ֥ב אֶת־אָבִ֖יךָ וְאֶת־אַחֶ֑יךָ יֵשְׁבוּ֙ בְּאֶ֣רֶץ גֹּ֔שֶׁן וְאִם־יָדַ֗עְתָּ וְיֶשׁ־בָּם֙ אַנְשֵׁי־חַ֔יִל וְשַׂמְתָּ֛ם שָׂרֵ֥י מִקְנֶ֖ה עַל־אֲשֶׁר־לִֽי׃ אֶרֶץ מִצְרַיִם לְפָנֶיךָ, הוֹשֵׁב אֶת אָבִיךָ וְאֶת אַחֶיךָ בָּאֵזוֹר הַטּוֹב בְּיוֹתֵר בָּאָרֶץ, יָגוּרוּ בְּאֶרֶץ גּוֹשֶׁן, וְאִם יָדוּעַ לְךָ שֶׁיֵּשׁ בָּהֶם אֲנָשִׁים מֻכְשָׁרִים תְּמַנֶּה אוֹתָם לְאַחְרָאִים עַל עֲדָרִים שֶׁלִּי.
ז וַיָּבֵ֤א יוֹסֵף֙ אֶת־יַֽעֲקֹ֣ב אָבִ֔יו וַיַּֽעֲמִדֵ֖הוּ לִפְנֵ֣י פַרְעֹ֑ה וַיְבָ֥רֶךְ יַעֲקֹ֖ב אֶת־פַּרְעֹֽה׃ יוֹסֵף הֵבִיא אֶת יַעֲקֹב אָבִיו וְהֶעֱמִיד אוֹתוֹ לִפְנֵי פַּרְעֹה, וְיַעֲקֹב בֵּרֵךְ אֶת פַּרְעֹה.
ח וַיֹּ֥אמֶר פַּרְעֹ֖ה אֶֽל־יַעֲקֹ֑ב כַּמָּ֕ה יְמֵ֖י שְׁנֵ֥י חַיֶּֽיךָ׃ פַּרְעֹה אָמַר לְיַעֲקֹב: בֵּן כַּמָּה שָׁנִים אַתָּה?
ט וַיֹּ֤אמֶר יַעֲקֹב֙ אֶל־פַּרְעֹ֔ה יְמֵי֙ שְׁנֵ֣י מְגוּרַ֔י שְׁלֹשִׁ֥ים וּמְאַ֖ת שָׁנָ֑ה מְעַ֣ט וְרָעִ֗ים הָיוּ֙ יְמֵי֙ שְׁנֵ֣י חַיַּ֔י וְלֹ֣א הִשִּׂ֗יגוּ אֶת־יְמֵי֙ שְׁנֵי֙ חַיֵּ֣י אֲבֹתַ֔י בִּימֵ֖י מְגוּרֵיהֶֽם׃ יַעֲקֹב אָמַר לְפַרְעֹה: שְׁנוֹת חַיַּי מֵאָה וּשְׁלוֹשִׁים שָׁנִים וְהָיִיתִי בָּהֶן מְהַגֵּר, יְמֵי שְׁנוֹת חַיַּי הָיוּ מְעַטִּים וְרָעִים, וְלֹא הִגִּיעוּ לִימֵי שְׁנוֹת חַיֵּי אֲבוֹתַי שֶׁהָיוּ מְהַגְּרִים.
י וַיְבָ֥רֶךְ יַעֲקֹ֖ב אֶת־פַּרְעֹ֑ה וַיֵּצֵ֖א מִלִּפְנֵ֥י פַרְעֹֽה׃ יַעֲקֹב בֵּרֵךְ אֶת פַּרְעֹה, וְיָצָא מִלִּפְנֵי פַּרְעֹה.

שביעי[עריכה]

יא וַיּוֹשֵׁ֣ב יוֹסֵף֮ אֶת־אָבִ֣יו וְאֶת־אֶחָיו֒ וַיִּתֵּ֨ן לָהֶ֤ם אֲחֻזָּה֙ בְּאֶ֣רֶץ מִצְרַ֔יִם בְּמֵיטַ֥ב הָאָ֖רֶץ בְּאֶ֣רֶץ רַעְמְסֵ֑ס כַּאֲשֶׁ֖ר צִוָּ֥ה פַרְעֹֽה׃ יוֹסֵף הוֹשִׁיב אֶת אָבִיו וְאֶת אֶחָיו, וְנָתַן לָהֶם אֲחֻזָּה בְּאֶרֶץ מִצְרַיִם בָּאֵזוֹר הַטּוֹב בְּיוֹתֵר בָּאָרֶץ שֶׁבְּאֶרֶץ רַעְמְסֵס, כְּמוֹ שֶׁפַּרְעֹה צִוָּה.
יב וַיְכַלְכֵּ֤ל יוֹסֵף֙ אֶת־אָבִ֣יו וְאֶת־אֶחָ֔יו וְאֵ֖ת כׇּל־בֵּ֣ית אָבִ֑יו לֶ֖חֶם לְפִ֥י הַטָּֽף׃ יוֹסֵף פִּרְנֵס אֶת אָבִיו וְאֶת אֶחָיו וְאֶת כָּל מִשְׁפַּחַת אָבִיו, וְנָתַן לֶחֶם כְּפִי צוֹרֶךְ הַיְּלָדִים.
פָּרָשַׁת קְנִיַּת מִצְרַיִם
יג וְלֶ֤חֶם אֵין֙ בְּכׇל־הָאָ֔רֶץ כִּֽי־כָבֵ֥ד הָרָעָ֖ב מְאֹ֑ד וַתֵּ֜לַהּ אֶ֤רֶץ מִצְרַ֙יִם֙ וְאֶ֣רֶץ כְּנַ֔עַן מִפְּנֵ֖י הָרָעָֽב׃ לֹא הָיָה לֶחֶם בְּכָל הָאֵזוֹר כִּי הָרָעָב קָשֶׁה מְאוֹד, תּוֹשְׁבֵי אֶרֶץ מִצְרַיִם וְאֶרֶץ כְּנַעַן נֶחְלְשׁוּ בִּגְלַל הָרָעָב.
יד וַיְלַקֵּ֣ט יוֹסֵ֗ף אֶת־כׇּל־הַכֶּ֙סֶף֙ הַנִּמְצָ֤א בְאֶֽרֶץ־מִצְרַ֙יִם֙ וּבְאֶ֣רֶץ כְּנַ֔עַן בַּשֶּׁ֖בֶר אֲשֶׁר־הֵ֣ם שֹׁבְרִ֑ים וַיָּבֵ֥א יוֹסֵ֛ף אֶת־הַכֶּ֖סֶף בֵּ֥יתָה פַרְעֹֽה׃ יוֹסֵף אָסַף אֶת כָּל הַכֶּסֶף שֶׁנִּמְצָא בְּאֶרֶץ מִצְרַיִם וּבְאֶרֶץ כְּנַעַן עַל יְדֵי מְכִירַת הָאוֹכֶל שֶׁהֵם קוֹנִים, וְיוֹסֵף הֵבִיא אֶת הַכֶּסֶף לְבֵית פַּרְעֹה.
טו וַיִּתֹּ֣ם הַכֶּ֗סֶף מֵאֶ֣רֶץ מִצְרַ֘יִם֮ וּמֵאֶ֣רֶץ כְּנַ֒עַן֒ וַיָּבֹ֩אוּ֩ כׇל־מִצְרַ֨יִם אֶל־יוֹסֵ֤ף לֵאמֹר֙ הָֽבָה־לָּ֣נוּ לֶ֔חֶם וְלָ֥מָּה נָמ֖וּת נֶגְדֶּ֑ךָ כִּ֥י אָפֵ֖ס כָּֽסֶף׃ הַכֶּסֶף נִגְמַר בְּאֶרֶץ מִצְרַיִם וּבְאֶרֶץ כְּנַעַן, וְכָל תּוֹשְׁבֵי מִצְרַיִם בָּאוּ אֶל יוֹסֵף וְאָמְרוּ לוֹ: תֵּן לָנוּ לֶחֶם, וְלָמָּה נָמוּת מוּלְךָ, כִּי נִגְמַר הַכֶּסֶף!
טז וַיֹּ֤אמֶר יוֹסֵף֙ הָב֣וּ מִקְנֵיכֶ֔ם וְאֶתְּנָ֥ה לָכֶ֖ם בְּמִקְנֵיכֶ֑ם אִם־אָפֵ֖ס כָּֽסֶף׃ יוֹסֵף אָמַר: תְּנוּ אֶת עֶדְרֵיכֶם, וְאֶתֵּן לָכֶם תְּמוּרַת עֶדְרֵיכֶם, אִם נִגְמַר הַכֶּסֶף.
יז וַיָּבִ֣יאוּ אֶת־מִקְנֵיהֶם֮ אֶל־יוֹסֵף֒ וַיִּתֵּ֣ן לָהֶם֩ יוֹסֵ֨ף לֶ֜חֶם בַּסּוּסִ֗ים וּבְמִקְנֵ֥ה הַצֹּ֛אן וּבְמִקְנֵ֥ה הַבָּקָ֖ר וּבַחֲמֹרִ֑ים וַיְנַהֲלֵ֤ם בַּלֶּ֙חֶם֙ בְּכׇל־מִקְנֵהֶ֔ם בַּשָּׁנָ֖ה הַהִֽוא׃ הֵם הֵבִיאוּ אֶת עֶדְרֵיהֶם אֶל יוֹסֵף, וְיוֹסֵף נָתַן לָהֶם לֶחֶם תְּמוּרַת הַסּוּסִים וְעֶדְרֵי הַצֹּאן וְעֶדְרֵי הַבָּקָר וְהַחֲמוֹרִים, וּפִרְנֵס אוֹתָם בְּלֶחֶם תְּמוּרַת כָּל עֶדְרֵיהֶם בְּאוֹתָהּ שָׁנָה.
יח וַתִּתֹּם֮ הַשָּׁנָ֣ה הַהִוא֒ וַיָּבֹ֨אוּ אֵלָ֜יו בַּשָּׁנָ֣ה הַשֵּׁנִ֗ית וַיֹּ֤אמְרוּ לוֹ֙ לֹֽא־נְכַחֵ֣ד מֵֽאֲדֹנִ֔י כִּ֚י אִם־תַּ֣ם הַכֶּ֔סֶף וּמִקְנֵ֥ה הַבְּהֵמָ֖ה אֶל־אֲדֹנִ֑י לֹ֤א נִשְׁאַר֙ לִפְנֵ֣י אֲדֹנִ֔י בִּלְתִּ֥י אִם־גְּוִיָּתֵ֖נוּ וְאַדְמָתֵֽנוּ׃ הַשָּׁנָה הַהִיא נִגְמְרָה, הֵם בָּאוּ אֵלָיו בַּשָּׁנָה הַשְּׁנִיָּה וְאָמְרוּ לוֹ: לֹא נַסְתִּיר מֵאֲדוֹנִי אֶלָּא נְגַלֶּה שֶׁנִּגְמְרוּ הַכֶּסֶף וְעֶדְרֵי הַבְּהֵמָה מִלָּתֵת לַאֲדוֹנִי, לֹא נִשְׁאַר לִפְנֵי אֲדוֹנִי אֶלָּא רַק גּוּפֵנוּ וְאַדְמָתֵנוּ.
יט לָ֧מָּה נָמ֣וּת לְעֵינֶ֗יךָ גַּם־אֲנַ֙חְנוּ֙ גַּ֣ם אַדְמָתֵ֔נוּ קְנֵֽה־אֹתָ֥נוּ וְאֶת־אַדְמָתֵ֖נוּ בַּלָּ֑חֶם וְנִֽהְיֶ֞ה אֲנַ֤חְנוּ וְאַדְמָתֵ֙נוּ֙ עֲבָדִ֣ים לְפַרְעֹ֔ה וְתֶן־זֶ֗רַע וְנִֽחְיֶה֙ וְלֹ֣א נָמ֔וּת וְהָאֲדָמָ֖ה לֹ֥א תֵשָֽׁם׃ לָמָּה נָמוּת לְעֵינֶיךָ גַּם אֲנַחְנוּ גַּם אַדְמָתֵנוּ? קְנֵה אוֹתָנוּ וְאֶת אַדְמָתֵנוּ תְּמוּרַת הַלֶּחֶם, וְנִהְיֶה אֲנַחְנוּ וְאַדְמָתֵנוּ עֲבָדִים לְפַרְעֹה, וְתֵן לָנוּ גַּם זֶרַע לִזְרוֹעַ, וְנִחְיֶה וְלֹא נָמוּת וְהָאֲדָמָה לֹא תִּהְיֶה שְׁמָמָה!
כ וַיִּ֨קֶן יוֹסֵ֜ף אֶת־כׇּל־אַדְמַ֤ת מִצְרַ֙יִם֙ לְפַרְעֹ֔ה כִּֽי־מָכְר֤וּ מִצְרַ֙יִם֙ אִ֣ישׁ שָׂדֵ֔הוּ כִּֽי־חָזַ֥ק עֲלֵהֶ֖ם הָרָעָ֑ב וַתְּהִ֥י הָאָ֖רֶץ לְפַרְעֹֽה׃ יוֹסֵף קָנָה אֶת כָּל אַדְמַת מִצְרַיִם לְפַרְעֹה, כִּי תּוֹשְׁבֵי מִצְרַיִם מָכְרוּ כָּל אֶחָד אֶת שָׂדֵהוּ, כִּי הָרָעָב הָיָה לָהֶם קָשֶׁה, וְהָאָרֶץ נִהְיְתָה שַׁיֶּכֶת לְפַרְעֹה.
כא וְאֶ֨ת־הָעָ֔ם הֶעֱבִ֥יר אֹת֖וֹ לֶעָרִ֑ים מִקְצֵ֥ה גְבוּל־מִצְרַ֖יִם וְעַד־קָצֵֽהוּ׃ כְּדֵי לְהַמְחִישׁ זֹאת – הוּא הֶעֱבִיר אֶת הָעָם לְעָרִים אֲחֵרוֹת, מִקְצֵה גְּבוּל מִצְרַיִם וְעַד הַקָּצֶה הַשֵּׁנִי.
כב רַ֛ק אַדְמַ֥ת הַכֹּהֲנִ֖ים לֹ֣א קָנָ֑ה כִּי֩ חֹ֨ק לַכֹּהֲנִ֜ים מֵאֵ֣ת פַּרְעֹ֗ה וְאָֽכְל֤וּ אֶת־חֻקָּם֙ אֲשֶׁ֨ר נָתַ֤ן לָהֶם֙ פַּרְעֹ֔ה עַל־כֵּ֕ן לֹ֥א מָכְר֖וּ אֶת־אַדְמָתָֽם׃ רַק אֶת אַדְמַת הַכְּמָרִים לֹא קָנָה, כִּי הָיְתָה קִצְבָּה לַכְּמָרִים מִפַּרְעֹה, וְהֵם אָכְלוּ אֶת הַקִצְבָּה שֶׁלָּהֶם שֶׁפַּרְעֹה נָתַן לָהֶם, לָכֵן לֹא מָכְרוּ אֶת אַדְמָתָם.
כג וַיֹּ֤אמֶר יוֹסֵף֙ אֶל־הָעָ֔ם הֵן֩ קָנִ֨יתִי אֶתְכֶ֥ם הַיּ֛וֹם וְאֶת־אַדְמַתְכֶ֖ם לְפַרְעֹ֑ה הֵֽא־לָכֶ֣ם זֶ֔רַע וּזְרַעְתֶּ֖ם אֶת־הָאֲדָמָֽה׃ יוֹסֵף אָמַר לָעָם: הֲרֵי קָנִיתִי הַיּוֹם אֶתְכֶם וְאֶת אַדְמַתְכֶם לְפַרְעֹה, הִנֵּה לָכֶם זֶרַע וְתִזְרְעוּ אֶת הָאֲדָמָה.
כד וְהָיָה֙ בַּתְּבוּאֹ֔ת וּנְתַתֶּ֥ם חֲמִישִׁ֖ית לְפַרְעֹ֑ה וְאַרְבַּ֣ע הַיָּדֹ֡ת יִהְיֶ֣ה לָכֶם֩ לְזֶ֨רַע הַשָּׂדֶ֧ה וּֽלְאׇכְלְכֶ֛ם וְלַאֲשֶׁ֥ר בְּבָתֵּיכֶ֖ם וְלֶאֱכֹ֥ל לְטַפְּכֶֽם׃ וְכָךְ יִהְיֶה בַּקָּצִיר שֶׁתָּבִיאוּ: חֲמִישִׁית תִּתְּנוּ לְפַרְעֹה וְאַרְבַּעַת הַחֲלָקִים יִהְיֶה לָכֶם לְזֶרַע לַשָּׂדֶה וּלְאוֹכֶל עֲבוּרְכֶם וְלָאֲנָשִׁים שֶׁבְּבָתֵּיכֶם, וּלְהַאֲכִיל אֶת יַלְדֵיכֶם.

מפטיר[עריכה]

כה וַיֹּאמְר֖וּ הֶחֱיִתָ֑נוּ נִמְצָא־חֵן֙ בְּעֵינֵ֣י אֲדֹנִ֔י וְהָיִ֥ינוּ עֲבָדִ֖ים לְפַרְעֹֽה׃ אָמְרוּ: הֶחֱיִיתָ אוֹתָנוּ! נִמְצָא חֵן בְּעֵינֵי אֲדוֹנִי וְנִהְיֶה עֲבָדִים לְפַרְעֹה.
כו וַיָּ֣שֶׂם אֹתָ֣הּ יוֹסֵ֡ף לְחֹק֩ עַד־הַיּ֨וֹם הַזֶּ֜ה עַל־אַדְמַ֥ת מִצְרַ֛יִם לְפַרְעֹ֖ה לַחֹ֑מֶשׁ רַ֞ק אַדְמַ֤ת הַכֹּֽהֲנִים֙ לְבַדָּ֔ם לֹ֥א הָיְתָ֖ה לְפַרְעֹֽה׃ יוֹסֵף קָבַע אֶת זֶה לְחוֹק עַד הַיּוֹם הַזֶּה עַל אַדְמַת מִצְרַיִם – שֶׁנּוֹתְנִים לְפַרְעֹה חֲמִישִׁית, רַק אַדְמַת הַכְּמָרִים בִּלְבַד לֹא הָיְתָה שַׁיֶּכֶת לְפַרְעֹה.
כז וַיֵּ֧שֶׁב יִשְׂרָאֵ֛ל בְּאֶ֥רֶץ מִצְרַ֖יִם בְּאֶ֣רֶץ גֹּ֑שֶׁן וַיֵּאָחֲז֣וּ בָ֔הּ וַיִּפְר֥וּ וַיִּרְבּ֖וּ מְאֹֽד׃ יִשְׂרָאֵל הִתְיַשְּׁבוּ בְּאֶרֶץ מִצְרַיִם בְּאֶרֶץ גּוֹשֶׁן, הֵם נֶאֶחְזוּ בָּאָרֶץ וְהוֹלִידוּ וְהִתְרַבּוּ מְאוֹד.
