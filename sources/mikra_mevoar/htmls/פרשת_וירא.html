ראשון[עריכה]

פ פָּרָשַׁת בִּקּוּר הַמַּלְאָכִים
יח א וַיֵּרָ֤א אֵלָיו֙ יְהֹוָ֔ה בְּאֵלֹנֵ֖י מַמְרֵ֑א וְה֛וּא יֹשֵׁ֥ב פֶּֽתַח־הָאֹ֖הֶל כְּחֹ֥ם הַיּֽוֹם׃ ה' הִתְגַּלָּה אֶל אַבְרָהָם בַּמִּישׁוֹר שֶׁל מַמְרֵא כְּשֶׁהוּא יָשַׁב בְּפֶתַח הָאוֹהֶל בְּחוֹם הַיּוֹם.
ב וַיִּשָּׂ֤א עֵינָיו֙ וַיַּ֔רְא וְהִנֵּה֙ שְׁלֹשָׁ֣ה אֲנָשִׁ֔ים נִצָּבִ֖ים עָלָ֑יו וַיַּ֗רְא וַיָּ֤רׇץ לִקְרָאתָם֙ מִפֶּ֣תַח הָאֹ֔הֶל וַיִּשְׁתַּ֖חוּ אָֽרְצָה׃ הוּא הֵרִים אֶת מַבָּטוֹ וְרָאָה וְהִנֵּה שְׁלוֹשָׁה אֲנָשִׁים עוֹמְדִים לְפָנָיו, הוּא רָאָה שֶׁהִתְעַכְּבוּ וְרָץ לִקְרָאתָם מִפֶּתַח הָאוֹהֶל וְהִשְׁתַּחֲוָה עַל הָאֲדָמָה.
ג וַיֹּאמַ֑ר אֲדֹנָ֗י אִם־נָ֨א מָצָ֤אתִי חֵן֙ בְּעֵינֶ֔יךָ אַל־נָ֥א תַעֲבֹ֖ר מֵעַ֥ל עַבְדֶּֽךָ׃ אָמַר: רַבּוֹתַי! אִם מָצָאתִי חֵן בְּעֵינֵיכֶם, בְּבַקָּשָׁה אַל תַּמְשִׁיכוּ הָלְאָה מֵעַבְדְּכֶם.
ד יֻקַּֽח־נָ֣א מְעַט־מַ֔יִם וְרַחֲצ֖וּ רַגְלֵיכֶ֑ם וְהִֽשָּׁעֲנ֖וּ תַּ֥חַת הָעֵֽץ׃ אֶלָּא יוּבָא בְּבַקָּשָׁה מְעַט מַיִם וְתִרְחֲצוּ אֶת רַגְלֵיכֶם וְתִשָּׁעֲנוּ מִתַּחַת לָעֵץ.
ה וְאֶקְחָ֨ה פַת־לֶ֜חֶם וְסַעֲד֤וּ לִבְּכֶם֙ אַחַ֣ר תַּעֲבֹ֔רוּ כִּֽי־עַל־כֵּ֥ן עֲבַרְתֶּ֖ם עַֽל־עַבְדְּכֶ֑ם וַיֹּ֣אמְר֔וּ כֵּ֥ן תַּעֲשֶׂ֖ה כַּאֲשֶׁ֥ר דִּבַּֽרְתָּ׃ וְאָבִיא פַּת לֶחֶם וְתִסְעָדוּ לָשׂוֹבַע וְאַחַר כָּךְ תַּמְשִׁיכוּ, כֵּיוָן שֶׁכְּבָר עֲבַרְתֶּם עַל יַד עַבְדְּכֶם, הֵם אָמְרוּ: כָּךְ תַּעֲשֶׂה, כְּמוֹ שֶׁדִּבַּרְתָּ.
ו וַיְמַהֵ֧ר אַבְרָהָ֛ם הָאֹ֖הֱלָה אֶל־שָׂרָ֑ה וַיֹּ֗אמֶר מַהֲרִ֞י שְׁלֹ֤שׁ סְאִים֙ קֶ֣מַח סֹ֔לֶת ל֖וּשִׁי וַעֲשִׂ֥י עֻגֽוֹת׃ אַבְרָהָם מִהֵר לָאוֹהֶל לְשָׂרָה, וְאָמַר: מַהֲרִי וּקְחִי שָׁלוֹשׁ סְאִים (כְּ21.6 לִיטֶר) שֶׁל קֶמַח סוֹלֶת, וְלוּשִׁי אוֹתָם וַעֲשִׂי מֵהֶם פִּתּוֹת עֲגוּלוֹת.
ז וְאֶל־הַבָּקָ֖ר רָ֣ץ אַבְרָהָ֑ם וַיִּקַּ֨ח בֶּן־בָּקָ֜ר רַ֤ךְ וָטוֹב֙ וַיִּתֵּ֣ן אֶל־הַנַּ֔עַר וַיְמַהֵ֖ר לַעֲשׂ֥וֹת אֹתֽוֹ׃ וְאַבְרָהָם רָץ אֶל הַבָּקָר, לָקַח עֵגֶל רַךְ וָטוֹב וְנָתַן לַמְּשָׁרֵת, וְהוּא מִהֵר לְהָכִין אוֹתוֹ.
ח וַיִּקַּ֨ח חֶמְאָ֜ה וְחָלָ֗ב וּבֶן־הַבָּקָר֙ אֲשֶׁ֣ר עָשָׂ֔ה וַיִּתֵּ֖ן לִפְנֵיהֶ֑ם וְהֽוּא־עֹמֵ֧ד עֲלֵיהֶ֛ם תַּ֥חַת הָעֵ֖ץ וַיֹּאכֵֽלוּ׃ הוּא לָקַח חֶמְאָה וְחָלָב וְאֶת הָעֵגֶל שֶׁהֵכִין וְנָתַן לִפְנֵיהֶם, וְהוּא עוֹמֵד לְיָדָם מִתַּחַת לָעֵץ לְשָׁרֵת אוֹתָם, וְהֵם אָכְלוּ.
ט וַיֹּאמְר֣וּ אֵׄלָ֔יׄוׄ אַיֵּ֖ה שָׂרָ֣ה אִשְׁתֶּ֑ךָ וַיֹּ֖אמֶר הִנֵּ֥ה בָאֹֽהֶל׃ אָמְרוּ לוֹ: הֵיכָן שָׂרָה אִשְׁתְּךָ? אָמַר: הִנֵּה בָּאוֹהֶל.
י וַיֹּ֗אמֶר שׁ֣וֹב אָשׁ֤וּב אֵלֶ֙יךָ֙ כָּעֵ֣ת חַיָּ֔ה וְהִנֵּה־בֵ֖ן לְשָׂרָ֣ה אִשְׁתֶּ֑ךָ וְשָׂרָ֥ה שֹׁמַ֛עַת פֶּ֥תַח הָאֹ֖הֶל וְה֥וּא אַחֲרָֽיו׃ אָמַר אֶחָד מֵהֶם: אֶחֱזוֹר אֵלֶיךָ בִּזְמַן זֶה בְּעוֹד שָׁנָה וְיִהְיֶה לְשָׂרָה אִשְׁתְּךָ בֵּן! וְשָׂרָה שׁוֹמַעַת בְּפֶתַח הָאוֹהֶל שֶׁהָיָה מֵאֲחוֹרֵי הָאִישׁ.
יא וְאַבְרָהָ֤ם וְשָׂרָה֙ זְקֵנִ֔ים בָּאִ֖ים בַּיָּמִ֑ים חָדַל֙ לִהְי֣וֹת לְשָׂרָ֔ה אֹ֖רַח כַּנָּשִֽׁים׃ וְאַבְרָהָם וְשָׂרָה זְקֵנִים – עָבְרוּ עֲלֵיהֶם שָׁנִים רַבּוֹת, וּכְבָר פָּסַק לְשָׂרָה מַחֲזוֹר דָּם נִדָּה כְּדֶרֶךְ הַנָּשִׁים שֶׁעֲדַיִן יוֹלְדוֹת.
יב וַתִּצְחַ֥ק שָׂרָ֖ה בְּקִרְבָּ֣הּ לֵאמֹ֑ר אַחֲרֵ֤י בְלֹתִי֙ הָֽיְתָה־לִּ֣י עֶדְנָ֔ה וַֽאדֹנִ֖י זָקֵֽן׃ שָׂרָה צָחֲקָה בְּלִבָּהּ וְחָשְׁבָה: וְכִי אַחֲרֵי שֶׁנַּעֲשֵׂתִי כְּפֶרַח שֶׁנָּבַל יָשׁוּבוּ לִי נְעוּרִים?! הֲרֵי גַּם אֲדוֹנִי אַבְרָהָם כְּבָר זָקֵן!
יג וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל־אַבְרָהָ֑ם לָ֣מָּה זֶּה֩ צָחֲקָ֨ה שָׂרָ֜ה לֵאמֹ֗ר הַאַ֥ף אֻמְנָ֛ם אֵלֵ֖ד וַאֲנִ֥י זָקַֽנְתִּי׃ אָמַר ה' לְאַבְרָהָם: לָמָּה שָׂרָה צָחֲקָה וְחָשְׁבָה: וְכִי בֶּאֱמֶת אֵלֵד וַהֲרֵי אֲנִי זָקַנְתִּי?
יד הֲיִפָּלֵ֥א מֵיְהֹוָ֖ה דָּבָ֑ר לַמּוֹעֵ֞ד אָשׁ֥וּב אֵלֶ֛יךָ כָּעֵ֥ת חַיָּ֖ה וּלְשָׂרָ֥ה בֵֽן׃ וְכִי יִמָּנַע מֵה' דָּבָר?! בִּזְמַן זֶה אֶחֱזוֹר אֵלֶיךָ בְּעוֹד שָׁנָה, וּלְשָׂרָה יִהְיֶה בֵּן!

שני[עריכה]

טו וַתְּכַחֵ֨שׁ שָׂרָ֧ה ׀ לֵאמֹ֛ר לֹ֥א צָחַ֖קְתִּי כִּ֣י ׀ יָרֵ֑אָה וַיֹּ֥אמֶר ׀ לֹ֖א כִּ֥י צָחָֽקְתְּ׃ וְשָׂרָה הִכְחִישָׁה, וְאָמְרָה: לֹא צָחַקְתִּי, כִּי הִיא פָּחֲדָה, אָמַר לָהּ אַבְרָהָם: לֹא נָכוֹן, אַתְּ כֵּן צָחָקְתְּ!
פָּרָשַׁת הַתְּפִלָּה עַל סְדֹם
טז וַיָּקֻ֤מוּ מִשָּׁם֙ הָֽאֲנָשִׁ֔ים וַיַּשְׁקִ֖פוּ עַל־פְּנֵ֣י סְדֹ֑ם וְאַ֨בְרָהָ֔ם הֹלֵ֥ךְ עִמָּ֖ם לְשַׁלְּחָֽם׃ הָאֲנָשִׁים קָמוּ מִשָּׁם וְהִשְׁקִיפוּ לְעֵבֶר סְדוֹם, וְאַבְרָהָם הוֹלֵךְ אִתָּם לְלַוּוֹת אוֹתָם בִּיצִיאָתָם.
יז וַֽיהֹוָ֖ה אָמָ֑ר הַֽמְכַסֶּ֤ה אֲנִי֙ מֵֽאַבְרָהָ֔ם אֲשֶׁ֖ר אֲנִ֥י עֹשֶֽׂה׃ ה' אָמַר: וְכִי אַסְתִּיר מֵאַבְרָהָם אֶת מַה שֶּׁאֲנִי עוֹשֶׂה?!
יח וְאַ֨בְרָהָ֔ם הָי֧וֹ יִֽהְיֶ֛ה לְג֥וֹי גָּד֖וֹל וְעָצ֑וּם וְנִ֨בְרְכוּ־ב֔וֹ כֹּ֖ל גּוֹיֵ֥י הָאָֽרֶץ׃ הֲרֵי מֵאַבְרָהָם יֵצֵא עָם גָּדוֹל וְעָצוּם, וְיִתְבָּרְכוּ בִּזְכוּתוֹ כָּל עַמֵּי הָעוֹלָם.
יט כִּ֣י יְדַעְתִּ֗יו לְמַ֩עַן֩ אֲשֶׁ֨ר יְצַוֶּ֜ה אֶת־בָּנָ֤יו וְאֶת־בֵּיתוֹ֙ אַחֲרָ֔יו וְשָֽׁמְרוּ֙ דֶּ֣רֶךְ יְהֹוָ֔ה לַעֲשׂ֥וֹת צְדָקָ֖ה וּמִשְׁפָּ֑ט לְמַ֗עַן הָבִ֤יא יְהֹוָה֙ עַל־אַבְרָהָ֔ם אֵ֥ת אֲשֶׁר־דִּבֶּ֖ר עָלָֽיו׃ אֲנִי מוֹדִיע לוֹ זֹאת כְּדֵי שֶׁיְּצַוֶּה אֶת בָּנָיו וְאֶת מִשְׁפַּחְתּוֹ אַחֲרָיו לְהַזְהִיר אוֹתָם שֶׁיִּשְׁמְרוּ אֶת דֶּרֶךְ ה' לַעֲשׂוֹת צְדָקָה וּמִשְׁפָּט, כְּדֵי שֶׁה' יָבִיא לְאַבְרָהָם אֶת מַה שֶּׁהִבְטִיחַ לוֹ.
כ וַיֹּ֣אמֶר יְהֹוָ֔ה זַעֲקַ֛ת סְדֹ֥ם וַעֲמֹרָ֖ה כִּי־רָ֑בָּה וְחַ֨טָּאתָ֔ם כִּ֥י כָבְדָ֖ה מְאֹֽד׃ ה' אָמַר: זַעֲקַת הַתְּלֻנּוֹת עַל סְדוֹם וַעֲמוֹרָה גְּדוֹלָה, וְהָעֲבֵרוֹת שֶׁלָּהֶם חֲמוּרוֹת מְאוֹד.
כא אֵֽרְﬞדָה־נָּ֣א וְאֶרְאֶ֔ה הַכְּצַעֲקָתָ֛הּ הַבָּ֥אָה אֵלַ֖י עָשׂ֣וּ ׀ כָּלָ֑ה וְאִם־לֹ֖א אֵדָֽעָה׃ לָכֵן אֵרֵד עַכְשָׁיו וְאֶרְאֶה: אִם אָכֵן עָשׂוּ כְּמוֹ שֶׁמִּתְלוֹנְנִים אֵלַי – אֲכַלֶּה אוֹתָם, וְאִם לֹא –אֵדַע שֶׁאֵינָם חַיָּבִים.
כב וַיִּפְנ֤וּ מִשָּׁם֙ הָֽאֲנָשִׁ֔ים וַיֵּלְכ֖וּ סְדֹ֑מָה וְאַ֨בְרָהָ֔ם עוֹדֶ֥נּוּ עֹמֵ֖ד לִפְנֵ֥י יְהֹוָֽה׃ הָאֲנָשִׁים פָּנוּ מִשָּׁם וְהָלְכוּ לִסְדוֹם בְּעוֹד אַבְרָהָם עֲדַיִן עוֹמֵד וּמְדַבֵּר לִפְנֵי ה'.
כג וַיִּגַּ֥שׁ אַבְרָהָ֖ם וַיֹּאמַ֑ר הַאַ֣ף תִּסְפֶּ֔ה צַדִּ֖יק עִם־רָשָֽׁע׃ אַבְרָהָם נִגַּשׁ וְאָמַר: וְכִי תַּשְׁמִיד צַדִּיק עִם רָשָׁע?!
כד אוּלַ֥י יֵ֛שׁ חֲמִשִּׁ֥ים צַדִּיקִ֖ם בְּת֣וֹךְ הָעִ֑יר הַאַ֤ף תִּסְפֶּה֙ וְלֹא־תִשָּׂ֣א לַמָּק֔וֹם לְמַ֛עַן חֲמִשִּׁ֥ים הַצַּדִּיקִ֖ם אֲשֶׁ֥ר בְּקִרְבָּֽהּ׃ אוּלַי יֵשׁ חֲמִשִּׁים צַדִּיקִים בְּתוֹךְ הָעִיר, וְכִי תַּשְׁמִיד אוֹתָם וְלֹא תְּוַתֵּר לַמָּקוֹם בִּשְׁבִיל חֲמִשִּׁים הַצַּדִּיקִים שֶׁבְּתוֹכָהּ?!
כה חָלִ֨לָה לְּךָ֜ מֵעֲשֹׂ֣ת ׀ כַּדָּבָ֣ר הַזֶּ֗ה לְהָמִ֤ית צַדִּיק֙ עִם־רָשָׁ֔ע וְהָיָ֥ה כַצַּדִּ֖יק כָּרָשָׁ֑ע חָלִ֣לָה לָּ֔ךְ הֲשֹׁפֵט֙ כׇּל־הָאָ֔רֶץ לֹ֥א יַעֲשֶׂ֖ה מִשְׁפָּֽט׃ חָלִילָה לְךָ לַעֲשׂוֹת כַּדָּבָר הַזֶּה, לְהָמִית צַדִּיק עִם רָשָׁע, שֶׁצַּדִּיק יֵעָנֵשׁ כְּמוֹ רָשָׁע, חָלִילָה לְךָ! הַאִם הַשּׁוֹפֵט שֶׁל כָּל הָעוֹלָם לֹא יַעֲשֶׂה מִשְׁפָּט צֶדֶק?!
כו וַיֹּ֣אמֶר יְהֹוָ֔ה אִם־אֶמְצָ֥א בִסְדֹ֛ם חֲמִשִּׁ֥ים צַדִּיקִ֖ם בְּת֣וֹךְ הָעִ֑יר וְנָשָׂ֥אתִי לְכׇל־הַמָּק֖וֹם בַּעֲבוּרָֽם׃ ה' אָמַר: אִם אֶמְצָא בִּסְדוֹם חֲמִשִּׁים צַדִּיקִים בְּתוֹךְ הָעִיר אֲוַתֵּר לְכָל הַמָּקוֹם בִּשְׁבִילָם.
כז וַיַּ֥עַן אַבְרָהָ֖ם וַיֹּאמַ֑ר הִנֵּה־נָ֤א הוֹאַ֙לְתִּי֙ לְדַבֵּ֣ר אֶל־אֲדֹנָ֔י וְאָנֹכִ֖י עָפָ֥ר וָאֵֽפֶר׃ אַבְרָהָם עָנָה וְאָמַר: אַמְשִׁיךְ בְּבַקָּשָׁה כִּי זָכִיתִי לְדַבֵּר אֶל ה', לַמְרוֹת שֶׁאֲנִי רַק עָפָר וְאֵפֶר.
כח א֠וּלַ֠י יַחְסְר֞וּן חֲמִשִּׁ֤ים הַצַּדִּיקִם֙ חֲמִשָּׁ֔ה הֲתַשְׁחִ֥ית בַּחֲמִשָּׁ֖ה אֶת־כׇּל־הָעִ֑יר וַיֹּ֙אמֶר֙ לֹ֣א אַשְׁחִ֔ית אִם־אֶמְצָ֣א שָׁ֔ם אַרְבָּעִ֖ים וַחֲמִשָּֽׁה׃ אוּלַי יַחְסְרוּ מֵחֲמִשִּׁים הַצַּדִּיקִים חֲמִשָּׁה, הַאִם בִּגְלַל חֲמִשָּׁה חֲסֵרִים תַּשְׁחִית אֶת כָּל הָעִיר?! אָמַר לוֹ: לֹא אַשְׁחִית אִם אֶמְצָא שָׁם אַרְבָּעִים וַחֲמִשָּׁה.
כט וַיֹּ֨סֶף ע֜וֹד לְדַבֵּ֤ר אֵלָיו֙ וַיֹּאמַ֔ר אוּלַ֛י יִמָּצְא֥וּן שָׁ֖ם אַרְבָּעִ֑ים וַיֹּ֙אמֶר֙ לֹ֣א אֶֽעֱשֶׂ֔ה בַּעֲב֖וּר הָאַרְבָּעִֽים׃ הוּא הוֹסִיף עוֹד לְדַבֵּר אֵלָיו, וְאָמַר: אוּלַי יִמָּצְאוּ שָׁם אַרְבָּעִים? אָמַר לוֹ: לֹא אֶעֱשֶׂה, בִּזְכוּת הָאַרְבָּעִים.
ל וַ֠יֹּ֠אמֶר אַל־נָ֞א יִ֤חַר לַֽאדֹנָי֙ וַאֲדַבֵּ֔רָה אוּלַ֛י יִמָּצְא֥וּן שָׁ֖ם שְׁלֹשִׁ֑ים וַיֹּ֙אמֶר֙ לֹ֣א אֶֽעֱשֶׂ֔ה אִם־אֶמְצָ֥א שָׁ֖ם שְׁלֹשִֽׁים׃ אָמַר לוֹ: בְּבַקָּשָׁה, שֶׁלֹּא יַכְעִיס הַדָּבָר אֶת ה' וַאֲדַבֵּר, אוּלַי יִמָּצְאוּ שָׁם שְׁלוֹשִׁים? אָמַר לוֹ: לֹא אֶעֱשֶׂה אִם אֶמְצָא שָׁם שְׁלוֹשִׁים.
לא וַיֹּ֗אמֶר הִנֵּֽה־נָ֤א הוֹאַ֙לְתִּי֙ לְדַבֵּ֣ר אֶל־אֲדֹנָ֔י אוּלַ֛י יִמָּצְא֥וּן שָׁ֖ם עֶשְׂרִ֑ים וַיֹּ֙אמֶר֙ לֹ֣א אַשְׁחִ֔ית בַּעֲב֖וּר הָֽעֶשְׂרִֽים׃ אָמַר לוֹ: בְּבַקָּשָׁה, הִנֵּה זָכִיתִי לְדַבֵּר אֶל ה', אוּלַי יִמָּצְאוּ שָׁם עֶשְׂרִים? אָמַר לוֹ: לֹא אַשְׁחִית בִּזְכוּת הָעֶשְׂרִים.
לב וַ֠יֹּ֠אמֶר אַל־נָ֞א יִ֤חַר לַֽאדֹנָי֙ וַאֲדַבְּרָ֣ה אַךְ־הַפַּ֔עַם אוּלַ֛י יִמָּצְא֥וּן שָׁ֖ם עֲשָׂרָ֑ה וַיֹּ֙אמֶר֙ לֹ֣א אַשְׁחִ֔ית בַּעֲב֖וּר הָעֲשָׂרָֽה׃ אָמַר לוֹ: בְּבַקָּשָׁה, שֶׁלֹּא יַכְעִיס הַדָּבָר אֶת ה' וַאֲדַבֵּר רַק הַפַּעַם, אוּלַי יִמָּצְאוּ שָׁם עֲשָׂרָה! אָמַר לוֹ: לֹא אַשְׁחִית בִּזְכוּת הָעֲשָׂרָה.
לג וַיֵּ֣לֶךְ יְהֹוָ֔ה כַּאֲשֶׁ֣ר כִּלָּ֔ה לְדַבֵּ֖ר אֶל־אַבְרָהָ֑ם וְאַבְרָהָ֖ם שָׁ֥ב לִמְקֹמֽוֹ׃ ה' הָלַךְ כְּשֶׁסִּיֵּם לְדַבֵּר אֶל אַבְרָהָם, וְאַבְרָהָם חָזַר לִמְקוֹמוֹ.

שלישי[עריכה]

פָּרָשַׁת הֲפִיכַת סְדוֹם
יט א וַ֠יָּבֹ֠אוּ שְׁנֵ֨י הַמַּלְאָכִ֤ים סְדֹ֙מָה֙ בָּעֶ֔רֶב וְל֖וֹט יֹשֵׁ֣ב בְּשַֽׁעַר־סְדֹ֑ם וַיַּרְא־לוֹט֙ וַיָּ֣קׇם לִקְרָאתָ֔ם וַיִּשְׁתַּ֥חוּ אַפַּ֖יִם אָֽרְצָה׃ שְׁנֵי הַמַּלְאָכִים בָּאוּ לִסְדוֹם בָּעֶרֶב, וְלוֹט הָיָה יוֹשֵׁב בְּשַׁעַר סְדוֹם, לוֹט רָאָה אוֹתָם קָם לִקְרָאתָם, וְהִשְׁתַּחֲוָה עַל פָּנָיו עַל הָאֲדָמָה.
ב וַיֹּ֜אמֶר הִנֶּ֣ה נָּא־אֲדֹנַ֗י ס֣וּרוּ נָ֠א אֶל־בֵּ֨ית עַבְדְּכֶ֤ם וְלִ֙ינוּ֙ וְרַחֲצ֣וּ רַגְלֵיכֶ֔ם וְהִשְׁכַּמְתֶּ֖ם וַהֲלַכְתֶּ֣ם לְדַרְכְּכֶ֑ם וַיֹּאמְר֣וּ לֹּ֔א כִּ֥י בָרְח֖וֹב נָלִֽין׃ אָמַר: אָנָּא רַבּוֹתַי, פְּנוּ בְּבַקָּשָׁה לְבֵית עַבְדְּכֶם לִישׁוֹן הַלַּיְלָה וְתִרְחֲצוּ אֶת רַגְלֵיכֶם וְתַשְׁכִּימוּ בַּבּוֹקֶר וְתֵלְכוּ לְדַרְכְּכֶם! אָמְרוּ לוֹ: לֹא! אֶלָּא נִישַׁן הַלַּיְלָה בָּרְחוֹב.
ג וַיִּפְצַר־בָּ֣ם מְאֹ֔ד וַיָּסֻ֣רוּ אֵלָ֔יו וַיָּבֹ֖אוּ אֶל־בֵּית֑וֹ וַיַּ֤עַשׂ לָהֶם֙ מִשְׁתֶּ֔ה וּמַצּ֥וֹת אָפָ֖ה וַיֹּאכֵֽלוּ׃ הוּא הִפְצִיר בָּהֶם מְאוֹד, הֵם פָּנוּ אֵלָיו וּבָאוּ לְבֵיתוֹ, הוּא עָשָׂה לָהֶם מִשְׁתֶּה וְאָפָה מַצּוֹת וְהֵם אָכְלוּ.
ד טֶ֘רֶם֮ יִשְׁכָּ֒בוּ֒ וְאַנְשֵׁ֨י הָעִ֜יר אַנְשֵׁ֤י סְדֹם֙ נָסַ֣בּוּ עַל־הַבַּ֔יִת מִנַּ֖עַר וְעַד־זָקֵ֑ן כׇּל־הָעָ֖ם מִקָּצֶֽה׃ לִפְנֵי שֶׁהִסְפִּיקוּ לִשְׁכַּב, אַנְשֵׁי הָעִיר אַנְשֵׁי סְדוֹם הִקִּיפוּ אֶת הַבַּיִת מִנַּעַר וְעַד זָקֵן – כָּל הָעָם מִקְּצֶה הָעִיר.
ה וַיִּקְרְא֤וּ אֶל־לוֹט֙ וַיֹּ֣אמְרוּ ל֔וֹ אַיֵּ֧ה הָאֲנָשִׁ֛ים אֲשֶׁר־בָּ֥אוּ אֵלֶ֖יךָ הַלָּ֑יְלָה הוֹצִיאֵ֣ם אֵלֵ֔ינוּ וְנֵדְעָ֖ה אֹתָֽם׃ הֵם קָרְאוּ לְלוֹט וְאָמְרוּ לוֹ: אֵיפֹה הָאֲנָשִׁים שֶׁבָּאוּ אֵלֶיךָ הַלַּיְלָה? הוֹצֵא אוֹתָם אֵלֵינוּ וְנֵדַע מַה לַּעֲשׂוֹת אִתָּם!
ו וַיֵּצֵ֧א אֲלֵהֶ֛ם ל֖וֹט הַפֶּ֑תְחָה וְהַדֶּ֖לֶת סָגַ֥ר אַחֲרָֽיו׃ לוֹט יָצָא אֲלֵהֶם לַפֶּתַח וְסָגַר אֶת הַדֶּלֶת אַחֲרָיו.
ז וַיֹּאמַ֑ר אַל־נָ֥א אַחַ֖י תָּרֵֽעוּ׃ וְאָמַר: אַחַי! בְּבַקָּשָׁה אַל תַּעֲשׂוּ רָע!
ח הִנֵּה־נָ֨א לִ֜י שְׁתֵּ֣י בָנ֗וֹת אֲשֶׁ֤ר לֹֽא־יָדְעוּ֙ אִ֔ישׁ אוֹצִֽיאָה־נָּ֤א אֶתְהֶן֙ אֲלֵיכֶ֔ם וַעֲשׂ֣וּ לָהֶ֔ן כַּטּ֖וֹב בְּעֵינֵיכֶ֑ם רַ֠ק לָֽאֲנָשִׁ֤ים הָאֵל֙ אַל־תַּעֲשׂ֣וּ דָבָ֔ר כִּֽי־עַל־כֵּ֥ן בָּ֖אוּ בְּצֵ֥ל קֹרָתִֽי׃ בְּבַקָּשָׁה, הִנֵּה יֵשׁ לִי שְׁתֵּי בָּנוֹת שֶׁלֹּא נִשְּׂאוּ לְאִישׁ, אוֹצִיא אוֹתָן בְּבַקָּשָׁה אֲלֵיכֶם וַעֲשׂוּ לָהֶן כַּטּוֹב בְּעֵינֵיכֶם, אֲבָל לָאֲנָשִׁים הָאֵלֶּה אַל תַּעֲשׂוּ שׁוּם דָּבָר כִּי הֵם בָּאוּ לְהִתְאָרֵחַ בְּצֵל קוֹרָתִי.
ט וַיֹּאמְר֣וּ ׀ גֶּשׁ־הָ֗לְאָה וַיֹּֽאמְרוּ֙ הָאֶחָ֤ד בָּֽא־לָגוּר֙ וַיִּשְׁפֹּ֣ט שָׁפ֔וֹט עַתָּ֕ה נָרַ֥ע לְךָ֖ מֵהֶ֑ם וַיִּפְצְר֨וּ בָאִ֤ישׁ בְּלוֹט֙ מְאֹ֔ד וַֽיִּגְּשׁ֖וּ לִשְׁבֹּ֥ר הַדָּֽלֶת׃ אָמְרוּ לוֹ: לֵךְ הָלְאָה! אָמְרוּ: אִישׁ אֶחָד רַק בָּא לָגוּר וּכְבָר מַנְהִיג אוֹתָנוּ?! עַכְשָׁיו נַעֲשֶׂה לְךָ רַע יוֹתֵר מֵאֲשֶׁר לָהֶם! הֵם לָחֲצוּ מְאוֹד עַל הָאִישׁ – עַל לוֹט, וְנִגְּשׁוּ לִשְׁבּוֹר אֶת הַדֶּלֶת.
י וַיִּשְׁלְח֤וּ הָֽאֲנָשִׁים֙ אֶת־יָדָ֔ם וַיָּבִ֧יאוּ אֶת־ל֛וֹט אֲלֵיהֶ֖ם הַבָּ֑יְתָה וְאֶת־הַדֶּ֖לֶת סָגָֽרוּ׃ הַמַּלְאָכִים הוֹשִׁיטוּ אֶת יְדֵיהֶם וְהֵבִיאוּ אֶת לוֹט אֲלֵיהֶם הַבַּיְתָה, וְסָגְרוּ אֶת הַדֶּלֶת.
יא וְֽאֶת־הָאֲנָשִׁ֞ים אֲשֶׁר־פֶּ֣תַח הַבַּ֗יִת הִכּוּ֙ בַּסַּנְוֵרִ֔ים מִקָּטֹ֖ן וְעַד־גָּד֑וֹל וַיִּלְא֖וּ לִמְצֹ֥א הַפָּֽתַח׃ וְאֶת הָאֲנָשִׁים שֶׁבְּפֶתַח הַבַּיִת הֵם סִנְוְרוּ – מִקָּטָן וְעַד גָּדוֹל, וְהֵם הִתְעַיְּפוּ מֵחִפּוּשִׂים לִמְצוֹא אֶת הַפֶּתַח.
יב וַיֹּאמְר֨וּ הָאֲנָשִׁ֜ים אֶל־ל֗וֹט עֹ֚ד מִֽי־לְךָ֣ פֹ֔ה חָתָן֙ וּבָנֶ֣יךָ וּבְנֹתֶ֔יךָ וְכֹ֥ל אֲשֶׁר־לְךָ֖ בָּעִ֑יר הוֹצֵ֖א מִן־הַמָּקֽוֹם׃ אָמְרוּ הַמַּלְאָכִים לְלוֹט: כָּל מִי שֶׁיֵּשׁ לְךָ עוֹד פֹּה – חָתָן וּבָנֶיךָ וּבְנוֹתֶיךָ, וְכָל מַה שֶּׁבָּעִיר שֶׁשַּׁיָּךְ לְךָ – הוֹצֵא מֵהַמָּקוֹם!
יג כִּֽי־מַשְׁחִתִ֣ים אֲנַ֔חְנוּ אֶת־הַמָּק֖וֹם הַזֶּ֑ה כִּֽי־גָדְלָ֤ה צַעֲקָתָם֙ אֶת־פְּנֵ֣י יְהֹוָ֔ה וַיְשַׁלְּחֵ֥נוּ יְהֹוָ֖ה לְשַׁחֲתָֽהּ׃ כִּי אֲנַחְנוּ מַשְׁמִידִים אֶת הַמָּקוֹם הַזֶּה, כִּי הַתְּלֻנּוֹת עֲלֵיהֶם אֶל ה' חֲמוּרוֹת, וַה' שָׁלַח אוֹתָנוּ לְהַשְׁמִיד אוֹתוֹ.
יד וַיֵּצֵ֨א ל֜וֹט וַיְדַבֵּ֣ר ׀ אֶל־חֲתָנָ֣יו ׀ לֹקְחֵ֣י בְנֹתָ֗יו וַיֹּ֙אמֶר֙ ק֤וּמוּ צְּאוּ֙ מִן־הַמָּק֣וֹם הַזֶּ֔ה כִּֽי־מַשְׁחִ֥ית יְהֹוָ֖ה אֶת־הָעִ֑יר וַיְהִ֥י כִמְצַחֵ֖ק בְּעֵינֵ֥י חֲתָנָֽיו׃ לוֹט יָצָא וְדִבֵּר אֶל חֲתָנָיו – בַּעֲלֵי בְּנוֹתָיו, וְאָמַר: קוּמוּ, צְאוּ מֵהַמָּקוֹם הַזֶּה, כִּי ה' מַשְׁחִית אֶת הָעִיר! אֲבָל בְּעֵינֵי חֲתָנָיו הוּא נִרְאָה כְּמִתְלוֹצֵץ.
טו וּכְמוֹ֙ הַשַּׁ֣חַר עָלָ֔ה וַיָּאִ֥יצוּ הַמַּלְאָכִ֖ים בְּל֣וֹט לֵאמֹ֑ר קוּם֩ קַ֨ח אֶֽת־אִשְׁתְּךָ֜ וְאֶת־שְׁתֵּ֤י בְנֹתֶ֙יךָ֙ הַנִּמְצָאֹ֔ת פֶּן־תִּסָּפֶ֖ה בַּעֲוֺ֥ן הָעִֽיר׃ וּכְשֶׁעָלָה הַשַּׁחַר הַמַּלְאָכִים הֵאִיצוּ בְּלוֹט וְאָמְרוּ: קוּם קַח אֶת אִשְׁתְּךָ וְאֶת שְׁתֵּי בְּנוֹתֶיךָ שֶׁנִּמְצָאוֹת בְּבֵיתְךָ כְּדֵי שֶׁלֹּא תֻּשְׁמַד בַּעֲווֹן הָעִיר!
טז וַֽיִּתְמַהְמָ֓הּ ׀ וַיַּחֲזִ֨יקוּ הָאֲנָשִׁ֜ים בְּיָד֣וֹ וּבְיַד־אִשְׁתּ֗וֹ וּבְיַד֙ שְׁתֵּ֣י בְנֹתָ֔יו בְּחֶמְלַ֥ת יְהֹוָ֖ה עָלָ֑יו וַיֹּצִאֻ֥הוּ וַיַּנִּחֻ֖הוּ מִח֥וּץ לָעִֽיר׃ הוּא הִתְעַכֵּב, הָאֲנָשִׁים הֶחֱזִיקוּ בְּיָדוֹ וּבְיַד אִשְׁתּוֹ וּבְיַד שְׁתֵּי בְּנוֹתָיו בִּגְלַל רַחֲמֵי ה' עָלָיו, וְהוֹצִיאוּ אוֹתוֹ וְהִנִּיחוּ אוֹתוֹ מִחוּץ לָעִיר.
יז וַיְהִי֩ כְהוֹצִיאָ֨ם אֹתָ֜ם הַח֗וּצָה וַיֹּ֙אמֶר֙ הִמָּלֵ֣ט עַל־נַפְשֶׁ֔ךָ אַל־תַּבִּ֣יט אַחֲרֶ֔יךָ וְאַֽל־תַּעֲמֹ֖ד בְּכׇל־הַכִּכָּ֑ר הָהָ֥רָה הִמָּלֵ֖ט פֶּן־תִּסָּפֶֽה׃ כְּשֶׁהוֹצִיאוּ אוֹתָם הַחוּצָה, אָמַר לוֹ הָאֶחָד: בְּרַח לְנַפְשְׁךָ אַל תַּבִּיט לַאֲחוֹרֶיךָ וְאַל תַּעֲצוֹר בְּכָל הַבִּקְעָה, בְּרַח לָהָר כְּדֵי שֶׁלֹּא תֻּשְׁמַד!
יח וַיֹּ֥אמֶר ל֖וֹט אֲלֵהֶ֑ם אַל־נָ֖א אֲדֹנָֽי׃ אָמַר לָהֶם לוֹט : רַבּוֹתַי! בְּבַקָּשָׁה אַל תֹּאמְרוּ כָּכָה.
יט הִנֵּה־נָ֠א מָצָ֨א עַבְדְּךָ֣ חֵן֮ בְּעֵינֶ֒יךָ֒ וַתַּגְדֵּ֣ל חַסְדְּךָ֗ אֲשֶׁ֤ר עָשִׂ֙יתָ֙ עִמָּדִ֔י לְהַחֲי֖וֹת אֶת־נַפְשִׁ֑י וְאָנֹכִ֗י לֹ֤א אוּכַל֙ לְהִמָּלֵ֣ט הָהָ֔רָה פֶּן־תִּדְבָּקַ֥נִי הָרָעָ֖ה וָמַֽתִּי׃ בְּבַקָּשָׁה הִנֵּה עַבְדְּךָ מָצָא חֵן בְּעֵינֶיךָ וְעָשִׂיתָ אִתִּי חֶסֶד גָּדוֹל שֶׁהֶחֱיֵיתָ אֶת נַפְשִׁי, וַאֲנִי לֹא אוּכַל לִבְרוֹחַ לָהָר הָרָחוֹק, שֶׁלֹּא תַּשִּׂיג אוֹתִי הָרָעָה וְאָמוּת.
כ הִנֵּה־נָ֠א הָעִ֨יר הַזֹּ֧את קְרֹבָ֛ה לָנ֥וּס שָׁ֖מָּה וְהִ֣וא מִצְעָ֑ר אִמָּלְטָ֨ה נָּ֜א שָׁ֗מָּה הֲלֹ֥א מִצְעָ֛ר הִ֖וא וּתְחִ֥י נַפְשִֽׁי׃ בְּבַקָּשָׁה, הִנֵּה הָעִיר הַזֹּאת קְרוֹבָה וְקַל לִבְרוֹחַ לְשָׁם וְהִיא קְטַנָּה, אֶבְרַח בְּבַקָּשָׁה לְשָׁם, וְלֹא תַּשְׁמִידוּ אוֹתָהּ – הֲרֵי הִיא קְטַנָּה, וְנַפְשִׁי תִּחְיֶה.

רביעי[עריכה]

כא וַיֹּ֣אמֶר אֵלָ֔יו הִנֵּה֙ נָשָׂ֣אתִי פָנֶ֔יךָ גַּ֖ם לַדָּבָ֣ר הַזֶּ֑ה לְבִלְתִּ֛י הׇפְכִּ֥י אֶת־הָעִ֖יר אֲשֶׁ֥ר דִּבַּֽרְתָּ׃ אָמַר לוֹ: הִנֵּה אֲנִי מַסְכִּים לִרְצוֹנְךָ גַּם בַּדָּבָר הַזֶּה, שֶׁלֹּא אֶהֱרוֹס אֶת הָעִיר שֶׁדִּבַּרְתָּ עָלֶיהָ.
כב מַהֵר֙ הִמָּלֵ֣ט שָׁ֔מָּה כִּ֣י לֹ֤א אוּכַל֙ לַעֲשׂ֣וֹת דָּבָ֔ר עַד־בֹּאֲךָ֖ שָׁ֑מָּה עַל־כֵּ֛ן קָרָ֥א שֵׁם־הָעִ֖יר צֽוֹעַר׃ תְּמַהֵר לִבְרוֹחַ לְשָׁם! כִּי לֹא אוּכַל לַעֲשׂוֹת שׁוּם דָּבָר עַד שֶׁתָּבוֹא לְשָׁם, לָכֵן קָרָא אֶת שֵׁם הָעִיר צוֹעַר – הַקְּטַנָּה (צָעִיר – זָעִיר).
כג הַשֶּׁ֖מֶשׁ יָצָ֣א עַל־הָאָ֑רֶץ וְל֖וֹט בָּ֥א צֹֽעֲרָה׃ הַשֶּׁמֶשׁ זָרְחָה עַל הָאָרֶץ וְלוֹט הִגִּיעַ לְצוֹעַר.
כד וַֽיהֹוָ֗ה הִמְטִ֧יר עַל־סְדֹ֛ם וְעַל־עֲמֹרָ֖ה גׇּפְרִ֣ית וָאֵ֑שׁ מֵאֵ֥ת יְהֹוָ֖ה מִן־הַשָּׁמָֽיִם׃ וַה' הִמְטִיר עַל סְדוֹם וְעַל עֲמוֹרָה גָּפְרִית וָאֵשׁ מֵאֵת ה' מֵהַשָּׁמַיִם.
כה וַֽיַּהֲפֹךְ֙ אֶת־הֶעָרִ֣ים הָאֵ֔ל וְאֵ֖ת כׇּל־הַכִּכָּ֑ר וְאֵת֙ כׇּל־יֹשְׁבֵ֣י הֶעָרִ֔ים וְצֶ֖מַח הָאֲדָמָֽה׃ הוּא הָרַס אֶת הֶעָרִים הָאֵלֶּה וְאֶת כָּל הַבִּקְעָה וְאֶת כָּל יוֹשְׁבֵי הֶעָרִים וְאֶת צֶמַח הָאֲדָמָה.
כו וַתַּבֵּ֥ט אִשְׁתּ֖וֹ מֵאַחֲרָ֑יו וַתְּהִ֖י נְצִ֥יב מֶֽלַח׃ אִשְׁתּוֹ הִבִּיטָה לַאֲחוֹרָיו וְהָפְכָה לְמַצֵּבָה מִמֶּלַח.
כז וַיַּשְׁכֵּ֥ם אַבְרָהָ֖ם בַּבֹּ֑קֶר אֶ֨ל־הַמָּק֔וֹם אֲשֶׁר־עָ֥מַד שָׁ֖ם אֶת־פְּנֵ֥י יְהֹוָֽה׃ אַבְרָהָם הִשְׁכִּים בַּבּוֹקֶר וְהָלַךְ לַמָּקוֹם שֶׁעָמַד שָׁם לְהִתְפַּלֵּל לִפְנֵי ה'.
כח וַיַּשְׁקֵ֗ף עַל־פְּנֵ֤י סְדֹם֙ וַעֲמֹרָ֔ה וְעַֽל־כׇּל־פְּנֵ֖י אֶ֣רֶץ הַכִּכָּ֑ר וַיַּ֗רְא וְהִנֵּ֤ה עָלָה֙ קִיטֹ֣ר הָאָ֔רֶץ כְּקִיטֹ֖ר הַכִּבְשָֽׁן׃ הוּא הִשְׁקִיף לְעֵבֶר סְדוֹם וַעֲמוֹרָה וּלְעֵבֶר כָּל שֶׁטַח הַבִּקְעָה, וְרָאָה וְהִנֵּה עָלָה עָשָׁן הָאָרֶץ כְּמוֹ עָשָׁן כִּבְשָׁן.
כט וַיְהִ֗י בְּשַׁחֵ֤ת אֱלֹהִים֙ אֶת־עָרֵ֣י הַכִּכָּ֔ר וַיִּזְכֹּ֥ר אֱלֹהִ֖ים אֶת־אַבְרָהָ֑ם וַיְשַׁלַּ֤ח אֶת־לוֹט֙ מִתּ֣וֹךְ הַהֲפֵכָ֔ה בַּהֲפֹךְ֙ אֶת־הֶ֣עָרִ֔ים אֲשֶׁר־יָשַׁ֥ב בָּהֵ֖ן לֽוֹט׃ כְּשֶׁאֱלֹהִים הִשְׁחִית אֶת עָרֵי הַבִּקְעָה, אֱלֹהִים זָכַר אֶת אַבְרָהָם, הוּא הוֹצִיא אֶת לוֹט מֵהַמַּהְפֵּכָה כְּשֶׁהָרַס אֶת הֶעָרִים שֶׁלּוֹט גָּר בָּהֵן.
פָּרָשַׁת בְּנוֹת לוֹט
ל וַיַּ֩עַל֩ ל֨וֹט מִצּ֜וֹעַר וַיֵּ֣שֶׁב בָּהָ֗ר וּשְׁתֵּ֤י בְנֹתָיו֙ עִמּ֔וֹ כִּ֥י יָרֵ֖א לָשֶׁ֣בֶת בְּצ֑וֹעַר וַיֵּ֙שֶׁב֙ בַּמְּעָרָ֔ה ה֖וּא וּשְׁתֵּ֥י בְנֹתָֽיו׃ לוֹט עָלָה מִצּוֹעַר וְהִתְיַשֵּׁב בָּהָר וּשְׁתֵּי בְּנוֹתָיו אִתּוֹ כִּי הוּא פָּחַד לְהִתְיַשֵּׁב בְּצוֹעַר, הוּא הִתְיַשֵּׁב בַּמְּעָרָה עִם שְׁתֵּי בְּנוֹתָיו.
לא וַתֹּ֧אמֶר הַבְּכִירָ֛ה אֶל־הַצְּעִירָ֖ה אָבִ֣ינוּ זָקֵ֑ן וְאִ֨ישׁ אֵ֤ין בָּאָ֙רֶץ֙ לָב֣וֹא עָלֵ֔ינוּ כְּדֶ֖רֶךְ כׇּל־הָאָֽרֶץ׃ אָמְרָה הַבְּכוֹרָה לַצְּעִירָה: אָבִינוּ זָקֵן וְאֵין בָּאָרֶץ אַף אִישׁ שֶׁיִּשָּׂא אוֹתָנוּ כְּדֶרֶךְ כָּל הָעוֹלָם.
לב לְכָ֨ה נַשְׁקֶ֧ה אֶת־אָבִ֛ינוּ יַ֖יִן וְנִשְׁכְּבָ֣ה עִמּ֑וֹ וּנְחַיֶּ֥ה מֵאָבִ֖ינוּ זָֽרַע׃ בּוֹאִי נִתֵּן לְאָבִינוּ לִשְׁתּוֹת יַיִן וְנִשְׁכַּב אִתּוֹ כְּדֵי שֶׁנֵּלֵד מֵאָבִינוּ בָּנִים.
לג וַתַּשְׁקֶ֧יןָ אֶת־אֲבִיהֶ֛ן יַ֖יִן בַּלַּ֣יְלָה ה֑וּא וַתָּבֹ֤א הַבְּכִירָה֙ וַתִּשְׁכַּ֣ב אֶת־אָבִ֔יהָ וְלֹֽא־יָדַ֥ע בְּשִׁכְבָ֖הּ וּבְקוּׄמָֽהּ׃ הֵן נָתְנוּ לַאֲבִיהֶן בְּאוֹתוֹ לַיְלָה לִשְׁתּוֹת יַיִן, הַבְּכוֹרָה בָּאָה וְשָׁכְבָה עִם אָבִיהָ וְהוּא לֹא יָדַע מִשְּׁכִיבָתָהּ וּמִקִּימָתָהּ.
לד וַֽיְהִי֙ מִֽמׇּחֳרָ֔ת וַתֹּ֤אמֶר הַבְּכִירָה֙ אֶל־הַצְּעִירָ֔ה הֵן־שָׁכַ֥בְתִּי אֶ֖מֶשׁ אֶת־אָבִ֑י נַשְׁקֶ֨נּוּ יַ֜יִן גַּם־הַלַּ֗יְלָה וּבֹ֙אִי֙ שִׁכְבִ֣י עִמּ֔וֹ וּנְחַיֶּ֥ה מֵאָבִ֖ינוּ זָֽרַע׃ לְמָחֳרָת הַבְּכוֹרָה אָמְרָה לַצְּעִירָה: הֲרֵי שָׁכַבְתִּי אֶמֶשׁ עִם אָבִי, נִתֵּן לוֹ לִשְׁתּוֹת יַיִן גַּם הַלַּיְלָה וּבוֹאִי וְשִׁכְבִי אִתּוֹ וְנֵלֵד בָּנִים מֵאָבִינוּ.
לה וַתַּשְׁקֶ֜יןָ גַּ֣ם בַּלַּ֧יְלָה הַה֛וּא אֶת־אֲבִיהֶ֖ן יָ֑יִן וַתָּ֤קׇם הַצְּעִירָה֙ וַתִּשְׁכַּ֣ב עִמּ֔וֹ וְלֹֽא־יָדַ֥ע בְּשִׁכְבָ֖הּ וּבְקֻמָֽהּ׃ גַּם בְּאוֹתוֹ לַיְלָה הֵן נָתְנוּ לַאֲבִיהֶן לִשְׁתּוֹת יַיִן, הַצְּעִירָה קָמָה וְשָׁכְבָה אִתּוֹ וְהוּא לֹא יָדַע מִשְּׁכִיבָתָהּ וּמִקִּימָתָהּ.
לו וַֽתַּהֲרֶ֛יןָ שְׁתֵּ֥י בְנֽוֹת־ל֖וֹט מֵאֲבִיהֶֽן׃ שְׁתֵּי בְּנוֹת לוֹט נִכְנְסוּ לְהֵרָיוֹן מֵאֲבִיהֶן.
לז וַתֵּ֤לֶד הַבְּכִירָה֙ בֵּ֔ן וַתִּקְרָ֥א שְׁמ֖וֹ מוֹאָ֑ב ה֥וּא אֲבִֽי־מוֹאָ֖ב עַד־הַיּֽוֹם׃ הַבְּכוֹרָה יָלְדָה בֵּן וְקָרְאָה אֶת שְׁמוֹ מוֹאָב, הוּא אֲבִי הָעָם מוֹאָב שֶׁקַּיָּמִים עַד הַיּוֹם.
לח וְהַצְּעִירָ֤ה גַם־הִוא֙ יָ֣לְדָה בֵּ֔ן וַתִּקְרָ֥א שְׁמ֖וֹ בֶּן־עַמִּ֑י ה֛וּא אֲבִ֥י בְנֵֽי־עַמּ֖וֹן עַד־הַיּֽוֹם׃ וְהַצְּעִירָה גַּם הִיא יָלְדָה בֵּן וְקָרְאָה אֶת שְׁמוֹ בֶּן עַמִּי, הוּא אֲבִי הָעָם עַמּוֹן שֶׁקַּיָּמִים עַד הַיּוֹם.
ס פָּרָשַׁת אֲבִימֶלֶךְ
כ א וַיִּסַּ֨ע מִשָּׁ֤ם אַבְרָהָם֙ אַ֣רְצָה הַנֶּ֔גֶב וַיֵּ֥שֶׁב בֵּין־קָדֵ֖שׁ וּבֵ֣ין שׁ֑וּר וַיָּ֖גׇר בִּגְרָֽר׃ מִשָּׁם נָסַע אַבְרָהָם לְאֶרֶץ הַנֶּגֶב וְהִתְיַשֵּׁב בֵּין קָדֵשׁ וּבֵין שׁוּר וְגָר בִּגְרָר.
ב וַיֹּ֧אמֶר אַבְרָהָ֛ם אֶל־שָׂרָ֥ה אִשְׁתּ֖וֹ אֲחֹ֣תִי הִ֑וא וַיִּשְׁלַ֗ח אֲבִימֶ֙לֶךְ֙ מֶ֣לֶךְ גְּרָ֔ר וַיִּקַּ֖ח אֶת־שָׂרָֽה׃ אַבְרָהָם אָמַר עַל שָׂרָה אִשְׁתּוֹ: "אֲחוֹתִי הִיא", וַאֲבִימֶלֶךְ מֶלֶךְ גְּרָר שָׁלַח שְׁלִיחִים וְלָקַח אֶת שָׂרָה.
ג וַיָּבֹ֧א אֱלֹהִ֛ים אֶל־אֲבִימֶ֖לֶךְ בַּחֲל֣וֹם הַלָּ֑יְלָה וַיֹּ֣אמֶר ל֗וֹ הִנְּךָ֥ מֵת֙ עַל־הָאִשָּׁ֣ה אֲשֶׁר־לָקַ֔חְתָּ וְהִ֖וא בְּעֻ֥לַת בָּֽעַל׃ אֱלֹהִים בָּא לַאֲבִימֶלֶךְ בַּחֲלוֹם הַלַּיְלָה וְאָמַר לוֹ: מַגִּיעַ לְךָ עוֹנֶשׁ מָוֶת עַל שֶׁלָּקַחְתָּ אֶת הָאִשָּׁה כִּי יֵשׁ לָהּ בַּעַל.
ד וַאֲבִימֶ֕לֶךְ לֹ֥א קָרַ֖ב אֵלֶ֑יהָ וַיֹּאמַ֕ר אֲדֹנָ֕י הֲג֥וֹי גַּם־צַדִּ֖יק תַּהֲרֹֽג׃ וַאֲבִימֶלֶךְ לֹא הִתְקָרֵב אֵלֶיהָ, וְאָמַר: ה'! הַאִם תַּהֲרוֹג גַּם עַם זַכַּאי כָּמוֹנוּ?!
ה הֲלֹ֨א ה֤וּא אָֽמַר־לִי֙ אֲחֹ֣תִי הִ֔וא וְהִֽיא־גַם־הִ֥וא אָֽמְרָ֖ה אָחִ֣י ה֑וּא בְּתׇם־לְבָבִ֛י וּבְנִקְיֹ֥ן כַּפַּ֖י עָשִׂ֥יתִי זֹֽאת׃ הֲרֵי הוּא אָמַר לִי: הִיא אֲחוֹתִי, וְגַם הִיא אָמְרָה: הוּא אָחִי, בְּתוֹם לֵב וּבְנִקְיוֹן כַּפַּיִם עָשִׂיתִי זֹאת!
ו וַיֹּ֩אמֶר֩ אֵלָ֨יו הָֽאֱלֹהִ֜ים בַּחֲלֹ֗ם גַּ֣ם אָנֹכִ֤י יָדַ֙עְתִּי֙ כִּ֤י בְתׇם־לְבָבְךָ֙ עָשִׂ֣יתָ זֹּ֔את וָאֶחְשֹׂ֧ךְ גַּם־אָנֹכִ֛י אֽוֹתְךָ֖ מֵחֲטוֹ־לִ֑י עַל־כֵּ֥ן לֹא־נְתַתִּ֖יךָ לִנְגֹּ֥עַ אֵלֶֽיהָ׃ אָמַר לוֹ הָאֱלֹהִים בַּחֲלוֹם: גַּם אֲנִי יָדַעְתִּי שֶׁעָשִׂיתָ זֹאת בְּתוֹם לֵב וְגַם אֲנִי מָנַעְתִּי מִמְּךָ לַחֲטוֹא לִי, לָכֵן לֹא נָתַתִּי לְךָ לִנְגּוֹעַ בָּהּ.
ז וְעַתָּ֗ה הָשֵׁ֤ב אֵֽשֶׁת־הָאִישׁ֙ כִּֽי־נָבִ֣יא ה֔וּא וְיִתְפַּלֵּ֥ל בַּֽעַדְךָ֖ וֶֽחְיֵ֑ה וְאִם־אֵֽינְךָ֣ מֵשִׁ֔יב דַּ֚ע כִּי־מ֣וֹת תָּמ֔וּת אַתָּ֖ה וְכׇל־אֲשֶׁר־לָֽךְ׃ וְעַכְשָׁיו הָשֵׁב אֶת אֵשֶׁת הָאִישׁ כִּי הוּא נָבִיא וְיִתְפַּלֵּל עָלֶיךָ וְתִחְיֶה, וְאִם אֵינְךָ מֵשִׁיב דַּע שֶׁתָּמוּת אַתָּה וְכָל מִי שֶׁשַּׁיָּךְ לְךָ.
ח וַיַּשְׁכֵּ֨ם אֲבִימֶ֜לֶךְ בַּבֹּ֗קֶר וַיִּקְרָא֙ לְכׇל־עֲבָדָ֔יו וַיְדַבֵּ֛ר אֶת־כׇּל־הַדְּבָרִ֥ים הָאֵ֖לֶּה בְּאׇזְנֵיהֶ֑ם וַיִּֽירְא֥וּ הָאֲנָשִׁ֖ים מְאֹֽד׃ אֲבִימֶלֶךְ הִשְׁכִּים בַּבּוֹקֶר, קָרָא לְכָל עֲבָדָיו, וְדִבֵּר אֶת כָּל הַדְּבָרִים הָאֵלֶּה בְּאָזְנֵיהֶם, וְהָאֲנָשִׁים פָּחֲדוּ מְאוֹד.
ט וַיִּקְרָ֨א אֲבִימֶ֜לֶךְ לְאַבְרָהָ֗ם וַיֹּ֨אמֶר ל֜וֹ מֶֽה־עָשִׂ֤יתָ לָּ֙נוּ֙ וּמֶֽה־חָטָ֣אתִי לָ֔ךְ כִּֽי־הֵבֵ֧אתָ עָלַ֛י וְעַל־מַמְלַכְתִּ֖י חֲטָאָ֣ה גְדֹלָ֑ה מַעֲשִׂים֙ אֲשֶׁ֣ר לֹא־יֵֽעָשׂ֔וּ עָשִׂ֖יתָ עִמָּדִֽי׃ אֲבִימֶלֶךְ קָרָא לְאַבְרָהָם וְאָמַר לוֹ: מֶה עָשִׂיתָ לָנוּ וּמֶה חָטָאתִי לְךָ שֶׁגָּרַמְתָּ לִי וּלְמַמְלַכְתִּי חֵטְא גָּדוֹל? מַעֲשִׂים שֶׁלֹּא יֵעָשׂוּ עָשִׂיתָ לִי!
י וַיֹּ֥אמֶר אֲבִימֶ֖לֶךְ אֶל־אַבְרָהָ֑ם מָ֣ה רָאִ֔יתָ כִּ֥י עָשִׂ֖יתָ אֶת־הַדָּבָ֥ר הַזֶּֽה׃ אֲבִימֶלֶךְ אָמַר לְאַבְרָהָם: מָה חָשַׁבְתָּ שֶׁעָשִׂיתָ אֶת הַדָּבָר הַזֶּה?
יא וַיֹּ֙אמֶר֙ אַבְרָהָ֔ם כִּ֣י אָמַ֗רְתִּי רַ֚ק אֵין־יִרְאַ֣ת אֱלֹהִ֔ים בַּמָּק֖וֹם הַזֶּ֑ה וַהֲרָג֖וּנִי עַל־דְּבַ֥ר אִשְׁתִּֽי׃ אָמַר אַבְרָהָם: כִּי אָמַרְתִּי שֶׁאֵין יִרְאַת אֱלֹהִים בַּמָּקוֹם הַזֶּה, וְיַהַרְגוּ אוֹתִי בִּשְׁבִיל לָקַחַת אֶת אִשְׁתִּי.
יב וְגַם־אׇמְנָ֗ה אֲחֹתִ֤י בַת־אָבִי֙ הִ֔וא אַ֖ךְ לֹ֣א בַת־אִמִּ֑י וַתְּהִי־לִ֖י לְאִשָּֽׁה׃ וְגַם אָמְנָם הִיא אֲחוֹתִי – הִיא בַּת הַבֵּן שֶׁל אָבִי, אֲבָל לֹא הַבַּת שֶׁל אִמִּי, וְהִיא נִשְׂאָה לִי.
יג וַיְהִ֞י כַּאֲשֶׁ֧ר הִתְע֣וּ אֹתִ֗י אֱלֹהִים֮ מִבֵּ֣ית אָבִי֒ וָאֹמַ֣ר לָ֔הּ זֶ֣ה חַסְדֵּ֔ךְ אֲשֶׁ֥ר תַּעֲשִׂ֖י עִמָּדִ֑י אֶ֤ל כׇּל־הַמָּקוֹם֙ אֲשֶׁ֣ר נָב֣וֹא שָׁ֔מָּה אִמְרִי־לִ֖י אָחִ֥י הֽוּא׃ וּכְשֶׁאֱלֹהִים הֶגְלָה אוֹתִי מִבֵּית אָבִי, אָמַרְתִּי לָהּ: זֶה הַחֶסֶד שֶׁתַּעֲשִׂי אִתִּי – בְּכָל מָקוֹם שֶׁנָּבוֹא לְשָׁם אִמְרִי עָלַי: "אָחִי הוּא".
יד וַיִּקַּ֨ח אֲבִימֶ֜לֶךְ צֹ֣אן וּבָקָ֗ר וַעֲבָדִים֙ וּשְׁפָחֹ֔ת וַיִּתֵּ֖ן לְאַבְרָהָ֑ם וַיָּ֣שֶׁב ל֔וֹ אֵ֖ת שָׂרָ֥ה אִשְׁתּֽוֹ׃ אֲבִימֶלֶךְ לָקַח צֹאן וּבָקָר וַעֲבָדִים וּשְׁפָחוֹת, וְנָתַן לְאַבְרָהָם, וְהוּא הֵשִׁיב לוֹ אֶת שָׂרָה אִשְׁתּוֹ.
טו וַיֹּ֣אמֶר אֲבִימֶ֔לֶךְ הִנֵּ֥ה אַרְצִ֖י לְפָנֶ֑יךָ בַּטּ֥וֹב בְּעֵינֶ֖יךָ שֵֽׁב׃ אֲבִימֶלֶךְ אָמַר: הִנֵּה אַרְצִי לְפָנֶיךָ, שֵׁב בְּאֵיזֶה מָקוֹם שֶׁטּוֹב בְּעֵינֶיךָ.
טז וּלְשָׂרָ֣ה אָמַ֗ר הִנֵּ֨ה נָתַ֜תִּי אֶ֤לֶף כֶּ֙סֶף֙ לְאָחִ֔יךְ הִנֵּ֤ה הוּא־לָךְ֙ כְּס֣וּת עֵינַ֔יִם לְכֹ֖ל אֲשֶׁ֣ר אִתָּ֑ךְ וְאֵ֥ת כֹּ֖ל וְנֹכָֽחַת׃ וּלְשָׂרָה אָמַר: הִנֵּה בַּהַתְחָלָה נָתַתִּי אֶלֶף כֶּסֶף לְאָחִיךְ תְּמוּרָתֵךְ, וְהִנֵּה הוּא בַּעֲלֵךְ! זֶה הוּכַח בִּפְנֵי כֻּלָּם, כִּי מֵאָז שֶׁנִּשֵּׂאת אַתְּ מִתְעַטֶּפֶת בְּמִטְפַּחַת לִצְנִיעוּת לְכִסּוּי עֵינַיִם בִּפְנֵי כָּל הָאֲנָשִׁים שֶׁאִתָּךְ.
יז וַיִּתְפַּלֵּ֥ל אַבְרָהָ֖ם אֶל־הָאֱלֹהִ֑ים וַיִּרְפָּ֨א אֱלֹהִ֜ים אֶת־אֲבִימֶ֧לֶךְ וְאֶת־אִשְׁתּ֛וֹ וְאַמְהֹתָ֖יו וַיֵּלֵֽדוּ׃ אַבְרָהָם הִתְפַּלֵּל לָאֱלֹהִים, וֵאלֹהִים רִפָּא אֶת אֲבִימֶלֶךְ וְאֶת אִשְׁתּוֹ וְשִׁפְחוֹתָיו וְהֵם יָלְדוּ.
יח כִּֽי־עָצֹ֤ר עָצַר֙ יְהֹוָ֔ה בְּעַ֥ד כׇּל־רֶ֖חֶם לְבֵ֣ית אֲבִימֶ֑לֶךְ עַל־דְּבַ֥ר שָׂרָ֖ה אֵ֥שֶׁת אַבְרָהָֽם׃ כִּי ה' עָצַר כָּל רֶחֶם בְּבֵית אֲבִימֶלֶךְ מִלָּלֶדֶת, בִּגְלַל לְקִיחַת שָׂרָה אֵשֶׁת אַבְרָהָם.
ס פָּרָשַׁת לֵידַת יִצְחָק
כא א וַֽיהֹוָ֛ה פָּקַ֥ד אֶת־שָׂרָ֖ה כַּאֲשֶׁ֣ר אָמָ֑ר וַיַּ֧עַשׂ יְהֹוָ֛ה לְשָׂרָ֖ה כַּאֲשֶׁ֥ר דִּבֵּֽר׃ וַה' זָכַר אֶת שָׂרָה כְּמוֹ שֶׁאָמָר, וְה' עָשָׂה לְשָׂרָה כְּמוֹ שֶׁדִּבֵּר.
ב וַתַּ֩הַר֩ וַתֵּ֨לֶד שָׂרָ֧ה לְאַבְרָהָ֛ם בֵּ֖ן לִזְקֻנָ֑יו לַמּוֹעֵ֕ד אֲשֶׁר־דִּבֶּ֥ר אֹת֖וֹ אֱלֹהִֽים׃ שָׂרָה נִכְנְסָה לְהֵרָיוֹן, וְיָלְדָה לְאַבְרָהָם בֵּן בִּימֵי זִקְנוּתוֹ, בַּזְּמָן שֶׁאֱלֹהִים אָמַר שֶׁתֵּלֵד.
ג וַיִּקְרָ֨א אַבְרָהָ֜ם אֶֽת־שֶׁם־בְּנ֧וֹ הַנּֽוֹלַד־ל֛וֹ אֲשֶׁר־יָלְדָה־לּ֥וֹ שָׂרָ֖ה יִצְחָֽק׃ אַבְרָהָם קָרָא אֶת שֵׁם בְּנוֹ שֶׁנּוֹלַד לוֹ שֶׁיָּלְדָה לוֹ שָׂרָה – יִצְחָק.
ד וַיָּ֤מׇל אַבְרָהָם֙ אֶת־יִצְחָ֣ק בְּנ֔וֹ בֶּן־שְׁמֹנַ֖ת יָמִ֑ים כַּאֲשֶׁ֛ר צִוָּ֥ה אֹת֖וֹ אֱלֹהִֽים׃ אַבְרָהָם מָל אֶת יִצְחָק בְּנוֹ בְּגִיל שְׁמוֹנָה יָמִים, כְּמוֹ שֶׁאֱלֹהִים צִוָּה אוֹתוֹ.

חמישי[עריכה]

ה וְאַבְרָהָ֖ם בֶּן־מְאַ֣ת שָׁנָ֑ה בְּהִוָּ֣לֶד ל֔וֹ אֵ֖ת יִצְחָ֥ק בְּנֽוֹ׃ וְאַבְרָהָם הָיָה בֶּן מֵאָה שָׁנָה כְּשֶׁנּוֹלַד לוֹ יִצְחָק בְּנוֹ.
ו וַתֹּ֣אמֶר שָׂרָ֔ה צְחֹ֕ק עָ֥שָׂה לִ֖י אֱלֹהִ֑ים כׇּל־הַשֹּׁמֵ֖עַ יִֽצְחַק־לִֽי׃ שָׂרָה אָמְרָה: שִׂמְחָה מַפְלִיאָה עָשָׂה לִי אֱלֹהִים – כָּל מִי שֶׁשּׁוֹמֵעַ מִתְפַּלֵּא עָלַי!
ז וַתֹּ֗אמֶר מִ֤י מִלֵּל֙ לְאַבְרָהָ֔ם הֵינִ֥יקָה בָנִ֖ים שָׂרָ֑ה כִּֽי־יָלַ֥דְתִּי בֵ֖ן לִזְקֻנָֽיו׃ וְאָמְרָה: מִי הֶאֱמִין שֶׁכָּךְ יִקְרֶה לְאַבְרָהָם – שֶׁשָּׂרָה מֵינִיקָה בָּנִים, הֲרֵי יָלַדְתִּי בֵּן בִּימֵי זִקְנוּתוֹ!
ח וַיִּגְדַּ֥ל הַיֶּ֖לֶד וַיִּגָּמַ֑ל וַיַּ֤עַשׂ אַבְרָהָם֙ מִשְׁתֶּ֣ה גָד֔וֹל בְּי֖וֹם הִגָּמֵ֥ל אֶת־יִצְחָֽק׃ הַיֶּלֶד גָּדַל וְנִגְמַל מֵהַיְּנִיקָה, אַבְרָהָם עָשָׂה מִשְׁתֶּה גָּדוֹל בַּיּוֹם שֶׁיִּצְחָק נִגְמַל.
פָּרָשַׁת גֵּרוּשׁ הָגָר
ט וַתֵּ֨רֶא שָׂרָ֜ה אֶֽת־בֶּן־הָגָ֧ר הַמִּצְרִ֛ית אֲשֶׁר־יָלְדָ֥ה לְאַבְרָהָ֖ם מְצַחֵֽק׃ שָׂרָה רָאֲתָה אֶת הַבֵּן שֶׁהָגָר הַמִּצְרִית יָלְדָה לְאַבְרָהָם מִתְנַהֵג בְּחוֹסֶר צְנִיעוּת.
י וַתֹּ֙אמֶר֙ לְאַבְרָהָ֔ם גָּרֵ֛שׁ הָאָמָ֥ה הַזֹּ֖את וְאֶת־בְּנָ֑הּ כִּ֣י לֹ֤א יִירַשׁ֙ בֶּן־הָאָמָ֣ה הַזֹּ֔את עִם־בְּנִ֖י עִם־יִצְחָֽק׃ הִיא אָמְרָה לְאַבְרָהָם: גָּרֵשׁ אֶת הַשִּׁפְחָה הַזֹּאת וְאֶת בְּנָהּ, כִּי הַבֵּן שֶׁל הָאָמָה הַזֹּאת אֵינוֹ רָאוּי לָרֶשֶׁת עִם בְּנִי יִצְחָק!
יא וַיֵּ֧רַע הַדָּבָ֛ר מְאֹ֖ד בְּעֵינֵ֣י אַבְרָהָ֑ם עַ֖ל אוֹדֹ֥ת בְּנֽוֹ׃ הַדָּבָר הָיָה רַע מְאוֹד בְּעֵינֵי אַבְרָהָם, עַל עִנְיַן בְּנוֹ.
יב וַיֹּ֨אמֶר אֱלֹהִ֜ים אֶל־אַבְרָהָ֗ם אַל־יֵרַ֤ע בְּעֵינֶ֙יךָ֙ עַל־הַנַּ֣עַר וְעַל־אֲמָתֶ֔ךָ כֹּל֩ אֲשֶׁ֨ר תֹּאמַ֥ר אֵלֶ֛יךָ שָׂרָ֖ה שְׁמַ֣ע בְּקֹלָ֑הּ כִּ֣י בְיִצְחָ֔ק יִקָּרֵ֥א לְךָ֖ זָֽרַע׃ אֱלֹהִים אָמַר לְאַבְרָהָם: שֶׁלֹּא יִהְיֶה רַע בְּעֵינֶיךָ עַל הַנַּעַר וְעַל שִׁפְחָתְךָ, אֶלָּא כָּל מַה שֶּׁשָּׂרָה תֹּאמַר לְךָ – שְׁמַע בְּקוֹלָהּ, כִּי רַק צֶאֱצָאִים שֶׁל יִצְחָק יֵחָשְׁבוּ לְצֶאֱצָאִים שֶׁלְּךָ.
יג וְגַ֥ם אֶת־בֶּן־הָאָמָ֖ה לְג֣וֹי אֲשִׂימֶ֑נּוּ כִּ֥י זַרְעֲךָ֖ הֽוּא׃ וְגַם מֵהַבֵּן שֶׁל הַשִּׁפְחָה אֶעֱשֶׂה עָם כִּי הוּא בִּנְךָ.
יד וַיַּשְׁכֵּ֣ם אַבְרָהָ֣ם ׀ בַּבֹּ֡קֶר וַיִּֽקַּֽח־לֶ֩חֶם֩ וְחֵ֨מַת מַ֜יִם וַיִּתֵּ֣ן אֶל־הָ֠גָ֠ר שָׂ֧ם עַל־שִׁכְמָ֛הּ וְאֶת־הַיֶּ֖לֶד וַֽיְשַׁלְּחֶ֑הָ וַתֵּ֣לֶךְ וַתֵּ֔תַע בְּמִדְבַּ֖ר בְּאֵ֥ר שָֽׁבַע׃ אַבְרָהָם הִשְׁכִּים בַּבּוֹקֶר, לָקַח לֶחֶם וּכְלִי מַיִם וְנָתַן לְהָגָר, הוּא שָׂם אוֹתָם עַל כְּתֵפָהּ, וְאֶת הַיֶּלֶד, וְגֵרֵשׁ אוֹתָהּ, הִיא הָלְכָה וְאִבְּדָה אֶת דַּרְכָּהּ בְּמִדְבַּר בְּאֵר שֶׁבַע.
טו וַיִּכְל֥וּ הַמַּ֖יִם מִן־הַחֵ֑מֶת וַתַּשְׁלֵ֣ךְ אֶת־הַיֶּ֔לֶד תַּ֖חַת אַחַ֥ד הַשִּׂיחִֽם׃ הַמַּיִם נִגְמְרוּ מֵהַכְּלִי, וְהִיא הִשְׁלִיכָה אֶת הַיֶּלֶד אֶל מִתַּחַת לְאַחַד הַשִּׂיחִים.
טז וַתֵּ֩לֶךְ֩ וַתֵּ֨שֶׁב לָ֜הּ מִנֶּ֗גֶד הַרְחֵק֙ כִּמְטַחֲוֵ֣י קֶ֔שֶׁת כִּ֣י אָֽמְרָ֔ה אַל־אֶרְאֶ֖ה בְּמ֣וֹת הַיָּ֑לֶד וַתֵּ֣שֶׁב מִנֶּ֔גֶד וַתִּשָּׂ֥א אֶת־קֹלָ֖הּ וַתֵּֽבְךְּ׃ הִיא הָלְכָה וְיָשְׁבָה מִמּוּל, הַרְחֵק כְּמֶרְחַק יֶרִי בְּקֶשֶׁת, כִּי אָמְרָה: שֶׁלֹּא אֶרְאֶה בְּמוֹת הַיֶּלֶד, יָשְׁבָה מִמּוּל וְהֵרִימָה אֶת קוֹלָהּ וּבָכְתָה.
יז וַיִּשְׁמַ֣ע אֱלֹהִים֮ אֶת־ק֣וֹל הַנַּ֒עַר֒ וַיִּקְרָא֩ מַלְאַ֨ךְ אֱלֹהִ֤ים ׀ אֶל־הָגָר֙ מִן־הַשָּׁמַ֔יִם וַיֹּ֥אמֶר לָ֖הּ מַה־לָּ֣ךְ הָגָ֑ר אַל־תִּ֣ירְאִ֔י כִּֽי־שָׁמַ֧ע אֱלֹהִ֛ים אֶל־ק֥וֹל הַנַּ֖עַר בַּאֲשֶׁ֥ר הוּא־שָֽׁם׃ אֱלֹהִים שָׁמַע אֶת קוֹל הַנַּעַר, מַלְאַךְ אֱלֹהִים קָרָא לְהָגָר מֵהַשָּׁמַיִם וְאָמַר לָהּ: מַה לָּךְ הָגָר? אַל תִּפְחֲדִי! כִּי אֱלֹהִים שָׁמַע לְקוֹל הַנַּעַר מֵהַמָּקוֹם שֶׁהוּא נִמְצָא בּוֹ.
יח ק֚וּמִי שְׂאִ֣י אֶת־הַנַּ֔עַר וְהַחֲזִ֥יקִי אֶת־יָדֵ֖ךְ בּ֑וֹ כִּֽי־לְג֥וֹי גָּד֖וֹל אֲשִׂימֶֽנּוּ׃ קוּמִי הֲרִימִי אֶת הַנַּעַר וְהַחֲזִיקִי בּוֹ בְּיָדֵךְ, כִּי אֶעֱשֶׂה אוֹתוֹ לְעָם גָּדוֹל.
יט וַיִּפְקַ֤ח אֱלֹהִים֙ אֶת־עֵינֶ֔יהָ וַתֵּ֖רֶא בְּאֵ֣ר מָ֑יִם וַתֵּ֜לֶךְ וַתְּמַלֵּ֤א אֶת־הַחֵ֙מֶת֙ מַ֔יִם וַתַּ֖שְׁקְ אֶת־הַנָּֽעַר׃ אֱלֹהִים פָּקַח אֶת עֵינֶיהָ וְהִיא רָאֲתָה בְּאֵר מַיִם, הִיא הָלְכָה וּמִלְּאָה אֶת הַכְּלִי בְּמַיִם וְהִשְׁקְתָה אֶת הַנָּעַר.
כ וַיְהִ֧י אֱלֹהִ֛ים אֶת־הַנַּ֖עַר וַיִּגְדָּ֑ל וַיֵּ֙שֶׁב֙ בַּמִּדְבָּ֔ר וַיְהִ֖י רֹבֶ֥ה קַשָּֽׁת׃ אֱלֹהִים הָיָה עִם הַנַּעַר לַעֲזוֹר לוֹ וְהוּא גָּדַל, הוּא גָּר בַּמִּדְבָּר וְהָיָה חָמוּשׁ יוֹרֶה בְּקֶשֶׁת.
כא וַיֵּ֖שֶׁב בְּמִדְבַּ֣ר פָּארָ֑ן וַתִּֽקַּֽח־ל֥וֹ אִמּ֛וֹ אִשָּׁ֖ה מֵאֶ֥רֶץ מִצְרָֽיִם׃ הוּא הִתְיַשֵּׁב בְּמִדְבַּר פָּארָן, וְאִמּוֹ לָקְחָה לוֹ אִשָּׁה מֵאֶרֶץ מִצְרַיִם.

ששי[עריכה]

פ פָּרָשַׁת בְּרִית בְּאֵר שֶׁבַע
כב וַֽיְהִי֙ בָּעֵ֣ת הַהִ֔וא וַיֹּ֣אמֶר אֲבִימֶ֗לֶךְ וּפִיכֹל֙ שַׂר־צְבָא֔וֹ אֶל־אַבְרָהָ֖ם לֵאמֹ֑ר אֱלֹהִ֣ים עִמְּךָ֔ בְּכֹ֥ל אֲשֶׁר־אַתָּ֖ה עֹשֶֽׂה׃ בְּאוֹתוֹ זְמָן, אָמְרוּ אֲבִימֶלֶךְ וּפִיכוֹל שַׂר הַצָּבָא שֶׁלּוֹ לְאַבְרָהָם: אֱלֹהִים עוֹשֶׂה שֶׁתַּצְלִיחַ בְּכָל מַה שֶּׁאַתָּה עוֹשֶׂה.
כג וְעַתָּ֗ה הִשָּׁ֨בְעָה לִּ֤י בֵֽאלֹהִים֙ הֵ֔נָּה אִם־תִּשְׁקֹ֣ר לִ֔י וּלְנִינִ֖י וּלְנֶכְדִּ֑י כַּחֶ֜סֶד אֲשֶׁר־עָשִׂ֤יתִי עִמְּךָ֙ תַּעֲשֶׂ֣ה עִמָּדִ֔י וְעִם־הָאָ֖רֶץ אֲשֶׁר־גַּ֥רְתָּה בָּֽהּ׃ וְעַכְשָׁיו הִשָּׁבַע לִי בֵּאלֹהִים כָּאן, וְלֹא תְּשַׁקֵּר לִי וְלִבְנִי וּלְנֶכְדִּי שֶׁכַּחֶסֶד שֶׁעָשִׂיתִי אִתְּךָ תַּעֲשֶׂה אִתִּי וְעִם תּוֹשָׁבֵי הָאָרֶץ שֶׁגַּרְתָּ בָּהּ.
כד וַיֹּ֙אמֶר֙ אַבְרָהָ֔ם אָנֹכִ֖י אִשָּׁבֵֽעַ׃ אָמַר אַבְרָהָם: אֲנִי אֶשָּׁבַע.
כה וְהוֹכִ֥חַ אַבְרָהָ֖ם אֶת־אֲבִימֶ֑לֶךְ עַל־אֹדוֹת֙ בְּאֵ֣ר הַמַּ֔יִם אֲשֶׁ֥ר גָּזְל֖וּ עַבְדֵ֥י אֲבִימֶֽלֶךְ׃ וְאַבְרָהָם הֵעִיר לַאֲבִימֶלֶךְ עַל עִנְיַן בְּאֵר הַמַּיִם שֶׁעַבְדֵי אֲבִימֶלֶךְ גָּזְלוּ.
כו וַיֹּ֣אמֶר אֲבִימֶ֔לֶךְ לֹ֣א יָדַ֔עְתִּי מִ֥י עָשָׂ֖ה אֶת־הַדָּבָ֣ר הַזֶּ֑ה וְגַם־אַתָּ֞ה לֹא־הִגַּ֣דְתָּ לִּ֗י וְגַ֧ם אָנֹכִ֛י לֹ֥א שָׁמַ֖עְתִּי בִּלְתִּ֥י הַיּֽוֹם׃ אָמַר אֲבִימֶלֶךְ: לֹא יָדַעְתִּי מִי עָשָׂה אֶת הַדָּבָר הַזֶּה, וְגַם אַתָּה לֹא סִפַּרְתָּ לִי עַל זֶה, וְגַם לֹא שָׁמַעְתִּי עַל זֶה עַד הַיּוֹם.
כז וַיִּקַּ֤ח אַבְרָהָם֙ צֹ֣אן וּבָקָ֔ר וַיִּתֵּ֖ן לַאֲבִימֶ֑לֶךְ וַיִּכְרְת֥וּ שְׁנֵיהֶ֖ם בְּרִֽית׃ אַבְרָהָם לָקַח צֹאן וּבָקָר וְנָתַן לַאֲבִימֶלֶךְ, וּשְׁנֵיהֶם עָשׂוּ הֶסְכֵּם.
כח וַיַּצֵּ֣ב אַבְרָהָ֗ם אֶת־שֶׁ֛בַע כִּבְשֹׂ֥ת הַצֹּ֖אן לְבַדְּהֶֽן׃ אַבְרָהָם הֶעֱמִיד אֶת שֶׁבַע כִּבְשׂוֹת הַצֹּאן שֶׁנָּתַן לַאֲבִימֶלֶךְ, לְבַד.
כט וַיֹּ֥אמֶר אֲבִימֶ֖לֶךְ אֶל־אַבְרָהָ֑ם מָ֣ה הֵ֗נָּה שֶׁ֤בַע כְּבָשֹׂת֙ הָאֵ֔לֶּה אֲשֶׁ֥ר הִצַּ֖בְתָּ לְבַדָּֽנָה׃ אָמַר אֲבִימֶלֶךְ לְאַבְרָהָם: מָה הֵן שֶׁבַע הַכְּבָשׂוֹת הָאֵלֶּה שֶׁהֶעֱמַדְתָּ לְבַד?
ל וַיֹּ֕אמֶר כִּ֚י אֶת־שֶׁ֣בַע כְּבָשֹׂ֔ת תִּקַּ֖ח מִיָּדִ֑י בַּעֲבוּר֙ תִּֽהְיֶה־לִּ֣י לְעֵדָ֔ה כִּ֥י חָפַ֖רְתִּי אֶת־הַבְּאֵ֥ר הַזֹּֽאת׃ אָמַר לוֹ: הֶעֱמַדְתִּי אוֹתָן לְבַד, כִּי אֶת שֶׁבַע הַכְּבָשׂוֹת הָאֵלֶּה תִּקַּח מִמֶּנִּי כְּדֵי שֶׁזֶּה יִהְיֶה סִימָן לְטוֹבָתִי שֶׁאֲנִי חָפַרְתִּי אֶת הַבְּאֵר הַזֹּאת.
לא עַל־כֵּ֗ן קָרָ֛א לַמָּק֥וֹם הַה֖וּא בְּאֵ֣ר שָׁ֑בַע כִּ֛י שָׁ֥ם נִשְׁבְּע֖וּ שְׁנֵיהֶֽם׃ לָכֵן הוּא קָרָא לַמָּקוֹם הַהוּא בְּאֵר שֶׁבַע, כִּי שָׁם שְׁנֵיהֶם נִשְׁבְּעוּ.
לב וַיִּכְרְת֥וּ בְרִ֖ית בִּבְאֵ֣ר שָׁ֑בַע וַיָּ֣קׇם אֲבִימֶ֗לֶךְ וּפִיכֹל֙ שַׂר־צְבָא֔וֹ וַיָּשֻׁ֖בוּ אֶל־אֶ֥רֶץ פְּלִשְׁתִּֽים׃ הֵם עָשׂוּ הֶסְכֵּם בִּבְאֵר שֶׁבַע, אֲבִימֶלֶךְ וּפִיכוֹל שַׂר הַצָּבָא שֶׁלּוֹ קָמוּ, וְחָזְרוּ לְאֶרֶץ פְּלִשְׁתִּים.
לג וַיִּטַּ֥ע אֶ֖שֶׁל בִּבְאֵ֣ר שָׁ֑בַע וַיִּ֨קְרָא־שָׁ֔ם בְּשֵׁ֥ם יְהֹוָ֖ה אֵ֥ל עוֹלָֽם׃ אַבְרָהָם נָטַע עֵץ בִּבְאֵר שֶׁבַע וְהִתְפַּלֵּל שָׁם אֶל ה' – אֵל הָעוֹלָם.
לד וַיָּ֧גׇר אַבְרָהָ֛ם בְּאֶ֥רֶץ פְּלִשְׁתִּ֖ים יָמִ֥ים רַבִּֽים׃ אַבְרָהָם גַּר בְּאֶרֶץ פְּלִשְׁתִּים זְמָן רַב.

שביעי[עריכה]

פ פָּרָשַׁת הָעֲקֵידָה
כב א וַיְהִ֗י אַחַר֙ הַדְּבָרִ֣ים הָאֵ֔לֶּה וְהָ֣אֱלֹהִ֔ים נִסָּ֖ה אֶת־אַבְרָהָ֑ם וַיֹּ֣אמֶר אֵלָ֔יו אַבְרָהָ֖ם וַיֹּ֥אמֶר הִנֵּֽנִי׃ אַחֲרֵי הַדְּבָרִים הָאֵלֶּה הָאֱלֹהִים נִסָּה אֶת אַבְרָהָם אִם יִשְׁמַע בְּקוֹלוֹ, וְאָמַר לוֹ: אַבְרָהָם! אָמַר לוֹ: הִנֵּה אֲנִי!
ב וַיֹּ֡אמֶר קַח־נָ֠א אֶת־בִּנְךָ֨ אֶת־יְחִֽידְךָ֤ אֲשֶׁר־אָהַ֙בְתָּ֙ אֶת־יִצְחָ֔ק וְלֶ֨ךְ־לְךָ֔ אֶל־אֶ֖רֶץ הַמֹּרִיָּ֑ה וְהַעֲלֵ֤הוּ שָׁם֙ לְעֹלָ֔ה עַ֚ל אַחַ֣ד הֶֽהָרִ֔ים אֲשֶׁ֖ר אֹמַ֥ר אֵלֶֽיךָ׃ אָמַר לוֹ: קַח בְּבַקָּשָׁה אֶת בִּנְךָ יְחִידְךָ שֶׁאָהַבְתָּ – אֶת יִצְחָק, וְלֵךְ לְאֶרֶץ הַמּוֹרִיָּה, וְתַעֲלֵה אוֹתוֹ שָׁם לְעוֹלָה עַל אַחַד הֶהָרִים שֶׁאוֹמַר לְךָ!
ג וַיַּשְׁכֵּ֨ם אַבְרָהָ֜ם בַּבֹּ֗קֶר וַֽיַּחֲבֹשׁ֙ אֶת־חֲמֹר֔וֹ וַיִּקַּ֞ח אֶת־שְׁנֵ֤י נְעָרָיו֙ אִתּ֔וֹ וְאֵ֖ת יִצְחָ֣ק בְּנ֑וֹ וַיְבַקַּע֙ עֲצֵ֣י עֹלָ֔ה וַיָּ֣קׇם וַיֵּ֔לֶךְ אֶל־הַמָּק֖וֹם אֲשֶׁר־אָֽמַר־ל֥וֹ הָאֱלֹהִֽים׃ אַבְרָהָם הִשְׁכִּים בַּבּוֹקֶר רָתַם אֶת חֲמוֹרוֹ לָקַח אֶת שְׁנֵי מְשָׁרְתָיו אִתּוֹ וְאֶת יִצְחָק בְּנוֹ, חָטַב עֵצִים לְקָרְבַּן עוֹלָה, קָם וְהָלַךְ לַמָּקוֹם שֶׁהָאֱלֹהִים אָמַר לוֹ.
ד בַּיּ֣וֹם הַשְּׁלִישִׁ֗י וַיִּשָּׂ֨א אַבְרָהָ֧ם אֶת־עֵינָ֛יו וַיַּ֥רְא אֶת־הַמָּק֖וֹם מֵרָחֹֽק׃ בַּיּוֹם הַשְּׁלִישִׁי אַבְרָהָם הֵרִים אֶת מַבָּטוֹ, וְרָאָה אֶת הַמָּקוֹם מֵרָחוֹק.
ה וַיֹּ֨אמֶר אַבְרָהָ֜ם אֶל־נְעָרָ֗יו שְׁבוּ־לָכֶ֥ם פֹּה֙ עִֽם־הַחֲמ֔וֹר וַאֲנִ֣י וְהַנַּ֔עַר נֵלְכָ֖ה עַד־כֹּ֑ה וְנִֽשְׁתַּחֲוֶ֖ה וְנָשׁ֥וּבָה אֲלֵיכֶֽם׃ אָמַר אַבְרָהָם לִנְעָרָיו: שְׁבוּ לָכֶם פֹּה עִם הַחֲמוֹר וַאֲנִי וְהַנַּעַר נֵלֵךְ עוֹד מֶרְחָק מְסוּיָם, וְנִשְׁתַּחֲוֶה לַה' וְנַחֲזוֹר אֲלֵיכֶם.
ו וַיִּקַּ֨ח אַבְרָהָ֜ם אֶת־עֲצֵ֣י הָעֹלָ֗ה וַיָּ֙שֶׂם֙ עַל־יִצְחָ֣ק בְּנ֔וֹ וַיִּקַּ֣ח בְּיָד֔וֹ אֶת־הָאֵ֖שׁ וְאֶת־הַֽמַּאֲכֶ֑לֶת וַיֵּלְכ֥וּ שְׁנֵיהֶ֖ם יַחְדָּֽו׃ אַבְרָהָם לָקַח אֶת עֲצֵי הָעוֹלָה וְהֶעֱמִיס אוֹתָם עַל יִצְחָק בְּנוֹ, וְלָקַח בְּיָדוֹ אֶת הָאֵשׁ וְאֶת הַסַּכִּין, וּשְׁנֵיהֶם הָלְכוּ יַחַד.
ז וַיֹּ֨אמֶר יִצְחָ֜ק אֶל־אַבְרָהָ֤ם אָבִיו֙ וַיֹּ֣אמֶר אָבִ֔י וַיֹּ֖אמֶר הִנֶּ֣נִּֽי בְנִ֑י וַיֹּ֗אמֶר הִנֵּ֤ה הָאֵשׁ֙ וְהָ֣עֵצִ֔ים וְאַיֵּ֥ה הַשֶּׂ֖ה לְעֹלָֽה׃ אָמַר יִצְחָק לְאַבְרָהָם אָבִיו: אָבִי! אָמַר לוֹ: הִנֵּה אֲנִי בְּנִי! אָמַר לוֹ: הִנֵּה הָאֵשׁ וְהָעֵצִים וְאֵיפֹה הַגְּדִי לְעוֹלָה?
ח וַיֹּ֙אמֶר֙ אַבְרָהָ֔ם אֱלֹהִ֞ים יִרְאֶה־לּ֥וֹ הַשֶּׂ֛ה לְעֹלָ֖ה בְּנִ֑י וַיֵּלְכ֥וּ שְׁנֵיהֶ֖ם יַחְדָּֽו׃ אָמַר אַבְרָהָם: אֱלֹהִים יִבְחַר לְעַצְמוֹ אֶת הַגְּדִי לְעוֹלָה, בְּנִי, וּשְׁנֵיהֶם הָלְכוּ יַחַד.
ט וַיָּבֹ֗אוּ אֶֽל־הַמָּקוֹם֮ אֲשֶׁ֣ר אָֽמַר־ל֣וֹ הָאֱלֹהִים֒ וַיִּ֨בֶן שָׁ֤ם אַבְרָהָם֙ אֶת־הַמִּזְבֵּ֔חַ וַֽיַּעֲרֹ֖ךְ אֶת־הָעֵצִ֑ים וַֽיַּעֲקֹד֙ אֶת־יִצְחָ֣ק בְּנ֔וֹ וַיָּ֤שֶׂם אֹתוֹ֙ עַל־הַמִּזְבֵּ֔חַ מִמַּ֖עַל לָעֵצִֽים׃ הֵם בָּאוּ לַמָּקוֹם שֶׁהָאֱלֹהִים אָמַר לוֹ, אַבְרָהָם בָּנָה שָׁם אֶת הַמִּזְבֵּחַ וְסִדֵּר אֶת הָעֵצִים, וְקָשַׁר אֶת יִצְחָק בְּנוֹ וְשָׂם אוֹתוֹ עַל הַמִּזְבֵּחַ מֵעַל הָעֵצִים.
י וַיִּשְׁלַ֤ח אַבְרָהָם֙ אֶת־יָד֔וֹ וַיִּקַּ֖ח אֶת־הַֽמַּאֲכֶ֑לֶת לִשְׁחֹ֖ט אֶת־בְּנֽוֹ׃ אַבְרָהָם הוֹשִׁיט אֶת יָדוֹ וְלָקַח אֶת הַסַּכִּין לִשְׁחוֹט אֶת בְּנוֹ.
יא וַיִּקְרָ֨א אֵלָ֜יו מַלְאַ֤ךְ יְהֹוָה֙ מִן־הַשָּׁמַ֔יִם וַיֹּ֖אמֶר אַבְרָהָ֣ם ׀ אַבְרָהָ֑ם וַיֹּ֖אמֶר הִנֵּֽנִי׃ קָרָא לוֹ מַלְאַךְ ה' מֵהַשָּׁמַיִם, וְאָמַר: אַבְרָהָם אַבְרָהָם! אָמַר לוֹ: הִנֵּה אֲנִי!
יב וַיֹּ֗אמֶר אַל־תִּשְׁלַ֤ח יָֽדְךָ֙ אֶל־הַנַּ֔עַר וְאַל־תַּ֥עַשׂ ל֖וֹ מְא֑וּמָה כִּ֣י ׀ עַתָּ֣ה יָדַ֗עְתִּי כִּֽי־יְרֵ֤א אֱלֹהִים֙ אַ֔תָּה וְלֹ֥א חָשַׂ֛כְתָּ אֶת־בִּנְךָ֥ אֶת־יְחִידְךָ֖ מִמֶּֽנִּי׃ אָמַר לוֹ: אַל תּוֹשִׁיט אֶת יָדְךָ לַנַּעַר וְאַל תַּעֲשֶׂה לוֹ כְּלוּם! כִּי עַכְשָׁיו אֲנִי יוֹדֵעַ שֶׁבֶּאֱמֶת אַתָּה יְרֵא אֱלֹהִים, וְלֹא נִמְנַעְתָּ מִלְּהַקְרִיב לִי אֶת בִּנְךָ הַיָּחִיד.
יג וַיִּשָּׂ֨א אַבְרָהָ֜ם אֶת־עֵינָ֗יו וַיַּרְא֙ וְהִנֵּה־אַ֔יִל אַחַ֕ר נֶאֱחַ֥ז בַּסְּבַ֖ךְ בְּקַרְנָ֑יו וַיֵּ֤לֶךְ אַבְרָהָם֙ וַיִּקַּ֣ח אֶת־הָאַ֔יִל וַיַּעֲלֵ֥הוּ לְעֹלָ֖ה תַּ֥חַת בְּנֽוֹ׃ אַבְרָהָם הֵרִים אֶת מַבָּטוֹ, וְרָאָה וְהִנֵּה אַיִל מֵאֲחוֹרֵי סְבַךְ הַשִּׂיחִים וְהוּא מִסְתַּבֵּךְ בָּהֶם בְּקַרְנָיו, אַבְרָהָם הָלַךְ וְלָקַח אֶת הָאַיִל, וְהֶעֱלָה אוֹתוֹ לְעוֹלָה בִּמְקוֹם בְּנוֹ.
יד וַיִּקְרָ֧א אַבְרָהָ֛ם שֵֽׁם־הַמָּק֥וֹם הַה֖וּא יְהֹוָ֣ה ׀ יִרְאֶ֑ה אֲשֶׁר֙ יֵאָמֵ֣ר הַיּ֔וֹם בְּהַ֥ר יְהֹוָ֖ה יֵרָאֶֽה׃ אַבְרָהָם קָרָא אֶת שֵׁם הַמָּקוֹם הַהוּא "ה' רוֹאֶה", כְּמוֹ שֶׁאוֹמְרִים הַיּוֹם: בְּהַר ה' יֵרָאֶה כָּל זְכוּרְךָ לִפְנֵי ה'.
טו וַיִּקְרָ֛א מַלְאַ֥ךְ יְהֹוָ֖ה אֶל־אַבְרָהָ֑ם שֵׁנִ֖ית מִן־הַשָּׁמָֽיִם׃ מַלְאַךְ ה' קָרָא לְאַבְרָהָם מֵהַשָּׁמַיִם, שֵׁנִית.
טז וַיֹּ֕אמֶר בִּ֥י נִשְׁבַּ֖עְתִּי נְאֻם־יְהֹוָ֑ה כִּ֗י יַ֚עַן אֲשֶׁ֤ר עָשִׂ֙יתָ֙ אֶת־הַדָּבָ֣ר הַזֶּ֔ה וְלֹ֥א חָשַׂ֖כְתָּ אֶת־בִּנְךָ֥ אֶת־יְחִידֶֽךָ׃ וְאָמַר: נִשְׁבַּעְתִּי בִּשְׁמִי, נְאֻם ה', שֶׁבִּזְכוּת שֶׁעָשִׂיתָ אֶת הַדָּבָר הַזֶּה – שֶׁלֹּא נִמְנַעְתָּ מִלְּהַקְרִיב אֶת בִּנְךָ הַיָּחִיד.
יז כִּֽי־בָרֵ֣ךְ אֲבָרֶכְךָ֗ וְהַרְבָּ֨ה אַרְבֶּ֤ה אֶֽת־זַרְעֲךָ֙ כְּכוֹכְבֵ֣י הַשָּׁמַ֔יִם וְכַח֕וֹל אֲשֶׁ֖ר עַל־שְׂפַ֣ת הַיָּ֑ם וְיִרַ֣שׁ זַרְעֲךָ֔ אֵ֖ת שַׁ֥עַר אֹיְבָֽיו׃ לָכֵן אֲבָרֵךְ אוֹתְךָ, וְאַרְבֶּה אֶת צֶאֱצָאֶיךָ כְּכוֹכְבֵי הַשָּׁמַיִם וְכַחוֹל שֶׁעַל שְׂפַת הַיָּם, וְצֶאֱצָאֶיךָ יִכְבְּשׁוּ אֶת עָרֵי אוֹיְבֵיהֶם.
יח וְהִתְבָּרְﬞכ֣וּ בְזַרְעֲךָ֔ כֹּ֖ל גּוֹיֵ֣י הָאָ֑רֶץ עֵ֕קֶב אֲשֶׁ֥ר שָׁמַ֖עְתָּ בְּקֹלִֽי׃ וּבִזְכוּת צֶאֱצָאֶיךָ יִתְבָּרְכוּ כָּל עַמֵּי הָאָרֶץ, בִּזְכוּת שֶׁשָּׁמַעְתָּ בְּקוֹלִי.
יט וַיָּ֤שׇׁב אַבְרָהָם֙ אֶל־נְעָרָ֔יו וַיָּקֻ֛מוּ וַיֵּלְכ֥וּ יַחְדָּ֖ו אֶל־בְּאֵ֣ר שָׁ֑בַע וַיֵּ֥שֶׁב אַבְרָהָ֖ם בִּבְאֵ֥ר שָֽׁבַע׃ אַבְרָהָם חָזַר אֶל נְעָרָיו וְהֵם קָמוּ וְהָלְכוּ יַחַד לִבְאֵר שֶׁבַע, וְאַבְרָהָם גָּר בִּבְאֵר שֶׁבַע.

מפטיר[עריכה]

פ פָּרָשַׁת בְּנֵי נָחוֹר
כ וַיְהִ֗י אַחֲרֵי֙ הַדְּבָרִ֣ים הָאֵ֔לֶּה וַיֻּגַּ֥ד לְאַבְרָהָ֖ם לֵאמֹ֑ר הִ֠נֵּ֠ה יָלְדָ֨ה מִלְכָּ֥ה גַם־הִ֛וא בָּנִ֖ים לְנָח֥וֹר אָחִֽיךָ׃ אַחֲרֵי הַדְּבָרִים הָאֵלֶּה, סִפְּרוּ לְאַבְרָהָם וְאָמְרוּ לוֹ: הִנֵּה, גַּם מִלְכָּה יָלְדָה בָּנִים לְנָחוֹר אָחִיךָ.
כא אֶת־ע֥וּץ בְּכֹר֖וֹ וְאֶת־בּ֣וּז אָחִ֑יו וְאֶת־קְמוּאֵ֖ל אֲבִ֥י אֲרָֽם׃ אֶת עוּץ בְּכוֹרוֹ וְאֶת בּוּז אָחִיו וְאֶת קְמוּאֵל – אָבִיו שֶׁל אֲרָם.
כב וְאֶת־כֶּ֣שֶׂד וְאֶת־חֲז֔וֹ וְאֶת־פִּלְדָּ֖שׁ וְאֶת־יִדְלָ֑ף וְאֵ֖ת בְּתוּאֵֽל׃ וְאֶת כֶּשֶׂד וְאֶת חֲזוֹ וְאֶת פִּלְדָּשׁ וְאֶת יִדְלָף וְאֶת בְּתוּאֵל.
כג וּבְתוּאֵ֖ל יָלַ֣ד אֶת־רִבְקָ֑ה שְׁמֹנָ֥ה אֵ֙לֶּה֙ יָלְדָ֣ה מִלְכָּ֔ה לְנָח֖וֹר אֲחִ֥י אַבְרָהָֽם׃ וּבְתוּאֵל הוֹלִיד אֶת רִבְקָה, אֶת הַשְּׁמוֹנָה הָאֵלֶּה יָלְדָה מִלְכָּה לְנָחוֹר אֲחִי אַבְרָהָם.
כד וּפִֽילַגְשׁ֖וֹ וּשְׁמָ֣הּ רְאוּמָ֑ה וַתֵּ֤לֶד גַּם־הִוא֙ אֶת־טֶ֣בַח וְאֶת־גַּ֔חַם וְאֶת־תַּ֖חַשׁ וְאֶֽת־מַעֲכָֽה׃ וְשִׁפְחָתוֹ שֶׁשְּׁמָהּ רְאוּמָה יָלְדָה גַּם הִיא – אֶת טֶבַח וְאֶת גַּחַם וְאֶת תַּחַשׁ וְאֶת מַעֲכָה.
