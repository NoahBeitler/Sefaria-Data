ראשון[עריכה]

פ פָּרָשַׁת הַמְּרַגְּלִים
יג א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב שְׁלַח־לְךָ֣ אֲנָשִׁ֗ים וְיָתֻ֙רוּ֙ אֶת־אֶ֣רֶץ כְּנַ֔עַן אֲשֶׁר־אֲנִ֥י נֹתֵ֖ן לִבְנֵ֣י יִשְׂרָאֵ֑ל אִ֣ישׁ אֶחָד֩ אִ֨ישׁ אֶחָ֜ד לְמַטֵּ֤ה אֲבֹתָיו֙ תִּשְׁלָ֔חוּ כֹּ֖ל נָשִׂ֥יא בָהֶֽם׃ שְׁלַח אֲנָשִׁים שֶׁיְּתַיְּרוּ בְּאֶרֶץ כְּנַעַן שֶׁאֲנִי נוֹתֵן לִבְנֵי יִשְׂרָאֵל, תִּשְׁלְחוּ אִישׁ אֶחָד מִכָּל שֵׁבֶט – כָּל אֶחָד נִכְבָּד בְּשִׁבְטוֹ.
ג וַיִּשְׁלַ֨ח אֹתָ֥ם מֹשֶׁ֛ה מִמִּדְבַּ֥ר פָּארָ֖ן עַל־פִּ֣י יְהֹוָ֑ה כֻּלָּ֣ם אֲנָשִׁ֔ים רָאשֵׁ֥י בְנֵֽי־יִשְׂרָאֵ֖ל הֵֽמָּה׃ מֹשֶׁה שָׁלַח אוֹתָם מִמִּדְבַּר פָּארָן עַל פִּי דִּבְרֵי ה', כֻּלָּם אֲנָשִׁים חֲשׁוּבִים – הֵם רָאשֵׁי בְּנֵי יִשְׂרָאֵל.
ד וְאֵ֖לֶּה שְׁמוֹתָ֑ם לְמַטֵּ֣ה רְאוּבֵ֔ן שַׁמּ֖וּעַ בֶּן־זַכּֽוּר׃ וְאֵלֶּה שְׁמוֹתָם, לְשֵׁבֶט רְאוּבֵן שַׁמּוּעַ בֶּן זַכּוּר.
ה לְמַטֵּ֣ה שִׁמְע֔וֹן שָׁפָ֖ט בֶּן־חוֹרִֽי׃ לְשֵׁבֶט שִׁמְעוֹן שָׁפָט בֶּן חוֹרִי.
ו לְמַטֵּ֣ה יְהוּדָ֔ה כָּלֵ֖ב בֶּן־יְפֻנֶּֽה׃ לְשֵׁבֶט יְהוּדָה כָּלֵב בֶּן יְפֻנֶּה.
ז לְמַטֵּ֣ה יִשָּׂשכָ֔ר יִגְאָ֖ל בֶּן־יוֹסֵֽף׃ לְשֵׁבֶט יִשָּׂשכָר יִגְאָל בֶּן יוֹסֵף.
ח לְמַטֵּ֥ה אֶפְרָ֖יִם הוֹשֵׁ֥עַ בִּן־נֽוּן׃ לְשֵׁבֶט אֶפְרַיִם הוֹשֵׁעַ בֶּן נוּן.
ט לְמַטֵּ֣ה בִנְיָמִ֔ן פַּלְטִ֖י בֶּן־רָפֽוּא׃ לְשֵׁבֶט בִּנְיָמִין פַּלְטִי בֶּן רָפוּא.
י לְמַטֵּ֣ה זְבוּלֻ֔ן גַּדִּיאֵ֖ל בֶּן־סוֹדִֽי׃ לְשֵׁבֶט זְבוּלוּן גַּדִּיאֵל בֶּן סוֹדִי.
יא לְמַטֵּ֥ה יוֹסֵ֖ף לְמַטֵּ֣ה מְנַשֶּׁ֑ה גַּדִּ֖י בֶּן־סוּסִֽי׃ בְּשֵׁבֶט יוֹסֵף: לְשֵׁבֶט מְנַשֶּׁה גַּדִּי בֶּן סוּסִי.
יב לְמַטֵּ֣ה דָ֔ן עַמִּיאֵ֖ל בֶּן־גְּמַלִּֽי׃ לְשֵׁבֶט דָּן עַמִּיאֵל בֶּן גְּמַלִּי.
יג לְמַטֵּ֣ה אָשֵׁ֔ר סְת֖וּר בֶּן־מִיכָאֵֽל׃ לְשֵׁבֶט אָשֵׁר סְתוּר בֶּן מִיכָאֵל.
יד לְמַטֵּ֣ה נַפְתָּלִ֔י נַחְבִּ֖י בֶּן־וׇפְסִֽי׃ לְשֵׁבֶט נַפְתָּלִי נַחְבִּי בֶּן וָפְסִי.
טו לְמַטֵּ֣ה גָ֔ד גְּאוּאֵ֖ל בֶּן־מָכִֽי׃ לְשֵׁבֶט גָּד גְּאוּאֵל בֶּן מָכִי.
טז אֵ֚לֶּה שְׁמ֣וֹת הָֽאֲנָשִׁ֔ים אֲשֶׁר־שָׁלַ֥ח מֹשֶׁ֖ה לָת֣וּר אֶת־הָאָ֑רֶץ וַיִּקְרָ֥א מֹשֶׁ֛ה לְהוֹשֵׁ֥עַ בִּן־נ֖וּן יְהוֹשֻֽׁעַ׃ אֵלֶּה שְׁמוֹת הָאֲנָשִׁים שֶׁמֹּשֶׁה שָׁלַח לְתַיֵּר בָּאָרֶץ, מֹשֶׁה קָרָא לְהוֹשֵׁעַ בִּן נוּן יְהוֹשֻׁעַ.
יז וַיִּשְׁלַ֤ח אֹתָם֙ מֹשֶׁ֔ה לָת֖וּר אֶת־אֶ֣רֶץ כְּנָ֑עַן וַיֹּ֣אמֶר אֲלֵהֶ֗ם עֲל֥וּ זֶה֙ בַּנֶּ֔גֶב וַעֲלִיתֶ֖ם אֶת־הָהָֽר׃ מֹשֶׁה שָׁלַח אוֹתָם לְתַיֵּר בְּאֶרֶץ כְּנַעַן וְאָמַר לָהֶם: עֲלוּ דֶּרֶךְ הַנֶּגֶב וַעֲלוּ בָּהָר.
יח וּרְאִיתֶ֥ם אֶת־הָאָ֖רֶץ מַה־הִ֑וא וְאֶת־הָעָם֙ הַיֹּשֵׁ֣ב עָלֶ֔יהָ הֶחָזָ֥ק הוּא֙ הֲרָפֶ֔ה הַמְעַ֥ט ה֖וּא אִם־רָֽב׃ וְתִרְאוּ אֶת הָאָרֶץ אֵיךְ הִיא, וְאֶת הָעָם שֶׁגָּר בָּהּ – אִם הוּא חָזָק אוֹ חַלָּשׁ, אִם הֵם מְעַטִּים אוֹ רַבִּים.
יט וּמָ֣ה הָאָ֗רֶץ אֲשֶׁר־הוּא֙ יֹשֵׁ֣ב בָּ֔הּ הֲטוֹבָ֥ה הִ֖וא אִם־רָעָ֑ה וּמָ֣ה הֶֽעָרִ֗ים אֲשֶׁר־הוּא֙ יוֹשֵׁ֣ב בָּהֵ֔נָּה הַבְּמַֽחֲנִ֖ים אִ֥ם בְּמִבְצָרִֽים׃ וְאֵיךְ הָאָרֶץ שֶׁהוּא יוֹשֵׁב בָּהּ – אִם הִיא טוֹבָה אוֹ רָעָה, וְאֵיךְ הֶעָרִים שֶׁהוּא גָּר בָּהֵן – אִם הֵן בְּמַחֲנוֹת פְּתוּחִים אוֹ בְּמִבְצָרִים.
כ וּמָ֣ה הָ֠אָ֠רֶץ הַשְּׁמֵנָ֨ה הִ֜וא אִם־רָזָ֗ה הֲיֵֽשׁ־בָּ֥הּ עֵץ֙ אִם־אַ֔יִן וְהִ֨תְחַזַּקְתֶּ֔ם וּלְקַחְתֶּ֖ם מִפְּרִ֣י הָאָ֑רֶץ וְהַ֨יָּמִ֔ים יְמֵ֖י בִּכּוּרֵ֥י עֲנָבִֽים׃ וְאֵיךְ הָאָרֶץ, אִם הִיא שְׁמֵנָה אוֹ רָזָה, אִם יֵשׁ בָּהּ עֵצִים אוֹ אֵין, וְתִתְחַזְּקוּ וְתִקְחוּ מִפֵּרוֹת הָאָרֶץ, וְאוֹתָם יָמִים הָיוּ יְמֵי הַבְשָׁלַת הָעֲנָבִים.

שני[עריכה]

כא וַֽיַּעֲל֖וּ וַיָּתֻ֣רוּ אֶת־הָאָ֑רֶץ מִמִּדְבַּר־צִ֥ן עַד־רְחֹ֖ב לְבֹ֥א חֲמָֽת׃ הֵם עָלוּ וְתִיְּרוּ בָּאָרֶץ מִמִּדְבַּר צִין שֶׁבַּנֶּגֶב עַד לִרְחָבַת הַכְּנִיסָה לְאַנְטְיוֹכְיָא שֶׁבִּדְרוֹם טוּרְקִיָּה.
כב וַיַּעֲל֣וּ בַנֶּ֘גֶב֮ וַיָּבֹ֣א עַד־חֶבְרוֹן֒ וְשָׁ֤ם אֲחִימַן֙ שֵׁשַׁ֣י וְתַלְמַ֔י יְלִידֵ֖י הָעֲנָ֑ק וְחֶבְר֗וֹן שֶׁ֤בַע שָׁנִים֙ נִבְנְתָ֔ה לִפְנֵ֖י צֹ֥עַן מִצְרָֽיִם׃ הֵם עָלוּ דֶּרֶךְ הַנֶּגֶב וּבָאוּ עַד חֶבְרוֹן, וְשָׁם אֲחִימַן שֵׁשַׁי וְתַלְמַי צֶאֱצָאֵי הָעֲנָקִים, וְחֶבְרוֹן נִבְנְתָה שֶׁבַע שָׁנִים לִפְנֵי הָעִיר צוֹעַן שֶׁבְּמִצְרַיִם.
כג וַיָּבֹ֜אוּ עַד־נַ֣חַל אֶשְׁכֹּ֗ל וַיִּכְרְת֨וּ מִשָּׁ֤ם זְמוֹרָה֙ וְאֶשְׁכּ֤וֹל עֲנָבִים֙ אֶחָ֔ד וַיִּשָּׂאֻ֥הוּ בַמּ֖וֹט בִּשְׁנָ֑יִם וּמִן־הָרִמֹּנִ֖ים וּמִן־הַתְּאֵנִֽים׃ הֵם בָּאוּ עַד נַחַל אֶשְׁכּוֹל, וְכָרְתוּ מִשָּׁם עָנָף עִם אֶשְׁכּוֹל עֲנָבִים אֶחָד, שְׁנֵי אֲנָשִׁים נָשְׂאוּ אוֹתוֹ בְּמוֹט, וְהֵבִיאוּ גַּם מֵהָרִמּוֹנִים וּמֵהַתְּאֵנִים.
כד לַמָּק֣וֹם הַה֔וּא קָרָ֖א נַ֣חַל אֶשְׁכּ֑וֹל עַ֚ל אֹד֣וֹת הָֽאֶשְׁכּ֔וֹל אֲשֶׁר־כָּרְת֥וּ מִשָּׁ֖ם בְּנֵ֥י יִשְׂרָאֵֽל׃ הַמָּקוֹם הַהוּא נִקְרָא נַחַל אֶשְׁכּוֹל עַל שֵׁם הָאֶשְׁכּוֹל שֶׁבְּנֵי יִשְׂרָאֵל כָּרְתוּ מִשָּׁם.
כה וַיָּשֻׁ֖בוּ מִתּ֣וּר הָאָ֑רֶץ מִקֵּ֖ץ אַרְבָּעִ֥ים יֽוֹם׃ הֵם חָזְרוּ מִלְּתַיֵּר בָּאָרֶץ בְּסוֹף אַרְבָּעִים יוֹם.
כו וַיֵּלְכ֡וּ וַיָּבֹ֩אוּ֩ אֶל־מֹשֶׁ֨ה וְאֶֽל־אַהֲרֹ֜ן וְאֶל־כׇּל־עֲדַ֧ת בְּנֵֽי־יִשְׂרָאֵ֛ל אֶל־מִדְבַּ֥ר פָּארָ֖ן קָדֵ֑שָׁה וַיָּשִׁ֨יבוּ אֹתָ֤ם דָּבָר֙ וְאֶת־כׇּל־הָ֣עֵדָ֔ה וַיַּרְא֖וּם אֶת־פְּרִ֥י הָאָֽרֶץ׃ הֵם הָלְכוּ לְמִדְבַּר פָּארָן לְקָדֵשׁ וּבָאוּ לְמֹשֶׁה וּלְאַהֲרֹן וּלְכָל הַקָּהָל שֶׁל בְּנֵי יִשְׂרָאֵל, וְהֵשִׁיבוּ לָהֶם וּלְכָל הַקָּהָל תְּשׁוּבוֹת, וְהֶרְאוּ לָהֶם אֶת פֵּרוֹת הָאָרֶץ.
כז וַיְסַפְּרוּ־לוֹ֙ וַיֹּ֣אמְר֔וּ בָּ֕אנוּ אֶל־הָאָ֖רֶץ אֲשֶׁ֣ר שְׁלַחְתָּ֑נוּ וְ֠גַ֠ם זָבַ֨ת חָלָ֥ב וּדְבַ֛שׁ הִ֖וא וְזֶה־פִּרְיָֽהּ׃ הֵם סִפְּרוּ לוֹ וְאָמְרוּ: בָּאנוּ לָאָרֶץ שֶׁשָּׁלַחְתָּ אוֹתָנוּ, וְהִיא אָמְנָם שׁוֹפַעַת חָלָב וּדְבַשׁ, וְאֵלֶּה פֵּרוֹתֶיהָ.
כח אֶ֚פֶס כִּֽי־עַ֣ז הָעָ֔ם הַיֹּשֵׁ֖ב בָּאָ֑רֶץ וְהֶֽעָרִ֗ים בְּצֻר֤וֹת גְּדֹלֹת֙ מְאֹ֔ד וְגַם־יְלִדֵ֥י הָֽעֲנָ֖ק רָאִ֥ינוּ שָֽׁם׃ אֲבָל הָעָם שֶׁיּוֹשֵׁב בָּאָרֶץ הוּא גִּבּוֹר, וְהֶעָרִים מְבֻצָּרוֹת וּגְדוֹלוֹת מְאוֹד, וְגַם רָאִינוּ שָׁם אֶת צֶאֱצָאֵי הָעֲנָקִים.
כט עֲמָלֵ֥ק יוֹשֵׁ֖ב בְּאֶ֣רֶץ הַנֶּ֑גֶב וְ֠הַֽחִתִּ֠י וְהַיְבוּסִ֤י וְהָֽאֱמֹרִי֙ יוֹשֵׁ֣ב בָּהָ֔ר וְהַֽכְּנַעֲנִי֙ יוֹשֵׁ֣ב עַל־הַיָּ֔ם וְעַ֖ל יַ֥ד הַיַּרְדֵּֽן׃ עֲמָלֵק יוֹשֵׁב בְּאֶרֶץ הַנֶּגֶב, וְהַחִתִּי וְהַיְּבוּסִי וְהָאֱמוֹרִי יוֹשְׁבִים בָּהָר, וְהַכְּנַעֲנִים יוֹשְׁבִים עַל יַד הַיָּם וְעַל יַד הַיַּרְדֵּן.
ל וַיַּ֧הַס כָּלֵ֛ב אֶת־הָעָ֖ם אֶל־מֹשֶׁ֑ה וַיֹּ֗אמֶר עָלֹ֤ה נַעֲלֶה֙ וְיָרַ֣שְׁנוּ אֹתָ֔הּ כִּֽי־יָכ֥וֹל נוּכַ֖ל לָֽהּ׃ כָּלֵב הִשְׁתִּיק אֶת הָעָם שֶׁיִּשְׁמְעוּ לְמֹשֶׁה, וְאָמַר: אֲנַחְנוּ נַעֲלֶה וְנִכְבּוֹשׁ אֶת הָאָרֶץ כִּי נוּכַל לִכְבּוֹשׁ אוֹתָהּ!
לא וְהָ֨אֲנָשִׁ֜ים אֲשֶׁר־עָל֤וּ עִמּוֹ֙ אָֽמְר֔וּ לֹ֥א נוּכַ֖ל לַעֲל֣וֹת אֶל־הָעָ֑ם כִּֽי־חָזָ֥ק ה֖וּא מִמֶּֽנּוּ׃ וְהָאֲנָשִׁים שֶׁעָלוּ אִתּוֹ אָמְרוּ: לֹא נוּכַל לַעֲלוֹת מוּל הָעָם הַזֶּה לְהִלָּחֵם בּוֹ כִּי הוּא יוֹתֵר חָזָק מֵאִתָּנוּ.
לב וַיֹּצִ֜יאוּ דִּבַּ֤ת הָאָ֙רֶץ֙ אֲשֶׁ֣ר תָּר֣וּ אֹתָ֔הּ אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר הָאָ֡רֶץ אֲשֶׁר֩ עָבַ֨רְנוּ בָ֜הּ לָת֣וּר אֹתָ֗הּ אֶ֣רֶץ אֹכֶ֤לֶת יוֹשְׁבֶ֙יהָ֙ הִ֔וא וְכׇל־הָעָ֛ם אֲשֶׁר־רָאִ֥ינוּ בְתוֹכָ֖הּ אַנְשֵׁ֥י מִדּֽוֹת׃ הֵם הוֹצִיאוּ שֵׁם רַע בְּאָזְנֵי בְּנֵי יִשְׂרָאֵל עַל הָאָרֶץ שֶׁתִּיְּרוּ בָּהּ, וְאָמְרוּ: הָאָרֶץ שֶׁעָבַרְנוּ בָּהּ לְתַיֵּר בָּהּ הִיא אֶרֶץ שֶׁהוֹרֶגֶת אֶת יוֹשְׁבֶיהָ, וְכָל הָעָם שֶׁרָאִינוּ בְּתוֹכָהּ הֵם אֲנָשִׁים גְּדוֹלִים.
לג וְשָׁ֣ם רָאִ֗ינוּ אֶת־הַנְּפִילִ֛ים בְּנֵ֥י עֲנָ֖ק מִן־הַנְּפִלִ֑ים וַנְּהִ֤י בְעֵינֵ֙ינוּ֙ כַּֽחֲגָבִ֔ים וְכֵ֥ן הָיִ֖ינוּ בְּעֵינֵיהֶֽם׃ וְשָׁם רָאִינוּ אֶת הַעֲנָקִים בְּנֵי הָאִישׁ 'עֲנָק' – מֵהָעֲנָקִים שֶׁהָיוּ בָּאָרֶץ בֶּעָבָר, הָיִינוּ בְּעֵינֵינוּ כְּמוֹ חֲגָבִים וְכָךְ הָיִינוּ גַּם בְּעֵינֵיהֶם.
יד א וַתִּשָּׂא֙ כׇּל־הָ֣עֵדָ֔ה וַֽיִּתְּנ֖וּ אֶת־קוֹלָ֑ם וַיִּבְכּ֥וּ הָעָ֖ם בַּלַּ֥יְלָה הַהֽוּא׃ כָּל הַקָּהָל – הָעָם, הֵרִימוּ אֶת קוֹלָם וּבָכוּ בְּאוֹתוֹ לַיְלָה.
ב וַיִּלֹּ֙נוּ֙ עַל־מֹשֶׁ֣ה וְעַֽל־אַהֲרֹ֔ן כֹּ֖ל בְּנֵ֣י יִשְׂרָאֵ֑ל וַֽיֹּאמְר֨וּ אֲלֵהֶ֜ם כׇּל־הָעֵדָ֗ה לוּ־מַ֙תְנוּ֙ בְּאֶ֣רֶץ מִצְרַ֔יִם א֛וֹ בַּמִּדְבָּ֥ר הַזֶּ֖ה לוּ־מָֽתְנוּ׃ כָּל בְּנֵי יִשְׂרָאֵל הִתְלוֹנְנוּ עַל מֹשֶׁה וְעַל אַהֲרֹן, וְכָל הַקָּהָל אָמְרוּ לָהֶם: הַלְוַאי שֶׁהָיִינוּ מֵתִים בְּאֶרֶץ מִצְרַיִם, אוֹ בַּמִּדְבָּר הַזֶּה הַלְוַאי שֶׁהָיִינוּ מֵתִים!
ג וְלָמָ֣ה יְ֠הֹוָ֠ה מֵבִ֨יא אֹתָ֜נוּ אֶל־הָאָ֤רֶץ הַזֹּאת֙ לִנְפֹּ֣ל בַּחֶ֔רֶב נָשֵׁ֥ינוּ וְטַפֵּ֖נוּ יִהְי֣וּ לָבַ֑ז הֲל֧וֹא ט֦וֹב לָ֖נוּ שׁ֥וּב מִצְרָֽיְמָה׃ וְלָמָּה ה' מֵבִיא אוֹתָנוּ לָאָרֶץ הַזֹּאת שֶׁנָּמוּת בַּמִּלְחָמָה, וְשֶׁיִּשְׁבּוּ אֶת נָשֵׁינוּ וִילָדֵנוּ? הֲרֵי עָדִיף לָנוּ לַחֲזוֹר לְמִצְרַיִם!
ד וַיֹּאמְר֖וּ אִ֣ישׁ אֶל־אָחִ֑יו נִתְּנָ֥ה רֹ֖אשׁ וְנָשׁ֥וּבָה מִצְרָֽיְמָה׃ הֵם אָמְרוּ אֶחָד לַשֵּׁנִי: נְמַנֶּה לָנוּ מַנְהִיג וְנַחֲזוֹר לְמִצְרַיִם!
ה וַיִּפֹּ֥ל מֹשֶׁ֛ה וְאַהֲרֹ֖ן עַל־פְּנֵיהֶ֑ם לִפְנֵ֕י כׇּל־קְהַ֥ל עֲדַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ מֹשֶׁה וְאַהֲרֹן הִשְׁתַּטְּחוּ עַל פְּנֵיהֶם לִפְנֵי כָּל צִבּוּר קְהַל בְּנֵי יִשְׂרָאֵל לְבַקֵּשׁ מֵהֶם שֶׁיַּפְסִיקוּ.
ו וִיהוֹשֻׁ֣עַ בִּן־נ֗וּן וְכָלֵב֙ בֶּן־יְפֻנֶּ֔ה מִן־הַתָּרִ֖ים אֶת־הָאָ֑רֶץ קָרְע֖וּ בִּגְדֵיהֶֽם׃ וִיהוֹשֻׁעַ בִּן נוּן וְכָלֵב בֶּן יְפֻנֶּה שֶׁהָיוּ מֵהַמְּתַיְּרִים בָּאָרֶץ קָרְעוּ אֶת בִּגְדֵיהֶם.
ז וַיֹּ֣אמְר֔וּ אֶל־כׇּל־עֲדַ֥ת בְּנֵֽי־יִשְׂרָאֵ֖ל לֵאמֹ֑ר הָאָ֗רֶץ אֲשֶׁ֨ר עָבַ֤רְנוּ בָהּ֙ לָת֣וּר אֹתָ֔הּ טוֹבָ֥ה הָאָ֖רֶץ מְאֹ֥ד מְאֹֽד׃ וְאָמְרוּ לְכָל הַקָּהָל שֶׁל בְּנֵי יִשְׂרָאֵל: הָאָרֶץ שֶׁעָבַרְנוּ בָּהּ לְתַיֵּר בָּהּ הִיא אֶרֶץ טוֹבָה מְאוֹד מְאוֹד!

שלישי[עריכה]

ח אִם־חָפֵ֥ץ בָּ֙נוּ֙ יְהֹוָ֔ה וְהֵבִ֤יא אֹתָ֙נוּ֙ אֶל־הָאָ֣רֶץ הַזֹּ֔את וּנְתָנָ֖הּ לָ֑נוּ אֶ֕רֶץ אֲשֶׁר־הִ֛וא זָבַ֥ת חָלָ֖ב וּדְבָֽשׁ׃ אִם ה' רוֹצֶה בָּנוּ הוּא יָבִיא אוֹתָנוּ לָאָרֶץ הַזֹּאת וְיִתֵּן לָנוּ אוֹתָהּ – אֶרֶץ שֶׁשּׁוֹפַעַת חָלָב וּדְבָשׁ!
ט אַ֣ךְ בַּיהֹוָה֮ אַל־תִּמְרֹ֒דוּ֒ וְאַתֶּ֗ם אַל־תִּֽירְאוּ֙ אֶת־עַ֣ם הָאָ֔רֶץ כִּ֥י לַחְמֵ֖נוּ הֵ֑ם סָ֣ר צִלָּ֧ם מֵעֲלֵיהֶ֛ם וַֽיהֹוָ֥ה אִתָּ֖נוּ אַל־תִּירָאֻֽם׃ רַק אַל תִּמְרְדוּ בַּה', וְאַתֶּם אַל תִּפְחֲדוּ מֵהָעָם שֶׁגָּר בָּאָרֶץ כִּי נִכְבּוֹשׁ אוֹתָם, הֲרֵי סָר כּוֹחָם מֵעֲלֵיהֶם וַה' אִתָּנוּ, אַל תִּפְחְדוּ מֵהֶם!
י וַיֹּֽאמְרוּ֙ כׇּל־הָ֣עֵדָ֔ה לִרְגּ֥וֹם אֹתָ֖ם בָּאֲבָנִ֑ים וּכְב֣וֹד יְהֹוָ֗ה נִרְאָה֙ בְּאֹ֣הֶל מוֹעֵ֔ד אֶֽל־כׇּל־בְּנֵ֖י יִשְׂרָאֵֽל׃ כָּל הַצִּבּוּר חָשְׁבוּ לִרְגּוֹם אוֹתָם בָּאֲבָנִים, וּכְבוֹד ה' הִתְגַּלָּה בְּאוֹהֶל הַהִתְכַּנְּסוּת אֶל כָּל בְּנֵי יִשְׂרָאֵל.
פ
יא וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה עַד־אָ֥נָה יְנַאֲצֻ֖נִי הָעָ֣ם הַזֶּ֑ה וְעַד־אָ֙נָה֙ לֹא־יַאֲמִ֣ינוּ בִ֔י בְּכֹל֙ הָֽאֹת֔וֹת אֲשֶׁ֥ר עָשִׂ֖יתִי בְּקִרְבּֽוֹ׃ ה' אָמַר לְמֹשֶׁה: עַד מָתַי הָעָם הַזֶּה יַכְעִיסוּ אוֹתִי, וְעַד מָתַי לֹא יַאֲמִינוּ בִּי לַמְרוֹת כָּל הַסִּימָנִים לְכוֹחִי שֶׁהֶרְאֵיתִי לוֹ?
יב אַכֶּ֥נּוּ בַדֶּ֖בֶר וְאוֹרִשֶׁ֑נּוּ וְאֶֽעֱשֶׂה֙ אֹֽתְךָ֔ לְגוֹי־גָּד֥וֹל וְעָצ֖וּם מִמֶּֽנּוּ׃ אַכֶּה אוֹתוֹ בְּמַחֲלָה קָשָׁה וַאֲגָרֵשׁ אוֹתוֹ, וְאֶעֱשֶׂה מִמְּךָ עַם יוֹתֵר גָּדוֹל וְעָצוּם מִמֶּנּוּ.
יג וַיֹּ֥אמֶר מֹשֶׁ֖ה אֶל־יְהֹוָ֑ה וְשָׁמְע֣וּ מִצְרַ֔יִם כִּֽי־הֶעֱלִ֧יתָ בְכֹחֲךָ֛ אֶת־הָעָ֥ם הַזֶּ֖ה מִקִּרְבּֽוֹ׃ מֹשֶׁה אָמַר לַה': כְּשֶׁמִּצְרַיִם שֶׁמִּתּוֹכָם הֶעֱלִיתָ אֶת הָעָם הַזֶּה בְּכוֹחֲךָ, יִשְׁמְעוּ זֹאת,
יד וְאָמְר֗וּ אֶל־יוֹשֵׁב֮ הָאָ֣רֶץ הַזֹּאת֒ שָֽׁמְעוּ֙ כִּֽי־אַתָּ֣ה יְהֹוָ֔ה בְּקֶ֖רֶב הָעָ֣ם הַזֶּ֑ה אֲשֶׁר־עַ֨יִן בְּעַ֜יִן נִרְאָ֣ה ׀ אַתָּ֣ה יְהֹוָ֗ה וַעֲנָֽנְךָ֙ עֹמֵ֣ד עֲלֵהֶ֔ם וּבְעַמֻּ֣ד עָנָ֗ן אַתָּ֨ה הֹלֵ֤ךְ לִפְנֵיהֶם֙ יוֹמָ֔ם וּבְעַמּ֥וּד אֵ֖שׁ לָֽיְלָה׃ הֵם יֹאמְרוּ לְתוֹשְׁבֵי הָאָרֶץ הַזֹּאת שֶׁשָּׁמְעוּ שֶׁאַתָּה ה' נִמְצָא בְּתוֹךְ הָעָם הַזֶּה – שֶׁרָאוּ בְּעַיִן שֶׁאַתָּה ה' מִתְגַּלֶּה אֲלֵיהֶם וַעֲנָנְךָ פָּרוּשׂ מֵעֲלֵיהֶם וְאַתָּה הוֹלֵךְ לִפְנֵיהֶם, בַּיּוֹם בְּעַמּוּד עָנָן, וּבְעַמּוּד אֵשׁ בַּלַּיְלָה.
טו וְהֵמַתָּ֛ה אֶת־הָעָ֥ם הַזֶּ֖ה כְּאִ֣ישׁ אֶחָ֑ד וְאָֽמְרוּ֙ הַגּוֹיִ֔ם אֲשֶׁר־שָׁמְע֥וּ אֶֽת־שִׁמְעֲךָ֖ לֵאמֹֽר׃ אִם תַּהֲרוֹג אֶת הָעָם הַזֶּה כְּאִישׁ אֶחָד, הָעַמִּים שֶׁשָּׁמְעוּ אֶת הַשְּׁמוּעוֹת עַל כּוֹחֲךָ יֹאמְרוּ כָּךְ:
טז מִבִּלְתִּ֞י יְכֹ֣לֶת יְהֹוָ֗ה לְהָבִיא֙ אֶת־הָעָ֣ם הַזֶּ֔ה אֶל־הָאָ֖רֶץ אֲשֶׁר־נִשְׁבַּ֣ע לָהֶ֑ם וַיִּשְׁחָטֵ֖ם בַּמִּדְבָּֽר׃ בִּגְלַל חוֹסֶר יְכָלְתוֹ שֶׁל ה' לְהָבִיא אֶת הָעָם הַזֶּה לָאָרֶץ שֶׁנִּשְׁבַּע לָהֶם הוּא שָׁחַט אוֹתָם בַּמִּדְבָּר.
יז וְעַתָּ֕ה יִגְדַּל־נָ֖א כֹּ֣חַ אֲדֹנָ֑י כַּאֲשֶׁ֥ר דִּבַּ֖רְתָּ לֵאמֹֽר׃ וְעַכְשָׁיו, יִגְדַּל בְּבַקָּשָׁה הַכֹּחַ שֶׁל אֲדֹנָי בְּעֵינֵי הָעַמִּים, וַעֲשֵׂה כְּמוֹ שֶׁדִּבַּרְתָּ:
יח יְהֹוָ֗ה אֶ֤רֶךְ אַפַּ֙יִם֙ וְרַב־חֶ֔סֶד נֹשֵׂ֥א עָוֺ֖ן וָפָ֑שַׁע וְנַקֵּה֙ לֹ֣א יְנַקֶּ֔ה פֹּקֵ֞ד עֲוֺ֤ן אָבוֹת֙ עַל־בָּנִ֔ים עַל־שִׁלֵּשִׁ֖ים וְעַל־רִבֵּעִֽים׃ "ה' סַבְלָן מִלִּכְעוֹס וּמַרְבֶּה חֶסֶד, מַבְלִיג עַל עָווֹן וְעַל פֶּשַׁע אֲבָל אֵינוֹ מוֹחֵק אוֹתָם, זוֹכֵר עֲווֹן אָבוֹת לְהַעֲנִישׁ בָּנִים נְכָדִים וְנִינִים".
יט סְלַֽח־נָ֗א לַעֲוֺ֛ן הָעָ֥ם הַזֶּ֖ה כְּגֹ֣דֶל חַסְדֶּ֑ךָ וְכַאֲשֶׁ֤ר נָשָׂ֙אתָה֙ לָעָ֣ם הַזֶּ֔ה מִמִּצְרַ֖יִם וְעַד־הֵֽנָּה׃ סְלַח בְּבַקָּשָׁה לַעֲווֹן הָעָם הַזֶּה כְּפִי גּוֹדֶל חַסְדְּךָ וּכְמוֹ שֶׁסָּלַחְתָּ לָעָם הַזֶּה עַל הָעֲווֹנוֹת שֶׁעָשׂוּ מִמִּצְרַיִם וְעַד עַכְשָׁיו.
כ וַיֹּ֣אמֶר יְהֹוָ֔ה סָלַ֖חְתִּי כִּדְבָרֶֽךָ׃ ה' אָמַר: סָלַחְתִּי כִּדְבָרְךָ.
כא וְאוּלָ֖ם חַי־אָ֑נִי וְיִמָּלֵ֥א כְבוֹד־יְהֹוָ֖ה אֶת־כׇּל־הָאָֽרֶץ׃ אֲבָל אֲנִי נִשְׁבַּע בְּחַיַּי שֶׁהֵם יֵעָנְשׁוּ, וּכְבוֹד ה' יִתְפַּרְסֵם בְּכָל הָעוֹלָם –
כב כִּ֣י כׇל־הָאֲנָשִׁ֗ים הָרֹאִ֤ים אֶת־כְּבֹדִי֙ וְאֶת־אֹ֣תֹתַ֔י אֲשֶׁר־עָשִׂ֥יתִי בְמִצְרַ֖יִם וּבַמִּדְבָּ֑ר וַיְנַסּ֣וּ אֹתִ֗י זֶ֚ה עֶ֣שֶׂר פְּעָמִ֔ים וְלֹ֥א שָׁמְע֖וּ בְּקוֹלִֽי׃ שֶׁכָּל הָאֲנָשִׁים שֶׁרָאוּ אֶת כְּבוֹדִי וְאֶת הַסִּימָנִים לְכוֹחִי שֶׁעָשִׂיתִי בְּמִצְרַיִם וּבַמִּדְבָּר, וְלַמְרוֹת זֹאת נִסּוּ אוֹתִי כְּבָר עֶשֶׂר פְּעָמִים וְלֹא שָׁמְעוּ בְּקוֹלִי,
כג אִם־יִרְאוּ֙ אֶת־הָאָ֔רֶץ אֲשֶׁ֥ר נִשְׁבַּ֖עְתִּי לַאֲבֹתָ֑ם וְכׇל־מְנַאֲצַ֖י לֹ֥א יִרְאֽוּהָ׃ שֶׁלֹּא יִרְאוּ אֶת הָאָרֶץ שֶׁנִּשְׁבַּעְתִּי לַאֲבוֹתֵיהֶם, וְכָל הַמַּרְגִּיזִים אוֹתִי לֹא יִרְאוּ אוֹתָהּ!
כד וְעַבְדִּ֣י כָלֵ֗ב עֵ֣קֶב הָֽיְתָ֞ה ר֤וּחַ אַחֶ֙רֶת֙ עִמּ֔וֹ וַיְמַלֵּ֖א אַחֲרָ֑י וַהֲבִֽיאֹתִ֗יו אֶל־הָאָ֙רֶץ֙ אֲשֶׁר־בָּ֣א שָׁ֔מָּה וְזַרְע֖וֹ יוֹרִשֶֽׁנָּה׃ וְכָלֵב עַבְדִּי, בִּגְלַל שֶׁהָיְתָה בְּלִבּוֹ רוּחַ אַחֶרֶת וְנֶאֱמָנוּתוֹ לְדַרְכִּי הָיְתָה שְׁלֵמָה, אָבִיא אוֹתוֹ לָאָרֶץ שֶׁבָּא אֵלֶיהָ וְצֶאֱצָאָיו יִירְשׁוּ אוֹתָהּ.
כה וְהָעֲמָֽלֵקִ֥י וְהַֽכְּנַעֲנִ֖י יוֹשֵׁ֣ב בָּעֵ֑מֶק מָחָ֗ר פְּנ֨וּ וּסְע֥וּ לָכֶ֛ם הַמִּדְבָּ֖ר דֶּ֥רֶךְ יַם־סֽוּף׃ וְהָעֲמָלֵקִים וְהַכְּנַעֲנִים גָּרִים בָּעֵמֶק, מָחָר פְּנוּ חֲזָרָה וּסְעוּ לַמִּדְבָּר דֶּרֶךְ יַם סוּף כְּדֵי שֶׁלֹּא לְפָגְשָׁם.

רביעי[עריכה]

פ
כו וַיְדַבֵּ֣ר יְהֹוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאֶל אַהֲרֹן וְאָמַר:
כז עַד־מָתַ֗י לָעֵדָ֤ה הָֽרָעָה֙ הַזֹּ֔את אֲשֶׁ֛ר הֵ֥מָּה מַלִּינִ֖ים עָלָ֑י אֶת־תְּלֻנּ֞וֹת בְּנֵ֣י יִשְׂרָאֵ֗ל אֲשֶׁ֨ר הֵ֧מָּה מַלִּינִ֛ים עָלַ֖י שָׁמָֽעְתִּי׃ עַד מָתַי הַצִּבּוּר הָרַע הַזֶּה יִתְלוֹנְנוּ עָלַי?! הֲרֵי שָׁמַעְתִּי אֶת תְּלֻנּוֹת בְּנֵי יִשְׂרָאֵל שֶׁהֵם מִתְלוֹנְנִים עָלַי.
כח אֱמֹ֣ר אֲלֵהֶ֗ם חַי־אָ֙נִי֙ נְאֻם־יְהֹוָ֔ה אִם־לֹ֕א כַּאֲשֶׁ֥ר דִּבַּרְתֶּ֖ם בְּאׇזְנָ֑י כֵּ֖ן אֶֽעֱשֶׂ֥ה לָכֶֽם׃ אֱמוֹר לָהֶם, ה' אָמַר: אֲנִי נִשְׁבַּע בְּחַיַּי שֶׁכְּמוֹ שֶׁדִּבַּרְתֶּם בְּאָזְנַי – כָּךְ אֶעֱשֶׂה לָכֶם!
כט בַּמִּדְבָּ֣ר הַ֠זֶּ֠ה יִפְּל֨וּ פִגְרֵיכֶ֜ם וְכׇל־פְּקֻדֵיכֶם֙ לְכׇל־מִסְפַּרְכֶ֔ם מִבֶּ֛ן עֶשְׂרִ֥ים שָׁנָ֖ה וָמָ֑עְלָה אֲשֶׁ֥ר הֲלִֽינֹתֶ֖ם עָלָֽי׃ פִּגְרֵיכֶם יִפְּלוּ בַּמִּדְבָּר הַזֶּה – כָּל אַנְשֵׁיכֶם מִגִּיל עֶשְׂרִים שָׁנָה וָמַעְלָה שֶׁנִּסְפְּרוּ, בִּגְלַל שֶׁהִתְלוֹנַנְתֶּם עָלַי.
ל אִם־אַתֶּם֙ תָּבֹ֣אוּ אֶל־הָאָ֔רֶץ אֲשֶׁ֤ר נָשָׂ֙אתִי֙ אֶת־יָדִ֔י לְשַׁכֵּ֥ן אֶתְכֶ֖ם בָּ֑הּ כִּ֚י אִם־כָּלֵ֣ב בֶּן־יְפֻנֶּ֔ה וִיהוֹשֻׁ֖עַ בִּן־נֽוּן׃ אֲנִי נִשְׁבַּע שֶׁלֹּא תָּבוֹאוּ לָאָרֶץ שֶׁנִּשְׁבַּעְתִּי בַּהֲרָמַת יַד לְהוֹשִׁיב אֶתְכֶם בָּהּ, חוּץ מִכָּלֵב בֶּן יְפֻנֶּה וִיהוֹשֻׁעַ בִּן נוּן.
לא וְטַ֨פְּכֶ֔ם אֲשֶׁ֥ר אֲמַרְתֶּ֖ם לָבַ֣ז יִהְיֶ֑ה וְהֵבֵיאתִ֣י אֹתָ֔ם וְיָֽדְעוּ֙ אֶת־הָאָ֔רֶץ אֲשֶׁ֥ר מְאַסְתֶּ֖ם בָּֽהּ׃ וְיַלְדֵיכֶם שֶׁאֲמַרְתֶּם שֶׁיִּהְיוּ שָׁלָל, אוֹתָם אָבִיא וְהֵם יַכִּירוּ אֶת הָאָרֶץ שֶׁמְּאַסְתֶּם בָּהּ.
לב וּפִגְרֵיכֶ֖ם אַתֶּ֑ם יִפְּל֖וּ בַּמִּדְבָּ֥ר הַזֶּֽה׃ וְאַתֶּם, הַפְּגָרִים שֶׁלָּכֶם יִפְּלוּ בַּמִּדְבָּר הַזֶּה.
לג וּ֠בְנֵיכֶ֠ם יִהְי֨וּ רֹעִ֤ים בַּמִּדְבָּר֙ אַרְבָּעִ֣ים שָׁנָ֔ה וְנָשְׂא֖וּ אֶת־זְנוּתֵיכֶ֑ם עַד־תֹּ֥ם פִּגְרֵיכֶ֖ם בַּמִּדְבָּֽר׃ וּבְנֵיכֶם יִסְתּוֹבְבוּ בַּמִּדְבָּר אַרְבָּעִים שָׁנָה וְיִסְבְּלוּ בִּגְלַל בְּגִידַתְכֶם, עַד שֶׁיִּגָּמְרוּ פִּגְרֵיכֶם בַּמִּדְבָּר.
לד בְּמִסְפַּ֨ר הַיָּמִ֜ים אֲשֶׁר־תַּרְתֶּ֣ם אֶת־הָאָ֘רֶץ֮ אַרְבָּעִ֣ים יוֹם֒ י֣וֹם לַשָּׁנָ֞ה י֣וֹם לַשָּׁנָ֗ה תִּשְׂאוּ֙ אֶת־עֲוֺנֹ֣תֵיכֶ֔ם אַרְבָּעִ֖ים שָׁנָ֑ה וִֽידַעְתֶּ֖ם אֶת־תְּנוּאָתִֽי׃ כְּמִסְפַּר הַיָּמִים שֶׁתִּיַּרְתֶּם בָּאָרֶץ – אַרְבָּעִים יוֹם, יוֹם לַשָּׁנָה יוֹם לַשָּׁנָה – כָּךְ תְּקַבְּלוּ אֶת עוֹנֶשׁ עֲוֹנוֹתֵיכֶם – אַרְבָּעִים שָׁנָה, וְתַפְנִימוּ אֶת עוֹנֶשׁ הֲפָרַת בְּרִיתִי.
לה אֲנִ֣י יְהֹוָה֮ דִּבַּ֒רְתִּי֒ אִם־לֹ֣א ׀ זֹ֣את אֶֽעֱשֶׂ֗ה לְכׇל־הָעֵדָ֤ה הָֽרָעָה֙ הַזֹּ֔את הַנּוֹעָדִ֖ים עָלָ֑י בַּמִּדְבָּ֥ר הַזֶּ֛ה יִתַּ֖מּוּ וְשָׁ֥ם יָמֻֽתוּ׃ אֲנִי ה' מַצְהִיר שֶׁכָּךְ אֶעֱשֶׂה לְכָל הַצִּבּוּר הָרַע הַזֶּה שֶׁהִתְאַסְּפוּ נֶגְדִּי, הֵם יֻשְׁמְדוּ בַּמִּדְבָּר הַזֶּה וְשָׁם יָמוּתוּ.
לו וְהָ֣אֲנָשִׁ֔ים אֲשֶׁר־שָׁלַ֥ח מֹשֶׁ֖ה לָת֣וּר אֶת־הָאָ֑רֶץ וַיָּשֻׁ֗בוּ וילונו וַיַּלִּ֤ינוּ עָלָיו֙ אֶת־כׇּל־הָ֣עֵדָ֔ה לְהוֹצִ֥יא דִבָּ֖ה עַל־הָאָֽרֶץ׃ וְהָאֲנָשִׁים שֶׁמֹּשֶׁה שָׁלַח לָתַיֵּר בָּאָרֶץ וְחָזְרוּ וְגָרְמוּ לְכָל הַצִּבּוּר לְהִתְלוֹנֵן עָלָיו, וּלְהוֹצִיא שֵׁם רַע עַל הָאָרֶץ –
לז וַיָּמֻ֙תוּ֙ הָֽאֲנָשִׁ֔ים מוֹצִאֵ֥י דִבַּת־הָאָ֖רֶץ רָעָ֑ה בַּמַּגֵּפָ֖ה לִפְנֵ֥י יְהֹוָֽה׃ הָאֲנָשִׁים הָאֵלֶּה שֶׁהוֹצִיאוּ שֵׁם רַע עַל הָאָרֶץ, ה' הָרַג אוֹתָם בַּמַּגֵּפָה.
לח וִיהוֹשֻׁ֣עַ בִּן־נ֔וּן וְכָלֵ֖ב בֶּן־יְפֻנֶּ֑ה חָיוּ֙ מִן־הָאֲנָשִׁ֣ים הָהֵ֔ם הַהֹֽלְכִ֖ים לָת֥וּר אֶת־הָאָֽרֶץ׃ וִיהוֹשֻׁעַ בִּן נוּן וְכָלֵב בֶּן יְפֻנֶּה הָיוּ הַיְּחִידִים מֵאוֹתָם הָאֲנָשִׁים שֶׁהָלְכוּ לְתַיֵּר בָּאָרֶץ שֶׁנִּשְׁאֲרוּ בַּחַיִּים.
לט וַיְדַבֵּ֤ר מֹשֶׁה֙ אֶת־הַדְּבָרִ֣ים הָאֵ֔לֶּה אֶֽל־כׇּל־בְּנֵ֖י יִשְׂרָאֵ֑ל וַיִּֽתְאַבְּל֥וּ הָעָ֖ם מְאֹֽד׃ מֹשֶׁה דִּבֵּר אֶת הַדְּבָרִים הָאֵלֶּה אֶל כָּל בְּנֵי יִשְׂרָאֵל, וְהָעָם הִתְאַבְּלוּ מְאוֹד.
מ וַיַּשְׁכִּ֣מוּ בַבֹּ֔קֶר וַיַּֽעֲל֥וּ אֶל־רֹאשׁ־הָהָ֖ר לֵאמֹ֑ר הִנֶּ֗נּוּ וְעָלִ֛ינוּ אֶל־הַמָּק֛וֹם אֲשֶׁר־אָמַ֥ר יְהֹוָ֖ה כִּ֥י חָטָֽאנוּ׃ הֵם הִשְׁכִּימוּ בַּבּוֹקֶר וְעָלוּ אֶל רֹאשׁ הָהָר וְאָמְרוּ: הִנֵּה אֲנַחְנוּ נַעֲלֶה לַמָּקוֹם שֶׁה' אָמַר לַעֲלוֹת אֵלָיו שֶׁהֲרֵי חָטָאנוּ שֶׁלֹּא רָצִינוּ!
מא וַיֹּ֣אמֶר מֹשֶׁ֔ה לָ֥מָּה זֶּ֛ה אַתֶּ֥ם עֹבְרִ֖ים אֶת־פִּ֣י יְהֹוָ֑ה וְהִ֖וא לֹ֥א תִצְלָֽח׃ מֹשֶׁה אָמַר: לָמָּה אַתֶּם עוֹבְרִים עַל דִּבְרֵי ה'? הֲרֵי זֶה לֹא יַצְלִיחַ!
מב אַֽל־תַּעֲל֔וּ כִּ֛י אֵ֥ין יְהֹוָ֖ה בְּקִרְבְּכֶ֑ם וְלֹא֙ תִּנָּ֣גְפ֔וּ לִפְנֵ֖י אֹיְבֵיכֶֽם׃ אַל תַּעֲלוּ כִּי ה' לֹא אִתְּכֶם, וְכָךְ לֹא תֻּכּוּ עַל יְדֵי אוֹיְבֵיכֶם.
מג כִּי֩ הָעֲמָלֵקִ֨י וְהַכְּנַעֲנִ֥י שָׁם֙ לִפְנֵיכֶ֔ם וּנְפַלְתֶּ֖ם בֶּחָ֑רֶב כִּֽי־עַל־כֵּ֤ן שַׁבְתֶּם֙ מֵאַחֲרֵ֣י יְהֹוָ֔ה וְלֹא־יִהְיֶ֥ה יְהֹוָ֖ה עִמָּכֶֽם׃ הֲרֵי הָעֲמָלֵקִים וְהַכְּנַעֲנִים שָׁם לִפְנֵיכֶם וְתָמוּתוּ בַּמִּלְחָמָה, בִּגְלַל שֶׁעֲזַבְתֶּם אֶת ה' – ה' לֹא יִהְיֶה אִתְּכֶם.
מד וַיַּעְפִּ֕לוּ לַעֲל֖וֹת אֶל־רֹ֣אשׁ הָהָ֑ר וַאֲר֤וֹן בְּרִית־יְהֹוָה֙ וּמֹשֶׁ֔ה לֹא־מָ֖שׁוּ מִקֶּ֥רֶב הַֽמַּחֲנֶֽה׃ הֵם הִתְעַקְּשׁוּ לַעֲלוֹת אֶל רֹאשׁ הָהָר, אֲבָל אֲרוֹן בְּרִית ה' וּמֹשֶׁה לֹא זָזוּ מִתּוֹךְ הַמַּחֲנֶה.
מה וַיֵּ֤רֶד הָעֲמָלֵקִי֙ וְהַֽכְּנַעֲנִ֔י הַיֹּשֵׁ֖ב בָּהָ֣ר הַה֑וּא וַיַּכּ֥וּם וַֽיַּכְּת֖וּם עַד־הַֽחׇרְמָֽה׃ הָעֲמָלֵקִים וְהַכְּנַעֲנִים שֶׁגָּרִים בָּהָר הַהוּא יָרְדוּ וְהִכּוּ אוֹתָם וְכָתְשׁוּ אוֹתָם עַד לְהַשְׁמָדָה.
פ פָּרָשַׁת הַמְּנָחוֹת וְהַנְּסָכִים
טו א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָמַרְתָּ֖ אֲלֵהֶ֑ם כִּ֣י תָבֹ֗אוּ אֶל־אֶ֙רֶץ֙ מוֹשְׁבֹ֣תֵיכֶ֔ם אֲשֶׁ֥ר אֲנִ֖י נֹתֵ֥ן לָכֶֽם׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְתֹאמַר לָהֶם: כְּשֶׁתָּבוֹאוּ לְאֶרֶץ מְגוּרֵיכֶם שֶׁאֲנִי נוֹתֵן לָכֶם.
ג וַעֲשִׂיתֶ֨ם אִשֶּׁ֤ה לַֽיהֹוָה֙ עֹלָ֣ה אוֹ־זֶ֔בַח לְפַלֵּא־נֶ֙דֶר֙ א֣וֹ בִנְדָבָ֔ה א֖וֹ בְּמֹעֲדֵיכֶ֑ם לַעֲשׂ֞וֹת רֵ֤יחַ נִיחֹ֙חַ֙ לַֽיהֹוָ֔ה מִן־הַבָּקָ֖ר א֥וֹ מִן־הַצֹּֽאן׃ וְתַעֲשׂוּ קָרְבָּן עוֹלֶה בָּאֵשׁ לַה' – עוֹלָה אוֹ זֶבַח שֶׁנֶּאֱכָל, עֲבוּר הִתְחַיְּבוּת בְּנֶדֶר אוֹ בִּנְדָבָה, אוֹ בְּמוֹעֲדֵיכֶם, כְּדֵי לַעֲשׂוֹת נַחַת רוּחַ לַה' מֵהַבָּקָר אוֹ מֵהַצֹּאן.
ד וְהִקְרִ֛יב הַמַּקְרִ֥יב קׇרְבָּנ֖וֹ לַֽיהֹוָ֑ה מִנְחָה֙ סֹ֣לֶת עִשָּׂר֔וֹן בָּל֕וּל בִּרְבִעִ֥ית הַהִ֖ין שָֽׁמֶן׃ הַמַּקְרִיב אֶת קָרְבָּנוֹ לַה' יַקְרִיב מִנְחָה שֶׁל עֲשִׂירִית אֵיפָה (כִּ2.16 לִיטֶר) סוֹלֶת, מְעֻרְבֶּבֶת בְּרֶבַע הִין (כְּ0.9 לִיטֶר) שֶׁמֶן.
ה וְיַ֤יִן לַנֶּ֙סֶךְ֙ רְבִיעִ֣ית הַהִ֔ין תַּעֲשֶׂ֥ה עַל־הָעֹלָ֖ה א֣וֹ לַזָּ֑בַח לַכֶּ֖בֶשׂ הָאֶחָֽד׃ וְתַקְרִיב עִם הָעוֹלָה אוֹ הַזֶּבַח רֶבַע הִין (כְּ0.9 לִיטֶר), יַיִן לְהַגָּשָׁה לַמִּזְבֵּחַ עֲבוּר כָּל כֶּבֶשׂ.
ו א֤וֹ לָאַ֙יִל֙ תַּעֲשֶׂ֣ה מִנְחָ֔ה סֹ֖לֶת שְׁנֵ֣י עֶשְׂרֹנִ֑ים בְּלוּלָ֥ה בַשֶּׁ֖מֶן שְׁלִשִׁ֥ית הַהִֽין׃ וְלָאַיִל תַּעֲשֶׂה מִנְחָה שֶׁל שְׁנֵי עֶשְׂרוֹנִים (כְּ4.33 לִיטֶר) סוֹלֶת, מְעֻרְבֶּבֶת בִּשְׁלִישׁ הִין (כְּ1.2 לִיטֶר) שֶׁמֶן.
ז וְיַ֥יִן לַנֶּ֖סֶךְ שְׁלִשִׁ֣ית הַהִ֑ין תַּקְרִ֥יב רֵֽיחַ־נִיחֹ֖חַ לַיהֹוָֽה׃ וְתַקְרִיב שְׁלִישׁ הִין (כְּ1.2 לִיטֶר) יַיִן לְהַגָּשָׁה לַמִּזְבֵּחַ לְנַחַת רוּחַ לַה'.

חמישי[עריכה]

ח וְכִֽי־תַעֲשֶׂ֥ה בֶן־בָּקָ֖ר עֹלָ֣ה אוֹ־זָ֑בַח לְפַלֵּא־נֶ֥דֶר אֽוֹ־שְׁלָמִ֖ים לַֽיהֹוָֽה׃ וְאִם תַּעֲשֶׂה בֶּן בָּקָר לְעוֹלָה אוֹ לְזֶבַח שֶׁנֶּאֱכַל, עֲבוּר הִתְחַיְּבוּת בְּנֶדֶר אוֹ נְדָבָה שֶׁל קָרְבַּן שְׁלָמִים, לַה' –
ט וְהִקְרִ֤יב עַל־בֶּן־הַבָּקָר֙ מִנְחָ֔ה סֹ֖לֶת שְׁלֹשָׁ֣ה עֶשְׂרֹנִ֑ים בָּל֥וּל בַּשֶּׁ֖מֶן חֲצִ֥י הַהִֽין׃ יַקְרִיב עִם בֶּן הַבָּקָר מִנְחָה – שָׁלוֹשׁ עֲשִׂירִיּוֹת הָאֵיפָה (כְּ6.5 לִיטֶר) סוֹלֶת מְעֻרְבֶּבֶת בַּחֲצִי הִין (כְּ1.8 לִיטֶר) שֶׁמֶן.
י וְיַ֛יִן תַּקְרִ֥יב לַנֶּ֖סֶךְ חֲצִ֣י הַהִ֑ין אִשֵּׁ֥ה רֵֽיחַ־נִיחֹ֖חַ לַיהֹוָֽה׃ וְתַקְרִיב חֲצִי הִין (כְּ1.8 לִיטֶר) יַיִן לְהַגָּשָׁה לַמִּזְבֵּחַ, קָרְבָּן עוֹלֶה בָּאֵשׁ לְנַחַת רוּחַ לַה'.
יא כָּ֣כָה יֵעָשֶׂ֗ה לַשּׁוֹר֙ הָֽאֶחָ֔ד א֖וֹ לָאַ֣יִל הָאֶחָ֑ד אֽוֹ־לַשֶּׂ֥ה בַכְּבָשִׂ֖ים א֥וֹ בָעִזִּֽים׃ כָּכָה יֵעָשֶׂה לְכָל שׁוֹר, לְכָל אַיִל, וּלְכָל טָלֶה כְּבָשִׂים אוֹ גְּדִי עִזִּים.
יב כַּמִּסְפָּ֖ר אֲשֶׁ֣ר תַּעֲשׂ֑וּ כָּ֛כָה תַּעֲשׂ֥וּ לָאֶחָ֖ד כְּמִסְפָּרָֽם׃ כְּמִסְפַּר הַבְּהֵמוֹת שֶׁתַּקְרִיבוּ כָּכָה תַּעֲשׂוּ עֲבוּר כָּל אֶחָד – כְּמִסְפָּרָם.
יג כׇּל־הָאֶזְרָ֥ח יַעֲשֶׂה־כָּ֖כָה אֶת־אֵ֑לֶּה לְהַקְרִ֛יב אִשֵּׁ֥ה רֵֽיחַ־נִיחֹ֖חַ לַֽיהֹוָֽה׃ כָּל אֶזְרָח יַעֲשֶׂה אֶת אֵלֶּה כָּכָה, לְהַקְרִיב קָרְבָּן עוֹלֶה בָּאֵשׁ לְנַחַת רוּחַ לַה'.
יד וְכִֽי־יָגוּר֩ אִתְּכֶ֨ם גֵּ֜ר א֤וֹ אֲשֶׁר־בְּתֽוֹכְכֶם֙ לְדֹרֹ֣תֵיכֶ֔ם וְעָשָׂ֛ה אִשֵּׁ֥ה רֵֽיחַ־נִיחֹ֖חַ לַיהֹוָ֑ה כַּאֲשֶׁ֥ר תַּעֲשׂ֖וּ כֵּ֥ן יַעֲשֶֽׂה׃ וְאִם יָבוֹא גֵּר לָגוּר אִתְּכֶם בְּאַחַד הַדּוֹרוֹת אוֹ שֶׁכְּבָר נִמְצָא אֶצְלֵיכֶם, וְיַעֲשֶׂה קָרְבָּן עוֹלֶה בָּאֵשׁ לְנַחַת רוּחַ לַה', כְּמוֹ שֶׁאַתֶּם עוֹשִׂים כָּךְ הוּא יַעֲשֶׂה.
טו הַקָּהָ֕ל חֻקָּ֥ה אַחַ֛ת לָכֶ֖ם וְלַגֵּ֣ר הַגָּ֑ר חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם כָּכֶ֛ם כַּגֵּ֥ר יִהְיֶ֖ה לִפְנֵ֥י יְהֹוָֽה׃ כָּל הַקָּהָל – חוֹק אֶחָד יִהְיֶה לָכֶם וְלַגֵּר שֶׁגָּר אִתְּכֶם, חוֹק לְעוֹלָם לְדוֹרוֹת, הַגֵּר יִהְיֶה כְּמוֹכֶם לִפְנֵי ה'.
טז תּוֹרָ֥ה אַחַ֛ת וּמִשְׁפָּ֥ט אֶחָ֖ד יִהְיֶ֣ה לָכֶ֑ם וְלַגֵּ֖ר הַגָּ֥ר אִתְּכֶֽם׃ הוֹרָאָה אַחַת וּמִשְׁפָּט אֶחָד יִהְיֶה לָכֶם וְלַגֵּר שֶׁגָּר אִתְּכֶם.

שישי[עריכה]

פ פָּרָשַׁת הַחַלָּה
יז וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יח דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָמַרְתָּ֖ אֲלֵהֶ֑ם בְּבֹֽאֲכֶם֙ אֶל־הָאָ֔רֶץ אֲשֶׁ֥ר אֲנִ֛י מֵבִ֥יא אֶתְכֶ֖ם שָֽׁמָּה׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְתֹאמַר לָהֶם: כְּשֶׁתָּבוֹאוּ לָאָרֶץ שֶׁאֲנִי מֵבִיא אֶתְכֶם לְשָׁם,
יט וְהָיָ֕ה בַּאֲכׇלְכֶ֖ם מִלֶּ֣חֶם הָאָ֑רֶץ תָּרִ֥ימוּ תְרוּמָ֖ה לַיהֹוָֽה׃ כְּשֶׁתֹּאכְלוּ אָז לֶחֶם מִתְּבוּאַת הָאָרֶץ, תַּפְרִישׁוּ תְּרוּמָה לַה'.
כ רֵאשִׁית֙ עֲרִסֹ֣תֵכֶ֔ם חַלָּ֖ה תָּרִ֣ימוּ תְרוּמָ֑ה כִּתְרוּמַ֣ת גֹּ֔רֶן כֵּ֖ן תָּרִ֥ימוּ אֹתָֽהּ׃ אֶת רֵאשִׁית הַבָּצֵק שֶׁלָּכֶם תַּפְרִישׁוּ לְחַלָּה, כְּמוֹ שֶׁמַּפְרִישִׁים מֵהַגּוֹרֶן כָּךְ תַּפְרִישׁוּ אוֹתָהּ.
כא מֵרֵאשִׁית֙ עֲרִסֹ֣תֵיכֶ֔ם תִּתְּנ֥וּ לַיהֹוָ֖ה תְּרוּמָ֑ה לְדֹרֹ֖תֵיכֶֽם׃ אֶת רֵאשִׁית הַבָּצֵק שֶׁלָּכֶם תִּתְּנוּ לַה' תְּרוּמָה, בְּכָל הַדּוֹרוֹת.
ס פָּרָשַׁת הֶעְלֵם דָּבָר
כב וְכִ֣י תִשְׁגּ֔וּ וְלֹ֣א תַעֲשׂ֔וּ אֵ֥ת כׇּל־הַמִּצְוֺ֖ת הָאֵ֑לֶּה אֲשֶׁר־דִּבֶּ֥ר יְהֹוָ֖ה אֶל־מֹשֶֽׁה׃ וְאִם תִּטְעוּ וְלֹא תַּעֲשׂוּ אֶת כָּל הַמִּצְווֹת הָאֵלֶּה שֶׁה' דִּבֵּר אֶל מֹשֶׁה –
כג אֵת֩ כׇּל־אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֲלֵיכֶ֖ם בְּיַד־מֹשֶׁ֑ה מִן־הַיּ֞וֹם אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה וָהָ֖לְאָה לְדֹרֹתֵיכֶֽם׃ אֶת כָּל מַה שֶּׁה' צִוָּה אֶתְכֶם עַל יְדֵי מֹשֶׁה מֵהַיּוֹם שֶׁה' צִוָּה וָהָלְאָה, אִם תִּטְעוּ בְּאַחַד הַדּוֹרוֹת –
כד וְהָיָ֗ה אִ֣ם מֵעֵינֵ֣י הָעֵדָה֮ נֶעֶשְׂתָ֣ה לִשְׁגָגָה֒ וְעָשׂ֣וּ כׇל־הָעֵדָ֡ה פַּ֣ר בֶּן־בָּקָר֩ אֶחָ֨ד לְעֹלָ֜ה לְרֵ֤יחַ נִיחֹ֙חַ֙ לַֽיהֹוָ֔ה וּמִנְחָת֥וֹ וְנִסְכּ֖וֹ כַּמִּשְׁפָּ֑ט וּשְׂעִיר־עִזִּ֥ים אֶחָ֖ד לְחַטָּֽת׃ אִם נֶעֶשְׂתָה טָעוּת בְּהוֹרָאַת הַסַּנְהֶדְרִין, כָּל הַצִּבּוּר יַקְרִיבוּ פַּר אֶחָד לְעוֹלָה לְנַחַת רוּחַ לַה', וּמִנְחָתוֹ וּשְׁפִיכַת הַיַּיִן שֶׁאִתּוֹ כַּדִּין, וְתַיִשׁ אֶחָד לְחַטָּאת.
כה וְכִפֶּ֣ר הַכֹּהֵ֗ן עַֽל־כׇּל־עֲדַ֛ת בְּנֵ֥י יִשְׂרָאֵ֖ל וְנִסְלַ֣ח לָהֶ֑ם כִּֽי־שְׁגָגָ֣ה הִ֔וא וְהֵם֩ הֵבִ֨יאוּ אֶת־קׇרְבָּנָ֜ם אִשֶּׁ֣ה לַֽיהֹוָ֗ה וְחַטָּאתָ֛ם לִפְנֵ֥י יְהֹוָ֖ה עַל־שִׁגְגָתָֽם׃ הַכֹּהֵן יְכַפֵּר עַל כָּל צִבּוּר בְּנֵי יִשְׂרָאֵל וְיִסָּלַח לָהֶם כִּי זוֹ הָיְתָה טָעוּת וְהֵם הֵבִיאוּ עַל טָעוּתָם אֶת הַקָּרְבַּן שֶׁעוֹלֶה בָּאֵשׁ לַה' וְאֶת קָרְבַּן חַטָּאתָם לִפְנֵי ה'.
כו וְנִסְלַ֗ח לְכׇל־עֲדַת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וְלַגֵּ֖ר הַגָּ֣ר בְּתוֹכָ֑ם כִּ֥י לְכׇל־הָעָ֖ם בִּשְׁגָגָֽה׃ וְיִסָּלַח לְכָל צִבּוּר בְּנֵי יִשְׂרָאֵל וְלַגֵּרִים שֶׁגָּרִים בְּתוֹכָם כִּי כָּל הָעָם עֲשׂוּ בְּטָעוּת.

שביעי[עריכה]

ס פָּרָשַׁת חֵטְא הַיָּחִיד
כז וְאִם־נֶ֥פֶשׁ אַחַ֖ת תֶּחֱטָ֣א בִשְׁגָגָ֑ה וְהִקְרִ֛יבָה עֵ֥ז בַּת־שְׁנָתָ֖הּ לְחַטָּֽאת׃ וְאִם אִישׁ יָחִיד יֶחֱטָא בְּטָעוּת, הוּא יַקְרִיב עֵז שֶׁגִּילָהּ עַד שָׁנָה לְחַטָּאת.
כח וְכִפֶּ֣ר הַכֹּהֵ֗ן עַל־הַנֶּ֧פֶשׁ הַשֹּׁגֶ֛גֶת בְּחֶטְאָ֥הֿ בִשְׁגָגָ֖ה לִפְנֵ֣י יְהֹוָ֑ה לְכַפֵּ֥ר עָלָ֖יו וְנִסְלַ֥ח לֽוֹ׃ הַכֹּהֵן יְכַפֵּר עַל הָאִישׁ הַטּוֹעֶה שֶׁחָטָא בְּטָעוּת לִפְנֵי ה', הוּא יְכַפֵּר עָלָיו וְיִסָּלַח לוֹ.
כט הָֽאֶזְרָח֙ בִּבְנֵ֣י יִשְׂרָאֵ֔ל וְלַגֵּ֖ר הַגָּ֣ר בְּתוֹכָ֑ם תּוֹרָ֤ה אַחַת֙ יִהְיֶ֣ה לָכֶ֔ם לָעֹשֶׂ֖ה בִּשְׁגָגָֽה׃ לָאֶזְרָח בִּבְנֵי יִשְׂרָאֵל וְלַגֵּר שֶׁגָּר בְּתוֹכָם, תִּהְיֶה לָכֶם הוֹרָאָה אַחַת לָעוֹשֶׂה בְּטָעוּת.
ל וְהַנֶּ֜פֶשׁ אֲשֶֽׁר־תַּעֲשֶׂ֣ה ׀ בְּיָ֣ד רָמָ֗ה מִן־הָֽאֶזְרָח֙ וּמִן־הַגֵּ֔ר אֶת־יְהֹוָ֖ה ה֣וּא מְגַדֵּ֑ף וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מִקֶּ֥רֶב עַמָּֽהּ׃ וְכָל אָדָם – אֶזְרָח אוֹ גֵּר, שֶׁיַּעֲשֶׂה בְּמֵזִיד וּבְיָד מוּרֶמֶת בְּחֻצְפָּה, הוּא מְבַזֶּה אֶת ה' – הַנֶּפֶשׁ הַהִיא תֹּאבַד מִתּוֹךְ עַמָּהּ.
לא כִּ֤י דְבַר־יְהֹוָה֙ בָּזָ֔ה וְאֶת־מִצְוָת֖וֹ הֵפַ֑ר הִכָּרֵ֧ת ׀ תִּכָּרֵ֛ת הַנֶּ֥פֶשׁ הַהִ֖וא עֲוֺנָ֥הֿ בָֽהּ׃ כִּי הוּא בִּזָּה אֶת דְּבַר ה', וְהֵפֵר אֶת מִצְוָתוֹ – הַנֶּפֶשׁ הַהִיא תֹּאבַד כִּי יֵשׁ בָּהּ עֲווֹן.
פ פָּרָשַׁת הַמְּקוֹשֵׁשׁ
לב וַיִּהְי֥וּ בְנֵֽי־יִשְׂרָאֵ֖ל בַּמִּדְבָּ֑ר וַֽיִּמְצְא֗וּ אִ֛ישׁ מְקֹשֵׁ֥שׁ עֵצִ֖ים בְּי֥וֹם הַשַּׁבָּֽת׃ בְּנֵי יִשְׂרָאֵל הָיוּ בַּמִּדְבָּר וּמָצְאוּ אִישׁ אוֹסֵף עֵצִים בְּיוֹם הַשַּׁבָּת.
לג וַיַּקְרִ֣יבוּ אֹת֔וֹ הַמֹּצְאִ֥ים אֹת֖וֹ מְקֹשֵׁ֣שׁ עֵצִ֑ים אֶל־מֹשֶׁה֙ וְאֶֽל־אַהֲרֹ֔ן וְאֶ֖ל כׇּל־הָעֵדָֽה׃ אֵלֶּה שֶׁמָּצְאוּ אוֹתוֹ אוֹסֵף עֵצִים הֵבִיאוּ אוֹתוֹ לְמֹשֶׁה וּלְאַהֲרֹן וּלְכָל הַצִּבּוּר.
לד וַיַּנִּ֥יחוּ אֹת֖וֹ בַּמִּשְׁמָ֑ר כִּ֚י לֹ֣א פֹרַ֔שׁ מַה־יֵּעָשֶׂ֖ה לֽוֹ׃ הֵם הֵנִיחוּ אוֹתוֹ בְּמַעֲצָר כִּי לֹא פֵּרְשׁוּ לָהֶם מַה לַּעֲשׂוֹת לוֹ.

לה וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה מ֥וֹת יוּמַ֖ת הָאִ֑ישׁ רָג֨וֹם אֹת֤וֹ בָֽאֲבָנִים֙ כׇּל־הָ֣עֵדָ֔ה מִח֖וּץ לַֽמַּחֲנֶֽה׃ ה' אָמַר לְמֹשֶׁה: הָאִישׁ יוּמַת, כָּל הַצִּבּוּר יִרְגְּמוּ אוֹתוֹ בַּאֲבָנִים מִחוּץ לַמַּחֲנֶה.
לו וַיֹּצִ֨יאוּ אֹת֜וֹ כׇּל־הָעֵדָ֗ה אֶל־מִחוּץ֙ לַֽמַּחֲנֶ֔ה וַיִּרְגְּמ֥וּ אֹת֛וֹ בָּאֲבָנִ֖ים וַיָּמֹ֑ת כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ כָּל הַצִּבּוּר הוֹצִיאוּ אוֹתוֹ אֶל מִחוּץ לַמַּחֲנֶה, וְרָגְמוּ אוֹתוֹ בַּאֲבָנִים והוּא מֵת, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

מפטיר[עריכה]

פ פָּרָשַׁת צִיצִית
לז וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' אָמַר לְמֹשֶׁה:
לח דַּבֵּ֞ר אֶל־בְּנֵ֤י יִשְׂרָאֵל֙ וְאָמַרְתָּ֣ אֲלֵהֶ֔ם וְעָשׂ֨וּ לָהֶ֥ם צִיצִ֛ת עַל־כַּנְפֵ֥י בִגְדֵיהֶ֖ם לְדֹרֹתָ֑ם וְנָ֥תְנ֛וּ עַל־צִיצִ֥ת הַכָּנָ֖ף פְּתִ֥יל תְּכֵֽלֶת׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְתֹאמַר לָהֶם שֶׂיַּעֲשׂוּ לָהֶם חוּטִים עַל קְצָווֹת בִּגְדֵיהֶם, בְּכָל הַדּוֹרוֹת, וְיָשִׂימוּ עַל הַחוּטִים שֶׁעַל קְצֵה הַבֶּגֶד פְּתִיל תְּכֵלֶת.
לט וְהָיָ֣ה לָכֶם֮ לְצִיצִת֒ וּרְאִיתֶ֣ם אֹת֗וֹ וּזְכַרְתֶּם֙ אֶת־כׇּל־מִצְוֺ֣ת יְהֹוָ֔ה וַעֲשִׂיתֶ֖ם אֹתָ֑ם וְלֹֽא־תָת֜וּרוּ אַחֲרֵ֤י לְבַבְכֶם֙ וְאַחֲרֵ֣י עֵֽינֵיכֶ֔ם אֲשֶׁר־אַתֶּ֥ם זֹנִ֖ים אַחֲרֵיהֶֽם׃ וְזֶה יִהְיֶה לָכֶם צִיצִית, וְתִרְאוּ אוֹתוֹ וְתִזְכְּרוּ אֶת כָּל מִצְווֹת ה' וְתַעֲשׂוּ אוֹתָם, וְלֹא תִּמָּשְׁכוּ אַחֲרֵי לִבְּכֶם וְאַחֲרֵי עֵינֵיכֶם שֶׁאַתֶּם סוֹטִים אַחֲרֵיהֶם.
מ לְמַ֣עַן תִּזְכְּר֔וּ וַעֲשִׂיתֶ֖ם אֶת־כׇּל־מִצְוֺתָ֑י וִהְיִיתֶ֥ם קְדֹשִׁ֖ים לֵאלֹֽהֵיכֶֽם׃ כְּדֵי שֶׁתִּזְכְּרוּ וְתַעֲשׂוּ אֶת כָּל מִצְווֹתַי, וְתִהְיוּ קְדוֹשִׁים לֵאלֹהֵיכֶם.
מא אֲנִ֞י יְהֹוָ֣ה אֱלֹֽהֵיכֶ֗ם אֲשֶׁ֨ר הוֹצֵ֤אתִי אֶתְכֶם֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לִהְי֥וֹת לָכֶ֖ם לֵאלֹהִ֑ים אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ אֲנִי ה' אֱלֹהֵיכֶם שֶׁהוֹצֵאתִי אֶתְכֶם מֵאֶרֶץ מִצְרַיִם כְּדֵי לִהְיוֹת לָכֶם לֵאלֹהִים, אֲנִי ה' אֱלֹהֵיכֶם.
