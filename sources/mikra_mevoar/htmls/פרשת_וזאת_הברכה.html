ראשון[עריכה]

פ פָּרָשַׁת בִּרְכַּת מֹשֶׁה
לג א וְזֹ֣את הַבְּרָכָ֗ה אֲשֶׁ֨ר בֵּרַ֥ךְ מֹשֶׁ֛ה אִ֥ישׁ הָאֱלֹהִ֖ים אֶת־בְּנֵ֣י יִשְׂרָאֵ֑ל לִפְנֵ֖י מוֹתֽוֹ׃ וְזֹאת הַבְּרָכָה שֶׁמֹּשֶׁה אִישׁ הָאֱלֹהִים בֵּרֵךְ אֶת בְּנֵי יִשְׂרָאֵל לִפְנֵי מוֹתוֹ.
ב וַיֹּאמַ֗ר יְהֹוָ֞ה מִסִּינַ֥י בָּא֙ וְזָרַ֤ח מִשֵּׂעִיר֙ לָ֔מוֹ הוֹפִ֙יעַ֙ מֵהַ֣ר פָּארָ֔ן וְאָתָ֖ה מֵרִבְבֹ֣ת קֹ֑דֶשׁ מִימִינ֕וֹ אשדת [אֵ֥שׁ דָּ֖ת] לָֽמוֹ׃ הוּא אָמַר: ה' בָּא מִסִּינַי בְּמַתַּן תּוֹרָה, וְהֵאִיר לְיִשְׂרָאֵל מִשֵּׂעִיר וְהוֹפִיעַ מֵהַר פָּארָן הַקְּרוֹבִים לְסִינַי, וּבָא מְלֻוֶּה בְּרִבְבוֹת מַלְאֲכֵי קוֹדֶשׁ, כְּשֶׁבִּימִינוֹ דַּת הַתּוֹרָה כְּתוּבָה בְּאֵשׁ לְיִשְׂרָאֵל.
ג אַ֚ף חֹבֵ֣ב עַמִּ֔ים כׇּל־קְדֹשָׁ֖יו בְּיָדֶ֑ךָ וְהֵם֙ תֻּכּ֣וּ לְרַגְלֶ֔ךָ יִשָּׂ֖א מִדַּבְּרֹתֶֽיךָ׃ ה' גַּם חִבֵּב אֶת הַשְּׁבָטִים – לָקַח בְּיָדוֹ אֶת כָּל יִשְׂרָאֵל קְדוֹשָׁיו, הֵם הֻכְנְסוּ תַּחַת כַּנְפֵי הַשְּׁכִינָה – מְרוֹמָמִים עַל יְדֵי דִּבּוּרָיו.
ד תּוֹרָ֥ה צִוָּה־לָ֖נוּ מֹשֶׁ֑ה מוֹרָשָׁ֖ה קְהִלַּ֥ת יַעֲקֹֽב׃ מֹשֶׁה צִוָּה לָנוּ אֶת הַתּוֹרָה, יְרֻשָּׁה לִקְהִלַּת יַעֲקֹב.
ה וַיְהִ֥י בִישֻׁר֖וּן מֶ֑לֶךְ בְּהִתְאַסֵּף֙ רָ֣אשֵׁי עָ֔ם יַ֖חַד שִׁבְטֵ֥י יִשְׂרָאֵֽל׃ מֹשֶׁה הָיָה מֶלֶךְ בְּיִשְׂרָאֵל, כְּשֶׁהִתְאַסְּפוּ רָאשֵׁי הָעָם, שִׁבְטֵי יִשְׂרָאֵל נִשְׁמְעוּ לוֹ יַחַד.
ו יְחִ֥י רְאוּבֵ֖ן וְאַל־יָמֹ֑ת וִיהִ֥י מְתָ֖יו מִסְפָּֽר׃ יְחִי רְאוּבֵן וְלֹא יָמוּת בְּצֵאתוֹ לַקְּרָב רִאשׁוֹן לִפְנֵי הַמַּחֲנֶה, וַאֲנָשָׁיו יָשׁוּבוּ מֵהַקְּרָב בַּמִּסְפָּר שֶׁיָּצְאוּ.
ס
ז וְזֹ֣את לִיהוּדָה֮ וַיֹּאמַר֒ שְׁמַ֤ע יְהֹוָה֙ ק֣וֹל יְהוּדָ֔ה וְאֶל־עַמּ֖וֹ תְּבִיאֶ֑נּוּ יָדָיו֙ רָ֣ב ל֔וֹ וְעֵ֥זֶר מִצָּרָ֖יו תִּהְיֶֽה׃ וְזֹאת לִיהוּדָה – אָמַר: שְׁמַע ה' אֶת קוֹל תְּפִלַּת יְהוּדָה בְּצֵאתוֹ לַקְּרָב וְתַחֲזִיר אוֹתוֹ בְּשָׁלוֹם לְעַמּוֹ, הוּא יַצְלִיחַ מְאֹד בְּמַעֲשָׂיו, וְתַעֲזוֹר לוֹ מִצּוֹרְרָיו.

שני[עריכה]

פ
ח וּלְלֵוִ֣י אָמַ֔ר תֻּמֶּ֥יךָ וְאוּרֶ֖יךָ לְאִ֣ישׁ חֲסִידֶ֑ךָ אֲשֶׁ֤ר נִסִּיתוֹ֙ בְּמַסָּ֔ה תְּרִיבֵ֖הוּ עַל־מֵ֥י מְרִיבָֽה׃ וְעַל לֵוִי אָמַר: ה', אֶת הָאֲבָנִים הַמְּאִירוֹת שֶׁעוֹנוֹת לִשְׁאֵלוֹת נָתַתָּ לְאַהֲרֹן – הָאִישׁ הַצַּדִּיק שֶׁלְּךָ, נִסִּיתָ אוֹתוֹ בְּנִסָּיוֹן וּבָחַנְתָּ אוֹתוֹ בָּאֵרוּעַ שֶׁל מֵי מְרִיבָה.
ט הָאֹמֵ֞ר לְאָבִ֤יו וּלְאִמּוֹ֙ לֹ֣א רְאִיתִ֔יו וְאֶת־אֶחָיו֙ לֹ֣א הִכִּ֔יר וְאֶת־בָּנָ֖ו לֹ֣א יָדָ֑ע כִּ֤י שָֽׁמְרוּ֙ אִמְרָתֶ֔ךָ וּבְרִיתְךָ֖ יִנְצֹֽרוּ׃ כְּשֶׁצָּרִיךְ הָיָה לְהַעֲנִישׁ וְלַהֲרוֹג בְּחֵטְא הָעֵגֶל – שֵׁבֶט לֵוִי הִתְיַחֵס לְאָבִיו וְאִמּוֹ כְּאִלּוּ לֹא רָאָה אוֹתָם מֵעוֹלָם וְלֹא הִכִּיר אֶת אֶחָיו וְלֹא יָדַע מִי הֵם בָּנָיו, כִּי הֵם הִקְפִּידוּ עַל מִצְווֹתֶיךָ וְהֵם יְקַיְּמוּ אֶת הַהֶסְכֵּם אִתְּךָ.
י יוֹר֤וּ מִשְׁפָּטֶ֙יךָ֙ לְיַֽעֲקֹ֔ב וְתוֹרָתְךָ֖ לְיִשְׂרָאֵ֑ל יָשִׂ֤ימוּ קְטוֹרָה֙ בְּאַפֶּ֔ךָ וְכָלִ֖יל עַֽל־מִזְבְּחֶֽךָ׃ הֵם יְלַמְּדוּ אֶת מִצְווֹתֶיךָ לְיַעֲקֹב וְאֶת תּוֹרָתְךָ לְיִשְׂרָאֵל, הֵם יָשִׂימוּ בַּמִּקְדָּשׁ קְטוֹרֶת בְּאַפְּךָ וְקָרְבָּן עַל מִזְבַּחֲךָ.
יא בָּרֵ֤ךְ יְהֹוָה֙ חֵיל֔וֹ וּפֹ֥עַל יָדָ֖יו תִּרְצֶ֑ה מְחַ֨ץ מׇתְנַ֧יִם קָמָ֛יו וּמְשַׂנְאָ֖יו מִן־יְקוּמֽוּן׃ בָּרֵךְ ה' אֶת נְכָסָיו, וְאֶת מַעֲשֵׂה יָדָיו תְּקַבֵּל בְּרָצוֹן, מְחַץ אֶת הַמָּתְנַיִם שֶׁל הַקָּמִים לְהִלָּחֵם בּוֹ וְשֶׁל שׂוֹנְאָיו, שֶׁלֹּא יָקוּמוּ מֵהַתְּבוּסָה.
ס
יב לְבִנְיָמִ֣ן אָמַ֔ר יְדִ֣יד יְהֹוָ֔ה יִשְׁכֹּ֥ן לָבֶ֖טַח עָלָ֑יו חֹפֵ֤ף עָלָיו֙ כׇּל־הַיּ֔וֹם וּבֵ֥ין כְּתֵפָ֖יו שָׁכֵֽן׃ עַל בִּנְיָמִין אָמַר שֶׁהוּא אָהוּב לַה', הוּא יִשְׁכּוֹן בְּאַרְצוֹ בְּבִטָּחוֹן בַּה', ה' מָגֵן עָלָיו כָּל הַיּוֹם, וּבֵית מִקְדָּשׁוֹ שׁוֹכֵן בֵּין הָרָיו.

שלישי[עריכה]

ס
יג וּלְיוֹסֵ֣ף אָמַ֔ר מְבֹרֶ֥כֶת יְהֹוָ֖ה אַרְצ֑וֹ מִמֶּ֤גֶד שָׁמַ֙יִם֙ מִטָּ֔ל וּמִתְּה֖וֹם רֹבֶ֥צֶת תָּֽחַת׃ וּלְיוֹסֵף אָמַר: אַרְצוֹ מְבוֹרֶכֶת בְּבִרְכַּת ה' – בְּפֵירוֹת מְתוּקִים שֶׁמּוּשְׁקִים מִטַּל הַשָּׁמַיִם וּמִמַּעְיְנוֹת הַתְּהוֹם שֶׁמִּתַּחַת לָאָרֶץ.
יד וּמִמֶּ֖גֶד תְּבוּאֹ֣ת שָׁ֑מֶשׁ וּמִמֶּ֖גֶד גֶּ֥רֶשׁ יְרָחִֽים׃ וּבְפֵירוֹת מְתוּקִים מֵהַשְׁפַּעַת הַשֶּׁמֶשׁ, וּבְפֵירוֹת מְתוּקִים מֵהַשְׁפַּעַת הַיָּרֵחַ.
טו וּמֵרֹ֖אשׁ הַרְרֵי־קֶ֑דֶם וּמִמֶּ֖גֶד גִּבְע֥וֹת עוֹלָֽם׃ וּבְהָרִים שֶׁפֵּירוֹתֵיהֶם מַקְדִּימִים לְהַבְשִׁיל רִאשׁוֹנִים, וּבִגְבָעוֹת שֶׁמַּצְמִיחוֹת תָּמִיד פֵּירוֹת מְתוּקִים.
טז וּמִמֶּ֗גֶד אֶ֚רֶץ וּמְלֹאָ֔הּ וּרְצ֥וֹן שֹׁכְנִ֖י סְנֶ֑ה תָּב֙וֹאתָה֙ לְרֹ֣אשׁ יוֹסֵ֔ף וּלְקׇדְקֹ֖ד נְזִ֥יר אֶחָֽיו׃ וּמִטּוּב אֶרֶץ וּמְלוֹאָהּ מֵרְצוֹן ה' שֶׁנִּגְלָה אֵלַי בַּסְּנֶה, תָּבוֹא בְּרָכָה זוֹ לְרֹאשׁ יוֹסֵף – לַקָּדְקוֹד שֶׁל הַמְּרוֹמָם שֶׁבְּאֶחָיו.
יז בְּכ֨וֹר שׁוֹר֜וֹ הָדָ֣ר ל֗וֹ וְקַרְנֵ֤י רְאֵם֙ קַרְנָ֔יו בָּהֶ֗ם עַמִּ֛ים יְנַגַּ֥ח יַחְדָּ֖ו אַפְסֵי־אָ֑רֶץ וְהֵם֙ רִבְב֣וֹת אֶפְרַ֔יִם וְהֵ֖ם אַלְפֵ֥י מְנַשֶּֽׁה׃ הוּא כְּמוֹ שׁוֹר בְּכוֹר חָזָק בַּעַל יוֹפִי וְהָדָר, וְקַרְנָיו הֵם קַרְנֵי רְאֵם בָּהֶם יְנַגֵּחַ יַחַד אֶת עַמֵּי אוֹיְבָיו עַד סוֹף הָאָרֶץ, וְהֵם רִבְבוֹת לוֹחֲמֵי אֶפְרַיִם וְאַלְפֵי לוֹחֲמֵי מְנַשֶּׁה.

רביעי[עריכה]

ס
יח וְלִזְבוּלֻ֣ן אָמַ֔ר שְׂמַ֥ח זְבוּלֻ֖ן בְּצֵאתֶ֑ךָ וְיִשָּׂשכָ֖ר בְּאֹהָלֶֽיךָ׃ וְלִזְבוּלוּן אָמַר: שְׂמַח זְבוּלוּן בְּצֵאתְךָ לַקְּרָב וּלְמִסְחָר, וְיִשָּׂשכָר בְּשִׁבְתְּךָ בְּאוֹהָלֶיךָ.
יט עַמִּים֙ הַר־יִקְרָ֔אוּ שָׁ֖ם יִזְבְּח֣וּ זִבְחֵי־צֶ֑דֶק כִּ֣י שֶׁ֤פַע יַמִּים֙ יִינָ֔קוּ וּשְׂפֻנֵ֖י טְמ֥וּנֵי חֽוֹל׃ אַנְשֵׁי זְבוּלוּן יִקְרְאוּ לָעַמִּים לְהַר הַבַּיִת, שָׁם יַקְרִיבוּ זִבְחֵי אֱמֶת, כִּי הֵם יִינְקוּ שֶׁפַע מֵהַיַּמִּים, וְאוֹצָרוֹת שֶׁמֻּחְבָּאִים וּטְמוּנִים בַּחוֹל.
ס
כ וּלְגָ֣ד אָמַ֔ר בָּר֖וּךְ מַרְחִ֣יב גָּ֑ד כְּלָבִ֣יא שָׁכֵ֔ן וְטָרַ֥ף זְר֖וֹעַ אַף־קׇדְקֹֽד׃ וּלְגָד אָמַר: בָּרוּךְ ה' שֶׁמַּרְחִיב אֶת גְּבוּלוֹ שֶׁל גָּד, הוּא שׁוֹכֵן בְּבִטָּחוֹן כְּאַרְיֵה וְיִטְרוֹף אֶת אוֹיְבָיו – זְרוֹעַ עִם הָרֹאשׁ.
כא וַיַּ֤רְא רֵאשִׁית֙ ל֔וֹ כִּי־שָׁ֛ם חֶלְקַ֥ת מְחֹקֵ֖ק סָפ֑וּן וַיֵּתֵא֙ רָ֣אשֵׁי עָ֔ם צִדְקַ֤ת יְהֹוָה֙ עָשָׂ֔ה וּמִשְׁפָּטָ֖יו עִם־יִשְׂרָאֵֽל׃ הוּא בָּחַר לְעַצְמוֹ אֶת הַחֵלֶק הָרִאשׁוֹן שֶׁנִּכְבַּשׁ, שֶׁשָּׁם חֶלְקַת הַקֶּבֶר שֶׁבָּהּ טָמוּן מֹשֶׁה מוֹסֵר הַתּוֹרָה, הוּא יֵצֵא בְּרֹאשׁ הָעָם וּבְכָךְ יְקַיֵּם אֶת הַבְטָחָתוֹ לַה' וְאֶת הִתְחַיְּבוּתוֹ לְיִשְׂרָאֵל.

חמישי[עריכה]

ס
כב וּלְדָ֣ן אָמַ֔ר דָּ֖ן גּ֣וּר אַרְיֵ֑ה יְזַנֵּ֖ק מִן־הַבָּשָֽׁן׃ וּלְדָן אָמַר: דָּן גּוּר אַרְיֵה יְזַנֵּק מֵהַבָּשָׁן.
כג וּלְנַפְתָּלִ֣י אָמַ֔ר נַפְתָּלִי֙ שְׂבַ֣ע רָצ֔וֹן וּמָלֵ֖א בִּרְכַּ֣ת יְהֹוָ֑ה יָ֥ם וְדָר֖וֹם יְרָֽשָׁה׃ וּלְנַפְתָּלִי אָמַר: נַפְתָּלִי שָׂבֵעַ וּמְרוּצֶה וּמָלֵא בְּבִרְכַּת ה', הוּא יִירַשׁ אֶת הַכִּנֶּרֶת וְאֶת הָעֵמֶק שֶׁלִּדְרוֹמָהּ.
ס
כד וּלְאָשֵׁ֣ר אָמַ֔ר בָּר֥וּךְ מִבָּנִ֖ים אָשֵׁ֑ר יְהִ֤י רְצוּי֙ אֶחָ֔יו וְטֹבֵ֥ל בַּשֶּׁ֖מֶן רַגְלֽוֹ׃ וּלְאָשֵׁר אָמַר: אָשֵׁר מְבוֹרָךְ בְּבָנָיו, יִהְיֶה רָצוּי לְאֶחָיו וְלֹא יְקַנְּאוּ בּוֹ, וְהוּא טוֹבֵל אֶת רַגְלוֹ בְּשֶׁמֶן בְּתַפְנוּקֵי מְלָכִים.
כה בַּרְזֶ֥ל וּנְחֹ֖שֶׁת מִנְעָלֶ֑ךָ וּכְיָמֶ֖יךָ דׇּבְאֶֽךָ׃ מַנְעוּלֵי שְׁעָרֶיךָ חֲזָקִים כְּבַרְזֶל וּנְחוֹשֶׁת, וְכִימֵי צְעִירוּתְךָ כָּךְ יִהְיוּ יְמֵי זִקְנוּתְךָ.
כו אֵ֥ין כָּאֵ֖ל יְשֻׁר֑וּן רֹכֵ֤ב שָׁמַ֙יִם֙ בְּעֶזְרֶ֔ךָ וּבְגַאֲוָת֖וֹ שְׁחָקִֽים׃ אֵין כֵּאלֹהֵי יִשְׂרָאֵל, הָרוֹכֵב בַּשָּׁמַיִם הוּא שֶׁעוֹזֵר לְךָ, וּכְבוֹדוֹ בַּשְּׁחָקִים.

ששי[עריכה]

כז מְעֹנָהֿ֙ אֱלֹ֣הֵי קֶ֔דֶם וּמִתַּ֖חַת זְרֹעֹ֣ת עוֹלָ֑ם וַיְגָ֧רֶשׁ מִפָּנֶ֛יךָ אוֹיֵ֖ב וַיֹּ֥אמֶר הַשְׁמֵֽד׃ אֱלֹהִים שֶׁהָיָה מֵאָז וּמִקֶּדֶם הוּא מַחֲסֶה לָהֶם מִלְמַעְלָה, וּזְרוֹעוֹתָיו מִתַּחַת שׁוֹמְרוֹת לְעוֹלָם, הוּא גֵּרֵשׁ מִפָּנֶיךָ אוֹיֵב וְאָמַר לְהַשְׁמִידוֹ.
כח וַיִּשְׁכֹּן֩ יִשְׂרָאֵ֨ל בֶּ֤טַח בָּדָד֙ עֵ֣ין יַֽעֲקֹ֔ב אֶל־אֶ֖רֶץ דָּגָ֣ן וְתִיר֑וֹשׁ אַף־שָׁמָ֖יו יַ֥עַרְפוּ טָֽל׃ יִשְׂרָאֵל שָׁכַן בְּבִטָּחוֹן לְלֹא אוֹיְבִים – כְּמוֹ שֶׁבֵּרְכָם יַעֲקֹב, הֵם יַגִּיעוּ לְאֶרֶץ מְגַדֶּלֶת דָּגָן וְיַיִן, גַּם הַשָּׁמַיִם יִנְטְפוּ עָלָיו טַל.
כט אַשְׁרֶ֨יךָ‏[1] יִשְׂרָאֵ֜ל מִ֣י כָמ֗וֹךָ עַ֚ם נוֹשַׁ֣ע בַּֽיהֹוָ֔ה מָגֵ֣ן עֶזְרֶ֔ךָ וַאֲשֶׁר־חֶ֖רֶב גַּאֲוָתֶ֑ךָ וְיִכָּחֲשׁ֤וּ אֹיְבֶ֙יךָ֙ לָ֔ךְ וְאַתָּ֖ה עַל־בָּמוֹתֵ֥ימוֹ תִדְרֹֽךְ׃ אַשְׁרֶיךָ יִשְׂרָאֵל מִי כָּמוֹךָ! עַם שֶׁה' מוֹשִׁיעוֹ, הוּא מָגֵן עָלֶיךָ וְעוֹזֵר לְךָ וּבוֹ אַתָּה מִתְגָּאֶה בְּנִצְחוֹנוֹתֶיךָ, הַגּוֹיִם יַכְחִישׁוּ שֶׁהֵם הָיוּ אוֹיְבֶיךָ כְּדֵי לְהַחְנִיף לְךָ, וְאַתָּה תִּדְרוֹךְ עַל בָּמוֹתֵיהֶם.

שביעי[עריכה]

ס פָּרָשַׁת הָעֲלִיָּה לְהַר נְבוֹ
לד א וַיַּ֨עַל מֹשֶׁ֜ה מֵעַרְבֹ֤ת מוֹאָב֙ אֶל־הַ֣ר נְב֔וֹ רֹ֚אשׁ הַפִּסְגָּ֔ה אֲשֶׁ֖ר עַל־פְּנֵ֣י יְרֵח֑וֹ וַיַּרְאֵ֨הוּ יְהֹוָ֧ה אֶת־כׇּל־הָאָ֛רֶץ אֶת־הַגִּלְעָ֖ד עַד־דָּֽן׃ מֹשֶׁה עָלָה מֵעַרְבוֹת מוֹאָב לְהַר נְבוֹ – לְרֹאשׁ הַפִּסְגָּה הַפּוֹנֶה לְמוּל יְרִיחוֹ, ה' הֶרְאָה לוֹ אֶת כָּל הָאָרֶץ – אֶת הַגִּלְעָד עַד הַבַּנְיָס.
ב וְאֵת֙ כׇּל־נַפְתָּלִ֔י וְאֶת־אֶ֥רֶץ אֶפְרַ֖יִם וּמְנַשֶּׁ֑ה וְאֵת֙ כׇּל־אֶ֣רֶץ יְהוּדָ֔ה עַ֖ד הַיָּ֥ם הָאַחֲרֽוֹן׃ וְאֶת כָּל נַפְתָּלִי וְאֶת אֶרֶץ אֶפְרַיִם וּמְנַשֶּׁה וְאֶת כָּל אֶרֶץ יְהוּדָה עַד הַיָּם הַמַּעֲרָבִי.
ג וְאֶת־הַנֶּ֗גֶב וְֽאֶת־הַכִּכָּ֞ר בִּקְעַ֧ת יְרֵח֛וֹ עִ֥יר הַתְּמָרִ֖ים עַד־צֹֽעַר׃ וְאֶת הַנֶּגֶב וְאֶת בִּקְעַת הַיַּרְדֵּן בִּקְעַת יְרִיחוֹ עִיר הַתְּמָרִים, עַד צוֹעַר.
ד וַיֹּ֨אמֶר יְהֹוָ֜ה אֵלָ֗יו זֹ֤את הָאָ֙רֶץ֙ אֲשֶׁ֣ר נִ֠שְׁבַּ֠עְתִּי לְאַבְרָהָ֨ם לְיִצְחָ֤ק וּֽלְיַעֲקֹב֙ לֵאמֹ֔ר לְזַרְעֲךָ֖ אֶתְּנֶ֑נָּה הֶרְאִיתִ֣יךָ בְעֵינֶ֔יךָ וְשָׁ֖מָּה לֹ֥א תַעֲבֹֽר׃ ה' אָמַר לוֹ: זֹאת הָאָרֶץ שֶׁנִּשְׁבַּעְתִּי לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב: "אֶתֵּן אוֹתָהּ לְצֶאֱצָאֶיךָ!", הֶרְאֵיתִי לְךָ אוֹתָהּ בְּעֵינֶיךָ, וְלֹא תַּעֲבוֹר לְשָׁם.
ה וַיָּ֨מׇת שָׁ֜ם מֹשֶׁ֧ה עֶבֶד־יְהֹוָ֛ה בְּאֶ֥רֶץ מוֹאָ֖ב עַל־פִּ֥י יְהֹוָֽה׃ מֹשֶׁה עֶבֶד ה' מֵת שָׁם בְּאֶרֶץ מוֹאָב עַל פִּי גְּזֵירַת ה'.
ו וַיִּקְבֹּ֨ר אֹת֤וֹ בַגַּי֙ בְּאֶ֣רֶץ מוֹאָ֔ב מ֖וּל בֵּ֣ית פְּע֑וֹר וְלֹא־יָדַ֥ע אִישׁ֙ אֶת־קְבֻ֣רָת֔וֹ עַ֖ד הַיּ֥וֹם הַזֶּֽה׃ ה' קָבַר אוֹתוֹ בָּעֵמֶק בְּאֶרֶץ מוֹאָב מוּל בֵּית פְּעוֹר, וְאַף אִישׁ לֹא יָדַע אֶת מְקוֹם קְבוּרָתוֹ עַד הַיּוֹם הַזֶּה.
ז וּמֹשֶׁ֗ה בֶּן־מֵאָ֧ה וְעֶשְׂרִ֛ים שָׁנָ֖ה בְּמֹת֑וֹ לֹא־כָהֲתָ֥ה עֵינ֖וֹ וְלֹא־נָ֥ס לֵחֹֽה׃ וּמֹשֶׁה הָיָה בֶּן מֵאָה וְעֶשְׂרִים שָׁנָה בְּמוֹתוֹ, וְלַמְרוֹת זֹאת עֵינָיו לֹא נֶחְלְשׁוּ וְלַחוּת נְעוּרָיו לֹא הִסְתַּלְּקָה.
ח וַיִּבְכּוּ֩ בְנֵ֨י יִשְׂרָאֵ֧ל אֶת־מֹשֶׁ֛ה בְּעַֽרְבֹ֥ת מוֹאָ֖ב שְׁלֹשִׁ֣ים י֑וֹם וַֽיִּתְּמ֔וּ יְמֵ֥י בְכִ֖י אֵ֥בֶל מֹשֶֽׁה׃ בְּנֵי יִשְׂרָאֵל בָּכוּ עַל מֹשֶׁה בְּעַרְבוֹת מוֹאָב שְׁלוֹשִׁים יוֹם, וְהִסְתַּיְּמוּ יְמֵי הַבֶּכִי שֶׁל הָאֵבֶל עַל מֹשֶׁה.
ט וִיהוֹשֻׁ֣עַ בִּן־נ֗וּן מָלֵא֙ ר֣וּחַ חׇכְמָ֔ה כִּֽי־סָמַ֥ךְ מֹשֶׁ֛ה אֶת־יָדָ֖יו עָלָ֑יו וַיִּשְׁמְע֨וּ אֵלָ֤יו בְּנֵֽי־יִשְׂרָאֵל֙ וַֽיַּעֲשׂ֔וּ כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ וִיהוֹשֻׁעַ בִּן נוּן מָלֵא רוּחַ חָכְמָה כִּי מֹשֶׁה סָמַךְ עָלָיו אֶת יָדָיו, בְּנֵי יִשְׂרָאֵל שָׁמְעוּ בְּקוֹלוֹ וְעָשׂוּ כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.
י וְלֹא־קָ֨ם נָבִ֥יא ע֛וֹד בְּיִשְׂרָאֵ֖ל כְּמֹשֶׁ֑ה אֲשֶׁר֙ יְדָע֣וֹ יְהֹוָ֔ה פָּנִ֖ים אֶל־פָּנִֽים׃ וְלֹא קָם בְּיִשְׂרָאֵל עוֹד נָבִיא כְּמוֹ מֹשֶׁה ­– שֶׁה' נִפְגַּשׁ אִתּוֹ פָּנִים אֶל פָּנִים.
יא לְכׇל־הָ֨אֹתֹ֜ת וְהַמּוֹפְתִ֗ים אֲשֶׁ֤ר שְׁלָחוֹ֙ יְהֹוָ֔ה לַעֲשׂ֖וֹת בְּאֶ֣רֶץ מִצְרָ֑יִם לְפַרְעֹ֥ה וּלְכׇל־עֲבָדָ֖יו וּלְכׇל־אַרְצֽוֹ׃ וְשֶׁעָשָׂה אֶת כָּל סִימָנֵי הַהוֹכָחָה לְכוֹחוֹ וְאֶת כָּל הַנִּסִּים שֶׁה' שָׁלַח אוֹתוֹ לַעֲשׂוֹת בְּאֶרֶץ מִצְרַיִם לְפַרְעֹה וּלְכָל עֲבָדָיו וּלְכָל אַרְצוֹ.
יב וּלְכֹל֙ הַיָּ֣ד הַחֲזָקָ֔ה וּלְכֹ֖ל הַמּוֹרָ֣א הַגָּד֑וֹל אֲשֶׁר֙ עָשָׂ֣ה מֹשֶׁ֔ה לְעֵינֵ֖י כׇּל־יִשְׂרָאֵֽל׃ וְכָל מַעֲשֵׂי הַיָּד הַחֲזָקָה וְכָל הַדְּבָרִים הַמְּעוֹרְרִים הַעֲרָצָה, כְּמוֹ שֶׁמֹּשֶׁה עָשָׂה לְעֵינֵי כָּל יִשְׂרָאֵל.
