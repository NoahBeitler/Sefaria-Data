ראשון[עריכה]

ס פָּרָשַׁת בִּרְכוֹת בִּלְעָם
כב ב וַיַּ֥רְא בָּלָ֖ק בֶּן־צִפּ֑וֹר אֵ֛ת כׇּל־אֲשֶׁר־עָשָׂ֥ה יִשְׂרָאֵ֖ל לָֽאֱמֹרִֽי׃ בָּלָק בֶּן צִפּוֹר רָאָה אֶת כָּל מַה שֶּׁעַם יִשְׂרָאֵל עָשָׂה לָאֱמוֹרִי.
ג וַיָּ֨גׇר מוֹאָ֜ב מִפְּנֵ֥י הָעָ֛ם מְאֹ֖ד כִּ֣י רַב־ה֑וּא וַיָּ֣קׇץ מוֹאָ֔ב מִפְּנֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל׃ הַמּוֹאָבִים פָּחֲדוּ מְאוֹד מֵעַם יִשְׂרָאֵל כִּי הֵם רַבִּים, וְנִמְאַס לְמוֹאָב מִבְּנֵי יִשְׂרָאֵל.
ד וַיֹּ֨אמֶר מוֹאָ֜ב אֶל־זִקְנֵ֣י מִדְיָ֗ן עַתָּ֞ה יְלַחֲכ֤וּ הַקָּהָל֙ אֶת־כׇּל־סְבִ֣יבֹתֵ֔ינוּ כִּלְחֹ֣ךְ הַשּׁ֔וֹר אֵ֖ת יֶ֣רֶק הַשָּׂדֶ֑ה וּבָלָ֧ק בֶּן־צִפּ֛וֹר מֶ֥לֶךְ לְמוֹאָ֖ב בָּעֵ֥ת הַהִֽוא׃ הַמּוֹאָבִים אָמְרוּ לְזִקְנֵי מִדְיָן: עַכְשָׁיו הַקָּהָל הַזֶּה יַשְׁמִיד אֶת כָּל הָעַמִּים שֶׁסְּבִיבֵנוּ כְּמוֹ שֶׁהַשּׁוֹר זוֹלֵל אֶת יֶרֶק הַשָּׂדֶה! וּבָלָק בֶּן צִפּוֹר הָיָה מֶלֶךְ מוֹאָב בְּאוֹתוֹ זְמָן.
ה וַיִּשְׁלַ֨ח מַלְאָכִ֜ים אֶל־בִּלְעָ֣ם בֶּן־בְּעֹ֗ר‏[1] פְּ֠ת֠וֹרָה אֲשֶׁ֧ר עַל־הַנָּהָ֛ר אֶ֥רֶץ בְּנֵי־עַמּ֖וֹ לִקְרֹא־ל֑וֹ לֵאמֹ֗ר הִ֠נֵּ֠ה עַ֣ם יָצָ֤א מִמִּצְרַ֙יִם֙ הִנֵּ֤ה כִסָּה֙ אֶת־עֵ֣ין הָאָ֔רֶץ וְה֥וּא יֹשֵׁ֖ב מִמֻּלִֽי׃ הוּא שָׁלַח שְׁלִיחִים לְבִלְעָם בֶּן בְּעוֹר אֶל הָעִיר פְּתוֹר שֶׁעַל שְׂפַת נְהַר פְּרָת, אֶל הָאָרֶץ שֶׁל בְּנֵי עַמּוֹ – לִקְרֹא לוֹ, וְלוֹמַר לוֹ: הִנֵּה עַם יָצָא מִמִּצְרַיִם, הִנֵּה הוּא כִּסָּה אֶת מַרְאֵה הָאָרֶץ וְהוּא חוֹנֶה מִמּוּלִי.
ו וְעַתָּה֩ לְכָה־נָּ֨א אָֽרָה־לִּ֜י אֶת־הָעָ֣ם הַזֶּ֗ה כִּֽי־עָצ֥וּם הוּא֙ מִמֶּ֔נִּי אוּלַ֤י אוּכַל֙ נַכֶּה־בּ֔וֹ וַאֲגָרְשֶׁ֖נּוּ מִן־הָאָ֑רֶץ כִּ֣י יָדַ֗עְתִּי אֵ֤ת אֲשֶׁר־תְּבָרֵךְ֙ מְבֹרָ֔ךְ וַאֲשֶׁ֥ר תָּאֹ֖ר יוּאָֽר׃ וְעַכְשָׁיו לֵךְ בְּבַקָּשָׁה קַלֵּל לִי אֶת הָעָם הַזֶּה כִּי הוּא עָצוּם, יוֹתֵר מִמֶּנִּי, אוּלַי אוּכַל לְהַכּוֹת בּוֹ וַאֲגָרֵשׁ אוֹתוֹ מֵהָאָרֶץ, כִּי אֲנִי יוֹדֵעַ שֶׁמִּי שֶׁתְּבָרֵךְ הוּא מְבוֹרָךְ, וּמִי שֶׁתְּקַלֵּל יְקֻלָּל.
ז וַיֵּ֨לְכ֜וּ זִקְנֵ֤י מוֹאָב֙ וְזִקְנֵ֣י מִדְיָ֔ן וּקְסָמִ֖ים בְּיָדָ֑ם וַיָּבֹ֙אוּ֙ אֶל־בִּלְעָ֔ם וַיְדַבְּר֥וּ אֵלָ֖יו דִּבְרֵ֥י בָלָֽק׃ זִקְנֵי מוֹאָב וְזִקְנֵי מִדְיָן הָלְכוּ וּבְיָדָם סִפְרֵי קְסָמִים, הֵם בָּאוּ לְבִלְעָם וְדִבְּרוּ אֵלָיו אֶת דִּבְרֵי בָּלָק.
ח וַיֹּ֣אמֶר אֲלֵיהֶ֗ם לִ֤ינוּ פֹה֙ הַלַּ֔יְלָה וַהֲשִׁבֹתִ֤י אֶתְכֶם֙ דָּבָ֔ר כַּאֲשֶׁ֛ר יְדַבֵּ֥ר יְהֹוָ֖ה אֵלָ֑י וַיֵּשְׁב֥וּ שָׂרֵֽי־מוֹאָ֖ב עִם־בִּלְעָֽם׃ הוּא אָמַר לָהֶם: הִשָּׁאֲרוּ פֹּה הַלַּיְלָה וְאָשִׁיב לָכֶם תְּשׁוּבָה לְפִי מַה שֶּׁה' יֹאמַר לִי, שָׂרֵי מוֹאָב הִתְאַכְסְנוּ אֵצֶל בִּלְעָם.
ט וַיָּבֹ֥א אֱלֹהִ֖ים אֶל־בִּלְעָ֑ם וַיֹּ֕אמֶר מִ֛י הָאֲנָשִׁ֥ים הָאֵ֖לֶּה עִמָּֽךְ׃ אֱלֹהִים בָּא לְבִלְעָם וְאָמַר: מִי הָאֲנָשִׁים הָאֵלֶּה שֶׁאִתְּךָ?
י וַיֹּ֥אמֶר בִּלְעָ֖ם אֶל־הָאֱלֹהִ֑ים בָּלָ֧ק בֶּן־צִפֹּ֛ר מֶ֥לֶךְ מוֹאָ֖ב שָׁלַ֥ח אֵלָֽי׃ בִּלְעָם אָמַר לָאֱלֹהִים: בָּלָק בֶּן צִפּוֹר מֶלֶךְ מוֹאָב שָׁלַח אוֹתָם אֵלַי לוֹמַר לִי:
יא הִנֵּ֤ה הָעָם֙ הַיֹּצֵ֣א מִמִּצְרַ֔יִם וַיְכַ֖ס אֶת־עֵ֣ין הָאָ֑רֶץ עַתָּ֗ה לְכָ֤ה קָֽבָה־לִּי֙ אֹת֔וֹ אוּלַ֥י אוּכַ֛ל לְהִלָּ֥חֶם בּ֖וֹ וְגֵרַשְׁתִּֽיו׃ הִנֵּה הָעָם שֶׁיָּצָא מִמִּצְרַיִם כִּסָּה אֶת מַרְאֵה הָאָרֶץ, עַכְשָׁיו לֵךְ תְּקַלֵּל לִי אוֹתוֹ אוּלַי אוּכַל לְהִלָּחֵם בּוֹ וּלְגָרְשׁוֹ.
יב וַיֹּ֤אמֶר אֱלֹהִים֙ אֶל־בִּלְעָ֔ם לֹ֥א תֵלֵ֖ךְ עִמָּהֶ֑ם לֹ֤א תָאֹר֙ אֶת־הָעָ֔ם כִּ֥י בָר֖וּךְ הֽוּא׃ אֱלֹהִים אָמַר לְבִלְעָם: אַל תֵּלֵךְ אִתָּם, אַל תְּקַלֵּל אֶת הָעָם כִּי הוּא מְבוֹרָךְ.

שני (חמישי במחובר)[עריכה]

יג וַיָּ֤קׇם בִּלְעָם֙ בַּבֹּ֔קֶר וַיֹּ֙אמֶר֙ אֶל־שָׂרֵ֣י בָלָ֔ק לְכ֖וּ אֶֽל־אַרְצְכֶ֑ם כִּ֚י מֵאֵ֣ן יְהֹוָ֔ה לְתִתִּ֖י לַהֲלֹ֥ךְ עִמָּכֶֽם׃ בִּלְעָם קָם בַּבּוֹקֶר וְאָמַר לְשָׂרֵי בָּלָק: לְכוּ לְאַרְצְכֶם כִּי ה' סֵרֵב לְאַפְשֵׁר לִי לָלֶכֶת אִתְּכֶם.
יד וַיָּק֙וּמוּ֙ שָׂרֵ֣י מוֹאָ֔ב וַיָּבֹ֖אוּ אֶל־בָּלָ֑ק וַיֹּ֣אמְר֔וּ מֵאֵ֥ן בִּלְעָ֖ם הֲלֹ֥ךְ עִמָּֽנוּ׃ שָׂרֵי מוֹאָב קָמוּ וּבָאוּ לְבָלָק וְאָמְרוּ: בִּלְעָם סֵרֵב לָלֶכֶת אִתָּנוּ.
טו וַיֹּ֥סֶף ע֖וֹד בָּלָ֑ק שְׁלֹ֣חַ שָׂרִ֔ים רַבִּ֥ים וְנִכְבָּדִ֖ים מֵאֵֽלֶּה׃ בָּלָק הוֹסִיף לִשְׁלוֹחַ עוֹד שָׂרִים, רַבִּים וְנִכְבָּדִים מֵאֵלֶּה שֶׁשָּׁלַח קוֹדֶם.
טז וַיָּבֹ֖אוּ אֶל־בִּלְעָ֑ם וַיֹּ֣אמְרוּ ל֗וֹ כֹּ֤ה אָמַר֙ בָּלָ֣ק בֶּן־צִפּ֔וֹר אַל־נָ֥א תִמָּנַ֖ע מֵהֲלֹ֥ךְ אֵלָֽי׃ הֵם בָּאוּ לְבִלְעָם וְאָמְרוּ לוֹ: כָּךְ אָמַר בָּלָק בֶּן צִפּוֹר: אָנָא, אַל תִּמָּנַע מִלָּלֶכֶת אֵלַי.
יז כִּֽי־כַבֵּ֤ד אֲכַבֶּדְךָ֙ מְאֹ֔ד וְכֹ֛ל אֲשֶׁר־תֹּאמַ֥ר אֵלַ֖י אֶֽעֱשֶׂ֑ה וּלְכָה־נָּא֙ קָֽבָה־לִּ֔י אֵ֖ת הָעָ֥ם הַזֶּֽה׃ כִּי אֲנִי אֲכַבֵּד אוֹתְךָ מְאוֹד וְאֶעֱשֶׂה כָּל מַה שֶּׁתֹּאמַר לִי, וְאָנָּא, לֵךְ קַלֵּל לִי אֶת הָעָם הַזֶּה.
יח וַיַּ֣עַן בִּלְעָ֗ם וַיֹּ֙אמֶר֙ אֶל־עַבְדֵ֣י בָלָ֔ק אִם־יִתֶּן־לִ֥י בָלָ֛ק מְלֹ֥א בֵית֖וֹ כֶּ֣סֶף וְזָהָ֑ב לֹ֣א אוּכַ֗ל לַעֲבֹר֙ אֶת־פִּי֙ יְהֹוָ֣ה אֱלֹהָ֔י לַעֲשׂ֥וֹת קְטַנָּ֖ה א֥וֹ גְדוֹלָֽה׃ בִּלְעָם עָנָה וְאָמַר לְעַבְדֵי בָּלָק: גַּם אִם בָּלָק יִתֵּן לִי כֶּסֶף וְזָהָב בְּכַמּוּת שֶׁיְּכוֹלָה לְמַלֵּא אֶת בֵּיתוֹ לֹא אוּכַל לַעֲבוֹר עַל דִּבְרֵי ה' אֱלֹהַי לַעֲשׂוֹת דָּבָר קָטָן אוֹ גָּדוֹל.
יט וְעַתָּ֗ה שְׁב֨וּ נָ֥א בָזֶ֛ה גַּם־אַתֶּ֖ם הַלָּ֑יְלָה וְאֵ֣דְעָ֔ה מַה־יֹּסֵ֥ף יְהֹוָ֖ה דַּבֵּ֥ר עִמִּֽי׃ וְעַכְשָׁיו בְּבַקָּשָׁה, תִּתְאַכְסְנוּ כָּאן גַּם אַתֶּם הַלַּיְלָה, עַד שֶׁאֵדַע מַה ה' יְדַבֵּר אִתִּי עוֹד.
כ וַיָּבֹ֨א אֱלֹהִ֥ים ׀ אֶל־בִּלְעָם֮ לַ֒יְלָה֒ וַיֹּ֣אמֶר ל֗וֹ אִם־לִקְרֹ֤א לְךָ֙ בָּ֣אוּ הָאֲנָשִׁ֔ים ק֖וּם לֵ֣ךְ אִתָּ֑ם וְאַ֗ךְ אֶת־הַדָּבָ֛ר אֲשֶׁר־אֲדַבֵּ֥ר אֵלֶ֖יךָ אֹת֥וֹ תַעֲשֶֽׂה׃ אֱלֹהִים בָּא לְבִלְעָם בַּלַּיְלָה, וְאָמַר לוֹ: אִם הָאֲנָשִׁים בָּאוּ רַק לִקְרֹא לְךָ קוּם לֵךְ אִתָּם, אֲבָל רַק אֶת הַדָּבָר שֶׁאֲדַבֵּר אֵלֶיךָ אוֹתוֹ תַּעֲשֶׂה.

שלישי[עריכה]

כא וַיָּ֤קׇם בִּלְעָם֙ בַּבֹּ֔קֶר וַֽיַּחֲבֹ֖שׁ אֶת־אֲתֹנ֑וֹ וַיֵּ֖לֶךְ עִם־שָׂרֵ֥י מוֹאָֽב׃ בִּלְעָם קָם בַּבּוֹקֶר, רָתַם אֶת אֲתוֹנוֹ, וְהָלַךְ עִם שָׂרֵי מוֹאָב.
כב וַיִּֽחַר־אַ֣ף אֱלֹהִים֮ כִּֽי־הוֹלֵ֣ךְ הוּא֒ וַיִּתְיַצֵּ֞ב מַלְאַ֧ךְ יְהֹוָ֛ה בַּדֶּ֖רֶךְ לְשָׂטָ֣ן ל֑וֹ וְהוּא֙ רֹכֵ֣ב עַל־אֲתֹנ֔וֹ וּשְׁנֵ֥י נְעָרָ֖יו עִמּֽוֹ׃ אֱלֹהִים כָּעַס עַל זֶה שֶׁהוּא הוֹלֵךְ, מַלְאַךְ ה' עָמַד בַּדֶּרֶךְ לְהַפְרִיעַ לוֹ, וְהוּא רוֹכֵב עַל אֲתוֹנוֹ וּשְׁנֵי נְעָרָיו אִתּוֹ.
כג וַתֵּ֣רֶא הָאָתוֹן֩ אֶת־מַלְאַ֨ךְ יְהֹוָ֜ה נִצָּ֣ב בַּדֶּ֗רֶךְ וְחַרְבּ֤וֹ שְׁלוּפָה֙ בְּיָד֔וֹ וַתֵּ֤ט הָֽאָתוֹן֙ מִן־הַדֶּ֔רֶךְ וַתֵּ֖לֶךְ בַּשָּׂדֶ֑ה וַיַּ֤ךְ בִּלְעָם֙ אֶת־הָ֣אָת֔וֹן לְהַטֹּתָ֖הּ הַדָּֽרֶךְ׃ הָאָתוֹן רָאֲתָה אֶת מַלְאַךְ ה' עוֹמֵד בַּדֶּרֶךְ וְחַרְבּוֹ שְׁלוּפָה בְּיָדוֹ, הָאָתוֹן סָטְתָה מֵהַדֶּרֶךְ וְהָלְכָה בַּשָּׂדֶה, בִּלְעָם הִכָּה אֶת הָאָתוֹן כְּדֵי לְהַפְנוֹת אוֹתָהּ לַדֶּרֶךְ.
כד וַֽיַּעֲמֹד֙ מַלְאַ֣ךְ יְהֹוָ֔ה בְּמִשְׁע֖וֹל הַכְּרָמִ֑ים גָּדֵ֥ר מִזֶּ֖ה וְגָדֵ֥ר מִזֶּֽה׃ מַלְאַךְ ה' עָמַד בַּשְּׁבִיל שֶׁבֵּין הַכְּרָמִים, גָּדֵר מִכָּאן וְגָדֵר מִכָּאן.
כה וַתֵּ֨רֶא הָאָת֜וֹן אֶת־מַלְאַ֣ךְ יְהֹוָ֗ה וַתִּלָּחֵץ֙ אֶל־הַקִּ֔יר וַתִּלְחַ֛ץ אֶת־רֶ֥גֶל בִּלְעָ֖ם אֶל־הַקִּ֑יר וַיֹּ֖סֶף לְהַכֹּתָֽהּ׃ הָאָתוֹן רָאֲתָה אֶת מַלְאַךְ ה', הִיא נִלְחֲצָה לַקִּיר וְלָחֲצָה אֶת רֶגֶל בִּלְעָם, הוּא הִכָּה אוֹתָהּ שׁוּב.
כו וַיּ֥וֹסֶף מַלְאַךְ־יְהֹוָ֖ה עֲב֑וֹר וַֽיַּעֲמֹד֙ בְּמָק֣וֹם צָ֔ר אֲשֶׁ֛ר אֵֽין־דֶּ֥רֶךְ לִנְט֖וֹת יָמִ֥ין וּשְׂמֹֽאול׃ מַלְאַךְ ה' הִמְשִׁיךְ לַעֲבוֹר, וְעָמַד בְּמָקוֹם צָר שֶׁאֵין בּוֹ דֶּרֶךְ לִסְטוֹת לְיָמִין אוֹ לִשְׂמֹאל.
כז וַתֵּ֤רֶא הָֽאָתוֹן֙ אֶת־מַלְאַ֣ךְ יְהֹוָ֔ה וַתִּרְבַּ֖ץ תַּ֣חַת בִּלְעָ֑ם וַיִּֽחַר־אַ֣ף בִּלְעָ֔ם וַיַּ֥ךְ אֶת־הָאָת֖וֹן בַּמַּקֵּֽל׃ הָאָתוֹן רָאֲתָה אֶת מַלְאַךְ ה', וְרָבְצָה תַּחַת בִּלְעָם, בִּלְעָם כָּעַס וְהִכָּה אֶת הָאָתוֹן בַּמַּקֵּל.
כח וַיִּפְתַּ֥ח יְהֹוָ֖ה אֶת־פִּ֣י הָאָת֑וֹן וַתֹּ֤אמֶר לְבִלְעָם֙ מֶה־עָשִׂ֣יתִֽי לְךָ֔ כִּ֣י הִכִּיתַ֔נִי זֶ֖ה שָׁלֹ֥שׁ רְגָלִֽים׃ ה' פָּתַח אֶת פִּי הָאָתוֹן וְהִיא אָמְרָה לְבִלְעָם: מֶה עָשִׂיתִי לְךָ שֶׁהִכִּיתָ אוֹתִי שָׁלוֹשׁ פְּעָמִים?
כט וַיֹּ֤אמֶר בִּלְעָם֙ לָֽאָת֔וֹן כִּ֥י הִתְעַלַּ֖לְתְּ בִּ֑י ל֤וּ יֶשׁ־חֶ֙רֶב֙ בְּיָדִ֔י כִּ֥י עַתָּ֖ה הֲרַגְתִּֽיךְ׃ בִּלְעָם אָמַר לָאָתוֹן: כִּי הִתְעַלַּלְתְּ בִּי, אִלּוּ הָיְתָה חֶרֶב בְּיָדִי הָיִיתִי עַכְשָׁיו הוֹרֵג אוֹתָךְ!
ל וַתֹּ֨אמֶר הָאָת֜וֹן אֶל־בִּלְעָ֗ם הֲלוֹא֩ אָנֹכִ֨י אֲתֹֽנְךָ֜ אֲשֶׁר־רָכַ֣בְתָּ עָלַ֗י מֵעֽוֹדְךָ֙ עַד־הַיּ֣וֹם הַזֶּ֔ה הַֽהַסְכֵּ֣ן הִסְכַּ֔נְתִּי לַעֲשׂ֥וֹת לְךָ֖ כֹּ֑ה וַיֹּ֖אמֶר לֹֽא׃ הָאָתוֹן אָמְרָה לְבִלְעָם: הֲרֵי אֲנִי אֲתוֹנְךָ שֶׁרָכַבְתָּ עָלַי מֵאָז שֶׁהִתְחַלְתָּ לִרְכּוֹב וְעַד הַיּוֹם הַזֶּה, הַאִם הָיִיתִי רְגִילָה לַעֲשׂוֹת לְךָ כָּךְ? אָמַר: לֹא.
לא וַיְגַ֣ל יְהֹוָה֮ אֶת־עֵינֵ֣י בִלְעָם֒ וַיַּ֞רְא אֶת־מַלְאַ֤ךְ יְהֹוָה֙ נִצָּ֣ב בַּדֶּ֔רֶךְ וְחַרְבּ֥וֹ שְׁלֻפָ֖ה בְּיָד֑וֹ וַיִּקֹּ֥ד וַיִּשְׁתַּ֖חוּ לְאַפָּֽיו׃ ה' פָּקַח אֶת עֵינֵי בִּלְעָם וְהוּא רָאָה אֶת מַלְאַךְ ה' עוֹמֵד בַּדֶּרֶךְ וְחַרְבּוֹ שְׁלוּפָה בְּיָדוֹ, הוּא קָד וְהִשְׁתַּחֲוָה עַל פָּנָיו.
לב וַיֹּ֤אמֶר אֵלָיו֙ מַלְאַ֣ךְ יְהֹוָ֔ה עַל־מָ֗ה הִכִּ֙יתָ֙ אֶת־אֲתֹ֣נְךָ֔ זֶ֖ה שָׁל֣וֹשׁ רְגָלִ֑ים הִנֵּ֤ה אָנֹכִי֙ יָצָ֣אתִי לְשָׂטָ֔ן כִּֽי־יָרַ֥ט הַדֶּ֖רֶךְ לְנֶגְדִּֽי׃ מַלְאַךְ ה' אָמַר לוֹ: לַמָּה הִכִּיתָ אֶת אֲתוֹנְךָ כְּבָר שָׁלוֹשׁ פְּעָמִים? הֲרֵי אֲנִי הוּא שֶׁיָּצָאתִי לְהַפְרִיעַ, כִּי דַּרְכְּךָ אֵינָהּ יְשָׁרָה בְּעֵינַי!
לג וַתִּרְאַ֙נִי֙ הָֽאָת֔וֹן וַתֵּ֣ט לְפָנַ֔י זֶ֖ה שָׁלֹ֣שׁ רְגָלִ֑ים אוּלַי֙ נָטְתָ֣ה מִפָּנַ֔י כִּ֥י עַתָּ֛ה גַּם־אֹתְכָ֥ה הָרַ֖גְתִּי וְאוֹתָ֥הּ הֶחֱיֵֽיתִי׃ הָאָתוֹן רָאֲתָה אוֹתִי וְסָטְתָה מִלְּפָנַי כְּבָר שָׁלוֹשׁ פְּעָמִים, אִם לֹא הָיְתָה סוֹטָה מִפָּנַי הָיִיתִי עַכְשָׁיו הוֹרֵג אוֹתְךָ, וְאוֹתָהּ הָיִיתִי מַשְׁאִיר בַּחַיִּים.
לד וַיֹּ֨אמֶר בִּלְעָ֜ם אֶל־מַלְאַ֤ךְ יְהֹוָה֙ חָטָ֔אתִי כִּ֚י לֹ֣א יָדַ֔עְתִּי כִּ֥י אַתָּ֛ה נִצָּ֥ב לִקְרָאתִ֖י בַּדָּ֑רֶךְ וְעַתָּ֛ה אִם־רַ֥ע בְּעֵינֶ֖יךָ אָשׁ֥וּבָה לִּֽי׃ בִּלְעָם אָמַר לְמַלְאַךְ ה': חָטָאתִי כִּי לֹא יָדַעְתִּי שֶׁאַתָּה עוֹמֵד מוּלִי בַּדֶּרֶךְ, וְעַכְשָׁיו, אִם זֶה רַע בְּעֵינֶיךָ שֶׁהָלַכְתִּי, אֶחֱזוֹר.
לה וַיֹּ֩אמֶר֩ מַלְאַ֨ךְ יְהֹוָ֜ה אֶל־בִּלְעָ֗ם לֵ֚ךְ עִם־הָ֣אֲנָשִׁ֔ים וְאֶ֗פֶס אֶת־הַדָּבָ֛ר אֲשֶׁר־אֲדַבֵּ֥ר אֵלֶ֖יךָ אֹת֣וֹ תְדַבֵּ֑ר וַיֵּ֥לֶךְ בִּלְעָ֖ם עִם־שָׂרֵ֥י בָלָֽק׃ מַלְאַךְ ה' אָמַר לְבִלְעָם: לֵךְ עִם הָאֲנָשִׁים, אֲבָל רַק אֶת הַדָּבָר שֶׁאֲדַבֵּר אֵלֶיךָ אוֹתוֹ תְּדַבֵּר! בִּלְעָם הָלַךְ עִם שָׂרֵי בָּלָק.
לו וַיִּשְׁמַ֥ע בָּלָ֖ק כִּ֣י בָ֣א בִלְעָ֑ם וַיֵּצֵ֨א לִקְרָאת֜וֹ אֶל־עִ֣יר מוֹאָ֗ב אֲשֶׁר֙ עַל־גְּב֣וּל אַרְנֹ֔ן אֲשֶׁ֖ר בִּקְצֵ֥ה הַגְּבֽוּל׃ בָּלָק שָׁמַע שֶׁבִּלְעָם בָּא, הוּא יָצָא לִקְרָאתוֹ לְעִיר הַבִּירָה שֶׁל מוֹאָב שֶׁנִּמְצֵאת עַל גְּבוּל נַחַל אַרְנוֹן שֶׁנִּמְצָא בִּקְצֵה גְּבוּל מוֹאָב.
לז וַיֹּ֨אמֶר בָּלָ֜ק אֶל־בִּלְעָ֗ם הֲלֹא֩ שָׁלֹ֨חַ שָׁלַ֤חְתִּי אֵלֶ֙יךָ֙ לִקְרֹא־לָ֔ךְ לָ֥מָּה לֹא־הָלַ֖כְתָּ אֵלָ֑י הַֽאֻמְנָ֔ם לֹ֥א אוּכַ֖ל כַּבְּדֶֽךָ׃ בָּלָק אָמַר לְבִלְעָם: הֲרֵי כְּבָר שָׁלַחְתִּי אֵלֶיךָ שְׁלִיחִים לִקְרֹא לְךָ, לָמָּה לֹא הָלַכְתָּ אֵלַי? הַאִם בֶּאֱמֶת חָשַׁבְתָּ שֶׁלֹּא אוּכַל לְכַבֵּד אוֹתְךָ?
לח וַיֹּ֨אמֶר בִּלְעָ֜ם אֶל־בָּלָ֗ק הִֽנֵּה־בָ֙אתִי֙ אֵלֶ֔יךָ עַתָּ֕ה הֲיָכֹ֥ל אוּכַ֖ל דַּבֵּ֣ר מְא֑וּמָה הַדָּבָ֗ר אֲשֶׁ֨ר יָשִׂ֧ים אֱלֹהִ֛ים בְּפִ֖י אֹת֥וֹ אֲדַבֵּֽר׃ בִּלְעָם אָמַר לְבָלָק: הִנֵּה בָּאתִי אֵלֶיךָ עַכְשָׁיו, הַאִם אֲנִי יָכוֹל לְדַבֵּר מַשֶּׁהוּ?! רַק אֶת הַדְּבָרִים שֶׁאֱלֹהִים יָשִׂים בְּפִי, אוֹתָם אֲדַבֵּר.

רביעי (שישי במחובר)[עריכה]

לט וַיֵּ֥לֶךְ בִּלְעָ֖ם עִם־בָּלָ֑ק וַיָּבֹ֖אוּ קִרְיַ֥ת חֻצֽוֹת׃ בִּלְעָם הָלַךְ עִם בָּלָק, וְהֵם בָּאוּ לָעִיר – לְחוּצוֹת עִיר הַבִּירָה.
מ וַיִּזְבַּ֥ח בָּלָ֖ק בָּקָ֣ר וָצֹ֑אן וַיְשַׁלַּ֣ח לְבִלְעָ֔ם וְלַשָּׂרִ֖ים אֲשֶׁ֥ר אִתּֽוֹ׃ בָּלָק שָׁחַט בָּקָר וָצֹאן, וְשָׁלַח מֵהַבָּשָׂר לְבִלְעָם וְלַשָּׂרִים שֶׁאִתּוֹ.
מא וַיְהִ֣י בַבֹּ֔קֶר וַיִּקַּ֤ח בָּלָק֙ אֶת־בִּלְעָ֔ם וַֽיַּעֲלֵ֖הוּ בָּמ֣וֹת בָּ֑עַל וַיַּ֥רְא מִשָּׁ֖ם קְצֵ֥ה הָעָֽם׃ בַּבּוֹקֶר, בָּלָק לָקַח אֶת בִּלְעָם וְהֶעֱלָה אוֹתוֹ לָרָמָה שֶׁעָבְדוּ עָלֶיהָ אֶת הַבַּעַל, וְהוּא רָאָה מִשָּׁם אֶת קְצֵה הָעָם.
כג א וַיֹּ֤אמֶר בִּלְעָם֙ אֶל־בָּלָ֔ק בְּנֵה־לִ֥י בָזֶ֖ה שִׁבְעָ֣ה מִזְבְּחֹ֑ת וְהָכֵ֥ן לִי֙ בָּזֶ֔ה שִׁבְעָ֥ה פָרִ֖ים וְשִׁבְעָ֥ה אֵילִֽים׃ בִּלְעָם אָמַר לְבָלָק: בְּנֵה לִי כָּאן שִׁבְעָה מִזְבְּחוֹת וְהָכֵן לִי כָּאן שִׁבְעָה פָּרִים וְשִׁבְעָה אֵילִים.
ב וַיַּ֣עַשׂ בָּלָ֔ק כַּאֲשֶׁ֖ר דִּבֶּ֣ר בִּלְעָ֑ם וַיַּ֨עַל בָּלָ֧ק וּבִלְעָ֛ם פָּ֥ר וָאַ֖יִל בַּמִּזְבֵּֽחַ׃ בָּלָק עָשָׂה כְּמוֹ שֶׁבִּלְעָם דִּבֵּר, וּבָלָק וּבִלְעָם הִקְרִיבוּ בְּכָל מִזְבֵּחַ פָּר וְאַיִל לְעוֹלָה.
ג וַיֹּ֨אמֶר בִּלְעָ֜ם לְבָלָ֗ק הִתְיַצֵּב֮ עַל־עֹלָתֶ֒ךָ֒ וְאֵֽלְכָ֗ה אוּלַ֞י יִקָּרֵ֤ה יְהֹוָה֙ לִקְרָאתִ֔י וּדְבַ֥ר מַה־יַּרְאֵ֖נִי וְהִגַּ֣דְתִּי לָ֑ךְ וַיֵּ֖לֶךְ שֶֽׁפִי׃ בִּלְעָם אָמַר לְבָלָק: עֲמוֹד לְיַד עוֹלָתְךָ וְאֵלֵךְ, אוּלַי בְּמִקְרֶה ה' יָבוֹא לִקְרָאתִי וְיַרְאֶה לִי דָּבָר בִּנְבוּאָה וְאַגִּיד לְךָ, וְהוּא הָלַךְ לְבָד.
ד וַיִּקָּ֥ר אֱלֹהִ֖ים אֶל־בִּלְעָ֑ם וַיֹּ֣אמֶר אֵלָ֗יו אֶת־שִׁבְעַ֤ת הַֽמִּזְבְּחֹת֙ עָרַ֔כְתִּי וָאַ֛עַל פָּ֥ר וָאַ֖יִל בַּמִּזְבֵּֽחַ׃ אֱלֹהִים הִזְדַּמֵּן לְבִלְעָם, אָמַר לוֹ: עָרַכְתִּי אֶת שִׁבְעַת הַמִּזְבְּחוֹת וְהִקְרַבְתִּי בְּכָל מִזְבֵּחַ פָּר וְאַיִל לְעוֹלָה.
ה וַיָּ֧שֶׂם יְהֹוָ֛ה דָּבָ֖ר בְּפִ֣י בִלְעָ֑ם וַיֹּ֛אמֶר שׁ֥וּב אֶל־בָּלָ֖ק וְכֹ֥ה תְדַבֵּֽר׃ ה' מָסַר דְּבַר נְבוּאָה לְבִלְעָם, וְאָמַר: שׁוּב לְבָלָק וְכָךְ תְּדַבֵּר.
ו וַיָּ֣שׇׁב אֵלָ֔יו וְהִנֵּ֥ה נִצָּ֖ב עַל־עֹלָת֑וֹ ה֖וּא וְכׇל־שָׂרֵ֥י מוֹאָֽב׃ הוּא חָזַר אֵלָיו וְהִנֵּה הוּא עוֹמֵד לְיַד עוֹלָתוֹ, וְכָל שָׂרֵי מוֹאָב אִתּוֹ.
ז וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר מִן־אֲ֠רָ֠ם יַנְחֵ֨נִי בָלָ֤ק מֶֽלֶךְ־מוֹאָב֙ מֵֽהַרְרֵי־קֶ֔דֶם לְכָה֙ אָֽרָה־לִּ֣י יַעֲקֹ֔ב וּלְכָ֖ה זֹעֲמָ֥ה יִשְׂרָאֵֽל׃ הוּא נָשָׂא אֶת נְאוּמוֹ וְאָמַר: בָּלָק מֶלֶךְ מוֹאָב הוֹבִיל אוֹתִי מֵאֲרָם – מֵהָרֵי הַמִּזְרָח, וּבִקֵּשׁ מִמֶּנִּי: לֵךְ קַלֵּל לִי אֶת יַעֲקֹב וְלֵךְ הֲבֵא זַעַם עַל יִשְׂרָאֵל.
ח מָ֣ה אֶקֹּ֔ב לֹ֥א קַבֹּ֖ה אֵ֑ל וּמָ֣ה אֶזְעֹ֔ם לֹ֥א זָעַ֖ם יְהֹוָֽה׃ אֵיךְ אֲקַלֵּל כְּשֶׁאֵין קִלְלַת אֵל, וְאֵיךְ אָבִיא זַעַם וַה' לֹא זָעַם?!
ט כִּֽי־מֵרֹ֤אשׁ צֻרִים֙ אֶרְאֶ֔נּוּ וּמִגְּבָע֖וֹת אֲשׁוּרֶ֑נּוּ הֶן־עָם֙ לְבָדָ֣ד יִשְׁכֹּ֔ן וּבַגּוֹיִ֖ם לֹ֥א יִתְחַשָּֽׁב׃ הֲרֵי אֲנִי רוֹאֶה אוֹתוֹ מֵרֹאשׁ הַצּוּקִים וּמַבִּיט בּוֹ מִגְּבָעוֹת – הִנֵּה עַם מְיוּחָד שֶׁיִּתְיַשֵּׁב בְּאַרְצוֹ, וְאֵינוֹ נֶחְשַׁב כְּאֶחָד מֵהָעַמִּים.
י מִ֤י מָנָה֙ עֲפַ֣ר יַעֲקֹ֔ב וּמִסְפָּ֖ר אֶת־רֹ֣בַע יִשְׂרָאֵ֑ל תָּמֹ֤ת נַפְשִׁי֙ מ֣וֹת יְשָׁרִ֔ים וּתְהִ֥י אַחֲרִיתִ֖י כָּמֹֽהוּ׃ מִי יָכוֹל לִמְנוֹת אֶת צֶאֱצָאֵי יַעֲקֹב הַמְּרֻבִּים כֶּעָפָר, וּמִי יָכוֹל לִסְפּוֹר אֲפִלּוּ רֶבַע אֶחָד שֶׁל מַחֲנֵה יִשְׂרָאֵל, הַלְוַאי שֶׁאָמוּת כְּצַדִּיק וְיִהְיֶה סוֹפִי כָּמוֹהוּ.
יא וַיֹּ֤אמֶר בָּלָק֙ אֶל־בִּלְעָ֔ם מֶ֥ה עָשִׂ֖יתָ לִ֑י לָקֹ֤ב אֹיְבַי֙ לְקַחְתִּ֔יךָ וְהִנֵּ֖ה בֵּרַ֥כְתָּ בָרֵֽךְ׃ בָּלָק אָמַר לְבִלְעָם: מֶה עָשִׂיתָ לִי?! לָקַחְתִּי אוֹתְךָ לְקַלֵּל אֶת אוֹיְבַי וְהִנֵּה בֵּרַכְתָּ אוֹתָם!
יב וַיַּ֖עַן וַיֹּאמַ֑ר הֲלֹ֗א אֵת֩ אֲשֶׁ֨ר יָשִׂ֤ים יְהֹוָה֙ בְּפִ֔י אֹת֥וֹ אֶשְׁמֹ֖ר לְדַבֵּֽר׃ עָנָה לוֹ וְאָמַר: הֲרֵי אֶת הַנְּבוּאָה שֶׁה' יָשִׂים בְּפִי, רַק אוֹתָהּ אַקְפִּיד לוֹמַר!

חמישי[עריכה]

יג וַיֹּ֨אמֶר אֵלָ֜יו בָּלָ֗ק לך לְכָה־נָּ֨א אִתִּ֜י אֶל־מָק֤וֹם אַחֵר֙ אֲשֶׁ֣ר תִּרְאֶ֣נּוּ מִשָּׁ֔ם אֶ֚פֶס קָצֵ֣הוּ תִרְאֶ֔ה וְכֻלּ֖וֹ לֹ֣א תִרְאֶ֑ה וְקׇבְנוֹ־לִ֖י מִשָּֽׁם׃ בָּלָק אָמַר לוֹ: לֵךְ בְּבַקָּשָׁה אִתִּי לְמָקוֹם אַחֵר שֶׁתִּרְאֶה אוֹתוֹ מִשָּׁם, תִּרְאֶה רַק אֶת קָצֵהוּ וְלֹא תִּרְאֶה אֶת כֻּלּוֹ, וְקַלֵּל לִי אוֹתוֹ מִשָּׁם.
יד וַיִּקָּחֵ֙הוּ֙ שְׂדֵ֣ה צֹפִ֔ים אֶל־רֹ֖אשׁ הַפִּסְגָּ֑ה וַיִּ֙בֶן֙ שִׁבְעָ֣ה מִזְבְּחֹ֔ת וַיַּ֛עַל פָּ֥ר וָאַ֖יִל בַּמִּזְבֵּֽחַ׃ הוּא לָקַח אוֹתוֹ לִשְׂדֵה הַתַּצְפִּית לְרֹאשׁ הַפִּסְגָּה, הוּא בָּנָה שִׁבְעָה מִזְבְּחוֹת וְהִקְרִיב בְּכָל מִזְבֵּחַ פָּר וְאַיִל לְעוֹלָה.
טו וַיֹּ֙אמֶר֙ אֶל־בָּלָ֔ק הִתְיַצֵּ֥ב כֹּ֖ה עַל־עֹלָתֶ֑ךָ וְאָנֹכִ֖י אִקָּ֥רֶה כֹּֽה׃ אָמַר לְבָלָק: עֲמוֹד כָּאן לְיַד עוֹלָתְךָ, וַאֲנִי, תִּזְדַּמֵּן לִי נְבוּאָה כָּאן.
טז וַיִּקָּ֤ר יְהֹוָה֙ אֶל־בִּלְעָ֔ם וַיָּ֥שֶׂם דָּבָ֖ר בְּפִ֑יו וַיֹּ֛אמֶר שׁ֥וּב אֶל־בָּלָ֖ק וְכֹ֥ה תְדַבֵּֽר׃ ה' הִזְדַּמֵּן לְבִלְעָם וּמָסַר לוֹ דְּבַר נְבוּאָה, וְאָמַר לוֹ: חֲזוֹר לְבָלָק וְכָךְ תְּדַבֵּר.
יז וַיָּבֹ֣א אֵלָ֗יו וְהִנּ֤וֹ נִצָּב֙ עַל־עֹ֣לָת֔וֹ וְשָׂרֵ֥י מוֹאָ֖ב אִתּ֑וֹ וַיֹּ֤אמֶר לוֹ֙ בָּלָ֔ק מַה־דִּבֶּ֖ר יְהֹוָֽה׃ הוּא בָּא אֵלָיו וְהִנֵּה הוּא עוֹמֵד לְיַד עוֹלָתוֹ וְשָׂרֵי מוֹאָב אִתּוֹ, אָמַר לוֹ בָּלָק: מַה ה' דִּבֵּר?
יח וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר ק֤וּם בָּלָק֙ וּֽשְׁמָ֔ע הַאֲזִ֥ינָה עָדַ֖י בְּנ֥וֹ צִפֹּֽר׃ הוּא נָשָׂא אֶת נְאוּמוֹ וְאָמַר: קוּם בָּלָק וּשְׁמָע, הַקְשֵׁב אֵלַי בְּנוֹ שֶׁל צִפּוֹר.
יט לֹ֣א אִ֥ישׁ אֵל֙ וִֽיכַזֵּ֔ב וּבֶן־אָדָ֖ם וְיִתְנֶחָ֑ם הַה֤וּא אָמַר֙ וְלֹ֣א יַעֲשֶׂ֔ה וְדִבֶּ֖ר וְלֹ֥א יְקִימֶֽנָּה׃ הָאֵל אֵינוֹ אִישׁ שֶׁמְּשַׁקֵּר, הוּא אֵינוֹ בֶּן אָדָם שֶׁמִּתְחָרֵט, הוּא אָמַר – וְכִי לֹא יַעֲשֶׂה?! הוּא דִּבֵּר – וְכִי לֹא יְקַיֵּם?!
כ הִנֵּ֥ה בָרֵ֖ךְ לָקָ֑חְתִּי וּבֵרֵ֖ךְ וְלֹ֥א אֲשִׁיבֶֽנָּה׃ הִנֵּה הֵבֵאתִי בְּרָכָה, וְלֹא אֶחֱזוֹר בִּי מִלְּבָרֵךְ.
כא לֹֽא־הִבִּ֥יט אָ֙וֶן֙ בְּיַעֲקֹ֔ב וְלֹא־רָאָ֥ה עָמָ֖ל בְּיִשְׂרָאֵ֑ל יְהֹוָ֤ה אֱלֹהָיו֙ עִמּ֔וֹ וּתְרוּעַ֥ת מֶ֖לֶךְ בּֽוֹ׃ לֹא נִרְאָה בְּיַעֲקֹב דָּבָר שֶׁאֵינוֹ הָגוּן, וְלֹא נִרְאָה בְּיִשְׂרָאֵל דְּבַר שֶׁקֶר, ה' אֱלֹהָיו אִתּוֹ וּתְרוּעַת הַמֶּלֶךְ נִשְׁמַעַת בּוֹ בְּהִתְגַּלּוֹתוֹ.
כב אֵ֖ל מוֹצִיאָ֣ם מִמִּצְרָ֑יִם כְּתוֹעֲפֹ֥ת רְאֵ֖ם לֽוֹ׃ הָאֵל שֶׁהוֹצִיא אוֹתוֹ מִמִּצְרַיִם נוֹתֵן לוֹ כֹּחַ כְּמוֹ שֶׁל שׁוֹר הַבָּר הַגָּדוֹל.
כג כִּ֤י לֹא־נַ֙חַשׁ֙ בְּיַעֲקֹ֔ב וְלֹא־קֶ֖סֶם בְּיִשְׂרָאֵ֑ל כָּעֵ֗ת יֵאָמֵ֤ר לְיַעֲקֹב֙ וּלְיִשְׂרָאֵ֔ל מַה־פָּ֖עַל אֵֽל׃ הֲרֵי נִיחוּשׁ עֲתִידוֹת אֵינוֹ מַזִּיק לְיַעֲקֹב, וְחִזּוּי בְּקֶסֶם לֹא מַשְׁפִּיעַ עַל יִשְׂרָאֵל, עַכְשָׁיו אוֹמַר עַל יַעֲקֹב וְעַל יִשְׂרָאֵל אֶת מַה שֶּׁהָאֵל יַעֲשֶׂה לוֹ:
כד הֶן־עָם֙ כְּלָבִ֣יא יָק֔וּם וְכַאֲרִ֖י יִתְנַשָּׂ֑א לֹ֤א יִשְׁכַּב֙ עַד־יֹ֣אכַל טֶ֔רֶף וְדַם־חֲלָלִ֖ים יִשְׁתֶּֽה׃ הִנֵּה עַם זֶה יָקוּם וְיִתְרוֹמֵם כְּמוֹ אַרְיֵה, הוּא לֹא יִשְׁכַּב לָנוּחַ עַד שֶׁיֹּאכַל טֶרֶף וְיִשְׁתֶּה אֶת דַּם הַהֲרוּגִים.
כה וַיֹּ֤אמֶר בָּלָק֙ אֶל־בִּלְעָ֔ם גַּם־קֹ֖ב לֹ֣א תִקֳּבֶ֑נּוּ גַּם־בָּרֵ֖ךְ לֹ֥א תְבָרְכֶֽנּוּ׃ בָּלָק אָמַר לְבִלְעָם: גַּם אִם לֹא תְּקַלֵּל אוֹתוֹ לְפָחוֹת אַל תְּבָרֵךְ אוֹתוֹ!
כו וַיַּ֣עַן בִּלְעָ֔ם וַיֹּ֖אמֶר אֶל־בָּלָ֑ק הֲלֹ֗א דִּבַּ֤רְתִּי אֵלֶ֙יךָ֙ לֵאמֹ֔ר כֹּ֛ל אֲשֶׁר־יְדַבֵּ֥ר יְהֹוָ֖ה אֹת֥וֹ אֶֽעֱשֶֽׂה׃ עָנָה בִּלְעָם וְאָמַר לְבָלָק: הֲרֵי דִּבַּרְתִּי אֵלֶיךָ וְאָמַרְתִּי: כָּל מַה שֶּׁה' יְדַבֵּר – אוֹתוֹ אֶעֱשֶׂה.

שישי (שביעי במחובר)[עריכה]

כז וַיֹּ֤אמֶר בָּלָק֙ אֶל־בִּלְעָ֔ם לְכָה־נָּא֙ אֶקָּ֣חֲךָ֔ אֶל־מָק֖וֹם אַחֵ֑ר אוּלַ֤י יִישַׁר֙ בְּעֵינֵ֣י הָאֱלֹהִ֔ים וְקַבֹּ֥תוֹ לִ֖י מִשָּֽׁם׃ בָּלָק אָמַר לְבִלְעָם: לֵךְ בְּבַקָּשָׁה אֶקַּח אוֹתְךָ לְמָקוֹם אַחֵר, אוּלַי יִיטַב בְּעֵינֵי הָאֱלֹהִים וּתְקַלֵּל לִי אוֹתוֹ מִשָּׁם.
כח וַיִּקַּ֥ח בָּלָ֖ק אֶת־בִּלְעָ֑ם רֹ֣אשׁ הַפְּע֔וֹר הַנִּשְׁקָ֖ף עַל־פְּנֵ֥י הַיְשִׁימֹֽן׃ בָּלָק לָקַח אֶת בִּלְעָם לְרֹאשׁ הַגִּבְעָה שֶׁעָבְדוּ עָלֶיהָ לָאֱלִיל פְּעוֹר שֶׁמַּשְׁקִיפִים מִשָּׁם לַמִּדְבָּר.
כט וַיֹּ֤אמֶר בִּלְעָם֙ אֶל־בָּלָ֔ק בְּנֵה־לִ֥י בָזֶ֖ה שִׁבְעָ֣ה מִזְבְּחֹ֑ת וְהָכֵ֥ן לִי֙ בָּזֶ֔ה שִׁבְעָ֥ה פָרִ֖ים וְשִׁבְעָ֥ה אֵילִֽם׃ בִּלְעָם אָמַר לְבָלָק: בְּנֵה לִי כָּאן שִׁבְעָה מִזְבְּחוֹת וְהָכֵן לִי כָּאן שִׁבְעָה פָּרִים וְשִׁבְעָה אֵילִים.
ל וַיַּ֣עַשׂ בָּלָ֔ק כַּאֲשֶׁ֖ר אָמַ֣ר בִּלְעָ֑ם וַיַּ֛עַל פָּ֥ר וָאַ֖יִל בַּמִּזְבֵּֽחַ׃ בָּלָק עָשָׂה כְּמוֹ שֶׁבִּלְעָם אָמַר וְהִקְרִיב בְּכָל מִזְבֵּחַ פָּר וְאַיִל לְעוֹלָה.
כד א וַיַּ֣רְא בִּלְעָ֗ם כִּ֣י ט֞וֹב בְּעֵינֵ֤י יְהֹוָה֙ לְבָרֵ֣ךְ אֶת־יִשְׂרָאֵ֔ל וְלֹא־הָלַ֥ךְ כְּפַֽעַם־בְּפַ֖עַם לִקְרַ֣את נְחָשִׁ֑ים וַיָּ֥שֶׁת אֶל־הַמִּדְבָּ֖ר פָּנָֽיו׃ בִּלְעָם רָאָה שֶׁטּוֹב בְּעֵינֵי ה' לְבָרֵךְ אֶת יִשְׂרָאֵל, לָכֵן הוּא לֹא הִשְׁתַּמֵּשׁ בְּאֶמְצָעִים לְנִיחוּשׁ עֲתִידוֹת כְּמוֹ בַּפְּעָמִים הַקּוֹדְמוֹת, אֶלָּא הִפְנָה אֶת פָּנָיו לַמִּדְבָּר.
ב וַיִּשָּׂ֨א בִלְעָ֜ם אֶת־עֵינָ֗יו וַיַּרְא֙ אֶת־יִשְׂרָאֵ֔ל שֹׁכֵ֖ן לִשְׁבָטָ֑יו וַתְּהִ֥י עָלָ֖יו ר֥וּחַ אֱלֹהִֽים׃ בִּלְעָם הֵרִים אֶת מַבָּטוֹ וְרָאָה אֶת יִשְׂרָאֵל חוֹנֶה מְסֻדָּר לְפִי שְׁבָטָיו, וְשָׁרְתָה עָלָיו רוּחַ אֱלֹהִים.
ג וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר נְאֻ֤ם בִּלְעָם֙ בְּנ֣וֹ בְעֹ֔ר וּנְאֻ֥ם הַגֶּ֖בֶר שְׁתֻ֥ם הָעָֽיִן׃ הוּא נָשָׂא אֶת נְאוּמוֹ וְאָמַר: נְאוּם בִּלְעָם בְּנוֹ שֶׁל בְּעוֹר, נְאוּם הַגֶּבֶר בַּעַל הָעַיִן הַמִּתְבּוֹנֶנֶת הֵיטֵב.
ד נְאֻ֕ם שֹׁמֵ֖עַ אִמְרֵי־אֵ֑ל אֲשֶׁ֨ר מַחֲזֵ֤ה שַׁדַּי֙ יֶֽחֱזֶ֔ה נֹפֵ֖ל וּגְל֥וּי עֵינָֽיִם׃ נְאוּם הַשּׁוֹמֵעַ דִּבּוּרֵי אֵל, שֶׁרוֹאֶה מַחֲזוֹת שַׁדַּי, נוֹפֵל בְּתַרְדֵּמָה וּמִתְגַּלָּה לְעֵינָיו נְבוּאָה.
ה מַה־טֹּ֥בוּ אֹהָלֶ֖יךָ יַעֲקֹ֑ב מִשְׁכְּנֹתֶ֖יךָ יִשְׂרָאֵֽל׃ כַּמָּה טוֹבִים אוֹהָלֶיךָ יַעֲקֹב, וּמְגוּרֶיךָ יִשְׂרָאֵל!
ו כִּנְחָלִ֣ים נִטָּ֔יוּ כְּגַנֹּ֖ת עֲלֵ֣י נָהָ֑ר כַּאֲהָלִים֙ נָטַ֣ע יְהֹוָ֔ה כַּאֲרָזִ֖ים עֲלֵי־מָֽיִם׃ כְּמוֹ נְחָלִים זוֹרְמִים, כְּמוֹ גַּנִּים עַל נָהָר, כְּמוֹ בְּשָׂמִים שֶׁה' נָטַע, כְּמוֹ אֲרָזִים עַל מַיִם.
ז יִֽזַּל־מַ֙יִם֙ מִדָּ֣לְיָ֔ו וְזַרְע֖וֹ בְּמַ֣יִם רַבִּ֑ים וְיָרֹ֤ם מֵֽאֲגַג֙ מַלְכּ֔וֹ וְתִנַּשֵּׂ֖א מַלְכֻתֽוֹ׃ יְטַפְטְפוּ מַיִם מֵעֲנָפָיו, וְזַרְעוֹ יִזָּרַע בְּמַיִם רַבִּים, מַלְכּוֹ יִהְיֶה מְרוֹמָם יוֹתֵר מֵאֲגַג מֶלֶךְ עֲמָלֵק, וּמַלְכוּתוֹ תִּתְעַלֶּה.
ח אֵ֚ל מוֹצִיא֣וֹ מִמִּצְרַ֔יִם כְּתוֹעֲפֹ֥ת רְאֵ֖ם ל֑וֹ יֹאכַ֞ל גּוֹיִ֣ם צָרָ֗יו וְעַצְמֹתֵיהֶ֛ם יְגָרֵ֖ם וְחִצָּ֥יו יִמְחָֽץ׃ הָאֵל שֶׁהוֹצִיא אוֹתוֹ מִמִּצְרַיִם נוֹתֵן לוֹ כֹּחַ כְּמוֹ שֶׁל שׁוֹר הַבָּר הַגָּדוֹל, הוּא יִכְבּוֹשׁ אֶת הָעַמִּים שׂוֹנְאָיו וְיִשְׁבּוֹר אֶת עַצְמוֹתֵיהֶם וְיִמְחַץ בָּהֶם אֶת חִצָּיו.
ט כָּרַ֨ע שָׁכַ֧ב כַּאֲרִ֛י וּכְלָבִ֖יא מִ֣י יְקִימֶ֑נּוּ מְבָרְכֶ֣יךָ בָר֔וּךְ וְאֹרְרֶ֖יךָ אָרֽוּר׃ יִרְבּוֹץ יִתְיַשֵּׁב וְיִשְׁכַּב כְּמוֹ אַרְיֵה – מִי יָקִים אוֹתוֹ! מְבָרְכֶיךָ מְבוֹרָכִים וּמְקַלְּלֶיךָ מְקֻלָּלִים.
י וַיִּֽחַר־אַ֤ף בָּלָק֙ אֶל־בִּלְעָ֔ם וַיִּסְפֹּ֖ק אֶת־כַּפָּ֑יו וַיֹּ֨אמֶר בָּלָ֜ק אֶל־בִּלְעָ֗ם לָקֹ֤ב אֹֽיְבַי֙ קְרָאתִ֔יךָ וְהִנֵּה֙ בֵּרַ֣כְתָּ בָרֵ֔ךְ זֶ֖ה שָׁלֹ֥שׁ פְּעָמִֽים׃ בָּלָק כָּעַס עַל בִּלְעָם וּמָחָה אֶת כַּפָּיו בְּכַעַס, בָּלָק אָמַר לְבִלְעָם: קָרָאתִי לְךָ לְקַלֵּל אֶת אוֹיְבַי וְהִנֵּה בֵּרַכְתָּ אוֹתָם כְּבָר שָׁלוֹשׁ פְּעָמִים!
יא וְעַתָּ֖ה בְּרַח־לְךָ֣ אֶל־מְקוֹמֶ֑ךָ אָמַ֙רְתִּי֙ כַּבֵּ֣ד אֲכַבֶּדְךָ֔ וְהִנֵּ֛ה מְנָעֲךָ֥ יְהֹוָ֖ה מִכָּבֽוֹד׃ וְעַכְשָׁיו בְּרַח לְךָ לִמְקוֹמְךָ! רָצִיתִי לְכַבֵּד אוֹתְךָ וְהִנֵּה ה' מָנַע מִמְּךָ כָּבוֹד.
יב וַיֹּ֥אמֶר בִּלְעָ֖ם אֶל־בָּלָ֑ק הֲלֹ֗א גַּ֧ם אֶל־מַלְאָכֶ֛יךָ אֲשֶׁר־שָׁלַ֥חְתָּ אֵלַ֖י דִּבַּ֥רְתִּי לֵאמֹֽר׃ בִּלְעָם אָמַר לְבָלָק: הֲרֵי גַּם לִשְׁלוּחֶיךָ שֶׁשָּׁלַחְתָּ אֵלַי דִּבַּרְתִּי וְאָמַרְתִּי כָּךְ:
יג אִם־יִתֶּן־לִ֨י בָלָ֜ק מְלֹ֣א בֵיתוֹ֮ כֶּ֣סֶף וְזָהָב֒ לֹ֣א אוּכַ֗ל לַעֲבֹר֙ אֶת־פִּ֣י יְהֹוָ֔ה לַעֲשׂ֥וֹת טוֹבָ֛ה א֥וֹ רָעָ֖ה מִלִּבִּ֑י אֲשֶׁר־יְדַבֵּ֥ר יְהֹוָ֖ה אֹת֥וֹ אֲדַבֵּֽר׃ גַּם אִם בָּלָק יִתֵּן לִי כֶּסֶף וְזָהָב בְּכַמּוּת שֶׁאֶפְשָׁר לְמַלֵּא בָּהּ אֶת בֵּיתוֹ – לֹא אוּכַל לַעֲבוֹר עַל דִּבְרֵי ה' אֱלֹהַי וְלַעֲשׂוֹת מִלִּבִּי טוֹב אוֹ רַע – רַק אֶת מַה שֶּׁה' יְדַבֵּר אוֹתוֹ אֲדַבֵּר.

שביעי[עריכה]

יד וְעַתָּ֕ה הִנְנִ֥י הוֹלֵ֖ךְ לְעַמִּ֑י לְכָה֙ אִיעָ֣צְךָ֔ אֲשֶׁ֨ר יַעֲשֶׂ֜ה הָעָ֥ם הַזֶּ֛ה לְעַמְּךָ֖ בְּאַחֲרִ֥ית הַיָּמִֽים׃ וְעַכְשָׁיו אֲנִי הוֹלֵךְ לְעַמִּי, בֹּא וַאֲיָעֵץ לְךָ מַה לַּעֲשׂוֹת וַאֲגַלֶּה לְךָ מַה יַּעֲשֶׂה הָעָם הַזֶּה לְעַמְּךָ בְּאַחֲרִית הַיָּמִים.
טו וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר נְאֻ֤ם בִּלְעָם֙ בְּנ֣וֹ בְעֹ֔ר וּנְאֻ֥ם הַגֶּ֖בֶר שְׁתֻ֥ם הָעָֽיִן׃ הוּא נָשָׂא אֶת נְאוּמוֹ וְאָמַר: נְאוּם בִּלְעָם בְּנוֹ שֶׁל בְּעוֹר וּנְאוּם הַגֶּבֶר בַּעַל הָעַיִן הַמִּתְבּוֹנֶנֶת הֵיטֵב.
טז נְאֻ֗ם שֹׁמֵ֙עַ֙ אִמְרֵי־אֵ֔ל וְיֹדֵ֖עַ דַּ֣עַת עֶלְי֑וֹן מַחֲזֵ֤ה שַׁדַּי֙ יֶֽחֱזֶ֔ה נֹפֵ֖ל וּגְל֥וּי עֵינָֽיִם׃ נְאוּם הַשּׁוֹמֵעַ דִּבּוּרֵי אֵל וְיוֹדֵעַ דַּעַת מֵאֵל עֶלְיוֹן, הָרוֹאֶה מַחֲזוֹת שַׁדַּי, נוֹפֵל בְּתַרְדֵּמָה וּמִתְגַּלָּה לְעֵינָיו נְבוּאָה.
יז אֶרְאֶ֙נּוּ֙ וְלֹ֣א עַתָּ֔ה אֲשׁוּרֶ֖נּוּ וְלֹ֣א קָר֑וֹב דָּרַ֨ךְ כּוֹכָ֜ב מִֽיַּעֲקֹ֗ב וְקָ֥ם שֵׁ֙בֶט֙ מִיִּשְׂרָאֵ֔ל וּמָחַץ֙ פַּאֲתֵ֣י מוֹאָ֔ב וְקַרְקַ֖ר כׇּל־בְּנֵי־שֵֽׁת׃ אֲנִי רוֹאֶה אוֹתוֹ וְלֹא עַכְשָׁיו, אֲנִי מַבִּיט בּוֹ וְהוּא אֵינוֹ קָרוֹב, בֶּעָתִיד יִזְרַח מֶלֶךְ עֶלְיוֹן מִיַּעֲקֹב וְיָקוּם שַׁלִּיט מִיִּשְׂרָאֵל, וְיִמְחַץ אֶת מוֹאָב שֶׁיּוֹשְׁבִים בִּגְבוּלוֹ, וְיִכְבּוֹשׁ אֶת כָּל בְּנֵי הָאָדָם.
יח וְהָיָ֨ה אֱד֜וֹם יְרֵשָׁ֗ה וְהָיָ֧ה יְרֵשָׁ֛ה שֵׂעִ֖יר אֹיְבָ֑יו וְיִשְׂרָאֵ֖ל עֹ֥שֶׂה חָֽיִל׃ אֱדוֹם יִהְיוּ לוֹ לִירוּשָׁה – שֵׂעִיר אוֹיְבָיו יִהְיוּ לוֹ לִירוּשָׁה, וְיִשְׂרָאֵל עוֹשֶׂה חַיִל.
יט וְיֵ֖רְדְּ מִֽיַּעֲקֹ֑ב וְהֶֽאֱבִ֥יד שָׂרִ֖יד מֵעִֽיר׃ וְאִישׁ מִיַּעֲקֹב יִשְׁלוֹט וְיַשְׁמִיד כָּל שָׂרִיד מֵהָעִיר.
כ וַיַּרְא֙ אֶת־עֲמָלֵ֔ק וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר רֵאשִׁ֤ית גּוֹיִם֙ עֲמָלֵ֔ק וְאַחֲרִית֖וֹ עֲדֵ֥י אֹבֵֽד׃ הוּא רָאָה אֶת עֲמָלֵק, נָשָׂא אֶת נְאוּמוֹ וְאָמַר: עֲמָלֵק הוּא הָעָם הָרִאשׁוֹן שֶׁנִּלְחַם בְּיִשְׂרָאֵל, וְסוֹפוֹ שֶׁיֹּאבַד.
כא וַיַּרְא֙ אֶת־הַקֵּינִ֔י וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר אֵיתָן֙ מֽוֹשָׁבֶ֔ךָ וְשִׂ֥ים בַּסֶּ֖לַע קִנֶּֽךָ׃ הוּא רָאָה אֶת בְּנֵי יִתְרוֹ, נָשָׂא אֶת נְאוּמוֹ וְאָמַר: חַזֵּק אֶת מְגוּרֶיךָ וְשִׂים בַּסֶּלַע אֶת בֵּיתְךָ.
כב כִּ֥י אִם־יִהְיֶ֖ה לְבָ֣עֵֽר קָ֑יִן עַד־מָ֖ה אַשּׁ֥וּר תִּשְׁבֶּֽךָּ׃ כִּי אִם יִרְצוּ לְהַשְׁמִיד אֶת שִׁלְטוֹן בְּנֵי יִתְרוֹ, לָמָּה שֶׁאַשּׁוּר יִקְחוּ אוֹתְךָ בַּשֶּׁבִי?
כג וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר א֕וֹי מִ֥י יִחְיֶ֖ה מִשֻּׂמ֥וֹ אֵֽל׃ הוּא נָשָׂא אֶת נְאוּמוֹ וְאָמַר: אוֹי! מִי יִשָּׁאֵר בַּחַיִּים כְּשֶׁהָאֵל יַעֲשֶׂה אֶת יִשְׂרָאֵל שַׁלִּיטִים.
כד וְצִים֙ מִיַּ֣ד כִּתִּ֔ים וְעִנּ֥וּ אַשּׁ֖וּר וְעִנּוּ־עֵ֑בֶר וְגַם־ה֖וּא עֲדֵ֥י אֹבֵֽד׃ וְצִיֵּי אֳנִיּוֹת יֵצְאוּ מֵאֶרֶץ כִּתִּים, יְעַנּוּ אֶת עַם אַשּׁוּר וִיעַנּוּ אֶת עַם עֵבֶר, וְגַם הוּא – עַם כִּתִּים סוֹפוֹ שֶׁיֹּאבַד.
כה וַיָּ֣קׇם בִּלְעָ֔ם וַיֵּ֖לֶךְ וַיָּ֣שׇׁב לִמְקֹמ֑וֹ וְגַם־בָּלָ֖ק הָלַ֥ךְ לְדַרְכּֽוֹ׃ בִּלְעָם קָם וְהָלַךְ וְחָזַר לִמְקוֹמוֹ, וְגַם בָּלָק הָלַךְ לְדַרְכּוֹ.
פ פָּרָשַׁת קִנְאַת פִּינְחָס
כה א וַיֵּ֥שֶׁב יִשְׂרָאֵ֖ל בַּשִּׁטִּ֑ים וַיָּ֣חֶל הָעָ֔ם לִזְנ֖וֹת אֶל־בְּנ֥וֹת מוֹאָֽב׃ יִשְׂרָאֵל חָנוּ בַּשִּׁטִּים, וְהָעָם הִתְחִילוּ לִזְנוֹת עִם בְּנוֹת מוֹאָב.
ב וַתִּקְרֶ֣אןָ לָעָ֔ם לְזִבְחֵ֖י אֱלֹהֵיהֶ֑ן וַיֹּ֣אכַל הָעָ֔ם וַיִּֽשְׁתַּחֲו֖וּ לֵאלֹֽהֵיהֶֽן׃ הֵן קָרְאוּ לָעָם לֶאֱכוֹל מִקָּרְבְּנוֹת אֱלֹהֵיהֶן, וְהָעָם אָכְלוּ וְהִשְׁתַּחֲווּ לֵאלֹהֵיהֶן.
ג וַיִּצָּ֥מֶד יִשְׂרָאֵ֖ל לְבַ֣עַל פְּע֑וֹר וַיִּֽחַר־אַ֥ף יְהֹוָ֖ה בְּיִשְׂרָאֵֽל׃ עַם יִשְׂרָאֵל הִתְחַבֵּר לָאֱלִיל בַּעַל פְּעוֹר, וַה' כָּעַס עַל יִשְׂרָאֵל.
ד וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה קַ֚ח אֶת־כׇּל־רָאשֵׁ֣י הָעָ֔ם וְהוֹקַ֥ע אוֹתָ֛ם לַיהֹוָ֖ה נֶ֣גֶד הַשָּׁ֑מֶשׁ וְיָשֹׁ֛ב חֲר֥וֹן אַף־יְהֹוָ֖ה מִיִּשְׂרָאֵֽל׃ ה' אָמַר לְמֹשֶׁה: קַח אִתְּךָ אֶת כָּל רָאשֵׁי הָעָם וּתְלֵה אֶת הַחוֹטְאִים לְמוּל הַשֶּׁמֶשׁ לְמַעַן כְּבוֹד ה', וְכָךְ יִפָּסֵק הַכַּעַס שֶׁל ה' מִיִּשְׂרָאֵל.
ה וַיֹּ֣אמֶר מֹשֶׁ֔ה אֶל־שֹׁפְטֵ֖י יִשְׂרָאֵ֑ל הִרְגוּ֙ אִ֣ישׁ אֲנָשָׁ֔יו הַנִּצְמָדִ֖ים לְבַ֥עַל פְּעֽוֹר׃ מֹשֶׁה אָמַר לְשׁוֹפְטֵי יִשְׂרָאֵל: הִרְגוּ כָּל אִישׁ אֶת אֲנָשָׁיו שֶׁהִתְחַבְּרוּ לְבַעַל פְּעוֹר!
ו וְהִנֵּ֡ה אִישׁ֩ מִבְּנֵ֨י יִשְׂרָאֵ֜ל בָּ֗א וַיַּקְרֵ֤ב אֶל־אֶחָיו֙ אֶת־הַמִּדְיָנִ֔ית לְעֵינֵ֣י מֹשֶׁ֔ה וּלְעֵינֵ֖י כׇּל־עֲדַ֣ת בְּנֵי־יִשְׂרָאֵ֑ל וְהֵ֣מָּה בֹכִ֔ים פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וְהִנֵּה אִישׁ מִבְּנֵי יִשְׂרָאֵל בָּא וְקֵרֵב אֶל מִשְׁפַּחְתּוֹ אֶת הַבַּת הַמִּדְיָנִית לְעֵינֵי מֹשֶׁה וּלְעֵינֵי כָּל הַקָּהָל שֶׁל בְּנֵי יִשְׂרָאֵל, וְהֵם בּוֹכִים לִפְנֵי פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.

מפטיר[עריכה]

ז וַיַּ֗רְא פִּֽינְחָס֙ בֶּן־אֶלְעָזָ֔ר בֶּֽן־אַהֲרֹ֖ן הַכֹּהֵ֑ן וַיָּ֙קׇם֙ מִתּ֣וֹךְ הָֽעֵדָ֔ה וַיִּקַּ֥ח רֹ֖מַח בְּיָדֽוֹ׃ פִּינְחָס בֶּן אֶלְעָזָר בֶּן אַהֲרֹן הַכֹּהֵן רָאָה, קָם מִתּוֹךְ הַצִּבּוּר וְלָקַח רוֹמַח בְּיָדוֹ.
ח וַ֠יָּבֹ֠א אַחַ֨ר אִֽישׁ־יִשְׂרָאֵ֜ל אֶל־הַקֻּבָּ֗ה וַיִּדְקֹר֙ אֶת־שְׁנֵיהֶ֔ם אֵ֚ת אִ֣ישׁ יִשְׂרָאֵ֔ל וְאֶת־הָאִשָּׁ֖ה אֶל־קֳבָתָ֑הּ וַתֵּֽעָצַר֙ הַמַּגֵּפָ֔ה מֵעַ֖ל בְּנֵ֥י יִשְׂרָאֵֽל׃ הוּא בָּא אַחֲרֵי הָאִישׁ מִבְּנֵי יִשְׂרָאֵל לָאוֹהֶל שֶׁבּוֹ שָׁכְבוּ, וְדָקַר אֶת שְׁנֵיהֶם – אֶת הָאִישׁ מִיִּשְׂרָאֵל וְאֶת הָאִשָּׁה בְּבִטְנָהּ, וְהַמַּגֵּפָה נֶעֶצְרָה מֵעַל בְּנֵי יִשְׂרָאֵל.
ט וַיִּהְי֕וּ הַמֵּתִ֖ים בַּמַּגֵּפָ֑ה אַרְבָּעָ֥ה וְעֶשְׂרִ֖ים אָֽלֶף׃ מִסְפַּר הַמֵּתִים בַּמַּגֵּפָה הָיוּ עֶשְׂרִים וְאַרְבָּעָה אֶלֶף.
