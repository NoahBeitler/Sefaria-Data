ראשון[עריכה]

פ פָּרָשַׁת מַחֲלֹקֶת קֹרַח
טז א וַיִּקַּ֣ח קֹ֔רַח בֶּן־יִצְהָ֥ר בֶּן־קְהָ֖ת בֶּן־לֵוִ֑י וְדָתָ֨ן וַאֲבִירָ֜ם בְּנֵ֧י אֱלִיאָ֛ב וְא֥וֹן בֶּן־פֶּ֖לֶת בְּנֵ֥י רְאוּבֵֽן׃ קוֹרַח בֶּן יִצְהָר בֶּן קְהָת בֶּן לֵוִי, וְדָתָן וַאֲבִירָם בְּנֵי אֱלִיאָב וְאוֹן בֶּן פַּלּוּא – בְּנֵי רְאוּבֵן, הִתְאַסְּפוּ.
ב וַיָּקֻ֙מוּ֙ לִפְנֵ֣י מֹשֶׁ֔ה וַאֲנָשִׁ֥ים מִבְּנֵֽי־יִשְׂרָאֵ֖ל חֲמִשִּׁ֣ים וּמָאתָ֑יִם נְשִׂיאֵ֥י עֵדָ֛ה קְרִאֵ֥י מוֹעֵ֖ד אַנְשֵׁי־שֵֽׁם׃ הֵם קָמוּ לִמְרוֹד בְּמֹשֶׁה, וְאִתָּם מָאתַיִם וַחֲמִשִּׁים אֲנָשִׁים מִבְּנֵי יִשְׂרָאֵל, נְשִׂיאֵי צִבּוּר, שֶׁמֻּזְמָנִים לַאֲסֵיפוֹת, אֲנָשִׁים מְפוּרְסָמִים.
ג וַיִּֽקָּהֲל֞וּ עַל־מֹשֶׁ֣ה וְעַֽל־אַהֲרֹ֗ן וַיֹּאמְר֣וּ אֲלֵהֶם֮ רַב־לָכֶם֒ כִּ֤י כׇל־הָֽעֵדָה֙ כֻּלָּ֣ם קְדֹשִׁ֔ים וּבְתוֹכָ֖ם יְהֹוָ֑ה וּמַדּ֥וּעַ תִּֽתְנַשְּׂא֖וּ עַל־קְהַ֥ל יְהֹוָֽה׃ הֵם הִתְאַסְּפוּ נֶגֶד מֹשֶׁה וְנֶגֶד אַהֲרֹן וְאָמְרוּ לָהֶם: מַסְפִּיק לָכֶם! הֲרֵי כָּל הַצִּבּוּר כֻּלָּם קְדוֹשִׁים וּבְתוֹכָם שׁוֹכֵן ה', וְלָמָּה אַתֶּם מִתְנַשְּׂאִים עַל קְהַל ה'?
ד וַיִּשְׁמַ֣ע מֹשֶׁ֔ה וַיִּפֹּ֖ל עַל־פָּנָֽיו׃ מֹשֶׁה שָׁמַע וְהִשְׁתַּטֵּחַ עַל פָּנָיו.
ה וַיְדַבֵּ֨ר אֶל־קֹ֜רַח וְאֶֽל־כׇּל־עֲדָתוֹ֮ לֵאמֹר֒ בֹּ֠קֶר וְיֹדַ֨ע יְהֹוָ֧ה אֶת־אֲשֶׁר־ל֛וֹ וְאֶת־הַקָּד֖וֹשׁ וְהִקְרִ֣יב אֵלָ֑יו וְאֵ֛ת אֲשֶׁ֥ר יִבְחַר־בּ֖וֹ יַקְרִ֥יב אֵלָֽיו׃ הוּא דִּבֵּר אֶל קוֹרַח וְאֶל כָּל חֲבוּרָתוֹ וְאָמַר: בַּבּוֹקֶר ה' יוֹדִיעַ מִי שֶׁלּוֹ וּמִי קָדוֹשׁ, וְהוּא יְקָרֵב אוֹתוֹ אֵלָיו, וְאֶת מִי שֶׁיִּבְחַר הוּא יְקָרֵב לַעֲבוֹדָתוֹ.
ו זֹ֖את עֲשׂ֑וּ קְחוּ־לָכֶ֣ם מַחְתּ֔וֹת קֹ֖רַח וְכׇל־עֲדָתֽוֹ׃ עֲשׂוּ כָּךְ: קְחוּ לָכֶם יָעִים, קוֹרַח וְכָל חֲבוּרָתוֹ.
ז וּתְנ֣וּ בָהֵ֣ן ׀ אֵ֡שׁ וְשִׂ֩ימוּ֩ עֲלֵיהֶ֨ן ׀ קְטֹ֜רֶת לִפְנֵ֤י יְהֹוָה֙ מָחָ֔ר וְהָיָ֗ה הָאִ֛ישׁ אֲשֶׁר־יִבְחַ֥ר יְהֹוָ֖ה ה֣וּא הַקָּד֑וֹשׁ רַב־לָכֶ֖ם בְּנֵ֥י לֵוִֽי׃ וְשִׂימוּ בָּהֵם גֶּחָלִים וְשִׂימוּ עֲלֵיהֶם קְטוֹרֶת וְתַגִּישׁוּ לִפְנֵי ה' מָחָר, וְאָז הָאִישׁ שֶׁה' יִבְחַר בּוֹ הוּא הַקָּדוֹשׁ, מַסְפִּיק לָכֶם בְּנֵי לֵוִי!
ח וַיֹּ֥אמֶר מֹשֶׁ֖ה אֶל־קֹ֑רַח שִׁמְעוּ־נָ֖א בְּנֵ֥י לֵוִֽי׃ מֹשֶׁה אָמַר לְקוֹרַח: שִׁמְעוּ בְּבַקָּשָׁה בְּנֵי לֵוִי!
ט הַמְעַ֣ט מִכֶּ֗ם כִּֽי־הִבְדִּיל֩ אֱלֹהֵ֨י יִשְׂרָאֵ֤ל אֶתְכֶם֙ מֵעֲדַ֣ת יִשְׂרָאֵ֔ל לְהַקְרִ֥יב אֶתְכֶ֖ם אֵלָ֑יו לַעֲבֹ֗ד אֶת־עֲבֹדַת֙ מִשְׁכַּ֣ן יְהֹוָ֔ה וְלַעֲמֹ֛ד לִפְנֵ֥י הָעֵדָ֖ה לְשָׁרְתָֽם׃ הַאִם לֹא מַסְפִּיק לָכֶם שֶׁאֱלֹהֵי יִשְׂרָאֵל הִבְדִּיל אֶתְכֶם מִצִּבּוּר יִשְׂרָאֵל וְקֵרֵב אֶתְכֶם אֵלָיו לַעֲבוֹד אֶת עֲבוֹדַת מִשְׁכַּן ה' וְלַעֲמוֹד לִפְנֵי הַצִּבּוּר לְשָׁרְתָם,
י וַיַּקְרֵב֙ אֹֽתְךָ֔ וְאֶת־כׇּל־אַחֶ֥יךָ בְנֵי־לֵוִ֖י אִתָּ֑ךְ וּבִקַּשְׁתֶּ֖ם גַּם־כְּהֻנָּֽה׃ וְהוּא קֵרֵב אֶתְכֶם וְאֶת כָּל אֲחֵיכֶם בְּנֵי לֵוִי אִתְּכֶם, וְאַתֶּם דּוֹרְשִׁים גַּם כְּהֻנָּה?!
יא לָכֵ֗ן אַתָּה֙ וְכׇל־עֲדָ֣תְךָ֔ הַנֹּעָדִ֖ים עַל־יְהֹוָ֑ה וְאַהֲרֹ֣ן מַה־ה֔וּא כִּ֥י תלונו תַלִּ֖ינוּ עָלָֽיו׃ לָכֵן אַתָּה וְכָל חֲבוּרָתְךָ הִתְאַסַּפְתֶּם נֶגֶד ה' – אַהֲרֹן מַה חֶטְאוֹ שֶׁהִתְלוֹנַנְתֶּם עָלָיו?
יב וַיִּשְׁלַ֣ח מֹשֶׁ֔ה לִקְרֹ֛א לְדָתָ֥ן וְלַאֲבִירָ֖ם בְּנֵ֣י אֱלִיאָ֑ב וַיֹּאמְר֖וּ לֹ֥א נַעֲלֶֽה׃ מֹשֶׁה שָׁלַח לִקְרֹא לְדָתָן וְלַאֲבִירָם בְּנֵי אֱלִיאָב, וְהֵם אָמְרוּ: לֹא נַעֲלֶה אֵלֶיךָ!
יג הַמְעַ֗ט כִּ֤י הֶֽעֱלִיתָ֙נוּ֙ מֵאֶ֨רֶץ זָבַ֤ת חָלָב֙ וּדְבַ֔שׁ לַהֲמִיתֵ֖נוּ בַּמִּדְבָּ֑ר כִּֽי־תִשְׂתָּרֵ֥ר עָלֵ֖ינוּ גַּם־הִשְׂתָּרֵֽר׃ הַאִם לֹא מַסְפִּיק שֶׁהֶעֱלִיתָ אוֹתָנוּ מֵאֶרֶץ שׁוֹפַעַת חָלָב וּדְבַשׁ כְּדֵי לַהֲרוֹג אוֹתָנוּ בַּמִּדְבָּר, שֶׁאַתָּה גַּם מִשְׁתַּלֵּט עָלֵינוּ?!

שני[עריכה]

יד אַ֡ף לֹ֣א אֶל־אֶ֩רֶץ֩ זָבַ֨ת חָלָ֤ב וּדְבַשׁ֙ הֲבִ֣יאֹתָ֔נוּ וַתִּ֨תֶּן־לָ֔נוּ נַחֲלַ֖ת שָׂדֶ֣ה וָכָ֑רֶם הַעֵינֵ֞י הָאֲנָשִׁ֥ים הָהֵ֛ם תְּנַקֵּ֖ר לֹ֥א נַעֲלֶֽה׃ גַּם אִם לֹא הֵבֵאתָ אוֹתָנוּ לְאֶרֶץ שׁוֹפַעַת חָלָב וּדְבַשׁ, לְפָחוֹת הָיִיתָ נוֹתֵן לָנוּ נַחֲלָה שֶׁל שָׂדֶה וְכֶרֶם! הַאִם אַתָּה מִתְיַחֵס אֵלֵינוּ כְּאֶל עִוְּרִים?! לֹא נַעֲלֶה אֵלֶיךָ!
טו וַיִּ֤חַר לְמֹשֶׁה֙ מְאֹ֔ד וַיֹּ֙אמֶר֙ אֶל־יְהֹוָ֔ה אַל־תֵּ֖פֶן אֶל־מִנְחָתָ֑ם לֹ֠א חֲמ֨וֹר אֶחָ֤ד מֵהֶם֙ נָשָׂ֔אתִי וְלֹ֥א הֲרֵעֹ֖תִי אֶת־אַחַ֥ד מֵהֶֽם׃ מֹשֶׁה הִצְטַעֵר מְאוֹד וְאָמַר לַה': אַל תִּפְנֶה אֶל מִנְחוֹתֵיהֶם, לֹא לָקַחְתִּי מֵהֶם אֲפִלּוּ חֲמוֹר אֶחָד וְלֹא עָשִׂיתִי רַע לְאַף אֶחָד מֵהֶם!
טז וַיֹּ֤אמֶר מֹשֶׁה֙ אֶל־קֹ֔רַח אַתָּה֙ וְכׇל־עֲדָ֣תְךָ֔ הֱי֖וּ לִפְנֵ֣י יְהֹוָ֑ה אַתָּ֥ה וָהֵ֛ם וְאַהֲרֹ֖ן מָחָֽר׃ מֹשֶׁה אָמַר לְקוֹרַח: אַתָּה וְכָל חֲבוּרָתְךָ עִמְדוּ לִפְנֵי ה', אַתָּה וְהֵם וְגַם אַהֲרֹן, מָחָר.
יז וּקְח֣וּ ׀ אִ֣ישׁ מַחְתָּת֗וֹ וּנְתַתֶּ֤ם עֲלֵיהֶם֙ קְטֹ֔רֶת וְהִקְרַבְתֶּ֞ם לִפְנֵ֤י יְהֹוָה֙ אִ֣ישׁ מַחְתָּת֔וֹ חֲמִשִּׁ֥ים וּמָאתַ֖יִם מַחְתֹּ֑ת וְאַתָּ֥ה וְאַהֲרֹ֖ן אִ֥ישׁ מַחְתָּתֽוֹ׃ וּקְחוּ כָּל אֶחָד אֶת הַיָּעֶה שֶׁלּוֹ וְשִׂימוּ עֲלֵיהֶם קְטוֹרֶת וְתַקְרִיבוּ לִפְנֵי ה' כָּל אֶחָד אֶת הַיָּעֶה שֶׁלּוֹ – מָאתַיִם חֲמִשִּׁים יָעִים, וְגַם אַתָּה וְאַהֲרֹן – כָּל אֶחָד אֶת הַיָּעֶה שֶׁלּוֹ.
יח וַיִּקְח֞וּ אִ֣ישׁ מַחְתָּת֗וֹ וַיִּתְּנ֤וּ עֲלֵיהֶם֙ אֵ֔שׁ וַיָּשִׂ֥ימוּ עֲלֵיהֶ֖ם קְטֹ֑רֶת וַֽיַּעַמְד֗וּ פֶּ֛תַח אֹ֥הֶל מוֹעֵ֖ד וּמֹשֶׁ֥ה וְאַהֲרֹֽן׃ הֵם לָקְחוּ כָּל אֶחָד אֶת הַיָּעֶה שֶׁלּוֹ וְשָׂמוּ עֲלֵיהֶם גֶּחָלִים וְשָׂמוּ עֲלֵיהֶם קְטוֹרֶת, וְעָמְדוּ מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת וּמֹשֶׁה וְאַהֲרֹן אִתָּם.
יט וַיַּקְהֵ֨ל עֲלֵיהֶ֥ם קֹ֙רַח֙ אֶת־כׇּל־הָ֣עֵדָ֔ה אֶל־פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וַיֵּרָ֥א כְבוֹד־יְהֹוָ֖ה אֶל־כׇּל־הָעֵדָֽה׃ קוֹרַח אָסַף לְמוּלָם אֶת כָּל הַצִּבּוּר אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת, וּכְבוֹד ה' הִתְגַּלָּה אֶל כָּל הַצִּבּוּר.

שלישי[עריכה]

ס פָּרָשַׁת עֹנֶשׁ עֲדַת קֹרַח
כ וַיְדַבֵּ֣ר יְהֹוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאֶל אַהֲרֹן וְאָמַר:
כא הִבָּ֣דְל֔וּ מִתּ֖וֹךְ הָעֵדָ֣ה הַזֹּ֑את וַאֲכַלֶּ֥ה אֹתָ֖ם כְּרָֽגַע׃ צְאוּ מִתּוֹךְ הַצִּבּוּר הַזֶּה וְאַשְׁמִיד אוֹתָם תּוֹךְ רֶגַע!
כב וַיִּפְּל֤וּ עַל־פְּנֵיהֶם֙ וַיֹּ֣אמְר֔וּ אֵ֕ל אֱלֹהֵ֥י הָרוּחֹ֖ת לְכׇל־בָּשָׂ֑ר הָאִ֤ישׁ אֶחָד֙ יֶחֱטָ֔א וְעַ֥ל כׇּל־הָעֵדָ֖ה תִּקְצֹֽף׃ הֵם הִשְׁתַּטְּחוּ עַל פְּנֵיהֶם, וְאָמְרוּ: אֵל, אֱלֹהֵי נִשְׁמוֹת כָּל הָאֲנָשִׁים! הַאִם בִּגְלַל שֶׁאִישׁ אֶחָד חָטָא תִּכְעַס עַל כָּל הַצִּבּוּר?!

כג וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
כד דַּבֵּ֥ר אֶל־הָעֵדָ֖ה לֵאמֹ֑ר הֵֽעָלוּ֙ מִסָּבִ֔יב לְמִשְׁכַּן־קֹ֖רַח דָּתָ֥ן וַאֲבִירָֽם׃ דַּבֵּר אֶל הַצִּבּוּר וֶאֱמוֹר: הִסְתַּלְּקוּ מִסָּבִיב לִמְגוּרֵי קוֹרַח דָּתָן וַאֲבִירָם.
כה וַיָּ֣קׇם מֹשֶׁ֔ה וַיֵּ֖לֶךְ אֶל־דָּתָ֣ן וַאֲבִירָ֑ם וַיֵּלְכ֥וּ אַחֲרָ֖יו זִקְנֵ֥י יִשְׂרָאֵֽל׃ מֹשֶׁה קָם וְהָלַךְ לְדָתָן וַאֲבִירָם, וְזִקְנֵי יִשְׂרָאֵל הָלְכוּ אַחֲרָיו.
כו וַיְדַבֵּ֨ר אֶל־הָעֵדָ֜ה לֵאמֹ֗ר ס֣וּרוּ נָ֡א מֵעַל֩ אׇהֳלֵ֨י הָאֲנָשִׁ֤ים הָֽרְשָׁעִים֙ הָאֵ֔לֶּה וְאַֽל־תִּגְּע֖וּ בְּכׇל־אֲשֶׁ֣ר לָהֶ֑ם פֶּן־תִּסָּפ֖וּ בְּכׇל־חַטֹּאתָֽם׃ הוּא דִּבֵּר אֶל הַקָּהָל וְאָמַר: סוּרוּ בְּבַקָּשָׁה מֵעַל אָהֳלֵי הָאֲנָשִׁים הָרְשָׁעִים הָאֵלֶּה וְאַל תִּגְּעוּ בְּכָל מַה שֶּׁשַּׁיָּךְ לָהֶם שֶׁלֹּא תֻּשְׁמְדוּ בִּגְלַל כָּל חֲטָאֵיהֶם.
כז וַיֵּעָל֗וּ מֵעַ֧ל מִשְׁכַּן־קֹ֛רַח דָּתָ֥ן וַאֲבִירָ֖ם מִסָּבִ֑יב וְדָתָ֨ן וַאֲבִירָ֜ם יָצְא֣וּ נִצָּבִ֗ים פֶּ֚תַח אׇֽהֳלֵיהֶ֔ם וּנְשֵׁיהֶ֥ם וּבְנֵיהֶ֖ם וְטַפָּֽם׃ הֵם הִסְתַּלְּקוּ מִמְּגוּרֵי קוֹרַח דָּתָן וַאֲבִירָם מִסָּבִיב, וְדָתָן וַאֲבִירָם יָצְאוּ זְקוּפִים בְּפֶתַח אָהֳלֵיהֶם וְאִתָּם נְשֵׁיהֶם וּבְנֵיהֶם וְיַלְדֵיהֶם.
כח וַיֹּ֘אמֶר֮ מֹשֶׁה֒ בְּזֹאת֙ תֵּֽדְע֔וּן כִּֽי־יְהֹוָ֣ה שְׁלָחַ֔נִי לַעֲשׂ֕וֹת אֵ֥ת כׇּל־הַֽמַּעֲשִׂ֖ים הָאֵ֑לֶּה כִּי־לֹ֖א מִלִּבִּֽי׃ מֹשֶׁה אָמַר: בְּזֹאת תֵּדְעוּ שֶׁה' שָׁלַח אוֹתִי לַעֲשׂוֹת אֶת כָּל הַמַּעֲשִׂים הָאֵלֶּה – שֶׁלֹּא הִמְצֵאתִי אוֹתָם מִלִּבִּי.
כט אִם־כְּמ֤וֹת כׇּל־הָֽאָדָם֙ יְמֻת֣וּן אֵ֔לֶּה וּפְקֻדַּת֙ כׇּל־הָ֣אָדָ֔ם יִפָּקֵ֖ד עֲלֵיהֶ֑ם לֹ֥א יְהֹוָ֖ה שְׁלָחָֽנִי׃ אִם אֵלֶּה יָמוּתוּ כְּכָל אָדָם, שֶׁתִּגָּזֵר עֲלֵיהֶם גְּזֵירַת מָוֶת רְגִילָה כְּמוֹ שֶׁנִּגְזֶרֶת עַל כָּל אָדָם –לֹא ה' שָׁלַח אוֹתִי!
ל וְאִם־בְּרִיאָ֞ה יִבְרָ֣א יְהֹוָ֗ה וּפָצְתָ֨ה הָאֲדָמָ֤ה אֶת־פִּ֙יהָ֙ וּבָלְעָ֤ה אֹתָם֙ וְאֶת־כׇּל־אֲשֶׁ֣ר לָהֶ֔ם וְיָרְד֥וּ חַיִּ֖ים שְׁאֹ֑לָה וִֽידַעְתֶּ֕ם כִּ֧י נִֽאֲצ֛וּ הָאֲנָשִׁ֥ים הָאֵ֖לֶּה אֶת־יְהֹוָֽה׃ וְאִם ה' יִבְרָא בְּרִיאָה חֲדָשָׁה, שֶׁהָאֲדָמָה תִּפְתַּח אֶת פִּיהָ וְתִבְלַע אוֹתָם וְאֶת כָּל מַה שֶּׁשַּׁיָּךְ לָהֶם, וְהֵם יֵרְדוּ לַקֶּבֶר בַּחַיִּים, אָז תֵּדְעוּ שֶׁהָאֲנָשִׁים הָאֵלֶּה הִכְעִיסוּ אֶת ה'.
לא וַיְהִי֙ כְּכַלֹּת֔וֹ לְדַבֵּ֕ר אֵ֥ת כׇּל־הַדְּבָרִ֖ים הָאֵ֑לֶּה וַתִּבָּקַ֥ע הָאֲדָמָ֖ה אֲשֶׁ֥ר תַּחְתֵּיהֶֽם׃ כְּשֶׁהוּא גָּמַר לְדַבֵּר אֶת כָּל הַדְּבָרִים הָאֵלֶּה הָאֲדָמָה שֶׁמִּתַּחְתֵּיהֶם נִבְקְעָה.
לב וַתִּפְתַּ֤ח הָאָ֙רֶץ֙ אֶת־פִּ֔יהָ וַתִּבְלַ֥ע אֹתָ֖ם וְאֶת־בָּתֵּיהֶ֑ם וְאֵ֤ת כׇּל־הָאָדָם֙ אֲשֶׁ֣ר לְקֹ֔רַח וְאֵ֖ת כׇּל־הָרְכֽוּשׁ׃ הָאָרֶץ פָּתְחָה אֶת פִּיהָ וּבָלְעָה אוֹתָם וְאֶת בָּתֵּיהֶם, וְאֶת כָּל אַנְשֵׁי קוֹרַח וְאֶת כָּל רְכוּשָׁם.
לג וַיֵּ֨רְד֜וּ הֵ֣ם וְכׇל־אֲשֶׁ֥ר לָהֶ֛ם חַיִּ֖ים שְׁאֹ֑לָה וַתְּכַ֤ס עֲלֵיהֶם֙ הָאָ֔רֶץ וַיֹּאבְד֖וּ מִתּ֥וֹךְ הַקָּהָֽל׃ הֵם וְכָל מַה שֶּׁשַּׁיָּךְ לָהֶם יָרְדוּ בַּחַיִּים לַקֶּבֶר, הָאָרֶץ כִּסְּתָה אוֹתָם וְהֵם אָבְדוּ מִתּוֹךְ הַקָּהָל.
לד וְכׇל־יִשְׂרָאֵ֗ל אֲשֶׁ֛ר סְבִיבֹתֵיהֶ֖ם נָ֣סוּ לְקֹלָ֑ם כִּ֣י אָֽמְר֔וּ פֶּן־תִּבְלָעֵ֖נוּ הָאָֽרֶץ׃ וְכָל יִשְׂרָאֵל שֶׁסְּבִיבָם בָּרְחוּ מִצַּעֲקוֹתֵיהֶם, כִּי אָמְרוּ: שֶׁלֹּא תִּבְלַע אוֹתָנוּ הָאֲדָמָה!
לה וְאֵ֥שׁ יָצְאָ֖ה מֵאֵ֣ת יְהֹוָ֑ה וַתֹּ֗אכַל אֵ֣ת הַחֲמִשִּׁ֤ים וּמָאתַ֙יִם֙ אִ֔ישׁ מַקְרִיבֵ֖י הַקְּטֹֽרֶת׃ וְיָצְאָה אֵשׁ מִלִּפְנֵי ה' וְשָׂרְפָה אֶת מָאתַיִם וַחֲמִשִּׁים הָאֲנָשִׁים מַקְרִיבֵי הַקְּטוֹרֶת.

יז א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב אֱמֹ֨ר אֶל־אֶלְעָזָ֜ר בֶּן־אַהֲרֹ֣ן הַכֹּהֵ֗ן וְיָרֵ֤ם אֶת־הַמַּחְתֹּת֙ מִבֵּ֣ין הַשְּׂרֵפָ֔ה וְאֶת־הָאֵ֖שׁ זְרֵה־הָ֑לְאָה כִּ֖י קָדֵֽשׁוּ׃ אֱמוֹר לְאֶלְעָזָר בֶּן אַהֲרֹן הַכֹּהֵן שֶׁיָּרִים אֶת הַיָּעִים מִבֵּין שְׁאֵרִיּוֹת הַשְּׂרֵפָה כִּי הֵם הִתְקַדְּשׁוּ, וְאֶת שְׁאֵרִיּוֹת הַגֶּחָלִים יַשְׁלִיךְ הָלְאָה.
ג אֵ֡ת מַחְתּוֹת֩ הַֽחַטָּאִ֨ים הָאֵ֜לֶּה בְּנַפְשֹׁתָ֗ם וְעָשׂ֨וּ אֹתָ֜ם רִקֻּעֵ֤י פַחִים֙ צִפּ֣וּי לַמִּזְבֵּ֔חַ כִּֽי־הִקְרִיבֻ֥ם לִפְנֵֽי־יְהֹוָ֖ה וַיִּקְדָּ֑שׁוּ וְיִֽהְי֥וּ לְא֖וֹת לִבְנֵ֥י יִשְׂרָאֵֽל׃ וְיִקְחוּ אֶת הַיָּעִים שֶׁל הַחוֹטְאִים הָאֵלֶּה שֶׁגָּרְמוּ מָוֶת לְעַצְמָם, וְיַעֲשׂוּ מֵהֶם רְדִידִים דַּקִּים לְצִפּוּי לַמִּזְבֵּחַ, כִּי הֵם הִקְרִיבוּ אוֹתָם לִפְנֵי ה' וְהֵם הִתְקַדְּשׁוּ, וְיִהְיוּ לְסִימַן תֵּעוּד לִבְנֵי יִשְׂרָאֵל.
ד וַיִּקַּ֞ח אֶלְעָזָ֣ר הַכֹּהֵ֗ן אֵ֚ת מַחְתּ֣וֹת הַנְּחֹ֔שֶׁת אֲשֶׁ֥ר הִקְרִ֖יבוּ הַשְּׂרֻפִ֑ים וַֽיְרַקְּע֖וּם צִפּ֥וּי לַמִּזְבֵּֽחַ׃ אֶלְעָזָר הַכֹּהֵן לָקַח אֶת הַיָּעִים מִנְּחוֹשֶׁת שֶׁהַשְּׂרוּפִים הִקְרִיבוּ, וְרִדְּדוּ אוֹתָם וְעָשׂוּ מֵהֶם צִפּוּי לַמִּזְבֵּחַ.
ה זִכָּר֞וֹן לִבְנֵ֣י יִשְׂרָאֵ֗ל לְ֠מַ֠עַן אֲשֶׁ֨ר לֹֽא־יִקְרַ֜ב אִ֣ישׁ זָ֗ר אֲ֠שֶׁ֠ר לֹ֣א מִזֶּ֤רַע אַהֲרֹן֙ ה֔וּא לְהַקְטִ֥יר קְטֹ֖רֶת לִפְנֵ֣י יְהֹוָ֑ה וְלֹֽא־יִהְיֶ֤ה כְקֹ֙רַח֙ וְכַ֣עֲדָת֔וֹ כַּאֲשֶׁ֨ר דִּבֶּ֧ר יְהֹוָ֛ה בְּיַד־מֹשֶׁ֖ה לֽוֹ׃ מַזְכֶּרֶת לִבְנֵי יִשְׂרָאֵל, כְּדֵי שֶׁאִישׁ זָר שֶׁאֵינוֹ מִצֶּאֱצָאֵי אַהֲרֹן לֹא יִגַּשׁ לְהַקְטִיר קְטוֹרֶת לִפְנֵי ה', וְלֹא יִקְרֶה לוֹ כְּמוֹ קוֹרַח וַחֲבוּרָתוֹ. אֶלְעָזָר עָשָׂה כְּמוֹ שֶׁה' דִּבֵּר אֵלָיו בְּאֶמְצָעוּת מֹשֶׁה.
פ פָּרָשַׁת הַמַּגֵּפָה
ו וַיִּלֹּ֜נוּ כׇּל־עֲדַ֤ת בְּנֵֽי־יִשְׂרָאֵל֙ מִֽמׇּחֳרָ֔ת עַל־מֹשֶׁ֥ה וְעַֽל־אַהֲרֹ֖ן לֵאמֹ֑ר אַתֶּ֥ם הֲמִתֶּ֖ם אֶת־עַ֥ם יְהֹוָֽה׃ כָּל קְהַל בְּנֵי יִשְׂרָאֵל הִתְלוֹנְנוּ לְמָחֳרָת עַל מֹשֶׁה וְעַל אַהֲרֹן וְאָמְרוּ: אַתֶּם גְּרַמְתֶּם מָוֶת לְעַם ה'!
ז וַיְהִ֗י בְּהִקָּהֵ֤ל הָֽעֵדָה֙ עַל־מֹשֶׁ֣ה וְעַֽל־אַהֲרֹ֔ן וַיִּפְנוּ֙ אֶל־אֹ֣הֶל מוֹעֵ֔ד וְהִנֵּ֥ה כִסָּ֖הוּ הֶעָנָ֑ן וַיֵּרָ֖א כְּב֥וֹד יְהֹוָֽה׃ כְּשֶׁהַקָּהָל הִתְאַסְּפוּ לְהִתְלוֹנֵן עַל מֹשֶׁה וְעַל אַהֲרֹן הֵם פָּנוּ לְאוֹהֶל הַהִתְכַּנְּסוּת וְהִנֵּה הֶעָנָן כִּסָּה אוֹתוֹ וּכְבוֹד ה' נִגְלָה.
ח וַיָּבֹ֤א מֹשֶׁה֙ וְאַהֲרֹ֔ן אֶל־פְּנֵ֖י אֹ֥הֶל מוֹעֵֽד׃ מֹשֶׁה וְאַהֲרֹן בָּאוּ אֶל לִפְנֵי אוֹהֶל הַהִתְכַּנְּסוּת.

רביעי[עריכה]

ס
ט וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
י הֵרֹ֗מּוּ מִתּוֹךְ֙ הָעֵדָ֣ה הַזֹּ֔את וַאֲכַלֶּ֥ה אֹתָ֖ם כְּרָ֑גַע וַֽיִּפְּל֖וּ עַל־פְּנֵיהֶֽם׃ הִסְתַּלְּקוּ מִתּוֹךְ הַצִּבּוּר הַזֶּה וְאַשְׁמִיד אוֹתָם תּוֹךְ רֶגַע! הֵם הִשְׁתַּטְּחוּ עַל פְּנֵיהֶם בִּתְפִלָּה.
יא וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶֽל־אַהֲרֹ֗ן קַ֣ח אֶת־הַ֠מַּחְתָּ֠ה וְתֶן־עָלֶ֨יהָ אֵ֜שׁ מֵעַ֤ל הַמִּזְבֵּ֙חַ֙ וְשִׂ֣ים קְטֹ֔רֶת וְהוֹלֵ֧ךְ מְהֵרָ֛ה אֶל־הָעֵדָ֖ה וְכַפֵּ֣ר עֲלֵיהֶ֑ם כִּֽי־יָצָ֥א הַקֶּ֛צֶף מִלִּפְנֵ֥י יְהֹוָ֖ה הֵחֵ֥ל הַנָּֽגֶף׃ מֹשֶׁה אָמַר לְאַהֲרֹן: קַח אֶת הַיָּעֶה וְשִׂים עָלָיו גֶּחָלִים מֵעַל הַמִּזְבֵּחַ וְשִׂים קְטוֹרֶת וְתוֹלִיךְ בִּמְהִירוּת אֶל הַצִּבּוּר וְכַפֵּר עֲלֵיהֶם, כִּי פְּעֻלַּת הַכַּעַס כְּבָר יָצְאָה מִלִּפְנֵי ה' וְהַמַּגֵּפָה הִתְחִילָה!
יב וַיִּקַּ֨ח אַהֲרֹ֜ן כַּאֲשֶׁ֣ר ׀ דִּבֶּ֣ר מֹשֶׁ֗ה וַיָּ֙רׇץ֙ אֶל־תּ֣וֹךְ הַקָּהָ֔ל וְהִנֵּ֛ה הֵחֵ֥ל הַנֶּ֖גֶף בָּעָ֑ם וַיִּתֵּן֙ אֶֽת־הַקְּטֹ֔רֶת וַיְכַפֵּ֖ר עַל־הָעָֽם׃ אַהֲרֹן לָקַח כְּמוֹ שֶׁמֹּשֶׁה דִּבֵּר הוּא רָץ אֶל תּוֹךְ הַקָּהָל וְהִנֵּה הַמַּגֵּפָה הִתְחִילָה בָּעָם, הוּא שָׂם אֶת הַקְּטוֹרֶת וְכִפֵּר עַל הָעָם.
יג וַיַּעֲמֹ֥ד בֵּֽין־הַמֵּתִ֖ים וּבֵ֣ין הַֽחַיִּ֑ים וַתֵּעָצַ֖ר הַמַּגֵּפָֽה׃ הוּא עָמַד בֵּין הַמֵּתִים וּבֵין הַחַיִּים וְהַמַּגֵּפָה נֶעֶצְרָה.
יד וַיִּהְי֗וּ הַמֵּתִים֙ בַּמַּגֵּפָ֔ה אַרְבָּעָ֥ה עָשָׂ֛ר אֶ֖לֶף וּשְׁבַ֣ע מֵא֑וֹת מִלְּבַ֥ד הַמֵּתִ֖ים עַל־דְּבַר־קֹֽרַח׃ הַמֵּתִים בַּמַּגֵּפָה הָיוּ אַרְבָּעָה עָשָׂר אֶלֶף וּשְׁבַע מֵאוֹת, בְּנוֹסָף לַמֵּתִים עַל מַחֲלוֹקֶת קוֹרַח.
טו וַיָּ֤שׇׁב אַהֲרֹן֙ אֶל־מֹשֶׁ֔ה אֶל־פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְהַמַּגֵּפָ֖ה נֶעֱצָֽרָה׃ אַהֲרֹן חָזַר לְמֹשֶׁה אֶל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת וְהַמַּגֵּפָה נֶעֶצְרָה.

חמישי[עריכה]

פ פָּרָשַׁת פְּרִיחַת הַמַּטֶּה
טז וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יז דַּבֵּ֣ר ׀ אֶל־בְּנֵ֣י יִשְׂרָאֵ֗ל וְקַ֣ח מֵֽאִתָּ֡ם מַטֶּ֣ה מַטֶּה֩ לְבֵ֨ית אָ֜ב מֵאֵ֤ת כׇּל־נְשִֽׂיאֵהֶם֙ לְבֵ֣ית אֲבֹתָ֔ם שְׁנֵ֥ים עָשָׂ֖ר מַטּ֑וֹת אִ֣ישׁ אֶת־שְׁמ֔וֹ תִּכְתֹּ֖ב עַל־מַטֵּֽהוּ׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְקַח מֵהֶם מַקֵּל אֶחָד מִכָּל מִשְׁפָּחָה מִכָּל נְשִׂיאֵי מִשְׁפְּחוֹת אֲבוֹתֵיהֶם – שְׁנֵים עָשָׂר מַקְלוֹת, וְתִכְתּוֹב אֶת הַשֵּׁם שֶׁל כָּל אֶחָד עַל מַקְלוֹ.
יח וְאֵת֙ שֵׁ֣ם אַהֲרֹ֔ן תִּכְתֹּ֖ב עַל־מַטֵּ֣ה לֵוִ֑י כִּ֚י מַטֶּ֣ה אֶחָ֔ד לְרֹ֖אשׁ בֵּ֥ית אֲבוֹתָֽם׃ וְאֶת שֵׁם אַהֲרֹן תִּכְתּוֹב עַל הַמַּקֵּל שֶׁל לֵוִי, שֶׁיִּהְיֶה מַקֵּל אֶחָד לַמַּנְהִיג שֶׁל כָּל מִשְׁפְּחוֹת אֲבוֹתֵיהֶם.
יט וְהִנַּחְתָּ֖ם בְּאֹ֣הֶל מוֹעֵ֑ד לִפְנֵי֙ הָֽעֵד֔וּת אֲשֶׁ֛ר אִוָּעֵ֥ד לָכֶ֖ם שָֽׁמָּה׃ וְתָנִיחַ אוֹתָם בְּאוֹהֶל הַהִתְכַּנְּסוּת לִפְנֵי אֲרוֹן לוּחוֹת הַבְּרִית שֶׁאֲנִי נִפְגָּשׁ אִתְּכֶם שָׁם.
כ וְהָיָ֗ה הָאִ֛ישׁ אֲשֶׁ֥ר אֶבְחַר־בּ֖וֹ מַטֵּ֣הוּ יִפְרָ֑ח וַהֲשִׁכֹּתִ֣י מֵֽעָלַ֗י אֶת־תְּלֻנּוֹת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁ֛ר הֵ֥ם מַלִּינִ֖ם עֲלֵיכֶֽם׃ וְהָאִישׁ שֶׁאֶבְחַר בּוֹ הַמַּקֵּל שֶׁלּוֹ יִפְרַח, כָּךְ אַשְׁקִיט אֶת הַתְּלֻנּוֹת שֶׁל בְּנֵי יִשְׂרָאֵל עָלַי שֶׁהֵם מִתְלוֹנְנִים עֲלֵיכֶם.
כא וַיְדַבֵּ֨ר מֹשֶׁ֜ה אֶל־בְּנֵ֣י יִשְׂרָאֵ֗ל וַיִּתְּנ֣וּ אֵלָ֣יו ׀ כׇּֽל־נְשִׂיאֵיהֶ֡ם מַטֶּה֩ לְנָשִׂ֨יא אֶחָ֜ד מַטֶּ֨ה לְנָשִׂ֤יא אֶחָד֙ לְבֵ֣ית אֲבֹתָ֔ם שְׁנֵ֥ים עָשָׂ֖ר מַטּ֑וֹת וּמַטֵּ֥ה אַהֲרֹ֖ן בְּת֥וֹךְ מַטּוֹתָֽם׃ מֹשֶׁה דִּבֵּר אֶל בְּנֵי יִשְׂרָאֵל, וְכָל נְשִׂיאֵיהֶם נָתְנוּ לוֹ מַקֵּל עֲבוּר כָּל נָשִׂיא שֶׁל מִשְׁפְּחוֹת אֲבוֹתֵיהֶם – שְׁנֵים עָשָׂר מַקְלוֹת, וְהַמַּקֵּל שֶׁל אַהֲרֹן בֵּין מַקְלוֹתֵיהֶם.
כב וַיַּנַּ֥ח מֹשֶׁ֛ה אֶת־הַמַּטֹּ֖ת לִפְנֵ֣י יְהֹוָ֑ה בְּאֹ֖הֶל הָעֵדֻֽת׃ מֹשֶׁה הִנִּיחַ אֶת הַמַּקְלוֹת לִפְנֵי ה' בְּאוֹהֶל לוּחוֹת הַבְּרִית.
כג וַיְהִ֣י מִֽמׇּחֳרָ֗ת וַיָּבֹ֤א מֹשֶׁה֙ אֶל־אֹ֣הֶל הָעֵד֔וּת וְהִנֵּ֛ה פָּרַ֥ח מַטֵּֽה־אַהֲרֹ֖ן לְבֵ֣ית לֵוִ֑י וַיֹּ֤צֵֽא פֶ֙רַח֙ וַיָּ֣צֵֽץ צִ֔יץ וַיִּגְמֹ֖ל שְׁקֵדִֽים׃ לְמָחֳרָת, מֹשֶׁה בָּא לְאוֹהֶל לוּחוֹת הַבְּרִית וְהִנֵּה הַמַּקֵּל שֶׁל אַהֲרֹן מִמִּשְׁפַּחַת לֵוִי פָּרַח – הוֹצִיא פֶּרַח הֵנֵץ נִצָּנִים וְגִדֵּל שְׁקֵדִים.
כד וַיֹּצֵ֨א מֹשֶׁ֤ה אֶת־כׇּל־הַמַּטֹּת֙ מִלִּפְנֵ֣י יְהֹוָ֔ה אֶֽל־כׇּל־בְּנֵ֖י יִשְׂרָאֵ֑ל וַיִּרְא֥וּ וַיִּקְח֖וּ אִ֥ישׁ מַטֵּֽהוּ׃ מֹשֶׁה הוֹצִיא אֶת כָּל הַמַּקְלוֹת מִלִּפְנֵי ה' אֶל כָּל בְּנֵי יִשְׂרָאֵל, הֵם רָאוּ וְלָקְחוּ כָּל אֶחָד אֶת מַקְלוֹ.

שישי[עריכה]

פ
כה וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה הָשֵׁ֞ב אֶת־מַטֵּ֤ה אַהֲרֹן֙ לִפְנֵ֣י הָעֵד֔וּת לְמִשְׁמֶ֥רֶת לְא֖וֹת לִבְנֵי־מֶ֑רִי וּתְכַ֧ל תְּלוּנֹּתָ֛ם מֵעָלַ֖י וְלֹ֥א יָמֻֽתוּ׃ ה' אָמַר לְמֹשֶׁה: הָשֵׁב אֶת הַמַּקֵּל שֶׁל אַהֲרֹן אֶל לִפְנֵי לוּחוֹת הַבְּרִית לְסִימַן תֵּעוּד לְמַזְכֶּרֶת לְסַרְבָנִים, וְיִפָּסְקוּ הַתְּלוּנּוֹת שֶׁלָּהֶם עָלַי וְלֹא יָמוּתוּ.
כו וַיַּ֖עַשׂ מֹשֶׁ֑ה כַּאֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֹת֖וֹ כֵּ֥ן עָשָֽׂה׃ מֹשֶׁה בִּצֵּעַ, כְּמוֹ שֶׁה' צִוָּה אוֹתוֹ כָּךְ הוּא עָשָׂה.
פ פָּרָשַׁת מִשְׁמֶרֶת הַמִּקְדָּשׁ
כז וַיֹּֽאמְרוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֶל־מֹשֶׁ֖ה לֵאמֹ֑ר הֵ֥ן גָּוַ֛עְנוּ אָבַ֖דְנוּ כֻּלָּ֥נוּ אָבָֽדְנוּ׃ בְּנֵי יִשְׂרָאֵל אָמְרוּ לְמֹשֶׁה: הֲרֵי מַתְנוּ, אָבַדְנוּ, כֻּלָּנוּ נֹאבַד!
כח כֹּ֣ל הַקָּרֵ֧ב ׀ הַקָּרֵ֛ב אֶל־מִשְׁכַּ֥ן יְהֹוָ֖ה יָמ֑וּת הַאִ֥ם תַּ֖מְנוּ לִגְוֺֽעַ׃ כָּל מִי שֶׁנִּגָּשׁ לְמִשְׁכַּן ה' יָמוּת! הַאִם סִיַּמְנוּ לָמוּת?

יח א וַיֹּ֤אמֶר יְהֹוָה֙ אֶֽל־אַהֲרֹ֔ן אַתָּ֗ה וּבָנֶ֤יךָ וּבֵית־אָבִ֙יךָ֙ אִתָּ֔ךְ תִּשְׂא֖וּ אֶת־עֲוֺ֣ן הַמִּקְדָּ֑שׁ וְאַתָּה֙ וּבָנֶ֣יךָ אִתָּ֔ךְ תִּשְׂא֖וּ אֶת־עֲוֺ֥ן כְּהֻנַּתְכֶֽם׃ ה' אָמַר לְאַהֲרֹן, אַתָּה וְאִתְּךָ בָּנֶיךָ וּמִשְׁפַּחַת אָבִיךָ, תִּקְחוּ אֶת הָאַחֲרָיוּת עַל עֲווֹן חִלּוּל הַמִּקְדָּשׁ, וְאַתָּה וּבָנֶיךָ אִתְּךָ תִּקְחוּ אֶת הָאַחֲרָיוּת עַל עֲווֹן חִלּוּל כְּהֻנַּתְכֶם.
ב וְגַ֣ם אֶת־אַחֶ֩יךָ֩ מַטֵּ֨ה לֵוִ֜י שֵׁ֤בֶט אָבִ֙יךָ֙ הַקְרֵ֣ב אִתָּ֔ךְ וְיִלָּו֥וּ עָלֶ֖יךָ וִישָֽׁרְת֑וּךָ וְאַתָּה֙ וּבָנֶ֣יךָ אִתָּ֔ךְ לִפְנֵ֖י אֹ֥הֶל הָעֵדֻֽת׃ וְקָרֵב אֵלֶיךָ גַּם אֶת אַחֶיךָ שְׁאָר שֵׁבֶט לֵוִי שֵׁבֶט אָבִיךָ, וְיִצְטָרְפוּ אֵלֶיךָ וִישָׁרְתוּ אוֹתְךָ, וְאַתָּה וְאִתְּךָ בָּנֶיךָ תִּהְיוּ לִפְנֵי אוֹהֶל לוּחוֹת הַבְּרִית.
ג וְשָֽׁמְרוּ֙ מִֽשְׁמַרְתְּךָ֔ וּמִשְׁמֶ֖רֶת כׇּל־הָאֹ֑הֶל אַךְ֩ אֶל־כְּלֵ֨י הַקֹּ֤דֶשׁ וְאֶל־הַמִּזְבֵּ֙חַ֙ לֹ֣א יִקְרָ֔בוּ וְלֹֽא־יָמֻ֥תוּ גַם־הֵ֖ם גַּם־אַתֶּֽם׃ הֵם יִשְׁמְרוּ אֶת הַשְּׁמִירָה שֶׁמֻּטֶּלֶת עָלֶיךָ וְאֶת שְׁמִירַת כָּל הָאוֹהֶל, אֲבָל לֹא יִגְּשׁוּ אֶל כְּלֵי הַקּוֹדֶשׁ וְאֶל הַמִּזְבֵּחַ וְכָךְ לֹא יָמוּתוּ, לֹא הֵם וְלֹא אַתֶּם.
ד וְנִלְו֣וּ עָלֶ֔יךָ וְשָֽׁמְר֗וּ אֶת־מִשְׁמֶ֙רֶת֙ אֹ֣הֶל מוֹעֵ֔ד לְכֹ֖ל עֲבֹדַ֣ת הָאֹ֑הֶל וְזָ֖ר לֹא־יִקְרַ֥ב אֲלֵיכֶֽם׃ הֵם יִצְטָרְפוּ אֵלֶיךָ וְיִשְׁמְרוּ אֶת שְׁמִירַת אוֹהֶל הַהִתְכַּנְּסוּת וְכָל עֲבוֹדַת הָאוֹהֶל, וְאָדָם זָר לֹא יִגַּשׁ אֲלֵיכֶם.
ה וּשְׁמַרְתֶּ֗ם אֵ֚ת מִשְׁמֶ֣רֶת הַקֹּ֔דֶשׁ וְאֵ֖ת מִשְׁמֶ֣רֶת הַמִּזְבֵּ֑חַ וְלֹֽא־יִהְיֶ֥ה ע֛וֹד קֶ֖צֶף עַל־בְּנֵ֥י יִשְׂרָאֵֽל׃ תִּשְׁמְרוּ אֶת שְׁמִירַת הַקּוֹדֶשׁ וְאֶת שְׁמִירַת הַמִּזְבֵּחַ וְכָךְ לֹא יִהְיֶה עוֹד כַּעַס עַל בְּנֵי יִשְׂרָאֵל.
ו וַאֲנִ֗י הִנֵּ֤ה לָקַ֙חְתִּי֙ אֶת־אֲחֵיכֶ֣ם הַלְוִיִּ֔ם מִתּ֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל לָכֶ֞ם מַתָּנָ֤ה נְתֻנִים֙ לַֽיהֹוָ֔ה לַעֲבֹ֕ד אֶת־עֲבֹדַ֖ת אֹ֥הֶל מוֹעֵֽד׃ וַאֲנִי הִנֵּה לָקַחְתִּי אֶת אֲחֵיכֶם הַלְוִיִּים מִתּוֹךְ בְּנֵי יִשְׂרָאֵל, הֵם נְתוּנִים לָכֶם לְצוֹרֶךְ עֲבוֹדַת ה' לַעֲבוֹד אֶת עֲבוֹדַת אוֹהֶל הַהִתְכַּנְּסוּת.
ז וְאַתָּ֣ה וּבָנֶ֣יךָ אִ֠תְּךָ֠ תִּשְׁמְר֨וּ אֶת־כְּהֻנַּתְכֶ֜ם לְכׇל־דְּבַ֧ר הַמִּזְבֵּ֛חַ וּלְמִבֵּ֥ית לַפָּרֹ֖כֶת וַעֲבַדְתֶּ֑ם עֲבֹדַ֣ת מַתָּנָ֗ה אֶתֵּן֙ אֶת־כְּהֻנַּתְכֶ֔ם וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת׃ וְאַתָּה וּבָנֶיךָ אִתְּךָ תִּשְׁמְרוּ עַל קְדֻשַּׁת כְּהֻנַּתְכֶם בְּכָל עִנְיְנֵי הַמִּזְבֵּחַ וּפְנִימָה מֵהַפָּרוֹכֶת, וְתַעַבְדוּ אֶת עֲבוֹדַת כְּהֻנַּתְכֶם שֶׁהִיא מַתָּנָה שֶׁנָּתַתִּי לָכֶם, וְזָר שֶׁיִּגַּשׁ לַעֲבוֹד יוּמַת.
פ פָּרָשַׁת מַתְּנוֹת הַכְּהֻנָּה
ח וַיְדַבֵּ֣ר יְהֹוָה֮ אֶֽל־אַהֲרֹן֒ וַאֲנִי֙ הִנֵּ֣ה נָתַ֣תִּֽי לְךָ֔ אֶת־מִשְׁמֶ֖רֶת תְּרוּמֹתָ֑י לְכׇל־קׇדְשֵׁ֣י בְנֵֽי־יִ֠שְׂרָאֵ֠ל לְךָ֨ נְתַתִּ֧ים לְמׇשְׁחָ֛ה וּלְבָנֶ֖יךָ לְחׇק־עוֹלָֽם׃ ה' דִּבֵּר אֶל אַהֲרֹן: וַאֲנִי הִנֵּה נָתַתִּי לְךָ אֶת תְּרוּמוֹתַי שֶׁתִּשְׁמוֹר אוֹתָם בְּטַהֲרָה, כָּל תְּרוּמָה שֶׁל בְּנֵי יִשְׂרָאֵל לַקּוֹדֶשׁ, נָתַתִּי אוֹתָם לְךָ לְכַבֵּד אוֹתְךָ, וְגַם לְבָנֶיךָ – חוֹק לְעוֹלָם.
ט זֶֽה־יִהְיֶ֥ה לְךָ֛ מִקֹּ֥דֶשׁ הַקֳּדָשִׁ֖ים מִן־הָאֵ֑שׁ כׇּל־קׇ֠רְבָּנָ֠ם לְֽכׇל־מִנְחָתָ֞ם וּלְכׇל־חַטָּאתָ֗ם וּלְכׇל־אֲשָׁמָם֙ אֲשֶׁ֣ר יָשִׁ֣יבוּ לִ֔י קֹ֣דֶשׁ קׇֽדָשִׁ֥ים לְךָ֛ ה֖וּא וּלְבָנֶֽיךָ׃ זֶה יִהְיֶה לְךָ מִקּוֹדֶשׁ הַקֳּדָשִׁים מֵהָאֵשׁ: כָּל קָרְבְּנוֹתֵיהֶם מִנְחוֹתֵיהֶם חַטֹּאתֵיהֶם וְכָל אַשְׁמוֹתֵיהֶם וַהֲשָׁבַת גֵּזֶל הַגֵּר שֶׁנּוֹתְנִים לַה', הוּא קוֹדֶשׁ קָדָשִׁים לְךָ וּלְבָנֶיךָ.
י בְּקֹ֥דֶשׁ הַקֳּדָשִׁ֖ים תֹּאכְלֶ֑נּוּ כׇּל־זָכָר֙ יֹאכַ֣ל אֹת֔וֹ קֹ֖דֶשׁ יִֽהְיֶה־לָּֽךְ׃ תֹּאכַל אוֹתוֹ בְּתוֹךְ הַמִּשְׁכָּן הַמְּקֻדָּשׁ, כָּל זָכָר יֹאכַל אוֹתוֹ, וְהוּא יִהְיֶה לְךָ קוֹדֶשׁ.
יא וְזֶה־לְּךָ֞ תְּרוּמַ֣ת מַתָּנָ֗ם לְכׇל־תְּנוּפֹת֮ בְּנֵ֣י יִשְׂרָאֵל֒ לְךָ֣ נְתַתִּ֗ים וּלְבָנֶ֧יךָ וְלִבְנֹתֶ֛יךָ אִתְּךָ֖ לְחׇק־עוֹלָ֑ם כׇּל־טָה֥וֹר בְּבֵיתְךָ֖ יֹאכַ֥ל אֹתֽוֹ׃ וְזֶה יִהְיֶה שַׁיָּךְ לְךָ מֵהַמַּתָּנוֹת שֶׁהֵם מַפְרִישִׁים, מִכָּל הַתְּרוּמוֹת שֶׁל בְּנֵי יִשְׂרָאֵל – נָתַתִּי אוֹתָם לְךָ וּלְבָנֶיךָ וְלִבְנוֹתֶיךָ אִתְּךָ, חוֹק לְעוֹלָם, כָּל טָהוֹר בְּבֵיתְךָ יֹאכַל אוֹתוֹ:
יב כֹּ֚ל חֵ֣לֶב יִצְהָ֔ר וְכׇל־חֵ֖לֶב תִּיר֣וֹשׁ וְדָגָ֑ן רֵאשִׁיתָ֛ם אֲשֶׁר־יִתְּנ֥וּ לַֽיהֹוָ֖ה לְךָ֥ נְתַתִּֽים׃ הַחֵלֶק הַמְּשֻׁבָּח מִכָּל הַשֶּׁמֶן וְהַחֵלֶק הַמְּשֻׁבָּח מִכָּל הַיַּיִן וְהַתְּבוּאָה, אֶת הַחֵלֶק הָרִאשׁוֹן שֶׁנּוֹתְנִים לַה' – נָתַתִּי אוֹתָם לְךָ.
יג בִּכּוּרֵ֞י כׇּל־אֲשֶׁ֧ר בְּאַרְצָ֛ם אֲשֶׁר־יָבִ֥יאוּ לַיהֹוָ֖ה לְךָ֣ יִהְיֶ֑ה כׇּל־טָה֥וֹר בְּבֵיתְךָ֖ יֹאכְלֶֽנּוּ׃ הַבִּכּוּרִים מִכָּל הַפֵּרוֹת שֶׁבְּאַרְצָם שֶׁיָּבִיאוּ לַה' יִהְיוּ שֶׁלְּךָ, כָּל אָדָם טָהוֹר בְּבֵיתְךָ יֹאכַל אוֹתָם.
יד כׇּל־חֵ֥רֶם בְּיִשְׂרָאֵ֖ל לְךָ֥ יִהְיֶֽה׃ כָּל דָּבָר שֶׁהֻקְדַּשׁ בְּיִשְׂרָאֵל בִּלְשׁוֹן 'חֵרֶם' – יִהְיֶה שֶׁלְּךָ.
טו כׇּל־פֶּ֣טֶר רֶ֠חֶם לְֽכׇל־בָּשָׂ֞ר אֲשֶׁר־יַקְרִ֧יבוּ לַֽיהֹוָ֛ה בָּאָדָ֥ם וּבַבְּהֵמָ֖ה יִֽהְיֶה־לָּ֑ךְ אַ֣ךְ ׀ פָּדֹ֣ה תִפְדֶּ֗ה אֵ֚ת בְּכ֣וֹר הָֽאָדָ֔ם וְאֵ֛ת בְּכֽוֹר־הַבְּהֵמָ֥ה הַטְּמֵאָ֖ה תִּפְדֶּֽה׃ כָּל בְּכוֹר – שֶׁיּוֹצֵא רִאשׁוֹן מִפֶּתַח הָרֶחֶם שֶׁל כָּל חַי, בָּאָדָם וּבַבְּהֵמָה, שֶׁיֻּקְדַּשׁ לַה' – יִהְיֶה שֶׁלְּךָ, אֲבָל אֶת בְּכוֹר הָאָדָם תִּפְדֶּה, וְגַם אֶת בְּכוֹר הַבְּהֵמָה הַטְּמֵאָה תִּפְדֶּה.
טז וּפְדוּיָו֙ מִבֶּן־חֹ֣דֶשׁ תִּפְדֶּ֔ה בְּעֶ֨רְכְּךָ֔ כֶּ֛סֶף חֲמֵ֥שֶׁת שְׁקָלִ֖ים בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ עֶשְׂרִ֥ים גֵּרָ֖ה הֽוּא׃ תִּפְדֶּה נִפְדִּים אֵלֶּה מִגִּיל חוֹדֶשׁ, בְּכֶסֶף בְּמִשְׁקַל חֲמִשָּׁה שְׁקָלִים בְּשִׁקְלֵי הַקּוֹדֶשׁ (70.5 גְּרַם), מִשְׁקַל שֶׁקֶל הַקּוֹדֶשׁ הוּא עֶשְׂרִים גֵּרָה (14.1 גְּרַם).
יז אַ֣ךְ בְּֽכוֹר־שׁ֡וֹר אֽוֹ־בְכ֨וֹר כֶּ֜שֶׂב אֽוֹ־בְכ֥וֹר עֵ֛ז לֹ֥א תִפְדֶּ֖ה קֹ֣דֶשׁ הֵ֑ם אֶת־דָּמָ֞ם תִּזְרֹ֤ק עַל־הַמִּזְבֵּ֙חַ֙ וְאֶת־חֶלְבָּ֣ם תַּקְטִ֔יר אִשֶּׁ֛ה לְרֵ֥יחַ נִיחֹ֖חַ לַֽיהֹוָֽה׃ אֲבָל אַל תִּפְדֶּה בְּכוֹר שׁוֹר אוֹ בְּכוֹר כֶּבֶשׂ אוֹ בְּכוֹר עֵז כִּי הֵם קוֹדֶשׁ, אֶת דָּמָם תִּזְרוֹק עַל הַמִּזְבֵּחַ וְאֶת חֶלְבָּם תַּקְטִיר קָרְבָּן עוֹלֶה בָּאֵשׁ לְנַחַת רוּחַ לַה'.
יח וּבְשָׂרָ֖ם יִהְיֶה־לָּ֑ךְ כַּחֲזֵ֧ה הַתְּנוּפָ֛ה וּכְשׁ֥וֹק הַיָּמִ֖ין לְךָ֥ יִהְיֶֽה׃ וּבְשָׂרָם יִהְיֶה שֶׁלְּךָ, הוּא יִהְיֶה שֶׁלְּךָ כְּמוֹ הֶחָזֶה הַמּוּנָף וּכְמוֹ הַשּׁוֹק הַיְּמִינִי.
יט כֹּ֣ל ׀ תְּרוּמֹ֣ת הַקֳּדָשִׁ֗ים אֲשֶׁ֨ר יָרִ֥ימוּ בְנֵֽי־יִשְׂרָאֵל֮ לַֽיהֹוָה֒ נָתַ֣תִּֽי לְךָ֗ וּלְבָנֶ֧יךָ וְלִבְנֹתֶ֛יךָ אִתְּךָ֖ לְחׇק־עוֹלָ֑ם בְּרִית֩ מֶ֨לַח עוֹלָ֥ם הִוא֙ לִפְנֵ֣י יְהֹוָ֔ה לְךָ֖ וּלְזַרְעֲךָ֥ אִתָּֽךְ׃ נָתַתִּי לְךָ וּלְבָנֶיךָ וְלִבְנוֹתֶיךָ אִתְּךָ אֶת כָּל תְּרוּמוֹת הַקֳּדָשִׁים שֶׁבְּנֵי יִשְׂרָאֵל יַפְרִישׁוּ לַה', חוֹק לְעוֹלָם, זוֹ הִתְחַיְּבוּת לְעוֹלָם לִפְנֵי ה' כְּמוֹ עֲמִידוּת הַמֶּלַח, לְךָ וּלְצֶאֱצָאֶיךָ אִתְּךָ.
כ וַיֹּ֨אמֶר יְהֹוָ֜ה אֶֽל־אַהֲרֹ֗ן בְּאַרְצָם֙ לֹ֣א תִנְחָ֔ל וְחֵ֕לֶק לֹא־יִהְיֶ֥ה לְךָ֖ בְּתוֹכָ֑ם אֲנִ֤י חֶלְקְךָ֙ וְנַחֲלָ֣תְךָ֔ בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל׃ ה' אָמַר לְאַהֲרֹן: לֹא תִּנְחַל בְּאַרְצָם, וְלֹא יִהְיֶה לְךָ חֵלֶק בְּתוֹכָם, אֲנִי חֶלְקְךָ וְנַחֲלָתְךָ בְּתוֹךְ בְּנֵי יִשְׂרָאֵל.

שביעי[עריכה]

ס פָּרָשַׁת הַמַּעֲשֵׂר
כא וְלִבְנֵ֣י לֵוִ֔י הִנֵּ֥ה נָתַ֛תִּי כׇּל־מַֽעֲשֵׂ֥ר בְּיִשְׂרָאֵ֖ל לְנַחֲלָ֑ה חֵ֤לֶף עֲבֹֽדָתָם֙ אֲשֶׁר־הֵ֣ם עֹֽבְדִ֔ים אֶת־עֲבֹדַ֖ת אֹ֥הֶל מוֹעֵֽד׃ וְהִנֵּה נָתַתִּי לִבְנֵי לֵוִי לְנַחֲלָה אֶת כָּל הַמַּעַשְׂרוֹת בְּיִשְׂרָאֵל תְּמוּרַת עֲבוֹדָתָם שֶׁהֵם עוֹבְדִים בַּעֲבוֹדַת אוֹהֶל הַהִתְכַּנְּסוּת.
כב וְלֹא־יִקְרְב֥וּ ע֛וֹד בְּנֵ֥י יִשְׂרָאֵ֖ל אֶל־אֹ֣הֶל מוֹעֵ֑ד לָשֵׂ֥את חֵ֖טְא לָמֽוּת׃ וּבְנֵי יִשְׂרָאֵל לֹא יִגְּשׁוּ עוֹד לְאוֹהֶל הַהִתְכַּנְּסוּת וְלֹא יְקַבְּלוּ עוֹנֶשׁ עַל חֶטְאָם וְיָמוּתוּ.
כג וְעָבַ֨ד הַלֵּוִ֜י ה֗וּא אֶת־עֲבֹדַת֙ אֹ֣הֶל מוֹעֵ֔ד וְהֵ֖ם יִשְׂא֣וּ עֲוֺנָ֑ם חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם וּבְתוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לֹ֥א יִנְחֲל֖וּ נַחֲלָֽה׃ וְהַלְּוִיִּים יַעַבְדוּ אֶת עֲבוֹדַת אוֹהֶל הַהִתְכַּנְּסוּת, וְהֵם יִקְחוּ אַחֲרָיוּת עַל עֲווֹן חִלּוּלוֹ, חוֹק לְעוֹלָם לְדוֹרוֹת, וְלֹא יִנְחֲלוּ נַחֲלָה עִם בְּנֵי יִשְׂרָאֵל.
כד כִּ֞י אֶת־מַעְשַׂ֣ר בְּנֵֽי־יִשְׂרָאֵ֗ל אֲשֶׁ֨ר יָרִ֤ימוּ לַֽיהֹוָה֙ תְּרוּמָ֔ה נָתַ֥תִּי לַלְוִיִּ֖ם לְנַחֲלָ֑ה עַל־כֵּן֙ אָמַ֣רְתִּי לָהֶ֔ם בְּתוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לֹ֥א יִנְחֲל֖וּ נַחֲלָֽה׃ כִּי אֶת הַמַּעֲשֵׂר שֶׁל בְּנֵי יִשְׂרָאֵל שֶׁהֵם יַפְרִישׁוּ כִּתְרוּמָה לַה' – נָתַתִּי לַלְוִיִּים לְנַחֲלָה, לָכֵן אָמַרְתִּי עֲלֵיהֶם שֶׁלֹּא יִנְחֲלוּ נַחֲלָה עִם בְּנֵי יִשְׂרָאֵל.
פ פָּרָשַׁת תְּרוּמַת הַמַּעֲשֵׂר
כה וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
כו וְאֶל־הַלְוִיִּ֣ם תְּדַבֵּר֮ וְאָמַרְתָּ֣ אֲלֵהֶם֒ כִּֽי־תִ֠קְח֠וּ מֵאֵ֨ת בְּנֵֽי־יִשְׂרָאֵ֜ל אֶת־הַֽמַּעֲשֵׂ֗ר אֲשֶׁ֨ר נָתַ֧תִּי לָכֶ֛ם מֵאִתָּ֖ם בְּנַחֲלַתְכֶ֑ם וַהֲרֵמֹתֶ֤ם מִמֶּ֙נּוּ֙ תְּרוּמַ֣ת יְהֹוָ֔ה מַעֲשֵׂ֖ר מִן־הַֽמַּעֲשֵֽׂר׃ תְּדַבֵּר אֶל הַלְוִיִּים וְתֹאמַר לָהֶם: כְּשֶׁתִּקְחוּ מִבְּנֵי יִשְׂרָאֵל אֶת הַמַּעֲשֵׂר שֶׁנָּתַתִּי לָכֶם מֵהֶם כְּנַחֲלָה, תַּפְרִישׁוּ מִמֶּנּוּ תְּרוּמָה לַה' – מַעֲשֵׂר מֵהַמַּעֲשֵׂר.
כז וְנֶחְשַׁ֥ב לָכֶ֖ם תְּרוּמַתְכֶ֑ם כַּדָּגָן֙ מִן־הַגֹּ֔רֶן וְכַֽמְלֵאָ֖ה מִן־הַיָּֽקֶב׃ תְּרוּמַתְכֶם זוֹ תֵּחָשֵׁב לָכֶם כְּמוֹ תְּרוּמַת הַתְּבוּאָה מֵהַגּוֹרֶן וּבִכּוּרִים מֵהַיֶּקֶב.
כח כֵּ֣ן תָּרִ֤ימוּ גַם־אַתֶּם֙ תְּרוּמַ֣ת יְהֹוָ֔ה מִכֹּל֙ מַעְשְׂרֹ֣תֵיכֶ֔ם אֲשֶׁ֣ר תִּקְח֔וּ מֵאֵ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל וּנְתַתֶּ֤ם מִמֶּ֙נּוּ֙ אֶת־תְּרוּמַ֣ת יְהֹוָ֔ה לְאַהֲרֹ֖ן הַכֹּהֵֽן׃ כָּךְ תִּתְרְמוּ גַּם אַתֶּם תְּרוּמַת ה' מִכָּל מַעְשְׂרוֹתֵיכֶם שֶׁתִּקְחוּ מֵאֵת בְּנֵי יִשְׂרָאֵל, וְתִתְּנוּ מִמֶּנּוּ תְּרוּמָה לַה' – לְאַהֲרֹן הַכֹּהֵן.
כט מִכֹּל֙ מַתְּנֹ֣תֵיכֶ֔ם תָּרִ֕ימוּ אֵ֖ת כׇּל־תְּרוּמַ֣ת יְהֹוָ֑ה מִכׇּ֨ל־חֶלְבּ֔וֹ אֶֽת־מִקְדְּשׁ֖וֹ מִמֶּֽנּוּ׃ תִּתְרְמוּ מִכָּל מַתְּנוֹתֵיכֶם אֶת כָּל תְּרוּמַת ה', מֵהַחֵלֶק הַמְּשֻׁבָּח שֶׁבּוֹ – תְּקַדְּשׁוּ וְתִקְחוּ מִמֶּנּוּ.

מפטיר[עריכה]

ל וְאָמַרְתָּ֖ אֲלֵהֶ֑ם בַּהֲרִֽימְכֶ֤ם אֶת־חֶלְבּוֹ֙ מִמֶּ֔נּוּ וְנֶחְשַׁב֙ לַלְוִיִּ֔ם כִּתְבוּאַ֥ת גֹּ֖רֶן וְכִתְבוּאַ֥ת יָֽקֶב׃ וְתֹאמַר לַלְוִיִּים: כְּשֶׁתַּפְרִישׁוּ אֶת הַמְּשֻׁבָּח שֶׁבּוֹ, הַשְּׁאָר יֵחָשֵׁב לַלְוִיִּים חוּלִין כִּתְבוּאַת הַגּוֹרֶן וְכִתְבוּאַת הַיֶּקֶב.
לא וַאֲכַלְתֶּ֤ם אֹתוֹ֙ בְּכׇל־מָק֔וֹם אַתֶּ֖ם וּבֵֽיתְכֶ֑ם כִּֽי־שָׂכָ֥ר הוּא֙ לָכֶ֔ם חֵ֥לֶף עֲבֹֽדַתְכֶ֖ם בְּאֹ֥הֶל מוֹעֵֽד׃ וְתֹאכְלוּ אוֹתוֹ בְּכָל מָקוֹם אַתֶּם וּמִשְׁפַּחְתְּכֶם, כִּי הוּא שָׂכָר עֲבוּרְכֶם תְּמוּרַת עֲבוֹדַתְכֶם בְּאוֹהֶל הַהִתְכַּנְּסוּת.
לב וְלֹֽא־תִשְׂא֤וּ עָלָיו֙ חֵ֔טְא בַּהֲרִֽימְכֶ֥ם אֶת־חֶלְבּ֖וֹ מִמֶּ֑נּוּ וְאֶת־קׇדְשֵׁ֧י בְנֵי־יִשְׂרָאֵ֛ל לֹ֥א תְחַלְּל֖וּ וְלֹ֥א תָמֽוּתוּ׃ וּבִזְכוּת שֶׁתַּפְרִישׁוּ מִמֶּנּוּ אֶת הַמְּשֻׁבָּח שֶׁבּוֹ לֹא תְּקַבְּלוּ עָלָיו עוֹנֶשׁ חֵטְא, וְכָךְ לֹא תְּחַלְּלוּ אֶת קָדְשֵׁי בְּנֵי יִשְׂרָאֵל וְלֹא תָּמוּתוּ.
