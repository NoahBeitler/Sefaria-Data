ראשון[עריכה]

פ פָּרָשַׁת הַכְּנִיסָה לַקּוֹדֶשׁ
טז א וַיְדַבֵּ֤ר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה אַחֲרֵ֣י מ֔וֹת שְׁנֵ֖י בְּנֵ֣י אַהֲרֹ֑ן בְּקׇרְבָתָ֥ם לִפְנֵי־יְהֹוָ֖ה וַיָּמֻֽתוּ׃ ה' דִּבֵּר אֶל מֹשֶׁה אַחֲרֵי שֶׁמֵּתוּ שְׁנֵי בְּנֵי אַהֲרֹן, כְּשֶׁנִּגְּשׁוּ לִפְנֵי ה' וּמֵתוּ.
ב וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה דַּבֵּר֮ אֶל־אַהֲרֹ֣ן אָחִ֒יךָ֒ וְאַל־יָבֹ֤א בְכׇל־עֵת֙ אֶל־הַקֹּ֔דֶשׁ מִבֵּ֖ית לַפָּרֹ֑כֶת אֶל־פְּנֵ֨י הַכַּפֹּ֜רֶת אֲשֶׁ֤ר עַל־הָאָרֹן֙ וְלֹ֣א יָמ֔וּת כִּ֚י בֶּֽעָנָ֔ן אֵרָאֶ֖ה עַל־הַכַּפֹּֽרֶת׃ ה' אָמַר לְמֹשֶׁה: דַּבֵּר אֶל אַהֲרֹן אָחִיךָ שֶׁלֹּא יָבוֹא בְּכָל זְמָן שֶׁיִּרְצֶה אֶל הַקּוֹדֶשׁ שֶׁמֵּעֵבֶר לַפָּרוֹכֶת מִבִּפְנִים – אֶל מוּל הַכִּסּוּי שֶׁעַל הָאָרוֹן, וְכָךְ לֹא יָמוּת, כִּי אֲנִי מִתְגַּלֶּה בֶּעָנָן שֶׁעַל כִּסּוּי הָאָרוֹן.
ג בְּזֹ֛את יָבֹ֥א אַהֲרֹ֖ן אֶל־הַקֹּ֑דֶשׁ בְּפַ֧ר בֶּן־בָּקָ֛ר לְחַטָּ֖את וְאַ֥יִל לְעֹלָֽה׃ עִם קָרְבָּנוֹת אֵלֶּה אַהֲרֹן יִכָּנֵס לַקּוֹדֶשׁ: עִם פַּר לְחַטָּאת וְאַיִל לְעוֹלָה.
ד כְּתֹֽנֶת־בַּ֨ד קֹ֜דֶשׁ יִלְבָּ֗שׁ וּמִֽכְנְסֵי־בַד֮ יִהְי֣וּ עַל־בְּשָׂרוֹ֒ וּבְאַבְנֵ֥ט בַּד֙ יַחְגֹּ֔ר וּבְמִצְנֶ֥פֶת בַּ֖ד יִצְנֹ֑ף בִּגְדֵי־קֹ֣דֶשׁ הֵ֔ם וְרָחַ֥ץ בַּמַּ֛יִם אֶת־בְּשָׂר֖וֹ וּלְבֵשָֽׁם׃ הוּא יִלְבַּשׁ כְּתוֹנֶת קְדוֹשָׁה מִבַּד, וְיִהְיוּ עַל גּוּפוֹ מִכְנְסֵי בַּד, וְיַחְגּוֹר חֲגוֹרָה מִבַּד, וְיַחֲבוֹשׁ לְרֹאשׁוֹ מִצְנֶפֶת בַּד – אֵלֶּה הֵם בִּגְדֵי קוֹדֶשׁ, הוּא יִטְבּוֹל בְּמַיִם בְּגוּפוֹ, וְיִלְבַּשׁ אוֹתָם.
ה וּמֵאֵ֗ת עֲדַת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל יִקַּ֛ח שְׁנֵֽי־שְׂעִירֵ֥י עִזִּ֖ים לְחַטָּ֑את וְאַ֥יִל אֶחָ֖ד לְעֹלָֽה׃ וְיִקַּח מִקְּהַל בְּנֵי יִשְׂרָאֵל שְׁנֵי תְּיָשִׁים לְחַטָּאת וְאַיִל אֶחָד לְעוֹלָה.
ו וְהִקְרִ֧יב אַהֲרֹ֛ן אֶת־פַּ֥ר הַחַטָּ֖את אֲשֶׁר־ל֑וֹ וְכִפֶּ֥ר בַּעֲד֖וֹ וּבְעַ֥ד בֵּיתֽוֹ׃ אַהֲרֹן יַגִּישׁ אֶת פַּר הַחַטָּאת שֶׁלּוֹ וִיכַפֵּר עַל עַצְמוֹ וְעַל מִשְׁפַּחְתּוֹ.
ז וְלָקַ֖ח אֶת־שְׁנֵ֣י הַשְּׂעִירִ֑ם וְהֶעֱמִ֤יד אֹתָם֙ לִפְנֵ֣י יְהֹוָ֔ה פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ הוּא יִקַּח אֶת שְׁנֵי הַתְּיָשִׁים וְיַעֲמִיד אוֹתָם לִפְנֵי ה' – מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.
ח וְנָתַ֧ן אַהֲרֹ֛ן עַל־שְׁנֵ֥י הַשְּׂעִירִ֖ם גֹּרָל֑וֹת גּוֹרָ֤ל אֶחָד֙ לַיהֹוָ֔ה וְגוֹרָ֥ל אֶחָ֖ד לַעֲזָאזֵֽל׃ אַהֲרֹן יַטִּיל גּוֹרָלוֹת בֵּין שְׁנֵי הַתְּיָשִׁים, עַל גּוֹרָל אֶחָד כָּתוּב: "לַה'", וְעַל גּוֹרָל אֶחָד כָּתוּב: "לַצּוּק הַתָּלוּל".
ט וְהִקְרִ֤יב אַהֲרֹן֙ אֶת־הַשָּׂעִ֔יר אֲשֶׁ֨ר עָלָ֥ה עָלָ֛יו הַגּוֹרָ֖ל לַיהֹוָ֑ה וְעָשָׂ֖הוּ חַטָּֽאת׃ אַהֲרֹן יַגִּישׁ אֶת הַתַּיִשׁ שֶׁעָלָה עָלָיו הַגּוֹרָל "לַה'" וְיַעֲשֶׂה אוֹתוֹ חַטָּאת.
י וְהַשָּׂעִ֗יר אֲשֶׁר֩ עָלָ֨ה עָלָ֤יו הַגּוֹרָל֙ לַעֲזָאזֵ֔ל יׇֽעֳמַד־חַ֛י לִפְנֵ֥י יְהֹוָ֖ה לְכַפֵּ֣ר עָלָ֑יו לְשַׁלַּ֥ח אֹת֛וֹ לַעֲזָאזֵ֖ל הַמִּדְבָּֽרָה׃ וְאֶת הַתַּיִשׁ שֶׁעָלָה עָלָיו הַגּוֹרָל "לַצּוּק הַתָּלוּל" יַעֲמִידוּ חַי לִפְנֵי ה' כְּדֵי לְהִתְוַדּוֹת עָלָיו לְכַפָּרָה, וְלִשְׁלוֹחַ אוֹתוֹ לַצּוּק הַתָּלוּל שֶׁבַּמִּדְבָּר.
יא וְהִקְרִ֨יב אַהֲרֹ֜ן אֶת־פַּ֤ר הַֽחַטָּאת֙ אֲשֶׁר־ל֔וֹ וְכִפֶּ֥ר בַּֽעֲד֖וֹ וּבְעַ֣ד בֵּית֑וֹ וְשָׁחַ֛ט אֶת־פַּ֥ר הַֽחַטָּ֖את אֲשֶׁר־לֽוֹ׃ אַהֲרֹן יַגִּישׁ אֶת פַּר הַחַטָּאת שֶׁלּוֹ וִיכַפֵּר עַל עַצְמוֹ וְעַל מִשְׁפַּחְתּוֹ וְיִשְׁחַט אֶת פַּר הַחַטָּאת שֶׁלּוֹ.
יב וְלָקַ֣ח מְלֹֽא־הַ֠מַּחְתָּ֠ה גַּֽחֲלֵי־אֵ֞שׁ מֵעַ֤ל הַמִּזְבֵּ֙חַ֙ מִלִּפְנֵ֣י יְהֹוָ֔ה וּמְלֹ֣א חׇפְנָ֔יו קְטֹ֥רֶת סַמִּ֖ים דַּקָּ֑ה וְהֵבִ֖יא מִבֵּ֥ית לַפָּרֹֽכֶת׃ וְיִקַּח מַחְתָּה מְלֵאָה שֶׁל גֶּחָלִים בּוֹעֲרוֹת מֵעַל הַמִּזְבֵּחַ מִלִּפְנֵי ה', וּמְלֹא חָפְנָיו קְטוֹרֶת בְּשָׂמִים דַּקָּה, וְיָבִיא פְּנִימָה אֶל מֵעֵבֶר לַפָּרוֹכֶת.
יג וְנָתַ֧ן אֶֽת־הַקְּטֹ֛רֶת עַל־הָאֵ֖שׁ לִפְנֵ֣י יְהֹוָ֑ה וְכִסָּ֣ה ׀ עֲנַ֣ן הַקְּטֹ֗רֶת אֶת־הַכַּפֹּ֛רֶת אֲשֶׁ֥ר עַל־הָעֵד֖וּת וְלֹ֥א יָמֽוּת׃ הוּא יָשִׂים אֶת הַקְּטוֹרֶת עַל הָאֵשׁ לִפְנֵי ה', וַעֲנַן הַקְּטוֹרֶת יְכַסֶּה אֶת הַכִּסּוּי שֶׁעַל אֲרוֹן לוּחוֹת הַבְּרִית, וְהוּא לֹא יָמוּת.
יד וְלָקַח֙ מִדַּ֣ם הַפָּ֔ר וְהִזָּ֧ה בְאֶצְבָּע֛וֹ עַל־פְּנֵ֥י הַכַּפֹּ֖רֶת קֵ֑דְמָה וְלִפְנֵ֣י הַכַּפֹּ֗רֶת יַזֶּ֧ה שֶֽׁבַע־פְּעָמִ֛ים מִן־הַדָּ֖ם בְּאֶצְבָּעֽוֹ׃ הוּא יִקַּח מִדַּם הַפָּר וְיַתִּיז בְּאֶצְבָּעוֹ אֶל מֵעַל הַכִּסּוּי מִמִּזְרָח, וּלְכִוּוּן הַכִּסּוּי יַתִּיז בְּאֶצְבָּעוֹ מֵהַדָּם שֶׁבַע פְּעָמִים.
טו וְשָׁחַ֞ט אֶת־שְׂעִ֤יר הַֽחַטָּאת֙ אֲשֶׁ֣ר לָעָ֔ם וְהֵבִיא֙ אֶת־דָּמ֔וֹ אֶל־מִבֵּ֖ית לַפָּרֹ֑כֶת וְעָשָׂ֣ה אֶת־דָּמ֗וֹ כַּאֲשֶׁ֤ר עָשָׂה֙ לְדַ֣ם הַפָּ֔ר וְהִזָּ֥ה אֹת֛וֹ עַל־הַכַּפֹּ֖רֶת וְלִפְנֵ֥י הַכַּפֹּֽרֶת׃ הוּא יִשְׁחַט אֶת תַּיִשׁ הַחַטָּאת שֶׁל הָעָם, וְיָבִיא אֶת דָּמוֹ פְּנִימָה אֶל מֵעֵבֶר לַפָּרוֹכֶת, וְיַעֲשֶׂה בְּדָמוֹ כְּמוֹ שֶׁעָשָׂה בְּדַם הַפָּר וְיַתִּיז אוֹתוֹ מֵעַל הַכִּסּוּי וּלְכִוּוּן הַכִּסּוּי.
טז וְכִפֶּ֣ר עַל־הַקֹּ֗דֶשׁ מִטֻּמְאֹת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וּמִפִּשְׁעֵיהֶ֖ם לְכׇל־חַטֹּאתָ֑ם וְכֵ֤ן יַעֲשֶׂה֙ לְאֹ֣הֶל מוֹעֵ֔ד הַשֹּׁכֵ֣ן אִתָּ֔ם בְּת֖וֹךְ טֻמְאֹתָֽם׃ הוּא יַתִּיז בִּמְקוֹם הַקּוֹדֶשׁ כְּדֵי לְכַפֵּר עַל טֻמְאוֹת בְּנֵי יִשְׂרָאֵל וְעַל פִּשְׁעֵיהֶם וְכָל חֲטָאֵיהֶם, וְיַתִּיז כָּךְ גַּם בְּאוֹהֶל הַהִתְכַּנְּסוּת שֶׁשּׁוֹכֵן אִתָּם בְּתוֹךְ טֻמְאוֹתֵיהֶם.
יז וְכׇל־אָדָ֞ם לֹא־יִהְיֶ֣ה ׀ בְּאֹ֣הֶל מוֹעֵ֗ד בְּבֹא֛וֹ לְכַפֵּ֥ר בַּקֹּ֖דֶשׁ עַד־צֵאת֑וֹ וְכִפֶּ֤ר בַּעֲדוֹ֙ וּבְעַ֣ד בֵּית֔וֹ וּבְעַ֖ד כׇּל־קְהַ֥ל יִשְׂרָאֵֽל׃ וְשׁוּם אָדָם לֹא יִהְיֶה בְּאוֹהֶל הַהִתְכַּנְּסוּת כְּשֶׁיִכָּנֵס לַקּוֹדֶשׁ לְכַפֵּר עַד שֶׁיֵּצֵא, וְהוּא יְכַפֵּר עַל עַצְמוֹ, וְעַל מִשְׁפַּחְתּוֹ, וְעַל כָּל קְהַל יִשְׂרָאֵל.

שני[עריכה]

יח וְיָצָ֗א אֶל־הַמִּזְבֵּ֛חַ אֲשֶׁ֥ר לִפְנֵֽי־יְהֹוָ֖ה וְכִפֶּ֣ר עָלָ֑יו וְלָקַ֞ח מִדַּ֤ם הַפָּר֙ וּמִדַּ֣ם הַשָּׂעִ֔יר וְנָתַ֛ן עַל־קַרְנ֥וֹת הַמִּזְבֵּ֖חַ סָבִֽיב׃ הוּא יֵצֵא לַמִּזְבֵּחַ שֶׁלִּפְנֵי ה' וְיַתִּיז עָלָיו כְּדֵי לְכַפֵּר, וְיִקַּח מִדַּם הַפָּר וּמִדַּם הַתַּיִשׁ, וְיָשִׂים עַל פִּנּוֹת הַמִּזְבֵּחַ מִסָּבִיב.
יט וְהִזָּ֨ה עָלָ֧יו מִן־הַדָּ֛ם בְּאֶצְבָּע֖וֹ שֶׁ֣בַע פְּעָמִ֑ים וְטִהֲר֣וֹ וְקִדְּשׁ֔וֹ מִטֻּמְאֹ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ הוּא יַתִּיז עַל הַמִּזְבֵּחַ מֵהַדָּם בְּאֶצְבָּעוֹ שֶׁבַע פְּעָמִים, וִיטַהֵר אוֹתוֹ וִיקַדֵּשׁ אוֹתוֹ מִטֻּמְאוֹת בְּנֵי יִשְׂרָאֵל.
כ וְכִלָּה֙ מִכַּפֵּ֣ר אֶת־הַקֹּ֔דֶשׁ וְאֶת־אֹ֥הֶל מוֹעֵ֖ד וְאֶת־הַמִּזְבֵּ֑חַ וְהִקְרִ֖יב אֶת־הַשָּׂעִ֥יר הֶחָֽי׃ כְּשֶׁיִּגְמוֹר לְהַתִּיז עַל הַקּוֹדֶשׁ וְעַל אוֹהֶל הַהִתְכַּנְּסוּת וְעַל הַמִּזְבֵּחַ כְּדֵי לְכַפֵּר, יַקְרִיב אֶת הַתַּיִשׁ הַחַי.
כא וְסָמַ֨ךְ אַהֲרֹ֜ן אֶת־שְׁתֵּ֣י יָדָ֗ו עַ֣ל רֹ֣אשׁ הַשָּׂעִיר֮ הַחַי֒ וְהִתְוַדָּ֣ה עָלָ֗יו אֶת־כׇּל־עֲוֺנֹת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וְאֶת־כׇּל־פִּשְׁעֵיהֶ֖ם לְכׇל־חַטֹּאתָ֑ם וְנָתַ֤ן אֹתָם֙ עַל־רֹ֣אשׁ הַשָּׂעִ֔יר וְשִׁלַּ֛ח בְּיַד־אִ֥ישׁ עִתִּ֖י הַמִּדְבָּֽרָה׃ אַהֲרֹן יַשְׁעִין אֶת שְׁתֵּי יָדָיו עַל רֹאשׁ הַתַּיִשׁ הַחַי וְיִתְוַדֶּה עָלָיו אֶת כָּל עֲוֹנוֹת בְּנֵי יִשְׂרָאֵל וְאֶת כָּל פִּשְׁעֵיהֶם וְכָל חֲטָאֵיהֶם, וְיָשִׂים אוֹתָם עַל רֹאשׁ הַתַּיִשׁ וִיגָרֵשׁ אוֹתוֹ לַמִּדְבָּר בְּיַד אִישׁ שֶׁמּוּנָה לְכָךְ.
כב וְנָשָׂ֨א הַשָּׂעִ֥יר עָלָ֛יו אֶת־כׇּל־עֲוֺנֹתָ֖ם אֶל־אֶ֣רֶץ גְּזֵרָ֑ה וְשִׁלַּ֥ח אֶת־הַשָּׂעִ֖יר בַּמִּדְבָּֽר׃ הַתַּיִשׁ יִסְחוֹב עָלָיו אֶת כָּל עֲוֹנוֹתֵיהֶם אֶל אֶרֶץ שְׁמָמָה, וְהוּא יַשְׁלִיךְ אֶת הַתַּיִשׁ בַּמִּדְבָּר.
כג וּבָ֤א אַהֲרֹן֙ אֶל־אֹ֣הֶל מוֹעֵ֔ד וּפָשַׁט֙ אֶת־בִּגְדֵ֣י הַבָּ֔ד אֲשֶׁ֥ר לָבַ֖שׁ בְּבֹא֣וֹ אֶל־הַקֹּ֑דֶשׁ וְהִנִּיחָ֖ם שָֽׁם׃ אַהֲרֹן יָבוֹא לְאוֹהֶל הַהִתְכַּנְּסוּת וְיִפְשׁוֹט אֶת בִּגְדֵי הַבָּד שֶׁלָּבַשׁ בְּבוֹאוֹ לַקּוֹדֶשׁ וְיַנִּיחַ אוֹתָם שָׁם.
כד וְרָחַ֨ץ אֶת־בְּשָׂר֤וֹ בַמַּ֙יִם֙ בְּמָק֣וֹם קָד֔וֹשׁ וְלָבַ֖שׁ אֶת־בְּגָדָ֑יו וְיָצָ֗א וְעָשָׂ֤ה אֶת־עֹֽלָתוֹ֙ וְאֶת־עֹלַ֣ת הָעָ֔ם וְכִפֶּ֥ר בַּעֲד֖וֹ וּבְעַ֥ד הָעָֽם׃ הוּא יִטְבּוֹל בְּגוּפוֹ בְּמַיִם בְּמָקוֹם קָדוֹשׁ וְיִלְבַּשׁ אֶת בְּגָדָיו, וְיֵצֵא וְיַעֲשֶׂה אֶת עוֹלָתוֹ וְאֶת עוֹלַת הָעָם וִיכַפֵּר עַל עַצְמוֹ וְעַל הָעָם.

שלישי (שני במחוברין)[עריכה]

כה וְאֵ֛ת חֵ֥לֶב הַֽחַטָּ֖את יַקְטִ֥יר הַמִּזְבֵּֽחָה׃ וְאֶת חֵלֶב הַחַטָּאת יַקְטִיר עַל הַמִּזְבֵּחַ.
כו וְהַֽמְשַׁלֵּ֤חַ אֶת־הַשָּׂעִיר֙ לַֽעֲזָאזֵ֔ל יְכַבֵּ֣ס בְּגָדָ֔יו וְרָחַ֥ץ אֶת־בְּשָׂר֖וֹ בַּמָּ֑יִם וְאַחֲרֵי־כֵ֖ן יָב֥וֹא אֶל־הַֽמַּחֲנֶֽה׃ וְהָאִישׁ שֶׁמַּשְׁלִיךְ אֶת הַתַּיִשׁ לַעֲזָאזֵל יַטְבִּיל אֶת בְּגָדָיו וְיַטְבִּיל אֶת גּוּפוֹ בְּמַיִם וְאַחֲרֵי כֵן יוּכַל לָבוֹא לַמַּחֲנֶה.
כז וְאֵת֩ פַּ֨ר הַֽחַטָּ֜את וְאֵ֣ת ׀ שְׂעִ֣יר הַֽחַטָּ֗את אֲשֶׁ֨ר הוּבָ֤א אֶת־דָּמָם֙ לְכַפֵּ֣ר בַּקֹּ֔דֶשׁ יוֹצִ֖יא אֶל־מִח֣וּץ לַֽמַּחֲנֶ֑ה וְשָׂרְפ֣וּ בָאֵ֔שׁ אֶת־עֹרֹתָ֥ם וְאֶת־בְּשָׂרָ֖ם וְאֶת־פִּרְשָֽׁם׃ וְאֶת פַּר הַחַטָּאת וְאֶת תַּיִשׁ הַחַטָּאת שֶׁדָּמָם הוּבָא לְכַפֵּר בַּקּוֹדֶשׁ יוֹצִיא אֶל מִחוּץ לַמַּחֲנֶה, וְיִשְׂרְפוּ בָּאֵשׁ אֶת הָעוֹרוֹת וְהַבָּשָׂר שֶׁלָּהֶם וְאֶת הָאוֹכֶל הַמְּעֻכָּל שֶׁבָּהֶם.
כח וְהַשֹּׂרֵ֣ף אֹתָ֔ם יְכַבֵּ֣ס בְּגָדָ֔יו וְרָחַ֥ץ אֶת־בְּשָׂר֖וֹ בַּמָּ֑יִם וְאַחֲרֵי־כֵ֖ן יָב֥וֹא אֶל־הַֽמַּחֲנֶֽה׃ וְהָאִישׁ שֶׁשּׂוֹרֵף אוֹתָם יַטְבִּיל אֶת בְּגָדָיו וְיַטְבִּיל אֶת גּוּפוֹ בְּמַיִם, וְאַחֲרֵי כֵן יוּכַל לָבוֹא לַמַּחֲנֶה.
כט וְהָיְתָ֥ה לָכֶ֖ם לְחֻקַּ֣ת עוֹלָ֑ם בַּחֹ֣דֶשׁ הַ֠שְּׁבִיעִ֠י בֶּֽעָשׂ֨וֹר לַחֹ֜דֶשׁ תְּעַנּ֣וּ אֶת־נַפְשֹֽׁתֵיכֶ֗ם וְכׇל־מְלָאכָה֙ לֹ֣א תַעֲשׂ֔וּ הָֽאֶזְרָ֔ח וְהַגֵּ֖ר הַגָּ֥ר בְּתוֹכְכֶֽם׃ זֶה יִהְיֶה לָכֶם חוֹק לְעוֹלָם, בַּחוֹדֶשׁ הַשְּׁבִיעִי בַּיּוֹם הָעֲשִׂירִי לַחוֹדֶשׁ תְּעַנּוּ אֶת נַפְשׁוֹתֵיכֶם בְּצוֹם, וְאַל תַּעֲשׂוּ שׁוּם מְלָאכָה – הָאֶזְרָח וְהַגֵּר שֶׁגָּר אִתְּכֶם.
ל כִּֽי־בַיּ֥וֹם הַזֶּ֛ה יְכַפֵּ֥ר עֲלֵיכֶ֖ם לְטַהֵ֣ר אֶתְכֶ֑ם מִכֹּל֙ חַטֹּ֣אתֵיכֶ֔ם לִפְנֵ֥י יְהֹוָ֖ה תִּטְהָֽרוּ׃ כִּי בַּיּוֹם הַזֶּה יְכַפֵּר עֲלֵיכֶם לְטַהֵר אֶתְכֶם מִכָּל חֲטָאֵיכֶם – תִּטָּהֲרוּ לִפְנֵי ה'.
לא שַׁבַּ֨ת שַׁבָּת֥וֹן הִיא֙ לָכֶ֔ם וְעִנִּיתֶ֖ם אֶת־נַפְשֹׁתֵיכֶ֑ם חֻקַּ֖ת עוֹלָֽם׃ הוּא יוֹם שַׁבָּתוֹן עֲבוּרְכֶם וּתְעַנּוּ אֶת נַפְשׁוֹתֵיכֶם בְּצוֹם – חוֹק לְעוֹלָם.
לב וְכִפֶּ֨ר הַכֹּהֵ֜ן אֲשֶׁר־יִמְשַׁ֣ח אֹת֗וֹ וַאֲשֶׁ֤ר יְמַלֵּא֙ אֶת־יָד֔וֹ לְכַהֵ֖ן תַּ֣חַת אָבִ֑יו וְלָבַ֛שׁ אֶת־בִּגְדֵ֥י הַבָּ֖ד בִּגְדֵ֥י הַקֹּֽדֶשׁ׃ הַכֹּהֵן שֶׁיִּמְשְׁחוּ אוֹתוֹ וְשֶׁיְּמוּנֶה לְכַהֵן בִּמְקוֹם אָבִיו הוּא יְכַפֵּר, וְיִלְבַּשׁ אֶת בִּגְדֵי הַבָּד – בִּגְדֵי הַקּוֹדֶשׁ.
לג וְכִפֶּר֙ אֶת־מִקְדַּ֣שׁ הַקֹּ֔דֶשׁ וְאֶת־אֹ֧הֶל מוֹעֵ֛ד וְאֶת־הַמִּזְבֵּ֖חַ יְכַפֵּ֑ר וְעַ֧ל הַכֹּהֲנִ֛ים וְעַל־כׇּל־עַ֥ם הַקָּהָ֖ל יְכַפֵּֽר׃ הוּא יַתִּיז עַל קוֹדֶשׁ הַקֳּדָשִׁים וְעַל שְׁאָר אוֹהֶל הַהִתְכַּנְּסוּת וְעַל הַמִּזְבֵּחַ כְּדֵי לְכַפֵּר, וְכָךְ יְכַפֵּר עַל הַכֹּהֲנִים וְעַל כָּל עַם הַקָּהָל.
לד וְהָֽיְתָה־זֹּ֨את לָכֶ֜ם לְחֻקַּ֣ת עוֹלָ֗ם לְכַפֵּ֞ר עַל־בְּנֵ֤י יִשְׂרָאֵל֙ מִכׇּל־חַטֹּאתָ֔ם אַחַ֖ת בַּשָּׁנָ֑ה וַיַּ֕עַשׂ כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ זֶה יִהְיֶה לָכֶם חוֹק לְעוֹלָם – לְכַפֵּר עַל בְּנֵי יִשְׂרָאֵל מִכָּל חֲטָאֵיהֶם פַּעַם אַחַת בַּשָּׁנָה. אַהֲרֹן עָשָׂה כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

רביעי[עריכה]

פ פָּרָשַׁת שְׁחוּטֵי חוּץ
יז א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב דַּבֵּ֨ר אֶֽל־אַהֲרֹ֜ן וְאֶל־בָּנָ֗יו וְאֶל֙ כׇּל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָמַרְתָּ֖ אֲלֵיהֶ֑ם זֶ֣ה הַדָּבָ֔ר אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה לֵאמֹֽר׃ דַּבֵּר אֶל אַהֲרֹן וְאֶל בָּנָיו וְאֶל כָּל בְּנֵי יִשְׂרָאֵל וֶאֱמוֹר לָהֶם: זֶה הַדָּבָר שֶׁה' צִוָּה:
ג אִ֥ישׁ אִישׁ֙ מִבֵּ֣ית יִשְׂרָאֵ֔ל אֲשֶׁ֨ר יִשְׁחַ֜ט שׁ֥וֹר אוֹ־כֶ֛שֶׂב אוֹ־עֵ֖ז בַּֽמַּחֲנֶ֑ה א֚וֹ אֲשֶׁ֣ר יִשְׁחַ֔ט מִח֖וּץ לַֽמַּחֲנֶֽה׃ כָּל אִישׁ מֵעַם יִשְׂרָאֵל שֶׁיִּשְׁחַט שׁוֹר אוֹ כֶּבֶשׂ אוֹ עֵז בַּמַּחֲנֶה אוֹ שֶׁיִּשְׁחַט מִחוּץ לַמַּחֲנֶה,
ד וְאֶל־פֶּ֜תַח אֹ֣הֶל מוֹעֵד֮ לֹ֣א הֱבִיאוֹ֒ לְהַקְרִ֤יב קׇרְבָּן֙ לַֽיהֹוָ֔ה לִפְנֵ֖י מִשְׁכַּ֣ן יְהֹוָ֑ה דָּ֣ם יֵחָשֵׁ֞ב לָאִ֤ישׁ הַהוּא֙ דָּ֣ם שָׁפָ֔ךְ וְנִכְרַ֛ת הָאִ֥ישׁ הַה֖וּא מִקֶּ֥רֶב עַמּֽוֹ׃ וְלֹא יָבִיא אוֹתוֹ אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת כְּדֵי לְהַקְרִיב אוֹתוֹ קָרְבָּן לַה' לִפְנֵי מִשְׁכַּן ה' – זֶה יֵחָשֵׁב לָאִישׁ הַהוּא לְדָם, כְּאִלּוּ רָצַח וְשָׁפַךְ דָּם, הָאִישׁ הַהוּא יֻשְׁמַד מִתּוֹךְ עַמּוֹ.
ה לְמַ֩עַן֩ אֲשֶׁ֨ר יָבִ֜יאוּ בְּנֵ֣י יִשְׂרָאֵ֗ל אֶֽת־זִבְחֵיהֶם֮ אֲשֶׁ֣ר הֵ֣ם זֹבְחִים֮ עַל־פְּנֵ֣י הַשָּׂדֶה֒ וֶֽהֱבִיאֻ֣ם לַֽיהֹוָ֗ה אֶל־פֶּ֛תַח אֹ֥הֶל מוֹעֵ֖ד אֶל־הַכֹּהֵ֑ן וְזָ֨בְח֜וּ זִבְחֵ֧י שְׁלָמִ֛ים לַֽיהֹוָ֖ה אוֹתָֽם׃ כְּדֵי שֶׁבְּנֵי יִשְׂרָאֵל יָבִיאוּ אֶת זִבְחֵיהֶם שֶׁהֵם רְגִילִים לִזְבּוֹחַ בַּשָּׂדֶה – שֶׁיָּבִיאוּ אוֹתָם לַה' אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת – לַכֹּהֵן, וְיִזְבְּחוּ אוֹתָם כְּזִבְחֵי שְׁלָמִים לַה'.
ו וְזָרַ֨ק הַכֹּהֵ֤ן אֶת־הַדָּם֙ עַל־מִזְבַּ֣ח יְהֹוָ֔ה פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְהִקְטִ֣יר הַחֵ֔לֶב לְרֵ֥יחַ נִיחֹ֖חַ לַיהֹוָֽה׃ וְהַכֹּהֵן יִזְרוֹק אֶת הַדָּם עַל מִזְבַּח ה' מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת, וְיַקְטִיר אֶת הַחֵלֶב לְנַחַת רוּחַ לַה'.
ז וְלֹא־יִזְבְּח֥וּ עוֹד֙ אֶת־זִבְחֵיהֶ֔ם לַשְּׂעִירִ֕ם אֲשֶׁ֛ר הֵ֥ם זֹנִ֖ים אַחֲרֵיהֶ֑ם חֻקַּ֥ת עוֹלָ֛ם תִּֽהְיֶה־זֹּ֥את לָהֶ֖ם לְדֹרֹתָֽם׃ וְלֹא יִזְבְּחוּ עוֹד אֶת זִבְחֵיהֶם לַשֵּׁדִים שֶׁהֵם נִמְשָׁכִים אַחֲרֵיהֶם, זֶה יִהְיֶה לָהֶם חוֹק לְעוֹלָם לְדוֹרוֹת.

חמישי (שלישי במחוברין)[עריכה]

ח וַאֲלֵהֶ֣ם תֹּאמַ֔ר אִ֥ישׁ אִישׁ֙ מִבֵּ֣ית יִשְׂרָאֵ֔ל וּמִן־הַגֵּ֖ר אֲשֶׁר־יָג֣וּר בְּתוֹכָ֑ם אֲשֶׁר־יַעֲלֶ֥ה עֹלָ֖ה אוֹ־זָֽבַח׃ וֶאֱמוֹר לָהֶם: כָּל אִישׁ מֵעַם יִשְׂרָאֵל וּמֵהַגֵּר שֶׁיָּגוּר אִתָּם שֶׁיַּקְרִיב עוֹלָה אוֹ זֶבַח,
ט וְאֶל־פֶּ֜תַח אֹ֤הֶל מוֹעֵד֙ לֹ֣א יְבִיאֶ֔נּוּ לַעֲשׂ֥וֹת אֹת֖וֹ לַיהֹוָ֑ה וְנִכְרַ֛ת הָאִ֥ישׁ הַה֖וּא מֵעַמָּֽיו׃ וְלֹא יָבִיא אוֹתוֹ אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת לְהַקְרִיב אוֹתוֹ לַה' – הָאִישׁ הַהוּא יֻשְׁמַד מֵעַמּוֹ.
פָּרָשַׁת אִסּוּר הַדָּם
י וְאִ֨ישׁ אִ֜ישׁ מִבֵּ֣ית יִשְׂרָאֵ֗ל וּמִן־הַגֵּר֙ הַגָּ֣ר בְּתוֹכָ֔ם אֲשֶׁ֥ר יֹאכַ֖ל כׇּל־דָּ֑ם וְנָתַתִּ֣י פָנַ֗י בַּנֶּ֙פֶשׁ֙ הָאֹכֶ֣לֶת אֶת־הַדָּ֔ם וְהִכְרַתִּ֥י אֹתָ֖הּ מִקֶּ֥רֶב עַמָּֽהּ׃ וְכָל אִישׁ מִבֵּית יִשְׂרָאֵל אוֹ מֵהַגֵּר הַגָּר בְּתוֹכָם שֶׁיֹּאכַל כָּל דָּם שֶׁהוּא, אֶכְעַס עַל הַנֶּפֶשׁ הָאוֹכֶלֶת אֶת הַדָּם, וְאֶעֱשֶׂה שֶׁתֹּאבַד מִתּוֹךְ עַמָּהּ.
יא כִּ֣י נֶ֣פֶשׁ הַבָּשָׂר֮ בַּדָּ֣ם הִוא֒ וַאֲנִ֞י נְתַתִּ֤יו לָכֶם֙ עַל־הַמִּזְבֵּ֔חַ לְכַפֵּ֖ר עַל־נַפְשֹׁתֵיכֶ֑ם כִּֽי־הַדָּ֥ם ה֖וּא בַּנֶּ֥פֶשׁ יְכַפֵּֽר׃ כִּי הַחַיִּים שֶׁל בַּעֲלֵי הַחַיִּים הֵם בַּדָּם, וַאֲנִי צִוִּיתִי אֶתְכֶם שֶׁתִּזְרְקוּ אוֹתוֹ עַל הַמִּזְבֵּחַ כְּדֵי לְכַפֵּר עַל נַפְשׁוֹתֵיכֶם, כִּי זְרִיקַת הַדָּם עַל הַמִּזְבֵּחַ מְכַפֶּרֶת עַל הַנֶּפֶשׁ שֶׁל מֵבִיא הַקָּרְבָּן.
יב עַל־כֵּ֤ן אָמַ֙רְתִּי֙ לִבְנֵ֣י יִשְׂרָאֵ֔ל כׇּל־נֶ֥פֶשׁ מִכֶּ֖ם לֹא־תֹ֣אכַל דָּ֑ם וְהַגֵּ֛ר הַגָּ֥ר בְּתוֹכְכֶ֖ם לֹא־יֹ֥אכַל דָּֽם׃ לָכֵן אָמַרְתִּי לִבְנֵי יִשְׂרָאֵל: שׁוּם אָדָם מֵאִתְּכֶם לֹא יֹאכַל דָּם, וְגַם הַגֵּר שֶׁגָּר אִתְּכֶם לֹא יֹאכַל דָּם.
יג וְאִ֨ישׁ אִ֜ישׁ מִבְּנֵ֣י יִשְׂרָאֵ֗ל וּמִן־הַגֵּר֙ הַגָּ֣ר בְּתוֹכָ֔ם אֲשֶׁ֨ר יָצ֜וּד צֵ֥יד חַיָּ֛ה אוֹ־ע֖וֹף אֲשֶׁ֣ר יֵאָכֵ֑ל וְשָׁפַךְ֙ אֶת־דָּמ֔וֹ וְכִסָּ֖הוּ בֶּעָפָֽר׃ וְכָל אִישׁ מִבְּנֵי יִשְׂרָאֵל אוֹ מֵהַגֵּר שֶׁגָּר אִתָּם שֶׁיָּצוּד חַיָּה אוֹ עוֹף שֶׁמֻּתָּר לֶאֱכוֹל אוֹתָם, יִשְׁפּוֹךְ אֶת דָּמוֹ וִיכַסֶּה אוֹתוֹ בֶּעָפָר.
יד כִּֽי־נֶ֣פֶשׁ כׇּל־בָּשָׂ֗ר דָּמ֣וֹ בְנַפְשׁוֹ֮ הוּא֒ וָֽאֹמַר֙ לִבְנֵ֣י יִשְׂרָאֵ֔ל דַּ֥ם כׇּל־בָּשָׂ֖ר לֹ֣א תֹאכֵ֑לוּ כִּ֣י נֶ֤פֶשׁ כׇּל־בָּשָׂר֙ דָּמ֣וֹ הִ֔וא כׇּל־אֹכְלָ֖יו יִכָּרֵֽת׃ כִּי הַחַיִּים שֶׁל כָּל בַּעֲלֵי הַחַיִּים הֵם בְּדָמָם, וְאָמַרְתִּי לִבְנֵי יִשְׂרָאֵל: אַל תֹּאכְלוּ שׁוּם דַּם שֶׁל בַּעַל חַי, כִּי הַחַיִּים שֶׁל כָּל בַּעֲלֵי הַחַיִּים הֵם בְּדָמָם, כָּל הָאוֹכֵל אוֹתוֹ יֹאבַד מֵעַמּוֹ.
טו וְכׇל־נֶ֗פֶשׁ אֲשֶׁ֨ר תֹּאכַ֤ל נְבֵלָה֙ וּטְרֵפָ֔ה בָּאֶזְרָ֖ח וּבַגֵּ֑ר וְכִבֶּ֨ס בְּגָדָ֜יו וְרָחַ֥ץ בַּמַּ֛יִם וְטָמֵ֥א עַד־הָעֶ֖רֶב וְטָהֵֽר׃ וְכָל אִישׁ – אֶזְרָח אוֹ גֵּר, שֶׁיֹּאכַל נְבֵלָה אוֹ טְרֵפָה, יַטְבִּיל אֶת בְּגָדָיו וְיִטְבּוֹל בְּמַיִם וְיִטְמָא עַד הָעֶרֶב, וְאָז יִהְיֶה טָהוֹר.
טז וְאִם֙ לֹ֣א יְכַבֵּ֔ס וּבְשָׂר֖וֹ לֹ֣א יִרְחָ֑ץ וְנָשָׂ֖א עֲוֺנֽוֹ׃ וְאִם לֹא יַטְבִּיל אֶת בְּגָדָיו וְלֹא יַטְבִּיל אֶת גּוּפוֹ וְיִגַּע בְּקוֹדֶשׁ הוּא יְקַבֵּל עוֹנֶשׁ עַל עֲווֹנוֹ.
פ פָּרָשַׁת אַזְהָרַת הָעֲרָיוֹת
יח א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָמַרְתָּ֖ אֲלֵהֶ֑ם אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וֶאֱמוֹר לָהֶם: אֲנִי ה' אֱלֹהֵיכֶם.
ג כְּמַעֲשֵׂ֧ה אֶֽרֶץ־מִצְרַ֛יִם אֲשֶׁ֥ר יְשַׁבְתֶּם־בָּ֖הּ לֹ֣א תַעֲשׂ֑וּ וּכְמַעֲשֵׂ֣ה אֶֽרֶץ־כְּנַ֡עַן אֲשֶׁ֣ר אֲנִי֩ מֵבִ֨יא אֶתְכֶ֥ם שָׁ֙מָּה֙ לֹ֣א תַעֲשׂ֔וּ וּבְחֻקֹּתֵיהֶ֖ם לֹ֥א תֵלֵֽכוּ׃ אַל תַּעֲשׂוּ כְּמַעֲשֵׂי אֶרֶץ מִצְרַיִם שֶׁגַּרְתֶּם בָּהּ, וְאַל תַּעֲשׂוּ כְּמַעֲשֵׂי אֶרֶץ כְּנַעַן שֶׁאֲנִי מֵבִיא אֶתְכֶם לְשָׁם, וְאַל תִּתְנַהֲגוּ לְפִי הַחֻקִּים שֶׁלָּהֶם.
ד אֶת־מִשְׁפָּטַ֧י תַּעֲשׂ֛וּ וְאֶת־חֻקֹּתַ֥י תִּשְׁמְר֖וּ לָלֶ֣כֶת בָּהֶ֑ם אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ תַּעֲשׂוּ אֶת מִשְׁפָּטַי וְתִשְׁמְרוּ אֶת חֻקַּי וְתִתְנַהֲגוּ עַל פִּיהֶם, אֲנִי ה' אֱלֹהֵיכֶם.
ה וּשְׁמַרְתֶּ֤ם אֶת־חֻקֹּתַי֙ וְאֶת־מִשְׁפָּטַ֔י אֲשֶׁ֨ר יַעֲשֶׂ֥ה אֹתָ֛ם הָאָדָ֖ם וָחַ֣י בָּהֶ֑ם אֲנִ֖י יְהֹוָֽה׃ שִׁמְרוּ אֶת חֻקַּי וְאֶת מִשְׁפָּטַי שֶׁאִם אָדָם יַעֲשֶׂה אוֹתָם הוּא יִחְיֶה בִּזְכוּתָם, אֲנִי ה'.

ט עֶרְוַ֨ת אֲחֽוֹתְךָ֤ בַת־אָבִ֙יךָ֙ א֣וֹ בַת־אִמֶּ֔ךָ מוֹלֶ֣דֶת בַּ֔יִת א֖וֹ מוֹלֶ֣דֶת ח֑וּץ לֹ֥א תְגַלֶּ֖ה עֶרְוָתָֽן׃ אַל תְּגַלֶּה אֶת הַמָּקוֹם הַמֻּצְנָע שֶׁל אֲחוֹתְךָ בַּת אָבִיךָ אוֹ בַּת אִמְּךָ, שֶׁנּוֹלְדָה בְּנִשּׂוּאִין אוֹ מִחוּץ לְנִשּׂוּאִין; אַל תְּגַלֶּה אֶת הַמָּקוֹם הַמֻּצְנָע שֶׁלָּהֵן.

יז עֶרְוַ֥ת אִשָּׁ֛ה וּבִתָּ֖הּ לֹ֣א תְגַלֵּ֑ה אֶֽת־בַּת־בְּנָ֞הּ וְאֶת־בַּת־בִּתָּ֗הּ לֹ֤א תִקַּח֙ לְגַלּ֣וֹת עֶרְוָתָ֔הּ שַׁאֲרָ֥ה הֵ֖נָּה זִמָּ֥ה הִֽוא׃ אֶת הַמָּקוֹם הַמֻּצְנָע שֶׁל אִשָּׁה וּבִתָּהּ אַל תְּגַלֶּה, אַל תִּתְחַתֵּן עִם בַּת בְּנָהּ וְעִם בַּת בִּתָּהּ כְּדֵי לְגַלּוֹת אֶת מְקוֹמַן הַמֻּצְנָע, הֵן קְרוֹבוֹת, זוֹ מַחְשֶׁבֶת חֵטְא.
יח וְאִשָּׁ֥ה אֶל־אֲחֹתָ֖הּ לֹ֣א תִקָּ֑ח לִצְרֹ֗ר לְגַלּ֧וֹת עֶרְוָתָ֛הּ עָלֶ֖יהָ בְּחַיֶּֽיהָ׃ וְאַל תִּקַּח אִשָּׁה עִם אֲחוֹתָהּ לְהִתְחַתֵּן עִמָּן שֶׁתִּהְיֶינָה צָרוֹת זוֹ לָזוֹ – לְגַלּוֹת אֶת הַמָּקוֹם הַמֻּצְנַע שֶׁל אַחַת מֵהֶן בְּחַיֶּיהָ שֶׁל אֲחוֹתָהּ.
יט וְאֶל־אִשָּׁ֖ה בְּנִדַּ֣ת טֻמְאָתָ֑הּ לֹ֣א תִקְרַ֔ב לְגַלּ֖וֹת עֶרְוָתָֽהּ׃ וְאַל תִּגַּשׁ אֶל אִשָּׁה כְּשֶׁהִיא טְמֵאָה בְּנִדָּתָהּ, לְגַלּוֹת אֶת הַמָּקוֹם הַמֻּצְנָע שֶׁלָּהּ.
כ וְאֶל־אֵ֙שֶׁת֙ עֲמִֽיתְךָ֔ לֹא־תִתֵּ֥ן שְׁכׇבְתְּךָ֖ לְזָ֑רַע לְטׇמְאָה־בָֽהּ׃ וְאַל תַּכְנִיס אֶת הַזֶּרַע שֶׁלְּךָ בְּאֵשֶׁת חֲבֵרְךָ וְתִטַּמֵּא בָּהּ.
כא וּמִֽזַּרְעֲךָ֥ לֹא־תִתֵּ֖ן לְהַעֲבִ֣יר לַמֹּ֑לֶךְ וְלֹ֧א תְחַלֵּ֛ל אֶת־שֵׁ֥ם אֱלֹהֶ֖יךָ אֲנִ֥י יְהֹוָֽה׃ וְאַל תִּתֵּן מִצֶּאֱצָאֶיךָ לְהַעֲבִיר בָּאֵשׁ לַעֲבוֹדַת הָאֱלִיל מוֹלֶךְ, וְלֹא תְּחַלֵּל בְּכָךְ אֶת שֵׁם אֱלֹהֶיךָ, אֲנִי ה'.

שביעי (רביעי במחוברין)[עריכה]

כב וְאֶ֨ת־זָכָ֔ר לֹ֥א תִשְׁכַּ֖ב מִשְׁכְּבֵ֣י אִשָּׁ֑ה תּוֹעֵבָ֖ה הִֽוא׃ וְאַל תִּשְׁכַּב עִם זָכָר כְּמוֹ בְּמִשְׁכַּב עִם אִשָּׁה כִּי זוֹ תּוֹעֵבָה.
כג וּבְכׇל־בְּהֵמָ֛ה לֹא־תִתֵּ֥ן שְׁכׇבְתְּךָ֖ לְטׇמְאָה־בָ֑הּ וְאִשָּׁ֗ה לֹֽא־תַעֲמֹ֞ד לִפְנֵ֧י בְהֵמָ֛ה לְרִבְעָ֖הּ תֶּ֥בֶל הֽוּא׃ וְאַל תַּכְנִיס אֶת הַזֶּרַע שֶׁלְּךָ בְּכָל בְּהֵמָה וְתִטַּמֵּא בָּהּ, וְאִשָּׁה לֹא תַּעֲמוֹד לִפְנֵי בְּהֵמָה מִמִּין זָכָר שֶׁיִּשְׁכַּב אִתָּהּ, זֶה עִרְבּוּב וּבִלְבּוּל.
כד אַל־תִּֽטַּמְּא֖וּ בְּכׇל־אֵ֑לֶּה כִּ֤י בְכׇל־אֵ֙לֶּה֙ נִטְמְא֣וּ הַגּוֹיִ֔ם אֲשֶׁר־אֲנִ֥י מְשַׁלֵּ֖חַ מִפְּנֵיכֶֽם׃ אַל תִּטַּמְּאוּ בְּכָל אֵלֶּה כִּי בְּכָל אֵלֶּה נִטְמְאוּ הָעַמִּים שֶׁאֲנִי מְגָרֵשׁ מִפְּנֵיכֶם.
כה וַתִּטְמָ֣א הָאָ֔רֶץ וָאֶפְקֹ֥ד עֲוֺנָ֖הּ עָלֶ֑יהָ וַתָּקִ֥א הָאָ֖רֶץ אֶת־יֹשְׁבֶֽיהָ׃ הָאָרֶץ נִטְמְאָה זָכַרְתִּי אֶת חֲטָאֶיהָ וְהֶעֱנַשְׁתִּי אוֹתָהּ, הָאָרֶץ הֵקִיאָה אֶת תּוֹשָׁבֶיהָ וְגֵרְשָׁה אוֹתָם.
כו וּשְׁמַרְתֶּ֣ם אַתֶּ֗ם אֶת־חֻקֹּתַי֙ וְאֶת־מִשְׁפָּטַ֔י וְלֹ֣א תַעֲשׂ֔וּ מִכֹּ֥ל הַתּוֹעֵבֹ֖ת הָאֵ֑לֶּה הָֽאֶזְרָ֔ח וְהַגֵּ֖ר הַגָּ֥ר בְּתוֹכְכֶֽם׃ אַתֶּם תִּשְׁמְרוּ אֶת חֻקַּי וְאֶת מִשְׁפָּטַי וְאַל תַּעֲשׂוּ מִכָּל הַתּוֹעֵבוֹת הָאֵלֶּה, גַּם הָאֶזְרָח וְגַם הַגֵּר שֶׁגָּר אִתְּכֶם.
כז כִּ֚י אֶת־כׇּל־הַתּוֹעֵבֹ֣ת הָאֵ֔ל עָשׂ֥וּ אַנְשֵֽׁי־הָאָ֖רֶץ אֲשֶׁ֣ר לִפְנֵיכֶ֑ם וַתִּטְמָ֖א הָאָֽרֶץ׃ כִּי אֶת כָּל הַתּוֹעֵבוֹת הָאֵלֶּה עָשׂוּ אַנְשֵׁי הָאָרֶץ שֶׁהָיוּ בָּהּ לִפְנֵיכֶם וְהָאָרֶץ נִטְמְאָה.

מפטיר[עריכה]

כח וְלֹֽא־תָקִ֤יא הָאָ֙רֶץ֙ אֶתְכֶ֔ם בְּטַֽמַּאֲכֶ֖ם אֹתָ֑הּ כַּאֲשֶׁ֥ר קָאָ֛ה אֶת־הַגּ֖וֹי אֲשֶׁ֥ר לִפְנֵיכֶֽם׃ וְכָךְ הָאָרֶץ לֹא תָּקִיא אֶתְכֶם בִּגְלַל שֶׁאַתֶּם מְטַמְּאִים אוֹתָהּ, כְּמוֹ שֶׁהֵקִיאָה אֶת הָעָם שֶׁהָיָה לִפְנֵיכֶם.
כט כִּ֚י כׇּל־אֲשֶׁ֣ר יַעֲשֶׂ֔ה מִכֹּ֥ל הַתּוֹעֵבֹ֖ת הָאֵ֑לֶּה וְנִכְרְת֛וּ הַנְּפָשׁ֥וֹת הָעֹשֹׂ֖ת מִקֶּ֥רֶב עַמָּֽם׃ כִּי כָּל מִי שֶׁיַּעֲשֶׂה מִכָּל הַתּוֹעֵבוֹת הָאֵלֶּה –הַנְּפָשׁוֹת שֶׁעוֹשׂוֹת כָּךְ יֻשְׁמְדוּ מִתּוֹךְ עַמָּם.
ל וּשְׁמַרְתֶּ֣ם אֶת־מִשְׁמַרְתִּ֗י לְבִלְתִּ֨י עֲשׂ֜וֹת מֵחֻקּ֤וֹת הַתּֽוֹעֵבֹת֙ אֲשֶׁ֣ר נַעֲשׂ֣וּ לִפְנֵיכֶ֔ם וְלֹ֥א תִֽטַּמְּא֖וּ בָּהֶ֑ם אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ תַּקְפִּידוּ עַל מִצְווֹתַי שֶׁלֹּא תַּעֲשׂוּ מֵהַחֻקִּים הַמְּתוֹעָבִים שֶׁנַּעֲשׂוּ לִפְנֵיכֶם, וְלֹא תִּטַּמְּאוּ בָּהֶם, אֲנִי ה' אֱלֹהֵיכֶם.
