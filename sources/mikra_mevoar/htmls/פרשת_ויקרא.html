ראשון[עריכה]

פָּרָשַׁת הָעוֹלָה
א א וַיִּקְרָ֖א אֶל־מֹשֶׁ֑ה וַיְדַבֵּ֤ר יְהֹוָה֙ אֵלָ֔יו מֵאֹ֥הֶל מוֹעֵ֖ד לֵאמֹֽר׃ ה' קָרָא לְמֹשֶׁה, וְדִבֵּר אֵלָיו מֵאוֹהֶל הַהִתְכַּנְּסוּת, וְאָמַר:
ב דַּבֵּ֞ר אֶל־בְּנֵ֤י יִשְׂרָאֵל֙ וְאָמַרְתָּ֣ אֲלֵהֶ֔ם אָדָ֗ם כִּֽי־יַקְרִ֥יב מִכֶּ֛ם קׇרְבָּ֖ן לַֽיהֹוָ֑ה מִן־הַבְּהֵמָ֗ה מִן־הַבָּקָר֙ וּמִן־הַצֹּ֔אן תַּקְרִ֖יבוּ אֶת־קׇרְבַּנְכֶֽם׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְתֹאמַר לָהֶם: כְּשֶׁאָדָם מִכֶּם יַקְרִיב קָרְבָּן לַה' – תַּקְרִיבוּ אֶת קָרְבַּנְכֶם מֵהַבְּהֵמוֹת: מֵהַבָּקָר, וּמֵהַצֹּאן.
ג אִם־עֹלָ֤ה קׇרְבָּנוֹ֙ מִן־הַבָּקָ֔ר זָכָ֥ר תָּמִ֖ים יַקְרִיבֶ֑נּוּ אֶל־פֶּ֜תַח אֹ֤הֶל מוֹעֵד֙ יַקְרִ֣יב אֹת֔וֹ לִרְצֹנ֖וֹ לִפְנֵ֥י יְהֹוָֽה׃ אִם קָרְבָּנוֹ קָרְבַּן עוֹלָה מֵהַבָּקָר, יַקְרִיב זָכָר שָׁלֵם לְלֹא מוּם, וְיַגִּישׁ אוֹתוֹ אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת כְּדֵי שֶׁה' יִתְרַצֶּה לוֹ.
ד וְסָמַ֣ךְ יָד֔וֹ עַ֖ל רֹ֣אשׁ הָעֹלָ֑ה וְנִרְצָ֥ה ל֖וֹ לְכַפֵּ֥ר עָלָֽיו׃ הוּא יַשְׁעִין אֶת יָדָיו עַל רֹאשׁ הָעוֹלָה, וַה' יִתְרַצֶּה לוֹ וִיכַפֵּר עַל עֲווֹנוֹתָיו.
ה וְשָׁחַ֛ט אֶת־בֶּ֥ן הַבָּקָ֖ר לִפְנֵ֣י יְהֹוָ֑ה וְ֠הִקְרִ֠יבוּ בְּנֵ֨י אַהֲרֹ֤ן הַכֹּֽהֲנִים֙ אֶת־הַדָּ֔ם וְזָרְק֨וּ אֶת־הַדָּ֤ם עַל־הַמִּזְבֵּ֙חַ֙ סָבִ֔יב אֲשֶׁר־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ הוּא יִשְׁחַט אֶת בֶּן הַבָּקָר לִפְנֵי ה', וְהַכֹּהֲנִים בְּנֵי אַהֲרֹן יַגִּישׁוּ אֶת הַדָּם וְיִזְרְקוּ אוֹתוֹ מִסָּבִיב עַל הַמִּזְבֵּחַ שֶׁמּוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.
ו וְהִפְשִׁ֖יט אֶת־הָעֹלָ֑ה וְנִתַּ֥ח אֹתָ֖הּ לִנְתָחֶֽיהָ׃ הוּא יַפְשִׁיט אֶת הָעוֹלָה מֵהָעוֹר, וִיחַלֵּק אוֹתָהּ לַחֲלָקִים.
ז וְ֠נָתְנ֠וּ בְּנֵ֨י אַהֲרֹ֧ן הַכֹּהֵ֛ן אֵ֖שׁ עַל־הַמִּזְבֵּ֑חַ וְעָרְכ֥וּ עֵצִ֖ים עַל־הָאֵֽשׁ׃ בְּנֵי אַהֲרֹן הַכֹּהֵן יָנִיחוּ אֵשׁ עַל הַמִּזְבֵּחַ וִיסַדְּרוּ עֵצִים עַל הָאֵשׁ.
ח וְעָרְכ֗וּ בְּנֵ֤י אַהֲרֹן֙ הַכֹּ֣הֲנִ֔ים אֵ֚ת הַנְּתָחִ֔ים אֶת־הָרֹ֖אשׁ וְאֶת־הַפָּ֑דֶר עַל־הָעֵצִים֙ אֲשֶׁ֣ר עַל־הָאֵ֔שׁ אֲשֶׁ֖ר עַל־הַמִּזְבֵּֽחַ׃ הַכֹּהֲנִים בְּנֵי אַהֲרֹן יְסַדְּרוּ אֶת הַחֲלָקִים – אֶת הָרֹאשׁ וְאֶת שׁוּמַן הַחֵלֶב, עַל הָעֵצִים שֶׁעַל הָאֵשׁ שֶׁעַל הַמִּזְבֵּחַ.
ט וְקִרְבּ֥וֹ וּכְרָעָ֖יו יִרְחַ֣ץ בַּמָּ֑יִם וְהִקְטִ֨יר הַכֹּהֵ֤ן אֶת־הַכֹּל֙ הַמִּזְבֵּ֔חָה עֹלָ֛ה אִשֵּׁ֥ה רֵֽיחַ־נִיח֖וֹחַ לַֽיהֹוָֽה׃ וְאֶת מֵעָיו וְרַגְלָיו יִרְחַץ בְּמַיִם, וְהַכֹּהֵן יַקְטִיר אֶת הַכֹּל עַל הַמִּזְבֵּחַ, שֶׁיַּעֲלֶה בָּאֵשׁ לְנַחַת רוּחַ לַה'.
ס
י וְאִם־מִן־הַצֹּ֨אן קׇרְבָּנ֧וֹ מִן־הַכְּשָׂבִ֛ים א֥וֹ מִן־הָעִזִּ֖ים לְעֹלָ֑ה זָכָ֥ר תָּמִ֖ים יַקְרִיבֶֽנּוּ׃ וְאִם קָרְבָּנוֹ מֵהַצֹּאן – מֵהַכְּבָשִׂים אוֹ מֵהָעִזִּים לְעוֹלָה, יַקְרִיב זָכָר שָׁלֵם לְלֹא מוּם.
יא וְשָׁחַ֨ט אֹת֜וֹ עַ֣ל יֶ֧רֶךְ הַמִּזְבֵּ֛חַ צָפֹ֖נָה לִפְנֵ֣י יְהֹוָ֑ה וְזָרְק֡וּ בְּנֵי֩ אַהֲרֹ֨ן הַכֹּהֲנִ֧ים אֶת־דָּמ֛וֹ עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ הוּא יִשְׁחַט אוֹתוֹ בִּקְצֵה הַמִּזְבֵּחַ שֶׁלְּצַד צָפוֹן לִפְנֵי ה', וְהַכֹּהֲנִים בְּנֵי אַהֲרֹן יִזְרְקוּ אֶת דָּמוֹ עַל הַמִּזְבֵּחַ מִסָּבִיב.
יב וְנִתַּ֤ח אֹתוֹ֙ לִנְתָחָ֔יו וְאֶת־רֹאשׁ֖וֹ וְאֶת־פִּדְר֑וֹ וְעָרַ֤ךְ הַכֹּהֵן֙ אֹתָ֔ם עַל־הָֽעֵצִים֙ אֲשֶׁ֣ר עַל־הָאֵ֔שׁ אֲשֶׁ֖ר עַל־הַמִּזְבֵּֽחַ׃ הוּא יְחַלֵּק אוֹתוֹ לַחֲלָקִים – אֶת רֹאשׁוֹ וְאֶת שׁוּמַן הַחֵלֶב שֶׁלּוֹ, וְהַכֹּהֵן יַעֲרוֹךְ אוֹתָם עַל הָעֵצִים שֶׁעַל הָאֵשׁ שֶׁעַל הַמִּזְבֵּחַ.
יג וְהַקֶּ֥רֶב וְהַכְּרָעַ֖יִם יִרְחַ֣ץ בַּמָּ֑יִם וְהִקְרִ֨יב הַכֹּהֵ֤ן אֶת־הַכֹּל֙ וְהִקְטִ֣יר הַמִּזְבֵּ֔חָה עֹלָ֣ה ה֗וּא אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַיהֹוָֽה׃ וְאֶת הַמֵּעַיִם וְאֶת הָרַגְלַיִם יִרְחַץ בְּמַיִם, וְהַכֹּהֵן יַקְרִיב אֶת הַכֹּל וְיַקְטִיר עַל הַמִּזְבֵּחַ, קָרְבַּן עוֹלָה הוּא – עוֹלֶה בָּאֵשׁ לְנַחַת רוּחַ לַה'.

שני[עריכה]

פ
יד וְאִ֧ם מִן־הָע֛וֹף עֹלָ֥ה קׇרְבָּנ֖וֹ לַֽיהֹוָ֑ה וְהִקְרִ֣יב מִן־הַתֹּרִ֗ים א֛וֹ מִן־בְּנֵ֥י הַיּוֹנָ֖ה אֶת־קׇרְבָּנֽוֹ׃ וְאִם קָרְבָּנוֹ לַה' הוּא עוֹלָה מֵהָעוֹף – יַקְרִיב אֶת קָרְבָּנוֹ מֵהַתּוֹרִים אוֹ מִבְּנֵי הַיּוֹנָה.
טו וְהִקְרִיב֤וֹ הַכֹּהֵן֙ אֶל־הַמִּזְבֵּ֔חַ וּמָלַק֙ אֶת־רֹאשׁ֔וֹ וְהִקְטִ֖יר הַמִּזְבֵּ֑חָה וְנִמְצָ֣ה דָמ֔וֹ עַ֖ל קִ֥יר הַמִּזְבֵּֽחַ׃ הַכֹּהֵן יַגִּישׁ אוֹתוֹ לַמִּזְבֵּחַ וְיַחֲתוֹךְ אֶת רֹאשׁוֹ בְּצִפּוֹרֶן הַבּוֹהֶן, וְיַקְטִיר עַל הַמִּזְבֵּחַ, וְדָמוֹ יִסָּחֵט עַל קִיר הַמִּזְבֵּחַ.
טז וְהֵסִ֥יר אֶת־מֻרְאָת֖וֹ בְּנֹצָתָ֑הּ וְהִשְׁלִ֨יךְ אֹתָ֜הּ אֵ֤צֶל הַמִּזְבֵּ֙חַ֙ קֵ֔דְמָה אֶל־מְק֖וֹם הַדָּֽשֶׁן׃ הוּא יָסִיר אֶת הַזֶּפֶק שֶׁלּוֹ עִם הַמֵּעַיִם שֶׁמְּלֵאִים בְּאוֹכֶל מְעֻכָּל, וְיַשְׁלִיךְ אוֹתָם קָרוֹב לַמִּזְבֵּחַ בְּצַד מִזְרַח – לִמְקוֹם הַשְׁלָכַת הָאֵפֶר.
יז וְשִׁסַּ֨ע אֹת֣וֹ בִכְנָפָיו֮ לֹ֣א יַבְדִּיל֒ וְהִקְטִ֨יר אֹת֤וֹ הַכֹּהֵן֙ הַמִּזְבֵּ֔חָה עַל־הָעֵצִ֖ים אֲשֶׁ֣ר עַל־הָאֵ֑שׁ עֹלָ֣ה ה֗וּא אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַיהֹוָֽה׃ הוּא יִתְפּוֹס בִּכְנָפָיו וְיִקְרַע אוֹתוֹ אֲבָל לֹא יַפְרִיד לְגַמְרֵי, וְהַכֹּהֵן יַקְטִיר אוֹתוֹ עַל הַמִּזְבֵּחַ עַל הָעֵצִים שֶׁעַל הָאֵשׁ, קָרְבַּן עוֹלָה הוּא – עוֹלֶה בָּאֵשׁ לְנַחַת רוּחַ לַה'.
ס פָּרָשַׁת הַמִּנְחָה
ב א וְנֶ֗פֶשׁ כִּֽי־תַקְרִ֞יב קׇרְבַּ֤ן מִנְחָה֙ לַֽיהֹוָ֔ה סֹ֖לֶת יִהְיֶ֣ה קׇרְבָּנ֑וֹ וְיָצַ֤ק עָלֶ֙יהָ֙ שֶׁ֔מֶן וְנָתַ֥ן עָלֶ֖יהָ לְבֹנָֽה׃ וּכְשֶׁאָדָם יַקְרִיב קָרְבַּן מִנְחָה לַה' – קָרְבָּנוֹ יִהְיֶה סוֹלֶת, הוּא יִצוֹק עָלֶיהָ שֶׁמֶן וְיָשִׂים עָלֶיהָ לְבוֹנָה.
ב וֶֽהֱבִיאָ֗הּ אֶל־בְּנֵ֣י אַהֲרֹן֮ הַכֹּהֲנִים֒ וְקָמַ֨ץ מִשָּׁ֜ם מְלֹ֣א קֻמְצ֗וֹ מִסׇּלְתָּהּ֙ וּמִשַּׁמְנָ֔הּ עַ֖ל כׇּל־לְבֹנָתָ֑הּ וְהִקְטִ֨יר הַכֹּהֵ֜ן אֶת־אַזְכָּרָתָהּ֙ הַמִּזְבֵּ֔חָה אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַיהֹוָֽה׃ הוּא יָבִיא אוֹתָהּ לַכֹּהֲנִים בְּנֵי אַהֲרֹן, וְהַכֹּהֵן יִקַּח מִשָּׁם חוֹפֶן מָלֵא מֵהַסּוֹלֶת וּמֵהַשֶּׁמֶן שֶׁלָּהּ בְּנוֹסַף עַל כָּל הַלְּבוֹנָה שֶׁלָּהּ, וְיַקְטִיר עַל הַמִּזְבֵּחַ אֶת הַחוֹפֶן הַזֶּה שֶׁגּוֹרֵם לְהִזָּכֵר לִפְנֵי ה', כְּדֵי שֶׁיַּעֲלֶה בָּאֵשׁ וְיִגְרוֹם נַחַת רוּחַ לַה'.
ג וְהַנּוֹתֶ֙רֶת֙ מִן־הַמִּנְחָ֔ה לְאַהֲרֹ֖ן וּלְבָנָ֑יו קֹ֥דֶשׁ קׇֽדָשִׁ֖ים מֵאִשֵּׁ֥י יְהֹוָֽה׃ וּמַה שֶּׁנִּשְׁאַר מֵהַמִּנְחָה יִנָּתֵן לְאַהֲרֹן וּלְבָנָיו – קוֹדֶשׁ קָדָשִׁים מִקָּרְבְּנוֹת ה'.

ד וְכִ֥י תַקְרִ֛ב קׇרְבַּ֥ן מִנְחָ֖ה מַאֲפֵ֣ה תַנּ֑וּר סֹ֣לֶת חַלּ֤וֹת מַצֹּת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֥ים בַּשָּֽׁמֶן׃ וּכְשֶׁתַּקְרִיב קָרְבַּן מִנְחָה מַאֲפֵה תַּנּוּר – תָּבִיא חַלּוֹת שֶׁלֹּא הֶחְמִיצוּ עֲשׂוּיוֹת מִסּוֹלֶת וּמְעֻרְבָּבוֹת בְּשֶׁמֶן, וּפִתּוֹת דַּקּוֹת שֶׁלֹּא הֶחְמִיצוּ מְרוּחוֹת בְּשֶׁמֶן.

ה וְאִם־מִנְחָ֥ה עַל־הַֽמַּחֲבַ֖ת קׇרְבָּנֶ֑ךָ סֹ֛לֶת בְּלוּלָ֥ה בַשֶּׁ֖מֶן מַצָּ֥ה תִהְיֶֽה׃ וְאִם קָרְבָּנְךָ מִנְחָה עַל מַחֲבַת שְׁטוּחָה, הִיא תִּהְיֶה מִסּוֹלֶת שֶׁלֹּא הֶחְמִיצָה מְעֻרְבֶּבֶת בְּשֶׁמֶן.

שלישי[עריכה]

ס
ז וְאִם־מִנְחַ֥ת מַרְחֶ֖שֶׁת קׇרְבָּנֶ֑ךָ סֹ֥לֶת בַּשֶּׁ֖מֶן תֵּעָשֶֽׂה׃ וְאִם קָרְבָּנְךָ מִנְחָה מְטֻגֶּנֶת בְּמַחֲבַת עֲמוּקָה, הִיא תֵּעָשֶׂה מִסּוֹלֶת בְּשֶׁמֶן.
ח וְהֵבֵאתָ֣ אֶת־הַמִּנְחָ֗ה אֲשֶׁ֧ר יֵעָשֶׂ֛ה מֵאֵ֖לֶּה לַיהֹוָ֑ה וְהִקְרִיבָהּ֙ אֶל־הַכֹּהֵ֔ן וְהִגִּישָׁ֖הּ אֶל־הַמִּזְבֵּֽחַ׃ תָּבִיא אֶת הַמִּנְחָה שֶׁתֵּעָשֶׂה בְּאַחַת מֵהָאֶפְשָׁרוּיוֹת הָאֵלֶּה לַה', הַמֵּבִיא יַגִּישׁ אוֹתָהּ לַכֹּהֵן שֶׁיַּגִּישׁ אוֹתָהּ לַמִּזְבֵּחַ.
ט וְהֵרִ֨ים הַכֹּהֵ֤ן מִן־הַמִּנְחָה֙ אֶת־אַזְכָּ֣רָתָ֔הּ וְהִקְטִ֖יר הַמִּזְבֵּ֑חָה אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַיהֹוָֽה׃ הַכֹּהֵן יָרִים מֵהַמִּנְחָה אֶת הַחוֹפֶן שֶׁגּוֹרֵם לְהִזָּכֵר לִפְנֵי ה', וְיַקְטִיר עַל הַמִּזְבֵּחַ, קָרְבָּן עוֹלֶה בָּאֵשׁ רֵיחַ לְנַחַת רוּחַ לַה'.
י וְהַנּוֹתֶ֙רֶת֙ מִן־הַמִּנְחָ֔ה לְאַהֲרֹ֖ן וּלְבָנָ֑יו קֹ֥דֶשׁ קׇֽדָשִׁ֖ים מֵאִשֵּׁ֥י יְהֹוָֽה׃ וּמַה שֶּׁנִּשְׁאַר מֵהַמִּנְחָה יִנָּתֵן לְאַהֲרֹן וּלְבָנָיו, קוֹדֶשׁ קָדָשִׁים מִקָּרְבְּנוֹת ה'.
יא כׇּל־הַמִּנְחָ֗ה אֲשֶׁ֤ר תַּקְרִ֙יבוּ֙ לַיהֹוָ֔ה לֹ֥א תֵעָשֶׂ֖ה חָמֵ֑ץ כִּ֤י כׇל־שְׂאֹר֙ וְכׇל־דְּבַ֔שׁ לֹֽא־תַקְטִ֧ירוּ מִמֶּ֛נּוּ אִשֶּׁ֖ה לַֽיהֹוָֽה׃ כָּל הַמְּנָחוֹת שֶׁתַּקְרִיבוּ לַה' לֹא יֵעָשׂוּ חָמֵץ, כִּי שׁוּם בָּצֵק תּוֹפֵחַ שֶׁמַּתְפִּיחַ בָּצֵק אַחֵר אוֹ דְּבַשׁ אָסוּר לָכֶם לְהַקְטִיר בָּאֵשׁ לַה'.
יב קׇרְבַּ֥ן רֵאשִׁ֛ית תַּקְרִ֥יבוּ אֹתָ֖ם לַיהֹוָ֑ה וְאֶל־הַמִּזְבֵּ֥חַ לֹא־יַעֲל֖וּ לְרֵ֥יחַ נִיחֹֽחַ׃ רַק לְקָרְבַּן רֵאשִׁית הַתְּבוּאָה שֶׁמַּקְרִיבִים בְּחַג הַשָּׁבוּעוֹת תַּקְרִיבוּ חָמֵץ וּדְבַשׁ לַה', וְגַם אָז לֹא יַעֲלוּ אוֹתָם לַמִּזְבֵּחַ לְנַחַת רוּחַ.
יג וְכׇל־קׇרְבַּ֣ן מִנְחָתְךָ֮ בַּמֶּ֣לַח תִּמְלָח֒ וְלֹ֣א תַשְׁבִּ֗ית מֶ֚לַח בְּרִ֣ית אֱלֹהֶ֔יךָ מֵעַ֖ל מִנְחָתֶ֑ךָ עַ֥ל כׇּל־קׇרְבָּנְךָ֖ תַּקְרִ֥יב מֶֽלַח׃ וְתָשִׂים מֶלַח עַל כָּל קָרְבְּנוֹת הַמִּנְחָה שֶׁתַּקְרִיב, וְלֹא תְּבַטֵּל מִמִּנְחָתְךָ אֶת הַמֶּלַח שֶׁמְּסַמֵּל בַּעֲמִידוּתוֹ אֶת נִצְחִיּוּת הַבְּרִית עִם אֱלֹהֶיךָ – עַל כָּל קָרְבְּנוֹתֶיךָ תַּקְרִיב מֶלַח.
ס
יד וְאִם־תַּקְרִ֛יב מִנְחַ֥ת בִּכּוּרִ֖ים לַיהֹוָ֑ה אָבִ֞יב קָל֤וּי בָּאֵשׁ֙ גֶּ֣רֶשׂ כַּרְמֶ֔ל תַּקְרִ֕יב אֵ֖ת מִנְחַ֥ת בִּכּוּרֶֽיךָ׃ וְכַאֲשֶׁר תַּקְרִיב מִנְחַת בִּכּוּרִים לַה' – לְמָחֳרַת הַפֶּסַח, תַּקְרִיב אֶת מִנְחַת הַבִּכּוּרִים שֶׁלְּךָ גְּרִיסִים רַכִּים מִתְּבוּאָה צְעִירָה קְלוּיָה בָּאֵשׁ.
טו וְנָתַתָּ֤ עָלֶ֙יהָ֙ שֶׁ֔מֶן וְשַׂמְתָּ֥ עָלֶ֖יהָ לְבֹנָ֑ה מִנְחָ֖ה הִֽוא׃ תָּשִׂים עָלֶיהָ שֶׁמֶן וְתָשִׂים עָלֶיהָ לְבוֹנָה – הִיא מִנְחָה.
טז וְהִקְטִ֨יר הַכֹּהֵ֜ן אֶת־אַזְכָּרָתָ֗הּ מִגִּרְשָׂהּ֙ וּמִשַּׁמְנָ֔הּ עַ֖ל כׇּל־לְבֹנָתָ֑הּ אִשֶּׁ֖ה לַיהֹוָֽה׃ וְהַכֹּהֵן יַקְטִיר מֵהַגְּרִיסִים וּמֵהַשֶּׁמֶן שֶׁלָּהּ אֶת הַחוֹפֶן שֶׁגּוֹרֵם לְהִזָּכֵר לִפְנֵי ה', בְּנוֹסַף לְכָל הַלְּבוֹנָה שֶׁלָּהּ, קָרְבָּן עוֹלֶה בָּאֵשׁ לַה'.

רביעי[עריכה]

פ פָּרָשַׁת הַשְּׁלָמִים
ג א וְאִם־זֶ֥בַח שְׁלָמִ֖ים קׇרְבָּנ֑וֹ אִ֤ם מִן־הַבָּקָר֙ ה֣וּא מַקְרִ֔יב אִם־זָכָר֙ אִם־נְקֵבָ֔ה תָּמִ֥ים יַקְרִיבֶ֖נּוּ לִפְנֵ֥י יְהֹוָֽה׃ וְאִם קָרְבָּנוֹ זֶבַח שְׁלָמִים: אִם הוּא מַקְרִיב מֵהַבָּקָר בֵּין זָכָר בֵּין נְקֵבָה – יַקְרִיב אוֹתוֹ לִפְנֵי ה' שָׁלֵם לְלֹא מוּם.
ב וְסָמַ֤ךְ יָדוֹ֙ עַל־רֹ֣אשׁ קׇרְבָּנ֔וֹ וּשְׁחָט֕וֹ פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְזָרְק֡וּ בְּנֵי֩ אַהֲרֹ֨ן הַכֹּהֲנִ֧ים אֶת־הַדָּ֛ם עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ הוּא יַשְׁעִין אֶת יָדָיו עַל רֹאשׁ קָרְבָּנוֹ וְיִשְׁחַט אוֹתוֹ מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת, וְהַכֹּהֲנִים בְּנֵי אַהֲרֹן יִזְרְקוּ אֶת הַדָּם עַל הַמִּזְבֵּחַ מִסָּבִיב.
ג וְהִקְרִיב֙ מִזֶּ֣בַח הַשְּׁלָמִ֔ים אִשֶּׁ֖ה לַיהֹוָ֑ה אֶת־הַחֵ֙לֶב֙ הַֽמְכַסֶּ֣ה אֶת־הַקֶּ֔רֶב וְאֵת֙ כׇּל־הַחֵ֔לֶב אֲשֶׁ֖ר עַל־הַקֶּֽרֶב׃ הוּא יַקְרִיב מִזֶּבַח הַשְּׁלָמִים לָאֵשׁ לַה' – אֶת הַחֵלֶב שֶׁמְּכַסֶּה אֶת הַמֵּעַיִם וְאֶת כָּל הַחֵלֶב שֶׁעַל הַמֵּעַיִם.
ד וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־הַחֵ֙לֶב֙ אֲשֶׁ֣ר עֲלֵהֶ֔ן אֲשֶׁ֖ר עַל־הַכְּסָלִ֑ים וְאֶת־הַיֹּתֶ֙רֶת֙ עַל־הַכָּבֵ֔ד עַל־הַכְּלָי֖וֹת יְסִירֶֽנָּה׃ וְאֶת שְׁתֵּי הַכְּלָיוֹת וְאֶת הַחֵלֶב שֶׁעֲלֵיהֶן וְעַל הַמָּתְנַיִם, וְאֶת הַתּוֹסֶפֶת שֶׁעַל הַכָּבֵד שֶׁיָּסִיר אוֹתָהּ עִם הַכְּלָיוֹת.
ה וְהִקְטִ֨ירוּ אֹת֤וֹ בְנֵֽי־אַהֲרֹן֙ הַמִּזְבֵּ֔חָה עַל־הָ֣עֹלָ֔ה אֲשֶׁ֥ר עַל־הָעֵצִ֖ים אֲשֶׁ֣ר עַל־הָאֵ֑שׁ אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַֽיהֹוָֽה׃ בְּנֵי אַהֲרֹן יַקְטִירוּ אוֹתָם עַל הַמִּזְבֵּחַ, עַל הָעוֹלָה שֶׁעַל הָעֵצִים שֶׁעַל הָאֵשׁ, קָרְבָּן עוֹלֶה בָּאֵשׁ לְנַחַת רוּחַ לַה'.

ו וְאִם־מִן־הַצֹּ֧אן קׇרְבָּנ֛וֹ לְזֶ֥בַח שְׁלָמִ֖ים לַיהֹוָ֑ה זָכָר֙ א֣וֹ נְקֵבָ֔ה תָּמִ֖ים יַקְרִיבֶֽנּוּ׃ וְאִם קָרְבָּנוֹ לַה' לְזֶבַח שְׁלָמִים מֵהַצֹּאן – יַקְרִיב אוֹתוֹ זָכָר אוֹ נְקֵבָה שָׁלֵם לְלֹא מוּם.
ז אִם־כֶּ֥שֶׂב הֽוּא־מַקְרִ֖יב אֶת־קׇרְבָּנ֑וֹ וְהִקְרִ֥יב אֹת֖וֹ לִפְנֵ֥י יְהֹוָֽה׃ אִם הוּא מַקְרִיב כֶּבֶשׂ לְקָרְבָּנוֹ – יַקְרִיב אוֹתוֹ לִפְנֵי ה'.
ח וְסָמַ֤ךְ אֶת־יָדוֹ֙ עַל־רֹ֣אשׁ קׇרְבָּנ֔וֹ וְשָׁחַ֣ט אֹת֔וֹ לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וְ֠זָרְק֠וּ בְּנֵ֨י אַהֲרֹ֧ן אֶת־דָּמ֛וֹ עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ וְיַשְׁעִין אֶת יָדָיו עַל רֹאשׁ קָרְבָּנוֹ וְיִשְׁחַט אוֹתוֹ לִפְנֵי אוֹהֶל הַהִתְכַּנְּסוּת, וּבְנֵי אַהֲרֹן יִזְרְקוּ אֶת דָּמוֹ עַל הַמִּזְבֵּחַ מִסָּבִיב.
ט וְהִקְרִ֨יב מִזֶּ֣בַח הַשְּׁלָמִים֮ אִשֶּׁ֣ה לַיהֹוָה֒ חֶלְבּוֹ֙ הָאַלְיָ֣ה תְמִימָ֔ה לְעֻמַּ֥ת הֶעָצֶ֖ה יְסִירֶ֑נָּה וְאֶת־הַחֵ֙לֶב֙ הַֽמְכַסֶּ֣ה אֶת־הַקֶּ֔רֶב וְאֵת֙ כׇּל־הַחֵ֔לֶב אֲשֶׁ֖ר עַל־הַקֶּֽרֶב׃ הוּא יַקְרִיב מִזֶּבַח הַשְּׁלָמִים קָרְבָּן עוֹלֶה בָּאֵשׁ לַה' – אֶת שׁוּמַן הַזָּנָב שֶׁיּוּסַר בִּשְׁלֵמוּת מִמּוּל עֶצֶם הַזָּנָב, וְאֶת הַחֵלֶב שֶׁמְּכַסֶּה אֶת הַמֵּעַיִם וְאֶת כָּל הַחֵלֶב שֶׁעַל הַמֵּעַיִם.
י וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־הַחֵ֙לֶב֙ אֲשֶׁ֣ר עֲלֵהֶ֔ן אֲשֶׁ֖ר עַל־הַכְּסָלִ֑ים וְאֶת־הַיֹּתֶ֙רֶת֙ עַל־הַכָּבֵ֔ד עַל־הַכְּלָיֹ֖ת יְסִירֶֽנָּה׃ וְאֶת שְׁתֵּי הַכְּלָיוֹת וְאֶת הַחֵלֶב שֶׁעֲלֵיהֶן וְעַל הַמָּתְנַיִם, וְאֶת הַתּוֹסֶפֶת שֶׁעַל הַכָּבֵד שֶׁיָּסִיר אוֹתָהּ עִם הַכְּלָיוֹת.
יא וְהִקְטִיר֥וֹ הַכֹּהֵ֖ן הַמִּזְבֵּ֑חָה לֶ֥חֶם אִשֶּׁ֖ה לַיהֹוָֽה׃ הַכֹּהֵן יַקְטִיר אוֹתוֹ עַל הַמִּזְבֵּחַ, מַאֲכָל לָאֵשׁ – לַה'.
פ
יב וְאִ֥ם עֵ֖ז קׇרְבָּנ֑וֹ וְהִקְרִיב֖וֹ לִפְנֵ֥י יְהֹוָֽה׃ וְאִם קָרְבָּנוֹ עֵז – יַקְרִיב אוֹתוֹ לִפְנֵי ה'.
יג וְסָמַ֤ךְ אֶת־יָדוֹ֙ עַל־רֹאשׁ֔וֹ וְשָׁחַ֣ט אֹת֔וֹ לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וְ֠זָרְק֠וּ בְּנֵ֨י אַהֲרֹ֧ן אֶת־דָּמ֛וֹ עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ וְיַשְׁעִין אֶת יָדָיו עַל רֹאשׁוֹ וְיִשְׁחַט אוֹתוֹ לִפְנֵי אוֹהֶל הַהִתְכַּנְּסוּת, וּבְנֵי אַהֲרֹן יִזְרְקוּ אֶת דָּמוֹ עַל הַמִּזְבֵּחַ מִסָּבִיב.
יד וְהִקְרִ֤יב מִמֶּ֙נּוּ֙ קׇרְבָּנ֔וֹ אִשֶּׁ֖ה לַֽיהֹוָ֑ה אֶת־הַחֵ֙לֶב֙ הַֽמְכַסֶּ֣ה אֶת־הַקֶּ֔רֶב וְאֵת֙ כׇּל־הַחֵ֔לֶב אֲשֶׁ֖ר עַל־הַקֶּֽרֶב׃ וְיַקְרִיב מִמֶּנּוּ אֶת קָרְבָּנוֹ לָאֵשׁ לַה', אֶת הַחֵלֶב שֶׁמְּכַסֶּה אֶת הַמֵּעַיִם וְאֶת כָּל הַחֵלֶב שֶׁעַל הַמֵּעַיִם.
טו וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־הַחֵ֙לֶב֙ אֲשֶׁ֣ר עֲלֵהֶ֔ן אֲשֶׁ֖ר עַל־הַכְּסָלִ֑ים וְאֶת־הַיֹּתֶ֙רֶת֙ עַל־הַכָּבֵ֔ד עַל־הַכְּלָיֹ֖ת יְסִירֶֽנָּה׃ וְאֶת שְׁתֵּי הַכְּלָיוֹת וְאֶת הַחֵלֶב שֶׁעֲלֵיהֶן וְעַל הַמָּתְנַיִם, וְאֶת הַתּוֹסֶפֶת שֶׁעַל הַכָּבֵד שֶׁיָּסִיר אוֹתָהּ עִם הַכְּלָיוֹת.
טז וְהִקְטִירָ֥ם הַכֹּהֵ֖ן הַמִּזְבֵּ֑חָה לֶ֤חֶם אִשֶּׁה֙ לְרֵ֣יחַ נִיחֹ֔חַ כׇּל־חֵ֖לֶב לַיהֹוָֽה׃ הַכֹּהֵן יַקְטִיר אוֹתָם עַל הַמִּזְבֵּחַ, כָּל חֵלֶב הוּא מַאֲכָל לָאֵשׁ לְנַחַת רוּחַ לַה'.
יז חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם בְּכֹ֖ל מוֹשְׁבֹֽתֵיכֶ֑ם כׇּל־חֵ֥לֶב וְכׇל־דָּ֖ם לֹ֥א תֹאכֵֽלוּ׃ חוֹק לְעוֹלָם לְדוֹרוֹת בְּכָל מְקוֹמוֹת מְגוּרֵיכֶם, אַל תֹּאכְלוּ שׁוּם חֵלֶב וְשׁוּם דָּם.

חמישי[עריכה]

פ פָּרָשַׁת חַטַּאת הַכֹּהֵן הַמָּשִׁיחַ
ד א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב דַּבֵּ֞ר אֶל־בְּנֵ֣י יִשְׂרָאֵל֮ לֵאמֹר֒ נֶ֗פֶשׁ כִּֽי־תֶחֱטָ֤א בִשְׁגָגָה֙ מִכֹּל֙ מִצְוֺ֣ת יְהֹוָ֔ה אֲשֶׁ֖ר לֹ֣א תֵעָשֶׂ֑ינָה וְעָשָׂ֕ה מֵאַחַ֖ת מֵהֵֽנָּה׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְתֹאמַר: אִם אָדָם יֶחֱטָא בִּשְׁגָגָה וְיַעֲבוֹר עַל אַחַת מִכָּל מִצְווֹת ה' שֶׁאָסוּר לַעֲשׂוֹתָן –
ג אִ֣ם הַכֹּהֵ֧ן הַמָּשִׁ֛יחַ יֶחֱטָ֖א לְאַשְׁמַ֣ת הָעָ֑ם וְהִקְרִ֡יב עַ֣ל חַטָּאתוֹ֩ אֲשֶׁ֨ר חָטָ֜א פַּ֣ר בֶּן־בָּקָ֥ר תָּמִ֛ים לַיהֹוָ֖ה לְחַטָּֽאת׃ אִם הַכֹּהֵן הַגָּדוֹל יֶחֱטָא וְיִגְרוֹם אַשְׁמָה לָעָם, הוּא יַקְרִיב לַה' עַל חֶטְאוֹ שֶׁחָטָא פַּר שָׁלֵם לְלֹא מוּם לְקָרְבַּן חַטָּאת.
ד וְהֵבִ֣יא אֶת־הַפָּ֗ר אֶל־פֶּ֛תַח אֹ֥הֶל מוֹעֵ֖ד לִפְנֵ֣י יְהֹוָ֑ה וְסָמַ֤ךְ אֶת־יָדוֹ֙ עַל־רֹ֣אשׁ הַפָּ֔ר וְשָׁחַ֥ט אֶת־הַפָּ֖ר לִפְנֵ֥י יְהֹוָֽה׃ הוּא יָבִיא אֶת הַפָּר לִפְנֵי ה' אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת, וְיַשְׁעִין אֶת יָדָיו עַל רֹאשׁ הַפָּר וְיִשְׁחַט אֶת הַפָּר לִפְנֵי ה'.
ה וְלָקַ֛ח הַכֹּהֵ֥ן הַמָּשִׁ֖יחַ מִדַּ֣ם הַפָּ֑ר וְהֵבִ֥יא אֹת֖וֹ אֶל־אֹ֥הֶל מוֹעֵֽד׃ הַכֹּהֵן הַגָּדוֹל יִקַּח מִדַּם הַפָּר וְיָבִיא אוֹתוֹ לְאוֹהֶל הַהִתְכַּנְּסוּת.
ו וְטָבַ֧ל הַכֹּהֵ֛ן אֶת־אֶצְבָּע֖וֹ בַּדָּ֑ם וְהִזָּ֨ה מִן־הַדָּ֜ם שֶׁ֤בַע פְּעָמִים֙ לִפְנֵ֣י יְהֹוָ֔ה אֶת־פְּנֵ֖י פָּרֹ֥כֶת הַקֹּֽדֶשׁ׃ הַכֹּהֵן יַטְבִּיל אֶת אֶצְבָּעוֹ בַּדָּם וְיַזֶּה מֵהַדָּם שֶׁבַע פְּעָמִים לִפְנֵי ה' לְעֵבֶר הַפָּרוֹכֶת שֶׁלִּפְנֵי הַקּוֹדֶשׁ.
ז וְנָתַן֩ הַכֹּהֵ֨ן מִן־הַדָּ֜ם עַל־קַ֠רְנ֠וֹת מִזְבַּ֨ח קְטֹ֤רֶת הַסַּמִּים֙ לִפְנֵ֣י יְהֹוָ֔ה אֲשֶׁ֖ר בְּאֹ֣הֶל מוֹעֵ֑ד וְאֵ֣ת ׀ כׇּל־דַּ֣ם הַפָּ֗ר יִשְׁפֹּךְ֙ אֶל־יְסוֹד֙ מִזְבַּ֣ח הָעֹלָ֔ה אֲשֶׁר־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ הַכֹּהֵן יָשִׂים מֵהַדָּם עַל קַרְנוֹת מִזְבַּח קְטוֹרֶת הַבְּשָׂמִים שֶׁלִּפְנֵי ה' בְּאוֹהֶל הַהִתְכַּנְּסוּת, וְאֶת כָּל שְׁאָר דַּם הַפָּר הוּא יִשְׁפּוֹךְ עַל יְסוֹד מִזְבַּח הָעוֹלָה שֶׁמּוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.
ח וְאֶת־כׇּל־חֵ֛לֶב פַּ֥ר הַֽחַטָּ֖את יָרִ֣ים מִמֶּ֑נּוּ אֶת־הַחֵ֙לֶב֙ הַֽמְכַסֶּ֣ה עַל־הַקֶּ֔רֶב וְאֵת֙ כׇּל־הַחֵ֔לֶב אֲשֶׁ֖ר עַל־הַקֶּֽרֶב׃ וְיַפְרִישׁ מִמֶּנּוּ אֶת כָּל חֵלֶב פַּר הַחַטָּאת – אֶת הַחֵלֶב שֶׁמְּכַסֶּה עַל הַמֵּעַיִם וְאֶת כָּל הַחֵלֶב שֶׁעַל הַמֵּעַיִם.
ט וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־הַחֵ֙לֶב֙ אֲשֶׁ֣ר עֲלֵיהֶ֔ן אֲשֶׁ֖ר עַל־הַכְּסָלִ֑ים וְאֶת־הַיֹּתֶ֙רֶת֙ עַל־הַכָּבֵ֔ד עַל־הַכְּלָי֖וֹת יְסִירֶֽנָּה׃ וְאֶת שְׁתֵּי הַכְּלָיוֹת וְאֶת הַחֵלֶב שֶׁעֲלֵיהֶן וְעַל הַמָּתְנַיִם, וְאֶת הַתּוֹסֶפֶת שֶׁעַל הַכָּבֵד שֶׁיָּסִיר אוֹתָהּ עִם הַכְּלָיוֹת.
י כַּאֲשֶׁ֣ר יוּרַ֔ם מִשּׁ֖וֹר זֶ֣בַח הַשְּׁלָמִ֑ים וְהִקְטִירָם֙ הַכֹּהֵ֔ן עַ֖ל מִזְבַּ֥ח הָעֹלָֽה׃ כְּמוֹ שֶׁמֻּפְרָשׁ מִשּׁוֹר זֶבַח הַשְּׁלָמִים, הַכֹּהֵן יַקְטִיר אוֹתָם עַל מִזְבַּח הָעוֹלָה.
יא וְאֶת־ע֤וֹר הַפָּר֙ וְאֶת־כׇּל־בְּשָׂר֔וֹ עַל־רֹאשׁ֖וֹ וְעַל־כְּרָעָ֑יו וְקִרְבּ֖וֹ וּפִרְשֽׁוֹ׃ וְאֶת עוֹר הַפָּר וְאֶת כָּל בְּשָׂרוֹ עִם רֹאשׁוֹ וְעִם רַגְלָיו וּמֵעָיו, וְהָאוֹכֶל הַמְּעֻכָּל שֶׁבּוֹ.
יב וְהוֹצִ֣יא אֶת־כׇּל־הַ֠פָּ֠ר אֶל־מִח֨וּץ לַֽמַּחֲנֶ֜ה אֶל־מָק֤וֹם טָהוֹר֙ אֶל־שֶׁ֣פֶךְ הַדֶּ֔שֶׁן וְשָׂרַ֥ף אֹת֛וֹ עַל־עֵצִ֖ים בָּאֵ֑שׁ עַל־שֶׁ֥פֶךְ הַדֶּ֖שֶׁן יִשָּׂרֵֽף׃ – אֶת כָּל הַפָּר יוֹצִיא אֶל מִחוּץ לַמַּחֲנֶה לְמָקוֹם טָהוֹר – לִמְקוֹם שְׁפִיכַת הָאֵפֶר, וְיִשְׂרוֹף אוֹתוֹ עַל עֵצִים בָּאֵשׁ – הוּא יִשָּׂרֵף בַּמָּקוֹם שֶׁשּׁוֹפְכִים אֶת הָאֵפֶר.
פ פָּרָשַׁת חַטַּאת הַקָּהָל
יג וְאִ֨ם כׇּל־עֲדַ֤ת יִשְׂרָאֵל֙ יִשְׁגּ֔וּ וְנֶעְלַ֣ם דָּבָ֔ר מֵעֵינֵ֖י הַקָּהָ֑ל וְ֠עָשׂ֠וּ אַחַ֨ת מִכׇּל־מִצְוֺ֧ת יְהֹוָ֛ה אֲשֶׁ֥ר לֹא־תֵעָשֶׂ֖ינָה וְאָשֵֽׁמוּ׃ וְאִם כָּל קְהַל יִשְׂרָאֵל יִטְעוּ כִּי מַנְהִיגֵי הַקָּהָל יִשְׁכְּחוּ דָּבָר, וְיַעַבְרוּ עַל אַחַת מִכָּל מִצְווֹת ה' שֶׁאָסוּר לַעֲשׂוֹתָן וְיִהְיוּ אֲשֵׁמִים,
יד וְנֽוֹדְעָה֙ הַֽחַטָּ֔את אֲשֶׁ֥ר חָטְא֖וּ עָלֶ֑יהָ וְהִקְרִ֨יבוּ הַקָּהָ֜ל פַּ֤ר בֶּן־בָּקָר֙ לְחַטָּ֔את וְהֵבִ֣יאוּ אֹת֔וֹ לִפְנֵ֖י אֹ֥הֶל מוֹעֵֽד׃ וְיִוָּדַע לָהֶם הַחֵטְא שֶׁחָטְאוּ וְהִתְחַיְּבוּ עָלָיו, הַקָּהָל יַקְרִיבוּ פַּר לְקָרְבַּן חַטָּאת וְיָבִיאוּ אוֹתוֹ אֶל לִפְנֵי אוֹהֶל הַהִתְכַּנְּסוּת.
טו וְ֠סָמְכ֠וּ זִקְנֵ֨י הָעֵדָ֧ה אֶת־יְדֵיהֶ֛ם עַל־רֹ֥אשׁ הַפָּ֖ר לִפְנֵ֣י יְהֹוָ֑ה וְשָׁחַ֥ט אֶת־הַפָּ֖ר לִפְנֵ֥י יְהֹוָֽה׃ זִקְנֵי הַצִּבּוּר יַשְׁעִינוּ אֶת יְדֵיהֶם עַל רֹאשׁ הַפָּר לִפְנֵי ה', וְיִשְׁחֲטוּ אֶת הַפָּר לִפְנֵי ה'.
טז וְהֵבִ֛יא הַכֹּהֵ֥ן הַמָּשִׁ֖יחַ מִדַּ֣ם הַפָּ֑ר אֶל־אֹ֖הֶל מוֹעֵֽד׃ הַכֹּהֵן הַגָּדוֹל יָבִיא מִדַּם הַפָּר לְאוֹהֶל הַהִתְכַּנְּסוּת.
יז וְטָבַ֧ל הַכֹּהֵ֛ן אֶצְבָּע֖וֹ מִן־הַדָּ֑ם וְהִזָּ֞ה שֶׁ֤בַע פְּעָמִים֙ לִפְנֵ֣י יְהֹוָ֔ה אֵ֖ת פְּנֵ֥י הַפָּרֹֽכֶת׃ הַכֹּהֵן יַטְבִּיל אֶת אֶצְבָּעוֹ וְיִקַּח מֵהַדָּם, וְיַזֶּה שֶׁבַע פְּעָמִים לִפְנֵי ה' לְעֵבֶר הַפָּרוֹכֶת.
יח וּמִן־הַדָּ֞ם יִתֵּ֣ן ׀ עַל־קַרְנֹ֣ת הַמִּזְבֵּ֗חַ אֲשֶׁר֙ לִפְנֵ֣י יְהֹוָ֔ה אֲשֶׁ֖ר בְּאֹ֣הֶל מוֹעֵ֑ד וְאֵ֣ת כׇּל־הַדָּ֗ם יִשְׁפֹּךְ֙ אֶל־יְסוֹד֙ מִזְבַּ֣ח הָעֹלָ֔ה אֲשֶׁר־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ הוּא יָשִׂים מֵהַדָּם עַל קַרְנוֹת הַמִּזְבֵּחַ שֶׁלִּפְנֵי ה' שֶׁבְּאוֹהֶל הַהִתְכַּנְּסוּת, וְאֶת כָּל שְׁאָר הַדָּם יִשְׁפּוֹךְ עַל יְסוֹד מִזְבַּח הָעוֹלָה שֶׁמּוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.
יט וְאֵ֥ת כׇּל־חֶלְבּ֖וֹ יָרִ֣ים מִמֶּ֑נּוּ וְהִקְטִ֖יר הַמִּזְבֵּֽחָה׃ וְאֶת כָּל חֶלְבּוֹ יַפְרִישׁ מִמֶּנּוּ וְיַקְטִיר אוֹתוֹ עַל הַמִּזְבֵּחַ.
כ וְעָשָׂ֣ה לַפָּ֔ר כַּאֲשֶׁ֤ר עָשָׂה֙ לְפַ֣ר הַֽחַטָּ֔את כֵּ֖ן יַעֲשֶׂה־לּ֑וֹ וְכִפֶּ֧ר עֲלֵהֶ֛ם הַכֹּהֵ֖ן וְנִסְלַ֥ח לָהֶֽם׃ וְיַעֲשֶׂה לַפָּר כְּמוֹ שֶׁעָשָׂה לְפַר הַחַטָּאת – כָּךְ יַעֲשֶׂה לוֹ, וְהַכֹּהֵן יְכַפֵּר עַל הַקָּהָל וְיִסָּלַח לָהֶם.
כא וְהוֹצִ֣יא אֶת־הַפָּ֗ר אֶל־מִחוּץ֙ לַֽמַּחֲנֶ֔ה וְשָׂרַ֣ף אֹת֔וֹ כַּאֲשֶׁ֣ר שָׂרַ֔ף אֵ֖ת הַפָּ֣ר הָרִאשׁ֑וֹן חַטַּ֥את הַקָּהָ֖ל הֽוּא׃ הוּא יוֹצִיא אֶת הַפָּר אֶל מִחוּץ לַמַּחֲנֶה וְיִשְׂרוֹף אוֹתוֹ כְּמוֹ שֶׁשָּׂרַף אֶת הַפָּר הָרִאשׁוֹן, זֶה קָרְבַּן הַחַטָּאת שֶׁל הַקָּהָל.
פ פָּרָשַׁת חַטַּאת הַנָּשִׂיא
כב אֲשֶׁ֥ר נָשִׂ֖יא יֶֽחֱטָ֑א וְעָשָׂ֡ה אַחַ֣ת מִכׇּל־מִצְוֺת֩ יְהֹוָ֨ה אֱלֹהָ֜יו אֲשֶׁ֧ר לֹא־תֵעָשֶׂ֛ינָה בִּשְׁגָגָ֖ה וְאָשֵֽׁם׃ כַּאֲשֶׁר נָשִׂיא יֶחֱטָא, וְיַעֲבוֹר בְּטָעוּת עַל אַחַת מִכָּל מִצְווֹת ה' אֱלֹהָיו שֶׁאָסוּר לַעֲשׂוֹת אוֹתָן וְיִהְיֶה אָשֵׁם,
כג אֽוֹ־הוֹדַ֤ע אֵלָיו֙ חַטָּאת֔וֹ אֲשֶׁ֥ר חָטָ֖א בָּ֑הּ וְהֵבִ֧יא אֶת־קׇרְבָּנ֛וֹ שְׂעִ֥יר עִזִּ֖ים זָכָ֥ר תָּמִֽים׃ כְּשֶׁיִתְגַּלֶּה לוֹ חֶטְאוֹ שֶׁחָטָא בּוֹ, יָבִיא אֶת קָרְבָּנוֹ – תַּיִשׁ זָכָר לְלֹא מוּם.
כד וְסָמַ֤ךְ יָדוֹ֙ עַל־רֹ֣אשׁ הַשָּׂעִ֔יר וְשָׁחַ֣ט אֹת֗וֹ בִּמְק֛וֹם אֲשֶׁר־יִשְׁחַ֥ט אֶת־הָעֹלָ֖ה לִפְנֵ֣י יְהֹוָ֑ה חַטָּ֖את הֽוּא׃ הוּא יַשְׁעִין אֶת יָדָיו עַל רֹאשׁ הַתַּיִשׁ וְיִשְׁחַט אוֹתוֹ בַּמָּקוֹם שֶׁשּׁוֹחֲטִים אֶת הָעוֹלָה לִפְנֵי ה', הוּא קָרְבַּן חַטָּאת.
כה וְלָקַ֨ח הַכֹּהֵ֜ן מִדַּ֤ם הַֽחַטָּאת֙ בְּאֶצְבָּע֔וֹ וְנָתַ֕ן עַל־קַרְנֹ֖ת מִזְבַּ֣ח הָעֹלָ֑ה וְאֶת־דָּמ֣וֹ יִשְׁפֹּ֔ךְ אֶל־יְס֖וֹד מִזְבַּ֥ח הָעֹלָֽה׃ הַכֹּהֵן יִקַּח מִדַּם הַחַטָּאת בְּאֶצְבָּעוֹ וְיָשִׂים עַל קַרְנוֹת מִזְבַּח הָעוֹלָה, וְאֶת שְׁאָר הַדָּם יִשְׁפּוֹךְ עַל יְסוֹד מִזְבַּח הָעוֹלָה.
כו וְאֶת־כׇּל־חֶלְבּוֹ֙ יַקְטִ֣יר הַמִּזְבֵּ֔חָה כְּחֵ֖לֶב זֶ֣בַח הַשְּׁלָמִ֑ים וְכִפֶּ֨ר עָלָ֧יו הַכֹּהֵ֛ן מֵחַטָּאת֖וֹ וְנִסְלַ֥ח לֽוֹ׃ וְאֶת כָּל חֶלְבּוֹ יַקְטִיר עַל הַמִּזְבֵּחַ כְּמוֹ חֵלֶב זֶבַח הַשְּׁלָמִים, הַכֹּהֵן יְכַפֵּר עָלָיו מֵחֶטְאוֹ וְיִסָּלַח לוֹ.

ששי[עריכה]

פ פָּרָשַׁת חַטַּאת אָדָם פָּשׁוּט
כז וְאִם־נֶ֧פֶשׁ אַחַ֛ת תֶּחֱטָ֥א בִשְׁגָגָ֖ה מֵעַ֣ם הָאָ֑רֶץ בַּ֠עֲשֹׂתָ֠הּ אַחַ֨ת מִמִּצְוֺ֧ת יְהֹוָ֛ה אֲשֶׁ֥ר לֹא־תֵעָשֶׂ֖ינָה וְאָשֵֽׁם׃ וְאִם אָדָם פָּשׁוּט יֶחֱטָא בְּטָעוּת וְיַעֲבוֹר עַל אַחַת מִמִּצְווֹת ה' שֶׁאָסוּר לַעֲשׂוֹת אוֹתָן וְיִהְיֶה אָשֵׁם.
כח א֚וֹ הוֹדַ֣ע אֵלָ֔יו חַטָּאת֖וֹ אֲשֶׁ֣ר חָטָ֑א וְהֵבִ֨יא קׇרְבָּנ֜וֹ שְׂעִירַ֤ת עִזִּים֙ תְּמִימָ֣ה נְקֵבָ֔ה עַל־חַטָּאת֖וֹ אֲשֶׁ֥ר חָטָֽא׃ כְּשֶׁיִּתְגַּלֶּה לוֹ חֶטְאוֹ שֶׁחָטָא, הוּא יָבִיא אֶת קָרְבָּנוֹ עַל חֶטְאוֹ שֶׁחָטָא – עֵז נְקֵבָה לְלֹא מוּם.
כט וְסָמַךְ֙ אֶת־יָד֔וֹ עַ֖ל רֹ֣אשׁ הַֽחַטָּ֑את וְשָׁחַט֙ אֶת־הַ֣חַטָּ֔את בִּמְק֖וֹם הָעֹלָֽה׃ הוּא יַשְׁעִין אֶת יָדָיו עַל רֹאשׁ הַחַטָּאת וְיִשְׁחַט אֶת הַחַטָּאת בַּמָּקוֹם שֶׁשּׁוֹחֲטִים אֶת הָעוֹלָה.
ל וְלָקַ֨ח הַכֹּהֵ֤ן מִדָּמָהּ֙ בְּאֶצְבָּע֔וֹ וְנָתַ֕ן עַל־קַרְנֹ֖ת מִזְבַּ֣ח הָעֹלָ֑ה וְאֶת־כׇּל־דָּמָ֣הּ יִשְׁפֹּ֔ךְ אֶל־יְס֖וֹד הַמִּזְבֵּֽחַ׃ הַכֹּהֵן יִקַּח מִדָּמָהּ בְּאֶצְבָּעוֹ וְיָשִׂים עַל קַרְנוֹת מִזְבַּח הָעוֹלָה, וְאֶת כָּל שְׁאָר הַדָּם הוּא יִשְׁפּוֹךְ עַל יְסוֹד הַמִּזְבֵּחַ.
לא וְאֶת־כׇּל־חֶלְבָּ֣הּ יָסִ֗יר כַּאֲשֶׁ֨ר הוּסַ֣ר חֵ֘לֶב֮ מֵעַ֣ל זֶ֣בַח הַשְּׁלָמִים֒ וְהִקְטִ֤יר הַכֹּהֵן֙ הַמִּזְבֵּ֔חָה לְרֵ֥יחַ נִיחֹ֖חַ לַיהֹוָ֑ה וְכִפֶּ֥ר עָלָ֛יו הַכֹּהֵ֖ן וְנִסְלַ֥ח לֽוֹ׃ וְיָסִיר אֶת כָּל הַחֵלֶב שֶׁלָּהּ כְּמוֹ שֶׁהוּסַר הַחֵלֶב מִתּוֹךְ קָרְבַּן הַשְּׁלָמִים וְהַכֹּהֵן יַקְטִיר אוֹתוֹ עַל הַמִּזְבֵּחַ לְנַחַת רוּחַ לַה', וְהַכֹּהֵן יְכַפֵּר עָלָיו וְיִסָּלַח לוֹ.
פ
לב וְאִם־כֶּ֛בֶשׂ יָבִ֥יא קׇרְבָּנ֖וֹ לְחַטָּ֑את נְקֵבָ֥ה תְמִימָ֖ה יְבִיאֶֽנָּה׃ אִם יָבִיא כֶּבֶשׂ לְקָרְבָּנוֹ לְחַטָּאת, יָבִיא נְקֵבָה לְלֹא מוּם.
לג וְסָמַךְ֙ אֶת־יָד֔וֹ עַ֖ל רֹ֣אשׁ הַֽחַטָּ֑את וְשָׁחַ֤ט אֹתָהּ֙ לְחַטָּ֔את בִּמְק֕וֹם אֲשֶׁ֥ר יִשְׁחַ֖ט אֶת־הָעֹלָֽה׃ הוּא יַשְׁעִין אֶת יָדָיו עַל רֹאשׁ קָרְבַּן הַחַטָּאת, וְיִשְׁחַט אוֹתָהּ לְקָרְבַּן חַטָּאת בַּמָּקוֹם שֶׁשּׁוֹחֲטִים אֶת קָרְבַּן הָעוֹלָה.
לד וְלָקַ֨ח הַכֹּהֵ֜ן מִדַּ֤ם הַֽחַטָּאת֙ בְּאֶצְבָּע֔וֹ וְנָתַ֕ן עַל־קַרְנֹ֖ת מִזְבַּ֣ח הָעֹלָ֑ה וְאֶת־כׇּל־דָּמָ֣הּ יִשְׁפֹּ֔ךְ אֶל־יְס֖וֹד הַמִּזְבֵּֽחַ׃ הַכֹּהֵן יִקַּח מִדַּם הַחַטָּאת בְּאֶצְבָּעוֹ וְיָשִׂים עַל קַרְנוֹת מִזְבַּח הָעוֹלָה, וְאֶת כָּל שְׁאָר הַדָּם יִשְׁפּוֹךְ עַל יְסוֹד הַמִּזְבֵּחַ.
לה וְאֶת־כׇּל־חֶלְבָּ֣הּ יָסִ֗יר כַּאֲשֶׁ֨ר יוּסַ֥ר חֵֽלֶב־הַכֶּ֘שֶׂב֮ מִזֶּ֣בַח הַשְּׁלָמִים֒ וְהִקְטִ֨יר הַכֹּהֵ֤ן אֹתָם֙ הַמִּזְבֵּ֔חָה עַ֖ל אִשֵּׁ֣י יְהֹוָ֑ה וְכִפֶּ֨ר עָלָ֧יו הַכֹּהֵ֛ן עַל־חַטָּאת֥וֹ אֲשֶׁר־חָטָ֖א וְנִסְלַ֥ח לֽוֹ׃ וְיָסִיר אֶת כָּל הַחֵלֶב שֶׁלָּהּ כְּמוֹ שֶׁמְּסִירִים אֶת חֵלֶב הַכֶּבֶשׂ מִזֶּבַח הַשְּׁלָמִים וְהַכֹּהֵן יַקְטִיר אוֹתוֹ עַל הַמִּזְבֵּחַ עִם הַקָּרְבְּנוֹת הָעוֹלִים בָּאֵשׁ לַה', וְהַכֹּהֵן יְכַפֵּר עָלָיו עַל חֶטְאוֹ שֶׁחָטָא, וְיִסָּלַח לוֹ.
פ פָּרָשַׁת חַטַּאת עוֹלֶה וְיוֹרֵד
ה א וְנֶ֣פֶשׁ כִּֽי־תֶחֱטָ֗א וְשָֽׁמְעָה֙ ק֣וֹל אָלָ֔ה וְה֣וּא עֵ֔ד א֥וֹ רָאָ֖ה א֣וֹ יָדָ֑ע אִם־ל֥וֹא יַגִּ֖יד וְנָשָׂ֥א עֲוֺנֽוֹ׃ וְאִם אָדָם יֶחֱטָא – שֶׁהִשְׁבִּיעוּ אוֹתוֹ לְהָעִיד וְהוּא אָכֵן הָיָה עֵד – אוֹ שֶׁרָאָה אוֹ שֶׁיָּדַע, אִם הוּא לֹא יָעִיד הוּא יְקַבֵּל עוֹנֶשׁ עַל עֲווֹנוֹ.
ב א֣וֹ נֶ֗פֶשׁ אֲשֶׁ֣ר תִּגַּע֮ בְּכׇל־דָּבָ֣ר טָמֵא֒ אוֹ֩ בְנִבְלַ֨ת חַיָּ֜ה טְמֵאָ֗ה א֚וֹ בְּנִבְלַת֙ בְּהֵמָ֣ה טְמֵאָ֔ה א֕וֹ בְּנִבְלַ֖ת שֶׁ֣רֶץ טָמֵ֑א וְנֶעְלַ֣ם מִמֶּ֔נּוּ וְה֥וּא טָמֵ֖א וְאָשֵֽׁם׃ אוֹ אָדָם שֶׁיִּגַּע בְּכָל דָּבָר טָמֵא אוֹ בִּנְבֵלָה שֶׁל חַיָּה טְמֵאָה אוֹ בִּנְבֵלָה שֶׁל בְּהֵמָה טְמֵאָה אוֹ בִּנְבֵלָה שֶׁל שֶׁרֶץ טָמֵא, וְהוּא שָׁכַח וְחָטָא וְנִכְנַס לַמִּקְדָּשׁ אוֹ אָכַל קָדָשִׁים בְּעוֹדוֹ טָמֵא.
ג א֣וֹ כִ֤י יִגַּע֙ בְּטֻמְאַ֣ת אָדָ֔ם לְכֹל֙ טֻמְאָת֔וֹ אֲשֶׁ֥ר יִטְמָ֖א בָּ֑הּ וְנֶעְלַ֣ם מִמֶּ֔נּוּ וְה֥וּא יָדַ֖ע וְאָשֵֽׁם׃ אוֹ אִם יִגַּע בְּטֻמְאַת אָדָם – בְּאַחַת הַטֻּמְאוֹת שֶׁיִּטָּמֵא בָּהֵן, וְהוּא יָדַע וְשָׁכַח וְחָטָא וְנִכְנַס לַמִּקְדָּשׁ אוֹ אָכַל קָדָשִׁים.
ד א֣וֹ נֶ֡פֶשׁ כִּ֣י תִשָּׁבַע֩ לְבַטֵּ֨א בִשְׂפָתַ֜יִם לְהָרַ֣ע ׀ א֣וֹ לְהֵיטִ֗יב לְ֠כֹ֠ל אֲשֶׁ֨ר יְבַטֵּ֧א הָאָדָ֛ם בִּשְׁבֻעָ֖ה וְנֶעְלַ֣ם מִמֶּ֑נּוּ וְהוּא־יָדַ֥ע וְאָשֵׁ֖ם לְאַחַ֥ת מֵאֵֽלֶּה׃ אוֹ אִם אָדָם יְבַטֵּא שְׁבוּעָה בְּפִיו לְהָרַע לְעַצְמוֹ אוֹ לְהֵיטִיב – כָּל מַה שֶּׁאָדָם יְבַטֵּא בִּשְׁבוּעָה, וְהוּא יָדַע וְשָׁכַח וְחָטָא בְּאַחַת מִשְּׁבוּעוֹת אֵלֶּה.
ה וְהָיָ֥ה כִֽי־יֶאְשַׁ֖ם לְאַחַ֣ת מֵאֵ֑לֶּה וְהִ֨תְוַדָּ֔ה אֲשֶׁ֥ר חָטָ֖א עָלֶֽיהָ׃ אִם יֶחֱטָא בְּאַחַת מֵהָעֲבֵרוֹת הָאֵלֶּה – יִתְוַדֶּה עַל הַחֵטְא שֶׁחָטָא בּוֹ.
ו וְהֵבִ֣יא אֶת־אֲשָׁמ֣וֹ לַיהֹוָ֡ה עַ֣ל חַטָּאתוֹ֩ אֲשֶׁ֨ר חָטָ֜א נְקֵבָ֨ה מִן־הַצֹּ֥אן כִּשְׂבָּ֛ה אֽוֹ־שְׂעִירַ֥ת עִזִּ֖ים לְחַטָּ֑את וְכִפֶּ֥ר עָלָ֛יו הַכֹּהֵ֖ן מֵחַטָּאתֽוֹ׃ וְיָבִיא אֶת אֲשָׁמוֹ לַה' עַל חֶטְאוֹ שֶׁחָטָא: נְקֵבָה מֵהַצֹּאן – כִּבְשָׂה אוֹ עִזָּה לְחַטָּאת, וְהַכֹּהֵן יְכַפֵּר עָלָיו מֵחֶטְאוֹ.
ז וְאִם־לֹ֨א תַגִּ֣יעַ יָדוֹ֮ דֵּ֣י שֶׂה֒ וְהֵבִ֨יא אֶת־אֲשָׁמ֜וֹ אֲשֶׁ֣ר חָטָ֗א שְׁתֵּ֥י תֹרִ֛ים אֽוֹ־שְׁנֵ֥י בְנֵֽי־יוֹנָ֖ה לַֽיהֹוָ֑ה אֶחָ֥ד לְחַטָּ֖את וְאֶחָ֥ד לְעֹלָֽה׃ וְאִם לֹא יִהְיֶה לוֹ מַסְפִּיק כֶּסֶף לִקְנוֹת כִּבְשָׂה אוֹ עִזָּה, יָבִיא לַה' אֶת קָרְבַּן אֲשָׁמוֹ עַל חֶטְאוֹ שֶׁחָטָא – שְׁתֵּי תּוֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה, אֶחָד לְחַטָּאת וְאֶחָד לְעוֹלָה.
ח וְהֵבִ֤יא אֹתָם֙ אֶל־הַכֹּהֵ֔ן וְהִקְרִ֛יב אֶת־אֲשֶׁ֥ר לַחַטָּ֖את רִאשׁוֹנָ֑ה וּמָלַ֧ק אֶת־רֹאשׁ֛וֹ מִמּ֥וּל עׇרְפּ֖וֹ וְלֹ֥א יַבְדִּֽיל׃ הוּא יָבִיא אוֹתָם לַכֹּהֵן וְיַקְרִיב אֶת שֶׁל הַחַטָּאת תְּחִלָּה, וְיַחֲתוֹךְ אֶת רֹאשׁוֹ מִצַּד עָרְפּוֹ בְּצִפּוֹרֶן הַבּוֹהֶן וְלֹא יַפְרִיד אֶת הָרֹאשׁ לְגַמְרֵי.
ט וְהִזָּ֞ה מִדַּ֤ם הַחַטָּאת֙ עַל־קִ֣יר הַמִּזְבֵּ֔חַ וְהַנִּשְׁאָ֣ר בַּדָּ֔ם יִמָּצֵ֖ה אֶל־יְס֣וֹד הַמִּזְבֵּ֑חַ חַטָּ֖את הֽוּא׃ וְיַזֶּה מִדַּם הַחַטָּאת עַל קִיר הַמִּזְבֵּחַ, וּמַה שֶּׁנִּשְׁאָר מֵהַדָּם יִסָּחֵט עַל יְסוֹד הַמִּזְבֵּחַ – הוּא חַטָּאת.
י וְאֶת־הַשֵּׁנִ֛י יַעֲשֶׂ֥ה עֹלָ֖ה כַּמִּשְׁפָּ֑ט וְכִפֶּ֨ר עָלָ֧יו הַכֹּהֵ֛ן מֵחַטָּאת֥וֹ אֲשֶׁר־חָטָ֖א וְנִסְלַ֥ח לֽוֹ׃ וְאֶת הַשֵּׁנִי יַעֲשֶׂה עוֹלָה כַּדִּין, וְהַכֹּהֵן יְכַפֵּר עָלָיו מֵחֶטְאוֹ שֶׁחָטָא, וְיִסָּלַח לוֹ.

ס
יא וְאִם־לֹא֩ תַשִּׂ֨יג יָד֜וֹ לִשְׁתֵּ֣י תֹרִ֗ים אוֹ֮ לִשְׁנֵ֣י בְנֵי־יוֹנָה֒ וְהֵבִ֨יא אֶת־קׇרְבָּנ֜וֹ אֲשֶׁ֣ר חָטָ֗א עֲשִׂירִ֧ת הָאֵפָ֛ה סֹ֖לֶת לְחַטָּ֑את לֹא־יָשִׂ֨ים עָלֶ֜יהָ שֶׁ֗מֶן וְלֹא־יִתֵּ֤ן עָלֶ֙יהָ֙ לְבֹנָ֔ה כִּ֥י חַטָּ֖את הִֽוא׃ וְאִם לֹא יַשִּׂיג כֶּסֶף לִשְׁתֵּי תּוֹרִים אוֹ לִשְׁנֵי בְּנֵי יוֹנָה, יָבִיא אֶת קָרְבָּנוֹ עַל שֶׁחָטָא, עֲשִׂירִית אֵיפָה (כִּ2.16 לִיטֶר) סוֹלֶת לְחַטָּאת, לֹא יָשִׂים עָלֶיהָ שֶׁמֶן וְלֹא יָשִׂים עָלֶיהָ לְבוֹנָה כִּי הִיא קָרְבַּן חַטָּאת.
יב וֶהֱבִיאָהּ֮ אֶל־הַכֹּהֵן֒ וְקָמַ֣ץ הַכֹּהֵ֣ן ׀ מִ֠מֶּ֠נָּה מְל֨וֹא קֻמְצ֜וֹ אֶת־אַזְכָּרָתָהּ֙ וְהִקְטִ֣יר הַמִּזְבֵּ֔חָה עַ֖ל אִשֵּׁ֣י יְהֹוָ֑ה חַטָּ֖את הִֽוא׃ הוּא יָבִיא אוֹתָהּ לַכֹּהֵן, הַכֹּהֵן יִקַּח מִמֶּנָּה חוֹפֶן מָלֵא – שֶׁגּוֹרֵם לְהִזָּכֵר לִפְנֵי ה', וְיַקְטִיר עַל הַמִּזְבֵּחַ עַל הָאֵשׁ שֶׁל קָרְבְּנוֹת ה', הִיא חַטָּאת.
יג וְכִפֶּר֩ עָלָ֨יו הַכֹּהֵ֜ן עַל־חַטָּאת֧וֹ אֲשֶׁר־חָטָ֛א מֵֽאַחַ֥ת מֵאֵ֖לֶּה וְנִסְלַ֣ח ל֑וֹ וְהָיְתָ֥ה לַכֹּהֵ֖ן כַּמִּנְחָֽה׃ הַכֹּהֵן יְכַפֵּר עָלָיו מֵחֶטְאוֹ שֶׁחָטָא – שֶׁעָבַר עַל אַחַת מֵהָעֲבֵרוֹת הָאֵלֶּה, וְיִסָּלַח לוֹ, וְהַסּוֹלֶת תִּנָּתֵן לַכֹּהֵן – כְּמוֹ הַמִּנְחָה.
ס פָּרָשַׁת אֲשַׁם מְעִילוֹת
יד וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
טו נֶ֚פֶשׁ כִּֽי־תִמְעֹ֣ל מַ֔עַל וְחָֽטְאָה֙ בִּשְׁגָגָ֔ה מִקׇּדְשֵׁ֖י יְהֹוָ֑ה וְהֵבִיא֩ אֶת־אֲשָׁמ֨וֹ לַֽיהֹוָ֜ה אַ֧יִל תָּמִ֣ים מִן־הַצֹּ֗אן בְּעֶרְכְּךָ֛ כֶּֽסֶף־שְׁקָלִ֥ים בְּשֶֽׁקֶל־הַקֹּ֖דֶשׁ לְאָשָֽׁם׃ אִם אָדָם יִמְעוֹל וְיֶחֱטָא בְּטָעוּת וְיֶהֱנֶה מִקָּדְשֵׁי ה', הוּא יָבִיא לַה' קָרְבָּן לְכַפֵּר עַל אַשְׁמָתוֹ אַיִל מֵהַצֹּאן לְלֹא מוּם שֶׁשָּׁוֶה כֶּסֶף בְּמִשְׁקַל שְׁנֵי שְׁקָלִים שֶׁל שֶׁקֶל הַקּוֹדֶשׁ (28.2 גְּרַם).
טז וְאֵ֣ת אֲשֶׁר֩ חָטָ֨א מִן־הַקֹּ֜דֶשׁ יְשַׁלֵּ֗ם וְאֶת־חֲמִֽישִׁתוֹ֙ יוֹסֵ֣ף עָלָ֔יו וְנָתַ֥ן אֹת֖וֹ לַכֹּהֵ֑ן וְהַכֹּהֵ֗ן יְכַפֵּ֥ר עָלָ֛יו בְּאֵ֥יל הָאָשָׁ֖ם וְנִסְלַ֥ח לֽוֹ׃ וִישַׁלֵּם אֶת הַשּׁוֹוִי שֶׁל מַה שֶּׁחָטָא וְנֶהֱנָה מֵהַקּוֹדֶשׁ, וְיוֹסִיף עָלָיו חֲמִישִׁית וְיִתֵּן אוֹתוֹ לַכֹּהֵן, וְהַכֹּהֵן יְכַפֵּר עָלָיו בְּאֵיל הָאָשָׁם וְיִסָּלַח לוֹ.
פ פָּרָשַׁת אָשָׁם תָּלוּי
יז וְאִם־נֶ֙פֶשׁ֙ כִּ֣י תֶֽחֱטָ֔א וְעָֽשְׂתָ֗ה אַחַת֙ מִכׇּל־מִצְוֺ֣ת יְהֹוָ֔ה אֲשֶׁ֖ר לֹ֣א תֵעָשֶׂ֑ינָה וְלֹֽא־יָדַ֥ע וְאָשֵׁ֖ם וְנָשָׂ֥א עֲוֺנֽוֹ׃ וְאִם אָדָם יֶחֱטָא וְיַעֲבוֹר עַל אַחַת מִכָּל מִצְווֹת ה' שֶׁאָסוּר לַעֲשׂוֹתָן, וְלֹא יֵדַע בְּוַדָּאוּת אִם חָטָא, הוּא יִתְחַיֵּב וְיִצְטָרֵךְ לְכַפֵּר עַל עֲווֹנוֹ.
יח וְ֠הֵבִ֠יא אַ֣יִל תָּמִ֧ים מִן־הַצֹּ֛אן בְּעֶרְכְּךָ֥ לְאָשָׁ֖ם אֶל־הַכֹּהֵ֑ן וְכִפֶּר֩ עָלָ֨יו הַכֹּהֵ֜ן עַ֣ל שִׁגְגָת֧וֹ אֲשֶׁר־שָׁגָ֛ג וְה֥וּא לֹֽא־יָדַ֖ע וְנִסְלַ֥ח לֽוֹ׃ הוּא יָבִיא לַכֹּהֵן לְקָרְבַּן אָשָׁם אַיִל שָׁלֵם לְלֹא מוּם מֵהַצֹּאן בָּעֵרֶךְ שֶׁהֻזְכַּר לִפְנֵי כֵן, וְהַכֹּהֵן יְכַפֵּר עָלָיו עַל טָעוּתוֹ שֶׁטָּעָה וְהוּא לֹא יָדַע, וְיִסָּלַח לוֹ.
יט אָשָׁ֖ם ה֑וּא אָשֹׁ֥ם אָשַׁ֖ם לַיהֹוָֽה׃ זֶה קָרְבַּן אָשָׁם, כִּי הוּא חָטָא לַה'.
פ פָּרָשַׁת אֲשַׁם גְּזֵלוֹת
כ וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
כא נֶ֚פֶשׁ כִּ֣י תֶחֱטָ֔א וּמָעֲלָ֥ה מַ֖עַל בַּיהֹוָ֑ה וְכִחֵ֨שׁ בַּעֲמִית֜וֹ בְּפִקָּד֗וֹן אֽוֹ־בִתְשׂ֤וּמֶת יָד֙ א֣וֹ בְגָזֵ֔ל א֖וֹ עָשַׁ֥ק אֶת־עֲמִיתֽוֹ׃ אִם אָדָם יֶחֱטָא וְיִמְעוֹל בַּה' וְיַכְחִישׁ לַחֲבֵרוֹ עַל פִּקָּדוֹן אוֹ שֻׁתָּפוּת אוֹ גֵּזֶל אוֹ שָׂכָר שֶׁהִתְחַיֵּב לַחֲבֵרוֹ.
כב אֽוֹ־מָצָ֧א אֲבֵדָ֛ה וְכִ֥חֶשׁ בָּ֖הּ וְנִשְׁבַּ֣ע עַל־שָׁ֑קֶר עַל־אַחַ֗ת מִכֹּ֛ל אֲשֶׁר־יַעֲשֶׂ֥ה הָאָדָ֖ם לַחֲטֹ֥א בָהֵֽנָּה׃ אוֹ שֶׁמָּצָא אֲבֵדָה וְהִכְחִישׁ שֶׁמָּצָא וְנִשְׁבַּע עַל שֶׁקֶר, אוֹ נִשְׁבַּע לַשֶּׁקֶר עַל אַחַת מִכָּל הָעֲבֵרוֹת בְּמָמוֹן שֶׁאָדָם חוֹטֵא בָּהֵן.
כג וְהָיָה֮ כִּֽי־יֶחֱטָ֣א וְאָשֵׁם֒ וְהֵשִׁ֨יב אֶת־הַגְּזֵלָ֜ה אֲשֶׁ֣ר גָּזָ֗ל א֤וֹ אֶת־הָעֹ֙שֶׁק֙ אֲשֶׁ֣ר עָשָׁ֔ק א֚וֹ אֶת־הַפִּקָּד֔וֹן אֲשֶׁ֥ר הׇפְקַ֖ד אִתּ֑וֹ א֥וֹ אֶת־הָאֲבֵדָ֖ה אֲשֶׁ֥ר מָצָֽא׃ אִם יֶחֱטָא וְיִתְחַיֵּב, הוּא יַחֲזִיר אֶת הַגְּזֵלָה שֶׁגָּזָל אוֹ אֶת הַשָּׂכָר שֶׂהִתְחַיֵּב לְשַׁלֵּם אוֹ אֶת הַפִּקָּדוֹן שֶׁהֻפְקַד אֶצְלוֹ אוֹ אֶת הָאֲבֵדָה שֶׁמָּצָא,

מפטיר[עריכה]

כד א֠וֹ מִכֹּ֞ל אֲשֶׁר־יִשָּׁבַ֣ע עָלָיו֮ לַשֶּׁ֒קֶר֒ וְשִׁלַּ֤ם אֹתוֹ֙ בְּרֹאשׁ֔וֹ וַחֲמִשִׁתָ֖יו יֹסֵ֣ף עָלָ֑יו לַאֲשֶׁ֨ר ה֥וּא ל֛וֹ יִתְּנֶ֖נּוּ בְּי֥וֹם אַשְׁמָתֽוֹ׃ אוֹ אֶת כָּל מַה שֶּׁיִּשָּׁבַע עָלָיו לַשֶּׁקֶר, וִישַׁלֵּם אֶת עֶרְכּוֹ וְיוֹסִיף עָלָיו חֲמִישִׁית, הוּא יִתֵּן אוֹתוֹ בְּיוֹם כַּפָּרַת חֶטְאוֹ לְמִי שֶׁהוּא חַיָּב לוֹ.
כה וְאֶת־אֲשָׁמ֥וֹ יָבִ֖יא לַיהֹוָ֑ה אַ֣יִל תָּמִ֧ים מִן־הַצֹּ֛אן בְּעֶרְכְּךָ֥ לְאָשָׁ֖ם אֶל־הַכֹּהֵֽן׃ וְיָבִיא לַה' אֶת הַקָּרְבָּן לְכַפָּרַת אַשְׁמָתוֹ – יָבִיא לַכֹּהֵן אַיִל לְלֹא מוּם מֵהַצֹּאן שֶׁשָּׁוֶה כְּמוֹ קָרְבַּן אֲשַׁם הַמְּעִילָה.
כו וְכִפֶּ֨ר עָלָ֧יו הַכֹּהֵ֛ן לִפְנֵ֥י יְהֹוָ֖ה וְנִסְלַ֣ח ל֑וֹ עַל־אַחַ֛ת מִכֹּ֥ל אֲשֶֽׁר־יַעֲשֶׂ֖ה לְאַשְׁמָ֥ה בָֽהּ׃ הַכֹּהֵן יְכַפֵּר עָלָיו לִפְנֵי ה' וְיִסָּלַח לוֹ – עַל כָּל אַחַת מֵעֲבֵרוֹת הַמָּמוֹן שֶׁהִתְחַיֵּב עָלֶיהָ.
