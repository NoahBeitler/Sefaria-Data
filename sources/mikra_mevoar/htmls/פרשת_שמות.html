ראשון[עריכה]

פָּרָשַׁת שְׁמוֹת הַבָּאִים
א א וְאֵ֗לֶּה שְׁמוֹת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל הַבָּאִ֖ים מִצְרָ֑יְמָה אֵ֣ת יַעֲקֹ֔ב אִ֥ישׁ וּבֵית֖וֹ בָּֽאוּ׃ וְאֵלֶּה שְׁמוֹת בְּנֵי יִשְׂרָאֵל שֶׁבָּאוּ לְמִצְרַיִם עִם יַעֲקֹב, הֵם בָּאוּ כָּל אֶחָד עִם מִשְׁפַּחְתּוֹ:
ב רְאוּבֵ֣ן שִׁמְע֔וֹן לֵוִ֖י וִיהוּדָֽה׃ רְאוּבֵן שִׁמְעוֹן לֵוִי וִיהוּדָה.
ג יִשָּׂשכָ֥ר זְבוּלֻ֖ן וּבִנְיָמִֽן׃ יִשָּׂשכָר זְבוּלוּן וּבִנְיָמִין.
ד דָּ֥ן וְנַפְתָּלִ֖י גָּ֥ד וְאָשֵֽׁר׃ דָּן וְנַפְתָּלִי גָּד וְאָשֵׁר.
ה וַֽיְהִ֗י כׇּל־נֶ֛פֶשׁ יֹצְאֵ֥י יֶֽרֶךְ־יַעֲקֹ֖ב שִׁבְעִ֣ים נָ֑פֶשׁ וְיוֹסֵ֖ף הָיָ֥ה בְמִצְרָֽיִם׃ מִסְפַּר כָּל הָאֲנָשִׁים צֶאֱצָאֵי יַעֲקֹב הָיָה שִׁבְעִים, יַחַד עִם יוֹסֵף שֶׁהָיָה בְּמִצְרַיִם.
ו וַיָּ֤מׇת יוֹסֵף֙ וְכׇל־אֶחָ֔יו וְכֹ֖ל הַדּ֥וֹר הַהֽוּא׃ יוֹסֵף וְכָל אֶחָיו וְכָל אוֹתוֹ הַדּוֹר מֵתוּ.
ז וּבְנֵ֣י יִשְׂרָאֵ֗ל פָּר֧וּ וַֽיִּשְׁרְצ֛וּ וַיִּרְבּ֥וּ וַיַּֽעַצְמ֖וּ בִּמְאֹ֣ד מְאֹ֑ד וַתִּמָּלֵ֥א הָאָ֖רֶץ אֹתָֽם׃ וּבְנֵי יִשְׂרָאֵל הוֹלִידוּ צֶאֱצָאִים רַבִּים וְהִתְרַבּוּ וְהִתְחַזְּקוּ מְאוֹד מְאוֹד, הָאָרֶץ הִתְמַלְּאָה מֵהֶם.
פ פָּרָשַׁת הַגְּזֵרוֹת
ח וַיָּ֥קׇם מֶֽלֶךְ־חָדָ֖שׁ עַל־מִצְרָ֑יִם אֲשֶׁ֥ר לֹֽא־יָדַ֖ע אֶת־יוֹסֵֽף׃ מָלַךְ בְּמִצְרַיִם מֶלֶךְ חָדָשׁ שֶׁלֹּא הִכִּיר אֶת יוֹסֵף.
ט וַיֹּ֖אמֶר אֶל־עַמּ֑וֹ הִנֵּ֗ה עַ֚ם בְּנֵ֣י יִשְׂרָאֵ֔ל רַ֥ב וְעָצ֖וּם מִמֶּֽנּוּ׃ וְאָמַר לְעַמּוֹ: הִנֵּה עַם בְּנֵי יִשְׂרָאֵל גָּדוֹל וְחָזָק יוֹתֵר מֵאִתָּנוּ.
י הָ֥בָה נִֽתְחַכְּמָ֖ה ל֑וֹ פֶּן־יִרְבֶּ֗ה וְהָיָ֞ה כִּֽי־תִקְרֶ֤אנָה מִלְחָמָה֙ וְנוֹסַ֤ף גַּם־הוּא֙ עַל־שֹׂ֣נְאֵ֔ינוּ וְנִלְחַם־בָּ֖נוּ וְעָלָ֥ה מִן־הָאָֽרֶץ׃ בּוֹאוּ נִפְעַל נֶגְדּוֹ בְּחָכְמָה כְּדֵי שֶׁלֹּא יִתְרַבֶּה, וְאָז אִם תִּהְיֶה מִלְחָמָה הוּא יִצְטָרֵף אֶל שׂוֹנְאֵינוּ וְיִלָּחֵם בָּנוּ וְיַעֲלֶה מֵהָאָרֶץ.
יא וַיָּשִׂ֤ימוּ עָלָיו֙ שָׂרֵ֣י מִסִּ֔ים לְמַ֥עַן עַנֹּת֖וֹ בְּסִבְלֹתָ֑ם וַיִּ֜בֶן עָרֵ֤י מִסְכְּנוֹת֙ לְפַרְעֹ֔ה אֶת־פִּתֹ֖ם וְאֶת־רַעַמְסֵֽס׃ מִצְרַיִם מִנּוּ עָלָיו שׁוֹטְרִים שֶׁיַּעֲבִידוּ אוֹתוֹ כְּדֵי לְהַשְׁפִּיל אוֹתוֹ בְּשִׁעְבּוּד, הוּא בָּנָה עָרִים שֶׁל מַחְסַנִּים לְפַרְעֹה, אֶת פִּתוֹם וְאֶת רַעַמְסֵס.
יב וְכַאֲשֶׁר֙ יְעַנּ֣וּ אֹת֔וֹ כֵּ֥ן יִרְבֶּ֖ה וְכֵ֣ן יִפְרֹ֑ץ וַיָּקֻ֕צוּ מִפְּנֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל׃ וּכְכָל שֶׁהִשְׁפִּילוּ אוֹתוֹ כָּךְ הִתְרַבָּה וְכָךְ הִתְפַּשֵּׁט, וְנִמְאַס לַמִּצְרִים בִּגְלַל בְּנֵי יִשְׂרָאֵל.
יג וַיַּעֲבִ֧דוּ מִצְרַ֛יִם אֶת־בְּנֵ֥י יִשְׂרָאֵ֖ל בְּפָֽרֶךְ׃ מִצְרַיִם הֶעֱבִידוּ אֶת בְּנֵי יִשְׂרָאֵל בַּעֲבוֹדָה קָשָׁה וְשׁוֹבֶרֶת.
יד וַיְמָרְר֨וּ אֶת־חַיֵּיהֶ֜ם בַּעֲבֹדָ֣ה קָשָׁ֗ה בְּחֹ֙מֶר֙ וּבִלְבֵנִ֔ים וּבְכׇל־עֲבֹדָ֖ה בַּשָּׂדֶ֑ה אֵ֚ת כׇּל־עֲבֹ֣דָתָ֔ם אֲשֶׁר־עָבְד֥וּ בָהֶ֖ם בְּפָֽרֶךְ׃ הֵם מֵרְרוּ אֶת חַיֵּיהֶם בַּעֲבוֹדָה קָשָׁה בְּמֶלֶט וּבִלְבֵנִים וּבְכָל מִינֵי עֲבוֹדָה בַּשָּׂדֶה – בְּכָל עֲבוֹדוֹתֵיהֶם שֶׁעָבְדוּ בָּהֶם עֲבוֹדָה קָשָׁה וְשׁוֹבֶרֶת.
טו וַיֹּ֙אמֶר֙ מֶ֣לֶךְ מִצְרַ֔יִם לַֽמְיַלְּדֹ֖ת הָֽעִבְרִיֹּ֑ת אֲשֶׁ֨ר שֵׁ֤ם הָֽאַחַת֙ שִׁפְרָ֔ה וְשֵׁ֥ם הַשֵּׁנִ֖ית פּוּעָֽה׃ מֶלֶךְ מִצְרַיִם אָמַר לַמְיַלְּדוֹת הָעִבְרִיּוֹת שֶׁשֵּׁם הָאַחַת שִׁפְרָה וְשֵׁם הַשְּׁנִיָּה פּוּעָה.
טז וַיֹּ֗אמֶר בְּיַלֶּדְכֶן֙ אֶת־הָֽעִבְרִיּ֔וֹת וּרְאִיתֶ֖ן עַל־הָאׇבְנָ֑יִם אִם־בֵּ֥ן הוּא֙ וַהֲמִתֶּ֣ן אֹת֔וֹ וְאִם־בַּ֥ת הִ֖וא וָחָֽיָה׃ הוּא אָמַר: כְּשֶׁאַתֶּן מְיַלְּדוֹת אֶת הַנָּשִׁים הָעִבְרִיּוֹת רְאוּ עַל מְקוֹם מוֹשַׁב הַיּוֹלֶדֶת, אִם הוּא בֵּן – הִרְגוּ אוֹתוֹ, וְאִם הִיא בַּת – תִּחְיֶה.
יז וַתִּירֶ֤אןָ הַֽמְיַלְּדֹת֙ אֶת־הָ֣אֱלֹהִ֔ים וְלֹ֣א עָשׂ֔וּ כַּאֲשֶׁ֛ר דִּבֶּ֥ר אֲלֵיהֶ֖ן מֶ֣לֶךְ מִצְרָ֑יִם וַתְּחַיֶּ֖יןָ אֶת־הַיְלָדִֽים׃ הַמְיַלְּדוֹת כִּבְּדוּ אֶת הָאֱלֹהִים וְלֹא עָשׂוּ כְּמוֹ שֶׁמֶּלֶךְ מִצְרַיִם דִּבֵּר אִתָּן, וְהֵן הִשְׁאִירוּ אֶת הַיְּלָדִים בַּחַיִּים.

שני[עריכה]

יח וַיִּקְרָ֤א מֶֽלֶךְ־מִצְרַ֙יִם֙ לַֽמְיַלְּדֹ֔ת וַיֹּ֣אמֶר לָהֶ֔ן מַדּ֥וּעַ עֲשִׂיתֶ֖ן הַדָּבָ֣ר הַזֶּ֑ה וַתְּחַיֶּ֖יןָ אֶת־הַיְלָדִֽים׃ מֶלֶךְ מִצְרַיִם קָרָא לַמְיַלְּדוֹת וְאָמַר לָהֶן: מַדּוּעַ עֲשִׂיתֶן אֶת הַדָּבָר הַזֶּה וְהִשְׁאַרְתֶּן אֶת הַיְּלָדִים בַּחַיִּים?
יט וַתֹּאמַ֤רְןָ הַֽמְיַלְּדֹת֙ אֶל־פַּרְעֹ֔ה כִּ֣י לֹ֧א כַנָּשִׁ֛ים הַמִּצְרִיֹּ֖ת הָֽעִבְרִיֹּ֑ת כִּֽי־חָי֣וֹת הֵ֔נָּה בְּטֶ֨רֶם תָּב֧וֹא אֲלֵהֶ֛ן‏[1] הַמְיַלֶּ֖דֶת וְיָלָֽדוּ׃ אָמְרוּ הַמְיַלְּדוֹת לְפַרְעֹה: כִּי הַנָּשִׁים הָעִבְרִיּוֹת אֵינָן כַּמִּצְרִיּוֹת, הֵן מַכִּירוֹת אֶת מִקְצוֹעַ הַמְיַלְּדוֹת, הֵן יוֹלְדוֹת לִפְנֵי שֶׁתָּבוֹא אֲלֵהֶן הַמְיַלֶּדֶת.
כ וַיֵּ֥יטֶב אֱלֹהִ֖ים לַֽמְיַלְּדֹ֑ת וַיִּ֧רֶב הָעָ֛ם וַיַּֽעַצְמ֖וּ מְאֹֽד׃ אֱלֹהִים הֵטִיב לַמְיַלְּדוֹת, וְהָעָם הִתְרַבּוּ וְהִתְחַזְּקוּ מְאוֹד.
כא וַיְהִ֕י כִּֽי־יָרְא֥וּ הַֽמְיַלְּדֹ֖ת אֶת־הָאֱלֹהִ֑ים וַיַּ֥עַשׂ לָהֶ֖ם בָּתִּֽים׃ כֵּיוָן שֶׁהַמְיַלְּדוֹת כִּבְּדוּ אֶת הָאֱלֹהִים הוּא נָתַן לָהֶן בָּנִים מְכֻבָּדִים.
כב וַיְצַ֣ו פַּרְעֹ֔ה לְכׇל־עַמּ֖וֹ לֵאמֹ֑ר כׇּל־הַבֵּ֣ן הַיִּלּ֗וֹד הַיְאֹ֙רָה֙ תַּשְׁלִיכֻ֔הוּ וְכׇל־הַבַּ֖ת תְּחַיּֽוּן׃ פַּרְעֹה צִוָּה לְכָל עַמּוֹ וְאָמַר: כָּל בֵּן שֶׁנּוֹלָד תַּשְׁלִיכוּ אוֹתוֹ לַיְאוֹר, וְכָל בַּת תְּחַיּוּ!
פ פָּרָשַׁת לֵידַת מֹשֶׁה
ב א וַיֵּ֥לֶךְ אִ֖ישׁ מִבֵּ֣ית לֵוִ֑י וַיִּקַּ֖ח אֶת־בַּת־לֵוִֽי׃ הָלַךְ אִישׁ מִמִּשְׁפַּחַת לֵוִי וְנָשָׂא לְאִשָּׁה אֶת הַבַּת שֶׁל לֵוִי.
ב וַתַּ֥הַר הָאִשָּׁ֖ה וַתֵּ֣לֶד בֵּ֑ן וַתֵּ֤רֶא אֹתוֹ֙ כִּי־ט֣וֹב ה֔וּא וַֽתִּצְפְּנֵ֖הוּ שְׁלֹשָׁ֥ה יְרָחִֽים׃ הָאִשָּׁה נִכְנְסָה לְהֵרָיוֹן וְיָלְדָה בֵּן, הִיא רָאֲתָה שֶׁהוּא בָּרִיא וְהֶחְבִּיאָה אוֹתוֹ שְׁלוֹשָׁה חֳדָשִׁים.
ג וְלֹא־יָכְלָ֣ה עוֹד֮ הַצְּפִינוֹ֒ וַתִּֽקַּֽח־לוֹ֙ תֵּ֣בַת גֹּ֔מֶא וַתַּחְמְרָ֥הֿ בַחֵמָ֖ר וּבַזָּ֑פֶת וַתָּ֤שֶׂם בָּהּ֙ אֶת־הַיֶּ֔לֶד וַתָּ֥שֶׂם בַּסּ֖וּף עַל־שְׂפַ֥ת הַיְאֹֽר׃ הִיא לֹא יָכְלָה לְהַחְבִּיא אוֹתוֹ עוֹד, לָקְחָה לוֹ אַרְגָּז מִנְּצָרִים, צִפְּתָה אוֹתוֹ בְּחֵמָר וּבְזֶפֶת, וְשָׂמָה בּוֹ אֶת הַיֶּלֶד, וְהִנִּיחָה אוֹתוֹ בֵּין צִמְחֵי הַסּוּף שֶׁעַל שְׂפַת הַיְּאוֹר.
ד וַתֵּתַצַּ֥ב אֲחֹת֖וֹ מֵרָחֹ֑ק לְדֵעָ֕ה מַה־יֵּעָשֶׂ֖ה לֽוֹ׃ אֲחוֹתוֹ נֶעֶמְדָה מֵרָחוֹק כְּדֵי לָדַעַת מַה יִּקְרֶה לוֹ.
ה וַתֵּ֤רֶד בַּת־פַּרְעֹה֙ לִרְחֹ֣ץ עַל־הַיְאֹ֔ר וְנַעֲרֹתֶ֥יהָ הֹלְכֹ֖ת עַל־יַ֣ד הַיְאֹ֑ר וַתֵּ֤רֶא אֶת־הַתֵּבָה֙ בְּת֣וֹךְ הַסּ֔וּף וַתִּשְׁלַ֥ח אֶת־אֲמָתָ֖הּ וַתִּקָּחֶֽהָ׃ בַּת פַּרְעֹה יָרְדָה לִרְחוֹץ בַּיְּאוֹר וְנַעֲרוֹתֶיהָ הוֹלְכוֹת לְיַד הַיְּאוֹר, הִיא רָאֲתָה אֶת הָאַרְגָּז בֵּין צִמְחֵי הַסּוּף, וְשָׁלְחָה אֶת שִׁפְחָתָהּ וְלָקְחָה אוֹתוֹ.
ו וַתִּפְתַּח֙ וַתִּרְאֵ֣הוּ אֶת־הַיֶּ֔לֶד וְהִנֵּה־נַ֖עַר בֹּכֶ֑ה וַתַּחְמֹ֣ל עָלָ֔יו וַתֹּ֕אמֶר מִיַּלְדֵ֥י הָֽעִבְרִ֖ים זֶֽה׃ הִיא פָּתְחָה וְרָאֲתָה אֶת הַיֶּלֶד וְהִנֵּה תִּינוֹק בּוֹכֶה, הִיא רִחֲמָה עָלָיו וְאָמְרָה: הוּא מִיַּלְדֵי הָעִבְרִים!
ז וַתֹּ֣אמֶר אֲחֹתוֹ֮ אֶל־בַּת־פַּרְעֹה֒ הַאֵלֵ֗ךְ וְקָרָ֤אתִי לָךְ֙ אִשָּׁ֣ה מֵינֶ֔קֶת מִ֖ן הָעִבְרִיֹּ֑ת וְתֵינִ֥ק לָ֖ךְ אֶת־הַיָּֽלֶד׃ אֲחוֹתוֹ אָמְרָה לְבַת פַּרְעֹה: הַאִם אֵלֵךְ וְאֶקְרָא לְאִשָּׁה מֵנִיקָה מֵהָעִבְרִיּוֹת שֶׁתֵּינִיק בִּשְׁבִילֵךְ אֶת הַיֶּלֶד?
ח וַתֹּֽאמֶר־לָ֥הּ בַּת־פַּרְעֹ֖ה לֵ֑כִי וַתֵּ֙לֶךְ֙ הָֽעַלְמָ֔ה וַתִּקְרָ֖א אֶת־אֵ֥ם הַיָּֽלֶד׃ אָמְרָה לָהּ בַּת פַּרְעֹה לְכִי! הַנַּעֲרָה הָלְכָה וְקָרְאָה לְאֵם הַיֶּלֶד.
ט וַתֹּ֧אמֶר לָ֣הּ בַּת־פַּרְעֹ֗ה הֵילִ֜יכִי אֶת־הַיֶּ֤לֶד הַזֶּה֙ וְהֵינִקִ֣הוּ לִ֔י וַאֲנִ֖י אֶתֵּ֣ן אֶת־שְׂכָרֵ֑ךְ וַתִּקַּ֧ח הָאִשָּׁ֛ה הַיֶּ֖לֶד וַתְּנִיקֵֽהוּ׃ אָמְרָה לָהּ בַּת פַּרְעֹה הוֹבִילִי אֶת הַיֶּלֶד הַזֶּה וְתָנִיקִי אוֹתוֹ עֲבוּרִי, וַאֲנִי אֶתֵּן לָךְ אֶת שְׂכָרֵךְ! הָאִשָּׁה לָקְחָה אֶת הַיֶּלֶד וְהֵנִיקָה אוֹתוֹ.
י וַיִּגְדַּ֣ל הַיֶּ֗לֶד וַתְּבִאֵ֙הוּ֙ לְבַת־פַּרְעֹ֔ה וַֽיְהִי־לָ֖הּ לְבֵ֑ן וַתִּקְרָ֤א שְׁמוֹ֙ מֹשֶׁ֔ה וַתֹּ֕אמֶר כִּ֥י מִן־הַמַּ֖יִם מְשִׁיתִֽהוּ׃ הַיֶּלֶד גָּדַל וְהִיא הֵבִיאָה אוֹתוֹ לְבַת פַּרְעֹה, וְהוּא הָיָה לָהּ כְּמוֹ בֵּן; הִיא קָרְאָה אֶת שְׁמוֹ מֹשֶׁה, וְאָמְרָה: כִּי מָשַׁכְתִּי אוֹתוֹ מֵהַמַּיִם.

שלישי[עריכה]

פָּרָשַׁת מַעֲשֵׂי מֹשֶׁה
יא וַיְהִ֣י ׀ בַּיָּמִ֣ים הָהֵ֗ם וַיִּגְדַּ֤ל מֹשֶׁה֙ וַיֵּצֵ֣א אֶל־אֶחָ֔יו וַיַּ֖רְא בְּסִבְלֹתָ֑ם וַיַּרְא֙ אִ֣ישׁ מִצְרִ֔י מַכֶּ֥ה אִישׁ־עִבְרִ֖י מֵאֶחָֽיו׃ בַּיָּמִים הָהֵם מֹשֶׁה גָּדַל וְיָצָא אֶל אֶחָיו וְרָאָה בְּסִבְלָם, הוּא רָאָה אִישׁ מִצְרִי מַכֶּה אִישׁ עִבְרִי מֵאֶחָיו.
יב וַיִּ֤פֶן כֹּה֙ וָכֹ֔ה וַיַּ֖רְא כִּ֣י אֵ֣ין אִ֑ישׁ וַיַּךְ֙ אֶת־הַמִּצְרִ֔י וַֽיִּטְמְנֵ֖הוּ בַּחֽוֹל׃ הוּא הִבִּיט לְכָאן וּלְכָאן וְרָאָה שֶׁאֵין אִישׁ, הוּא הָרַג אֶת הַמִּצְרִי וְקָבַר אוֹתוֹ בַּחוֹל.
יג וַיֵּצֵא֙ בַּיּ֣וֹם הַשֵּׁנִ֔י וְהִנֵּ֛ה שְׁנֵֽי־אֲנָשִׁ֥ים עִבְרִ֖ים נִצִּ֑ים וַיֹּ֙אמֶר֙ לָֽרָשָׁ֔ע לָ֥מָּה תַכֶּ֖ה רֵעֶֽךָ׃ הוּא יָצָא בַּיּוֹם הַשֵּׁנִי וְהִנֵּה שְׁנֵי אֲנָשִׁים עִבְרִים רָבִים, הוּא אָמַר לָרָשָׁע: לָמָּה אַתָּה מַכֶּה אֶת חֲבֵרְךָ!
יד וַ֠יֹּ֠אמֶר מִ֣י שָֽׂמְךָ֞ לְאִ֨ישׁ שַׂ֤ר וְשֹׁפֵט֙ עָלֵ֔ינוּ הַלְהׇרְגֵ֙נִי֙ אַתָּ֣ה אֹמֵ֔ר כַּאֲשֶׁ֥ר הָרַ֖גְתָּ אֶת־הַמִּצְרִ֑י וַיִּירָ֤א מֹשֶׁה֙ וַיֹּאמַ֔ר אָכֵ֖ן נוֹדַ֥ע הַדָּבָֽר׃ אָמַר לוֹ: מִי מִנָּה אוֹתְךָ לְשַׂר וְשׁוֹפֵט עָלֵינוּ, הַאִם אַתָּה חוֹשֵׁב לַהֲרוֹג אוֹתִי כְּמוֹ שֶׁהָרַגְתָּ אֶת הַמִּצְרִי?! מֹשֶׁה פָּחַד וְאָמַר: אָכֵן הַדָּבָר הִתְפַּרְסֵם!
טו וַיִּשְׁמַ֤ע פַּרְעֹה֙ אֶת־הַדָּבָ֣ר הַזֶּ֔ה וַיְבַקֵּ֖שׁ לַהֲרֹ֣ג אֶת־מֹשֶׁ֑ה וַיִּבְרַ֤ח מֹשֶׁה֙ מִפְּנֵ֣י פַרְעֹ֔ה וַיֵּ֥שֶׁב בְּאֶֽרֶץ־מִדְיָ֖ן וַיֵּ֥שֶׁב עַֽל־הַבְּאֵֽר׃ פַּרְעֹה שָׁמַע אֶת הַדָּבָר הַזֶּה וְרָצָה לַהֲרוֹג אֶת מֹשֶׁה, מֹשֶׁה בָּרַח מִפַּרְעֹה וְהִתְיַשֵּׁב בְּאֶרֶץ מִדְיָן וְיָשַׁב עַל יַד הַבְּאֵר.
טז וּלְכֹהֵ֥ן מִדְיָ֖ן שֶׁ֣בַע בָּנ֑וֹת וַתָּבֹ֣אנָה וַתִּדְלֶ֗נָה וַתְּמַלֶּ֙אנָה֙ אֶת־הָ֣רְהָטִ֔ים לְהַשְׁק֖וֹת צֹ֥אן אֲבִיהֶֽן׃ לַכּוֹמֶר שֶׁל מִדְיָן הָיוּ שֶׁבַע בָּנוֹת, הֵן בָּאוּ, שָׁאֲבוּ, וּמִלְּאוּ אֶת הַשְּׁקָתוֹת כְּדֵי לְהַשְׁקוֹת אֶת צֹאן אֲבִיהֶן.
יז וַיָּבֹ֥אוּ הָרֹעִ֖ים וַיְגָרְשׁ֑וּם וַיָּ֤קׇם מֹשֶׁה֙ וַיּ֣וֹשִׁעָ֔ן וַיַּ֖שְׁקְ אֶת־צֹאנָֽם׃ בָּאוּ הָרוֹעִים וְגֵרְשׁוּ אוֹתָן, מֹשֶׁה קָם וְהוֹשִׁיעַ אוֹתָן וְהִשְׁקָה אֶת צֹאנָן.
יח וַתָּבֹ֕אנָה אֶל־רְעוּאֵ֖ל אֲבִיהֶ֑ן וַיֹּ֕אמֶר מַדּ֛וּעַ מִהַרְתֶּ֥ן בֹּ֖א הַיּֽוֹם׃ הֵן בָּאוּ לִרְעוּאֵל אֲבִיהֶן, אָמַר לָהֵן: מַדּוּעַ מִהַרְתֶּן לָבוֹא הַיּוֹם?
יט וַתֹּאמַ֕רְןָ אִ֣ישׁ מִצְרִ֔י הִצִּילָ֖נוּ מִיַּ֣ד הָרֹעִ֑ים וְגַם־דָּלֹ֤ה דָלָה֙ לָ֔נוּ וַיַּ֖שְׁקְ אֶת־הַצֹּֽאן׃ אָמְרוּ לוֹ: אִישׁ מִצְרִי הִצִּיל אוֹתָנוּ מֵהָרוֹעִים וְגַם שָׁאַב לָנוּ וְהִשְׁקָה אֶת הַצֹּאן.
כ וַיֹּ֥אמֶר אֶל־בְּנֹתָ֖יו וְאַיּ֑וֹ לָ֤מָּה זֶּה֙ עֲזַבְתֶּ֣ן אֶת־הָאִ֔ישׁ קִרְאֶ֥ן ל֖וֹ וְיֹ֥אכַל לָֽחֶם׃ הוּא אָמַר לִבְנוֹתָיו: וְהֵיכָן הוּא? לָמָּה עֲזַבְתֶּן אֶת הָאִישׁ? קִרְאוּ לוֹ וְיֹאכַל סְעוּדָה!
כא וַיּ֥וֹאֶל מֹשֶׁ֖ה לָשֶׁ֣בֶת אֶת־הָאִ֑ישׁ וַיִּתֵּ֛ן אֶת־צִפֹּרָ֥ה בִתּ֖וֹ לְמֹשֶֽׁה׃ מֹשֶׁה הִסְכִּים לָגוּר עִם הָאִישׁ, וְהוּא נָתַן אֶת צִפּוֹרָה בִּתּוֹ לְמֹשֶׁה.
כב וַתֵּ֣לֶד בֵּ֔ן וַיִּקְרָ֥א אֶת־שְׁמ֖וֹ גֵּרְשֹׁ֑ם כִּ֣י אָמַ֔ר גֵּ֣ר הָיִ֔יתִי בְּאֶ֖רֶץ נׇכְרִיָּֽה׃ הִיא יָלְדָה בֵּן וְהוּא קָרָא אֶת שְׁמוֹ גֵּרְשׁוֹם, כִּי אָמַר: הָיִיתִי גֵּר שָׁם – בְּאֶרֶץ זָרָה.
פ פָּרָשַׁת זַעֲקַת יִשְׂרָאֵל
כג וַיְהִי֩ בַיָּמִ֨ים הָֽרַבִּ֜ים הָהֵ֗ם וַיָּ֙מׇת֙ מֶ֣לֶךְ מִצְרַ֔יִם וַיֵּאָנְח֧וּ בְנֵֽי־יִשְׂרָאֵ֛ל מִן־הָעֲבֹדָ֖ה וַיִּזְעָ֑קוּ וַתַּ֧עַל שַׁוְעָתָ֛ם אֶל־הָאֱלֹהִ֖ים מִן־הָעֲבֹדָֽה׃ בַּיָּמִים הָרַבִּים הָהֵם מֶלֶךְ מִצְרַיִם מֵת, וּבְנֵי יִשְׂרָאֵל נֶאֶנְחוּ מֵהָעֲבוֹדָה וְזָעֲקוּ, וּתְפִלָּתָם לְהַצָּלָה מֵהָעֲבוֹדָה עָלְתָה אֶל הָאֱלֹהִים.
כד וַיִּשְׁמַ֥ע אֱלֹהִ֖ים אֶת־נַאֲקָתָ֑ם וַיִּזְכֹּ֤ר אֱלֹהִים֙ אֶת־בְּרִית֔וֹ אֶת־אַבְרָהָ֖ם אֶת־יִצְחָ֥ק וְאֶֽת־יַעֲקֹֽב׃ אֱלֹהִים שָׁמַע אֶת צַעֲקָתָם, וֶאֱלֹהִים זָכַר אֶת הַהֶסְכֵּם שֶׁעָשָׂה עִם אַבְרָהָם עִם יִצְחָק וְעִם יַעֲקֹב.
כה וַיַּ֥רְא אֱלֹהִ֖ים אֶת־בְּנֵ֣י יִשְׂרָאֵ֑ל וַיֵּ֖דַע אֱלֹהִֽים׃ אֱלֹהִים רָאָה אֶת מַצָּבָם שֶׁל בְּנֵי יִשְׂרָאֵל, אֱלֹהִים הֵבִין וְהִשְׁתַּתֵּף בְּצַעֲרָם.

רביעי[עריכה]

ס פָּרָשַׁת הַסְּנֶה
ג א וּמֹשֶׁ֗ה הָיָ֥ה רֹעֶ֛ה אֶת־צֹ֛אן יִתְר֥וֹ חֹתְנ֖וֹ כֹּהֵ֣ן מִדְיָ֑ן וַיִּנְהַ֤ג אֶת־הַצֹּאן֙ אַחַ֣ר הַמִּדְבָּ֔ר וַיָּבֹ֛א אֶל־הַ֥ר הָאֱלֹהִ֖ים חֹרֵֽבָה׃ וּמֹשֶׁה הָיָה רוֹעֶה אֶת צֹאן יִתְרוֹ חוֹתְנוֹ הַכּוֹמֶר שֶׁל מִדְיָן, הוּא הוֹבִיל אֶת הַצֹּאן לְעֵבֶר הַמִּדְבָּר, וּבָא אֶל הַר הָאֱלֹהִים – לְחוֹרֵב.
ב וַ֠יֵּרָ֠א מַלְאַ֨ךְ יְהֹוָ֥ה אֵלָ֛יו בְּלַבַּת־אֵ֖שׁ מִתּ֣וֹךְ הַסְּנֶ֑ה וַיַּ֗רְא וְהִנֵּ֤ה הַסְּנֶה֙ בֹּעֵ֣ר בָּאֵ֔שׁ וְהַסְּנֶ֖ה אֵינֶ֥נּוּ אֻכָּֽל׃ מַלְאַךְ ה' נִרְאָה אֵלָיו בְּלֶהָבַת אֵשׁ מִתּוֹךְ הַסְּנֶה, הוּא רָאָה וְהִנֵּה הַסְּנֶה בּוֹעֵר בָּאֵשׁ וְהַסְּנֶה לֹא נִשְׂרָף.
ג וַיֹּ֣אמֶר מֹשֶׁ֔ה אָסֻֽרָה־נָּ֣א וְאֶרְאֶ֔ה אֶת־הַמַּרְאֶ֥ה הַגָּדֹ֖ל הַזֶּ֑ה מַדּ֖וּעַ לֹא־יִבְעַ֥ר הַסְּנֶֽה׃ מֹשֶׁה אָמַר: אֶסְטֶה מִדַּרְכִּי וְאֶרְאֶה אֶת הַמַּרְאֶה הַגָּדוֹל הַזֶּה, לָמָּה הַסְּנֶה לֹא בּוֹעֵר.
ד וַיַּ֥רְא יְהֹוָ֖ה כִּ֣י סָ֣ר לִרְא֑וֹת וַיִּקְרָא֩ אֵלָ֨יו אֱלֹהִ֜ים מִתּ֣וֹךְ הַסְּנֶ֗ה וַיֹּ֛אמֶר מֹשֶׁ֥ה מֹשֶׁ֖ה וַיֹּ֥אמֶר הִנֵּֽנִי׃ ה' רָאָה שֶׁהוּא סָטָה כְּדֵי לִרְאוֹת, אֱלֹהִים קָרָא לוֹ מִתּוֹךְ הַסְּנֶה וְאָמַר: מֹשֶׁה מֹשֶׁה! הוּא אָמַר: הִנֵּה אֲנִי.
ה וַיֹּ֖אמֶר אַל־תִּקְרַ֣ב הֲלֹ֑ם שַׁל־נְעָלֶ֙יךָ֙ מֵעַ֣ל רַגְלֶ֔יךָ כִּ֣י הַמָּק֗וֹם אֲשֶׁ֤ר אַתָּה֙ עוֹמֵ֣ד עָלָ֔יו אַדְמַת־קֹ֖דֶשׁ הֽוּא׃ אָמַר: אַל תִּגַּשׁ לְכָאן! חֲלוֹץ אֶת נְעָלֶיךָ מֵעַל רַגְלֶיךָ כִּי הַמָּקוֹם שֶׁאַתָּה עוֹמֵד עָלָיו הוּא אַדְמַת קוֹדֶשׁ.
ו וַיֹּ֗אמֶר אָנֹכִי֙ אֱלֹהֵ֣י אָבִ֔יךָ אֱלֹהֵ֧י אַבְרָהָ֛ם אֱלֹהֵ֥י יִצְחָ֖ק וֵאלֹהֵ֣י יַעֲקֹ֑ב וַיַּסְתֵּ֤ר מֹשֶׁה֙ פָּנָ֔יו כִּ֣י יָרֵ֔א מֵהַבִּ֖יט אֶל־הָאֱלֹהִֽים׃ וְאָמַר: אֲנִי אֱלֹהֵי אָבִיךָ אֱלֹהֵי אַבְרָהָם אֱלֹהֵי יִצְחָק וֵאלֹהֵי יַעֲקֹב, מֹשֶׁה הִסְתִּיר אֶת פָּנָיו כִּי פָּחַד לְהַבִּיט אֶל הִתְגַּלּוּת כְּבוֹד הָאֱלֹהִים.
ז וַיֹּ֣אמֶר יְהֹוָ֔ה רָאֹ֥ה רָאִ֛יתִי אֶת־עֳנִ֥י עַמִּ֖י אֲשֶׁ֣ר בְּמִצְרָ֑יִם וְאֶת־צַעֲקָתָ֤ם שָׁמַ֙עְתִּי֙ מִפְּנֵ֣י נֹֽגְשָׂ֔יו כִּ֥י יָדַ֖עְתִּי אֶת־מַכְאֹבָֽיו׃ ה' אָמַר: רָאִיתִי אֶת הַשְׁפָּלַת עַמִּי שֶׁבְּמִצְרַיִם, וְשָׁמַעְתִּי אֶת צַעֲקָתָם שֶׁצָּעֲקוּ בִּגְלַל מַעֲבִידֵיהֶם הֲרֵי אֲנִי יוֹדֵעַ אֶת כְּאֵבֵיהֶם.
ח וָאֵרֵ֞ד לְהַצִּיל֣וֹ ׀ מִיַּ֣ד מִצְרַ֗יִם וּֽלְהַעֲלֹתוֹ֮ מִן־הָאָ֣רֶץ הַהִוא֒ אֶל־אֶ֤רֶץ טוֹבָה֙ וּרְחָבָ֔ה אֶל־אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָ֑שׁ אֶל־מְק֤וֹם הַֽכְּנַעֲנִי֙ וְהַ֣חִתִּ֔י וְהָֽאֱמֹרִי֙ וְהַפְּרִזִּ֔י וְהַחִוִּ֖י וְהַיְבוּסִֽי׃ וְהֶחֱלַטְתִּי לָרֶדֶת לְהַצִּיל אוֹתוֹ מֵהַמִּצְרִים וּלְהַעֲלוֹת אוֹתוֹ מֵהָאָרֶץ הַהִיא לְאֶרֶץ טוֹבָה וּרְחָבָה, לְאֶרֶץ שׁוֹפַעַת חָלָב וּדְבָשׁ, אֶל מְקוֹם הַכְּנַעֲנִי הַחִתִּי הָאֱמוֹרִי הַפְּרִזִּי הַחִוִּי וְהַיְּבוּסִי.
ט וְעַתָּ֕ה הִנֵּ֛ה צַעֲקַ֥ת בְּנֵי־יִשְׂרָאֵ֖ל בָּ֣אָה אֵלָ֑י וְגַם־רָאִ֙יתִי֙ אֶת־הַלַּ֔חַץ אֲשֶׁ֥ר מִצְרַ֖יִם לֹחֲצִ֥ים אֹתָֽם׃ וְעַכְשָׁיו, הִנֵּה צַעֲקַת בְּנֵי יִשְׂרָאֵל בָּאָה אֵלַי וְגַם רָאִיתִי אֶת הַלַּחַץ שֶׁמִּצְרַיִם לוֹחֲצִים אוֹתָם.
י וְעַתָּ֣ה לְכָ֔ה וְאֶֽשְׁלָחֲךָ֖ אֶל־פַּרְעֹ֑ה וְהוֹצֵ֛א אֶת־עַמִּ֥י בְנֵֽי־יִשְׂרָאֵ֖ל מִמִּצְרָֽיִם׃ וְעַכְשָׁיו לֵךְ וְאֶשְׁלַח אוֹתְךָ אֶל פַּרְעֹה וְהוֹצֵא אֶת עַמִּי בְּנֵי יִשְׂרָאֵל מִמִּצְרַיִם.
יא וַיֹּ֤אמֶר מֹשֶׁה֙ אֶל־הָ֣אֱלֹהִ֔ים מִ֣י אָנֹ֔כִי כִּ֥י אֵלֵ֖ךְ אֶל־פַּרְעֹ֑ה וְכִ֥י אוֹצִ֛יא אֶת־בְּנֵ֥י יִשְׂרָאֵ֖ל מִמִּצְרָֽיִם׃ מֹשֶׁה אָמַר לָאֱלֹהִים: מִי אֲנִי שֶׁאֵלֵךְ אֶל פַּרְעֹה?! וְכִי אַצְלִיחַ לְהוֹצִיא אֶת בְּנֵי יִשְׂרָאֵל מִמִּצְרַיִם?!
יב וַיֹּ֙אמֶר֙ כִּֽי־אֶֽהְיֶ֣ה עִמָּ֔ךְ וְזֶה־לְּךָ֣ הָא֔וֹת כִּ֥י אָנֹכִ֖י שְׁלַחְתִּ֑יךָ בְּהוֹצִֽיאֲךָ֤ אֶת־הָעָם֙ מִמִּצְרַ֔יִם תַּֽעַבְדוּן֙ אֶת־הָ֣אֱלֹהִ֔ים עַ֖ל הָהָ֥ר הַזֶּֽה׃ אָמַר: אַתָּה תַּצְלִיחַ כִּי אֶהְיֶה אִתְּךָ, וְזֶה הַסִּימָן שֶׁלְךָ - שֶׁאֲנִי שֶׁשָּׁלַחְתִּי אוֹתְךָ! כְּשֶׁתּוֹצִיא אֶת הָעָם מִמִּצְרַיִם תַּעַבְדוּ אֶת הָאֱלֹהִים עַל הָהָר הַזֶּה.
יג וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶל־הָֽאֱלֹהִ֗ים הִנֵּ֨ה אָנֹכִ֣י בָא֮ אֶל־בְּנֵ֣י יִשְׂרָאֵל֒ וְאָמַרְתִּ֣י לָהֶ֔ם אֱלֹהֵ֥י אֲבוֹתֵיכֶ֖ם שְׁלָחַ֣נִי אֲלֵיכֶ֑ם וְאָֽמְרוּ־לִ֣י מַה־שְּׁמ֔וֹ מָ֥ה אֹמַ֖ר אֲלֵהֶֽם׃ מֹשֶׁה אָמַר לָאֱלֹהִים: הִנֵּה אֲנִי בָּא לִבְנֵי יִשְׂרָאֵל וְאוֹמַר לָהֶם אֱלֹהֵי אֲבוֹתֵיכֶם שָׁלַח אוֹתִי אֲלֵיכֶם, יֹאמְרוּ לִי: בְּאֵיזֶה שֵׁם הוּא הִתְגַּלָּה אֵלֶיךָ? מָה אוֹמַר לָהֶם?
יד וַיֹּ֤אמֶר אֱלֹהִים֙ אֶל־מֹשֶׁ֔ה אֶֽהְיֶ֖ה אֲשֶׁ֣ר אֶֽהְיֶ֑ה וַיֹּ֗אמֶר כֹּ֤ה תֹאמַר֙ לִבְנֵ֣י יִשְׂרָאֵ֔ל אֶֽהְיֶ֖ה שְׁלָחַ֥נִי אֲלֵיכֶֽם׃ אָמַר אֱלֹהִים לְמֹשֶׁה: שְׁמִי 'אֶהְיֶה' כִּי אֶהְיֶה תָּמִיד, וְאָמַר: כָּךְ תֹּאמַר לִבְנֵי יִשְׂרָאֵל: 'אֶהְיֶה' שָׁלַח אוֹתִי אֲלֵיכֶם!
טו וַיֹּ֩אמֶר֩ ע֨וֹד אֱלֹהִ֜ים אֶל־מֹשֶׁ֗ה כֹּֽה־תֹאמַר֮ אֶל־בְּנֵ֣י יִשְׂרָאֵל֒ יְהֹוָ֞ה אֱלֹהֵ֣י אֲבֹתֵיכֶ֗ם אֱלֹהֵ֨י אַבְרָהָ֜ם אֱלֹהֵ֥י יִצְחָ֛ק וֵאלֹהֵ֥י יַעֲקֹ֖ב שְׁלָחַ֣נִי אֲלֵיכֶ֑ם זֶה־שְּׁמִ֣י לְעֹלָ֔ם וְזֶ֥ה זִכְרִ֖י לְדֹ֥ר דֹּֽר׃ אֱלֹהִים אָמַר עוֹד לְמֹשֶׁה: כָּךְ תֹּאמַר לִבְנֵי יִשְׂרָאֵל: ה' אֱלֹהֵי אֲבוֹתֵיכֶם אֱלֹהֵי אַבְרָהָם אֱלֹהֵי יִצְחָק וֵאלֹהֵי יַעֲקֹב שָׁלַח אוֹתִי אֲלֵיכֶם, זֶה יִהְיֶה שְׁמִי לְעוֹלָם, וּבַשֵּׁם הַזֶּה יַזְכִּירוּ אוֹתִי בְּכָל דּוֹר וָדוֹר.

חמישי[עריכה]

טז לֵ֣ךְ וְאָֽסַפְתָּ֞ אֶת־זִקְנֵ֣י יִשְׂרָאֵ֗ל וְאָמַרְתָּ֤ אֲלֵהֶם֙ יְהֹוָ֞ה אֱלֹהֵ֤י אֲבֹֽתֵיכֶם֙ נִרְאָ֣ה אֵלַ֔י אֱלֹהֵ֧י אַבְרָהָ֛ם יִצְחָ֥ק וְיַעֲקֹ֖ב לֵאמֹ֑ר פָּקֹ֤ד פָּקַ֙דְתִּי֙ אֶתְכֶ֔ם וְאֶת־הֶעָשׂ֥וּי לָכֶ֖ם בְּמִצְרָֽיִם׃ לֵךְ וֶאֱסוֹף אֶת חַכְמֵי יִשְׂרָאֵל וְתֹאמַר לָהֶם: ה' אֱלֹהֵי אֲבוֹתֵיכֶם – אֱלֹהֵי אַבְרָהָם יִצְחָק וְיַעֲקֹב, הִתְגַּלָּה אֵלַי וְאָמַר: אֲנִי זוֹכֵר אֶתְכֶם וְרוֹאֶה מַה עוֹשִׂים לָכֶם בְּמִצְרַיִם.
יז וָאֹמַ֗ר אַעֲלֶ֣ה אֶתְכֶם֮ מֵעֳנִ֣י מִצְרַ֒יִם֒ אֶל־אֶ֤רֶץ הַֽכְּנַעֲנִי֙ וְהַ֣חִתִּ֔י וְהָֽאֱמֹרִי֙ וְהַפְּרִזִּ֔י וְהַחִוִּ֖י וְהַיְבוּסִ֑י אֶל־אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָֽשׁ׃ וְאָמַרְתִּי: אַעֲלֶה אֶתְכֶם מֵהַשִּׁפְלוּת בְּמִצְרַיִם אֶל אֶרֶץ הַכְּנַעֲנִי הַחִתִּי הָאֱמוֹרִי הַפְּרִזִּי הַחִוִּי וְהַיְּבוּסִי, אֶל אֶרֶץ שׁוֹפַעַת חָלָב וּדְבָשׁ.
יח וְשָׁמְע֖וּ לְקֹלֶ֑ךָ וּבָאתָ֡ אַתָּה֩ וְזִקְנֵ֨י יִשְׂרָאֵ֜ל אֶל־מֶ֣לֶךְ מִצְרַ֗יִם וַאֲמַרְתֶּ֤ם אֵלָיו֙ יְהֹוָ֞ה אֱלֹהֵ֤י הָֽעִבְרִיִּים֙ נִקְרָ֣ה עָלֵ֔ינוּ וְעַתָּ֗ה נֵֽלְﬞכָה־נָּ֞א דֶּ֣רֶךְ שְׁלֹ֤שֶׁת יָמִים֙ בַּמִּדְבָּ֔ר וְנִזְבְּחָ֖ה לַֽיהֹוָ֥ה אֱלֹהֵֽינוּ׃ הֵם יְקַבְּלוּ אֶת דְּבָרֶיךָ, וְתָבוֹא אַתָּה וְחַכְמֵי יִשְׂרָאֵל לְמֶלֶךְ מִצְרַיִם, וְתֹאמְרוּ לוֹ: ה' אֱלֹהֵי הָעִבְרִים הִתְגַּלָּה אֵלֵינוּ, וְעַכְשָׁיו נֵלֶךְ בְּבַקָּשָׁה לְדֶרֶךְ שֶׁל שְׁלוֹשָׁה יָמִים בַּמִּדְבָּר, וְנַקְרִיב קָרְבָּנוֹת לַה' אֱלֹהֵינוּ.
יט וַאֲנִ֣י יָדַ֔עְתִּי כִּ֠י לֹֽא־יִתֵּ֥ן אֶתְכֶ֛ם מֶ֥לֶךְ מִצְרַ֖יִם לַהֲלֹ֑ךְ וְלֹ֖א בְּיָ֥ד חֲזָקָֽה׃ וַאֲנִי יוֹדֵעַ שֶׁמֶּלֶךְ מִצְרַיִם לֹא יִתֵּן לָכֶם לָלֶכֶת, הוּא לֹא יִתֵּן לַמְרוֹת שֶׁאַכֶּה אוֹתוֹ בְּיָד חֲזָקָה.
כ וְשָׁלַחְתִּ֤י אֶת־יָדִי֙ וְהִכֵּיתִ֣י אֶת־מִצְרַ֔יִם בְּכֹל֙ נִפְלְאֹתַ֔י אֲשֶׁ֥ר אֶֽעֱשֶׂ֖ה בְּקִרְבּ֑וֹ וְאַחֲרֵי־כֵ֖ן יְשַׁלַּ֥ח אֶתְכֶֽם׃ אוֹשִׁיט אֶת יָדִי וְאַכֶּה אֶת מִצְרַיִם בְּכֹל הַמַּכּוֹת הַמֻּפְלָאוֹת שֶׁאֶעֱשֶׂה אֶצְלוֹ, וְרַק אַחֲרֵי כֵן הוּא יְשַׁחְרֵר אֶתְכֶם.
כא וְנָתַתִּ֛י אֶת־חֵ֥ן הָֽעָם־הַזֶּ֖ה בְּעֵינֵ֣י מִצְרָ֑יִם וְהָיָה֙ כִּ֣י תֵֽלֵכ֔וּן לֹ֥א תֵלְכ֖וּ רֵיקָֽם׃ אֶעֱשֶׂה שֶׁהָעָם הַזֶּה יִמְצָא חֵן בְּעֵינֵי מִצְרַיִם, וּכְשֶׁתֵּלְכוּ לֹא תֵּלְכוּ רֵקִים מֵרְכוּשׁ.
כב וְשָׁאֲלָ֨ה אִשָּׁ֤ה מִשְּׁכֶנְתָּהּ֙ וּמִגָּרַ֣ת בֵּיתָ֔הּ כְּלֵי־כֶ֛סֶף וּכְלֵ֥י זָהָ֖ב וּשְׂמָלֹ֑ת וְשַׂמְתֶּ֗ם עַל־בְּנֵיכֶם֙ וְעַל־בְּנֹ֣תֵיכֶ֔ם וְנִצַּלְתֶּ֖ם אֶת־מִצְרָֽיִם׃ כָּל אִשָּׁה תְּבַקֵּשׁ מִשְּׁכֶנְתָּהּ וּמִמִּי שֶׁגָּרָה לְיַד בֵּיתָהּ כְּלֵי כֶּסֶף וּכְלֵי זָהָב וּבְגָדִים וְתָשִׂימוּ עַל בְּנֵיכֶם וְעַל בְּנוֹתֵיכֶם, וְתַצִּילוּ מִמִּצְרַיִם אֶת שְׂכַר שִׁעְבֻּדְכֶם.
ד א וַיַּ֤עַן מֹשֶׁה֙ וַיֹּ֔אמֶר וְהֵן֙ לֹֽא־יַאֲמִ֣ינוּ לִ֔י וְלֹ֥א יִשְׁמְע֖וּ בְּקֹלִ֑י כִּ֣י יֹֽאמְר֔וּ לֹֽא־נִרְאָ֥ה אֵלֶ֖יךָ יְהֹוָֽה׃ מֹשֶׁה עָנָה וְאָמַר: וַהֲרֵי לֹא יַאֲמִינוּ לִי וְלֹא יִשְׁמְעוּ בְּקוֹלִי אֶלָּא יֹאמְרוּ: ה' לֹא הִתְגַּלָּה אֵלֶיךָ.
ב וַיֹּ֧אמֶר אֵלָ֛יו יְהֹוָ֖ה מזה [מַה־זֶּ֣ה] בְיָדֶ֑ךָ וַיֹּ֖אמֶר מַטֶּֽה׃ ה' אָמַר לוֹ: מַה זֶּה בְּיָדְךָ? אָמַר: מַקֵּל.
ג וַיֹּ֙אמֶר֙ הַשְׁלִיכֵ֣הוּ אַ֔רְצָה וַיַּשְׁלִכֵ֥הוּ אַ֖רְצָה וַיְהִ֣י לְנָחָ֑שׁ וַיָּ֥נׇס מֹשֶׁ֖ה מִפָּנָֽיו׃ אָמַר: הַשְׁלֵךְ אוֹתוֹ לָאָרֶץ! הוּא הִשְׁלִיךְ אוֹתוֹ לָאָרֶץ, וְהוּא נֶהֱפַךְ לְנָחָשׁ, מֹשֶׁה בָּרַח מִמֶּנּוּ.
ד וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה שְׁלַח֙ יָֽדְךָ֔ וֶאֱחֹ֖ז בִּזְנָב֑וֹ וַיִּשְׁלַ֤ח יָדוֹ֙ וַיַּ֣חֲזֶק בּ֔וֹ וַיְהִ֥י לְמַטֶּ֖ה בְּכַפּֽוֹ׃ אָמַר ה' לְמֹשֶׁה: הוֹשֵׁט אֶת יָדְךָ וְהַחֲזֵק בִּזְנָבוֹ! הוּא הוֹשִׁיט אֶת יָדוֹ וְהֶחֱזִיק בּוֹ, וְהוּא נֶהֱפַךְ לְמַקֵּל בְּיָדוֹ.
ה לְמַ֣עַן יַאֲמִ֔ינוּ כִּֽי־נִרְאָ֥ה אֵלֶ֛יךָ יְהֹוָ֖ה אֱלֹהֵ֣י אֲבֹתָ֑ם אֱלֹהֵ֧י אַבְרָהָ֛ם אֱלֹהֵ֥י יִצְחָ֖ק וֵאלֹהֵ֥י יַעֲקֹֽב׃ עֲשֵׂה כָּךְ כְּדֵי שֶׁיַּאֲמִינוּ שֶׁה' אֱלֹהֵי אֲבוֹתֵיהֶם הִתְגַּלָּה אֵלֶיךָ, אֱלֹהֵי אַבְרָהָם אֱלֹהֵי יִצְחָק וֵאלֹהֵי יַעֲקֹב!
ו וַיֹּ֩אמֶר֩ יְהֹוָ֨ה ל֜וֹ ע֗וֹד הָֽבֵא־נָ֤א יָֽדְךָ֙ בְּחֵיקֶ֔ךָ וַיָּבֵ֥א יָד֖וֹ בְּחֵיק֑וֹ וַיּ֣וֹצִאָ֔הּ וְהִנֵּ֥ה יָד֖וֹ מְצֹרַ֥עַת כַּשָּֽׁלֶג׃ ה' אָמַר לוֹ עוֹד: הַכְנֵס אֶת יָדְךָ תַּחַת זְרוֹעֲךָ, הוּא הִכְנִיס אֶת יָדוֹ תַּחַת זְרוֹעוֹ, הוּא הוֹצִיא אוֹתָהּ וְהִנֵּה יָדוֹ מְצוֹרַעַת כַּשֶּׁלֶג.
ז וַיֹּ֗אמֶר הָשֵׁ֤ב יָֽדְךָ֙ אֶל־חֵיקֶ֔ךָ וַיָּ֥שֶׁב יָד֖וֹ אֶל־חֵיק֑וֹ וַיּֽוֹצִאָהּ֙ מֵֽחֵיק֔וֹ וְהִנֵּה־שָׁ֖בָה כִּבְשָׂרֽוֹ׃ אָמַר: הָשֵׁב אֶת יָדְךָ אֶל תַּחַת זְרוֹעֲךָ, הוּא הֵשִׁיב אֶת יָדוֹ אֶל תַּחַת זְרוֹעוֹ, וְהוֹצִיא אוֹתָהּ וְהִיא חָזְרָה לִהְיוֹת כִּבְשָׂרוֹ הַבָּרִיא.
ח וְהָיָה֙ אִם־לֹ֣א יַאֲמִ֣ינוּ לָ֔ךְ וְלֹ֣א יִשְׁמְע֔וּ לְקֹ֖ל הָאֹ֣ת הָרִאשׁ֑וֹן וְהֶֽאֱמִ֔ינוּ לְקֹ֖ל הָאֹ֥ת הָאַחֲרֽוֹן׃ אִם לֹא יַאֲמִינוּ לְךָ וְלֹא יְקַבְּלוּ אֶת הַסִּימָן הָרִאשׁוֹן – יַאֲמִינוּ בִּגְלַל הַסִּימָן הָאַחֵר.
ט וְהָיָ֡ה אִם־לֹ֣א יַאֲמִ֡ינוּ גַּם֩ לִשְׁנֵ֨י הָאֹת֜וֹת הָאֵ֗לֶּה וְלֹ֤א יִשְׁמְעוּן֙ לְקֹלֶ֔ךָ וְלָקַחְתָּ֙ מִמֵּימֵ֣י הַיְאֹ֔ר וְשָׁפַכְתָּ֖ הַיַּבָּשָׁ֑ה וְהָי֤וּ הַמַּ֙יִם֙ אֲשֶׁ֣ר תִּקַּ֣ח מִן־הַיְאֹ֔ר וְהָי֥וּ לְדָ֖ם בַּיַּבָּֽשֶׁת׃ וְאִם לֹא יַאֲמִינוּ גַּם לִשְׁנֵי הַסִּימָנִים הָאֵלֶּה וְלֹא יִשְׁמְעוּ לְקוֹלְךָ, תִּקַּח מִמֵּימֵי הַיְּאוֹר וְתִשְׁפּוֹךְ לַיַּבָּשָׁה וְהַמַּיִם שֶׁתִּקַּח מֵהַיְּאוֹר יֵהָפְכוּ לְדָם בַּיַּבָּשָׁה.
י וַיֹּ֨אמֶר מֹשֶׁ֣ה אֶל־יְהֹוָה֮ בִּ֣י אֲדֹנָי֒ לֹא֩ אִ֨ישׁ דְּבָרִ֜ים אָנֹ֗כִי גַּ֤ם מִתְּמוֹל֙ גַּ֣ם מִשִּׁלְשֹׁ֔ם גַּ֛ם מֵאָ֥ז דַּבֶּרְךָ֖ אֶל־עַבְדֶּ֑ךָ כִּ֧י כְבַד־פֶּ֛ה וּכְבַ֥ד לָשׁ֖וֹן אָנֹֽכִי׃ מֹשֶׁה אָמַר לַה': בְּבַקָּשָׁה ה', אֲנִי לֹא אִישׁ דִּבּוּרִים גַּם מֵאֶתְמוֹל גַּם מִשִּׁלְשׁוֹם גַּם מֵאָז שֶׁדִּבַּרְתָּ עִם עַבְדְּךָ, כִּי אֲנִי מְגַמְגֵּם וְקָשֶׁה לִי לְדַבֵּר.
יא וַיֹּ֨אמֶר יְהֹוָ֜ה אֵלָ֗יו מִ֣י שָׂ֣ם פֶּה֮ לָֽאָדָם֒ א֚וֹ מִֽי־יָשׂ֣וּם אִלֵּ֔ם א֣וֹ חֵרֵ֔שׁ א֥וֹ פִקֵּ֖חַ א֣וֹ עִוֵּ֑ר הֲלֹ֥א אָנֹכִ֖י יְהֹוָֽה׃ ה' אָמַר לוֹ: מִי שָׂם פֶּה לָאָדָם? אוֹ מִי בָּרָא אִלֵּם אוֹ חֵרֵשׁ אוֹ רוֹאֶה אוֹ עִוֵּר? הֲרֵי רַק אֲנִי – ה'!
יב וְעַתָּ֖ה לֵ֑ךְ וְאָנֹכִי֙ אֶֽהְיֶ֣ה עִם־פִּ֔יךָ וְהוֹרֵיתִ֖יךָ אֲשֶׁ֥ר תְּדַבֵּֽר׃ וְעַכְשָׁיו לֵךְ, וַאֲנִי אֶהְיֶה עִם פִּיךָ וְאַדְרִיךְ אוֹתְךָ מַה לְּדַבֵּר.
יג וַיֹּ֖אמֶר בִּ֣י אֲדֹנָ֑י שְֽׁלַֽח־נָ֖א בְּיַד־תִּשְׁלָֽח׃ אָמַר: בְּבַקָּשָׁה ה', שְׁלַח בְּבַקָּשָׁה אָדָם אַחֵר שֶׁתּוּכַל לִשְׁלוֹחַ.
יד וַיִּֽחַר־אַ֨ף יְהֹוָ֜ה בְּמֹשֶׁ֗ה וַיֹּ֙אמֶר֙ הֲלֹ֨א אַהֲרֹ֤ן אָחִ֙יךָ֙ הַלֵּוִ֔י יָדַ֕עְתִּי כִּֽי־דַבֵּ֥ר יְדַבֵּ֖ר ה֑וּא וְגַ֤ם הִנֵּה־הוּא֙ יֹצֵ֣א לִקְרָאתֶ֔ךָ וְרָאֲךָ֖ וְשָׂמַ֥ח בְּלִבּֽוֹ׃ ה' כָּעַס עַל מֹשֶׁה וְאָמַר: הֲרֵי אַהֲרֹן אָחִיךָ הַלֵּוִי יָדַעְתִּי שֶׁהוּא מֵטִיב לְדַבֵּר וְגַם הִנֵּה הוּא יוֹצֵא לִקְרָאתְךָ, וְהוּא יִרְאֶה אוֹתְךָ וְיִשְׂמַח בְּלִבּוֹ.
טו וְדִבַּרְתָּ֣ אֵלָ֔יו וְשַׂמְתָּ֥ אֶת־הַדְּבָרִ֖ים בְּפִ֑יו וְאָנֹכִ֗י אֶֽהְיֶ֤ה עִם־פִּ֙יךָ֙ וְעִם־פִּ֔יהוּ וְהוֹרֵיתִ֣י אֶתְכֶ֔ם אֵ֖ת אֲשֶׁ֥ר תַּעֲשֽׂוּן׃ דַּבֵּר אֵלָיו וֶאֱמוֹר לוֹ אֶת הַדְּבָרִים שֶׁיֹּאמַר, וַאֲנִי אֶעֱזוֹר לְפִיךָ וּלְפִיו וְאַדְרִיךְ אֶתְכֶם מַה לַּעֲשׂוֹת.
טז וְדִבֶּר־ה֥וּא לְךָ֖ אֶל־הָעָ֑ם וְהָ֤יָה הוּא֙ יִֽהְיֶה־לְּךָ֣ לְפֶ֔ה וְאַתָּ֖ה תִּֽהְיֶה־לּ֥וֹ לֵֽאלֹהִֽים׃ הוּא יְדַבֵּר עֲבוּרְךָ אֶל הָעָם, הוּא יִהְיֶה לְךָ לְפֶה וְאַתָּה תִּהְיֶה לוֹ לִמְפַקֵּד.
יז וְאֶת־הַמַּטֶּ֥ה הַזֶּ֖ה תִּקַּ֣ח בְּיָדֶ֑ךָ אֲשֶׁ֥ר תַּעֲשֶׂה־בּ֖וֹ אֶת־הָאֹתֹֽת׃ וְאֶת הַמַּקֵּל הַזֶּה תִּקַּח בְּיָדְךָ, שֶׁתַּעֲשֶׂה בּוֹ אֶת הַסִּימָנִים.

ששי[עריכה]

פ פָּרָשַׁת הַיְּצִיאָה מִמִּדְיָן
יח וַיֵּ֨לֶךְ מֹשֶׁ֜ה וַיָּ֣שׇׁב ׀ אֶל־יֶ֣תֶר חֹֽתְנ֗וֹ וַיֹּ֤אמֶר לוֹ֙ אֵ֣לְﬞכָה נָּ֗א וְאָשׁ֙וּבָה֙ אֶל־אַחַ֣י אֲשֶׁר־בְּמִצְרַ֔יִם וְאֶרְאֶ֖ה הַעוֹדָ֣ם חַיִּ֑ים וַיֹּ֧אמֶר יִתְר֛וֹ לְמֹשֶׁ֖ה לֵ֥ךְ לְשָׁלֽוֹם׃ מֹשֶׁה הָלַךְ וְחָזַר לְיִתְרוֹ חוֹתְנוֹ, וְאָמַר לוֹ: אֵלֶךְ בְּבַקָּשָׁה וְאֶחֱזוֹר אֶל אַחַי שֶׁבְּמִצְרַיִם וְאֶרְאֶה אִם הֵם עוֹד חַיִּים, אָמַר יִתְרוֹ לְמֹשֶׁה: לֵךְ לְשָׁלוֹם!
יט וַיֹּ֨אמֶר יְהֹוָ֤ה אֶל־מֹשֶׁה֙ בְּמִדְיָ֔ן לֵ֖ךְ שֻׁ֣ב מִצְרָ֑יִם כִּי־מֵ֙תוּ֙ כׇּל־הָ֣אֲנָשִׁ֔ים הַֽמְבַקְשִׁ֖ים אֶת־נַפְשֶֽׁךָ׃ ה' אָמַר לְמֹשֶׁה בְּמִדְיָן: לֵךְ חֲזוֹר לְמִצְרַיִם, כִּי מֵתוּ כָּל הָאֲנָשִׁים שֶׁרוֹצִים לַהֲרוֹג אוֹתְךָ.
כ וַיִּקַּ֨ח מֹשֶׁ֜ה אֶת־אִשְׁתּ֣וֹ וְאֶת־בָּנָ֗יו וַיַּרְכִּבֵם֙ עַֽל־הַחֲמֹ֔ר וַיָּ֖שׇׁב אַ֣רְצָה מִצְרָ֑יִם וַיִּקַּ֥ח מֹשֶׁ֛ה אֶת־מַטֵּ֥ה הָאֱלֹהִ֖ים בְּיָדֽוֹ׃ מֹשֶׁה לָקַח אֶת אִשְׁתּוֹ וְאֶת בָּנָיו וְהִרְכִּיב אוֹתָם עַל הַחֲמוֹר וְחָזַר לְאֶרֶץ מִצְרַיִם, מֹשֶׁה לָקַח אֶת מַקֵּל הָאֱלֹהִים בְּיָדוֹ.
כא וַיֹּ֣אמֶר יְהֹוָה֮ אֶל־מֹשֶׁה֒ בְּלֶכְתְּךָ֙ לָשׁ֣וּב מִצְרַ֔יְמָה רְאֵ֗ה כׇּל־הַמֹּֽפְתִים֙ אֲשֶׁר־שַׂ֣מְתִּי בְיָדֶ֔ךָ וַעֲשִׂיתָ֖ם לִפְנֵ֣י פַרְעֹ֑ה וַאֲנִי֙ אֲחַזֵּ֣ק אֶת־לִבּ֔וֹ וְלֹ֥א יְשַׁלַּ֖ח אֶת־הָעָֽם׃ ה' אָמַר לְמֹשֶׁה: כְּשֶׁאַתָּה חוֹזֵר לְמִצְרַיִם, שִׁים לֵב, אֶת כָּל הַנִּסִּים שֶׁנָּתַתִּי לְךָ לַעֲשׂוֹת – עֲשֵׂה אוֹתָם לִפְנֵי פַּרְעֹה, וַאֲנִי אַקְשִׁיחַ אֶת לִבּוֹ וְהוּא לֹא יְשַׁחְרֵר אֶת הָעָם.
כב וְאָמַרְתָּ֖ אֶל־פַּרְעֹ֑ה כֹּ֚ה אָמַ֣ר יְהֹוָ֔ה בְּנִ֥י בְכֹרִ֖י יִשְׂרָאֵֽל׃ תֹּאמַר לְפַרְעֹה – כָּךְ אָמַר ה': בְּנִי בְּכוֹרִי – יִשְׂרָאֵל.
כג וָאֹמַ֣ר אֵלֶ֗יךָ שַׁלַּ֤ח אֶת־בְּנִי֙ וְיַֽעַבְדֵ֔נִי וַתְּמָאֵ֖ן לְשַׁלְּח֑וֹ הִנֵּה֙ אָנֹכִ֣י הֹרֵ֔ג אֶת־בִּנְךָ֖ בְּכֹרֶֽךָ׃ אָמַרְתִּי לְךָ שַׁחְרֵר אֶת בְּנִי שֶׁיַּעֲבוֹד אוֹתִי! וְסֵרַבְתָּ לְשַׁחְרֵר אוֹתוֹ – הִנֵּה אֲנִי הוֹרֵג אֶת בִּנְךָ בְּכוֹרְךָ.
כד וַיְהִ֥י בַדֶּ֖רֶךְ בַּמָּל֑וֹן וַיִּפְגְּשֵׁ֣הוּ יְהֹוָ֔ה וַיְבַקֵּ֖שׁ הֲמִיתֽוֹ׃ בַּדֶּרֶךְ בַּחֲנִיַּת הַלַּיְלָה ה' פָּגַשׁ אוֹתוֹ וְרָצָה לַהֲרוֹג אוֹתוֹ.
כה וַתִּקַּ֨ח צִפֹּרָ֜ה צֹ֗ר וַתִּכְרֹת֙ אֶת־עׇרְלַ֣ת בְּנָ֔הּ וַתַּגַּ֖ע לְרַגְלָ֑יו וַתֹּ֕אמֶר כִּ֧י חֲתַן־דָּמִ֛ים אַתָּ֖ה לִֽי׃ צִפּוֹרָה לָקְחָה אֶבֶן צוֹר, וְחָתְכָה אֶת עָרְלַת בְּנָהּ וְהִנִּיחָה אוֹתָהּ לְרַגְלֵי מֹשֶׁה, וְאָמְרָה: אַתָּה בַּעֲלִי בְּסַכָּנַת מָוֶת.
כו וַיִּ֖רֶף מִמֶּ֑נּוּ אָ֚ז אָֽמְרָ֔ה חֲתַ֥ן דָּמִ֖ים לַמּוּלֹֽת׃ ה' הֵנִיחַ לוֹ וְלֹא הֲרָגוֹ, אָז אָמְרָה: אַתָּה בַּעַל שֶׁהִתְחַיֵּב בְּמָוֶת בִּגְלַל בְּרִית הַמִּילָה.
פ פָּרָשַׁת הַשִּׁיבָה לְמִצְרַיִם
כז וַיֹּ֤אמֶר יְהֹוָה֙ אֶֽל־אַהֲרֹ֔ן לֵ֛ךְ לִקְרַ֥את מֹשֶׁ֖ה הַמִּדְבָּ֑רָה וַיֵּ֗לֶךְ וַֽיִּפְגְּשֵׁ֛הוּ בְּהַ֥ר הָאֱלֹהִ֖ים וַיִּשַּׁק־לֽוֹ׃ ה' אָמַר לְאַהֲרֹן לֵךְ לִקְרַאת מֹשֶׁה לַמִּדְבָּר, הוּא הָלַךְ וּפָגַשׁ אוֹתוֹ בְּהַר הָאֱלֹהִים, וְהוּא נָשַׁק לוֹ.
כח וַיַּגֵּ֤ד מֹשֶׁה֙ לְאַֽהֲרֹ֔ן אֵ֛ת כׇּל־דִּבְרֵ֥י יְהֹוָ֖ה אֲשֶׁ֣ר שְׁלָח֑וֹ וְאֵ֥ת כׇּל־הָאֹתֹ֖ת אֲשֶׁ֥ר צִוָּֽהוּ׃ מֹשֶׁה סִפֵּר לְאַהֲרֹן אֶת כָּל הַדְּבָרִים שֶׁה' שָׁלַח אוֹתוֹ לוֹמַר וְאֶת כָּל הַסִּימָנִים שֶׁהוּא צִוָּה אוֹתוֹ לַעֲשׂוֹת.
כט וַיֵּ֥לֶךְ מֹשֶׁ֖ה וְאַהֲרֹ֑ן וַיַּ֣אַסְפ֔וּ אֶת־כׇּל־זִקְנֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל׃ מֹשֶׁה וְאַהֲרֹן הָלְכוּ וְאָסְפוּ אֶת כָּל זִקְנֵי בְּנֵי יִשְׂרָאֵל.
ל וַיְדַבֵּ֣ר אַהֲרֹ֔ן אֵ֚ת כׇּל־הַדְּבָרִ֔ים אֲשֶׁר־דִּבֶּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֑ה וַיַּ֥עַשׂ הָאֹתֹ֖ת לְעֵינֵ֥י הָעָֽם׃ אַהֲרֹן דִּבֵּר אֶת כָּל הַדְּבָרִים שֶׁה' דִּבֵּר אֶל מֹשֶׁה, וּמֹשֶׁה עָשָׂה אֶת הַסִּימָנִים לְעֵינֵי הָעָם.
לא וַֽיַּאֲמֵ֖ן הָעָ֑ם וַֽיִּשְׁמְע֡וּ כִּֽי־פָקַ֨ד יְהֹוָ֜ה אֶת־בְּנֵ֣י יִשְׂרָאֵ֗ל וְכִ֤י רָאָה֙ אֶת־עׇנְיָ֔ם וַֽיִּקְּד֖וּ וַיִּֽשְׁתַּחֲוֽוּ׃ הָעָם הֶאֱמִין, וְשָׁמְעוּ שֶׁה' זָכַר אֶת בְּנֵי יִשְׂרָאֵל וְשֶׁהוּא רָאָה אֶת הַשְׁפָּלָתָם, הֵם קָדוּ וְהִשְׁתַּחֲווּ.

שביעי[עריכה]

ה א וְאַחַ֗ר בָּ֚אוּ מֹשֶׁ֣ה וְאַהֲרֹ֔ן וַיֹּאמְר֖וּ אֶל־פַּרְעֹ֑ה כֹּֽה־אָמַ֤ר יְהֹוָה֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל שַׁלַּח֙ אֶת־עַמִּ֔י וְיָחֹ֥גּוּ לִ֖י בַּמִּדְבָּֽר׃ וְאַחַר כָּךְ מֹשֶׁה וְאַהֲרֹן בָּאוּ וְאָמְרוּ לְפַרְעֹה: כָּךְ אָמַר ה' אֱלֹהֵי יִשְׂרָאֵל: שַׁחְרֵר אֶת עַמִּי שֶׁיַּחְגְּגוּ לִי בַּמִּדְבָּר.
ב וַיֹּ֣אמֶר פַּרְעֹ֔ה מִ֤י יְהֹוָה֙ אֲשֶׁ֣ר אֶשְׁמַ֣ע בְּקֹל֔וֹ לְשַׁלַּ֖ח אֶת־יִשְׂרָאֵ֑ל לֹ֤א יָדַ֙עְתִּי֙ אֶת־יְהֹוָ֔ה וְגַ֥ם אֶת־יִשְׂרָאֵ֖ל לֹ֥א אֲשַׁלֵּֽחַ׃ פַּרְעֹה אָמַר: מִי הוּא ה' שֶׁאֶשְׁמַע בְּקוֹלוֹ לְשַׁחְרֵר אֶת יִשְׂרָאֵל?! אֵינֶנִּי מַכִּיר אֶת ה' וְגַם אֶת יִשְׂרָאֵל לֹא אֲשַׁחְרֵר!
ג וַיֹּ֣אמְר֔וּ אֱלֹהֵ֥י הָעִבְרִ֖ים נִקְרָ֣א עָלֵ֑ינוּ נֵ֣לְﬞכָה נָּ֡א דֶּ֩רֶךְ֩ שְׁלֹ֨שֶׁת יָמִ֜ים בַּמִּדְבָּ֗ר וְנִזְבְּחָה֙ לַֽיהֹוָ֣ה אֱלֹהֵ֔ינוּ פֶּ֨ן־יִפְגָּעֵ֔נוּ בַּדֶּ֖בֶר א֥וֹ בֶחָֽרֶב׃ אָמְרוּ לוֹ: אֱלֹהֵי הָעִבְרִים הִתְגַּלָּה אֱלֵינוּ, נֵלֶךְ בְּבַקָּשָׁה לְדֶרֶךְ שֶׁל שְׁלוֹשָׁה יָמִים בַּמִּדְבָּר וְנַקְרִיב לַה' אֱלֹהֵינוּ, שֶׁלֹּא יַהֲרוֹג אוֹתָנוּ בְּדֶבֶר אוֹ בְּמִלְחָמָה.
ד וַיֹּ֤אמֶר אֲלֵהֶם֙ מֶ֣לֶךְ מִצְרַ֔יִם לָ֚מָּה מֹשֶׁ֣ה וְאַהֲרֹ֔ן תַּפְרִ֥יעוּ אֶת־הָעָ֖ם מִמַּֽעֲשָׂ֑יו לְכ֖וּ לְסִבְלֹתֵיכֶֽם׃ אָמַר לָהֶם מֶלֶךְ מִצְרַיִם: מֹשֶׁה וְאַהֲרֹן, לָמָּה אַתֶּם מַפְרִיעִים לָעָם מִמַּעֲשָׂיו? לְכוּ לַעֲבוֹדַתְכֶם!
ה וַיֹּ֣אמֶר פַּרְעֹ֔ה הֵן־רַבִּ֥ים עַתָּ֖ה עַ֣ם הָאָ֑רֶץ וְהִשְׁבַּתֶּ֥ם אֹתָ֖ם מִסִּבְלֹתָֽם׃ פַּרְעֹה אָמַר: הֲרֵי תּוֹשְׁבֵי הָאָרֶץ רַבִּים עַכְשָׁיו, וְכִי תְּבַטְּלוּ אוֹתָם מֵעֲבוֹדָתָם?!
ו וַיְצַ֥ו פַּרְעֹ֖ה בַּיּ֣וֹם הַה֑וּא אֶת־הַנֹּגְשִׂ֣ים בָּעָ֔ם וְאֶת־שֹׁטְרָ֖יו לֵאמֹֽר׃ פַּרְעֹה צִוָּה בַּיּוֹם הַהוּא אֶת הַשּׁוֹטְרִים הַמִּצְרִים שֶׁמְּשַׁעְבְּדִים אֶת הָעָם, וְאֶת הַשּׁוֹטְרִים מִבְּנֵי יִשְׂרָאֵל, וְאָמַר:
ז לֹ֣א תֹאסִפ֞וּן לָתֵ֨ת תֶּ֧בֶן לָעָ֛ם לִלְבֹּ֥ן הַלְּבֵנִ֖ים כִּתְמ֣וֹל שִׁלְשֹׁ֑ם הֵ֚ם יֵֽלְכ֔וּ וְקֹשְׁשׁ֥וּ לָהֶ֖ם תֶּֽבֶן׃ אַל תּוֹסִיפוּ לָתֵת קַשׁ לָעָם לְיַצֵּר מִמֶּנּוּ אֶת הַלְּבֵנִים כְּמוֹ אֶתְמוֹל וְשִׁלְשׁוֹם, הֵם בְּעַצְמָם יֵלְכוּ וְיַאַסְפוּ לְעַצְמָם קַשׁ.
ח וְאֶת־מַתְכֹּ֨נֶת הַלְּבֵנִ֜ים אֲשֶׁ֣ר הֵם֩ עֹשִׂ֨ים תְּמ֤וֹל שִׁלְשֹׁם֙ תָּשִׂ֣ימוּ עֲלֵיהֶ֔ם לֹ֥א תִגְרְע֖וּ מִמֶּ֑נּוּ כִּֽי־נִרְפִּ֣ים הֵ֔ם עַל־כֵּ֗ן הֵ֤ם צֹֽעֲקִים֙ לֵאמֹ֔ר נֵלְכָ֖ה נִזְבְּחָ֥ה לֵאלֹהֵֽינוּ׃ וְאֶת מִסְפַּר הַלְּבֵנִים שֶׁעָשׂוּ אֶתְמוֹל וְשִׁלְשׁוֹם תָּשִׂימוּ עֲלֵיהֶם לְחוֹבָה, אַל תַּמְעִיטוּ מִמֶּנּוּ, כִּי הֵם בַּטְלָנִים לָכֵן הֵם צוֹעֲקִים: נֵלֵךְ נַקְרִיב קָרְבָּנוֹת לֵאלֹהֵינוּ.
ט תִּכְבַּ֧ד הָעֲבֹדָ֛ה עַל־הָאֲנָשִׁ֖ים וְיַעֲשׂוּ־בָ֑הּ וְאַל־יִשְׁע֖וּ בְּדִבְרֵי־שָֽׁקֶר׃ וְכָךְ תִּקְשֶׁה הָעֲבוֹדָה לָאֲנָשִׁים וְהֵם יַעַסְקוּ בָּהּ וְלֹא יְנַסּוּ לְהִנָּצֵל בְּאֶמְצָעוּת דְּבָרִים בְּטֵלִים.
י וַיֵּ֨צְא֜וּ נֹגְשֵׂ֤י הָעָם֙ וְשֹׁ֣טְרָ֔יו וַיֹּאמְר֥וּ אֶל־הָעָ֖ם לֵאמֹ֑ר כֹּ֚ה אָמַ֣ר פַּרְעֹ֔ה אֵינֶ֛נִּי נֹתֵ֥ן לָכֶ֖ם תֶּֽבֶן׃ הַשּׁוֹטְרִים הַמִּצְרִים שֶׁמְּשַׁעְבְּדִים אֶת הָעָם וְהַשּׁוֹטְרִים מִבְּנֵי יִשְׂרָאֵל יָצְאוּ וְאָמְרוּ לָעָם – כָּךְ אָמַר פַּרְעֹה: אֵינֶנִּי נוֹתֵן לָכֶם קַשׁ.
יא אַתֶּ֗ם לְכ֨וּ קְח֤וּ לָכֶם֙ תֶּ֔בֶן מֵאֲשֶׁ֖ר תִּמְצָ֑אוּ כִּ֣י אֵ֥ין נִגְרָ֛ע מֵעֲבֹדַתְכֶ֖ם דָּבָֽר׃ אַתֶּם לְכוּ קְחוּ לָכֶם קַשׁ מִמָּקוֹם שֶׁתִּמְצְאוּ כִּי לֹא מִתְמַעֵט שׁוּם דָּבָר מֵחוֹבַת עֲבוֹדַתְכֶם.
יב וַיָּ֥פֶץ הָעָ֖ם בְּכׇל־אֶ֣רֶץ מִצְרָ֑יִם לְקֹשֵׁ֥שׁ קַ֖שׁ לַתֶּֽבֶן׃ הָעָם הִתְפַּזֵּר בְּכָל אֶרֶץ מִצְרַיִם לֶאֱסוֹף קַשׁ בִּמְקוֹם הַקַּשׁ שֶׁהָיָה נִתַּן לָהֶם.
יג וְהַנֹּגְשִׂ֖ים אָצִ֣ים לֵאמֹ֑ר כַּלּ֤וּ מַעֲשֵׂיכֶם֙ דְּבַר־י֣וֹם בְּיוֹמ֔וֹ כַּאֲשֶׁ֖ר בִּהְי֥וֹת הַתֶּֽבֶן׃ וְהַשּׁוֹטְרִים הַמִּצְרִים מְאִיצִים בָּהֶם וְאוֹמְרִים: הַשְׁלִימוּ אֶת מַעֲשֵׂיכֶם כְּמוֹ שֶׁעֲשִׂיתֶם בְּכָל יוֹם בַּזְּמַן שֶׁקִּבַּלְתֶּם אֶת הַקַּשׁ!
יד וַיֻּכּ֗וּ שֹֽׁטְרֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁר־שָׂ֣מוּ עֲלֵהֶ֔ם נֹגְשֵׂ֥י פַרְעֹ֖ה לֵאמֹ֑ר מַדּ֡וּעַ לֹא֩ כִלִּיתֶ֨ם חׇקְכֶ֤ם לִלְבֹּן֙ כִּתְמ֣וֹל שִׁלְשֹׁ֔ם גַּם־תְּמ֖וֹל גַּם־הַיּֽוֹם׃ הַשּׁוֹטְרִים מִבְּנֵי יִשְׂרָאֵל שֶׁשּׁוֹטְרֵי פַּרְעֹה מִנּוּ עַל בְּנֵי יִשְׂרָאֵל הֻכּוּ, בַּטַּעֲנָה: לָמָּה גַּם אֶתְמוֹל וְגַם הַיּוֹם לֹא הִשְׁלַמְתֶּם אֶת חוֹבַתְכֶם לְיַצֵּר לְבֵנִים כְּמוֹ שִׁלְשׁוֹם?!
טו וַיָּבֹ֗אוּ שֹֽׁטְרֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַיִּצְעֲק֥וּ אֶל־פַּרְעֹ֖ה לֵאמֹ֑ר לָ֧מָּה תַעֲשֶׂ֦ה כֹ֖ה לַעֲבָדֶֽיךָ׃ הַשּׁוֹטְרִים מִבְּנֵי יִשְׂרָאֵל בָּאוּ וְצָעֲקוּ אֶל פַּרְעֹה וְאָמְרוּ: לָמָּה אַתָּה עוֹשֶׂה כָּכָה לַעֲבָדֶיךָ?
טז תֶּ֗בֶן אֵ֤ין נִתָּן֙ לַעֲבָדֶ֔יךָ וּלְבֵנִ֛ים אֹמְרִ֥ים לָ֖נוּ עֲשׂ֑וּ וְהִנֵּ֧ה עֲבָדֶ֛יךָ מֻכִּ֖ים וְחָטָ֥את עַמֶּֽךָ׃ לֹא נוֹתְנִים לַעֲבָדֶיךָ קַשׁ, וְאוֹמְרִים לָנוּ: עֲשׂוּ לְבֵנִים, וְהִנֵּה אָנוּ עֲבָדֶיךָ מֻכִּים, וְעַמְּךָ נֶחְשָׁבִים חוֹטְאִים!
יז וַיֹּ֛אמֶר נִרְפִּ֥ים אַתֶּ֖ם נִרְפִּ֑ים עַל־כֵּן֙ אַתֶּ֣ם אֹֽמְרִ֔ים נֵלְכָ֖ה נִזְבְּחָ֥ה לַֽיהֹוָֽה׃ אָמַר לָהֶם: אַתֶּם בַּטְלָנִים – בַּטְלָנִים! לָכֵן אַתֶּם אוֹמְרִים: נֵלֵךְ וְנַקְרִיב קָרְבָּנוֹת לַה'!
יח וְעַתָּה֙ לְכ֣וּ עִבְד֔וּ וְתֶ֖בֶן לֹא־יִנָּתֵ֣ן לָכֶ֑ם וְתֹ֥כֶן לְבֵנִ֖ים תִּתֵּֽנוּ׃ וְעַכְשָׂיו, לְכוּ עִבְדוּ, וְלֹא יִנָּתֵן לָכֶם קַשׁ, וּתְסַפְּקוּ אֶת כַּמּוּת הַלְּבֵנִים.
יט וַיִּרְא֞וּ שֹֽׁטְרֵ֧י בְנֵֽי־יִשְׂרָאֵ֛ל אֹתָ֖ם בְּרָ֣ע לֵאמֹ֑ר לֹא־תִגְרְע֥וּ מִלִּבְנֵיכֶ֖ם דְּבַר־י֥וֹם בְּיוֹמֽוֹ׃ שׁוֹטְרֵי בְּנֵי יִשְׂרָאֵל רָאוּ אֶת עַצְמָם בְּמַצָּבָם הָרָע – שֶׁנֶּאֱמַר לָהֶם: אַל תַּמְעִיטוּ מִלְּיַצֵּר אֶת מִכְסַת הַלְּבֵנִים הַיּוֹמִית.
כ וַֽיִּפְגְּעוּ֙ אֶת־מֹשֶׁ֣ה וְאֶֽת־אַהֲרֹ֔ן נִצָּבִ֖ים לִקְרָאתָ֑ם בְּצֵאתָ֖ם מֵאֵ֥ת פַּרְעֹֽה׃ וְהֵם פָּגְשׁוּ אֶת מֹשֶׁה וְאֶת אַהֲרֹן עוֹמְדִים לִקְרָאתָם כְּשֶׁיָּצְאוּ מִפַּרְעֹה.
כא וַיֹּאמְר֣וּ אֲלֵהֶ֔ם יֵ֧רֶא יְהֹוָ֛ה עֲלֵיכֶ֖ם וְיִשְׁפֹּ֑ט אֲשֶׁ֧ר הִבְאַשְׁתֶּ֣ם אֶת־רֵיחֵ֗נוּ בְּעֵינֵ֤י פַרְעֹה֙ וּבְעֵינֵ֣י עֲבָדָ֔יו לָֽתֶת־חֶ֥רֶב בְּיָדָ֖ם לְהׇרְגֵֽנוּ׃ הֵם אָמְרוּ לָהֶם: ה' יִרְאֶה אֶת מַעֲשֵׂיכֶם וְיַעֲנִישׁ אֶתְכֶם עַל שֶׁהִסְרַחְתֶּם אֶת רֵיחֵנוּ בְּעֵינֵי פַּרְעֹה וּבְעֵינֵי עֲבָדָיו, כְּאִלּוּ לָתֵת לָהֶם חֶרֶב לַהֲרוֹג אוֹתָנוּ.

מפטיר[עריכה]

כב וַיָּ֧שׇׁב מֹשֶׁ֛ה אֶל־יְהֹוָ֖ה וַיֹּאמַ֑ר אֲדֹנָ֗י לָמָ֤ה הֲרֵעֹ֙תָה֙ לָעָ֣ם הַזֶּ֔ה לָ֥מָּה זֶּ֖ה שְׁלַחְתָּֽנִי׃ מֹשֶׁה חָזַר לְדַבֵּר אֶל ה' וְאָמַר: ה', לָמָה הֵרַעְתָּ לָעָם הַזֶּה, לָמָּה שָׁלַחְתָּ אוֹתִי?
כג וּמֵאָ֞ז בָּ֤אתִי אֶל־פַּרְעֹה֙ לְדַבֵּ֣ר בִּשְׁמֶ֔ךָ הֵרַ֖ע לָעָ֣ם הַזֶּ֑ה וְהַצֵּ֥ל לֹא־הִצַּ֖לְתָּ אֶת־עַמֶּֽךָ׃ וּמֵאָז שֶׁבָּאתִי לְפַרְעֹה לְדַבֵּר בְּשִׁמְךָ נִהְיָה יוֹתֵר רַע לָעָם הַזֶּה, וְלֹא הִצַּלְתָּ אֶת עַמְּךָ.
ו א וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה עַתָּ֣ה תִרְאֶ֔ה אֲשֶׁ֥ר אֶֽעֱשֶׂ֖ה לְפַרְעֹ֑ה כִּ֣י בְיָ֤ד חֲזָקָה֙ יְשַׁלְּחֵ֔ם וּבְיָ֣ד חֲזָקָ֔ה יְגָרְשֵׁ֖ם מֵאַרְצֽוֹ׃ אָמַר ה' לְמֹשֶׁה: עַכְשָׁיו תִּרְאֶה אֶת מַה שֶּׁאֶעֱשֶׂה לְפַרְעֹה, שֶׁבִּגְלַל יָד חֲזָקָה שֶׁתַּכֶּה בּוֹ הוּא יְשַׁחְרֵר אוֹתָם – בִּגְלַל יָד חֲזָקָה הוּא יְגָרֵשׁ אוֹתָם מֵאַרְצוֹ.
