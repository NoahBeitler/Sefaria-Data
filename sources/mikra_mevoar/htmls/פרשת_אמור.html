ראשון[עריכה]

פ פָּרָשַׁת אִסּוּרֵי הַכֹּהֵן
כא א וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה אֱמֹ֥ר אֶל־הַכֹּהֲנִ֖ים בְּנֵ֣י אַהֲרֹ֑ן וְאָמַרְתָּ֣ אֲלֵהֶ֔ם לְנֶ֥פֶשׁ לֹֽא־יִטַּמָּ֖א בְּעַמָּֽיו׃ ה' אָמַר לְמֹשֶׁה: דַּבֵּר אֶל הַכֹּהֲנִים בְּנֵי אַהֲרֹן, וֶאֱמוֹר לָהֶם שֶׁאַף אֶחָד מֵהֶם לֹא יִטַּמָּא לְאָדָם מֵת מֵעַמּוֹ.
ב כִּ֚י אִם־לִשְׁאֵר֔וֹ הַקָּרֹ֖ב אֵלָ֑יו לְאִמּ֣וֹ וּלְאָבִ֔יו וְלִבְנ֥וֹ וּלְבִתּ֖וֹ וּלְאָחִֽיו׃ אֶלָּא רַק לִקְרוֹב מִשְׁפַּחְתּוֹ הַקָּרוֹב אֵלָיו – לְאִמּוֹ וּלְאָבִיו, וְלִבְנוֹ וּלְבִתּוֹ וּלְאָחִיו.
ג וְלַאֲחֹת֤וֹ הַבְּתוּלָה֙ הַקְּרוֹבָ֣ה אֵלָ֔יו אֲשֶׁ֥ר לֹֽא־הָיְתָ֖ה לְאִ֑ישׁ לָ֖הּ יִטַּמָּֽא׃ וְלַאֲחוֹתוֹ הַבְּתוּלָה הַקְּרוֹבָה אֵלָיו שֶׁלֹּא הִתְחַתְּנָה עִם אִישׁ – לָהּ מֻתָּר לוֹ לְהִטַּמֵּא.
ד לֹ֥א יִטַּמָּ֖א בַּ֣עַל בְּעַמָּ֑יו לְהֵ֖חַלּֽוֹ׃ כֹּהֵן שֶׁנָּשׂוּי לְאִשָּׁה מֵעַמּוֹ שֶׁאֲסוּרָה לוֹ וּבָהּ מִתְחַלְּלִים צֶאֱצָאָיו – לֹא יִטַּמָּא לָהּ.
ה לֹֽא־יִקְרְח֤וּ יקרחה קׇרְחָה֙ בְּרֹאשָׁ֔ם וּפְאַ֥ת זְקָנָ֖ם לֹ֣א יְגַלֵּ֑חוּ וּבִ֨בְשָׂרָ֔ם לֹ֥א יִשְׂרְט֖וּ שָׂרָֽטֶת׃ הֵם לֹא יַעֲשׂוּ קָרַחַת בְּרֹאשָׁם וְלֹא יְגַלְּחוּ אֶת קְצָווֹת זְקָנָם, וְלֹא יִשְׂרְטוּ שְׂרִיטָה בְּגוּפָם בַּאֲבֵלוּתָם.
ו קְדֹשִׁ֤ים יִהְיוּ֙ לֵאלֹ֣הֵיהֶ֔ם וְלֹ֣א יְחַלְּל֔וּ שֵׁ֖ם אֱלֹהֵיהֶ֑ם כִּי֩ אֶת־אִשֵּׁ֨י יְהֹוָ֜ה לֶ֧חֶם אֱלֹהֵיהֶ֛ם הֵ֥ם מַקְרִיבִ֖ם וְהָ֥יוּ קֹֽדֶשׁ׃ הֵם יִהְיוּ קְדוֹשִׁים לֵאלֹהֵיהֶם וְלֹא יְחַלְּלוּ אֶת שֵׁם אֱלֹהֵיהֶם, כִּי אֶת קָרְבָּנוֹת ה' – אֶת לֶחֶם אֱלֹהֵיהֶם הֵם מַקְרִיבִים, לָכֵן יִהְיוּ קוֹדֶשׁ.
ז אִשָּׁ֨ה זֹנָ֤ה וַחֲלָלָה֙ לֹ֣א יִקָּ֔חוּ וְאִשָּׁ֛ה גְּרוּשָׁ֥ה מֵאִישָׁ֖הּ לֹ֣א יִקָּ֑חוּ כִּֽי־קָדֹ֥שׁ ה֖וּא לֵאלֹהָֽיו׃ לֹא יִשְׂאוּ אִשָּׁה שֶׁזִּנְּתָה עִם אִישׁ הָאָסוּר לָהּ אוֹ עִם כֹּהֵן הָאָסוּר לָהּ, וְלֹא יִשְׂאוּ אִשָּׁה גְּרוּשָׁה מִבַּעֲלָהּ, כִּי הַכֹּהֵן קָדוֹשׁ לֵאלֹהָיו.
ח וְקִ֨דַּשְׁתּ֔וֹ כִּֽי־אֶת־לֶ֥חֶם אֱלֹהֶ֖יךָ ה֣וּא מַקְרִ֑יב קָדֹשׁ֙ יִֽהְיֶה־לָּ֔ךְ כִּ֣י קָד֔וֹשׁ אֲנִ֥י יְהֹוָ֖ה מְקַדִּשְׁכֶֽם׃ קַדֵּשׁ אוֹתוֹ כִּי הוּא מַקְרִיב אֶת קָרְבְּנוֹת אֱלֹהֶיךָ, הוּא יִהְיֶה קָדוֹשׁ עֲבוּרְךָ כִּי אֲנִי – ה' שֶׁמְּקַדֵּשׁ אֶתְכֶם – קָדוֹשׁ.
ט וּבַת֙ אִ֣ישׁ כֹּהֵ֔ן כִּ֥י תֵחֵ֖ל לִזְנ֑וֹת אֶת־אָבִ֙יהָ֙ הִ֣יא מְחַלֶּ֔לֶת בָּאֵ֖שׁ תִּשָּׂרֵֽף׃ וּבַת נְשׁוּאָה שֶׁל אִישׁ כֹּהֵן, אִם תִּתְחַלֵּל בִּזְנוּת – הִיא מְחַלֶּלֶת אֶת אָבִיהָ, לָכֵן תִּשָּׂרֵף בָּאֵשׁ.
ס פָּרָשַׁת אִסּוּרֵי הַכֹּהֵן הַגָּדוֹל
י וְהַכֹּהֵן֩ הַגָּד֨וֹל מֵאֶחָ֜יו אֲֽשֶׁר־יוּצַ֥ק עַל־רֹאשׁ֣וֹ ׀ שֶׁ֤מֶן הַמִּשְׁחָה֙ וּמִלֵּ֣א אֶת־יָד֔וֹ לִלְבֹּ֖שׁ אֶת־הַבְּגָדִ֑ים אֶת־רֹאשׁוֹ֙ לֹ֣א יִפְרָ֔ע וּבְגָדָ֖יו לֹ֥א יִפְרֹֽם׃ וְהַכֹּהֵן הַגָּדוֹל מֵאֶחָיו שֶׁיָּצְקוּ עַל רֹאשׁוֹ אֶת הַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ, וְקִבֵּל אֶת הַתַּפְקִיד לִלְבּוֹשׁ אֶת בְּגָדָיו הַמְּיוּחָדִים, לֹא יְגַדֵּל מִדַּי אֶת שְׂעַר רֹאשׁוֹ, וְלֹא יִפְרוֹם אֶת בְּגָדָיו בַּאֲבֵלוּת.
יא וְעַ֛ל כׇּל־נַפְשֹׁ֥ת מֵ֖ת לֹ֣א יָבֹ֑א לְאָבִ֥יו וּלְאִמּ֖וֹ לֹ֥א יִטַּמָּֽא׃ וְלֹא יָבוֹא אֶל שׁוּם אָדָם מֵת – אֲפִלּוּ לְאָבִיו וּלְאִמּוֹ לֹא יִטַּמָּא.
יב וּמִן־הַמִּקְדָּשׁ֙ לֹ֣א יֵצֵ֔א וְלֹ֣א יְחַלֵּ֔ל אֵ֖ת מִקְדַּ֣שׁ אֱלֹהָ֑יו כִּ֡י נֵ֠זֶר שֶׁ֣מֶן מִשְׁחַ֧ת אֱלֹהָ֛יו עָלָ֖יו אֲנִ֥י יְהֹוָֽה׃ וְלֹא יֵצֵא מֵעֲבוֹדַת הַמִּקְדָּשׁ וְלֹא יְחַלֵּל אֶת מִקְדַּשׁ אֱלֹהָיו, כִּי יֵשׁ עָלָיו "כֶּתֶר" שֶׁל הַשֶּׁמֶן שֶׁמָּרְחוּ עָלָיו לִכְבוֹד אֱלֹהָיו, אֲנִי ה'.
יג וְה֕וּא אִשָּׁ֥ה בִבְתוּלֶ֖יהָ יִקָּֽח׃ וְהוּא יִשָּׂא רַק אִשָּׁה שֶׁאַף אִישׁ לֹא שָׁכַב אִתָּהּ.
יד אַלְמָנָ֤ה וּגְרוּשָׁה֙ וַחֲלָלָ֣ה זֹנָ֔ה אֶת־אֵ֖לֶּה לֹ֣א יִקָּ֑ח כִּ֛י אִם־בְּתוּלָ֥ה מֵעַמָּ֖יו יִקַּ֥ח אִשָּֽׁה׃ אַלְמָנָה אוֹ גְּרוּשָׁה אוֹ אִשָּׁה שֶׁזִּנְּתָה עִם כֹּהֵן הָאָסוּר לָהּ אוֹ עִם אִישׁ אַחֵר הָאָסוּר לָהּ – אֶת אֵלֶּה לֹא יִשָּׂא, אֶלָּא יִשָּׂא רַק אִשָּׁה מֵעַמּוֹ שֶׁאִישׁ לֹא שָׁכַב אִתָּהּ.
טו וְלֹֽא־יְחַלֵּ֥ל זַרְע֖וֹ בְּעַמָּ֑יו כִּ֛י אֲנִ֥י יְהֹוָ֖ה מְקַדְּשֽׁוֹ׃ וְלֹא יְחַלֵּל בְּנָשִׁים אֵלֶּה אֶת צֶאֱצָאָיו שֶׁלֹּא יִהְיוּ קְדוֹשִׁים בְּעַמּוֹ, כִּי אֲנִי ה' מְקַדֵּשׁ אוֹתוֹ.

שני[עריכה]

ס פָּרָשַׁת הַמּוּמִים
טז וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יז דַּבֵּ֥ר אֶֽל־אַהֲרֹ֖ן לֵאמֹ֑ר אִ֣ישׁ מִֽזַּרְעֲךָ֞ לְדֹרֹתָ֗ם אֲשֶׁ֨ר יִהְיֶ֥ה בוֹ֙ מ֔וּם לֹ֣א יִקְרַ֔ב לְהַקְרִ֖יב לֶ֥חֶם אֱלֹהָֽיו׃ דַּבֵּר אֶל אַהֲרֹן וֶאֱמוֹר: אִישׁ מִצֶּאֱצָאֶיךָ בְּכָל הַדּוֹרוֹת שֶׁיִּהְיֶה בּוֹ מוּם לֹא יִגַּשׁ לְהַקְרִיב אֶת קָרְבַּן אֱלֹהָיו.
יח כִּ֥י כׇל־אִ֛ישׁ אֲשֶׁר־בּ֥וֹ מ֖וּם לֹ֣א יִקְרָ֑ב אִ֤ישׁ עִוֵּר֙ א֣וֹ פִסֵּ֔חַ א֥וֹ חָרֻ֖ם א֥וֹ שָׂרֽוּעַ׃ כִּי כָּל אִישׁ שֶׁיֵּשׁ בּוֹ מוּם לֹא יִגַּשׁ – אִישׁ עִוֵּר אוֹ צוֹלֵעַ אוֹ בַּעַל אָף שָׁקוּעַ אוֹ בַּעַל אֵיבָר גָּדוֹל מִדַּי.
יט א֣וֹ אִ֔ישׁ אֲשֶׁר־יִהְיֶ֥ה ב֖וֹ שֶׁ֣בֶר רָ֑גֶל א֖וֹ שֶׁ֥בֶר יָֽד׃ אוֹ אִישׁ שֶׁיִּהְיֶה לוֹ שֶׁבֶר בָּרֶגֶל אוֹ שֶׁבֶר בַּיָּד.
כ אֽוֹ־גִבֵּ֣ן אוֹ־דַ֔ק א֖וֹ תְּבַלֻּ֣ל בְּעֵינ֑וֹ א֤וֹ גָרָב֙ א֣וֹ יַלֶּ֔פֶת א֖וֹ מְר֥וֹחַ אָֽשֶׁךְ׃ אוֹ גִּבֵּן, אוֹ גַּמָּד, אוֹ שֶׁיֵּשׁ כְּעֵין שַׁבְּלוּל בְּעֵינוֹ, אוֹ בַּעַל שְׁחִין יָבֵשׁ אוֹ בַּעַל שְׁחִין לַח אוֹ שֶׁאֲשָׁכָיו מְעוּכִים.
כא כׇּל־אִ֞ישׁ אֲשֶׁר־בּ֣וֹ מ֗וּם מִזֶּ֙רַע֙ אַהֲרֹ֣ן הַכֹּהֵ֔ן לֹ֣א יִגַּ֔שׁ לְהַקְרִ֖יב אֶת־אִשֵּׁ֣י יְהֹוָ֑ה מ֣וּם בּ֔וֹ אֵ֚ת לֶ֣חֶם אֱלֹהָ֔יו לֹ֥א יִגַּ֖שׁ לְהַקְרִֽיב׃ כָּל אִישׁ מִצֶּאֱצָאֵי אַהֲרֹן הַכֹּהֵן שֶׁיֵּשׁ בּוֹ מוּם לֹא יִגַּשׁ לְהַקְרִיב אֶת קָרְבְּנוֹת ה', כֵּיוָן שֶׁיֵּשׁ בּוֹ מוּם הוּא לֹא יִגַּשׁ לְהַקְרִיב אֶת קָרְבְּנוֹת אֱלֹהָיו.
כב לֶ֣חֶם אֱלֹהָ֔יו מִקׇּדְשֵׁ֖י הַקֳּדָשִׁ֑ים וּמִן־הַקֳּדָשִׁ֖ים יֹאכֵֽל׃ אֲבָל הוּא רַשַּׁאי לֶאֱכוֹל מִקָּרְבְּנוֹת אֱלֹהָיו – מִקָּדְשֵׁי הַקֳּדָשִׁים וּמֵהַקֳּדָשִׁים.
כג אַ֣ךְ אֶל־הַפָּרֹ֜כֶת לֹ֣א יָבֹ֗א וְאֶל־הַמִּזְבֵּ֛חַ לֹ֥א יִגַּ֖שׁ כִּֽי־מ֣וּם בּ֑וֹ וְלֹ֤א יְחַלֵּל֙ אֶת־מִקְדָּשַׁ֔י כִּ֛י אֲנִ֥י יְהֹוָ֖ה מְקַדְּשָֽׁם׃ אֲבָל לֹא יָבוֹא אֶל מֵעֵבֶר לַפָּרוֹכֶת וְלֹא יִגַּשׁ אֶל הַמִּזְבֵּחַ כִּי יֵשׁ בּוֹ מוּם, וְלֹא יְחַלֵּל אֶת קָדָשַׁי כִּי אֲנִי ה' מְקַדֵּשׁ אוֹתָם.
כד וַיְדַבֵּ֣ר מֹשֶׁ֔ה אֶֽל־אַהֲרֹ֖ן וְאֶל־בָּנָ֑יו וְאֶֽל־כׇּל־בְּנֵ֖י יִשְׂרָאֵֽל׃ מֹשֶׁה אָמַר דְּבָרִים אֵלֶּה לְאַהֲרֹן וּלְבָנָיו וּלְכָל בְּנֵי יִשְׂרָאֵל.
פ פָּרָשַׁת זְהִירוּת הַכֹּהֲנִים מִטֻּמְאָה
כב א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב דַּבֵּ֨ר אֶֽל־אַהֲרֹ֜ן וְאֶל־בָּנָ֗יו וְיִנָּֽזְרוּ֙ מִקׇּדְשֵׁ֣י בְנֵֽי־יִשְׂרָאֵ֔ל וְלֹ֥א יְחַלְּל֖וּ אֶת־שֵׁ֣ם קׇדְשִׁ֑י אֲשֶׁ֨ר הֵ֧ם מַקְדִּשִׁ֛ים לִ֖י אֲנִ֥י יְהֹוָֽה׃ דַּבֵּר אֶל אַהֲרֹן וְאֶל בָּנָיו שֶׁיִּפְרְשׁוּ מִקָּדְשֵׁי בְּנֵי יִשְׂרָאֵל בִּזְמַן שֶׁהֵם טְמֵאִים, וְכָךְ לֹא יְחַלְּלוּ אֶת שֵׁם קָדְשִׁי וְאֶת מַה שֶּׁהֵם מַקְדִּישִׁים לִכְבוֹדִי, אֲנִי ה'.
ג אֱמֹ֣ר אֲלֵהֶ֗ם לְדֹרֹ֨תֵיכֶ֜ם כׇּל־אִ֣ישׁ ׀ אֲשֶׁר־יִקְרַ֣ב מִכׇּל־זַרְעֲכֶ֗ם אֶל־הַקֳּדָשִׁים֙ אֲשֶׁ֨ר יַקְדִּ֤ישׁוּ בְנֵֽי־יִשְׂרָאֵל֙ לַֽיהֹוָ֔ה וְטֻמְאָת֖וֹ עָלָ֑יו וְנִכְרְתָ֞ה הַנֶּ֧פֶשׁ הַהִ֛וא מִלְּפָנַ֖י אֲנִ֥י יְהֹוָֽה׃ אֱמוֹר לָהֶם: בְּכָל הַדּוֹרוֹת, כָּל אִישׁ מִכָּל צֶאֱצָאֵיכֶם שֶׁיִּגַּשׁ כְּשֶׁהוּא טָמֵא אֶל הַקֳּדָשִׁים שֶׁבְּנֵי יִשְׂרָאֵל יַקְדִּישׁוּ לַה' – הַנֶּפֶשׁ הַהִיא תֹּאבַד מִלְּפָנַי, אֲנִי ה'.
ד אִ֣ישׁ אִ֞ישׁ מִזֶּ֣רַע אַהֲרֹ֗ן וְה֤וּא צָר֙וּעַ֙ א֣וֹ זָ֔ב בַּקֳּדָשִׁים֙ לֹ֣א יֹאכַ֔ל עַ֖ד אֲשֶׁ֣ר יִטְהָ֑ר וְהַנֹּגֵ֙עַ֙ בְּכׇל־טְמֵא־נֶ֔פֶשׁ א֣וֹ אִ֔ישׁ אֲשֶׁר־תֵּצֵ֥א מִמֶּ֖נּוּ שִׁכְבַת־זָֽרַע׃ כָּל אִישׁ מִצֶּאֱצָאֵי אַהֲרֹן שֶׁהוּא צָרוּעַ אוֹ זָב לֹא יֹאכַל בַּקֳּדָשִׁים עַד שֶׁיִּטְהַר, וְכֵן הַנּוֹגֵעַ בְּכָל מִי שֶׁנִּטְמָא לְנֶפֶשׁ, אוֹ אִישׁ שֶׁתֵּצֵא מִמֶּנּוּ שִׁכְבַת זֶרַע.
ה אוֹ־אִישׁ֙ אֲשֶׁ֣ר יִגַּ֔ע בְּכׇל־שֶׁ֖רֶץ אֲשֶׁ֣ר יִטְמָא־ל֑וֹ א֤וֹ בְאָדָם֙ אֲשֶׁ֣ר יִטְמָא־ל֔וֹ לְכֹ֖ל טֻמְאָתֽוֹ׃ אוֹ אִישׁ שֶׁיִּגַּע בְּכָל שֶׁרֶץ שֶׁמְּטַמֵּא, אוֹ בְּאָדָם שֶׁיִּטָּמֵא בְּכָל טֻמְאָה שֶׁמְּטַמֵּאת.
ו נֶ֚פֶשׁ אֲשֶׁ֣ר תִּגַּע־בּ֔וֹ וְטָמְאָ֖ה עַד־הָעָ֑רֶב וְלֹ֤א יֹאכַל֙ מִן־הַקֳּדָשִׁ֔ים כִּ֛י אִם־רָחַ֥ץ בְּשָׂר֖וֹ בַּמָּֽיִם׃ אִישׁ שֶׁיִּגַּע בּוֹ – יִהְיֶה טָמֵא עַד הָעֶרֶב, וְלֹא יֹאכַל מֵהַקֳּדָשִׁים אֶלָּא אִם כֵּן הִטְבִּיל אֶת גּוּפוֹ בְּמַיִם.
ז וּבָ֥א הַשֶּׁ֖מֶשׁ וְטָהֵ֑ר וְאַחַר֙ יֹאכַ֣ל מִן־הַקֳּדָשִׁ֔ים כִּ֥י לַחְמ֖וֹ הֽוּא׃ כְּשֶׁתִּשְׁקַע הַשֶּׁמֶשׁ הוּא יִהְיֶה טָהוֹר, וְאַחַר כָּךְ יֹאכַל מֵהַקֳּדָשִׁים כִּי זֶהוּ מְזוֹנוֹ.
ח נְבֵלָ֧ה וּטְרֵפָ֛ה לֹ֥א יֹאכַ֖ל לְטׇמְאָה־בָ֑הּ אֲנִ֖י יְהֹוָֽה׃ לֹא יֹאכַל נְבֵלָה וּטְרֵפָה לְהִטָּמֵא בָּהּ, אֲנִי ה'.
ט וְשָׁמְר֣וּ אֶת־מִשְׁמַרְתִּ֗י וְלֹֽא־יִשְׂא֤וּ עָלָיו֙ חֵ֔טְא וּמֵ֥תוּ ב֖וֹ כִּ֣י יְחַלְּלֻ֑הוּ אֲנִ֥י יְהֹוָ֖ה מְקַדְּשָֽׁם׃ הֵם יִשְׁמְרוּ אֶת מִצְווֹתַי וְכָךְ לֹא יְקַבְּלוּ עוֹנֶשׁ עַל חֵטְא – וְיָמוּתוּ בְּכָךְ כְּשֶׁיְּחַלְּלוּ אֶת שְׁמִי, אֲנִי ה' מְקַדֵּשׁ אוֹתָם.
י וְכׇל־זָ֖ר לֹא־יֹ֣אכַל קֹ֑דֶשׁ תּוֹשַׁ֥ב כֹּהֵ֛ן וְשָׂכִ֖יר לֹא־יֹ֥אכַל קֹֽדֶשׁ׃ וְכָל זָר לֹא יֹאכַל קוֹדֶשׁ, עֶבֶד עִבְרִי שֶׁל כֹּהֵן וּשְׂכִירוֹ לֹא יֹאכְלוּ קוֹדֶשׁ.
יא וְכֹהֵ֗ן כִּֽי־יִקְנֶ֥ה נֶ֙פֶשׁ֙ קִנְיַ֣ן כַּסְפּ֔וֹ ה֖וּא יֹ֣אכַל בּ֑וֹ וִילִ֣יד בֵּית֔וֹ הֵ֖ם יֹאכְל֥וּ בְלַחְמֽוֹ׃ וְאִם כֹּהֵן יִקְנֶה בְּכַסְפּוֹ גּוֹי שֶׁיִּהְיֶה עַבְדּוֹ וּרְכוּשׁוֹ – הוּא יֹאכַל מֵהַקּוֹדֶשׁ, וְכֵן עֶבֶד שֶׁנּוֹלַד מִשִּׁפְחָתוֹ – הֵם יֹאכְלוּ מִמַּאֲכָלָיו.
יב וּבַ֨ת־כֹּהֵ֔ן כִּ֥י תִהְיֶ֖ה לְאִ֣ישׁ זָ֑ר הִ֕וא בִּתְרוּמַ֥ת הַקֳּדָשִׁ֖ים לֹ֥א תֹאכֵֽל׃ וּבַת כֹּהֵן שֶׁתִּנָּשֵׂא לְאִישׁ שֶׁאֵינוֹ כֹּהֵן – הִיא לֹא תֹּאכַל מִתְּרוּמַת הַקֳּדָשִׁים.
יג וּבַת־כֹּהֵן֩ כִּ֨י תִהְיֶ֜ה אַלְמָנָ֣ה וּגְרוּשָׁ֗ה וְזֶ֘רַע֮ אֵ֣ין לָהּ֒ וְשָׁבָ֞ה אֶל־בֵּ֤ית אָבִ֙יהָ֙ כִּנְעוּרֶ֔יהָ מִלֶּ֥חֶם אָבִ֖יהָ תֹּאכֵ֑ל וְכׇל־זָ֖ר לֹא־יֹ֥אכַל בּֽוֹ׃ וּבַת כֹּהֵן שֶׁתִּהְיֶה אַלְמָנָה אוֹ גְּרוּשָׁה וְאֵין לָהּ צֶאֱצָאִים – הִיא תַּחְזוֹר אֶל בֵּית אָבִיהָ כִּימֵי נְעוּרֶיהָ וְתֹאכַל מֵהָאוֹכֶל שֶׁל אָבִיהָ, וְכָל מִי שֶׁאֵינוֹ כֹּהֵן לֹא יֹאכַל מִמֶּנּוּ.
יד וְאִ֕ישׁ כִּֽי־יֹאכַ֥ל קֹ֖דֶשׁ בִּשְׁגָגָ֑ה וְיָסַ֤ף חֲמִֽשִׁיתוֹ֙ עָלָ֔יו וְנָתַ֥ן לַכֹּהֵ֖ן אֶת־הַקֹּֽדֶשׁ׃ וְאִישׁ שֶׁיֹּאכַל קוֹדֶשׁ בִּשְׁגָגָה, הוּא יוֹסִיף עָלָיו חֲמִישִׁית וְיִתֵּן לַכֹּהֵן אֶת פִּדְיוֹן הַקּוֹדֶשׁ.
טו וְלֹ֣א יְחַלְּל֔וּ אֶת־קׇדְשֵׁ֖י בְּנֵ֣י יִשְׂרָאֵ֑ל אֵ֥ת אֲשֶׁר־יָרִ֖ימוּ לַיהֹוָֽה׃ וְלֹא יְחַלְּלוּ אֶת קָדְשֵׁי בְּנֵי יִשְׂרָאֵל – אֶת מַה שֶּׁיִּתְרְמוּ לַה' –
טז וְהִשִּׂ֤יאוּ אוֹתָם֙ עֲוֺ֣ן אַשְׁמָ֔ה בְּאׇכְלָ֖ם אֶת־קׇדְשֵׁיהֶ֑ם כִּ֛י אֲנִ֥י יְהֹוָ֖ה מְקַדְּשָֽׁם׃ וִיחַיְּבוּ אֶת עַצְמָם בְּאַשְׁמַת עֲווֹן אִם יֹאכְלוּ אֶת קָדְשֵׁיהֶם, כִּי אֲנִי ה' מְקַדֵּשׁ אוֹתָם.

שלישי[עריכה]

פ פָּרָשַׁת מוּמֵי הַקָּרְבָּנוֹת
יז וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יח דַּבֵּ֨ר אֶֽל־אַהֲרֹ֜ן וְאֶל־בָּנָ֗יו וְאֶל֙ כׇּל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָמַרְתָּ֖ אֲלֵהֶ֑ם אִ֣ישׁ אִישׁ֩ מִבֵּ֨ית יִשְׂרָאֵ֜ל וּמִן־הַגֵּ֣ר בְּיִשְׂרָאֵ֗ל אֲשֶׁ֨ר יַקְרִ֤יב קׇרְבָּנוֹ֙ לְכׇל־נִדְרֵיהֶם֙ וּלְכׇל־נִדְבוֹתָ֔ם אֲשֶׁר־יַקְרִ֥יבוּ לַיהֹוָ֖ה לְעֹלָֽה׃ דַּבֵּר אֶל אַהֲרֹן וְאֶל בָּנָיו וְאֶל כָּל בְּנֵי יִשְׂרָאֵל וֶאֱמוֹר לָהֶם: כָּל אִישׁ מֵעַם יִשְׂרָאֵל וּמֵהַגֵּר בְּיִשְׂרָאֵל שֶׁיַּקְרִיב אֶת קָרְבָּנוֹ – אֶת כָּל נִדְרֵיהֶם אוֹ כָּל נִדְבוֹתֵיהֶם שֶׁיַּקְרִיבוּ לַה' לְעוֹלָה.
יט לִֽרְצֹנְכֶ֑ם תָּמִ֣ים זָכָ֔ר בַּבָּקָ֕ר בַּכְּשָׂבִ֖ים וּבָֽעִזִּֽים׃ שֶׁה' יִהְיֶה מְרֻצֶּה מִמְּכֶם – תַּקְרִיבוּ אוֹתוֹ שָׁלֵם לְלֹא מוּם, זָכָר, מִבָּקָר מִכְּבָשִׂים וּמֵעִזִּים.
כ כֹּ֛ל אֲשֶׁר־בּ֥וֹ מ֖וּם לֹ֣א תַקְרִ֑יבוּ כִּי־לֹ֥א לְרָצ֖וֹן יִהְיֶ֥ה לָכֶֽם׃ אַל תַּקְרִיבוּ כָּל מַה שֶּׁיֵּשׁ בּוֹ מוּם, כִּי זֶה לֹא יְרַצֶּה אֶתְכֶם בְּעֵינֵי ה'.
כא וְאִ֗ישׁ כִּֽי־יַקְרִ֤יב זֶֽבַח־שְׁלָמִים֙ לַיהֹוָ֔ה לְפַלֵּא־נֶ֙דֶר֙ א֣וֹ לִנְדָבָ֔ה בַּבָּקָ֖ר א֣וֹ בַצֹּ֑אן תָּמִ֤ים יִֽהְיֶה֙ לְרָצ֔וֹן כׇּל־מ֖וּם לֹ֥א יִהְיֶה־בּֽוֹ׃ וְאִם אִישׁ יַקְרִיב קָרְבַּן שְׁלָמִים לַה' אֶת מַה שֶּׁהִפְרִישׁ עֲבוּר נֶדֶר אוֹ נְדָבָה מִבָּקָר אוֹ מִצֹּאן, הוּא יִהְיֶה שָׁלֵם לְרַצּוֹת אֶת ה' – לֹא יִהְיֶה בּוֹ שׁוּם מוּם.
כב עַוֶּ֩רֶת֩ א֨וֹ שָׁב֜וּר אוֹ־חָר֣וּץ אֽוֹ־יַבֶּ֗לֶת א֤וֹ גָרָב֙ א֣וֹ יַלֶּ֔פֶת לֹא־תַקְרִ֥יבוּ אֵ֖לֶּה לַיהֹוָ֑ה וְאִשֶּׁ֗ה לֹא־תִתְּנ֥וּ מֵהֶ֛ם עַל־הַמִּזְבֵּ֖חַ לַיהֹוָֽה׃ עִוֵּר אוֹ בַּעַל שֶׁבֶר אוֹ בַּעַל חָרִיץ אוֹ שׁוּמָה אוֹ שְׁחִין יָבֵשׁ אוֹ שְׁחִין לַח – אַל תַּקְרִיבוּ אֶת אֵלֶּה לַה', וְאַל תַּעֲשׂוּ מֵהֶם קָרְבָּן עוֹלֶה בָּאֵשׁ עַל הַמִּזְבֵּחַ לַה'.
כג וְשׁ֥וֹר וָשֶׂ֖ה שָׂר֣וּעַ וְקָל֑וּט נְדָבָה֙ תַּעֲשֶׂ֣ה אֹת֔וֹ וּלְנֵ֖דֶר לֹ֥א יֵרָצֶֽה׃ וְשׁוֹר וְשֶׂה בַּעַל אֵבֶר גָּדוֹל מִדַּי אוֹ חָסֵר – אוֹתוֹ תִּתֵּן נְדָבָה לַמִּקְדָּשׁ, אֲבָל הוּא לֹא יֵרָצֶה כְּתַשְׁלוּם לְנֵדֶר שֶׁל קָרְבָּן.
כד וּמָע֤וּךְ וְכָתוּת֙ וְנָת֣וּק וְכָר֔וּת לֹ֥א תַקְרִ֖יבוּ לַֽיהֹוָ֑ה וּֽבְאַרְצְכֶ֖ם לֹ֥א תַעֲשֽׂוּ׃ אַל תַּקְרִיבוּ לַה' בַּעַל אֵבֶר מָעוּךְ אוֹ מְרֻסָּק אוֹ מְנֻתָּק אוֹ כָּרוּת, וּבְכָל אַרְצוֹתֵיכֶם אַל תַּעֲשׂוּ כָּךְ לַחַיּוֹת.
כה וּמִיַּ֣ד בֶּן־נֵכָ֗ר לֹ֥א תַקְרִ֛יבוּ אֶת־לֶ֥חֶם אֱלֹהֵיכֶ֖ם מִכׇּל־אֵ֑לֶּה כִּ֣י מׇשְׁחָתָ֤ם בָּהֶם֙ מ֣וּם בָּ֔ם לֹ֥א יֵרָצ֖וּ לָכֶֽם׃ גַּם מִיַּד בֵּן שֶׁל עַם אַחֵר אַל תַּקְרִיבוּ מִכָּל אֵלֶּה אֶת קָרְבַּן אֱלֹהֵיכֶם, כִּי יֵשׁ בָּהֶם הַשְׁחָתָה – יֵשׁ בָּהֶם מוּם – הֵם לֹא יְרַצּוּ אֶתְכֶם בְּעֵינֵי ה'.
ס פָּרָשַׁת פְּסוּלֵי קָרְבָּנוֹת
כו וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
כז שׁ֣וֹר אוֹ־כֶ֤שֶׂב אוֹ־עֵז֙ כִּ֣י יִוָּלֵ֔ד וְהָיָ֛ה שִׁבְעַ֥ת יָמִ֖ים תַּ֣חַת אִמּ֑וֹ וּמִיּ֤וֹם הַשְּׁמִינִי֙ וָהָ֔לְאָה יֵרָצֶ֕ה לְקׇרְבַּ֥ן אִשֶּׁ֖ה לַיהֹוָֽה׃ אִם יִוָּלֵד שׁוֹר אוֹ כֶּבֶשׂ אוֹ עֵז הוּא יִהְיֶה שִׁבְעָה יָמִים עִם אִמּוֹ, וּמֵהַיּוֹם הַשְּׁמִינִי וָהָלְאָה יִהְיֶה רָצוּי לְהַקְרִיבוֹ לָאֵשׁ לַה'.
כח וְשׁ֖וֹר אוֹ־שֶׂ֑ה אֹת֣וֹ וְאֶת־בְּנ֔וֹ לֹ֥א תִשְׁחֲט֖וּ בְּי֥וֹם אֶחָֽד׃ וְשׁוֹר אוֹ שֶׂה – אֶת הָאֵם וְאֶת בְּנָהּ אַל תִּשְׁחֲטוּ בְּיוֹם אֶחָד.
כט וְכִֽי־תִזְבְּח֥וּ זֶֽבַח־תּוֹדָ֖ה לַיהֹוָ֑ה לִֽרְצֹנְכֶ֖ם תִּזְבָּֽחוּ׃ וְאִם תִּזְבְּחוּ זֶבַח תּוֹדָה לַה', תִּזְבְּחוּ אוֹתוֹ בְּאוֹפֶן שֶׁה' יִהְיֶה מְרוּצֶה מִמַּעֲשֵׂיכֶם.
ל בַּיּ֤וֹם הַהוּא֙ יֵאָכֵ֔ל לֹֽא־תוֹתִ֥ירוּ מִמֶּ֖נּוּ עַד־בֹּ֑קֶר אֲנִ֖י יְהֹוָֽה׃ בְּאוֹתוֹ הַיּוֹם יֵאָכֵל, אַל תַּשְׁאִירוּ מִמֶּנּוּ עַד הַבּוֹקֶר, אֲנִי ה'.
לא וּשְׁמַרְתֶּם֙ מִצְוֺתַ֔י וַעֲשִׂיתֶ֖ם אֹתָ֑ם אֲנִ֖י יְהֹוָֽה׃ שִׁמְרוּ אֶת מִצְווֹתַי וַעֲשׂוּ אוֹתָן, אֲנִי ה'.
לב וְלֹ֤א תְחַלְּלוּ֙ אֶת־שֵׁ֣ם קׇדְשִׁ֔י וְנִ֨קְדַּשְׁתִּ֔י בְּת֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל אֲנִ֥י יְהֹוָ֖ה מְקַדִּשְׁכֶֽם׃ אַל תְּחַלְּלוּ אֶת שֵׁם קָדְשִׁי, וְאֶתְקַדֵּשׁ בְּתוֹךְ בְּנֵי יִשְׂרָאֵל, אֲנִי ה' שֶׁמְּקַדֵּשׁ אֶתְכֶם,
לג הַמּוֹצִ֤יא אֶתְכֶם֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לִהְי֥וֹת לָכֶ֖ם לֵאלֹהִ֑ים אֲנִ֖י יְהֹוָֽה׃ שֶׁמּוֹצִיא אֶתְכֶם מֵאֶרֶץ מִצְרַיִם לִהְיוֹת לָכֶם לֵאלֹהִים, אֲנִי ה'.

רביעי[עריכה]

פ פָּרָשַׁת הַמּוֹעֲדִים
כג א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב דַּבֵּ֞ר אֶל־בְּנֵ֤י יִשְׂרָאֵל֙ וְאָמַרְתָּ֣ אֲלֵהֶ֔ם מוֹעֲדֵ֣י יְהֹוָ֔ה אֲשֶׁר־תִּקְרְא֥וּ אֹתָ֖ם מִקְרָאֵ֣י קֹ֑דֶשׁ אֵ֥לֶּה הֵ֖ם מוֹעֲדָֽי׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וֶאֱמוֹר לָהֶם: זְמַנֵּי חַגֵּי ה' שֶׁתַּכְרִיזוּ עֲלֵיהֶם כִּמְאוֹרְעוֹת קוֹדֶשׁ – אֵלֶּה הֵם זְמַנֵּי מוֹעֲדַי:
ג שֵׁ֣שֶׁת יָמִים֮ תֵּעָשֶׂ֣ה מְלָאכָה֒ וּבַיּ֣וֹם הַשְּׁבִיעִ֗י שַׁבַּ֤ת שַׁבָּתוֹן֙ מִקְרָא־קֹ֔דֶשׁ כׇּל־מְלָאכָ֖ה לֹ֣א תַעֲשׂ֑וּ שַׁבָּ֥ת הִוא֙ לַֽיהֹוָ֔ה בְּכֹ֖ל מוֹשְׁבֹֽתֵיכֶֽם׃ שִׁשָּׁה יָמִים תֵּעָשֶׂה מְלָאכָה, וּבַיּוֹם הַשְּׁבִיעִי שְׁבִיתַת מְנוּחָה – מְאוֹרָע קוֹדֶשׁ, אַל תַּעֲשׂוּ שׁוּם מְלָאכָה – הִיא שַׁבָּת לַה' בְּכָל מְקוֹמוֹת מְגוּרֵיכֶם.
ד אֵ֚לֶּה מוֹעֲדֵ֣י יְהֹוָ֔ה מִקְרָאֵ֖י קֹ֑דֶשׁ אֲשֶׁר־תִּקְרְא֥וּ אֹתָ֖ם בְּמוֹעֲדָֽם׃ אֵלֶּה חַגֵּי ה' – מְאוֹרְעוֹת קוֹדֶשׁ, שֶׁתִּקְבְּעוּ אוֹתָם בִּזְמַנָּם:
פ
ה בַּחֹ֣דֶשׁ הָרִאשׁ֗וֹן בְּאַרְבָּעָ֥ה עָשָׂ֛ר לַחֹ֖דֶשׁ בֵּ֣ין הָעַרְבָּ֑יִם פֶּ֖סַח לַיהֹוָֽה׃ בַּחוֹדֶשׁ הָרִאשׁוֹן בְּאַרְבָּעָה עָשָׂר לַחוֹדֶשׁ אַחַר הַצָּהֳרַיִם – תַּקְרִיבוּ קָרְבַּן פֶּסַח לַה'.
ו וּבַחֲמִשָּׁ֨ה עָשָׂ֥ר יוֹם֙ לַחֹ֣דֶשׁ הַזֶּ֔ה חַ֥ג הַמַּצּ֖וֹת לַיהֹוָ֑ה שִׁבְעַ֥ת יָמִ֖ים מַצּ֥וֹת תֹּאכֵֽלוּ׃ וּבַחֲמִשָּׁה עָשָׂר יוֹם לַחוֹדֶשׁ הַזֶּה חַג הַמַּצּוֹת לַה' – שִׁבְעָה יָמִים תֹּאכְלוּ מַצּוֹת.
ז בַּיּוֹם֙ הָֽרִאשׁ֔וֹן מִקְרָא־קֹ֖דֶשׁ יִהְיֶ֣ה לָכֶ֑ם כׇּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַעֲשֽׂוּ׃ בַּיּוֹם הָרִאשׁוֹן מְאוֹרָע קוֹדֶשׁ יִהְיֶה לָכֶם, אַל תַּעֲשׂוּ שׁוּם מְלֶאכֶת עֲבוֹדָה.
ח וְהִקְרַבְתֶּ֥ם אִשֶּׁ֛ה לַיהֹוָ֖ה שִׁבְעַ֣ת יָמִ֑ים בַּיּ֤וֹם הַשְּׁבִיעִי֙ מִקְרָא־קֹ֔דֶשׁ כׇּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַעֲשֽׂוּ׃ הַקְרִיבוּ קָרְבָּן עוֹלֶה בָּאֵשׁ לַה' שִׁבְעָה יָמִים, בַּיּוֹם הַשְּׁבִיעִי מְאוֹרָע קוֹדֶשׁ, אַל תַּעֲשׂוּ שׁוּם מְלֶאכֶת עֲבוֹדָה.

ט וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
י דַּבֵּ֞ר אֶל־בְּנֵ֤י יִשְׂרָאֵל֙ וְאָמַרְתָּ֣ אֲלֵהֶ֔ם כִּֽי־תָבֹ֣אוּ אֶל־הָאָ֗רֶץ אֲשֶׁ֤ר אֲנִי֙ נֹתֵ֣ן לָכֶ֔ם וּקְצַרְתֶּ֖ם אֶת־קְצִירָ֑הּ וַהֲבֵאתֶ֥ם אֶת־עֹ֛מֶר רֵאשִׁ֥ית קְצִירְכֶ֖ם אֶל־הַכֹּהֵֽן׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וֶאֱמוֹר לָהֶם: כְּשֶׁתָּבוֹאוּ לָאָרֶץ שֶׁאֲנִי נוֹתֵן לָכֶם וְתִקְצְרוּ אֶת קְצִירָהּ, הֲבִיאוּ אֶת הָעוֹמֶר הָרִאשׁוֹן שֶׁתִּקְצְרוּ לַכֹּהֵן.
יא וְהֵנִ֧יף אֶת־הָעֹ֛מֶר לִפְנֵ֥י יְהֹוָ֖ה לִֽרְצֹנְכֶ֑ם מִֽמׇּחֳרַת֙ הַשַּׁבָּ֔ת יְנִיפֶ֖נּוּ הַכֹּהֵֽן׃ הוּא יָנִיף אֶת הָעוֹמֶר לִפְנֵי ה' כְּדֵי שֶׁה' יִתְרַצֶּה לָכֶם, הַכֹּהֵן יָנִיף אוֹתוֹ לְמָחֳרַת יוֹם הַשְּׁבִיתָה.
יב וַעֲשִׂיתֶ֕ם בְּי֥וֹם הֲנִֽיפְכֶ֖ם אֶת־הָעֹ֑מֶר כֶּ֣בֶשׂ תָּמִ֧ים בֶּן־שְׁנָת֛וֹ לְעֹלָ֖ה לַיהֹוָֽה׃ וּבַיּוֹם שֶׁתָּנִיפוּ אֶת הָעוֹמֶר תַּקְרִיבוּ כֶּבֶשׂ לְעוֹלָה לַה' שָׁלֵם לְלֹא מוּם בִּשְׁנָתוֹ הָרִאשׁוֹנָה.
יג וּמִנְחָתוֹ֩ שְׁנֵ֨י עֶשְׂרֹנִ֜ים סֹ֣לֶת בְּלוּלָ֥ה בַשֶּׁ֛מֶן אִשֶּׁ֥ה לַיהֹוָ֖ה רֵ֣יחַ נִיחֹ֑חַ וְנִסְכֹּ֥ה יַ֖יִן רְבִיעִ֥ת הַהִֽין׃ וּמִנְחָתוֹ שְׁנֵי עֶשְׂרוֹנִים (כְּ⅓4 לִיטֶר) סוֹלֶת מְעֻרְבֶּבֶת בְּשֶׁמֶן קָרְבָּן עוֹלֶה בָּאֵשׁ לַה' לְנַחַת רוּחַ, וְלַהַגָּשָׁה אִתּוֹ רְבִיעִית הַהִין יַיִן (כִּ900 מ"ל).
יד וְלֶ֩חֶם֩ וְקָלִ֨י וְכַרְמֶ֜ל לֹ֣א תֹֽאכְל֗וּ עַד־עֶ֙צֶם֙ הַיּ֣וֹם הַזֶּ֔ה עַ֚ד הֲבִ֣יאֲכֶ֔ם אֶת־קׇרְבַּ֖ן אֱלֹהֵיכֶ֑ם חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם בְּכֹ֖ל מֹשְׁבֹֽתֵיכֶֽם׃ וְאַל תֹּאכְלוּ לֶחֶם וְחִטָּה קְלוּיָה אוֹ מְרֻסֶּקֶת מֵהַתְּבוּאָה הַחֲדָשָׁה, עַד עֶצֶם הַיּוֹם הַזֶּה – עַד שֶׁתָּבִיאוּ אֶת קָרְבַּן אֱלֹהֵיכֶם, חוֹק לְעוֹלָם לְדוֹרוֹת בְּכָל מְקוֹמוֹת מְגוּרֵיכֶם.

טו וּסְפַרְתֶּ֤ם לָכֶם֙ מִמׇּחֳרַ֣ת הַשַּׁבָּ֔ת מִיּוֹם֙ הֲבִ֣יאֲכֶ֔ם אֶת־עֹ֖מֶר הַתְּנוּפָ֑ה שֶׁ֥בַע שַׁבָּת֖וֹת תְּמִימֹ֥ת תִּהְיֶֽינָה׃ סִפְרוּ מִלְּמָחֳרַת יוֹם הַשְּׁבִיתָה – מֵהַיּוֹם שֶׁתָּבִיאוּ אֶת הָעוֹמֶר הַמּוּנָף, שִׁבְעָה שָׁבוּעוֹת שְׁלֵימִים.
טז עַ֣ד מִֽמׇּחֳרַ֤ת הַשַּׁבָּת֙ הַשְּׁבִיעִ֔ת תִּסְפְּר֖וּ חֲמִשִּׁ֣ים י֑וֹם וְהִקְרַבְתֶּ֛ם מִנְחָ֥ה חֲדָשָׁ֖ה לַיהֹוָֽה׃ תִּסְפְּרוּ חֲמִשִּׁים יוֹם עַד לְמָחֳרַת הַשָּׁבוּעַ הַשְּׁבִיעִי, וְתַקְרִיבוּ לַה' מִנְחָה מֵהַתְּבוּאָה הַחֲדָשָׁה.
יז מִמּוֹשְׁבֹ֨תֵיכֶ֜ם תָּבִ֣יאּוּ ׀ לֶ֣חֶם תְּנוּפָ֗ה שְׁ֚תַּיִם שְׁנֵ֣י עֶשְׂרֹנִ֔ים סֹ֣לֶת תִּהְיֶ֔ינָה חָמֵ֖ץ תֵּאָפֶ֑ינָה בִּכּוּרִ֖ים לַֽיהֹוָֽה׃ תָּבִיאוּ לֶחֶם לַהֲנָפָה מִמְּקוֹמוֹת מְגוּרֵיכֶם, שְׁתֵּי חַלּוֹת מִשְּׁתֵּי עֲשִׂירִיּוֹת הָאֵיפָה, הֵן תִּהְיֶינָה סוֹלֶת וְתֵאָפֶינָה חָמֵץ – בִּכּוּרִים לַה'.
יח וְהִקְרַבְתֶּ֣ם עַל־הַלֶּ֗חֶם שִׁבְעַ֨ת כְּבָשִׂ֤ים תְּמִימִם֙ בְּנֵ֣י שָׁנָ֔ה וּפַ֧ר בֶּן־בָּקָ֛ר אֶחָ֖ד וְאֵילִ֣ם שְׁנָ֑יִם יִהְי֤וּ עֹלָה֙ לַֽיהֹוָ֔ה וּמִנְחָתָם֙ וְנִסְכֵּיהֶ֔ם אִשֵּׁ֥ה רֵֽיחַ־נִיחֹ֖חַ לַיהֹוָֽה׃ וְתַקְרִיבוּ בְּנוֹסָף לַלֶּחֶם: שִׁבְעָה כְּבָשִׂים לְלֹא מוּם שֶׁגִּילָם עַד שָׁנָה וּפַר אֶחָד וּשְׁנֵי אֵילִים, אֵלֶּה יִהְיוּ עוֹלָה לַה', וּמִנְחוֹתֵיהֶם וּשְׁפִיכוֹת הַיַּיִן שֶׁאִתָּם, קָרְבָּן עוֹלֶה בָּאֵשׁ לְנַחַת רוּחַ לַה'.
יט וַעֲשִׂיתֶ֛ם שְׂעִיר־עִזִּ֥ים אֶחָ֖ד לְחַטָּ֑את וּשְׁנֵ֧י כְבָשִׂ֛ים בְּנֵ֥י שָׁנָ֖ה לְזֶ֥בַח שְׁלָמִֽים׃ וְתַעֲשׂוּ תַּיִשׁ אֶחָד לְחַטָּאת וּשְׁנֵי כְּבָשִׂים שֶׁגִּילָם עַד שָׁנָה לְזֶבַח שְׁלָמִים.
כ וְהֵנִ֣יף הַכֹּהֵ֣ן ׀ אֹתָ֡ם עַל֩ לֶ֨חֶם הַבִּכֻּרִ֤ים תְּנוּפָה֙ לִפְנֵ֣י יְהֹוָ֔ה עַל־שְׁנֵ֖י כְּבָשִׂ֑ים קֹ֛דֶשׁ יִהְי֥וּ לַיהֹוָ֖ה לַכֹּהֵֽן׃ הַכֹּהֵן יָנִיף אוֹתָם לִפְנֵי ה' – אֶת לֶחֶם הַבִּכּוּרִים הַמּוּנָף לִפְנֵי ה' עִם שְׁנֵי הַכְּבָשִׂים, הֵם יִהְיוּ קוֹדֶשׁ לַה' וְיִנָּתְנוּ לַכֹּהֵן.
כא וּקְרָאתֶ֞ם בְּעֶ֣צֶם ׀ הַיּ֣וֹם הַזֶּ֗ה מִֽקְרָא־קֹ֙דֶשׁ֙ יִהְיֶ֣ה לָכֶ֔ם כׇּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֣א תַעֲשׂ֑וּ חֻקַּ֥ת עוֹלָ֛ם בְּכׇל־מוֹשְׁבֹ֥תֵיכֶ֖ם לְדֹרֹֽתֵיכֶֽם׃ וְתִקְבְּעוּ בַּיּוֹם הַזֶּה עַצְמוֹ – מְאוֹרָע קוֹדֶשׁ יִהְיֶה לָכֶם, אַל תַּעֲשׂוּ שׁוּם מְלֶאכֶת עֲבוֹדָה, חוֹק לְעוֹלָם בְּכָל מְקוֹמוֹת מְגוּרֵיכֶם לְדוֹרוֹת.
כב וּֽבְקֻצְרְכֶ֞ם אֶת־קְצִ֣יר אַרְצְכֶ֗ם לֹֽא־תְכַלֶּ֞ה פְּאַ֤ת שָֽׂדְךָ֙ בְּקֻצְרֶ֔ךָ וְלֶ֥קֶט קְצִירְךָ֖ לֹ֣א תְלַקֵּ֑ט לֶֽעָנִ֤י וְלַגֵּר֙ תַּעֲזֹ֣ב אֹתָ֔ם אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ וּכְשֶׁתִּקְצְרוּ אֶת קְצִיר אַרְצְכֶם אַל תְּחַסֵּל בִּקְצִירָתְךָ אֶת פִּנַּת שָׂדְךָ, וְאַל תְּלַקֵּט אֶת הַשִּׁבּוֹלִים שֶׁנּוֹפְלוֹת – תַּשְׁאִיר אוֹתָן לֶעָנִי וְלַגֵּר, אֲנִי ה' אֱלֹהֵיכֶם.

כג וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
כד דַּבֵּ֛ר אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר בַּחֹ֨דֶשׁ הַשְּׁבִיעִ֜י בְּאֶחָ֣ד לַחֹ֗דֶשׁ יִהְיֶ֤ה לָכֶם֙ שַׁבָּת֔וֹן זִכְר֥וֹן תְּרוּעָ֖ה מִקְרָא־קֹֽדֶשׁ׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וֶאֱמוֹר: בַּחוֹדֶשׁ הַשְּׁבִיעִי בָּרִאשׁוֹן לַחוֹדֶשׁ יִהְיֶה לָכֶם מְנוּחָה, הַזְכָּרַת תְּרוּעָה, מְאוֹרָע קוֹדֶשׁ.

כו וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
כז אַ֡ךְ בֶּעָשׂ֣וֹר לַחֹ֩דֶשׁ֩ הַשְּׁבִיעִ֨י הַזֶּ֜ה י֧וֹם הַכִּפֻּרִ֣ים ה֗וּא מִֽקְרָא־קֹ֙דֶשׁ֙ יִהְיֶ֣ה לָכֶ֔ם וְעִנִּיתֶ֖ם אֶת־נַפְשֹׁתֵיכֶ֑ם וְהִקְרַבְתֶּ֥ם אִשֶּׁ֖ה לַיהֹוָֽה׃ אֲבָל בַּעֲשִׂירִי לַחוֹדֶשׁ הַשְּׁבִיעִי הַזֶּה, הוּא יוֹם הַכִּפּוּרִים, יִהְיֶה לָכֶם מְאוֹרָע קוֹדֶשׁ, תִּתְעַנּוּ בְּצוֹם, וְתַקְרִיבוּ קָרְבָּן עוֹלֶה בָּאֵשׁ לַה'.
כח וְכׇל־מְלָאכָה֙ לֹ֣א תַעֲשׂ֔וּ בְּעֶ֖צֶם הַיּ֣וֹם הַזֶּ֑ה כִּ֣י י֤וֹם כִּפֻּרִים֙ ה֔וּא לְכַפֵּ֣ר עֲלֵיכֶ֔ם לִפְנֵ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ וְאַל תַּעֲשׂוּ שׁוּם מְלָאכָה בַּיּוֹם הַזֶּה עַצְמוֹ, כִּי הוּא יוֹם כַּפָּרַת חֲטָאִים לְכַפֵּר עֲלֵיכֶם לִפְנֵי ה' אֱלֹהֵיכֶם.
כט כִּ֤י כׇל־הַנֶּ֙פֶשׁ֙ אֲשֶׁ֣ר לֹֽא־תְעֻנֶּ֔ה בְּעֶ֖צֶם הַיּ֣וֹם הַזֶּ֑ה וְנִכְרְתָ֖ה מֵֽעַמֶּֽיהָ׃ כִּי כָּל אָדָם שֶׁלֹּא יִתְעַנֶּה בְּצוֹם בַּיּוֹם הַזֶּה עַצְמוֹ – יֹאבַד מֵעַמּוֹ.
ל וְכׇל־הַנֶּ֗פֶשׁ אֲשֶׁ֤ר תַּעֲשֶׂה֙ כׇּל־מְלָאכָ֔ה בְּעֶ֖צֶם הַיּ֣וֹם הַזֶּ֑ה וְהַֽאֲבַדְתִּ֛י אֶת־הַנֶּ֥פֶשׁ הַהִ֖וא מִקֶּ֥רֶב עַמָּֽהּ׃ וְכָל אָדָם שֶׁיַּעֲשֶׂה מְלָאכָה כָּלְשֶׁהִי בַּיּוֹם הַזֶּה עַצְמוֹ, אַאֲבִיד אוֹתוֹ מִתּוֹךְ עַמּוֹ.
לא כׇּל־מְלָאכָ֖ה לֹ֣א תַעֲשׂ֑וּ חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם בְּכֹ֖ל מֹשְׁבֹֽתֵיכֶֽם׃ אַל תַּעֲשׂוּ שׁוּם מְלָאכָה, חוֹק לְעוֹלָם לְדוֹרוֹת בְּכָל מְקוֹמוֹת מְגוּרֵיכֶם.
לב שַׁבַּ֨ת שַׁבָּת֥וֹן הוּא֙ לָכֶ֔ם וְעִנִּיתֶ֖ם אֶת־נַפְשֹׁתֵיכֶ֑ם בְּתִשְׁעָ֤ה לַחֹ֙דֶשׁ֙ בָּעֶ֔רֶב מֵעֶ֣רֶב עַד־עֶ֔רֶב תִּשְׁבְּת֖וּ שַׁבַּתְּכֶֽם׃ הוּא יוֹם שְׁבִיתָה מֻחְלֶטֶת בִּשְׁבִילְכֶם וְתִתְעַנּוּ בְּצוֹם, בְּתִשְׁעָה לַחוֹדֶשׁ בָּעֶרֶב – מֵעֶרֶב עַד עֶרֶב תִּשְׁבְּתוּ בַּשְּׁבִיתָה שֶׁלָּכֶם.

ששי[עריכה]

פ
לג וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
לד דַּבֵּ֛ר אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר בַּחֲמִשָּׁ֨ה עָשָׂ֜ר י֗וֹם לַחֹ֤דֶשׁ הַשְּׁבִיעִי֙ הַזֶּ֔ה חַ֧ג הַסֻּכּ֛וֹת שִׁבְעַ֥ת יָמִ֖ים לַיהֹוָֽה׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְתֹאמַר: בַּחֲמִשָּׁה עָשָׂר יוֹם לַחוֹדֶשׁ הַשְּׁבִיעִי הַזֶּה, יִהְיֶה חַג הַסֻּכּוֹת שִׁבְעָה יָמִים לַה'.
לה בַּיּ֥וֹם הָרִאשׁ֖וֹן מִקְרָא־קֹ֑דֶשׁ כׇּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַעֲשֽׂוּ׃ בַּיּוֹם הָרִאשׁוֹן מְאוֹרָע קוֹדֶשׁ, אַל תַּעֲשׂוּ שׁוּם מְלֶאכֶת עֲבוֹדָה.
לו שִׁבְעַ֣ת יָמִ֔ים תַּקְרִ֥יבוּ אִשֶּׁ֖ה לַיהֹוָ֑ה בַּיּ֣וֹם הַשְּׁמִינִ֡י מִקְרָא־קֹ֩דֶשׁ֩ יִהְיֶ֨ה לָכֶ֜ם וְהִקְרַבְתֶּ֨ם אִשֶּׁ֤ה לַֽיהֹוָה֙ עֲצֶ֣רֶת הִ֔וא כׇּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַעֲשֽׂוּ׃ שִׁבְעָה יָמִים תַּקְרִיבוּ קָרְבָּן עוֹלֶה בָּאֵשׁ לַה', בַּיּוֹם הַשְּׁמִינִי יִהְיֶה לָכֶם מְאוֹרָע קוֹדֶשׁ וְתַקְרִיבוּ קָרְבָּן עוֹלֶה בָּאֵשׁ לַה', הוּא יוֹם הִתְכַּנְּסוּת, אַל תַּעֲשׂוּ שׁוּם מְלָאכָה שֶׁל עֲבוֹדָה.
לז אֵ֚לֶּה מוֹעֲדֵ֣י יְהֹוָ֔ה אֲשֶׁר־תִּקְרְא֥וּ אֹתָ֖ם מִקְרָאֵ֣י קֹ֑דֶשׁ לְהַקְרִ֨יב אִשֶּׁ֜ה לַיהֹוָ֗ה עֹלָ֧ה וּמִנְחָ֛ה זֶ֥בַח וּנְסָכִ֖ים דְּבַר־י֥וֹם בְּיוֹמֽוֹ׃ אֵלֶּה מוֹעֲדֵי ה' שֶׁתַּכְרִיזוּ עֲלֵיהֶם כִּמְאוֹרְעוֹת קוֹדֶשׁ, לְהַקְרִיב קָרְבָּן עוֹלֶה בָּאֵשׁ לַה' – עוֹלָה וּמִנְחָה, קָרְבַּן שְׁלָמִים וּשְׁפִיכוֹת יַיִן, הַקָּרְבָּנוֹת שֶׁל כָּל יוֹם בְּאוֹתוֹ יוֹם.
לח מִלְּבַ֖ד שַׁבְּתֹ֣ת יְהֹוָ֑ה וּמִלְּבַ֣ד מַתְּנֽוֹתֵיכֶ֗ם וּמִלְּבַ֤ד כׇּל־נִדְרֵיכֶם֙ וּמִלְּבַד֙ כׇּל־נִדְבֹ֣תֵיכֶ֔ם אֲשֶׁ֥ר תִּתְּנ֖וּ לַיהֹוָֽה׃ בְּנוֹסַף לְשַׁבְּתוֹת ה', וּבְנוֹסַף לְמַתְּנוֹתֵיכֶם וּבְנוֹסַף לְכָל נִדְרֵיכֶם וּבְנוֹסַף לְכָל נִדְבוֹתֵיכֶם שֶׁתִּתְּנוּ לַה'.
לט אַ֡ךְ בַּחֲמִשָּׁה֩ עָשָׂ֨ר י֜וֹם לַחֹ֣דֶשׁ הַשְּׁבִיעִ֗י בְּאׇסְפְּכֶם֙ אֶת־תְּבוּאַ֣ת הָאָ֔רֶץ תָּחֹ֥גּוּ אֶת־חַג־יְהֹוָ֖ה שִׁבְעַ֣ת יָמִ֑ים בַּיּ֤וֹם הָֽרִאשׁוֹן֙ שַׁבָּת֔וֹן וּבַיּ֥וֹם הַשְּׁמִינִ֖י שַׁבָּתֽוֹן׃ אֲבָל בַּחֲמִשָּׁה עָשָׂר יוֹם לַחוֹדֶשׁ הַשְּׁבִיעִי כְּשֶׁתֶּאֶסְפוּ אֶת תְּבוּאַת הָאָרֶץ תַּחְגְּגוּ אֶת חַג ה' שִׁבְעָה יָמִים, בַּיּוֹם הָרִאשׁוֹן מְנוּחָה וּבַיּוֹם הַשְּׁמִינִי מְנוּחָה.
מ וּלְקַחְתֶּ֨ם לָכֶ֜ם בַּיּ֣וֹם הָרִאשׁ֗וֹן פְּרִ֨י עֵ֤ץ הָדָר֙ כַּפֹּ֣ת תְּמָרִ֔ים וַעֲנַ֥ף עֵץ־עָבֹ֖ת וְעַרְבֵי־נָ֑חַל וּשְׂמַחְתֶּ֗ם לִפְנֵ֛י יְהֹוָ֥ה אֱלֹהֵיכֶ֖ם שִׁבְעַ֥ת יָמִֽים׃ וְתִקְחוּ לָכֶם בַּיּוֹם הָרִאשׁוֹן פְּרִי עֵץ אֶתְרוֹג, לוּלְבֵי עֵץ תְּמָרִים, וַעֲנַף עֵץ הֲדַס, וְעַרְבֵי נַחַל, וְתִשְׂמְחוּ לִפְנֵי ה' אֱלֹהֵיכֶם שִׁבְעָה יָמִים.
מא וְחַגֹּתֶ֤ם אֹתוֹ֙ חַ֣ג לַֽיהֹוָ֔ה שִׁבְעַ֥ת יָמִ֖ים בַּשָּׁנָ֑ה חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם בַּחֹ֥דֶשׁ הַשְּׁבִיעִ֖י תָּחֹ֥גּוּ אֹתֽוֹ׃ וְתַחְגְּגוּ אוֹתוֹ חַג לַה' שִׁבְעָה יָמִים בַּשָּׁנָה, חוֹק לְעוֹלָם לְדוֹרוֹת, תַּחְגְּגוּ אוֹתוֹ בַּחוֹדֶשׁ הַשְּׁבִיעִי.
מב בַּסֻּכֹּ֥ת תֵּשְׁב֖וּ שִׁבְעַ֣ת יָמִ֑ים כׇּל־הָֽאֶזְרָח֙ בְּיִשְׂרָאֵ֔ל יֵשְׁב֖וּ בַּסֻּכֹּֽת׃ שִׁבְעָה יָמִים תֵּשְׁבוּ בְּסֻכּוֹת, כָּל הָאֶזְרָחִים שֶׁבְּעַם יִשְׂרָאֵל יֵשְׁבוּ בְּסֻכּוֹת.
מג לְמַ֘עַן֮ יֵדְע֣וּ דֹרֹֽתֵיכֶם֒ כִּ֣י בַסֻּכּ֗וֹת הוֹשַׁ֙בְתִּי֙ אֶת־בְּנֵ֣י יִשְׂרָאֵ֔ל בְּהוֹצִיאִ֥י אוֹתָ֖ם מֵאֶ֣רֶץ מִצְרָ֑יִם אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ כְּדֵי שֶׁצֶּאֱצָאֵיכֶם יֵדְעוּ שֶׁהוֹשַׁבְתִּי אֶת בְּנֵי יִשְׂרָאֵל בְּסֻכּוֹת כְּשֶׁהוֹצֵאתִי אוֹתָם מֵאֶרֶץ מִצְרַיִם, אֲנִי ה' אֱלֹהֵיכֶם.
מד וַיְדַבֵּ֣ר מֹשֶׁ֔ה אֶת־מֹעֲדֵ֖י יְהֹוָ֑ה אֶל־בְּנֵ֖י יִשְׂרָאֵֽל׃ מֹשֶׁה דִּבֵּר אֶת מֹעֲדֵי ה' אֶל בְּנֵי יִשְׂרָאֵל.

שביעי[עריכה]

פ פָּרָשַׁת הַשֶּׁמֶן לַמָּאוֹר
כד א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב צַ֞ו אֶת־בְּנֵ֣י יִשְׂרָאֵ֗ל וְיִקְח֨וּ אֵלֶ֜יךָ שֶׁ֣מֶן זַ֥יִת זָ֛ךְ כָּתִ֖ית לַמָּא֑וֹר לְהַעֲלֹ֥ת נֵ֖ר תָּמִֽיד׃ צַוֵּה אֶת בְּנֵי יִשְׂרָאֵל וְיָבִיאוּ לְךָ שֶׁמֶן זַיִת צָלוּל, כָּתוּשׁ, בִּשְׁבִיל לְהָאִיר, לְהַדְלִיק נֵר תָּמִיד.
ג מִחוּץ֩ לְפָרֹ֨כֶת הָעֵדֻ֜ת בְּאֹ֣הֶל מוֹעֵ֗ד יַעֲרֹךְ֩ אֹת֨וֹ אַהֲרֹ֜ן מֵעֶ֧רֶב עַד־בֹּ֛קֶר לִפְנֵ֥י יְהֹוָ֖ה תָּמִ֑יד חֻקַּ֥ת עוֹלָ֖ם לְדֹרֹֽתֵיכֶֽם׃ אַהֲרֹן יְסַדֵּר אוֹתוֹ מִחוּץ לַפָּרוֹכֶת שֶׁלִּפְנֵי אֲרוֹן לוּחוֹת הַבְּרִית בְּאוֹהֶל הַהִתְכַּנְּסוּת, שֶׁיִּדְלַק לִפְנֵי ה' תָּמִיד מֵעֶרֶב עַד בּוֹקֶר, חוֹק לְעוֹלָם לְדוֹרוֹת.
ד עַ֚ל הַמְּנֹרָ֣ה הַטְּהֹרָ֔ה יַעֲרֹ֖ךְ אֶת־הַנֵּר֑וֹת לִפְנֵ֥י יְהֹוָ֖ה תָּמִֽיד׃ הוּא יְסַדֵּר אֶת הַנֵּרוֹת עַל הַמְּנוֹרָה הַטְּהוֹרָה, לִפְנֵי ה' תָּמִיד.
פ פָּרָשַׁת לֶחֶם הַפָּנִים
ה וְלָקַחְתָּ֣ סֹ֔לֶת וְאָפִיתָ֣ אֹתָ֔הּ שְׁתֵּ֥ים עֶשְׂרֵ֖ה חַלּ֑וֹת שְׁנֵי֙ עֶשְׂרֹנִ֔ים יִהְיֶ֖ה הַֽחַלָּ֥ה הָאֶחָֽת׃ קַח סוֹלֶת וְתֹאפֶה אוֹתָהּ שְׁתֵּים עֶשְׂרֵה חַלּוֹת, כָּל חַלָּה תִּהְיֶה מִשְׁתֵּי עֲשִׂירִיּוֹת הָאֵיפָה (כּ5 לִיטֶר).
ו וְשַׂמְתָּ֥ אוֹתָ֛ם שְׁתַּ֥יִם מַֽעֲרָכ֖וֹת שֵׁ֣שׁ הַֽמַּעֲרָ֑כֶת עַ֛ל הַשֻּׁלְחָ֥ן הַטָּהֹ֖ר לִפְנֵ֥י יְהֹוָֽה׃ וְתָשִׂים אוֹתָם בִּשְׁתֵּי עֲרֵמוֹת מְסֻדָּרוֹת – שֵׁשׁ חַלּוֹת כָּל עֲרֵמָה, עַל הַשֻּׁלְחָן הַטָּהוֹר לִפְנֵי ה'.
ז וְנָתַתָּ֥ עַל־הַֽמַּעֲרֶ֖כֶת לְבֹנָ֣ה זַכָּ֑ה וְהָיְתָ֤ה לַלֶּ֙חֶם֙ לְאַזְכָּרָ֔ה אִשֶּׁ֖ה לַֽיהֹוָֽה׃ וְתָנִיחַ עַל הָעֲרֵמָה בּוֹשֶׂם לְבוֹנָה נְקִיָּה שֶׁתֻּקְטַר בָּאֵשׁ לְהַזְכִּיר אֶת הַלֶּחֶם לַה'.
ח בְּי֨וֹם הַשַּׁבָּ֜ת בְּי֣וֹם הַשַּׁבָּ֗ת יַֽעַרְכֶ֛נּוּ לִפְנֵ֥י יְהֹוָ֖ה תָּמִ֑יד מֵאֵ֥ת בְּנֵֽי־יִשְׂרָאֵ֖ל בְּרִ֥ית עוֹלָֽם׃ בְּכָל יוֹם שַׁבָּת יַעֲרוֹךְ אֶת הַשֻּׁלְחָן לִפְנֵי ה', תָּמִיד, הִתְחַיְּבוּת לְעוֹלָם מֵאֵת בְּנֵי יִשְׂרָאֵל.
ט וְהָֽיְתָה֙ לְאַהֲרֹ֣ן וּלְבָנָ֔יו וַאֲכָלֻ֖הוּ בְּמָק֣וֹם קָדֹ֑שׁ כִּ֡י קֹ֩דֶשׁ֩ קׇֽדָשִׁ֨ים ה֥וּא ל֛וֹ מֵאִשֵּׁ֥י יְהֹוָ֖ה חׇק־עוֹלָֽם׃ הַלֶּחֶם יִהְיֶה לְאַהֲרֹן וּלְבָנָיו, וְיֹאכְלוּ אוֹתוֹ בְּמָקוֹם קָדוֹשׁ, כִּי הוּא עֲבוּרוֹ קוֹדֶשׁ קָדָשִׁים מִקָּרְבְּנוֹת ה' – חוֹק לְעוֹלָם.
ס פָּרָשַׁת הַמְקַלֵּל
י וַיֵּצֵא֙ בֶּן־אִשָּׁ֣ה יִשְׂרְאֵלִ֔ית וְהוּא֙ בֶּן־אִ֣ישׁ מִצְרִ֔י בְּת֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל וַיִּנָּצוּ֙ בַּֽמַּחֲנֶ֔ה בֶּ֚ן הַיִּשְׂרְאֵלִ֔ית וְאִ֖ישׁ הַיִּשְׂרְאֵלִֽי׃ בֵּן שֶׁל אִשָּׁה יִשְׂרְאֵלִית וְאִישׁ מִצְרִי יָצָא וְהִסְתּוֹבֵב בֵּין בְּנֵי יִשְׂרָאֵל, הַבֵּן שֶׁל הַיִּשְׂרְאֵלִית וְאִישׁ יִשְׂרְאֵלִי רָבוּ בַּמַּחֲנֶה.
יא וַ֠יִּקֹּ֠ב בֶּן־הָֽאִשָּׁ֨ה הַיִּשְׂרְאֵלִ֤ית אֶת־הַשֵּׁם֙ וַיְקַלֵּ֔ל וַיָּבִ֥יאוּ אֹת֖וֹ אֶל־מֹשֶׁ֑ה וְשֵׁ֥ם אִמּ֛וֹ שְׁלֹמִ֥ית בַּת־דִּבְרִ֖י לְמַטֵּה־דָֽן׃ בֶּן הָאִשָּׁה הַיִּשְׂרְאֵלִית הִזְכִּיר אֶת שֵׁם הַשֵּׁם בִּמְפוֹרָש וְקִלֵּל אוֹתוֹ, וְהֵבִיאוּ אוֹתוֹ לְמֹשֶׁה. וְשֵׁם אִמּוֹ הָיָה שְׁלוֹמִית בַּת דִּבְרִי מִשֵּׁבֶט דָּן.
יב וַיַּנִּיחֻ֖הוּ בַּמִּשְׁמָ֑ר לִפְרֹ֥שׁ לָהֶ֖ם עַל־פִּ֥י יְהֹוָֽה׃ הִנִּיחוּ אוֹתוֹ בַּמַּעֲצָר בְּהַמְתָּנָה עַד שֶׁיְּפוֹרַשׁ לָהֶם דִּינוֹ עַל פִּי ה'.
פ
יג וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יד הוֹצֵ֣א אֶת־הַֽמְקַלֵּ֗ל אֶל־מִחוּץ֙ לַֽמַּחֲנֶ֔ה וְסָמְכ֧וּ כׇֽל־הַשֹּׁמְעִ֛ים אֶת־יְדֵיהֶ֖ם עַל־רֹאשׁ֑וֹ וְרָגְמ֥וּ אֹת֖וֹ כׇּל־הָעֵדָֽה׃ הוֹצֵא אֶת הַמְקַלֵּל אֶל מִחוּץ לַמַּחֲנֶה וְכָל הָעֵדִים שֶׁשָּׁמְעוּ יַשְׁעִינוּ אֶת יְדֵיהֶם עַל רֹאשׁוֹ וְכָל הַצִּבּוּר יִזְרְקוּ עָלָיו אֲבָנִים.
טו וְאֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל תְּדַבֵּ֣ר לֵאמֹ֑ר אִ֥ישׁ אִ֛ישׁ כִּֽי־יְקַלֵּ֥ל אֱלֹהָ֖יו וְנָשָׂ֥א חֶטְאֽוֹ׃ וְאֶל בְּנֵי יִשְׂרָאֵל תְּדַבֵּר וְתֹאמַר: כָּל אִישׁ שֶׁיְּקַלֵּל אֶת אֱלֹהָיו יְקַבֵּל אֶת הָעוֹנֶשׁ עַל חֶטְאוֹ.
טז וְנֹקֵ֤ב שֵׁם־יְהֹוָה֙ מ֣וֹת יוּמָ֔ת רָג֥וֹם יִרְגְּמוּ־ב֖וֹ כׇּל־הָעֵדָ֑ה כַּגֵּר֙ כָּֽאֶזְרָ֔ח בְּנׇקְבוֹ־שֵׁ֖ם יוּמָֽת׃ וּמִי שֶׁיַּזְכִּיר אֶת שֵׁם ה' וִיקַלֵּל אוֹתוֹ יוּמַת – כָּל הַצִּבּוּר יִזְרְקוּ עָלָיו אֲבָנִים, דִּין הַגֵּר כְּדִין הָאֶזְרָח – אִם יַזְכִּיר אֶת הַשֵּׁם וִיקַלֵּל אוֹתוֹ – יוּמַת.
יז וְאִ֕ישׁ כִּ֥י יַכֶּ֖ה כׇּל־נֶ֣פֶשׁ אָדָ֑ם מ֖וֹת יוּמָֽת׃ וְאִישׁ שֶׁיַּכֶּה אָדָם כָּלְשֶׁהוּ לַמָּוֶת – יוּמַת.
יח וּמַכֵּ֥ה נֶֽפֶשׁ־בְּהֵמָ֖ה יְשַׁלְּמֶ֑נָּה נֶ֖פֶשׁ תַּ֥חַת נָֽפֶשׁ׃ וְהַמַּכֵּה בְּהֵמָה וּמֵמִית אוֹתָהּ יְשַׁלֵּם אֶת עֶרְכָּהּ – עֵרֶךְ שֶׁל נֶפֶשׁ עֲבוּר נֶפֶשׁ.
יט וְאִ֕ישׁ כִּֽי־יִתֵּ֥ן מ֖וּם בַּעֲמִית֑וֹ כַּאֲשֶׁ֣ר עָשָׂ֔ה כֵּ֖ן יֵעָ֥שֶׂה לּֽוֹ׃ וְאִישׁ שֶׁיַּעֲשֶׂה מוּם בַּחֲבֵירוֹ, כְּמוֹ שֶׁעָשָׂה כָּךְ יֵעָשֶׂה לוֹ:
כ שֶׁ֚בֶר תַּ֣חַת שֶׁ֔בֶר עַ֚יִן תַּ֣חַת עַ֔יִן שֵׁ֖ן תַּ֣חַת שֵׁ֑ן כַּאֲשֶׁ֨ר יִתֵּ֥ן מוּם֙ בָּֽאָדָ֔ם כֵּ֖ן יִנָּ֥תֶן בּֽוֹ׃ פִּיצוּיֵי שֶׁבֶר עֲבוּר שֶׁבֶר, עֵרֶךְ שֶׁל עַיִן עַל הַשְׁחָתַת עַיִן, עֵרֶךְ שֶׁל שֵׁן עַל הַשְׁחָתַת שֵׁן – אוֹתוֹ נֶזֶק שֶׁעָשָׂה לְאָדָם כָּךְ יַעֲשׂוּ לוֹ.

מפטיר[עריכה]

כא וּמַכֵּ֥ה בְהֵמָ֖ה יְשַׁלְּמֶ֑נָּה וּמַכֵּ֥ה אָדָ֖ם יוּמָֽת׃ וּמַכֵּה בְּהֵמָה יְשַׁלֵּם אֶת עֶרְכָּהּ, וּמַכֵּה אָדָם בְּמֵזִיד יוּמַת.
כב מִשְׁפַּ֤ט אֶחָד֙ יִהְיֶ֣ה לָכֶ֔ם כַּגֵּ֥ר כָּאֶזְרָ֖ח יִהְיֶ֑ה כִּ֛י אֲנִ֥י יְהֹוָ֖ה אֱלֹהֵיכֶֽם׃ מִשְׁפַּט אֶחָד יִהְיֶה לָכֶם – דִּין הַגֵּר יִהְיֶה כְּדִין הָאֶזְרָח, כִּי אֲנִי ה' אֱלֹהֵיכֶם.
כג וַיְדַבֵּ֣ר מֹשֶׁה֮ אֶל־בְּנֵ֣י יִשְׂרָאֵל֒ וַיּוֹצִ֣יאוּ אֶת־הַֽמְקַלֵּ֗ל אֶל־מִחוּץ֙ לַֽמַּחֲנֶ֔ה וַיִּרְגְּמ֥וּ אֹת֖וֹ אָ֑בֶן וּבְנֵֽי־יִשְׂרָאֵ֣ל עָשׂ֔וּ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ מֹשֶׁה דִּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְהֵם הוֹצִיאוּ אֶת הַמְקַלֵּל אֶל מִחוּץ לַמַּחֲנֶה וְרָגְמוּ אוֹתוֹ בָּאֲבָנִים, בְּנֵי יִשְׂרָאֵל עָשׂוּ כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.
