ראשון[עריכה]

פ פָּרָשַׁת בִּיאַת יִתְרוֹ
יח א וַיִּשְׁמַ֞ע יִתְר֨וֹ כֹהֵ֤ן מִדְיָן֙ חֹתֵ֣ן מֹשֶׁ֔ה אֵת֩ כׇּל־אֲשֶׁ֨ר עָשָׂ֤ה אֱלֹהִים֙ לְמֹשֶׁ֔ה וּלְיִשְׂרָאֵ֖ל עַמּ֑וֹ כִּֽי־הוֹצִ֧יא יְהֹוָ֛ה אֶת־יִשְׂרָאֵ֖ל מִמִּצְרָֽיִם׃ יִתְרוֹ הַכּוֹמֶר שֶׁל מִדְיָן חָמִיו שֶׁל מֹשֶׁה שָׁמַע אֶת כָּל מַה שֶּׁאֱלֹהִים עָשָׂה לְמֹשֶׁה וּלְיִשְׂרָאֵל עַמּוֹ כְּשֶׁה' הוֹצִיא אֶת יִשְׂרָאֵל מִמִּצְרַיִם.
ב וַיִּקַּ֗ח יִתְרוֹ֙ חֹתֵ֣ן מֹשֶׁ֔ה אֶת־צִפֹּרָ֖ה אֵ֣שֶׁת מֹשֶׁ֑ה אַחַ֖ר שִׁלּוּחֶֽיהָ׃ לָקַח יִתְרוֹ חָמִיו שֶׁל מֹשֶׁה אֶת צִפּוֹרָה אִשְׁתּוֹ שֶׁל מֹשֶׁה, אַחֲרֵי שֶׁמֹּשֶׁה שָׁלַח אוֹתָהּ לְמִדְיָן.
ג וְאֵ֖ת שְׁנֵ֣י בָנֶ֑יהָ אֲשֶׁ֨ר שֵׁ֤ם הָֽאֶחָד֙ גֵּֽרְשֹׁ֔ם כִּ֣י אָמַ֔ר גֵּ֣ר הָיִ֔יתִי בְּאֶ֖רֶץ נׇכְרִיָּֽה׃ וְאֶת שְׁנֵי בָּנֶיהָ שֶׁשֵּׁם אֶחָד מֵהֶם גֵּרְשׁוֹם, כִּי אָמַר מֹשֶׁה גֵּר הָיִיתִי בְּאֶרֶץ נָכְרִיָּה.
ד וְשֵׁ֥ם הָאֶחָ֖ד אֱלִיעֶ֑זֶר כִּֽי־אֱלֹהֵ֤י אָבִי֙ בְּעֶזְרִ֔י וַיַּצִּלֵ֖נִי מֵחֶ֥רֶב פַּרְעֹֽה׃ וְשֵׁם הַשֵּׁנִי אֱלִיעֶזֶר, כִּי אָמַר מֹשֶׁה: אֱלֹהֵי אָבִי עָזַר לִי וְהִצִּיל אוֹתִי מֵהַחֶרֶב שֶׁל פַּרְעֹה.
ה וַיָּבֹ֞א יִתְר֨וֹ חֹתֵ֥ן מֹשֶׁ֛ה וּבָנָ֥יו וְאִשְׁתּ֖וֹ אֶל־מֹשֶׁ֑ה אֶל־הַמִּדְבָּ֗ר אֲשֶׁר־ה֛וּא חֹנֶ֥ה שָׁ֖ם הַ֥ר הָאֱלֹהִֽים׃ יִתְרוֹ חָמִיו שֶׁל מֹשֶׁה וּבְנֵי מֹשֶׁה וְאִשְׁתּוֹ בָּאוּ אֶל מֹשֶׁה לַמִּדְבָּר שֶׁהוּא חוֹנֶה שָׁם – לְיַד הַר הָאֱלֹהִים.
ו וַיֹּ֙אמֶר֙ אֶל־מֹשֶׁ֔ה אֲנִ֛י חֹתֶנְךָ֥ יִתְר֖וֹ בָּ֣א אֵלֶ֑יךָ וְאִ֨שְׁתְּךָ֔ וּשְׁנֵ֥י בָנֶ֖יהָ עִמָּֽהּ׃ הוּא אָמַר לְמֹשֶׁה: אֲנִי חָמִיךָ יִתְרוֹ בָּא אֵלֶיךָ, וְגַם אִשְׁתְּךָ בָּאָה וּשְׁנֵי בָּנֶיהָ אִתָּהּ.
ז וַיֵּצֵ֨א מֹשֶׁ֜ה לִקְרַ֣את חֹֽתְנ֗וֹ וַיִּשְׁתַּ֙חוּ֙ וַיִּשַּׁק־ל֔וֹ וַיִּשְׁאֲל֥וּ אִישׁ־לְרֵעֵ֖הוּ לְשָׁל֑וֹם וַיָּבֹ֖אוּ הָאֹֽהֱלָה׃ מֹשֶׁה יָצָא לִקְרַאת חָמִיו, הִשְׁתַּחֲוָה לוֹ וְנִשֵּׁק אוֹתוֹ, הֵם שָׁאֲלוּ זֶה אֶת זֶה לִשְׁלוֹמָם וּבָאוּ לָאוֹהֶל.
ח וַיְסַפֵּ֤ר מֹשֶׁה֙ לְחֹ֣תְנ֔וֹ אֵת֩ כׇּל־אֲשֶׁ֨ר עָשָׂ֤ה יְהֹוָה֙ לְפַרְעֹ֣ה וּלְמִצְרַ֔יִם עַ֖ל אוֹדֹ֣ת יִשְׂרָאֵ֑ל אֵ֤ת כׇּל־הַתְּלָאָה֙ אֲשֶׁ֣ר מְצָאָ֣תַם בַּדֶּ֔רֶךְ וַיַּצִּלֵ֖ם יְהֹוָֽה׃ מֹשֶׁה סִפֵּר לְחָמִיו אֶת כָּל מַה שֶּׁה' עָשָׂה לְפַרְעֹה וּלְמִצְרַיִם בְּעִנְיַן יִשְׂרָאֵל, וְאֶת כָּל הַצָּרוֹת שֶׁקָּרוּ לָהֶם בַּדֶּרֶךְ וַה' הִצִּיל אוֹתָם.
ט וַיִּ֣חַדְּ יִתְר֔וֹ עַ֚ל כׇּל־הַטּוֹבָ֔ה אֲשֶׁר־עָשָׂ֥ה יְהֹוָ֖ה לְיִשְׂרָאֵ֑ל אֲשֶׁ֥ר הִצִּיל֖וֹ מִיַּ֥ד מִצְרָֽיִם׃ יִתְרוֹ שָׂמַח עַל כָּל הַטּוֹבָה שֶׁה' עָשָׂה לְיִשְׂרָאֵל – שֶׁהִצִּיל אוֹתָם מִיַּד מִצְרַיִם.
י וַיֹּ֘אמֶר֮ יִתְרוֹ֒ בָּר֣וּךְ יְהֹוָ֔ה אֲשֶׁ֨ר הִצִּ֥יל אֶתְכֶ֛ם מִיַּ֥ד מִצְרַ֖יִם וּמִיַּ֣ד פַּרְעֹ֑ה אֲשֶׁ֤ר הִצִּיל֙ אֶת־הָעָ֔ם מִתַּ֖חַת יַד־מִצְרָֽיִם׃ יִתְרוֹ אָמַר: בָּרוּךְ ה' שֶׁהִצִּיל אֶתְכֶם מִמִּצְרַיִם וּמִפַּרְעֹה – שֶׁהִצִּיל אֶת הָעָם מֵהַשִּׁעְבּוּד לְמִצְרַיִם!
יא עַתָּ֣ה יָדַ֔עְתִּי כִּֽי־גָד֥וֹל יְהֹוָ֖ה מִכׇּל־הָאֱלֹהִ֑ים כִּ֣י בַדָּבָ֔ר אֲשֶׁ֥ר זָד֖וּ עֲלֵיהֶֽם׃ עַכְשָׁיו אֲנִי יוֹדֵעַ שֶׁה' גָּדוֹל יוֹתֵר מִכָּל הַמְּכוּנִים "אֱלֹהִים", שֶׁהֲרֵי מִצְרַיִם נֶעֶנְשׁוּ בְּמַיִם – בְּאוֹתוֹ הַדָּבָר שֶׁעָשׂוּ לְיִשְׂרָאֵל בְּרִשְׁעוּת.
יב וַיִּקַּ֞ח יִתְר֨וֹ חֹתֵ֥ן מֹשֶׁ֛ה עֹלָ֥ה וּזְבָחִ֖ים לֵֽאלֹהִ֑ים וַיָּבֹ֨א אַהֲרֹ֜ן וְכֹ֣ל ׀ זִקְנֵ֣י יִשְׂרָאֵ֗ל לֶאֱכׇל־לֶ֛חֶם עִם־חֹתֵ֥ן מֹשֶׁ֖ה לִפְנֵ֥י הָאֱלֹהִֽים׃ יִתְרוֹ חָמִיו שֶׁל מֹשֶׁה לָקַח עוֹלָה וּזְבָחִים וְהִקְרִיב לֵאלֹהִים, אַהֲרֹן וְכָל זִקְנֵי יִשְׂרָאֵל בָּאוּ עִם חָמִיו שֶׁל מֹשֶׁה לִסְעוּדַת הוֹדָיָה לִפְנֵי הָאֱלֹהִים.

שני[עריכה]

יג וַיְהִי֙ מִֽמׇּחֳרָ֔ת וַיֵּ֥שֶׁב מֹשֶׁ֖ה לִשְׁפֹּ֣ט אֶת־הָעָ֑ם וַיַּעֲמֹ֤ד הָעָם֙ עַל־מֹשֶׁ֔ה מִן־הַבֹּ֖קֶר עַד־הָעָֽרֶב׃ לְמָחֳרָת, מֹשֶׁה יָשַׁב לִשְׁפּוֹט אֶת הָעָם, הָעָם עָמַד בַּתּוֹר לְמֹשֶׁה מֵהַבּוֹקֶר עַד הָעֶרֶב.
יד וַיַּרְא֙ חֹתֵ֣ן מֹשֶׁ֔ה אֵ֛ת כׇּל־אֲשֶׁר־ה֥וּא עֹשֶׂ֖ה לָעָ֑ם וַיֹּ֗אמֶר מָֽה־הַדָּבָ֤ר הַזֶּה֙ אֲשֶׁ֨ר אַתָּ֤ה עֹשֶׂה֙ לָעָ֔ם מַדּ֗וּעַ אַתָּ֤ה יוֹשֵׁב֙ לְבַדֶּ֔ךָ וְכׇל־הָעָ֛ם נִצָּ֥ב עָלֶ֖יךָ מִן־בֹּ֥קֶר עַד־עָֽרֶב׃ חָמִיו שֶׁל מֹשֶׁה רָאָה אֶת כָּל מַה שֶּׁהוּא עוֹשֶׂה עִם הָעָם וְאָמַר: מָה הַדָּבָר הַזֶּה שֶׁאַתָּה עוֹשֶׂה לָעָם? מַדּוּעַ אַתָּה יוֹשֵׁב לְבַדְּךָ וְכָל הָעָם עוֹמְדִים לְפָנֶיךָ מִבּוֹקֶר עַד עֶרֶב?!
טו וַיֹּ֥אמֶר מֹשֶׁ֖ה לְחֹתְנ֑וֹ כִּֽי־יָבֹ֥א אֵלַ֛י הָעָ֖ם לִדְרֹ֥שׁ אֱלֹהִֽים׃ אָמַר מֹשֶׁה לְחָמִיו: כִּי הָעָם בָּא אֵלַי לְבַקֵּשׁ דִּבְרֵי אֱלֹהִים.
טז כִּֽי־יִהְיֶ֨ה לָהֶ֤ם דָּבָר֙ בָּ֣א אֵלַ֔י וְשָׁ֣פַטְתִּ֔י בֵּ֥ין אִ֖ישׁ וּבֵ֣ין רֵעֵ֑הוּ וְהוֹדַעְתִּ֛י אֶת־חֻקֵּ֥י הָאֱלֹהִ֖ים וְאֶת־תּוֹרֹתָֽיו׃ כְּשֶׁיֵּשׁ לָהֶם דָּבָר לְבָרֵר הֵם בָּאִים אֵלַי, וַאֲנִי שׁוֹפֵט בֵּין אִישׁ וּבֵין חֲבֵרוֹ, וַאֲנִי מוֹדִיעַ אֶת חֻקֵּי הָאֱלֹהִים וְאֶת מִצְווֹתָיו.
יז וַיֹּ֛אמֶר חֹתֵ֥ן מֹשֶׁ֖ה אֵלָ֑יו לֹא־טוֹב֙ הַדָּבָ֔ר אֲשֶׁ֥ר אַתָּ֖ה עֹשֶֽׂה׃ חָמִיו שֶׁל מֹשֶׁה אָמַר לוֹ: לֹא טוֹב הַדָּבָר שֶׁאַתָּה עוֹשֶׂה.
יח נָבֹ֣ל תִּבֹּ֔ל גַּם־אַתָּ֕ה גַּם־הָעָ֥ם הַזֶּ֖ה אֲשֶׁ֣ר עִמָּ֑ךְ כִּֽי־כָבֵ֤ד מִמְּךָ֙ הַדָּבָ֔ר לֹא־תוּכַ֥ל עֲשֹׂ֖הוּ לְבַדֶּֽךָ׃ תִּתְעָיֵף וְתִשָּׁחֵק גַּם אַתָּה וְגַם הָעָם הַזֶּה שֶׁאִתְּךָ, כִּי הַדָּבָר הַזֶּה קָשֶׁה מִדַּי עֲבוּרְךָ – לֹא תּוּכַל לַעֲשׂוֹת אוֹתוֹ לְבַדְּךָ!
יט עַתָּ֞ה שְׁמַ֤ע בְּקֹלִי֙ אִיעָ֣צְךָ֔ וִיהִ֥י אֱלֹהִ֖ים עִמָּ֑ךְ הֱיֵ֧ה אַתָּ֣ה לָעָ֗ם מ֚וּל הָֽאֱלֹהִ֔ים וְהֵבֵאתָ֥ אַתָּ֛ה אֶת־הַדְּבָרִ֖ים אֶל־הָאֱלֹהִֽים׃ עַכְשָׁיו שְׁמַע בְּקוֹלִי וַאֲיָעֵץ לְךָ וֶאֱלֹהִים יַעֲזוֹר לְךָ, אַתָּה תִּהְיֶה שָׁלִיחַ שֶׁל הָעָם לִפְנֵי הָאֱלֹהִים, וְתָבִיא אַתָּה אֶת הַדְּבָרִים שֶׁבִּרְצוֹנָם לְהַפְנוֹת לָאֱלֹהִים.
כ וְהִזְהַרְתָּ֣ה אֶתְהֶ֔ם אֶת־הַחֻקִּ֖ים וְאֶת־הַתּוֹרֹ֑ת וְהוֹדַעְתָּ֣ לָהֶ֗ם אֶת־הַדֶּ֙רֶךְ֙ יֵ֣לְכוּ בָ֔הּ וְאֶת־הַֽמַּעֲשֶׂ֖ה אֲשֶׁ֥ר יַעֲשֽׂוּן׃ וְתַזְהִיר אוֹתָם אֶת הַחֻקִּים וְאֶת הַמִּצְווֹת, וְתוֹדִיעַ לָהֶם אֶת הַדֶּרֶךְ שֶׁיֵּלְכוּ בָּהּ וְאֶת הַמַּעֲשִׂים שֶׁיַּעֲשׂוּ.
כא וְאַתָּ֣ה תֶחֱזֶ֣ה מִכׇּל־הָ֠עָ֠ם אַנְשֵׁי־חַ֜יִל יִרְאֵ֧י אֱלֹהִ֛ים אַנְשֵׁ֥י אֱמֶ֖ת שֹׂ֣נְאֵי בָ֑צַע וְשַׂמְתָּ֣ עֲלֵהֶ֗ם שָׂרֵ֤י אֲלָפִים֙ שָׂרֵ֣י מֵא֔וֹת שָׂרֵ֥י חֲמִשִּׁ֖ים וְשָׂרֵ֥י עֲשָׂרֹֽת׃ וְאַתָּה תִּבְחַר מִכָּל הָעָם אֲנָשִׁים גִּבּוֹרִים יִרְאֵי אֱלֹהִים אַנְשֵׁי אֱמֶת שׂוֹנְאֵי שׁוֹחַד, וּתְמַנֶּה אוֹתָם עַל הָעָם – לְשָׂרִים הַמְּמוּנִים עַל אֲלָפִים, שָׂרִים הַמְּמוּנִים עַל מֵאוֹת, שָׂרִים הַמְּמוּנִים עַל חֲמִשִּׁים, וְשָׂרִים הַמְּמוּנִים עַל עֲשָׂרוֹת.
כב וְשָׁפְט֣וּ אֶת־הָעָם֮ בְּכׇל־עֵת֒ וְהָיָ֞ה כׇּל־הַדָּבָ֤ר הַגָּדֹל֙ יָבִ֣יאוּ אֵלֶ֔יךָ וְכׇל־הַדָּבָ֥ר הַקָּטֹ֖ן יִשְׁפְּטוּ־הֵ֑ם וְהָקֵל֙ מֵֽעָלֶ֔יךָ וְנָשְׂא֖וּ אִתָּֽךְ׃ וְכָךְ יוּכְלוּ לִשְׁפּוֹט אֶת הָעָם בְּכָל זְמַן שֶׁיִּרְצוּ, כָּל דָּבָר גָּדוֹל יָבִיאוּ אֵלֶיךָ וְכָל דָּבָר קָטָן הֵם יִשְׁפְּטוּ, וְזֶה יָקֵל עָלֶיךָ וְהֵם יְבַצְּעוּ אֶת הַמְּשִׂימָה אִתְּךָ.
כג אִ֣ם אֶת־הַדָּבָ֤ר הַזֶּה֙ תַּעֲשֶׂ֔ה וְצִוְּךָ֣ אֱלֹהִ֔ים וְיָֽכׇלְתָּ֖ עֲמֹ֑ד וְגַם֙ כׇּל־הָעָ֣ם הַזֶּ֔ה עַל־מְקֹמ֖וֹ יָבֹ֥א בְשָׁלֽוֹם׃ אִם תַּעֲשֶׂה אֶת הַדָּבָר הַזֶּה תּוּכַל לַעֲמוֹד בְּצִוּוּיֵי הָאֱלֹהִים, וְגַם כָּל הָעָם הַזֶּה יָבוֹא אֶל מְקוֹמוֹ בְּשָׁלוֹם.

שלישי[עריכה]

כד וַיִּשְׁמַ֥ע מֹשֶׁ֖ה לְק֣וֹל חֹתְנ֑וֹ וַיַּ֕עַשׂ כֹּ֖ל אֲשֶׁ֥ר אָמָֽר׃ מֹשֶׁה שָׁמַע לְקוֹל חָמִיו, וְעָשָׂה אֶת כָּל מַה שֶּׁהוּא אָמָר.
כה וַיִּבְחַ֨ר מֹשֶׁ֤ה אַנְשֵׁי־חַ֙יִל֙ מִכׇּל־יִשְׂרָאֵ֔ל וַיִּתֵּ֥ן אֹתָ֛ם רָאשִׁ֖ים עַל־הָעָ֑ם שָׂרֵ֤י אֲלָפִים֙ שָׂרֵ֣י מֵא֔וֹת שָׂרֵ֥י חֲמִשִּׁ֖ים וְשָׂרֵ֥י עֲשָׂרֹֽת׃ מֹשֶׁה בָּחַר אֲנָשִׁים גִּבּוֹרִים מִכָּל יִשְׂרָאֵל וּמִנָּה אוֹתָם לְמַנְהִיגִים עַל הָעָם, שָׂרִים הַמְּמוּנִים עַל אֲלָפִים, שָׂרִים הַמְּמוּנִים עַל מֵאוֹת, שָׂרִים הַמְּמוּנִים עַל חֲמִשִּׁים, וְשָׂרִים הַמְּמוּנִים עַל עֲשָׂרוֹת.
כו וְשָׁפְט֥וּ אֶת־הָעָ֖ם בְּכׇל־עֵ֑ת אֶת־הַדָּבָ֤ר הַקָּשֶׁה֙ יְבִיא֣וּן אֶל־מֹשֶׁ֔ה וְכׇל־הַדָּבָ֥ר הַקָּטֹ֖ן יִשְׁפּוּט֥וּ הֵֽם׃ שֶׁיִּשְׁפְּטוּ אֶת הָעָם בְּכָל זְמַן שֶׁיִּרְצוּ, דָּבָר קָשֶׁה יָבִיאוּ לְמֹשֶׁה וְכָל דָּבָר קָטָן הֵם יִשְׁפְּטוּ.
כז וַיְשַׁלַּ֥ח מֹשֶׁ֖ה אֶת־חֹתְנ֑וֹ וַיֵּ֥לֶךְ ל֖וֹ אֶל־אַרְצֽוֹ׃ מֹשֶׁה לִוָּה אֶת חָמִיו, וְהוּא הָלַךְ לְאַרְצוֹ.

רביעי[עריכה]

פ פָּרָשַׁת הַהַגְבָּלָה
יט א בַּחֹ֙דֶשׁ֙ הַשְּׁלִישִׁ֔י לְצֵ֥את בְּנֵי־יִשְׂרָאֵ֖ל מֵאֶ֣רֶץ מִצְרָ֑יִם בַּיּ֣וֹם הַזֶּ֔ה בָּ֖אוּ מִדְבַּ֥ר סִינָֽי׃ בְּרֹאשׁ הַחוֹדֶשׁ הַשְּׁלִישִׁי לְצֵאת בְּנֵי יִשְׂרָאֵל מֵאֶרֶץ מִצְרַיִם, בְּאוֹתוֹ הַיּוֹם בָּאוּ לְמִדְבַּר סִינַי.
ב וַיִּסְע֣וּ מֵרְפִידִ֗ים וַיָּבֹ֙אוּ֙ מִדְבַּ֣ר סִינַ֔י וַֽיַּחֲנ֖וּ בַּמִּדְבָּ֑ר וַיִּֽחַן־שָׁ֥ם יִשְׂרָאֵ֖ל נֶ֥גֶד הָהָֽר׃ הֵם נָסְעוּ מֵרְפִידִים וּבָאוּ לְמִדְבַּר סִינַי וְחָנוּ בַּמִּדְבָּר, יִשְׂרָאֵל חָנוּ שָׁם מוּל הָהָר.
ג וּמֹשֶׁ֥ה עָלָ֖ה אֶל־הָאֱלֹהִ֑ים וַיִּקְרָ֨א אֵלָ֤יו יְהֹוָה֙ מִן־הָהָ֣ר לֵאמֹ֔ר כֹּ֤ה תֹאמַר֙ לְבֵ֣ית יַעֲקֹ֔ב וְתַגֵּ֖יד לִבְנֵ֥י יִשְׂרָאֵֽל׃ וּמֹשֶׁה עָלָה אֶל הָאֱלֹהִים, ה' קָרָא לוֹ מֵהָהָר וְאָמַר: כָּךְ תֹּאמַר לְבֵית יַעֲקֹב וְתַגִּיד לִבְנֵי יִשְׂרָאֵל:
ד אַתֶּ֣ם רְאִיתֶ֔ם אֲשֶׁ֥ר עָשִׂ֖יתִי לְמִצְרָ֑יִם וָאֶשָּׂ֤א אֶתְכֶם֙ עַל־כַּנְפֵ֣י נְשָׁרִ֔ים וָאָבִ֥א אֶתְכֶ֖ם אֵלָֽי׃ אַתֶּם רְאִיתֶם אֶת מַה שֶּׁעָשִׂיתִי לְמִצְרַיִם, וְשֶׁלָּקַחְתִּי אֶתְכֶם כְּמוֹ עַל כַּנְפֵי נְשָׁרִים וְהֵבֵאתִי אֶתְכֶם אֵלַי.
ה וְעַתָּ֗ה אִם־שָׁמ֤וֹעַ תִּשְׁמְעוּ֙ בְּקֹלִ֔י וּשְׁמַרְתֶּ֖ם אֶת־בְּרִיתִ֑י וִהְיִ֨יתֶם לִ֤י סְגֻלָּה֙ מִכׇּל־הָ֣עַמִּ֔ים כִּי־לִ֖י כׇּל־הָאָֽרֶץ׃ וְעַכְשָׁיו, אִם תִּשְׁמְעוּ בְּקוֹלִי וְתִשְׁמְרוּ אֶת הַהֶסְכֵּם אִתִּי – תִּהְיוּ הָעָם הַמְּיוּחָד וְהַמְּקוֹרָב אֱלַי יוֹתֵר מִכָּל הָעַמִּים, שֶׁהֲרֵי כָּל הָאָרֶץ שֶׁלִּי.
ו וְאַתֶּ֧ם תִּהְיוּ־לִ֛י מַמְלֶ֥כֶת כֹּהֲנִ֖ים וְג֣וֹי קָד֑וֹשׁ אֵ֚לֶּה הַדְּבָרִ֔ים אֲשֶׁ֥ר תְּדַבֵּ֖ר אֶל־בְּנֵ֥י יִשְׂרָאֵֽל׃ וְאַתֶּם תִּהְיוּ לִי מַמְלֶכֶת כֹּהֲנִים וְעַם קָדוֹשׁ. אֵלֶּה הַדְּבָרִים שֶׁתְּדַבֵּר אֶל בְּנֵי יִשְׂרָאֵל.

חמישי[עריכה]

ז וַיָּבֹ֣א מֹשֶׁ֔ה וַיִּקְרָ֖א לְזִקְנֵ֣י הָעָ֑ם וַיָּ֣שֶׂם לִפְנֵיהֶ֗ם אֵ֚ת כׇּל־הַדְּבָרִ֣ים הָאֵ֔לֶּה אֲשֶׁ֥ר צִוָּ֖הוּ יְהֹוָֽה׃ מֹשֶׁה בָּא וְקָרָא לְזִקְנֵי הָעָם, וְהִסְבִּיר לָהֶם אֶת כָּל הַדְּבָרִים הָאֵלֶּה שֶׁה' צִוָּה אוֹתוֹ.
ח וַיַּעֲנ֨וּ כׇל־הָעָ֤ם יַחְדָּו֙ וַיֹּ֣אמְר֔וּ כֹּ֛ל אֲשֶׁר־דִּבֶּ֥ר יְהֹוָ֖ה נַעֲשֶׂ֑ה וַיָּ֧שֶׁב מֹשֶׁ֛ה אֶת־דִּבְרֵ֥י הָעָ֖ם אֶל־יְהֹוָֽה׃ כָּל הָעָם עָנוּ יַחַד וְאָמְרוּ: כָּל מַה שֶּׁה' דִּבֵּר נַעֲשֶׂה! מֹשֶׁה הֵשִׁיב אֶת דִּבְרֵי הָעָם אֶל ה'.
ט וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה הִנֵּ֨ה אָנֹכִ֜י בָּ֣א אֵלֶ֘יךָ֮ בְּעַ֣ב הֶֽעָנָן֒ בַּעֲב֞וּר יִשְׁמַ֤ע הָעָם֙ בְּדַבְּרִ֣י עִמָּ֔ךְ וְגַם־בְּךָ֖ יַאֲמִ֣ינוּ לְעוֹלָ֑ם וַיַּגֵּ֥ד מֹשֶׁ֛ה אֶת־דִּבְרֵ֥י הָעָ֖ם אֶל־יְהֹוָֽה׃ ה' אָמַר לְמֹשֶׁה: הִנֵּה אֲנִי בָּא אֵלֶיךָ בָּעֲרָפֶל שֶׁל הֶעָנָן כְּדֵי שֶׁהָעָם יִשְׁמַע שֶׁאֲנִי מְדַבֵּר אִתְּךָ, וְיַאֲמִינוּ לְעוֹלָם גַם בְּךָ שֶׁאַתָּה שְׁלִיחִי, מֹשֶׁה סִפֵּר אֶת דִּבְרֵי הָעָם לַה'.
י וַיֹּ֨אמֶר יְהֹוָ֤ה אֶל־מֹשֶׁה֙ לֵ֣ךְ אֶל־הָעָ֔ם וְקִדַּשְׁתָּ֥ם הַיּ֖וֹם וּמָחָ֑ר וְכִבְּס֖וּ שִׂמְלֹתָֽם׃ ה' אָמַר לְמֹשֶׁה: לֵךְ לָעָם וְתָכִין אוֹתָם הַיּוֹם וּמָחָר שֶׁיִּטְבְּלוּ וִיכַבְּסוּ אֶת בִּגְדֵיהֶם.
יא וְהָי֥וּ נְכֹנִ֖ים לַיּ֣וֹם הַשְּׁלִישִׁ֑י כִּ֣י ׀ בַּיּ֣וֹם הַשְּׁלִשִׁ֗י יֵרֵ֧ד יְהֹוָ֛ה לְעֵינֵ֥י כׇל־הָעָ֖ם עַל־הַ֥ר סִינָֽי׃ וְיִהְיוּ מוּכָנִים לַיּוֹם הַשְּׁלִישִׁי, כִּי בַּיּוֹם הַשְּׁלִישִׁי ה' יִתְגַּלֶּה לְעֵינֵי כָּל הָעָם עַל הַר סִינַי.
יב וְהִגְבַּלְתָּ֤ אֶת־הָעָם֙ סָבִ֣יב לֵאמֹ֔ר הִשָּׁמְר֥וּ לָכֶ֛ם עֲל֥וֹת בָּהָ֖ר וּנְגֹ֣עַ בְּקָצֵ֑הוּ כׇּל־הַנֹּגֵ֥עַ בָּהָ֖ר מ֥וֹת יוּמָֽת׃ וּתְסַמֵּן מִסָּבִיב לָהָר גְּבוּל לָעָם, וְתֹאמַר: הִזָּהֲרוּ לָכֶם מִלַּעֲלוֹת בָּהָר וּמִלִּנְגּוֹעַ בְּקָצֵהוּ – כָּל מִי שֶׁנּוֹגֵעַ בָּהָר יוּמַת.
יג לֹא־תִגַּ֨ע בּ֜וֹ יָ֗ד כִּֽי־סָק֤וֹל יִסָּקֵל֙ אוֹ־יָרֹ֣ה יִיָּרֶ֔ה אִם־בְּהֵמָ֥ה אִם־אִ֖ישׁ לֹ֣א יִחְיֶ֑ה בִּמְשֹׁךְ֙ הַיֹּבֵ֔ל הֵ֖מָּה יַעֲל֥וּ בָהָֽר׃ אַל תִּגַּע בּוֹ יָד כִּי הַנּוֹגֵעַ יִסָּקֵל אוֹ יוּשְׁלַךְ מִגּוֹבַהּ, בֵּין אִם בְּהֵמָה בֵּין אִם אִישׁ – לֹא יִחְיֶה, כְּשֶׁתִּהְיֶה תְּקִיעַת הַשּׁוֹפָר הֵם יַעֲלוּ בָּהָר.
יד וַיֵּ֧רֶד מֹשֶׁ֛ה מִן־הָהָ֖ר אֶל־הָעָ֑ם וַיְקַדֵּשׁ֙ אֶת־הָעָ֔ם וַֽיְכַבְּס֖וּ שִׂמְלֹתָֽם׃ מֹשֶׁה יָרַד מֵהָהָר אֶל הָעָם וְהֵכִין אֶת הָעָם וְהֵם טָבְלוּ וְכִבְּסוּ אֶת בִּגְדֵיהֶם.
טו וַיֹּ֙אמֶר֙ אֶל־הָעָ֔ם הֱי֥וּ נְכֹנִ֖ים לִשְׁלֹ֣שֶׁת יָמִ֑ים אַֽל־תִּגְּשׁ֖וּ אֶל־אִשָּֽׁה׃ אָמַר לָעָם: הֱיוּ מוּכָנִים לְעוֹד שְׁלוֹשָׁה יָמִים – אַל תִּגְּשׁוּ לִשְׁכַּב עִם אִשָּׁה.
טז וַיְהִי֩ בַיּ֨וֹם הַשְּׁלִישִׁ֜י בִּֽהְיֹ֣ת הַבֹּ֗קֶר וַיְהִי֩ קֹלֹ֨ת וּבְרָקִ֜ים וְעָנָ֤ן כָּבֵד֙ עַל־הָהָ֔ר וְקֹ֥ל שֹׁפָ֖ר חָזָ֣ק מְאֹ֑ד וַיֶּחֱרַ֥ד כׇּל־הָעָ֖ם אֲשֶׁ֥ר בַּֽמַּחֲנֶֽה׃ בַּיּוֹם הַשְּׁלִישִׁי כְּשֶׁעָלָה הַבּוֹקֶר, הָיוּ עַל הָהָר קוֹלוֹת וּבְרָקִים וְעָנָן כָּבֵד וְהָיָה קוֹל שׁוֹפָר חָזָק מְאוֹד, כָּל הָעָם שֶׁבַּמַּחֲנֶה נֶחְרְדוּ.
יז וַיּוֹצֵ֨א מֹשֶׁ֧ה אֶת־הָעָ֛ם לִקְרַ֥את הָֽאֱלֹהִ֖ים מִן־הַֽמַּחֲנֶ֑ה וַיִּֽתְיַצְּב֖וּ בְּתַחְתִּ֥ית הָהָֽר׃ מֹשֶׁה הוֹצִיא אֶת הָעָם מֵהַמַּחֲנֶה לִקְרַאת הָאֱלֹהִים, וְהֵם הִתְיַצְּבוּ לְמַרְגְּלוֹת הָהָר.
יח וְהַ֤ר סִינַי֙ עָשַׁ֣ן כֻּלּ֔וֹ מִ֠פְּנֵ֠י אֲשֶׁ֨ר יָרַ֥ד עָלָ֛יו יְהֹוָ֖ה בָּאֵ֑שׁ וַיַּ֤עַל עֲשָׁנוֹ֙ כְּעֶ֣שֶׁן הַכִּבְשָׁ֔ן וַיֶּחֱרַ֥ד כׇּל־הָהָ֖ר מְאֹֽד׃ וְהַר סִינַי כֻּלּוֹ הֶעֱלָה עָשָׁן מִפְּנֵי שֶׁה' יָרַד עָלָיו בָּאֵשׁ, וַעֲשָׁנוֹ עָלָה כְּעָשָׁן הַכִּבְשָׁן, כָּל הָהָר הִזְדַּעֲזֵעַ מְאוֹד.
יט וַיְהִי֙ ק֣וֹל הַשֹּׁפָ֔ר הוֹלֵ֖ךְ וְחָזֵ֣ק מְאֹ֑ד מֹשֶׁ֣ה יְדַבֵּ֔ר וְהָאֱלֹהִ֖ים יַעֲנֶ֥נּוּ בְקֽוֹל׃ קוֹל הַשּׁוֹפָר הָיָה הוֹלֵךְ וּמִתְחַזֵּק מְאוֹד, מֹשֶׁה מְדַבֵּר וְהָאֱלֹהִים עוֹנֶה לוֹ בְּקוֹל.

ששי[עריכה]

כ וַיֵּ֧רֶד יְהֹוָ֛ה עַל־הַ֥ר סִינַ֖י אֶל־רֹ֣אשׁ הָהָ֑ר וַיִּקְרָ֨א יְהֹוָ֧ה לְמֹשֶׁ֛ה אֶל־רֹ֥אשׁ הָהָ֖ר וַיַּ֥עַל מֹשֶֽׁה׃ ה' הִתְגַּלָּה עַל הַר סִינַי עַל רֹאשׁ הָהָר, ה' קָרָא לְמֹשֶׁה אֶל רֹאשׁ הָהָר, מֹשֶׁה עָלָה.
כא וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה רֵ֖ד הָעֵ֣ד בָּעָ֑ם פֶּן־יֶהֶרְס֤וּ אֶל־יְהֹוָה֙ לִרְא֔וֹת וְנָפַ֥ל מִמֶּ֖נּוּ רָֽב׃ ה' אָמַר לְמֹשֶׁה: רֵד הַזְהֵר אֶת הָעָם שֶׁלֹּא יִתְפָּרְצוּ לַעֲלוֹת אֶל ה' לִרְאוֹת וְיָמוּתוּ מֵהֶם אֲנָשִׁים רַבִּים.
כב וְגַ֧ם הַכֹּהֲנִ֛ים הַנִּגָּשִׁ֥ים אֶל־יְהֹוָ֖ה יִתְקַדָּ֑שׁוּ פֶּן־יִפְרֹ֥ץ בָּהֶ֖ם יְהֹוָֽה׃ וְגַם הַכֹּהֲנִים שֶׁנִּגָּשִׁים תָּמִיד אֶל ה' יִזָּהֲרוּ לְהִשָּׁאֵר בִּמְקוֹמָם, כְּדֵי שֶׁה' לֹא יַהֲרוֹג אוֹתָם.
כג וַיֹּ֤אמֶר מֹשֶׁה֙ אֶל־יְהֹוָ֔ה לֹא־יוּכַ֣ל הָעָ֔ם לַעֲלֹ֖ת אֶל־הַ֣ר סִינָ֑י כִּֽי־אַתָּ֞ה הַעֵדֹ֤תָה בָּ֙נוּ֙ לֵאמֹ֔ר הַגְבֵּ֥ל אֶת־הָהָ֖ר וְקִדַּשְׁתּֽוֹ׃ מֹשֶׁה אָמַר לַה': הֲרֵי הָעָם לֹא יוּכַל לַעֲלוֹת אֶל הַר סִינַי כִּי אַתָּה הִזְהַרְתָּ אוֹתָנוּ וְאָמַרְתָּ: "סַמֵּן אֶת גְּבוּל הָהָר, וְהַכְרֵז עַל קְדֻשָּׁתוֹ"!
כד וַיֹּ֨אמֶר אֵלָ֤יו יְהֹוָה֙ לֶךְ־רֵ֔ד וְעָלִ֥יתָ אַתָּ֖ה וְאַהֲרֹ֣ן עִמָּ֑ךְ וְהַכֹּהֲנִ֣ים וְהָעָ֗ם אַל־יֶֽהֶרְס֛וּ לַעֲלֹ֥ת אֶל־יְהֹוָ֖ה פֶּן־יִפְרׇץ־בָּֽם׃ אָמַר לוֹ ה': בְּכָל זֹאת לֵךְ רֵד, וְתַעֲלֶה רַק אַתָּה וְאַהֲרֹן אִתְּךָ, וְהַכֹּהֲנִים וְהָעָם לֹא יִתְפָּרְצוּ לַעֲלוֹת אֶל ה' שֶׁלֹּא יַהֲרוֹג אוֹתָם.
כה וַיֵּ֥רֶד מֹשֶׁ֖ה אֶל־הָעָ֑ם וַיֹּ֖אמֶר אֲלֵהֶֽם׃ מֹשֶׁה יָרַד לָעָם וְאָמַר לָהֶם:
ס עֲשֶׂרֶת הַדִּבְּרוֹת
כ א וַיְדַבֵּ֣ר אֱלֹהִ֔ים אֵ֛ת כׇּל־הַדְּבָרִ֥ים הָאֵ֖לֶּה לֵאמֹֽר׃ אֱלֹהִים דִּבֵּר אֶת כָּל הַדְּבָרִים הָאֵלֶּה, וְאָמַר:
ס
ב אָֽנֹכִי֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ אֲשֶׁ֧ר הוֹצֵאתִ֛יךָ מֵאֶ֥רֶץ מִצְרַ֖יִם מִבֵּ֣ית עֲבָדִ֑ים לֹֽא־יִהְיֶ֥ה לְךָ֛ אֱלֹהִ֥ים אֲחֵרִ֖ים עַל־פָּנָֽי׃ אֲנִי ה' אֱלֹהֶיךָ שֶׁהוֹצֵאתִי אוֹתְךָ מֵאֶרֶץ מִצְרַיִם מִבֵּית עֲבָדִים. לֹא יִהְיֶה לְךָ אֱלֹהִים אֲחֵרִים חוּץ מִמֶּנִּי.
ג לֹֽא־תַעֲשֶׂ֨ה לְךָ֥ פֶ֙סֶל֙ וְכׇל־תְּמוּנָ֔ה אֲשֶׁ֤ר בַּשָּׁמַ֙יִם֙ מִמַּ֔עַל וַֽאֲשֶׁ֥ר בָּאָ֖רֶץ מִתָּ֑חַת וַאֲשֶׁ֥ר בַּמַּ֖יִם מִתַּ֥חַת לָאָֽרֶץ׃ אַל תַּעֲשֶׂה לְךָ פֶּסֶל וְשׁוּם דְּמוּת שֶׁל דָּבָר שֶׁבַּשָּׁמַיִם לְמַעְלָה אוֹ שֶׁבָּאָרֶץ מִתַּחַת, אוֹ שֶׁבַּמַּיִם – שֶׁנְּמוּכִים מֵהַיַּבָּשָׁה.
ד לֹֽא־תִשְׁתַּחֲוֶ֥ה לָהֶ֖ם וְלֹ֣א תׇעׇבְדֵ֑ם כִּ֣י אָֽנֹכִ֞י יְהֹוָ֤ה אֱלֹהֶ֙יךָ֙ אֵ֣ל קַנָּ֔א פֹּ֠קֵ֠ד עֲוֺ֨ן אָבֹ֧ת עַל־בָּנִ֛ים עַל־שִׁלֵּשִׁ֥ים וְעַל־רִבֵּעִ֖ים לְשֹׂנְאָֽי׃ אַל תִּשְׁתַּחֲוֶה לָהֶם וְאַל תַּעֲבוֹד אוֹתָם כִּי אֲנִי ה' אֱלֹהֶיךָ אֵל נוֹקֵם, לַשּׂוֹנְאִים אוֹתִי אֲנִי זוֹכֵר עֲווֹן שֶׁל אָבוֹת – לַבָּנִים לַנְּכָדִים וְלַנִּינִים וּמַעֲנִישׁ אוֹתָם.
ה וְעֹ֥שֶׂה חֶ֖סֶד לַאֲלָפִ֑ים לְאֹהֲבַ֖י וּלְשֹׁמְרֵ֥י מִצְוֺתָֽי׃ וְעוֹשֶׂה חֶסֶד לְאַלְפֵי דּוֹרוֹת – לְאוֹהֲבַי וּלְשׁוֹמְרֵי מִצְווֹתַי.
ס
ו לֹ֥א תִשָּׂ֛א אֶת־שֵֽׁם־יְהֹוָ֥ה אֱלֹהֶ֖יךָ לַשָּׁ֑וְא כִּ֣י לֹ֤א יְנַקֶּה֙ יְהֹוָ֔ה אֵ֛ת אֲשֶׁר־יִשָּׂ֥א אֶת־שְׁמ֖וֹ לַשָּֽׁוְא׃ אַל תַּזְכִּיר אֶת שֵׁם ה' אֱלֹהֶיךָ לְחִנָּם, כִּי ה' לֹא יִסְלַח לְמִי שֶׁיַּזְכִּיר אֶת שְׁמוֹ לְחִנָּם.
פ
ז זָכ֛וֹר אֶת־י֥וֹם הַשַּׁבָּ֖ת לְקַדְּשֽׁוֹ׃ זְכוֹר אֶת יוֹם הַשַּׁבָּת לְקַדְּשׁוֹ.
ח שֵׁ֤שֶׁת יָמִים֙ תַּֽעֲבֹ֔ד וְעָשִׂ֖יתָ כׇּל־מְלַאכְתֶּֽךָ׃ שִׁשָּׁה יָמִים תַּעֲבוֹד וְתַעֲשֶׂה אֶת כָּל מְלַאכְתְּךָ.
ט וְיוֹם֙ הַשְּׁבִיעִ֔י שַׁבָּ֖ת לַיהֹוָ֣ה אֱלֹהֶ֑יךָ לֹֽא־תַעֲשֶׂ֨ה כׇל־מְלָאכָ֜ה אַתָּ֣ה ׀ וּבִנְךָ֣ וּבִתֶּ֗ךָ עַבְדְּךָ֤ וַאֲמָֽתְךָ֙ וּבְהֶמְתֶּ֔ךָ וְגֵרְךָ֖ אֲשֶׁ֥ר בִּשְׁעָרֶֽיךָ׃ וְיוֹם הַשְּׁבִיעִי יוֹם שְׁבִיתָה לַה' אֱלֹהֶיךָ – אַל תַּעֲשֶׂה שׁוּם מְלָאכָה, אַתָּה וּבִנְךָ וּבִתְּךָ עַבְדְּךָ וְשִׁפְחָתְךָ וּבְהֶמְתְּךָ וְגֵרְךָ שֶׁבְּעָרֶיךָ.
י כִּ֣י שֵֽׁשֶׁת־יָמִים֩ עָשָׂ֨ה יְהֹוָ֜ה אֶת־הַשָּׁמַ֣יִם וְאֶת־הָאָ֗רֶץ אֶת־הַיָּם֙ וְאֶת־כׇּל־אֲשֶׁר־בָּ֔ם וַיָּ֖נַח בַּיּ֣וֹם הַשְּׁבִיעִ֑י עַל־כֵּ֗ן בֵּרַ֧ךְ יְהֹוָ֛ה אֶת־י֥וֹם הַשַּׁבָּ֖ת וַֽיְקַדְּשֵֽׁהוּ׃ כִּי שִׁשָּׁה יָמִים ה' עָשָׂה אֶת הַשָּׁמַיִם וְאֶת הָאָרֶץ אֶת הַיָּם וְאֶת כָּל מַה שֶּׁבָּהֶם, וְהוּא נָח בַּיּוֹם הַשְּׁבִיעִי, לָכֵן ה' בֵּרַךְ אֶת יוֹם הַשַּׁבָּת וְקִדֵּשׁ אוֹתוֹ.

יא כַּבֵּ֥ד אֶת־אָבִ֖יךָ וְאֶת־אִמֶּ֑ךָ לְמַ֙עַן֙ יַאֲרִכ֣וּן יָמֶ֔יךָ עַ֚ל הָאֲדָמָ֔ה אֲשֶׁר־יְהֹוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֥ן לָֽךְ׃ כַּבֵּד אֶת אָבִיךָ וְאֶת אִמְּךָ כְּדֵי שֶׁיִּתְאָרְכוּ חַיֶּיךָ עַל הָאֲדָמָה שֶׁה' אֱלֹהֶיךָ נוֹתֵן לְךָ.

יג לֹ֥א תַחְמֹ֖ד בֵּ֣ית רֵעֶ֑ךָ  לֹֽא־תַחְמֹ֞ד אֵ֣שֶׁת רֵעֶ֗ךָ וְעַבְדּ֤וֹ וַאֲמָתוֹ֙ וְשׁוֹר֣וֹ וַחֲמֹר֔וֹ וְכֹ֖ל אֲשֶׁ֥ר לְרֵעֶֽךָ׃ אַל תַּחְמוֹד אֶת בֵּית חֲבֵרְךָ, אַל תַּחְמוֹד אֶת אֵשֶׁת חֲבֵרְךָ וְעַבְדּוֹ וְשִׁפְחָתוֹ וְשׁוֹרוֹ וַחֲמוֹרוֹ וְכָל מַה שֶּׁשַּׁיָּךְ לַחֲבֵרְךָ.

שביעי[עריכה]

פ פָּרָשַׁת יִרְאַת מַעֲמַד הַר סִינַי
יד וְכׇל־הָעָם֩ רֹאִ֨ים אֶת־הַקּוֹלֹ֜ת וְאֶת־הַלַּפִּידִ֗ם וְאֵת֙ ק֣וֹל הַשֹּׁפָ֔ר וְאֶת־הָהָ֖ר עָשֵׁ֑ן וַיַּ֤רְא הָעָם֙ וַיָּנֻ֔עוּ וַיַּֽעַמְד֖וּ מֵֽרָחֹֽק׃ וְכָל הָעָם רוֹאִים וְשׁוֹמְעִים אֶת הַקּוֹלוֹת וְאֶת הַלַּפִּידִים וְאֶת קוֹל הַשּׁוֹפָר, וְאֶת הָהָר מַעֲלֵה עָשָׁן, הָעָם רָאוּ וְהֵם זָזוּ וְעָמְדוּ מֵרָחוֹק.
טו וַיֹּֽאמְרוּ֙ אֶל־מֹשֶׁ֔ה דַּבֵּר־אַתָּ֥ה עִמָּ֖נוּ וְנִשְׁמָ֑עָה וְאַל־יְדַבֵּ֥ר עִמָּ֛נוּ אֱלֹהִ֖ים פֶּן־נָמֽוּת׃ הֵם אָמְרוּ לְמֹשֶׁה: דַּבֵּר אִתָּנוּ אַתָּה וְנוּכַל לִשְׁמוֹעַ, וְלֹא יְדַבֵּר אִתָּנוּ אֱלֹהִים, שֶׁלֹּא נָמוּת.
טז וַיֹּ֨אמֶר מֹשֶׁ֣ה אֶל־הָעָם֮ אַל־תִּירָ֒אוּ֒ כִּ֗י לְבַֽעֲבוּר֙ נַסּ֣וֹת אֶתְכֶ֔ם בָּ֖א הָאֱלֹהִ֑ים וּבַעֲב֗וּר תִּהְיֶ֧ה יִרְאָת֛וֹ עַל־פְּנֵיכֶ֖ם לְבִלְתִּ֥י תֶחֱטָֽאוּ׃ אָמַר מֹשֶׁה לָעָם: אַל תִּפְחֲדוּ! כִּי הָאֱלֹהִים בָּא בִּשְׁבִיל לְרוֹמֵם אֶתְכֶם, וּכְדֵי שֶׁתִּהְיֶה נִכֶּרֶת רוֹמְמוּתוֹ עַל פְּנֵיכֶם כְּדֵי שֶׁלֹּא תֶּחֶטְאוּ.
יז וַיַּעֲמֹ֥ד הָעָ֖ם מֵרָחֹ֑ק וּמֹשֶׁה֙ נִגַּ֣שׁ אֶל־הָֽעֲרָפֶ֔ל אֲשֶׁר־שָׁ֖ם הָאֱלֹהִֽים׃ הָעָם עָמַד מֵרָחוֹק, וּמֹשֶׁה נִגַּשׁ אֶל הָעֲרָפֶל שֶׁשָּׁם הִתְגַּלּוּת הָאֱלֹהִים.

מפטיר[עריכה]

ס
יח וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה כֹּ֥ה תֹאמַ֖ר אֶל־בְּנֵ֣י יִשְׂרָאֵ֑ל אַתֶּ֣ם רְאִיתֶ֔ם כִּ֚י מִן־הַשָּׁמַ֔יִם דִּבַּ֖רְתִּי עִמָּכֶֽם׃ ה' אָמַר לְמֹשֶׁה: כָּךְ תֹּאמַר לִבְנֵי יִשְׂרָאֵל: אַתֶּם רְאִיתֶם שֶׁדִּבַּרְתִּי אִתְּכֶם מֵהַשָּׁמַיִם.
יט לֹ֥א תַעֲשׂ֖וּן אִתִּ֑י אֱלֹ֤הֵי כֶ֙סֶף֙ וֵאלֹהֵ֣י זָהָ֔ב לֹ֥א תַעֲשׂ֖וּ לָכֶֽם׃ לָכֵן, אַל תַּעֲשׂוּ לָכֶם בְּנוֹסָף עָלַי שׁוּם אֱלֹהֵי כֶּסֶף וֵאלֹהֵי זָהָב.
כ מִזְבַּ֣ח אֲדָמָה֮ תַּעֲשֶׂה־לִּי֒ וְזָבַחְתָּ֣ עָלָ֗יו אֶת־עֹלֹתֶ֙יךָ֙ וְאֶת־שְׁלָמֶ֔יךָ אֶת־צֹֽאנְךָ֖ וְאֶת־בְּקָרֶ֑ךָ בְּכׇל־הַמָּקוֹם֙ אֲשֶׁ֣ר אַזְכִּ֣יר אֶת־שְׁמִ֔י אָב֥וֹא אֵלֶ֖יךָ וּבֵרַכְתִּֽיךָ׃ עֲשֵׂה לִי מִזְבֵּחַ מֵאֲדָמָה, וְתַקְרִיב עָלָיו אֶת עוֹלוֹתֶיךָ וְאֶת שְׁלָמֶיךָ אֶת צֹאנְךָ וְאֶת בְּקָרְךָ, בְּכָל מָקוֹם שֶׁאֶתְגַּלֶּה בּוֹ – אָבוֹא אֵלֶיךָ וַאֲבָרֵךְ אוֹתְךָ.
כא וְאִם־מִזְבַּ֤ח אֲבָנִים֙ תַּֽעֲשֶׂה־לִּ֔י לֹֽא־תִבְנֶ֥ה אֶתְהֶ֖ן גָּזִ֑ית כִּ֧י חַרְבְּךָ֛ הֵנַ֥פְתָּ עָלֶ֖יהָ וַתְּחַֽלְלֶֽהָ׃ וּכְשֶׁתַּעֲשֶׂה לִי מִזְבֵּחַ מֵאֲבָנִים, אַל תִּבְנֶה אוֹתָן מְסֻתָּתוֹת בִּכְלֵי מַתֶּכֶת, כִּי אֶת חַרְבְּךָ הֵנַפְתָּ עָלָיו – וְחִלַּלְתָּ אוֹתוֹ.
כב וְלֹֽא־תַעֲלֶ֥ה בְמַעֲלֹ֖ת עַֽל־מִזְבְּחִ֑י אֲשֶׁ֛ר לֹֽא־תִגָּלֶ֥ה עֶרְוָתְךָ֖ עָלָֽיו׃ וְאַל תַּעֲלֶה בְּמַדְרֵגוֹת עַל מִזְבְּחִי, כְּדֵי שֶׁלֹּא יִתְגַּלֶּה עָלָיו הַמָּקוֹם הַמֻּצְנָע שֶׁלְּךָ.
