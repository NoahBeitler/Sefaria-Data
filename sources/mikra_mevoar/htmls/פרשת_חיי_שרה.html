ראשון[עריכה]

פ פָּרָשַׁת מְעָרַת הַמַּכְפֵּלָה
כג א וַיִּהְיוּ֙ חַיֵּ֣י שָׂרָ֔ה מֵאָ֥ה שָׁנָ֛ה וְעֶשְׂרִ֥ים שָׁנָ֖ה וְשֶׁ֣בַע שָׁנִ֑ים שְׁנֵ֖י חַיֵּ֥י שָׂרָֽה׃ שָׂרָה חָיָה מֵאָה וְעֶשְׂרִים וְשֶׁבַע שָׁנִים – אֵלֶּה שְׁנוֹת חַיֵּי שָׂרָה.
ב וַתָּ֣מׇת שָׂרָ֗ה בְּקִרְיַ֥ת אַרְבַּ֛ע הִ֥וא חֶבְר֖וֹן בְּאֶ֣רֶץ כְּנָ֑עַן וַיָּבֹא֙ אַבְרָהָ֔ם לִסְפֹּ֥ד לְשָׂרָ֖ה וְלִבְכֹּתָֽהּ׃ שָׂרָה מֵתָה בְּקִרְיַת אַרְבַּע הִיא חֶבְרוֹן בְּאֶרֶץ כְּנַעַן, וְאַבְרָהָם בָּא לְהַסְפִּיד אֶת שָׂרָה וְלִבְכּוֹת עָלֶיהָ.
ג וַיָּ֙קׇם֙ אַבְרָהָ֔ם מֵעַ֖ל פְּנֵ֣י מֵת֑וֹ וַיְדַבֵּ֥ר אֶל־בְּנֵי־חֵ֖ת לֵאמֹֽר׃ אַבְרָהָם קָם מִלִּפְנֵי מֵתוֹ, וְדִבֵּר אֶל בְּנֵי חֵת וְאָמַר:
ד גֵּר־וְתוֹשָׁ֥ב אָנֹכִ֖י עִמָּכֶ֑ם תְּנ֨וּ לִ֤י אֲחֻזַּת־קֶ֙בֶר֙ עִמָּכֶ֔ם וְאֶקְבְּרָ֥ה מֵתִ֖י מִלְּפָנָֽי׃ אֲנִי אֶצְלְכֶם זָר וְתוֹשַׁב זְמַנִּי לְלֹא קַרְקַע, תְּנוּ לִי חֶלְקַת קֶבֶר בְּאַדְמַתְכֶם וְאֶקְבּוֹר אֶת הַמֵּת שֶׁאֶצְלִי.
ה וַיַּעֲנ֧וּ בְנֵי־חֵ֛ת אֶת־אַבְרָהָ֖ם לֵאמֹ֥ר לֽוֹ׃ בְּנֵי חֵת עָנוּ לְאַבְרָהָם וְאָמְרוּ לוֹ:
ו שְׁמָעֵ֣נוּ ׀ אֲדֹנִ֗י נְשִׂ֨יא אֱלֹהִ֤ים אַתָּה֙ בְּתוֹכֵ֔נוּ בְּמִבְחַ֣ר קְבָרֵ֔ינוּ קְבֹ֖ר אֶת־מֵתֶ֑ךָ אִ֣ישׁ מִמֶּ֔נּוּ אֶת־קִבְר֛וֹ לֹֽא־יִכְלֶ֥ה מִמְּךָ֖ מִקְּבֹ֥ר מֵתֶֽךָ׃ שְׁמַע אוֹתָנוּ אֲדוֹנִי, אַתָּה נָשִׂיא מְכֻבָּד אֶצְלֵנוּ, קְבוֹר אֶת מֵתְךָ בַּמֻּבְחַר שֶׁבִּקְבָרֵינוּ, אַף אֶחָד מֵאִתָּנוּ לֹא יִמְנַע מִמְּךָ מִלִּקְבוֹר אֶת מֵתְךָ בְּאַדְמָתוֹ.
ז וַיָּ֧קׇם אַבְרָהָ֛ם וַיִּשְׁתַּ֥חוּ לְעַם־הָאָ֖רֶץ לִבְנֵי־חֵֽת׃ אַבְרָהָם קָם וְהִשְׁתַּחֲוָה לְהוֹדוֹת לָעָם שֶׁגָּר בָּאָרֶץ – לִבְנֵי חֵת.
ח וַיְדַבֵּ֥ר אִתָּ֖ם לֵאמֹ֑ר אִם־יֵ֣שׁ אֶֽת־נַפְשְׁכֶ֗ם לִקְבֹּ֤ר אֶת־מֵתִי֙ מִלְּפָנַ֔י שְׁמָע֕וּנִי וּפִגְעוּ־לִ֖י בְּעֶפְר֥וֹן בֶּן־צֹֽחַר׃ הוּא דִּבֵּר אִתָּם וְאָמַר: אִם יֵשׁ בִּרְצוֹנְכֶם לִקְבּוֹר אֶת הַמֵּת שֶׁאֶצְלִי, שִׁמְעוּ בְּקוֹלִי וּבַקְּשׁוּ עֲבוּרִי מֵעֶפְרוֹן בֶּן צוֹחַר.
ט וְיִתֶּן־לִ֗י אֶת־מְעָרַ֤ת הַמַּכְפֵּלָה֙ אֲשֶׁר־ל֔וֹ אֲשֶׁ֖ר בִּקְצֵ֣ה שָׂדֵ֑הוּ בְּכֶ֨סֶף מָלֵ֜א יִתְּנֶ֥נָּה לִּ֛י בְּתוֹכְכֶ֖ם לַאֲחֻזַּת־קָֽבֶר׃ וְהוּא יִתֵּן לִי אֶת מְעָרַת הַמַּכְפֵּלָה שֶׁשַּׁיֶּכֶת לוֹ שֶׁבִּקְצֵה שָׂדֵהוּ, שֶׁיִּתֵּן לִי אוֹתָהּ לַאֲחֻזַּת קֶבֶר בֵּינֵיכֶם תְּמוּרַת כֶּסֶף בִּמְחִיר מָלֵא.
י וְעֶפְר֥וֹן יֹשֵׁ֖ב בְּת֣וֹךְ בְּנֵי־חֵ֑ת וַיַּ֩עַן֩ עֶפְר֨וֹן הַחִתִּ֤י אֶת־אַבְרָהָם֙ בְּאׇזְנֵ֣י בְנֵי־חֵ֔ת לְכֹ֛ל בָּאֵ֥י שַֽׁעַר־עִיר֖וֹ לֵאמֹֽר׃ וְעֶפְרוֹן הָיָה יוֹשֵׁב יַחַד עִם בְּנֵי חֵת, עָנָה עֶפְרוֹן הַחִתִּי לְאַבְרָהָם בְּאָזְנֵי בְּנֵי חֵת כְּדֵי שֶׁיִּשְׁמְעוּ כָּל הָעוֹבְרִים בְּשַׁעַר עִירוֹ, וְאָמַר:
יא לֹֽא־אֲדֹנִ֣י שְׁמָעֵ֔נִי הַשָּׂדֶה֙ נָתַ֣תִּי לָ֔ךְ וְהַמְּעָרָ֥ה אֲשֶׁר־בּ֖וֹ לְךָ֣ נְתַתִּ֑יהָ לְעֵינֵ֧י בְנֵי־עַמִּ֛י נְתַתִּ֥יהָ לָּ֖ךְ קְבֹ֥ר מֵתֶֽךָ׃ לֹא אֲדוֹנִי, שְׁמַע אוֹתִי, אֲנִי נוֹתֵן לְךָ בְּמַתָּנָה אֶת הַשָּׂדֶה, וְגַם אֶת הַמְּעָרָה שֶׁבָּהּ אֲנִי נוֹתֵן לְךָ – לְעֵינֵי בְּנֵי עַמִּי, קְבוֹר אֶת מֵתְךָ!
יב וַיִּשְׁתַּ֙חוּ֙ אַבְרָהָ֔ם לִפְנֵ֖י עַ֥ם הָאָֽרֶץ׃ אַבְרָהָם הִשְׁתַּחֲוָה לִפְנֵי הָעָם שֶׁגָּר בָּאָרֶץ.
יג וַיְדַבֵּ֨ר אֶל־עֶפְר֜וֹן בְּאׇזְנֵ֤י עַם־הָאָ֙רֶץ֙ לֵאמֹ֔ר אַ֛ךְ אִם־אַתָּ֥ה ל֖וּ שְׁמָעֵ֑נִי נָתַ֜תִּי כֶּ֤סֶף הַשָּׂדֶה֙ קַ֣ח מִמֶּ֔נִּי וְאֶקְבְּרָ֥ה אֶת־מֵתִ֖י שָֽׁמָּה׃ הוּא דִּבֵּר אֶל עֶפְרוֹן בְּאָזְנֵי הָעָם שֶׁגָּר בָּאָרֶץ וְאָמַר: אֲבָל אִם רְצוֹנְךָ לַעֲשׂוֹת לִי טוֹבָה הַלְוַאי שֶׁתִּשְׁמַע בְּקוֹלִי, הִנֵּה אֲנִי נוֹתֵן אֶת כֶּסֶף מְחִיר הַשָּׂדֶה – קַח מִמֶּנִּי! וְאֶקְבּוֹר שָׁם אֶת מֵתִי.
יד וַיַּ֧עַן עֶפְר֛וֹן אֶת־אַבְרָהָ֖ם לֵאמֹ֥ר לֽוֹ׃ עֶפְרוֹן עָנָה לְאַבְרָהָם וְאָמַר לוֹ:
טו אֲדֹנִ֣י שְׁמָעֵ֔נִי אֶ֩רֶץ֩ אַרְבַּ֨ע מֵאֹ֧ת שֶֽׁקֶל־כֶּ֛סֶף בֵּינִ֥י וּבֵֽינְךָ֖ מַה־הִ֑וא וְאֶת־מֵתְךָ֖ קְבֹֽר׃ אֲדוֹנִי שְׁמַע אוֹתִי, מַה כְּבָר חֲשׁוּבָה בֵּינִי וּבֵינְךָ אֶרֶץ שֶׁשָּׁוָה כֶּסֶף בְּמִשְׁקַל אַרְבַּע מֵאוֹת שֶׁקֶל? (5 קִילוֹ וְ640 גְּרַם), קְבוֹר כְּבָר אֶת מֵתְךָ!
טז וַיִּשְׁמַ֣ע אַבְרָהָם֮ אֶל־עֶפְרוֹן֒ וַיִּשְׁקֹ֤ל אַבְרָהָם֙ לְעֶפְרֹ֔ן אֶת־הַכֶּ֕סֶף אֲשֶׁ֥ר דִּבֶּ֖ר בְּאׇזְנֵ֣י בְנֵי־חֵ֑ת אַרְבַּ֤ע מֵאוֹת֙ שֶׁ֣קֶל כֶּ֔סֶף עֹבֵ֖ר לַסֹּחֵֽר׃ אַבְרָהָם הִסְכִּים לְדִבְרֵי עֶפְרוֹן, הוּא שָׁקַל אֶת הַכֶּסֶף שֶׁעֶפְרוֹן הִזְכִּיר בְּאָזְנֵי בְּנֵי חֵת – כֶּסֶף בְּמִשְׁקַל אַרְבַּע מֵאוֹת שֶׁקֶל שֶׁמִּתְקַבֵּל עַל יְדֵי כָּל סוֹחֵר, וְנָתַן לְעֶפְרוֹן.

שני[עריכה]

יז וַיָּ֣קׇם ׀ שְׂדֵ֣ה עֶפְר֗וֹן אֲשֶׁר֙ בַּמַּכְפֵּלָ֔ה אֲשֶׁ֖ר לִפְנֵ֣י מַמְרֵ֑א הַשָּׂדֶה֙ וְהַמְּעָרָ֣ה אֲשֶׁר־בּ֔וֹ וְכׇל־הָעֵץ֙ אֲשֶׁ֣ר בַּשָּׂדֶ֔ה אֲשֶׁ֥ר בְּכׇל־גְּבֻל֖וֹ סָבִֽיב׃ וְכָךְ נַעֲשָׂה שְׂדֵה עֶפְרוֹן שֶׁבְּעֵמֶק הַמַּכְפֵּלָה שֶׁמּוּל מַמְרֵא – הַשָּׂדֶה וְהַמְּעָרָה שֶׁבּוֹ וְכָל הָעֵצִים שֶׁבַּשָּׂדֶה שֶׁבְּכָל גְּבוּלוֹ סָבִיב,
יח לְאַבְרָהָ֥ם לְמִקְנָ֖ה לְעֵינֵ֣י בְנֵי־חֵ֑ת בְּכֹ֖ל בָּאֵ֥י שַֽׁעַר־עִירֽוֹ׃ רְכוּשׁ שֶׁל אַבְרָהָם לְעֵינֵי בְּנֵי חֵת בְּמַעֲמַד כָּל הָעוֹבְרִים בְּשַׁעַר עִירוֹ.
יט וְאַחֲרֵי־כֵן֩ קָבַ֨ר אַבְרָהָ֜ם אֶת־שָׂרָ֣ה אִשְׁתּ֗וֹ אֶל־מְעָרַ֞ת שְׂדֵ֧ה הַמַּכְפֵּלָ֛ה עַל־פְּנֵ֥י מַמְרֵ֖א הִ֣וא חֶבְר֑וֹן בְּאֶ֖רֶץ כְּנָֽעַן׃ וְאַחֲרֵי כֵן קָבַר אַבְרָהָם אֶת שָׂרָה אִשְׁתּוֹ בִּמְעָרַת שְׂדֵה הַמַּכְפֵּלָה שֶׁמּוּל מַמְרֵא – הִיא חֶבְרוֹן, בְּאֶרֶץ כְּנַעַן.
כ וַיָּ֨קׇם הַשָּׂדֶ֜ה וְהַמְּעָרָ֧ה אֲשֶׁר־בּ֛וֹ לְאַבְרָהָ֖ם לַאֲחֻזַּת־קָ֑בֶר מֵאֵ֖ת בְּנֵי־חֵֽת׃ הַשָּׂדֶה וְהַמְּעָרָה שֶׁבּוֹ נִקְנוּ לְאַבְרָהָם מִבְּנֵי חֵת לְצוֹרֶךְ אֲחֻזַּת קֶבֶר.
ס פָּרָשַׁת לְקִיחַת רִבְקָה
כד א וְאַבְרָהָ֣ם זָקֵ֔ן בָּ֖א בַּיָּמִ֑ים וַֽיהֹוָ֛ה בֵּרַ֥ךְ אֶת־אַבְרָהָ֖ם בַּכֹּֽל׃ וְאַבְרָהָם זָקֵן עָבְרוּ עָלָיו שָׁנִים רַבּוֹת, וַה' בֵּרֵךְ אֶת אַבְרָהָם בְּכָל טוֹב.
ב וַיֹּ֣אמֶר אַבְרָהָ֗ם אֶל־עַבְדּוֹ֙ זְקַ֣ן בֵּית֔וֹ הַמֹּשֵׁ֖ל בְּכׇל־אֲשֶׁר־ל֑וֹ שִֽׂים־נָ֥א יָדְךָ֖ תַּ֥חַת יְרֵכִֽי׃ אַבְרָהָם אָמַר לְעַבְדּוֹ מְנַהֵל בֵּיתוֹ הָאַחְרַאי עַל כָּל הַשַּׁיָּךְ לוֹ: בְּבַקָּשָׁה, שִׂים אֶת יָדְךָ מִתַּחַת לִירֵכִי כְּמִנְהַג הָעֲבָדִים שֶׁנִּשְׁבָּעִים לַאֲדוֹנָם.
ג וְאַשְׁבִּ֣יעֲךָ֔ בַּֽיהֹוָה֙ אֱלֹהֵ֣י הַשָּׁמַ֔יִם וֵֽאלֹהֵ֖י הָאָ֑רֶץ אֲשֶׁ֨ר לֹֽא־תִקַּ֤ח אִשָּׁה֙ לִבְנִ֔י מִבְּנוֹת֙ הַֽכְּנַעֲנִ֔י אֲשֶׁ֥ר אָנֹכִ֖י יוֹשֵׁ֥ב בְּקִרְבּֽוֹ׃ וְאַשְׁבִּיעַ אוֹתְךָ בְּשֵׁם ה' אֱלֹהֵי הַשָּׁמַיִם וֵאלֹהֵי הָאָרֶץ שֶׁלֹּא תִּקַּח אִשָּׁה לִבְנִי מִבְּנוֹת הָעָם הַכְּנַעֲנִי שֶׁאֲנִי גָּר בְּתוֹכוֹ.
ד כִּ֧י אֶל־אַרְצִ֛י וְאֶל־מוֹלַדְתִּ֖י תֵּלֵ֑ךְ וְלָקַחְתָּ֥ אִשָּׁ֖ה לִבְנִ֥י לְיִצְחָֽק׃ אֶלָּא תֵּלֵךְ לְאַרְצִי וּלְמִשְׁפַּחְתִּי, וּמִשָּׁם תִּקַּח אִשָּׁה לִבְנִי לְיִצְחָק.
ה וַיֹּ֤אמֶר אֵלָיו֙ הָעֶ֔בֶד אוּלַי֙ לֹא־תֹאבֶ֣ה הָֽאִשָּׁ֔ה לָלֶ֥כֶת אַחֲרַ֖י אֶל־הָאָ֣רֶץ הַזֹּ֑את הֶֽהָשֵׁ֤ב אָשִׁיב֙ אֶת־בִּנְךָ֔ אֶל־הָאָ֖רֶץ אֲשֶׁר־יָצָ֥אתָ מִשָּֽׁם׃ אָמַר לוֹ הָעֶבֶד: אוּלַי הָאִשָּׁה לֹא תִּרְצֶה לָלֶכֶת אַחֲרַי לָאָרֶץ הַזֹּאת, הַאִם אַחֲזִיר אֶת בִּנְךָ לָאָרֶץ שֶׁיָּצָאתָ מִשָּׁם?
ו וַיֹּ֥אמֶר אֵלָ֖יו אַבְרָהָ֑ם הִשָּׁ֣מֶר לְךָ֔ פֶּן־תָּשִׁ֥יב אֶת־בְּנִ֖י שָֽׁמָּה׃ אָמַר לוֹ אַבְרָהָם: הִזָּהֵר לְךָ שֶׁלֹּא תַּחֲזִיר אֶת בְּנִי לְשָׁם.
ז יְהֹוָ֣ה ׀ אֱלֹהֵ֣י הַשָּׁמַ֗יִם אֲשֶׁ֨ר לְקָחַ֜נִי מִבֵּ֣ית אָבִי֮ וּמֵאֶ֣רֶץ מֽוֹלַדְתִּי֒ וַאֲשֶׁ֨ר דִּבֶּר־לִ֜י וַאֲשֶׁ֤ר נִֽשְׁבַּֽע־לִי֙ לֵאמֹ֔ר לְזַ֨רְעֲךָ֔ אֶתֵּ֖ן אֶת־הָאָ֣רֶץ הַזֹּ֑את ה֗וּא יִשְׁלַ֤ח מַלְאָכוֹ֙ לְפָנֶ֔יךָ וְלָקַחְתָּ֥ אִשָּׁ֛ה לִבְנִ֖י מִשָּֽׁם׃ ה' אֱלֹהֵי הַשָּׁמַיִם שֶׁלָּקַח אוֹתִי מִבֵּית אָבִי וּמֵאֶרֶץ הֻלַּדְתִּי וְשֶׁדִּבֵּר אֵלַי וְשֶׁנִּשְׁבַּע לִי וְאָמַר: לְצֶאֱצָאֶיךָ אֶתֵּן אֶת הָאָרֶץ הַזֹּאת – הוּא יִשְׁלַח אֶת שְׁלִיחוֹ לְפָנֶיךָ לְהַצְלָחַתְךָ, וְתִקַּח מִשָּׁם אִשָּׁה לִבְנִי.
ח וְאִם־לֹ֨א תֹאבֶ֤ה הָֽאִשָּׁה֙ לָלֶ֣כֶת אַחֲרֶ֔יךָ וְנִקִּ֕יתָ מִשְּׁבֻעָתִ֖י זֹ֑את רַ֣ק אֶת־בְּנִ֔י לֹ֥א תָשֵׁ֖ב שָֽׁמָּה׃ וְאִם הָאִשָּׁה לֹא תִּרְצֶה לָלֶכֶת אַחֲרֶיךָ תִּהְיֶה פָּטוּר מֵהַשְּׁבוּעָה הַזֹּאת שֶׁלִּי, רַק אַל תַּחֲזִיר לְשָׁם אֶת בְּנִי.
ט וַיָּ֤שֶׂם הָעֶ֙בֶד֙ אֶת־יָד֔וֹ תַּ֛חַת יֶ֥רֶךְ אַבְרָהָ֖ם אֲדֹנָ֑יו וַיִּשָּׁ֣בַֽע ל֔וֹ עַל־הַדָּבָ֖ר הַזֶּֽה׃ הָעֶבֶד שָׂם אֶת יָדוֹ מִתַּחַת לַיָּרֵךְ שֶׁל אַבְרָהָם אֲדוֹנוֹ, וְנִשְׁבַּע לוֹ עַל הַדָּבָר הַזֶּה.

שלישי[עריכה]

י וַיִּקַּ֣ח הָ֠עֶ֠בֶד עֲשָׂרָ֨ה גְמַלִּ֜ים מִגְּמַלֵּ֤י אֲדֹנָיו֙ וַיֵּ֔לֶךְ וְכׇל־ט֥וּב אֲדֹנָ֖יו בְּיָד֑וֹ וַיָּ֗קׇם וַיֵּ֛לֶךְ אֶל־אֲרַ֥ם נַֽהֲרַ֖יִם אֶל־עִ֥יר נָחֽוֹר׃ הָעֶבֶד לָקַח עֲשָׂרָה גְּמַלִּים מֵהַגְּמַלִּים שֶׁל אֲדוֹנוֹ וְהָלַךְ כְּשֶׁבְּיָדוֹ מִכָּל טוּב שֶׁל אֲדוֹנוֹ, הוּא קָם וְהָלַךְ לַאֲרַם נַהֲרַיִם אֶל הָעִיר שֶׁל נָחוֹר.
יא וַיַּבְרֵ֧ךְ הַגְּמַלִּ֛ים מִח֥וּץ לָעִ֖יר אֶל־בְּאֵ֣ר הַמָּ֑יִם לְעֵ֣ת עֶ֔רֶב לְעֵ֖ת צֵ֥את הַשֹּׁאֲבֹֽת׃ הוּא הוֹשִׁיב אֶת הַגְּמַלִּים מִחוּץ לָעִיר לְיַד בְּאֵר הַמַּיִם לְעֵת עֶרֶב בִּזְמַן שֶׁהַבָּנוֹת יוֹצְאוֹת לִשְׁאוֹב.
יב וַיֹּאמַ֓ר ׀ יְהֹוָ֗ה אֱלֹהֵי֙ אֲדֹנִ֣י אַבְרָהָ֔ם הַקְרֵה־נָ֥א לְפָנַ֖י הַיּ֑וֹם וַעֲשֵׂה־חֶ֕סֶד עִ֖ם אֲדֹנִ֥י אַבְרָהָֽם׃ וְאָמַר: ה' אֱלֹהֵי אֲדוֹנִי אַבְרָהָם, הַזְמֵן לְפָנַי הַיּוֹם אֶת בַּקָּשָׁתִי, וַעֲשֵׂה חֶסֶד עִם אֲדוֹנִי אַבְרָהָם.
יג הִנֵּ֛ה אָנֹכִ֥י נִצָּ֖ב עַל־עֵ֣ין הַמָּ֑יִם וּבְנוֹת֙ אַנְשֵׁ֣י הָעִ֔יר יֹצְאֹ֖ת לִשְׁאֹ֥ב מָֽיִם׃ הִנֵּה אֲנִי עוֹמֵד עַל יַד מַעְיַן הַמַּיִם, וּבְנוֹת אַנְשֵׁי הָעִיר יוֹצְאוֹת לִשְׁאוֹב מַיִם.
יד וְהָיָ֣ה הַֽנַּעֲרָ֗ אֲשֶׁ֨ר אֹמַ֤ר אֵלֶ֙יהָ֙ הַטִּי־נָ֤א כַדֵּךְ֙ וְאֶשְׁתֶּ֔ה וְאָמְרָ֣ה שְׁתֵ֔ה וְגַם־גְּמַלֶּ֖יךָ אַשְׁקֶ֑ה אֹתָ֤הּ הֹכַ֙חְתָּ֙ לְעַבְדְּךָ֣ לְיִצְחָ֔ק וּבָ֣הּ אֵדַ֔ע כִּי־עָשִׂ֥יתָ חֶ֖סֶד עִם־אֲדֹנִֽי׃ וְכָךְ יִהְיֶה, הַנַּעֲרָה שֶׁאוֹמַר אֵלֶיהָ: הַטִּי בְּבַקָּשָׁה אֶת כַּדֵּךְ וְאֶשְׁתֶּה, וְהִיא תֹּאמַר: שְׁתֵה, וְגַם אֶת גְּמַלֶּיךָ אַשְׁקֶה! – אוֹתָהּ זִמַּנְתָּ לְעַבְדְּךָ לְיִצְחָק, וְאֵדַע שֶׁבָּהּ עָשִׂיתָ חֶסֶד עִם אֲדוֹנִי.
טו וַֽיְהִי־ה֗וּא טֶ֘רֶם֮ כִּלָּ֣ה לְדַבֵּר֒ וְהִנֵּ֧ה רִבְקָ֣ה יֹצֵ֗את אֲשֶׁ֤ר יֻלְּדָה֙ לִבְתוּאֵ֣ל בֶּן־מִלְכָּ֔ה אֵ֥שֶׁת נָח֖וֹר אֲחִ֣י אַבְרָהָ֑ם וְכַדָּ֖הּ עַל־שִׁכְמָֽהּ׃ לִפְנֵי שֶׁהוּא סִיֵּם לְדַבֵּר, הִנֵּה יוֹצֵאת רִבְקָה – שֶׁנּוֹלְדָה לִבְתוּאֵל הַבֵּן שֶׁל מִלְכָּה אֵשֶׁת נָחוֹר אֲחִיו שֶׁל אַבְרָהָם, וְהַכַּד שֶׁלָּהּ עַל כְּתֵפָהּ.
טז וְהַֽנַּעֲרָ֗ טֹבַ֤ת מַרְאֶה֙ מְאֹ֔ד בְּתוּלָ֕ה וְאִ֖ישׁ לֹ֣א יְדָעָ֑הּ וַתֵּ֣רֶד הָעַ֔יְנָה וַתְּמַלֵּ֥א כַדָּ֖הּ וַתָּֽעַל׃ וְהַנַּעֲרָה יָפָה מְאוֹד, בְּתוּלָה – שֶׁאִישׁ לֹא שָׁכַב אִתָּהּ, הִיא יָרְדָה לַמַּעְיָן, מִלְּאָה אֶת הַכַּד שֶׁלָּהּ, וְעָלְתָה.
יז וַיָּ֥רׇץ הָעֶ֖בֶד לִקְרָאתָ֑הּ וַיֹּ֕אמֶר הַגְמִיאִ֥ינִי נָ֛א מְעַט־מַ֖יִם מִכַּדֵּֽךְ׃ הָעֶבֶד רָץ לִקְרָאתָהּ, וְאָמַר לָהּ: תְּנִי לִי בְּבַקָּשָׁה לִשְׁתּוֹת מְעַט מַיִם מִכַּדֵּךְ!
יח וַתֹּ֖אמֶר שְׁתֵ֣ה אֲדֹנִ֑י וַתְּמַהֵ֗ר וַתֹּ֧רֶד כַּדָּ֛הּ עַל־יָדָ֖הּ וַתַּשְׁקֵֽהוּ׃ אָמְרָה: שְׁתֵה אֲדוֹנִי! וּמִהֲרָה וְהוֹרִידָה אֶת הַכַּד שֶׁלָּהּ אֶל יָדָהּ, וְנָתְנָה לוֹ לִשְׁתּוֹת.
יט וַתְּכַ֖ל לְהַשְׁקֹת֑וֹ וַתֹּ֗אמֶר גַּ֤ם לִגְמַלֶּ֙יךָ֙ אֶשְׁאָ֔ב עַ֥ד אִם־כִּלּ֖וּ לִשְׁתֹּֽת׃ הִיא סִיְּמָה לָתֵת לוֹ לִשְׁתּוֹת, וְאָמְרָה: אֶשְׁאַב גַּם לַגְּמַלִּים שֶׁלְּךָ, עַד שֶׁיְּסַיְּמוּ לִשְׁתּוֹת!
כ וַתְּמַהֵ֗ר וַתְּעַ֤ר כַּדָּהּ֙ אֶל־הַשֹּׁ֔קֶת וַתָּ֥רׇץ ע֛וֹד אֶֽל־הַבְּאֵ֖ר לִשְׁאֹ֑ב וַתִּשְׁאַ֖ב לְכׇל־גְּמַלָּֽיו׃ הִיא מִהֲרָה וְרוֹקְנָה אֶת הַכַּד שֶׁלָּהּ לַשּׁוֹקֶת, וְרָצָה עוֹד לַבְּאֵר לִשְׁאוֹב, וְשָׁאֲבָה לְכָל הַגְּמַלִּים שֶׁלּוֹ.
כא וְהָאִ֥ישׁ מִשְׁתָּאֵ֖ה לָ֑הּ מַחֲרִ֕ישׁ לָדַ֗עַת הַֽהִצְלִ֧יחַ יְהֹוָ֛ה דַּרְכּ֖וֹ אִם־לֹֽא׃ וְהָאִישׁ מִשְׁתּוֹמֵם עָלֶיהָ, וְשׁוֹתֵק, לָדַעַת הַאִם ה' הִצְלִיחַ אֶת דַּרְכּוֹ, אוֹ לֹא.
כב וַיְהִ֗י כַּאֲשֶׁ֨ר כִּלּ֤וּ הַגְּמַלִּים֙ לִשְׁתּ֔וֹת וַיִּקַּ֤ח הָאִישׁ֙ נֶ֣זֶם זָהָ֔ב בֶּ֖קַע מִשְׁקָל֑וֹ וּשְׁנֵ֤י צְמִידִים֙ עַל־יָדֶ֔יהָ עֲשָׂרָ֥ה זָהָ֖ב מִשְׁקָלָֽם׃ כְּשֶׁהַגְּמַלִּים סִיְּמוּ לִשְׁתּוֹת, הָאִישׁ לָקַח נֶזֶם זָהָב שֶׁמִּשְׁקָלוֹ חֲצִי שֶׁקֶל (כְּ7 גְּרַם) וְנָתַן לָהּ, וְשָׂם עַל יָדֶיהָ שְׁנֵי צְמִידִים שֶׁמִּשְׁקָלָם עֲשָׂרָה שִׁקְלֵי זָהָב (141 גְּרַם).
כג וַיֹּ֙אמֶר֙ בַּת־מִ֣י אַ֔תְּ הַגִּ֥ידִי נָ֖א לִ֑י הֲיֵ֧שׁ בֵּית־אָבִ֛יךְ מָק֥וֹם לָ֖נוּ לָלִֽין׃ וְאָמַר: הַבַּת שֶׁל מִי אַתְּ? הַגִּידִי לִי בְּבַקָּשָׁה, הַאִם יֵשׁ לָנוּ מָקוֹם בְּבֵית אָבִיךְ לִישׁוֹן הַלַּיְלָה?
כד וַתֹּ֣אמֶר אֵלָ֔יו בַּת־בְּתוּאֵ֖ל אָנֹ֑כִי בֶּן־מִלְכָּ֕ה אֲשֶׁ֥ר יָלְדָ֖ה לְנָחֽוֹר׃ אָמְרָה לוֹ: אֲנִי בַּת שֶׁל בְּתוּאֵל בֶּן מִלְכָּה שֶׁיָּלְדָה אוֹתוֹ לְנָחוֹר.
כה וַתֹּ֣אמֶר אֵלָ֔יו גַּם־תֶּ֥בֶן גַּם־מִסְפּ֖וֹא רַ֣ב עִמָּ֑נוּ גַּם־מָק֖וֹם לָלֽוּן׃ אָמְרָה לוֹ: יֵשׁ אֶצְלֵנוּ גַּם הַרְבֵּה קַשׁ כְּדֵי שֶׁיִּשְׁכְּבוּ עָלָיו הַגְּמַלִּים, וְגַם הַרְבֵּה מִסְפּוֹא לְמַאֲכָלָם, וְגַם מָקוֹם לָאֲנָשִׁים לִישׁוֹן בַּלַּיְלָה.
כו וַיִּקֹּ֣ד הָאִ֔ישׁ וַיִּשְׁתַּ֖חוּ לַֽיהֹוָֽה׃ הָאִישׁ קָד וְהִשְׁתַּחֲוָה לַה'.

רביעי[עריכה]

כז וַיֹּ֗אמֶר בָּר֤וּךְ יְהֹוָה֙ אֱלֹהֵי֙ אֲדֹנִ֣י אַבְרָהָ֔ם אֲ֠שֶׁ֠ר לֹֽא־עָזַ֥ב חַסְדּ֛וֹ וַאֲמִתּ֖וֹ מֵעִ֣ם אֲדֹנִ֑י אָנֹכִ֗י בַּדֶּ֙רֶךְ֙ נָחַ֣נִי יְהֹוָ֔ה בֵּ֖ית אֲחֵ֥י אֲדֹנִֽי׃ וְאָמַר: בָּרוּךְ ה' אֱלֹהֵי אֲדוֹנִי אַבְרָהָם שֶׁלֹּא הִפְסִיק אֶת חַסְדּוֹ וְאֶת הַבְטָחָתוֹ הָאֲמִתִּית מֵאֲדוֹנִי שֶׁבִּזְכוּתוֹ ה' הוֹבִיל אוֹתִי בַּדֶּרֶךְ לְבֵית אֲחֵי אֲדוֹנִי.
כח וַתָּ֙רׇץ֙ הַֽנַּעֲרָ֔ וַתַּגֵּ֖ד לְבֵ֣ית אִמָּ֑הּ כַּדְּבָרִ֖ים הָאֵֽלֶּה׃ הַנַּעֲרָה רָצָה, וְסִפְּרָה לְמִשְׁפַּחַת הָאִמָּא שֶׁלָּהּ אֶת הַדְּבָרִים הָאֵלֶּה.
כט וּלְרִבְקָ֥ה אָ֖ח וּשְׁמ֣וֹ לָבָ֑ן וַיָּ֨רׇץ לָבָ֧ן אֶל־הָאִ֛ישׁ הַח֖וּצָה אֶל־הָעָֽיִן׃ וּלְרִבְקָה אָח וּשְׁמוֹ לָבָן, לָבָן רָץ אֶל הָאִישׁ הַחוּצָה אֶל הַמַּעְיָן.
ל וַיְהִ֣י ׀ כִּרְאֹ֣ת אֶת־הַנֶּ֗זֶם וְֽאֶת־הַצְּמִדִים֮ עַל־יְדֵ֣י אֲחֹתוֹ֒ וּכְשׇׁמְע֗וֹ אֶת־דִּבְרֵ֞י רִבְקָ֤ה אֲחֹתוֹ֙ לֵאמֹ֔ר כֹּֽה־דִבֶּ֥ר אֵלַ֖י הָאִ֑ישׁ וַיָּבֹא֙ אֶל־הָאִ֔ישׁ וְהִנֵּ֛ה עֹמֵ֥ד עַל־הַגְּמַלִּ֖ים עַל־הָעָֽיִן׃ כְּשֶׁרָאָה אֶת הַנֶּזֶם וְאֶת הַצְּמִידִים עַל יְדֵי אֲחוֹתוֹ, וּכְשֶׁשָּׁמַע אֶת דִּבְרֵי רִבְקָה אֲחוֹתוֹ שֶׁאָמְרָה: "כָּךְ וְכָךְ אָמַר לִי הָאִישׁ" – הוּא בָּא אֶל הָאִישׁ, וְהִנֵּה הוּא עוֹמֵד לְיַד הַגְּמַלִּים לְיַד הַמַּעְיָן.
לא וַיֹּ֕אמֶר בּ֖וֹא בְּר֣וּךְ יְהֹוָ֑ה לָ֤מָּה תַעֲמֹד֙ בַּח֔וּץ וְאָנֹכִי֙ פִּנִּ֣יתִי הַבַּ֔יִת וּמָק֖וֹם לַגְּמַלִּֽים׃ אָמַר לוֹ: בּוֹא הַמְּבוֹרָךְ מֵה'! לָמָּה תַּעֲמוֹד בַּחוּץ? הֲרֵי אֲנִי פִּנִּיתִי עֲבוּרְכֶם אֶת הַבַּיִת וְהֵכַנְתִּי מָקוֹם לַגְּמַלִּים!
לב וַיָּבֹ֤א הָאִישׁ֙ הַבַּ֔יְתָה וַיְפַתַּ֖ח הַגְּמַלִּ֑ים וַיִּתֵּ֨ן תֶּ֤בֶן וּמִסְפּוֹא֙ לַגְּמַלִּ֔ים וּמַ֙יִם֙ לִרְחֹ֣ץ רַגְלָ֔יו וְרַגְלֵ֥י הָאֲנָשִׁ֖ים אֲשֶׁ֥ר אִתּֽוֹ׃ הָאִישׁ בָּא הַבַּיְתָה פָּתַח אֶת זְמָמֵי הַגְּמַלִּים, שָׂם קַשׁ וּמִסְפּוֹא לַגְּמַלִּים וּמַיִם לִרְחוֹץ אֶת רַגְלָיו וְאֶת רַגְלֵי הָאֲנָשִׁים שֶׁאִתּוֹ.
לג ויישם וַיּוּשַׂ֤ם לְפָנָיו֙ לֶאֱכֹ֔ל וַיֹּ֙אמֶר֙ לֹ֣א אֹכַ֔ל עַ֥ד אִם־דִּבַּ֖רְתִּי דְּבָרָ֑י וַיֹּ֖אמֶר דַּבֵּֽר׃ שָׂמוּ לְפָנָיו אוֹכֶל לֶאֱכוֹל, אָמַר לָהֶם: לֹא אוֹכַל עַד שֶׁאוֹמַר אֶת דְּבָרַי! אָמַר לוֹ: דַּבֵּר!
לד וַיֹּאמַ֑ר עֶ֥בֶד אַבְרָהָ֖ם אָנֹֽכִי׃ אָמַר: אֲנִי עֶבֶד שֶׁל אַבְרָהָם.
לה וַיהֹוָ֞ה בֵּרַ֧ךְ אֶת־אֲדֹנִ֛י מְאֹ֖ד וַיִּגְדָּ֑ל וַיִּתֶּן־ל֞וֹ צֹ֤אן וּבָקָר֙ וְכֶ֣סֶף וְזָהָ֔ב וַעֲבָדִם֙ וּשְׁפָחֹ֔ת וּגְמַלִּ֖ים וַחֲמֹרִֽים׃ וַה' בֵּרֵךְ אֶת אֲדוֹנִי מְאוֹד וְהוּא הִתְעַשֵּׁר, וְנָתַן לוֹ צֹאן וּבָקָר וְכֶסֶף וְזָהָב וַעֲבָדִים וּשְׁפָחוֹת וּגְמַלִּים וַחֲמוֹרִים.
לו וַתֵּ֡לֶד שָׂרָה֩ אֵ֨שֶׁת אֲדֹנִ֥י בֵן֙ לַֽאדֹנִ֔י אַחֲרֵ֖י זִקְנָתָ֑הּ וַיִּתֶּן־ל֖וֹ אֶת־כׇּל־אֲשֶׁר־לֽוֹ׃ שָׂרָה אֵשֶׁת אֲדוֹנִי יָלְדָה בֵּן לַאדֹנִי אַחֲרֵי זִקְנָתָהּ, וְהוּא נָתַן לוֹ אֶת כָּל מַה שֶּׁיֵּשׁ לוֹ.
לז וַיַּשְׁבִּעֵ֥נִי אֲדֹנִ֖י לֵאמֹ֑ר לֹא־תִקַּ֤ח אִשָּׁה֙ לִבְנִ֔י מִבְּנוֹת֙ הַֽכְּנַעֲנִ֔י אֲשֶׁ֥ר אָנֹכִ֖י יֹשֵׁ֥ב בְּאַרְצֽוֹ׃ אֲדוֹנִי הִשְׁבִּיעַ אוֹתִי וְאָמַר: אַל תִּקַּח לִבְנִי אִשָּׁה מִבְּנוֹת הָעָם הַכְּנַעֲנִי שֶׁאֲנִי גָּר בְּאַרְצוֹ!
לח אִם־לֹ֧א אֶל־בֵּית־אָבִ֛י תֵּלֵ֖ךְ וְאֶל־מִשְׁפַּחְתִּ֑י וְלָקַחְתָּ֥ אִשָּׁ֖ה לִבְנִֽי׃ אֶלָּא לֵךְ אֶל בֵּית אָבִי וְאֶל מִשְׁפַּחְתִּי וְתִקַּח אִשָּׁה לִבְנִי.
לט וָאֹמַ֖ר אֶל־אֲדֹנִ֑י אֻלַ֛י לֹא־תֵלֵ֥ךְ הָאִשָּׁ֖ה אַחֲרָֽי׃ אָמַרְתִּי לַאֲדוֹנִי: אוּלַי הָאִשָּׁה לֹא תֵּלֵךְ אַחֲרַי!
מ וַיֹּ֖אמֶר אֵלָ֑י יְהֹוָ֞ה אֲשֶׁר־הִתְהַלַּ֣כְתִּי לְפָנָ֗יו יִשְׁלַ֨ח מַלְאָכ֤וֹ אִתָּךְ֙ וְהִצְלִ֣יחַ דַּרְכֶּ֔ךָ וְלָקַחְתָּ֤ אִשָּׁה֙ לִבְנִ֔י מִמִּשְׁפַּחְתִּ֖י וּמִבֵּ֥ית אָבִֽי׃ וְאָמַר לִי: ה' שֶׁהִתְנָהַגְתִּי בִּדְרָכָיו יִשְׁלַח אֶת שְׁלִיחוֹ אִתְּךָ וְיַעֲשֶׂה שֶׁתַּצְלִיחַ בְּדַרְכְּךָ, וְתִקַּח לִבְנִי אִשָּׁה מִמִּשְׁפַּחְתִּי וּמִבֵּית אָבִי.
מא אָ֤ז תִּנָּקֶה֙ מֵאָ֣לָתִ֔י כִּ֥י תָב֖וֹא אֶל־מִשְׁפַּחְתִּ֑י וְאִם־לֹ֤א יִתְּנוּ֙ לָ֔ךְ וְהָיִ֥יתָ נָקִ֖י מֵאָלָתִֽי׃ וְרַק אָז תִּהְיֶה פָּטוּר מִשְּׁבוּעָתִי – כְּשֶׁתָּבוֹא לְמִשְׁפַּחְתִּי, וְאִם לֹא יִתְּנוּ לְךָ – אָז תִּהְיֶה פָּטוּר מִשְּׁבוּעָתִי.
מב וָאָבֹ֥א הַיּ֖וֹם אֶל־הָעָ֑יִן וָאֹמַ֗ר יְהֹוָה֙ אֱלֹהֵי֙ אֲדֹנִ֣י אַבְרָהָ֔ם אִם־יֶשְׁךָ־נָּא֙ מַצְלִ֣יחַ דַּרְכִּ֔י אֲשֶׁ֥ר אָנֹכִ֖י הֹלֵ֥ךְ עָלֶֽיהָ׃ בָּאתִי הַיּוֹם לַמַּעְיָן, וְאָמַרְתִּי: ה' אֱלֹהֵי אֲדוֹנִי אַבְרָהָם! אִם בִּרְצוֹנְךָ לַעֲשׂוֹת שֶׁאַצְלִיחַ בְּדַרְכִּי שֶׁאֲנִי הוֹלֵךְ בָּהּ,
מג הִנֵּ֛ה אָנֹכִ֥י נִצָּ֖ב עַל־עֵ֣ין הַמָּ֑יִם וְהָיָ֤ה הָֽעַלְמָה֙ הַיֹּצֵ֣את לִשְׁאֹ֔ב וְאָמַרְתִּ֣י אֵלֶ֔יהָ הַשְׁקִֽינִי־נָ֥א מְעַט־מַ֖יִם מִכַּדֵּֽךְ׃ הִנֵּה אֲנִי עוֹמֵד לְיַד מַעְיַן הַמַּיִם, וְכָךְ יִהְיֶה: הַנַּעֲרָה שֶׁיּוֹצֵאת לִשְׁאוֹב וְאוֹמַר לָהּ: תְּנִי לִי בְּבַקָּשָׁה לִשְׁתּוֹת מְעַט מַיִם מִכַּדֵּךְ!
מד וְאָמְרָ֤ה אֵלַי֙ גַּם־אַתָּ֣ה שְׁתֵ֔ה וְגַ֥ם לִגְמַלֶּ֖יךָ אֶשְׁאָ֑ב הִ֣וא הָֽאִשָּׁ֔ה אֲשֶׁר־הֹכִ֥יחַ יְהֹוָ֖ה לְבֶן־אֲדֹנִֽי׃ וְהִיא תֹּאמַר אֵלַי: גַּם אַתָּה שְׁתֵה, וְגַם לִגְמַלֶּיךָ אֶשְׁאַב! – הִיא הָאִשָּׁה שֶׁה' זִמֵּן לַבֵּן שֶׁל אֲדוֹנִי!
מה אֲנִי֩ טֶ֨רֶם אֲכַלֶּ֜ה לְדַבֵּ֣ר אֶל־לִבִּ֗י וְהִנֵּ֨ה רִבְקָ֤ה יֹצֵאת֙ וְכַדָּ֣הּ עַל־שִׁכְמָ֔הּ וַתֵּ֥רֶד הָעַ֖יְנָה וַתִּשְׁאָ֑ב וָאֹמַ֥ר אֵלֶ֖יהָ הַשְׁקִ֥ינִי נָֽא׃ לִפְנֵי שֶׁאֲנִי סִיַּמְתִּי לְהִתְפַּלֵּל מִתּוֹךְ לִבִּי, וְהִנֵּה רִבְקָה יוֹצֵאת וְהַכַּד שֶׁלָּהּ עַל כְּתֵפָהּ וְהִיא יָרְדָה לַמַּעְיָן וְשָׁאֲבָה, אָמַרְתִּי לָהּ: תְּנִי לִי לִשְׁתּוֹת בְּבַקָּשָׁה!
מו וַתְּמַהֵ֗ר וַתּ֤וֹרֶד כַּדָּהּ֙ מֵֽעָלֶ֔יהָ וַתֹּ֣אמֶר שְׁתֵ֔ה וְגַם־גְּמַלֶּ֖יךָ אַשְׁקֶ֑ה וָאֵ֕שְׁתְּ וְגַ֥ם הַגְּמַלִּ֖ים הִשְׁקָֽתָה׃ הִיא מִהֲרָה וְהוֹרִידָה אֶת הַכַּד שֶׁלָּהּ מֵעָלֶיהָ, וְאָמְרָה: שְׁתֵה, וְגַם לִגְמַלֶּיךָ אֶתֵּן לִשְׁתּוֹת! שָׁתִיתִי וְהִיא נָתְנָה גַּם לַגְּמַלִּים לִשְׁתּוֹת.
מז וָאֶשְׁאַ֣ל אֹתָ֗הּ וָאֹמַר֮ בַּת־מִ֣י אַתְּ֒ וַתֹּ֗אמֶר בַּת־בְּתוּאֵל֙ בֶּן־נָח֔וֹר אֲשֶׁ֥ר יָֽלְדָה־לּ֖וֹ מִלְכָּ֑ה וָאָשִׂ֤ם הַנֶּ֙זֶם֙ עַל־אַפָּ֔הּ וְהַצְּמִידִ֖ים עַל־יָדֶֽיהָ׃ שָׁאַלְתִּי אוֹתָהּ וְאָמַרְתִּי: הַבַּת שֶׁל מִי אַתְּ? אָמְרָה: הַבַּת שֶׁל בְּתוּאֵל הַבֵּן שֶׁל נָחוֹר – שֶׁמִּלְכָּה יָלְדָה לוֹ, שַׂמְתִּי אֶת הַנֶּזֶם עַל פָּנֶיהָ וְאֶת הַצְּמִידִים עַל יָדֶיהָ.
מח וָאֶקֹּ֥ד וָֽאֶשְׁתַּחֲוֶ֖ה לַיהֹוָ֑ה וָאֲבָרֵ֗ךְ אֶת־יְהֹוָה֙ אֱלֹהֵי֙ אֲדֹנִ֣י אַבְרָהָ֔ם אֲשֶׁ֤ר הִנְחַ֙נִי֙ בְּדֶ֣רֶךְ אֱמֶ֔ת לָקַ֛חַת אֶת־בַּת־אֲחִ֥י אֲדֹנִ֖י לִבְנֽוֹ׃ קַדְתִּי וְהִשְׁתַּחֲוֵיתִי לַה', וּבֵרַכְתִּי אֶת ה' אֱלֹהֵי אֲדוֹנִי אַבְרָהָם, שֶׁהוֹבִיל אוֹתִי בְּדֶרֶךְ נְכוֹנָה – לָקַחַת אֶת הַבַּת שֶׁל אֲחִיו שֶׁל אֲדוֹנִי לִבְנוֹ.
מט וְ֠עַתָּ֠ה אִם־יֶשְׁכֶ֨ם עֹשִׂ֜ים חֶ֧סֶד וֶֽאֱמֶ֛ת אֶת־אֲדֹנִ֖י הַגִּ֣ידוּ לִ֑י וְאִם־לֹ֕א הַגִּ֣ידוּ לִ֔י וְאֶפְנֶ֥ה עַל־יָמִ֖ין א֥וֹ עַל־שְׂמֹֽאל׃ וְעַכְשָׁיו, אִם אַתֶּם עוֹשִׂים חֶסֶד אֲמִתִּי עִם אֲדוֹנִי, הַגִּידוּ לִי, וְאִם לֹא, הַגִּידוּ לִי וְאֶפְנֶה יָמִינָה אוֹ שְׂמֹאלָה לְחַפֵּשׂ אִשָּׁה אַחֶרֶת.
נ וַיַּ֨עַן לָבָ֤ן וּבְתוּאֵל֙ וַיֹּ֣אמְר֔וּ מֵיְהֹוָ֖ה יָצָ֣א הַדָּבָ֑ר לֹ֥א נוּכַ֛ל דַּבֵּ֥ר אֵלֶ֖יךָ רַ֥ע אוֹ־טֽוֹב׃ עָנוּ לָבָן וּבְתוּאֵל, וְאָמְרוּ: מֵה' יָצָא הַדָּבָר! לֹא נוּכַל לְדַבֵּר אֵלֶיךָ נֶגֶד גְּזֵירָתוֹ אוֹ בַּעֲדָהּ.
נא הִנֵּֽה־רִבְקָ֥ה לְפָנֶ֖יךָ קַ֣ח וָלֵ֑ךְ וּתְהִ֤י אִשָּׁה֙ לְבֶן־אֲדֹנֶ֔יךָ כַּאֲשֶׁ֖ר דִּבֶּ֥ר יְהֹוָֽה׃ הִנֵּה רִבְקָה לְפָנֶיךָ קַח וְלֵךְ, וְהִיא תִּהְיֶה אִשָּׁה לַבֵּן שֶׁל אֲדוֹנְךָ כְּמוֹ שֶׁה' דִּבֵּר.
נב וַיְהִ֕י כַּאֲשֶׁ֥ר שָׁמַ֛ע עֶ֥בֶד אַבְרָהָ֖ם אֶת־דִּבְרֵיהֶ֑ם וַיִּשְׁתַּ֥חוּ אַ֖רְצָה לַֽיהֹוָֽה׃ כְּשֶׁשָּׁמַע עֶבֶד אַבְרָהָם אֶת דִּבְרֵיהֶם הוּא הִשְׁתַּחֲוָה עַל הָאָרֶץ לַה'.

חמישי[עריכה]

נג וַיּוֹצֵ֨א הָעֶ֜בֶד כְּלֵי־כֶ֨סֶף וּכְלֵ֤י זָהָב֙ וּבְגָדִ֔ים וַיִּתֵּ֖ן לְרִבְקָ֑ה וּמִ֨גְדָּנֹ֔ת נָתַ֥ן לְאָחִ֖יהָ וּלְאִמָּֽהּ׃ הָעֶבֶד הוֹצִיא כְּלֵי כֶּסֶף וּכְלֵי זָהָב וּבְגָדִים וְנָתַן לְרִבְקָה, וְנָתַן מַתָּנוֹת לְאָחִיהָ וּלְאִמָּהּ.
נד וַיֹּאכְל֣וּ וַיִּשְׁתּ֗וּ ה֛וּא וְהָאֲנָשִׁ֥ים אֲשֶׁר־עִמּ֖וֹ וַיָּלִ֑ינוּ וַיָּק֣וּמוּ בַבֹּ֔קֶר וַיֹּ֖אמֶר שַׁלְּחֻ֥נִי לַֽאדֹנִֽי׃ הוּא וְהָאֲנָשִׁים שֶׁאִתּוֹ אָכְלוּ וְשָׁתוּ וְיָשְׁנוּ בַּלַּיְלָה, הֵם קָמוּ בַּבּוֹקֶר וְהוּא אָמַר: שַׁחְרְרוּ אוֹתִי לַאֲדוֹנִי.
נה וַיֹּ֤אמֶר אָחִ֙יהָ֙ וְאִמָּ֔הּ תֵּשֵׁ֨ב הַנַּעֲרָ֥ אִתָּ֛נוּ יָמִ֖ים א֣וֹ עָשׂ֑וֹר אַחַ֖ר תֵּלֵֽךְ׃ אָמְרוּ לוֹ אָחִיהָ וְאִמָּהּ: הַנַּעֲרָה תָּגוּר אִתָּנוּ עוֹד שָׁנָה אוֹ עֲשָׂרָה חֳדָשִׁים, וְאַחַר כָּךְ הִיא תֵּלֵךְ.
נו וַיֹּ֤אמֶר אֲלֵהֶם֙ אַל־תְּאַחֲר֣וּ אֹתִ֔י וַֽיהֹוָ֖ה הִצְלִ֣יחַ דַּרְכִּ֑י שַׁלְּח֕וּנִי וְאֵלְכָ֖ה לַֽאדֹנִֽי׃ אָמַר לָהֶם: אַל תְּעַכְּבוּ אוֹתִי הֲרֵי ה' עָשָׂה שֶׁאַצְלִיחַ בְּדַרְכִּי, שַׁחְרְרוּ אוֹתִי וְאֵלֵךְ לַאֲדוֹנִי!
נז וַיֹּאמְר֖וּ נִקְרָ֣א לַֽנַּעֲרָ֑ וְנִשְׁאֲלָ֖ה אֶת־פִּֽיהָ׃ אָמְרוּ לוֹ: נִקְרָא לַנַּעֲרָה וְנִשְׁאַל אוֹתָהּ מַה הִיא אוֹמֶרֶת.
נח וַיִּקְרְא֤וּ לְרִבְקָה֙ וַיֹּאמְר֣וּ אֵלֶ֔יהָ הֲתֵלְכִ֖י עִם־הָאִ֣ישׁ הַזֶּ֑ה וַתֹּ֖אמֶר אֵלֵֽךְ׃ הֵם קָרְאוּ לְרִבְקָה וְאָמְרוּ לָהּ: הַאִם תֵּלְכִי עִם הָאִישׁ הַזֶּה? הִיא אָמְרָה: אֵלֵךְ.
נט וַֽיְשַׁלְּח֛וּ אֶת־רִבְקָ֥ה אֲחֹתָ֖ם וְאֶת־מֵנִקְתָּ֑הּ וְאֶת־עֶ֥בֶד אַבְרָהָ֖ם וְאֶת־אֲנָשָֽׁיו׃ הֵם שִׁלְּחוּ אֶת רִבְקָה אֲחוֹתָם וְאֶת שִׁפְחָתָהּ שֶׁהָיְתָה הַמֵּינֶקֶת שֶׁלָּהּ, וְאֶת עֶבֶד אַבְרָהָם וְאֶת הָאֲנָשִׁים שֶׁאִתּוֹ.
ס וַיְבָרְﬞכ֤וּ אֶת־רִבְקָה֙ וַיֹּ֣אמְרוּ לָ֔הּ אֲחֹתֵ֕נוּ אַ֥תְּ הֲיִ֖י לְאַלְפֵ֣י רְבָבָ֑ה וְיִירַ֣שׁ זַרְעֵ֔ךְ אֵ֖ת שַׁ֥עַר שֹׂנְאָֽיו׃ הֵם בֵּרְכוּ אֶת רִבְקָה, וְאָמְרוּ לָהּ: אֲחוֹתֵנוּ! יִהְיוּ לָךְ אַלְפֵי רְבָבוֹת צֶאֱצָאִים, וְיִכְבְּשׁוּ צֶאֱצָאַיִךְ אֶת עָרֵי שׂוֹנְאֵיהֶם!
סא וַתָּ֨קׇם רִבְקָ֜ה וְנַעֲרֹתֶ֗יהָ וַתִּרְכַּ֙בְנָה֙ עַל־הַגְּמַלִּ֔ים וַתֵּלַ֖כְנָה אַחֲרֵ֣י הָאִ֑ישׁ וַיִּקַּ֥ח הָעֶ֛בֶד אֶת־רִבְקָ֖ה וַיֵּלַֽךְ׃ רִבְקָה וְנַעֲרוֹתֶיהָ קָמוּ וְרָכְבוּ עַל הַגְּמַלִּים, וְהָלְכוּ אַחֲרֵי הָאִישׁ, הָעֶבֶד לָקַח אֶת רִבְקָה וְהָלַךְ.
סב וְיִצְחָק֙ בָּ֣א מִבּ֔וֹא בְּאֵ֥ר לַחַ֖י רֹאִ֑י וְה֥וּא יוֹשֵׁ֖ב בְּאֶ֥רֶץ הַנֶּֽגֶב׃ וְיִצְחָק בָּא מֵהֲלִיכָתוֹ אֶל בְּאֵר לַחַי רוֹאִי, וְהוּא הָיָה גָּר בְּאֶרֶץ הַנֶּגֶב.
סג וַיֵּצֵ֥א יִצְחָ֛ק לָשׂ֥וּחַ בַּשָּׂדֶ֖ה לִפְנ֣וֹת עָ֑רֶב וַיִּשָּׂ֤א עֵינָיו֙ וַיַּ֔רְא וְהִנֵּ֥ה גְמַלִּ֖ים בָּאִֽים׃ יִצְחָק יָצָא לְהִתְפַּלֵּל בַּשָּׂדֶה לִפְנוֹת עֶרֶב, הוּא הֵרִים אֶת מַבָּטוֹ, וְרָאָה וְהִנֵּה גְּמַלִּים בָּאִים.
סד וַתִּשָּׂ֤א רִבְקָה֙ אֶת־עֵינֶ֔יהָ וַתֵּ֖רֶא אֶת־יִצְחָ֑ק וַתִּפֹּ֖ל מֵעַ֥ל הַגָּמָֽל׃ רִבְקָה הֵרִימָה אֶת מַבָּטָהּ וְרָאֲתָה אֶת יִצְחָק וְהִנְמִיכָה אֶת עַצְמָה מֵעַל הַגָּמָל.
סה וַתֹּ֣אמֶר אֶל־הָעֶ֗בֶד מִֽי־הָאִ֤ישׁ הַלָּזֶה֙ הַהֹלֵ֤ךְ בַּשָּׂדֶה֙ לִקְרָאתֵ֔נוּ וַיֹּ֥אמֶר הָעֶ֖בֶד ה֣וּא אֲדֹנִ֑י וַתִּקַּ֥ח הַצָּעִ֖יף וַתִּתְכָּֽס׃ וְאָמְרָה לָעֶבֶד: מִי הָאִישׁ הַזֶּה הַהוֹלֵךְ בַּשָּׂדֶה לִקְרָאתֵנוּ? אָמַר הָעֶבֶד: הוּא אֲדוֹנִי! הִיא לָקְחָה אֶת הַצָּעִיף וְכִסְּתָה אֶת עַצְמָהּ בִּצְנִיעוּת.
סו וַיְסַפֵּ֥ר הָעֶ֖בֶד לְיִצְחָ֑ק אֵ֥ת כׇּל־הַדְּבָרִ֖ים אֲשֶׁ֥ר עָשָֽׂה׃ הָעֶבֶד סִפֵּר לְיִצְחָק אֶת כָּל הַדְּבָרִים שֶׁעָשָׂה.
סז וַיְבִאֶ֣הָ יִצְחָ֗ק הָאֹ֙הֱלָה֙ שָׂרָ֣ה אִמּ֔וֹ וַיִּקַּ֧ח אֶת־רִבְקָ֛ה וַתְּהִי־ל֥וֹ לְאִשָּׁ֖ה וַיֶּאֱהָבֶ֑הָ וַיִּנָּחֵ֥ם יִצְחָ֖ק אַחֲרֵ֥י אִמּֽוֹ׃ יִצְחָק הֵבִיא אוֹתָהּ לָאוֹהֶל שֶׁל שָׂרָה אִמּוֹ, הוּא לָקַח אֶת רִבְקָה הִיא נִהְיְתָה אִשְׁתּוֹ וְהוּא אָהַב אוֹתָהּ, וְיִצְחָק הִתְנַחֵם אָז אַחֲרֵי מוֹת אִמּוֹ.

ששי[עריכה]

פ פָּרָשַׁת תּוֹלְדֹת אַבְרָהָם
כה א וַיֹּ֧סֶף אַבְרָהָ֛ם וַיִּקַּ֥ח אִשָּׁ֖ה וּשְׁמָ֥הּ קְטוּרָֽה׃ אַבְרָהָם הוֹסִיף וְלָקַח אִשָּׁה וּשְׁמָהּ קְטוּרָה.
ב וַתֵּ֣לֶד ל֗וֹ אֶת־זִמְרָן֙ וְאֶת־יׇקְשָׁ֔ן וְאֶת־מְדָ֖ן וְאֶת־מִדְיָ֑ן וְאֶת־יִשְׁבָּ֖ק וְאֶת־שֽׁוּחַ׃ הִיא יָלְדָה לוֹ אֶת זִמְרָן וְאֶת יָקְשָׁן וְאֶת מְדָן וְאֶת מִדְיָן וְאֶת יִשְׁבָּק וְאֶת שׁוּחַ.
ג וְיׇקְשָׁ֣ן יָלַ֔ד אֶת־שְׁבָ֖א וְאֶת־דְּדָ֑ן וּבְנֵ֣י דְדָ֔ן הָי֛וּ אַשּׁוּרִ֥ם וּלְטוּשִׁ֖ם וּלְאֻמִּֽים׃ וְיָקְשָׁן יָלַד אֶת שְׁבָא וְאֶת דְּדָן, וּבְנֵי דְּדָן הָיוּ סוֹחֲרִים וְאֻמָּנִים וְרָאשֵׁי אֻמּוֹת.
ד וּבְנֵ֣י מִדְיָ֗ן עֵיפָ֤ה וָעֵ֙פֶר֙ וַחֲנֹ֔ךְ וַאֲבִידָ֖ע וְאֶלְדָּעָ֑ה כׇּל־אֵ֖לֶּה בְּנֵ֥י קְטוּרָֽה׃ וּבְנֵי מִדְיָן: עֵיפָה וְעֵפֶר וַחֲנוֹךְ וַאֲבִידָע וְאֶלְדָּעָה, כָּל אֵלֶּה בְּנֵי קְטוּרָה.
ה וַיִּתֵּ֧ן אַבְרָהָ֛ם אֶת־כׇּל־אֲשֶׁר־ל֖וֹ לְיִצְחָֽק׃ אַבְרָהָם נָתַן אֶת כָּל מַה שֶּׁיֵּשׁ לוֹ לְיִצְחָק.
ו וְלִבְנֵ֤י הַפִּֽילַגְשִׁים֙ אֲשֶׁ֣ר לְאַבְרָהָ֔ם נָתַ֥ן אַבְרָהָ֖ם מַתָּנֹ֑ת וַֽיְשַׁלְּחֵ֞ם מֵעַ֨ל יִצְחָ֤ק בְּנוֹ֙ בְּעוֹדֶ֣נּוּ חַ֔י קֵ֖דְמָה אֶל־אֶ֥רֶץ קֶֽדֶם׃ וְלִבְנֵי הַשְּׁפָחוֹת שֶׁל אַבְרָהָם הוּא נָתַן מַתָּנוֹת, וּבְעוֹדוֹ חַי הוּא שִׁלַּח אוֹתָם מֵעַל יִצְחָק בְּנוֹ מִזְרָחָה – לְאַרְצוֹת הַמִּזְרָח.
ז וְאֵ֗לֶּה יְמֵ֛י שְׁנֵֽי־חַיֵּ֥י אַבְרָהָ֖ם אֲשֶׁר־חָ֑י מְאַ֥ת שָׁנָ֛ה וְשִׁבְעִ֥ים שָׁנָ֖ה וְחָמֵ֥שׁ שָׁנִֽים׃ וְזֶה מִסְפַּר שְׁנוֹת חַיֵּי אַבְרָהָם שֶׁהוּא חַי: מֵאָה וְשִׁבְעִים וְחָמֵשׁ שָׁנִים.
ח וַיִּגְוַ֨ע וַיָּ֧מׇת אַבְרָהָ֛ם בְּשֵׂיבָ֥ה טוֹבָ֖ה זָקֵ֣ן וְשָׂבֵ֑עַ וַיֵּאָ֖סֶף אֶל־עַמָּֽיו׃ נִשְׁמַת אַבְרָהָם יָצְאָה וְהוּא מֵת בְּשֵׂיבָה טוֹבָה זָקֵן וְשָׂבֵעַ מִטּוּב, נִשְׁמָתוֹ נֶאֶסְפָה לִמְקוֹם הַנְּשָׁמוֹת.
ט וַיִּקְבְּר֨וּ אֹת֜וֹ יִצְחָ֤ק וְיִשְׁמָעֵאל֙ בָּנָ֔יו אֶל־מְעָרַ֖ת הַמַּכְפֵּלָ֑ה אֶל־שְׂדֵ֞ה עֶפְרֹ֤ן בֶּן־צֹ֙חַר֙ הַֽחִתִּ֔י אֲשֶׁ֖ר עַל־פְּנֵ֥י מַמְרֵֽא׃ יִצְחָק וְיִשְׁמָעֵאל בָּנָיו קָבְרוּ אוֹתוֹ בִּמְעָרַת הַמַּכְפֵּלָה שֶׁבִּשְׂדֵה עֶפְרוֹן בֶּן צוֹחַר הַחִתִּי שֶׁמּוּל מַמְרֵא.
י הַשָּׂדֶ֛ה אֲשֶׁר־קָנָ֥ה אַבְרָהָ֖ם מֵאֵ֣ת בְּנֵי־חֵ֑ת שָׁ֛מָּה קֻבַּ֥ר אַבְרָהָ֖ם וְשָׂרָ֥ה אִשְׁתּֽוֹ׃ בַּשָּׂדֶה שֶׁאַבְרָהָם קָנָה מֵאֵת בְּנֵי חֵת – שָׁם נִקְבְּרוּ אַבְרָהָם וְשָׂרָה אִשְׁתּוֹ.
יא וַיְהִ֗י אַחֲרֵי֙ מ֣וֹת אַבְרָהָ֔ם וַיְבָ֥רֶךְ אֱלֹהִ֖ים אֶת־יִצְחָ֣ק בְּנ֑וֹ וַיֵּ֣שֶׁב יִצְחָ֔ק עִם־בְּאֵ֥ר לַחַ֖י רֹאִֽי׃ אַחֲרֵי שֶׁאַבְרָהָם מֵת, אֱלֹהִים בֵּרֵךְ אֶת יִצְחָק בְּנוֹ, וְיִצְחָק גָּר לְיַד בְּאֵר לַחַי רוֹאִי.

שביעי[עריכה]

פ פָּרָשַׁת תּוֹלְדֹת יִשְׁמָעֵאל
יב וְאֵ֛לֶּה תֹּלְדֹ֥ת יִשְׁמָעֵ֖אל בֶּן־אַבְרָהָ֑ם אֲשֶׁ֨ר יָלְדָ֜ה הָגָ֧ר הַמִּצְרִ֛ית שִׁפְחַ֥ת שָׂרָ֖ה לְאַבְרָהָֽם׃ אֵלֶּה צֶאֱצָאֵי יִשְׁמָעֵאל בֶּן אַבְרָהָם שֶׁהָגָר הַמִּצְרִית שִׁפְחַת שָׂרָה יָלְדָה לְאַבְרָהָם.
יג וְאֵ֗לֶּה שְׁמוֹת֙ בְּנֵ֣י יִשְׁמָעֵ֔אל בִּשְׁמֹתָ֖ם לְתוֹלְדֹתָ֑ם בְּכֹ֤ר יִשְׁמָעֵאל֙ נְבָיֹ֔ת וְקֵדָ֥ר וְאַדְבְּאֵ֖ל וּמִבְשָֽׂם׃ וְאֵלֶּה שְׁמוֹת בְּנֵי יִשְׁמָעֵאל בִּשְׁמוֹתָם לְפִי סֵדֶר לֵידָתָם: בְּכוֹר יִשְׁמָעֵאל – נְבָיוֹת, וְקֵדָר, וְאַדְבְּאֵל, וּמִבְשָׂם.
יד וּמִשְׁמָ֥ע וְדוּמָ֖ה וּמַשָּֽׂא׃ וּמִשְׁמָע וְדוּמָה וּמַשָּׂא.
טו חֲדַ֣ד וְתֵימָ֔א יְט֥וּר נָפִ֖ישׁ וָקֵֽדְמָה׃ חֲדַד וְתֵימָא, יְטוּר נָפִישׁ וְקֵדְמָה.

מפטיר[עריכה]

טז אֵ֣לֶּה הֵ֞ם בְּנֵ֤י יִשְׁמָעֵאל֙ וְאֵ֣לֶּה שְׁמֹתָ֔ם בְּחַצְרֵיהֶ֖ם וּבְטִֽירֹתָ֑ם שְׁנֵים־עָשָׂ֥ר נְשִׂיאִ֖ם לְאֻמֹּתָֽם׃ אֵלֶּה הֵם בְּנֵי יִשְׁמָעֵאל – אֵלֶּה שְׁמוֹתָם, שֶׁגָּרִים בְּעָרִים לְלֹא חוֹמָה וְשֶׁגָּרִים בְּמִבְצָרִים, שְׁנֵים עָשָׂר נְשִׂיאִים לְאֻמּוֹתֵיהֶם הַשּׁוֹנוֹת.
יז וְאֵ֗לֶּה שְׁנֵי֙ חַיֵּ֣י יִשְׁמָעֵ֔אל מְאַ֥ת שָׁנָ֛ה וּשְׁלֹשִׁ֥ים שָׁנָ֖ה וְשֶׁ֣בַע שָׁנִ֑ים וַיִּגְוַ֣ע וַיָּ֔מׇת וַיֵּאָ֖סֶף אֶל־עַמָּֽיו׃ וְזֶה מִסְפַּר שְׁנוֹת חַיֵּי יִשְׁמָעֵאל: מֵאָה וּשְׁלוֹשִׁים וְשֶׁבַע שָׁנִים, רוּחוֹ יָצְאָה וְהוּא מֵת וְנִשְׁמָתוֹ נֶאֶסְפָה לִמְקוֹם הַנְּשָׁמוֹת.
יח וַיִּשְׁכְּנ֨וּ מֵֽחֲוִילָ֜ה עַד־שׁ֗וּר אֲשֶׁר֙ עַל־פְּנֵ֣י מִצְרַ֔יִם בֹּאֲכָ֖ה אַשּׁ֑וּרָה עַל־פְּנֵ֥י כׇל־אֶחָ֖יו נָפָֽל׃ הֵם שָׁכְנוּ מֵהֹדּוּ עַד חֲלוּצָה שֶׁמּוּל מִצְרַיִם, וְעַד לְאַשּׁוּר, הוּא הִשְׁתַּכֵּן מוּל כָּל אֶחָיו.
