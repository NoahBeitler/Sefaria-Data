ראשון[עריכה]

פ פָּרָשַׁת הָעֹלָה
ו א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב צַ֤ו אֶֽת־אַהֲרֹן֙ וְאֶת־בָּנָ֣יו לֵאמֹ֔ר זֹ֥את תּוֹרַ֖ת הָעֹלָ֑ה הִ֣וא הָעֹלָ֡ה עַל֩ מוֹקְדָ֨הֿ‏[1] עַל־הַמִּזְבֵּ֤חַ כׇּל־הַלַּ֙יְלָה֙ עַד־הַבֹּ֔קֶר וְאֵ֥שׁ הַמִּזְבֵּ֖חַ תּ֥וּקַד בּֽוֹ׃ צַוֵּה אֶת אַהֲרֹן וְאֶת בָּנָיו וֶאֱמוֹר: אֵלֶּה דִּינֵי הָעוֹלָה: הִיא בְּהֵמָה שֶׁנִּשְׂרֶפֶת בַּמְּדוּרָה שֶׁלָּהּ עַל הַמִּזְבֵּחַ – בָּאֵשׁ שֶׁתִּבְעַר עַל הַמִּזְבֵּחַ כָּל הַלַּיְלָה עַד הַבּוֹקֶר.
ג וְלָבַ֨שׁ הַכֹּהֵ֜ן מִדּ֣וֹ בַ֗ד וּמִֽכְנְסֵי־בַד֮ יִלְבַּ֣שׁ עַל־בְּשָׂרוֹ֒ וְהֵרִ֣ים אֶת־הַדֶּ֗שֶׁן אֲשֶׁ֨ר תֹּאכַ֥ל הָאֵ֛שׁ אֶת־הָעֹלָ֖ה עַל־הַמִּזְבֵּ֑חַ וְשָׂמ֕וֹ אֵ֖צֶל הַמִּזְבֵּֽחַ׃ הַכֹּהֵן יִלְבַּשׁ אֶת הַמַּדִּים שֶׁלּוֹ שֶׁעֲשׂוּיִים מִבַּד, וְיִלְבַּשׁ עַל גּוּפוֹ מִכְנָסַיִם מִבַּד, וְיָרִים מֵעַל הַמִּזְבֵּחַ אֶת הָאֵפֶר שֶׁנִּשְׁאַר מֵהָעוֹלָה שֶׁהָאֵשׁ שָׂרְפָה, וְיָשִׂים אוֹתוֹ לְיַד הַמִּזְבֵּחַ.
ד וּפָשַׁט֙ אֶת־בְּגָדָ֔יו וְלָבַ֖שׁ בְּגָדִ֣ים אֲחֵרִ֑ים וְהוֹצִ֤יא אֶת־הַדֶּ֙שֶׁן֙ אֶל־מִח֣וּץ לַֽמַּחֲנֶ֔ה אֶל־מָק֖וֹם טָהֽוֹר׃ וְיִפְשׁוֹט אֶת בְּגָדָיו וְיִלְבַּשׁ בְּגָדִים אֲחֵרִים, וְיוֹצִיא אֶת הַאֵפֶר אֶל מִחוּץ לַמַּחֲנֶה לְמָקוֹם טָהוֹר.
ה וְהָאֵ֨שׁ עַל־הַמִּזְבֵּ֤חַ תּֽוּקַד־בּוֹ֙ לֹ֣א תִכְבֶּ֔ה וּבִעֵ֨ר עָלֶ֧יהָ הַכֹּהֵ֛ן עֵצִ֖ים בַּבֹּ֣קֶר בַּבֹּ֑קֶר וְעָרַ֤ךְ עָלֶ֙יהָ֙ הָֽעֹלָ֔ה וְהִקְטִ֥יר עָלֶ֖יהָ חֶלְבֵ֥י הַשְּׁלָמִֽים׃ וְהָאֵשׁ שֶׁעַל הַמִּזְבֵּחַ תִּבְעַר עָלָיו וְלֹא תִּכְבֶּה, וְהַכֹּהֵן יַדְלִיק עָלֶיהָ עֵצִים מֻקְדָּם בַּבּוֹקֶר, וִיסַדֵּר עָלֶיהָ אֶת הָעוֹלָה וְיַקְטִיר עָלֶיהָ אֶת גּוּשֵׁי הַחֵלֶב שֶׁל קָרְבְּנוֹת הַשְּׁלָמִים.
ו אֵ֗שׁ תָּמִ֛יד תּוּקַ֥ד עַל־הַמִּזְבֵּ֖חַ לֹ֥א תִכְבֶּֽה׃ תָּמִיד תִּבְעַר אֵשׁ עַל הַמִּזְבֵּחַ וְלֹא תִּכְבֶּה.
ס פָּרָשַׁת הַמִּנְחָה
ז וְזֹ֥את תּוֹרַ֖ת הַמִּנְחָ֑ה הַקְרֵ֨ב אֹתָ֤הּ בְּנֵֽי־אַהֲרֹן֙ לִפְנֵ֣י יְהֹוָ֔ה אֶל־פְּנֵ֖י הַמִּזְבֵּֽחַ׃ אֵלֶּה דִּינֵי הַמִּנְחָה: בְּנֵי אַהֲרֹן יַגִּישׁוּ אוֹתָהּ אֶל לִפְנֵי ה' לְקִדְמַת הַמִּזְבֵּחַ.
ח וְהֵרִ֨ים מִמֶּ֜נּוּ בְּקֻמְצ֗וֹ מִסֹּ֤לֶת הַמִּנְחָה֙ וּמִשַּׁמְנָ֔הּ וְאֵת֙ כׇּל־הַלְּבֹנָ֔ה אֲשֶׁ֖ר עַל־הַמִּנְחָ֑ה וְהִקְטִ֣יר הַמִּזְבֵּ֗חַ רֵ֧יחַ נִיחֹ֛חַ אַזְכָּרָתָ֖הּ לַיהֹוָֽה׃ הַכֹּהֵן יַפְרִישׁ מִמֶּנָּה בְּיָדוֹ חוֹפֶן מֵהַסּוֹלֶת וּמֵהַשֶּׁמֶן שֶׁל הַמִּנְחָה, וְיִקַּח אֶת כָּל הַלְּבוֹנָה שֶׁעַל הַמִּנְחָה, וְיַקְטִיר עַל הַמִּזְבֵּחַ לַה' חוֹפֶן זֶה שֶׁגּוֹרֵם לְהִזָּכֵר לִפְנֵי ה' לְנַחַת רוּחַ.
ט וְהַנּוֹתֶ֣רֶת מִמֶּ֔נָּה יֹאכְל֖וּ אַהֲרֹ֣ן וּבָנָ֑יו מַצּ֤וֹת תֵּֽאָכֵל֙ בְּמָק֣וֹם קָדֹ֔שׁ בַּחֲצַ֥ר אֹֽהֶל־מוֹעֵ֖ד יֹאכְלֽוּהָ׃ וְאֶת מַה שֶּׁנִּשְׁאָר מֵהַמִּנְחָה יֹאכְלוּ אַהֲרֹן וּבָנָיו, הִיא תֵּאָכֵל בְּלִי שֶׁתַּחְמִיץ, הֵם יֹאכְלוּ אוֹתָהּ בְּמָקוֹם קָדוֹשׁ – בַּחֲצַר אוֹהֶל הַהִתְכַּנְּסוּת.
י לֹ֤א תֵאָפֶה֙ חָמֵ֔ץ חֶלְקָ֛ם נָתַ֥תִּי אֹתָ֖הּ מֵאִשָּׁ֑י קֹ֤דֶשׁ קׇֽדָשִׁים֙ הִ֔וא כַּחַטָּ֖את וְכָאָשָֽׁם׃ הִיא לֹא תֵּאָפֶה חָמֵץ, נָתַתִּי אוֹתָהּ לְחֶלְקָם שֶׁל אַהֲרֹן וּבָנָיו מִקָּרְבְּנוֹתַי, הִיא קוֹדֶשׁ קָדָשִׁים כְּמוֹ הַחַטָּאת וְהָאָשָׁם.
יא כׇּל־זָכָ֞ר בִּבְנֵ֤י אַהֲרֹן֙ יֹֽאכְלֶ֔נָּה חׇק־עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם מֵאִשֵּׁ֖י יְהֹוָ֑ה כֹּ֛ל אֲשֶׁר־יִגַּ֥ע בָּהֶ֖ם יִקְדָּֽשׁ׃ כָּל זָכָר בִּבְנֵי אַהֲרֹן יֹאכַל אוֹתָהּ, חוֹק לְעוֹלָם לְדוֹרוֹת מִקָּרְבְּנוֹת ה', כָּל מִי שֶׁיִּגַּע בָּהֶם צָרִיךְ לְהִתְקַדֵּשׁ.

שני[עריכה]

פ פָּרָשַׁת מִנְחַת הַחִנּוּךְ
יב וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יג זֶ֡ה קׇרְבַּן֩ אַהֲרֹ֨ן וּבָנָ֜יו אֲשֶׁר־יַקְרִ֣יבוּ לַֽיהֹוָ֗ה בְּיוֹם֙ הִמָּשַׁ֣ח אֹת֔וֹ עֲשִׂירִ֨ת הָאֵפָ֥ה סֹ֛לֶת מִנְחָ֖ה תָּמִ֑יד מַחֲצִיתָ֣הּ בַּבֹּ֔קֶר וּמַחֲצִיתָ֖הּ בָּעָֽרֶב׃ זֶה הַקָּרְבַּן שֶׁל אַהֲרֹן אוֹ אֶחָד מִבָּנָיו שֶׁיַּקְרִיבוּ לַה' תָּמִיד בַּיּוֹם שֶׁמּוֹרְחִים עָלָיו שֶׁמֶן לְמַנּוֹתוֹ לְכֹהֵן גָּדוֹל: עֲשִׂירִית הָאֵפָה סוֹלֶת (כִּ2 לִיטֶר) מִנְחָה, מַחֲצִיתָהּ בַּבּוֹקֶר וּמַחֲצִיתָהּ בָּעֶרֶב.
יד עַֽל־מַחֲבַ֗ת בַּשֶּׁ֛מֶן תֵּעָשֶׂ֖ה מֻרְבֶּ֣כֶת תְּבִיאֶ֑נָּה תֻּפִינֵי֙ מִנְחַ֣ת פִּתִּ֔ים תַּקְרִ֥יב רֵֽיחַ־נִיחֹ֖חַ לַיהֹוָֽה׃ הִיא תֵּעָשֶׂה בְּשֶׁמֶן עַל מַחֲבַת שְׁטוּחָה, תָּבִיא אוֹתָהּ מְעוֹרֶבֶת בְּשֶׁמֶן, תַּגִּישׁ מִנְחַת חֲתִיכוֹת שֶׁל מַאֲפִים לְנַחַת רוּחַ לִפְנֵי ה'.
טו וְהַכֹּהֵ֨ן הַמָּשִׁ֧יחַ תַּחְתָּ֛יו מִבָּנָ֖יו יַעֲשֶׂ֣ה אֹתָ֑הּ חׇק־עוֹלָ֕ם לַיהֹוָ֖ה כָּלִ֥יל תׇּקְטָֽר׃ וְגַם הַכֹּהֵן הַגָּדוֹל שֶׁיִּתְמַנֶּה בִּמְקוֹמוֹ מִבָּנָיו יַעֲשֶׂה אוֹתָהּ, חוֹק לְעוֹלָם שֶׁיַּקְטִירוּ אֶת כֻּלָּהּ לַה'.
טז וְכׇל־מִנְחַ֥ת כֹּהֵ֛ן כָּלִ֥יל תִּהְיֶ֖ה לֹ֥א תֵאָכֵֽל׃ וְכָל מִנְחָה שֶׁל כֹּהֵן יַקְטִירוּ אֶת כֻּלָּהּ, הִיא לֹא תֵּאָכֵל.
פ פָּרָשַׁת הַחַטָּאת
יז וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יח דַּבֵּ֤ר אֶֽל־אַהֲרֹן֙ וְאֶל־בָּנָ֣יו לֵאמֹ֔ר זֹ֥את תּוֹרַ֖ת הַֽחַטָּ֑את בִּמְק֡וֹם אֲשֶׁר֩ תִּשָּׁחֵ֨ט הָעֹלָ֜ה תִּשָּׁחֵ֤ט הַֽחַטָּאת֙ לִפְנֵ֣י יְהֹוָ֔ה קֹ֥דֶשׁ קׇֽדָשִׁ֖ים הִֽוא׃ דַּבֵּר אֶל אַהֲרֹן וְאֶל בָּנָיו וֶאֱמוֹר: אֵלֶּה דִּינֵי הַחַטָּאת: בַּמָּקוֹם שֶׁתִּשָּׁחֵט הָעוֹלָה שָׁם תִּשָּׁחֵט הַחַטָּאת לִפְנֵי ה', הִיא קוֹדֶשׁ קָדָשִׁים.
יט הַכֹּהֵ֛ן הַֽמְחַטֵּ֥א אֹתָ֖הּ יֹאכְלֶ֑נָּה בְּמָק֤וֹם קָדֹשׁ֙ תֵּֽאָכֵ֔ל בַּחֲצַ֖ר אֹ֥הֶל מוֹעֵֽד׃ הַכֹּהֵן שֶׁמַּקְרִיב אוֹתָהּ וּמְכַפֵּר בָּהּ הוּא יֹאכַל אוֹתָהּ, הִיא תֵּאָכֵל בְּמָקוֹם קָדוֹשׁ – בַּחֲצַר אוֹהֶל הַהִתְכַּנְּסוּת.
כ כֹּ֛ל אֲשֶׁר־יִגַּ֥ע בִּבְשָׂרָ֖הּ יִקְדָּ֑שׁ וַאֲשֶׁ֨ר יִזֶּ֤ה מִדָּמָהּ֙ עַל־הַבֶּ֔גֶד אֲשֶׁר֙ יִזֶּ֣ה עָלֶ֔יהָ תְּכַבֵּ֖ס בְּמָק֥וֹם קָדֹֽשׁ׃ כָּל מַה שֶּׁיִּגַּע בָּהּ וִיקַבֵּל טַעַם מִבְּשָׂרָהּ יִתְקַדֵּשׁ כָּמוֹהָ, וְאִם יוּתַּז מִדָּמָהּ עַל הַבֶּגֶד – תְּכַבֵּס בְּמָקוֹם קָדוֹשׁ אֶת הַבֶּגֶד שֶׁיּוּתַּז עָלָיו.
כא וּכְלִי־חֶ֛רֶשׂ אֲשֶׁ֥ר תְּבֻשַּׁל־בּ֖וֹ יִשָּׁבֵ֑ר וְאִם־בִּכְלִ֤י נְחֹ֙שֶׁת֙ בֻּשָּׁ֔לָה וּמֹרַ֥ק וְשֻׁטַּ֖ף בַּמָּֽיִם׃ וּכְלִי חֶרֶס שֶׁהִיא תְּבֻשַּׁל בּוֹ – יִשָּׁבֵר, וְאִם נִתְבַּשְּׁלָה בִּכְלִי נְחוֹשֶׁת – יִשְׁטְפוּ אוֹתוֹ וְיַרְתִּיחוּ אוֹתוֹ בְּמַיִם.
כב כׇּל־זָכָ֥ר בַּכֹּהֲנִ֖ים יֹאכַ֣ל אֹתָ֑הּ קֹ֥דֶשׁ קׇֽדָשִׁ֖ים הִֽוא׃ כָּל זָכָר בַּכֹּהֲנִים יֹאכַל אוֹתָהּ, הִיא קוֹדֶשׁ קָדָשִׁים.
כג וְכׇל־חַטָּ֡את אֲשֶׁר֩ יוּבָ֨א מִדָּמָ֜הּ אֶל־אֹ֧הֶל מוֹעֵ֛ד לְכַפֵּ֥ר בַּקֹּ֖דֶשׁ לֹ֣א תֵאָכֵ֑ל בָּאֵ֖שׁ תִּשָּׂרֵֽף׃ וְכָל חַטָּאת שֶׁיּוּכְנַס מִדָּמָהּ לְאוֹהֶל הַהִתְכַּנְּסוּת לְכַפֵּר בַּקּוֹדֶשׁ – לֹא תֵּאָכֵל, אֶלָּא תִּשָּׂרֵף בָּאֵשׁ.
פ פָּרָשַׁת הָאָשָׁם
ז א וְזֹ֥את תּוֹרַ֖ת הָאָשָׁ֑ם קֹ֥דֶשׁ קׇֽדָשִׁ֖ים הֽוּא׃ אֵלֶּה דִּינֵי הָאָשָׁם: הוּא קוֹדֶשׁ קָדָשִׁים.
ב בִּמְק֗וֹם אֲשֶׁ֤ר יִשְׁחֲטוּ֙ אֶת־הָ֣עֹלָ֔ה יִשְׁחֲט֖וּ אֶת־הָאָשָׁ֑ם וְאֶת־דָּמ֛וֹ יִזְרֹ֥ק עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ בַּמָּקוֹם שֶׁיִּשְׁחֲטוּ אֶת הָעוֹלָה שָׁם יִשְׁחֲטוּ אֶת הָאָשָׁם, וְהַכֹּהֵן יִזְרוֹק אֶת דָּמוֹ עַל הַמִּזְבֵּחַ סָבִיב.
ג וְאֵ֥ת כׇּל־חֶלְבּ֖וֹ יַקְרִ֣יב מִמֶּ֑נּוּ אֵ֚ת הָֽאַלְיָ֔ה וְאֶת־הַחֵ֖לֶב הַֽמְכַסֶּ֥ה אֶת־הַקֶּֽרֶב׃ וְיַסִּיר מִמֶּנּוּ אֶת כָּל הַחֵלֶב וְיַגִּישׁ – אֶת שׁוּמַן הַזָּנָב וְאֶת הַחֵלֶב שֶׁמְּכַסֶּה אֶת הַמֵּעַיִם,
ד וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־הַחֵ֙לֶב֙ אֲשֶׁ֣ר עֲלֵיהֶ֔ן אֲשֶׁ֖ר עַל־הַכְּסָלִ֑ים וְאֶת־הַיֹּתֶ֙רֶת֙ עַל־הַכָּבֵ֔ד עַל־הַכְּלָיֹ֖ת יְסִירֶֽנָּה׃ וְאֶת שְׁתֵּי הַכְּלָיוֹת וְאֶת הַחֵלֶב שֶׁעֲלֵיהֶן וְעַל הַמָּתְנַיִם, וְאֶת הַתּוֹסֶפֶת שֶׁעַל הַכָּבֵד שֶׁיָּסִיר אוֹתָהּ עִם הַכְּלָיוֹת.
ה וְהִקְטִ֨יר אֹתָ֤ם הַכֹּהֵן֙ הַמִּזְבֵּ֔חָה אִשֶּׁ֖ה לַיהֹוָ֑ה אָשָׁ֖ם הֽוּא׃ הַכֹּהֵן יִשְׂרוֹף אוֹתָם עַל הַמִּזְבֵּחַ – קָרְבָּן עוֹלֶה בָּאֵשׁ לַה', זֶה אָשָׁם.
ו כׇּל־זָכָ֥ר בַּכֹּהֲנִ֖ים יֹאכְלֶ֑נּוּ בְּמָק֤וֹם קָדוֹשׁ֙ יֵאָכֵ֔ל קֹ֥דֶשׁ קׇֽדָשִׁ֖ים הֽוּא׃ כָּל כֹּהֵן זָכָר יֹאכַל אוֹתוֹ, הוּא יֵאָכֵל בְּמָקוֹם קָדוֹשׁ, הוּא קוֹדֶשׁ קָדָשִׁים.
ז כַּֽחַטָּאת֙ כָּֽאָשָׁ֔ם תּוֹרָ֥ה אַחַ֖ת לָהֶ֑ם הַכֹּהֵ֛ן אֲשֶׁ֥ר יְכַפֶּר־בּ֖וֹ ל֥וֹ יִהְיֶֽה׃ כְּדִין הַחַטָּאת כָּךְ דִּין הָאָשָׁם, דִּין אֶחָד לָהֶם, הַכֹּהֵן שֶׁיְּכַפֵּר בּוֹ – הַקָּרְבָּן יִהְיֶה שֶׁלּוֹ.
ח וְהַ֨כֹּהֵ֔ן הַמַּקְרִ֖יב אֶת־עֹ֣לַת אִ֑ישׁ ע֤וֹר הָֽעֹלָה֙ אֲשֶׁ֣ר הִקְרִ֔יב לַכֹּהֵ֖ן ל֥וֹ יִהְיֶֽה׃ וְהַכֹּהֵן שֶׁמַּקְרִיב עוֹלָה שֶׁל אִישׁ, עוֹר הָעוֹלָה שֶׁהִקְרִיב יִהְיֶה שַׁיָּךְ לְאוֹתוֹ כֹּהֵן.
ט וְכׇל־מִנְחָ֗ה אֲשֶׁ֤ר תֵּֽאָפֶה֙ בַּתַּנּ֔וּר וְכׇל־נַעֲשָׂ֥ה בַמַּרְחֶ֖שֶׁת וְעַֽל־מַחֲבַ֑ת לַכֹּהֵ֛ן הַמַּקְרִ֥יב אֹתָ֖הּ ל֥וֹ תִֽהְיֶֽה׃ וְכָל מִנְחָה שֶׁתֵּאָפֶה בַּתַּנּוּר וְכָל שֶׁנַּעֲשֵׂית בְּמַחֲבַת עֲמוּקָה אוֹ שְׁטוּחָה, הִיא תִּהְיֶה לַכֹּהֵן שֶׁמַּקְרִיב אוֹתָהּ.
י וְכׇל־מִנְחָ֥ה בְלוּלָֽה־בַשֶּׁ֖מֶן וַחֲרֵבָ֑ה לְכׇל־בְּנֵ֧י אַהֲרֹ֛ן תִּהְיֶ֖ה אִ֥ישׁ כְּאָחִֽיו׃ וְכָל מִנְחָה מְעֻרְבֶּבֶת בְּשֶׁמֶן אוֹ יְבֵשָׁה תִּהְיֶה לְכָל בְּנֵי אַהֲרֹן – לְכָל אֶחָד חֵלֶק שָׁוֶה כְּמוֹ לַחֲבֵרוֹ.

שלישי[עריכה]

פ פָּרָשַׁת הַשְּׁלָמִים
יא וְזֹ֥את תּוֹרַ֖ת זֶ֣בַח הַשְּׁלָמִ֑ים אֲשֶׁ֥ר יַקְרִ֖יב לַיהֹוָֽה׃ וְאֵלֶּה דִּינֵי זֶבַח הַשְּׁלָמִים שֶׁיַּקְרִיב לַה':
יב אִ֣ם עַל־תּוֹדָה֮ יַקְרִיבֶ֒נּוּ֒ וְהִקְרִ֣יב ׀ עַל־זֶ֣בַח הַתּוֹדָ֗ה חַלּ֤וֹת מַצּוֹת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֣ים בַּשָּׁ֑מֶן וְסֹ֣לֶת מֻרְבֶּ֔כֶת חַלֹּ֖ת בְּלוּלֹ֥ת בַּשָּֽׁמֶן׃ אִם יַקְרִיב אוֹתוֹ כְּקָרְבַּן הוֹדָיָה, יַקְרִיב בְּנוֹסָף לְזֶבַח הַתּוֹדָה חַלּוֹת שֶׁלֹּא הֶחְמִיצוּ מְעֻרְבָּבוֹת בְּשֶׁמֶן, וּפִתּוֹת דַּקּוֹת שֶׁלֹּא הֶחְמִיצוּ מְרוּחִים בְּשֶׁמֶן, וְסוֹלֶת מְעוֹרֶבֶת בְּשֶׁמֶן – חַלּוֹת מְעֻרְבָּבוֹת בְּשֶׁמֶן.
יג עַל־חַלֹּת֙ לֶ֣חֶם חָמֵ֔ץ יַקְרִ֖יב קׇרְבָּנ֑וֹ עַל־זֶ֖בַח תּוֹדַ֥ת שְׁלָמָֽיו׃ הוּא יַקְרִיב אֶת קָרְבָּנוֹ עִם חַלּוֹת לֶחֶם חָמֵץ עִם קָרְבַּן הַשְּׁלָמִים שֶׁמֵּבִיא לְהוֹדָיָה.
יד וְהִקְרִ֨יב מִמֶּ֤נּוּ אֶחָד֙ מִכׇּל־קׇרְבָּ֔ן תְּרוּמָ֖ה לַיהֹוָ֑ה לַכֹּהֵ֗ן הַזֹּרֵ֛ק אֶת־דַּ֥ם הַשְּׁלָמִ֖ים ל֥וֹ יִהְיֶֽה׃ הוּא יַגִּישׁ מֵהֶם אֶחָד מִכָּל סוּג מַאֲפֶה כִּתְרוּמָה לַה', וְיִהְיֶה שַׁיָּךְ לַכֹּהֵן שֶׁזּוֹרֵק אֶת דַּם הַשְּׁלָמִים.
טו וּבְשַׂ֗ר זֶ֚בַח תּוֹדַ֣ת שְׁלָמָ֔יו בְּי֥וֹם קׇרְבָּנ֖וֹ יֵאָכֵ֑ל לֹֽא־יַנִּ֥יחַ מִמֶּ֖נּוּ עַד־בֹּֽקֶר׃ וּבְשַׂר קָרְבַּן הַשְּׁלָמִים שֶׁמֵּבִיא לְהוֹדָיָה יֵאָכֵל בְּיוֹם הַקְרָבָתוֹ – לֹא יַשְׁאִיר מִמֶּנּוּ עַד הַבּוֹקֶר.
טז וְאִם־נֶ֣דֶר ׀ א֣וֹ נְדָבָ֗ה זֶ֚בַח קׇרְבָּנ֔וֹ בְּי֛וֹם הַקְרִיב֥וֹ אֶת־זִבְח֖וֹ יֵאָכֵ֑ל וּמִֽמׇּחֳרָ֔ת וְהַנּוֹתָ֥ר מִמֶּ֖נּוּ יֵאָכֵֽל׃ וְאִם זֶבַח קָרְבָּנוֹ הוּא נֶדֶר אוֹ נְדָבָה שֶׁאֵינוֹ קָרְבַּן תּוֹדָה – הוּא יֵאָכֵל בַּיּוֹם שֶׁיַּקְרִיב אֶת זִבְחוֹ, וּמַה שֶּׁיִּשָּׁאֵר מִמֶּנּוּ יֵאָכֵל לְמָחֳרָת.
יז וְהַנּוֹתָ֖ר מִבְּשַׂ֣ר הַזָּ֑בַח בַּיּוֹם֙ הַשְּׁלִישִׁ֔י בָּאֵ֖שׁ יִשָּׂרֵֽף׃ וּמַה שֶּׁיִּשָּׁאֵר מִבְּשַׂר הַזֶּבַח יִשָּׂרֵף בָּאֵשׁ בַּיּוֹם הַשְּׁלִישִׁי.
יח וְאִ֣ם הֵאָכֹ֣ל יֵ֠אָכֵ֠ל מִבְּשַׂר־זֶ֨בַח שְׁלָמָ֜יו בַּיּ֣וֹם הַשְּׁלִישִׁי֮ לֹ֣א יֵרָצֶה֒ הַמַּקְרִ֣יב אֹת֗וֹ לֹ֧א יֵחָשֵׁ֛ב ל֖וֹ פִּגּ֣וּל יִהְיֶ֑ה וְהַנֶּ֛פֶשׁ הָאֹכֶ֥לֶת מִמֶּ֖נּוּ עֲוֺנָ֥הּ תִּשָּֽׂא׃ וְאִם יֵאָכֵל מִבְּשַׂר זֶבַח שְׁלָמָיו בַּיּוֹם הַשְּׁלִישִׁי – הוּא לֹא יִתְקַבֵּל בְּרָצוֹן, זֶה לֹא יֵחָשֵׁב מִצְוָה לַמַּקְרִיב אוֹתוֹ, זֶה קָרְבָּן מָאוּס, וְהָאָדָם שֶׁאוֹכֵל מִמֶּנּוּ יְקַבֵּל עוֹנֶשׁ עַל עֲווֹנוֹ.
יט וְהַבָּשָׂ֞ר אֲשֶׁר־יִגַּ֤ע בְּכׇל־טָמֵא֙ לֹ֣א יֵֽאָכֵ֔ל בָּאֵ֖שׁ יִשָּׂרֵ֑ף וְהַ֨בָּשָׂ֔ר כׇּל־טָה֖וֹר יֹאכַ֥ל בָּשָֽׂר׃ וּבָשָׂר שֶׁיִּגַּע בְּכָל דָּבָר טָמֵא לֹא יֵאָכֵל אֶלָּא יִשָּׂרֵף בָּאֵשׁ, וּבְשַׂר שְׁלָמִים – כָּל מִי שֶׁטָּהוֹר יֹאכַל מֵהַבָּשָׂר.
כ וְהַנֶּ֜פֶשׁ אֲשֶׁר־תֹּאכַ֣ל בָּשָׂ֗ר מִזֶּ֤בַח הַשְּׁלָמִים֙ אֲשֶׁ֣ר לַיהֹוָ֔ה וְטֻמְאָת֖וֹ עָלָ֑יו וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵעַמֶּֽיהָ׃ וְהָאָדָם שֶׁיֹּאכַל בָּשָׂר מִזֶּבַח הַשְּׁלָמִים שֶׁל ה' כְּשֶׁהוּא טָמֵא – הַנֶּפֶשׁ הַהִיא תֹּאבַד מֵעַמָּהּ.
כא וְנֶ֜פֶשׁ כִּֽי־תִגַּ֣ע בְּכׇל־טָמֵ֗א בְּטֻמְאַ֤ת אָדָם֙ א֣וֹ ׀ בִּבְהֵמָ֣ה טְמֵאָ֗ה א֚וֹ בְּכׇל־שֶׁ֣קֶץ טָמֵ֔א וְאָכַ֛ל מִבְּשַׂר־זֶ֥בַח הַשְּׁלָמִ֖ים אֲשֶׁ֣ר לַיהֹוָ֑ה וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵעַמֶּֽיהָ׃ וְאָדָם שֶׁיִּגַּע בְּכָל דָּבָר טָמֵא בְּטֻמְאַת אָדָם אוֹ בִּבְהֵמָה טְמֵאָה אוֹ בְּכָל שֶׁקֶץ טָמֵא, וְיֹאכַל מִבְּשַׂר זֶבַח הַשְּׁלָמִים שֶׁל ה' – הַנֶּפֶשׁ הַהִיא תֹּאבַד מֵעַמָּהּ.
פ (בספרי תימן) פָּרָשַׁת מַתְּנוֹת חֵלֶב וְדָם
כב וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
כג דַּבֵּ֛ר אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר כׇּל־חֵ֜לֶב שׁ֥וֹר וְכֶ֛שֶׂב וָעֵ֖ז לֹ֥א תֹאכֵֽלוּ׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וֶאֱמוֹר: אַל תֹּאכְלוּ שׁוּם חֵלֶב שֶׁל שׁוֹר, כֶּבֶשׂ, וְעֵז.
כד וְחֵ֤לֶב נְבֵלָה֙ וְחֵ֣לֶב טְרֵפָ֔ה יֵעָשֶׂ֖ה לְכׇל־מְלָאכָ֑ה וְאָכֹ֖ל לֹ֥א תֹאכְלֻֽהוּ׃ וְחֵלֶב שֶׁל נְבֵלָה וְשֶׁל טְרֵפָה תִּשְׁתַּמְּשׁוּ בּוֹ לְכָל מְלָאכָה אֲבָל אַל תֹּאכְלוּ אוֹתוֹ.
כה כִּ֚י כׇּל־אֹכֵ֣ל חֵ֔לֶב מִ֨ן־הַבְּהֵמָ֔ה אֲשֶׁ֨ר יַקְרִ֥יב מִמֶּ֛נָּה אִשֶּׁ֖ה לַיהֹוָ֑ה וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הָאֹכֶ֖לֶת מֵֽעַמֶּֽיהָ׃ כִּי כָּל מִי שֶׁאוֹכֵל חֵלֶב מֵהַבְּהֵמָה שֶׁמַּקְרִיבִים מִמֶּנָּה קָרְבָּן עוֹלֶה בָּאֵשׁ לַה' – הַנֶּפֶשׁ שֶׁאוֹכֶלֶת תֹּאבַד מֵעַמָּהּ.
כו וְכׇל־דָּם֙ לֹ֣א תֹאכְל֔וּ בְּכֹ֖ל מוֹשְׁבֹתֵיכֶ֑ם לָע֖וֹף וְלַבְּהֵמָֽה׃ וְאַל תֹּאכְלוּ שׁוּם דָּם שֶׁל עוֹף אוֹ בְּהֵמָה בְּכָל מְקוֹמוֹת מְגוּרֵיכֶם.
כז כׇּל־נֶ֖פֶשׁ אֲשֶׁר־תֹּאכַ֣ל כׇּל־דָּ֑ם וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵֽעַמֶּֽיהָ׃ כָּל אָדָם שֶׁיֹּאכַל דַּם שֶׁל כָּל חַיָּה – הַנֶּפֶשׁ הַהִיא תֹּאבַד מֵעַמָּהּ.
פ (בספרי ספרד ואשכנז) פָּרָשַׁת הַתְּרוּמָה מֵהַשְּׁלָמִים
כח וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
כט דַּבֵּ֛ר אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר הַמַּקְרִ֞יב אֶת־זֶ֤בַח שְׁלָמָיו֙ לַיהֹוָ֔ה יָבִ֧יא אֶת־קׇרְבָּנ֛וֹ לַיהֹוָ֖ה מִזֶּ֥בַח שְׁלָמָֽיו׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וֶאֱמוֹר: מִי שֶׁמַּקְרִיב זֶבַח שְׁלָמִים שֶׁלּוֹ לַה' יָבִיא אֶת הַחֵלֶק שֶׁשַּׁיָּךְ לַה' מִקָּרְבַּן הַשְּׁלָמִים שֶׁלּוֹ.
ל יָדָ֣יו תְּבִיאֶ֔ינָה אֵ֖ת אִשֵּׁ֣י יְהֹוָ֑ה אֶת־הַחֵ֤לֶב עַל־הֶֽחָזֶה֙ יְבִיאֶ֔נּוּ אֵ֣ת הֶחָזֶ֗ה לְהָנִ֥יף אֹת֛וֹ תְּנוּפָ֖ה לִפְנֵ֥י יְהֹוָֽה׃ יָדָיו יָבִיאוּ אֶת קָרְבְּנוֹת ה' – הוּא יָבִיא אֶת הַחֵלֶב מוּנָח עַל הֶחָזֶה, וְאֶת הֶחָזֶה יָבִיא לְהָנִיף אוֹתוֹ לִפְנֵי ה'.
לא וְהִקְטִ֧יר הַכֹּהֵ֛ן אֶת־הַחֵ֖לֶב הַמִּזְבֵּ֑חָה וְהָיָה֙ הֶֽחָזֶ֔ה לְאַהֲרֹ֖ן וּלְבָנָֽיו׃ הַכֹּהֵן יַקְטִיר אֶת הַחֵלֶב עַל הַמִּזְבֵּחַ, וְהֶחָזֶה יִהְיֶה לְאַהֲרֹן וּלְבָנָיו.
לב וְאֵת֙ שׁ֣וֹק הַיָּמִ֔ין תִּתְּנ֥וּ תְרוּמָ֖ה לַכֹּהֵ֑ן מִזִּבְחֵ֖י שַׁלְמֵיכֶֽם׃ וְאֶת הַשּׁוֹק הַיְּמָנִי תִּתְּנוּ תְּרוּמָה לַכֹּהֵן מִזִּבְחֵי הַשְּׁלָמִים שֶׁלָּכֶם.
לג הַמַּקְרִ֞יב אֶת־דַּ֧ם הַשְּׁלָמִ֛ים וְאֶת־הַחֵ֖לֶב מִבְּנֵ֣י אַהֲרֹ֑ן ל֧וֹ תִהְיֶ֛ה שׁ֥וֹק הַיָּמִ֖ין לְמָנָֽה׃ הַכֹּהֵן מִבְּנֵי אַהֲרֹן שֶׁמַּקְרִיב אֶת דַּם הַשְּׁלָמִים וְאֶת הַחֵלֶב – הַשּׁוֹק הַיְּמָנִי יִהְיֶה הַמָּנָה שֶׁלּוֹ.
לד כִּי֩ אֶת־חֲזֵ֨ה הַתְּנוּפָ֜ה וְאֵ֣ת ׀ שׁ֣וֹק הַתְּרוּמָ֗ה לָקַ֙חְתִּי֙ מֵאֵ֣ת בְּנֵֽי־יִשְׂרָאֵ֔ל מִזִּבְחֵ֖י שַׁלְמֵיהֶ֑ם וָאֶתֵּ֣ן אֹ֠תָ֠ם לְאַהֲרֹ֨ן הַכֹּהֵ֤ן וּלְבָנָיו֙ לְחׇק־עוֹלָ֔ם מֵאֵ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ כִּי אֶת הֶחָזֶה הַמּוּנָף וְאֶת הַשּׁוֹק שֶׁהוּרַם לָקַחְתִּי מֵאֵת בְּנֵי יִשְׂרָאֵל מִזִּבְחֵי הַשְּׁלָמִים שֶׁלָּהֶם, וְנָתַתִּי אוֹתָם לְאַהֲרֹן הַכֹּהֵן וּלְבָנָיו – חוֹק לְעוֹלָם מִבְּנֵי יִשְׂרָאֵל.
לה זֹ֣את מִשְׁחַ֤ת אַהֲרֹן֙ וּמִשְׁחַ֣ת בָּנָ֔יו מֵאִשֵּׁ֖י יְהֹוָ֑ה בְּיוֹם֙ הִקְרִ֣יב אֹתָ֔ם לְכַהֵ֖ן לַיהֹוָֽה׃ זֹאת מַתְּנַת הַמִּנּוּי שֶׁל אַהֲרֹן וְשֶׁל בָּנָיו מִקָּרְבְּנוֹת ה' שֶׁנִּתְּנוּ בַּיּוֹם שֶׁקֵּרְבוּ אוֹתָם לְכַהֵן לַה'.
לו אֲשֶׁר֩ צִוָּ֨ה יְהֹוָ֜ה לָתֵ֣ת לָהֶ֗ם בְּיוֹם֙ מׇשְׁח֣וֹ אֹתָ֔ם מֵאֵ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל חֻקַּ֥ת עוֹלָ֖ם לְדֹרֹתָֽם׃ שֶׁה' צִוָּה שֶׁיִּנָּתֵן לָהֶם מִבְּנֵי יִשְׂרָאֵל בַּיּוֹם שֶׁמְּמַנִּים אוֹתָם, חוֹק לְעוֹלָם לְדוֹרוֹת.
לז זֹ֣את הַתּוֹרָ֗ה לָֽעֹלָה֙ לַמִּנְחָ֔ה וְלַֽחַטָּ֖את וְלָאָשָׁ֑ם וְלַ֨מִּלּוּאִ֔ים וּלְזֶ֖בַח הַשְּׁלָמִֽים׃ אֵלֶּה הַדִּינִים לָעוֹלָה לַמִּנְחָה וְלַחַטָּאת וְלָאָשָׁם וּלְקָרְבָּן הַמִּנּוּי וּלְזֶבַח הַשְּׁלָמִים.
לח אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֶת־מֹשֶׁ֖ה בְּהַ֣ר סִינָ֑י בְּי֨וֹם צַוֺּת֜וֹ אֶת־בְּנֵ֣י יִשְׂרָאֵ֗ל לְהַקְרִ֧יב אֶת־קׇרְבְּנֵיהֶ֛ם לַיהֹוָ֖ה בְּמִדְבַּ֥ר סִינָֽי׃ שֶׁה' צִוָּה אֶת מֹשֶׁה בְּהַר סִינַי בַּיּוֹם שֶׁצִּוָּה אֶת בְּנֵי יִשְׂרָאֵל לְהַקְרִיב אֶת קָרְבְּנוֹתֵיהֶם לַה' בְּמִדְבַּר סִינַי.

רביעי[עריכה]

פ פָּרָשַׁת הַמִּלּוּאִים
ח א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב קַ֤ח אֶֽת־אַהֲרֹן֙ וְאֶת־בָּנָ֣יו אִתּ֔וֹ וְאֵת֙ הַבְּגָדִ֔ים וְאֵ֖ת שֶׁ֣מֶן הַמִּשְׁחָ֑ה וְאֵ֣ת ׀ פַּ֣ר הַֽחַטָּ֗את וְאֵת֙ שְׁנֵ֣י הָֽאֵילִ֔ים וְאֵ֖ת סַ֥ל הַמַּצּֽוֹת׃ קַח אֶת אַהֲרֹן וְאֶת בָּנָיו אִתּוֹ וְאֶת הַבְּגָדִים וְאֶת הַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ וְאֶת פַּר הַחַטָּאת וְאֶת שְׁנֵי הָאֵילִים וְאֶת סַל הַלֶּחֶם שֶׁלֹּא הֶחְמִיץ.
ג וְאֵ֥ת כׇּל־הָעֵדָ֖ה הַקְהֵ֑ל אֶל־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וְכַנֵּס אֶת כָּל הַצִּבּוּר אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.
ד וַיַּ֣עַשׂ מֹשֶׁ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֹת֑וֹ וַתִּקָּהֵל֙ הָֽעֵדָ֔ה אֶל־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ מֹשֶׁה עָשָׂה כְּמוֹ שֶׁה' צִוָּה אוֹתוֹ, וְהַצִּבּוּר הִתְאַסֵּף אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.
ה וַיֹּ֥אמֶר מֹשֶׁ֖ה אֶל־הָעֵדָ֑ה זֶ֣ה הַדָּבָ֔ר אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה לַעֲשֽׂוֹת׃ מֹשֶׁה אָמַר לַצִּבּוּר: זֶה הַדָּבָר שֶׁה' צִוָּה לַעֲשׂוֹת:
ו וַיַּקְרֵ֣ב מֹשֶׁ֔ה אֶֽת־אַהֲרֹ֖ן וְאֶת־בָּנָ֑יו וַיִּרְחַ֥ץ אֹתָ֖ם בַּמָּֽיִם׃ מֹשֶׁה קֵרֵב אֶת אַהֲרֹן וְאֶת בָּנָיו וְהִטְבִּיל אוֹתָם בְּמַיִם.
ז וַיִּתֵּ֨ן עָלָ֜יו אֶת־הַכֻּתֹּ֗נֶת וַיַּחְגֹּ֤ר אֹתוֹ֙ בָּֽאַבְנֵ֔ט וַיַּלְבֵּ֤שׁ אֹתוֹ֙ אֶֽת־הַמְּעִ֔יל וַיִּתֵּ֥ן עָלָ֖יו אֶת־הָאֵפֹ֑ד וַיַּחְגֹּ֣ר אֹת֗וֹ בְּחֵ֙שֶׁב֙ הָֽאֵפֹ֔ד וַיֶּאְפֹּ֥ד ל֖וֹ בּֽוֹ׃ הוּא שָׂם עָלָיו אֶת הַכֻּתּוֹנֶת, חָגַר אוֹתוֹ בָּאַבְנֵט, הִלְבִּישׁ אוֹתוֹ בַּמְּעִיל, שָׂם עָלָיו אֶת הָאֵפוֹד, חָגַר אוֹתוֹ בַּחֲגוֹרַת הָאֵפוֹד וְהִלְבִּישׁ אוֹתוֹ עָלָיו.
ח וַיָּ֥שֶׂם עָלָ֖יו אֶת־הַחֹ֑שֶׁן וַיִּתֵּן֙ אֶל־הַחֹ֔שֶׁן אֶת־הָאוּרִ֖ים וְאֶת־הַתֻּמִּֽים׃ הוּא שָׂם עָלָיו אֶת הַחוֹשֶׁן, וְשָׂם בַּחוֹשֶׁן אֶת הָאֲבָנִים הַמְּאִירוֹת שֶׁעוֹנוֹת לִשְׁאֵלוֹת.
ט וַיָּ֥שֶׂם אֶת־הַמִּצְנֶ֖פֶת עַל־רֹאשׁ֑וֹ וַיָּ֨שֶׂם עַֽל־הַמִּצְנֶ֜פֶת אֶל־מ֣וּל פָּנָ֗יו אֵ֣ת צִ֤יץ הַזָּהָב֙ נֵ֣זֶר הַקֹּ֔דֶשׁ כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ הוּא שָׂם אֶת הַמִּצְנֶפֶת עַל רֹאשׁוֹ וְשָׂם עַל הַמִּצְנֶפֶת כְּלַפֵּי חוּץ אֶת צִיץ הַזָּהָב – כֶּתֶר הַקּוֹדֶשׁ, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.
י וַיִּקַּ֤ח מֹשֶׁה֙ אֶת־שֶׁ֣מֶן הַמִּשְׁחָ֔ה וַיִּמְשַׁ֥ח אֶת־הַמִּשְׁכָּ֖ן וְאֶת־כׇּל־אֲשֶׁר־בּ֑וֹ וַיְקַדֵּ֖שׁ אֹתָֽם׃ מֹשֶׁה לָקַח אֶת הַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ וּמָרַח עַל הַמִּשְׁכָּן וְעַל כָּל מַה שֶּׁבּוֹ וְקִדֵּשׁ אוֹתָם.
יא וַיַּ֥ז מִמֶּ֛נּוּ עַל־הַמִּזְבֵּ֖חַ שֶׁ֣בַע פְּעָמִ֑ים וַיִּמְשַׁ֨ח אֶת־הַמִּזְבֵּ֜חַ וְאֶת־כׇּל־כֵּלָ֗יו וְאֶת־הַכִּיֹּ֛ר וְאֶת־כַּנּ֖וֹ לְקַדְּשָֽׁם׃ הוּא הִתִּיז מִמֶּנּוּ עַל הַמִּזְבֵּחַ שֶׁבַע פְּעָמִים, וּמָרַח עַל הַמִּזְבֵּחַ וְעַל כָּל כֵּלָיו וְעַל הַכִּיּוֹר וְעַל כַּנּוֹ כְּדֵי לְקַדֵּשׁ אוֹתָם.
יב וַיִּצֹק֙ מִשֶּׁ֣מֶן הַמִּשְׁחָ֔ה עַ֖ל רֹ֣אשׁ אַהֲרֹ֑ן וַיִּמְשַׁ֥ח אֹת֖וֹ לְקַדְּשֽׁוֹ׃ הוּא שָׁפַךְ מֵהַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ עַל הָרֹאשׁ שֶׁל אַהֲרֹן, וּמָרַח עָלָיו כְּדֵי לְקַדֵּשׁ אוֹתוֹ.
יג וַיַּקְרֵ֨ב מֹשֶׁ֜ה אֶת־בְּנֵ֣י אַהֲרֹ֗ן וַיַּלְבִּשֵׁ֤ם כֻּתֳּנֹת֙ וַיַּחְגֹּ֤ר אֹתָם֙ אַבְנֵ֔ט וַיַּחֲבֹ֥שׁ לָהֶ֖ם מִגְבָּע֑וֹת כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ מֹשֶׁה קֵרֵב אֶת בְּנֵי אַהֲרֹן וְהִלְבִּישׁ אוֹתָם כֻּתּוֹנוֹת, חָגַר אוֹתָם בַּחֲגוֹרָה וְחָבַשׁ לָהֶם מִגְבָּעוֹת, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

חמישי[עריכה]

יד וַיַּגֵּ֕שׁ אֵ֖ת פַּ֣ר הַֽחַטָּ֑את וַיִּסְמֹ֨ךְ אַהֲרֹ֤ן וּבָנָיו֙ אֶת־יְדֵיהֶ֔ם עַל־רֹ֖אשׁ פַּ֥ר הַֽחַטָּֽאת׃ הוּא הִגִּישׁ אֶת פַּר הַחַטָּאת, וְאַהֲרֹן וּבָנָיו הִשְׁעִינוּ אֶת יְדֵיהֶם עַל רֹאשׁ פַּר הַחַטָּאת.
טו וַיִּשְׁחָ֗ט וַיִּקַּ֨ח מֹשֶׁ֤ה אֶת־הַדָּם֙ וַ֠יִּתֵּ֠ן עַל־קַרְנ֨וֹת הַמִּזְבֵּ֤חַ סָבִיב֙ בְּאֶצְבָּע֔וֹ וַיְחַטֵּ֖א אֶת־הַמִּזְבֵּ֑חַ וְאֶת־הַדָּ֗ם יָצַק֙ אֶל־יְס֣וֹד הַמִּזְבֵּ֔חַ וַֽיְקַדְּשֵׁ֖הוּ לְכַפֵּ֥ר עָלָֽיו׃ מֹשֶׁה שָׁחַט אוֹתוֹ וְלָקַח אֶת הַדָּם וְשָׂם בְּאֶצְבָּעוֹ עַל פִּנּוֹת הַמִּזְבֵּחַ מִסָּבִיב וְטִהֵר אֶת הַמִּזְבֵּחַ, וְאֶת שְׁאָר הַדָּם שָׁפַךְ עַל יְסוֹד הַמִּזְבֵּחַ, וְקִדֵּשׁ אוֹתוֹ שֶׁיִּהְיֶה אֶפְשָׁר לְכַפֵּר עָלָיו.
טז וַיִּקַּ֗ח אֶֽת־כׇּל־הַחֵ֘לֶב֮ אֲשֶׁ֣ר עַל־הַקֶּ֒רֶב֒ וְאֵת֙ יֹתֶ֣רֶת הַכָּבֵ֔ד וְאֶת־שְׁתֵּ֥י הַכְּלָיֹ֖ת וְאֶֽת־חֶלְבְּהֶ֑ן וַיַּקְטֵ֥ר מֹשֶׁ֖ה הַמִּזְבֵּֽחָה׃ הוּא לָקַח אֶת כָּל הַחֵלֶב שֶׁעַל הַמֵּעַיִם וְאֶת הַתּוֹסֶפֶת שֶׁעַל הַכָּבֵד וְאֶת שְׁתֵּי הַכְּלָיוֹת וְאֶת הַחֵלֶב שֶׁעֲלֵיהֶן, וּמֹשֶׁה הִקְטִיר אוֹתָם עַל הַמִּזְבֵּחַ.
יז וְאֶת־הַפָּ֤ר וְאֶת־עֹרוֹ֙ וְאֶת־בְּשָׂר֣וֹ וְאֶת־פִּרְשׁ֔וֹ שָׂרַ֣ף בָּאֵ֔שׁ מִח֖וּץ לַֽמַּחֲנֶ֑ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ וְאֶת הַפָּר וְאֶת עוֹרוֹ וְאֶת בְּשָׂרוֹ וְאֶת הָאוֹכֶל הַמְּעֻכָּל שֶׁבּוֹ הוּא שָׂרַף בָּאֵשׁ מִחוּץ לַמַּחֲנֶה, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.
יח וַיַּקְרֵ֕ב אֵ֖ת אֵ֣יל הָעֹלָ֑ה וַֽיִּסְמְכ֞וּ אַהֲרֹ֧ן וּבָנָ֛יו אֶת־יְדֵיהֶ֖ם עַל־רֹ֥אשׁ הָאָֽיִל׃ הוּא הִגִּישׁ אֶת אֵיל הָעוֹלָה, אַהֲרֹן וּבָנָיו הִשְׁעִינוּ אֶת יְדֵיהֶם עַל רֹאשׁ הָאַיִל.
יט וַיִּשְׁחָ֑ט וַיִּזְרֹ֨ק מֹשֶׁ֧ה אֶת־הַדָּ֛ם עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ מֹשֶׁה שָׁחַט אוֹתוֹ וְזָרַק אֶת הַדָּם עַל הַמִּזְבֵּחַ סָבִיב.
כ וְאֶ֨ת־הָאַ֔יִל נִתַּ֖ח לִנְתָחָ֑יו וַיַּקְטֵ֤ר מֹשֶׁה֙ אֶת־הָרֹ֔אשׁ וְאֶת־הַנְּתָחִ֖ים וְאֶת־הַפָּֽדֶר׃ וְחִלֵּק אֶת הָאַיִל לַחֲלָקָיו, וּמֹשֶׁה הִקְטִיר אֶת הָרֹאשׁ וְאֶת הַחֲלָקִים וְאֶת הַחֵלֶב.
כא וְאֶת־הַקֶּ֥רֶב וְאֶת־הַכְּרָעַ֖יִם רָחַ֣ץ בַּמָּ֑יִם וַיַּקְטֵר֩ מֹשֶׁ֨ה אֶת־כׇּל־הָאַ֜יִל הַמִּזְבֵּ֗חָה עֹלָ֨ה ה֤וּא לְרֵֽיחַ־נִיחֹ֙חַ֙ אִשֶּׁ֥ה הוּא֙ לַיהֹוָ֔ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ וְאֶת הַמֵּעַיִם וְאֶת הַכְּרָעַיִם רָחַץ בְּמַיִם, וּמֹשֶׁה הִקְטִיר אֶת כָּל הָאַיִל עַל הַמִּזְבֵּחַ, הוּא עוֹלָה לְנַחַת רוּחַ, הוּא קָרְבָּן עוֹלֶה בָּאֵשׁ לַה', כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

ששי[עריכה]

כב וַיַּקְרֵב֙ אֶת־הָאַ֣יִל הַשֵּׁנִ֔י אֵ֖יל הַמִּלֻּאִ֑ים וַֽיִּסְמְכ֞וּ אַהֲרֹ֧ן וּבָנָ֛יו אֶת־יְדֵיהֶ֖ם עַל־רֹ֥אשׁ הָאָֽיִל׃ הוּא הִגִּישׁ אֶת הָאַיִל הַשֵּׁנִי – אֵיל קָרְבַּן הַמִּנּוּי, אַהֲרֹן וּבָנָיו הִשְׁעִינוּ אֶת יְדֵיהֶם עַל רֹאשׁ הָאַיִל.
כג וַיִּשְׁחָ֓ט ׀ וַיִּקַּ֤ח מֹשֶׁה֙ מִדָּמ֔וֹ וַיִּתֵּ֛ן עַל־תְּנ֥וּךְ אֹֽזֶן־אַהֲרֹ֖ן הַיְמָנִ֑ית וְעַל־בֹּ֤הֶן יָדוֹ֙ הַיְמָנִ֔ית וְעַל־בֹּ֥הֶן רַגְל֖וֹ הַיְמָנִֽית׃ מֹשֶׁה שָׁחַט אוֹתוֹ וְלָקַח מִדָּמוֹ וְשָׂם עַל תְּנוּךְ אוֹזֶן אַהֲרֹן הַיְּמָנִית, וְעַל בּוֹהֶן יָדוֹ הַיְּמָנִית, וְעַל בּוֹהֶן רַגְלוֹ הַיְּמָנִית.
כד וַיַּקְרֵ֞ב אֶת־בְּנֵ֣י אַהֲרֹ֗ן וַיִּתֵּ֨ן מֹשֶׁ֤ה מִן־הַדָּם֙ עַל־תְּנ֤וּךְ אׇזְנָם֙ הַיְמָנִ֔ית וְעַל־בֹּ֤הֶן יָדָם֙ הַיְמָנִ֔ית וְעַל־בֹּ֥הֶן רַגְלָ֖ם הַיְמָנִ֑ית וַיִּזְרֹ֨ק מֹשֶׁ֧ה אֶת־הַדָּ֛ם עַל־הַֽמִּזְבֵּ֖חַ סָבִֽיב׃ הוּא קֵרֵב אֶת בְּנֵי אַהֲרֹן, וּמֹשֶׁה שָׂם מֵהַדָּם עַל תְּנוּךְ אָזְנָם הַיְּמָנִית, וְעַל בּוֹהֶן יָדָם הַיְּמָנִית, וְעַל בּוֹהֶן רַגְלָם הַיְּמָנִית, וּמֹשֶׁה זָרַק אֶת הַדָּם עַל הַמִּזְבֵּחַ מִסָּבִיב.
כה וַיִּקַּ֞ח אֶת־הַחֵ֣לֶב וְאֶת־הָֽאַלְיָ֗ה וְאֶֽת־כׇּל־הַחֵ֘לֶב֮ אֲשֶׁ֣ר עַל־הַקֶּ֒רֶב֒ וְאֵת֙ יֹתֶ֣רֶת הַכָּבֵ֔ד וְאֶת־שְׁתֵּ֥י הַכְּלָיֹ֖ת וְאֶֽת־חֶלְבְּהֶ֑ן וְאֵ֖ת שׁ֥וֹק הַיָּמִֽין׃ הוּא לָקַח אֶת הַחֵלֶב וְאֶת שׁוּמַן הַזָּנָב, וְאֶת כָּל הַחֵלֶב שֶׁעַל הַמֵּעַיִם, וְאֶת הַתּוֹסֶפֶת שֶׁעַל הַכָּבֵד, וְאֶת שְׁתֵּי הַכְּלָיוֹת וְאֶת הַחֵלֶב שֶׁעֲלֵיהֶן, וְאֶת הַשּׁוֹק הַיְּמִינִי.
כו וּמִסַּ֨ל הַמַּצּ֜וֹת אֲשֶׁ֣ר ׀ לִפְנֵ֣י יְהֹוָ֗ה לָ֠קַ֠ח חַלַּ֨ת מַצָּ֤ה אַחַת֙ וְֽחַלַּ֨ת לֶ֥חֶם שֶׁ֛מֶן אַחַ֖ת וְרָקִ֣יק אֶחָ֑ד וַיָּ֙שֶׂם֙ עַל־הַ֣חֲלָבִ֔ים וְעַ֖ל שׁ֥וֹק הַיָּמִֽין׃ וּמִסַּל הַלֶּחֶם שֶׁלֹּא הֶחְמִיץ שֶׁלִּפְנֵי ה' לָקַח חַלָּה שֶׁלֹּא הֶחְמִיצָה אַחַת, וּפִתָּה עָבָה מְטֻגֶּנֶת בְּשֶׁמֶן אַחַת, וּפִתָּה דַּקָּה אַחַת, וְשָׂם עַל גּוּשֵׁי הַחֵלֶב וְעַל הַשּׁוֹק הַיְּמָנִי.
כז וַיִּתֵּ֣ן אֶת־הַכֹּ֔ל עַ֚ל כַּפֵּ֣י אַהֲרֹ֔ן וְעַ֖ל כַּפֵּ֣י בָנָ֑יו וַיָּ֧נֶף אֹתָ֛ם תְּנוּפָ֖ה לִפְנֵ֥י יְהֹוָֽה׃ הוּא שָׂם אֶת הַכֹּל עַל כַּפּוֹת הַיָּדַיִם שֶׁל אַהֲרֹן וְשֶׁל בָּנָיו, וְהֵנִיף אוֹתָם לִפְנֵי ה'.
כח וַיִּקַּ֨ח מֹשֶׁ֤ה אֹתָם֙ מֵעַ֣ל כַּפֵּיהֶ֔ם וַיַּקְטֵ֥ר הַמִּזְבֵּ֖חָה עַל־הָעֹלָ֑ה מִלֻּאִ֥ים הֵם֙ לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֥ה ה֖וּא לַיהֹוָֽה׃ מֹשֶׁה לָקַח אוֹתָם מֵעַל כַּפּוֹת יְדֵיהֶם וְהִקְטִיר עַל הַמִּזְבֵּחַ עַל הָעוֹלָה, הֵם קָרְבָּן לְמִנּוּי לְתַפְקִיד, כְּדֵי לַעֲשׂוֹת נַחַת רוּחַ, זֶה קָרְבָּן עוֹלֶה בָּאֵשׁ לַה'.
כט וַיִּקַּ֤ח מֹשֶׁה֙ אֶת־הֶ֣חָזֶ֔ה וַיְנִיפֵ֥הוּ תְנוּפָ֖ה לִפְנֵ֣י יְהֹוָ֑ה מֵאֵ֣יל הַמִּלֻּאִ֗ים לְמֹשֶׁ֤ה הָיָה֙ לְמָנָ֔ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ מֹשֶׁה לָקַח אֶת הֶחָזֶה וְהֵנִיף אוֹתוֹ לִפְנֵי ה' – הוּא הָיָה מָנָה לְמֹשֶׁה מִקָּרְבַּן הָאַיִל שֶׁל הַמִּנּוּי, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

שביעי[עריכה]

ל וַיִּקַּ֨ח מֹשֶׁ֜ה מִשֶּׁ֣מֶן הַמִּשְׁחָ֗ה וּמִן־הַדָּם֮ אֲשֶׁ֣ר עַל־הַמִּזְבֵּ֒חַ֒ וַיַּ֤ז עַֽל־אַהֲרֹן֙ עַל־בְּגָדָ֔יו וְעַל־בָּנָ֛יו וְעַל־בִּגְדֵ֥י בָנָ֖יו אִתּ֑וֹ וַיְקַדֵּ֤שׁ אֶֽת־אַהֲרֹן֙ אֶת־בְּגָדָ֔יו וְאֶת־בָּנָ֛יו וְאֶת־בִּגְדֵ֥י בָנָ֖יו אִתּֽוֹ׃ מֹשֶׁה לָקַח מֵהַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ וּמֵהַדָּם שֶׁעַל הַמִּזְבֵּחַ, וְהִתִּיז עַל אַהֲרֹן וְעַל בְּגָדָיו וְעַל בָּנָיו וְעַל בִּגְדֵי בָּנָיו אִתּוֹ, וְקִדֵּשׁ אֶת אַהֲרֹן וְאֶת בְּגָדָיו וְאֶת בָּנָיו וְאֶת בִּגְדֵי בָּנָיו אִתּוֹ.
לא וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶל־אַהֲרֹ֣ן וְאֶל־בָּנָ֗יו בַּשְּׁל֣וּ אֶת־הַבָּשָׂר֮ פֶּ֣תַח אֹ֣הֶל מוֹעֵד֒ וְשָׁם֙ תֹּאכְל֣וּ אֹת֔וֹ וְאֶ֨ת־הַלֶּ֔חֶם אֲשֶׁ֖ר בְּסַ֣ל הַמִּלֻּאִ֑ים כַּאֲשֶׁ֤ר צִוֵּ֙יתִי֙ לֵאמֹ֔ר אַהֲרֹ֥ן וּבָנָ֖יו יֹאכְלֻֽהוּ׃ מֹשֶׁה אָמַר לְאַהֲרֹן וּלְבָנָיו: בַּשְּׁלוּ אֶת הַבָּשָׂר לִפְנֵי פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת וְשָׁם תֹּאכְלוּ אוֹתוֹ וְאֶת הַלֶּחֶם שֶׁבְּסַל קָרְבַּן הַמִּנּוּי, כְּמוֹ שֶׁצִּוִּיתִי: שֶׁאַהֲרֹן וּבָנָיו יֹאכְלוּ אוֹתוֹ.
לב וְהַנּוֹתָ֥ר בַּבָּשָׂ֖ר וּבַלָּ֑חֶם בָּאֵ֖שׁ תִּשְׂרֹֽפוּ׃ וְאֶת מַה שֶּׁיִּשָּׁאֵר מֵהַבָּשָׂר וּמֵהַלֶּחֶם תִּשְׂרְפוּ בָּאֵשׁ.

מפטיר[עריכה]

לג וּמִפֶּ֩תַח֩ אֹ֨הֶל מוֹעֵ֜ד לֹ֤א תֵֽצְאוּ֙ שִׁבְעַ֣ת יָמִ֔ים עַ֚ד י֣וֹם מְלֹ֔את יְמֵ֖י מִלֻּאֵיכֶ֑ם כִּ֚י שִׁבְעַ֣ת יָמִ֔ים יְמַלֵּ֖א אֶת־יֶדְכֶֽם׃ וְאַל תֵּצְאוּ מִפֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת שִׁבְעָה יָמִים – עַד לְיוֹם הַשְׁלָמַת יְמֵי הַמִּנּוּי שֶׁלָּכֶם, כִּי שִׁבְעָה יָמִים יִהְיֶה טֶקֶס הַמִּנּוּי שֶׁלָּכֶם.
לד כַּאֲשֶׁ֥ר עָשָׂ֖ה בַּיּ֣וֹם הַזֶּ֑ה צִוָּ֧ה יְהֹוָ֛ה לַעֲשֹׂ֖ת לְכַפֵּ֥ר עֲלֵיכֶֽם׃ כְּמוֹ שֶׁנַּעֲשָׂה בַּיּוֹם הַזֶּה – כָּךְ ה' צִוָּה לַעֲשׂוֹת שִׁבְעָה יָמִים כְּדֵי לְכַפֵּר עֲלֵיכֶם.
לה וּפֶ֩תַח֩ אֹ֨הֶל מוֹעֵ֜ד תֵּשְׁב֨וּ יוֹמָ֤ם וָלַ֙יְלָה֙ שִׁבְעַ֣ת יָמִ֔ים וּשְׁמַרְתֶּ֛ם אֶת־מִשְׁמֶ֥רֶת יְהֹוָ֖ה וְלֹ֣א תָמ֑וּתוּ כִּי־כֵ֖ן צֻוֵּֽיתִי׃ וְשִׁבְעָה יָמִים תֵּשְׁבוּ בַּיּוֹם וּבַלַּיְלָה בְּפֶתַח אוֹהֶל הַהִתְכַּנְּסוּת, וּתְשָׁרְתוּ בְּמִשְׁמֶרֶת ה' – וְכָךְ לֹא תָּמוּתוּ, כִּי כָּךְ הִצְטַוֵּיתִי.
לו וַיַּ֥עַשׂ אַהֲרֹ֖ן וּבָנָ֑יו אֵ֚ת כׇּל־הַדְּבָרִ֔ים אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה בְּיַד־מֹשֶֽׁה׃ אַהֲרֹן וּבָנָיו עָשׂוּ אֶת כָּל הַדְּבָרִים שֶׁה' צִוָּה בְּאֶמְצָעוּת מֹשֶׁה.
