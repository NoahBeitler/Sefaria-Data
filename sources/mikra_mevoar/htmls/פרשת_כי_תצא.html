ראשון[עריכה]

ס פָּרָשַׁת אֵשֶׁת יְפַת תּוֹאַר
כא י כִּֽי־תֵצֵ֥א לַמִּלְחָמָ֖ה עַל־אֹיְבֶ֑יךָ וּנְתָנ֞וֹ יְהֹוָ֧ה אֱלֹהֶ֛יךָ בְּיָדֶ֖ךָ וְשָׁבִ֥יתָ שִׁבְיֽוֹ׃ אִם תֵּצֵא לַמִּלְחָמָה עַל אוֹיְבֶיךָ וַה' אֱלֹהֶיךָ יִתֵּן אוֹתָם בְּיָדְךָ וְתִלְכּוֹד מֵהֶם שְׁבוּיִים.
יא וְרָאִ֙יתָ֙ בַּשִּׁבְיָ֔ה אֵ֖שֶׁת יְפַת־תֹּ֑אַר וְחָשַׁקְתָּ֣ בָ֔הּ וְלָקַחְתָּ֥ לְךָ֖ לְאִשָּֽׁה׃ וְתִרְאֶה בַּשֶּׁבִי אִשָּׁה יְפַת תּוֹאַר וְתַחֲשׁוֹק בָּהּ, תִּקַּח אוֹתָהּ לְךָ לְאִשָּׁה.
יב וַהֲבֵאתָ֖הּ אֶל־תּ֣וֹךְ בֵּיתֶ֑ךָ וְגִלְּחָה֙ אֶת־רֹאשָׁ֔הּ וְעָשְׂתָ֖ה אֶת־צִפׇּרְנֶֽיהָ׃ תָּבִיא אוֹתָהּ לְתוֹךְ בֵּיתְךָ וְהִיא תְּגַלֵּחַ אֶת רֹאשָׁהּ וְתִקְצוֹץ אֶת צִפָּרְנֶיהָ.
יג וְהֵסִ֩ירָה֩ אֶת־שִׂמְלַ֨ת שִׁבְיָ֜הּ מֵעָלֶ֗יהָ וְיָֽשְׁבָה֙ בְּבֵיתֶ֔ךָ וּבָ֥כְתָ֛ה אֶת־אָבִ֥יהָ וְאֶת־אִמָּ֖הּ יֶ֣רַח יָמִ֑ים וְאַ֨חַר כֵּ֜ן תָּב֤וֹא אֵלֶ֙יהָ֙ וּבְעַלְתָּ֔הּ וְהָיְתָ֥ה לְךָ֖ לְאִשָּֽׁה׃ הִיא תִּפְשׁוֹט מֵעָלֶיהָ אֶת הַבְּגָדִים שֶׁנִּשְׁבְּתָה בָּהֶם וְתָגוּר בְּבֵיתְךָ, וְתִבְכֶּה עַל אָבִיהָ וְעַל אִמָּהּ חוֹדֶשׁ יָמִים, וְאַחַר כָּךְ תָּבוֹא אֵלֶיהָ וְתִשְׁכַּב אִתָּהּ וְהִיא תִּהְיֶה לְךָ לְאִשָּׁה.
יד וְהָיָ֞ה אִם־לֹ֧א חָפַ֣צְתָּ בָּ֗הּ וְשִׁלַּחְתָּהּ֙ לְנַפְשָׁ֔הּ וּמָכֹ֥ר לֹא־תִמְכְּרֶ֖נָּה בַּכָּ֑סֶף לֹא־תִתְעַמֵּ֣ר בָּ֔הּ תַּ֖חַת אֲשֶׁ֥ר עִנִּיתָֽהּ׃ וְאִם לֹא תַּחְפּוֹץ בָּהּ תְּשַׁחְרֵר אוֹתָהּ לְנַפְשָׁהּ וְאַל תִּמְכּוֹר אוֹתָהּ תְּמוּרַת כֶּסֶף, אַל תִּסְחוֹר בָּהּ כִּי הִשְׁפַּלְתָּ אוֹתָהּ.
ס פָּרָשַׁת הַבְּכוֹר
טו כִּֽי־תִהְיֶ֨יןָ לְאִ֜ישׁ שְׁתֵּ֣י נָשִׁ֗ים הָאַחַ֤ת אֲהוּבָה֙ וְהָאַחַ֣ת שְׂנוּאָ֔ה וְיָֽלְדוּ־ל֣וֹ בָנִ֔ים הָאֲהוּבָ֖ה וְהַשְּׂנוּאָ֑ה וְהָיָ֛ה הַבֵּ֥ן הַבְּכֹ֖ר לַשְּׂנִיאָֽה׃ אִם תִּהְיֶינָה לְאִישׁ שְׁתֵּי נָשִׁים הָאַחַת אֲהוּבָה וְהָאַחַת שְׂנוּאָה, וְהֵן יֵלְדוּ לוֹ בָּנִים – הָאֲהוּבָה וְהַשְּׂנוּאָה, וְהַבֵּן הַבְּכוֹר יִהְיֶה שֶׁל הַשְּׂנוּאָה.
טז וְהָיָ֗ה בְּיוֹם֙ הַנְחִיל֣וֹ אֶת־בָּנָ֔יו אֵ֥ת אֲשֶׁר־יִהְיֶ֖ה ל֑וֹ לֹ֣א יוּכַ֗ל לְבַכֵּר֙ אֶת־בֶּן־הָ֣אֲהוּבָ֔ה עַל־פְּנֵ֥י בֶן־הַשְּׂנוּאָ֖ה הַבְּכֹֽר׃ בַּיּוֹם שֶׁיַּנְחִיל לְבָנָיו אֶת הָרְכוּשׁ שֶׁלּוֹ, הוּא לֹא יוּכַל לְהַעֲדִיף אֶת הַבֵּן שֶׁל הָאֲהוּבָה עַל פְּנֵי הַבֵּן שֶׁל הַשְּׂנוּאָה שֶׁהוּא הַבְּכוֹר.
יז כִּי֩ אֶת־הַבְּכֹ֨ר בֶּן־הַשְּׂנוּאָ֜ה יַכִּ֗יר לָ֤תֶת לוֹ֙ פִּ֣י שְׁנַ֔יִם בְּכֹ֥ל אֲשֶׁר־יִמָּצֵ֖א ל֑וֹ כִּי־הוּא֙ רֵאשִׁ֣ית אֹנ֔וֹ ל֖וֹ מִשְׁפַּ֥ט הַבְּכֹרָֽה׃ אֶלָּא יַכִּיר בִּזְכוּתוֹ שֶׁל הַבְּכוֹר בֶּן הַשְּׂנוּאָה וְיִתֵּן לוֹ פִּי שְׁנַיִם מִכָּל מַה שֶּׁיֵּשׁ לוֹ, כִּי הוּא הָרִאשׁוֹן שֶׁנּוֹלַד מִכּוֹחוֹ – לוֹ דִּין הַבְּכוֹרָה.
ס פָּרָשַׁת בֵּן סוֹרֵר וּמוֹרֶה
יח כִּֽי־יִהְיֶ֣ה לְאִ֗ישׁ בֵּ֚ן סוֹרֵ֣ר וּמוֹרֶ֔ה אֵינֶ֣נּוּ שֹׁמֵ֔עַ בְּק֥וֹל אָבִ֖יו וּבְק֣וֹל אִמּ֑וֹ וְיִסְּר֣וּ אֹת֔וֹ וְלֹ֥א יִשְׁמַ֖ע אֲלֵיהֶֽם׃ אִם יִהְיֶה לְאִישׁ בֵּן לֹא מְמֻשְׁמָע וּמוֹרֵד – שֶׁאֵינוֹ שׁוֹמֵעַ בְּקוֹל אָבִיו וּבְקוֹל אִמּוֹ, וְהֵם יְיַסְּרוּ אוֹתוֹ וְהוּא לֹא יִשְׁמַע אֲלֵיהֶם.
יט וְתָ֥פְשׂוּ ב֖וֹ אָבִ֣יו וְאִמּ֑וֹ וְהוֹצִ֧יאוּ אֹת֛וֹ אֶל־זִקְנֵ֥י עִיר֖וֹ וְאֶל־שַׁ֥עַר מְקֹמֽוֹ׃ אָבִיו וְאִמּוֹ יִתְפְּשׂוּ אוֹתוֹ וְיוֹצִיאוּ אוֹתוֹ אֶל זִקְנֵי עִירוֹ וּלְבֵית הַדִּין שֶׁל עִיר מְגוּרָיו.
כ וְאָמְר֞וּ אֶל־זִקְנֵ֣י עִיר֗וֹ בְּנֵ֤נוּ זֶה֙ סוֹרֵ֣ר וּמֹרֶ֔ה אֵינֶ֥נּוּ שֹׁמֵ֖עַ בְּקֹלֵ֑נוּ זוֹלֵ֖ל וְסֹבֵֽא׃ וְיֹאמְרוּ לְזִקְנֵי עִירוֹ: בְּנֵנוּ זֶה אֵינוֹ מְמֻשְׁמָע וּמוֹרֵד, אֵינוֹ שׁוֹמֵעַ בְּקוֹלֵנוּ, זוֹלֵל וּמִשְׁתַּכֵּר.
כא וּ֠רְגָמֻ֠הוּ כׇּל־אַנְשֵׁ֨י עִיר֤וֹ בָֽאֲבָנִים֙ וָמֵ֔ת וּבִֽעַרְתָּ֥ הָרָ֖ע מִקִּרְבֶּ֑ךָ וְכׇל־יִשְׂרָאֵ֖ל יִשְׁמְע֥וּ וְיִרָֽאוּ׃ כָּל אַנְשֵׁי עִירוֹ יִרְגְּמוּ אוֹתוֹ בַּאֲבָנִים וְהוּא יָמוּת, וּתְבַעֵר אֶת הָרָע מִתּוֹכְךָ, וְכָל יִשְׂרָאֵל יִשְׁמְעוּ וְיִפְחֲדוּ.

שני[עריכה]

ס פָּרָשַׁת הֲלָנַת הַתָּלוּי
כב וְכִֽי־יִהְיֶ֣ה בְאִ֗ישׁ חֵ֛טְא מִשְׁפַּט־מָ֖וֶת וְהוּמָ֑ת וְתָלִ֥יתָ אֹת֖וֹ עַל־עֵֽץ׃ וְאִם אִישׁ יִתְחַיֵּב בְּמָוֶת וְיוּמַת, תִּתְלֶה אוֹתוֹ עַל עֵץ.
כג לֹא־תָלִ֨ין נִבְלָת֜וֹ עַל־הָעֵ֗ץ כִּֽי־קָב֤וֹר תִּקְבְּרֶ֙נּוּ֙ בַּיּ֣וֹם הַה֔וּא כִּֽי־קִלְלַ֥ת אֱלֹהִ֖ים תָּל֑וּי וְלֹ֤א תְטַמֵּא֙ אֶת־אַדְמָ֣תְךָ֔ אֲשֶׁר֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ נֹתֵ֥ן לְךָ֖ נַחֲלָֽה׃ אַל תַּשְׁאִיר אֶת נִבְלָתוֹ בַּלַּיְלָה עַל הָעֵץ, אֶלָּא קְבוֹר אוֹתוֹ בְּאוֹתוֹ יוֹם, כִּי אָדָם תָּלוּי הוּא זִלְזוּל בֵּאלֹהִים, וְאַל תְּטַמֵּא אֶת אַדְמָתְךָ שֶׁה' אֱלֹהֶיךָ נוֹתֵן לְךָ לְנַחֲלָה.
ס פָּרָשַׁת הֲשָׁבַת אֲבֵדָה, וּטְעִינָה
כב א לֹֽא־תִרְאֶה֩ אֶת־שׁ֨וֹר אָחִ֜יךָ א֤וֹ אֶת־שֵׂיוֹ֙ נִדָּחִ֔ים וְהִתְעַלַּמְתָּ֖ מֵהֶ֑ם הָשֵׁ֥ב תְּשִׁיבֵ֖ם לְאָחִֽיךָ׃ אִם תִּרְאֶה אֶת שׁוֹר אָחִיךָ אוֹ אֶת הַשֶּׂה שֶׁלּוֹ תּוֹעִים בַּדֶּרֶךְ, אַל תִּתְעַלֵּם מֵהֶם, אֶלָּא תַּחֲזִיר אוֹתָם לְאָחִיךָ.
ב וְאִם־לֹ֨א קָר֥וֹב אָחִ֛יךָ אֵלֶ֖יךָ וְלֹ֣א יְדַעְתּ֑וֹ וַאֲסַפְתּוֹ֙ אֶל־תּ֣וֹךְ בֵּיתֶ֔ךָ וְהָיָ֣ה עִמְּךָ֗ עַ֣ד דְּרֹ֤שׁ אָחִ֙יךָ֙ אֹת֔וֹ וַהֲשֵׁבֹת֖וֹ לֽוֹ׃ וְאִם אָחִיךָ אֵינוֹ קָרוֹב אֵלֶיךָ אוֹ שֶׁאֵינְךָ מַכִּיר אוֹתוֹ, תַּאֲסוֹף אוֹתוֹ לְתוֹךְ בֵּיתְךָ וְיִהְיֶה אִתְּךָ עַד שֶׁאָחִיךָ יִדְרוֹשׁ אוֹתוֹ וְתַחֲזִיר לוֹ אוֹתוֹ.
ג וְכֵ֧ן תַּעֲשֶׂ֣ה לַחֲמֹר֗וֹ וְכֵ֣ן תַּעֲשֶׂה֮ לְשִׂמְלָתוֹ֒ וְכֵ֣ן תַּעֲשֶׂ֗ה לְכׇל־אֲבֵדַ֥ת אָחִ֛יךָ אֲשֶׁר־תֹּאבַ֥ד מִמֶּ֖נּוּ וּמְצָאתָ֑הּ לֹ֥א תוּכַ֖ל לְהִתְעַלֵּֽם׃ וְכָךְ תַּעֲשֶׂה לַחֲמוֹרוֹ וְכָךְ תַּעֲשֶׂה לְבִגְדּוֹ וְכָךְ תַּעֲשֶׂה לְכָל אֲבֵדָה שֶׁל אָחִיךָ שֶׁתֹּאבַד מִמֶּנּוּ וְתִמְצָא אוֹתָהּ – אֵינְךָ רַשַּׁאי לְהִתְעַלֵּם.
ס
ד לֹא־תִרְאֶה֩ אֶת־חֲמ֨וֹר אָחִ֜יךָ א֤וֹ שׁוֹרוֹ֙ נֹפְלִ֣ים בַּדֶּ֔רֶךְ וְהִתְעַלַּמְתָּ֖ מֵהֶ֑ם הָקֵ֥ם תָּקִ֖ים עִמּֽוֹ׃ אִם תִּרְאֶה חֲמוֹר שֶׁל אָחִיךָ אוֹ שׁוֹרוֹ נוֹפְלִים בַּדֶּרֶךְ אַל תִּתְעַלֵּם מֵהֶם, אֶלָּא הֲקֵם אוֹתָם אִתּוֹ.
ס פָּרָשַׁת לֹא יִלְבַּשׁ

ס פָּרָשַׁת מוֹצִיא שֵׁם רָע
יג כִּֽי־יִקַּ֥ח אִ֖ישׁ אִשָּׁ֑ה וּבָ֥א אֵלֶ֖יהָ וּשְׂנֵאָֽהּ׃ אִם אִישׁ יְקַדֵּשׁ אִשָּׁה וְיִשְׁכַּב אִתָּהּ וְיִשְׂנָא אוֹתָהּ.

טו וְלָקַ֛ח אֲבִ֥י הַֽנַּעֲרָ֖ וְאִמָּ֑הּ וְהוֹצִ֜יאוּ אֶת־בְּתוּלֵ֧י הַֽנַּעֲרָ֛ אֶל־זִקְנֵ֥י הָעִ֖יר הַשָּֽׁעְרָה׃ אֲבִי הַנַּעֲרָה וְאִמָּהּ יִקְחוּ אֶת דַּם הַבְּתוּלִים שֶׁל הַנַּעֲרָה וְיוֹצִיאוּ לְבֵית הַדִּין אֶל זִקְנֵי הָעִיר.
טז וְאָמַ֛ר אֲבִ֥י הַֽנַּעֲרָ֖ אֶל־הַזְּקֵנִ֑ים אֶת־בִּתִּ֗י נָתַ֜תִּי לָאִ֥ישׁ הַזֶּ֛ה לְאִשָּׁ֖ה וַיִּשְׂנָאֶֽהָ׃ וַאֲבִי הַנַּעֲרָה יֹאמַר לַזְּקֵנִים: נָתַתִּי אֶת בִּתִּי לָאִישׁ הַזֶּה לְאִשָּׁה וְהוּא שׂוֹנֵא אוֹתָהּ.
יז וְהִנֵּה־ה֡וּא שָׂם֩ עֲלִילֹ֨ת דְּבָרִ֜ים לֵאמֹ֗ר לֹֽא־מָצָ֤אתִי לְבִתְּךָ֙ בְּתוּלִ֔ים וְאֵ֖לֶּה בְּתוּלֵ֣י בִתִּ֑י וּפָֽרְשׂוּ֙ הַשִּׂמְלָ֔ה לִפְנֵ֖י זִקְנֵ֥י הָעִֽיר׃ וְהוּא הֶעֱלִיל עָלֶיהָ עֲלִילוֹת וְאָמַר: לֹא מָצָאתִי לְבִתְּךָ דַּם בְּתוּלִים. וְהִנֵּה דַּם הַבְּתוּלִים שֶׁל בִּתִּי! וְיִפְרְשׂוּ אֶת הַשִּׂמְלָה שֶׁלָּהּ לִפְנֵי זִקְנֵי הָעִיר.
יח וְלָ֥קְח֛וּ זִקְנֵ֥י הָֽעִיר־הַהִ֖וא אֶת־הָאִ֑ישׁ וְיִסְּר֖וּ אֹתֽוֹ׃ זִקְנֵי הָעִיר הַהִיא יִקְחוּ אֶת הָאִישׁ וִייַסְּרוּ אוֹתוֹ.
יט וְעָנְשׁ֨וּ אֹת֜וֹ מֵ֣אָה כֶ֗סֶף וְנָתְנוּ֙ לַאֲבִ֣י הַֽנַּעֲרָ֔ה כִּ֤י הוֹצִיא֙ שֵׁ֣ם רָ֔ע עַ֖ל בְּתוּלַ֣ת יִשְׂרָאֵ֑ל וְלֽוֹ־תִהְיֶ֣ה לְאִשָּׁ֔ה לֹא־יוּכַ֥ל לְשַׁלְּחָ֖הּ כׇּל־יָמָֽיו׃ וְיַעֲנִישׁוּ אוֹתוֹ לְשַׁלֵּם כֶּסֶף בְּמִשְׁקַל מֵאָה שְׁקָלִים (1.41 קִילוֹ) וְיִתְּנוּ לַאֲבִי הַנַּעֲרָה כִּי הוּא הוֹצִיא שֵׁם רָע עַל בְּתוּלָה בְּיִשְׂרָאֵל, וְאִם הִיא תִּרְצֶה הִיא תִּהְיֶה לוֹ לְאִשָּׁה – הוּא לֹא יוּכַל לְגָרֵשׁ אוֹתָהּ בְּכָל חַיָּיו.

טו כִּי֩ יְהֹוָ֨ה אֱלֹהֶ֜יךָ מִתְהַלֵּ֣ךְ ׀ בְּקֶ֣רֶב מַחֲנֶ֗ךָ לְהַצִּֽילְךָ֙ וְלָתֵ֤ת אֹיְבֶ֙יךָ֙ לְפָנֶ֔יךָ וְהָיָ֥ה מַחֲנֶ֖יךָ קָד֑וֹשׁ וְלֹֽא־יִרְאֶ֤ה בְךָ֙ עֶרְוַ֣ת דָּבָ֔ר וְשָׁ֖ב מֵאַחֲרֶֽיךָ׃ כִּי ה' אֱלֹהֶיךָ מִתְהַלֵּךְ בְּתוֹךְ הַמַּחֲנֶה שֶׁלְּךָ לְהַצִּיל אוֹתְךָ וְלִמְסוֹר בְּיָדְךָ אֶת אוֹיְבֶיךָ, לָכֵן הַמַּחֲנֶה שֶׁלְּךָ יִהְיֶה קָדוֹשׁ, וְהוּא לֹא יִרְאֶה בְּךָ דָּבָר מְגֻנֶּה וְיַעֲזוֹב אוֹתְךָ.

טז לֹא־תַסְגִּ֥יר עֶ֖בֶד אֶל־אֲדֹנָ֑יו אֲשֶׁר־יִנָּצֵ֥ל אֵלֶ֖יךָ מֵעִ֥ם אֲדֹנָֽיו׃ אַל תַּסְגִּיר אֶל אֲדוֹנוֹ עֶבֶד מֵעַם אַחֵר שֶׁיִּנָּצֵל מֵאֲדוֹנוֹ וְיִבְרַח אֵלֶיךָ לָאָרֶץ.
יז עִמְּךָ֞ יֵשֵׁ֣ב בְּקִרְבְּךָ֗ בַּמָּק֧וֹם אֲשֶׁר־יִבְחַ֛ר בְּאַחַ֥ד שְׁעָרֶ֖יךָ בַּטּ֣וֹב ל֑וֹ לֹ֖א תּוֹנֶֽנּוּ׃ אֶלָּא הוּא יָגוּר אֶצְלְךָ בַּמָּקוֹם שֶׁיִּבְחַר בְּאַחַת מֵעָרֶיךָ בַּמָּקוֹם שֶׁטּוֹב לוֹ, אַל תְּצַעֵר אוֹתוֹ.
ס פָּרָשַׁת קָדֵשׁ וְאֶתְנַן
יח לֹא־תִהְיֶ֥ה קְדֵשָׁ֖ה מִבְּנ֣וֹת יִשְׂרָאֵ֑ל וְלֹֽא־יִהְיֶ֥ה קָדֵ֖שׁ מִבְּנֵ֥י יִשְׂרָאֵֽל׃ לֹא תִּהְיֶה זוֹנָה מִבְּנוֹת יִשְׂרָאֵל וְלֹא יִהְיֶה זוֹנֶה מִבְּנֵי יִשְׂרָאֵל.
יט לֹא־תָבִיא֩ אֶתְנַ֨ן זוֹנָ֜ה וּמְחִ֣יר כֶּ֗לֶב בֵּ֛ית יְהֹוָ֥ה אֱלֹהֶ֖יךָ לְכׇל־נֶ֑דֶר כִּ֧י תוֹעֲבַ֛ת יְהֹוָ֥ה אֱלֹהֶ֖יךָ גַּם־שְׁנֵיהֶֽם׃ אַל תַּקְרִיב לְבֵית ה' אֱלֹהֶיךָ כְּתַשְׁלוּם עֲבוּר כָּל נֶדֶר בְּהֵמָה שֶׁנִּתְּנָה כְּתַשְׁלוּם לְזוֹנָה אוֹ שֶׁהֻחְלְפָה תְּמוּרַת כֶּלֶב, כִּי שְׁנֵיהֶם תּוֹעֵבָה בְּעֵינֵי ה' אֱלֹהֶיךָ.
ס פָּרָשַׁת הַנֶּשֶׁךְ
כ לֹא־תַשִּׁ֣יךְ לְאָחִ֔יךָ נֶ֥שֶׁךְ כֶּ֖סֶף נֶ֣שֶׁךְ אֹ֑כֶל נֶ֕שֶׁךְ כׇּל־דָּבָ֖ר אֲשֶׁ֥ר יִשָּֽׁךְ׃ אַל תַּלְוֶה בְּרִבִּית לְאָחִיךָ, בֵּין רִבִּית שֶׁל כֶּסֶף, בֵּין רִבִּית שֶׁל אוֹכֶל, אוֹ רִבִּית שֶׁל כָּל דָּבָר שֶׁיִּגָּבֶה יוֹתֵר מֵהַחוֹב.
כא לַנׇּכְרִ֣י תַשִּׁ֔יךְ וּלְאָחִ֖יךָ לֹ֣א תַשִּׁ֑יךְ לְמַ֨עַן יְבָרֶכְךָ֜ יְהֹוָ֣ה אֱלֹהֶ֗יךָ בְּכֹל֙ מִשְׁלַ֣ח יָדֶ֔ךָ עַל־הָאָ֕רֶץ אֲשֶׁר־אַתָּ֥ה בָא־שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ לְנָכְרִי תּוּכַל לְהַלְווֹת בְּרִבִּית וּלְאָחִיךָ אַל תַּלְוֶה בְּרִבִּית, כְּדֵי שֶׁה' אֱלֹהֶיךָ יְבָרֵךְ אוֹתְךָ בְּכָל מַה שֶּׁתַּעֲשֶׂה בָּאָרֶץ שֶׁאַתָּה בָּא לְשָׁם לִכְבּוֹשׁ אוֹתָהּ.
ס פָּרָשַׁת נְדָרִים
כב כִּֽי־תִדֹּ֥ר נֶ֙דֶר֙ לַיהֹוָ֣ה אֱלֹהֶ֔יךָ לֹ֥א תְאַחֵ֖ר לְשַׁלְּמ֑וֹ כִּֽי־דָרֹ֨שׁ יִדְרְשֶׁ֜נּוּ יְהֹוָ֤ה אֱלֹהֶ֙יךָ֙ מֵֽעִמָּ֔ךְ וְהָיָ֥ה בְךָ֖ חֵֽטְא׃ אִם תִּדּוֹר נֶדֶר לַה' אֱלֹהֶיךָ אַל תִּתְעַכֵּב מִלְּשַׁלֵּם אוֹתוֹ, כִּי ה' אֱלֹהֶיךָ יִדְרוֹשׁ אוֹתוֹ מִמְּךָ וְיִהְיֶה בְּךָ חֵטְא.
כג וְכִ֥י תֶחְדַּ֖ל לִנְדֹּ֑ר לֹֽא־יִהְיֶ֥ה בְךָ֖ חֵֽטְא׃ וְאִם לֹא תִּדּוֹר לֹא יִהְיֶה בְּךָ חֵטְא.
כד מוֹצָ֥א שְׂפָתֶ֖יךָ תִּשְׁמֹ֣ר וְעָשִׂ֑יתָ כַּאֲשֶׁ֨ר נָדַ֜רְתָּ לַיהֹוָ֤ה אֱלֹהֶ֙יךָ֙ נְדָבָ֔ה אֲשֶׁ֥ר דִּבַּ֖רְתָּ בְּפִֽיךָ׃ הַקְפֵּד לְקַיֵּם אֶת מַה שֶׁיּוֹצֵא מִשְּׂפָתֶיךָ – כְּמוֹ שֶׁנָּדַרְתָּ נְדָבָה לַה' אֱלֹהֶיךָ, שֶׁדִּבַּרְתָּ בְּפִיךָ.

חמישי[עריכה]

ס פָּרָשַׁת פּוֹעֵל
כה כִּ֤י תָבֹא֙ בְּכֶ֣רֶם רֵעֶ֔ךָ וְאָכַלְתָּ֧ עֲנָבִ֛ים כְּנַפְשְׁךָ֖ שׇׂבְעֶ֑ךָ וְאֶֽל־כֶּלְיְךָ֖ לֹ֥א תִתֵּֽן׃ אִם תָּבוֹא לִהְיוֹת פּוֹעֵל בְּכֶרֶם חֲבֵרְךָ, אֱכוֹל עֲנָבִים כִּרְצוֹנְךָ עַד שֶׁתִּשְׂבַּע, אַךְ אַל תָּשִׂים בְּתִיקְךָ.
ס
כו כִּ֤י תָבֹא֙ בְּקָמַ֣ת רֵעֶ֔ךָ וְקָטַפְתָּ֥ מְלִילֹ֖ת בְּיָדֶ֑ךָ וְחֶרְמֵשׁ֙ לֹ֣א תָנִ֔יף עַ֖ל קָמַ֥ת רֵעֶֽךָ׃ אִם תָּבוֹא כְּפוֹעֵל לַעֲבוֹד בִּתְבוּאָה שֶׁל חֲבֵרְךָ, קְטוֹף שִׁבּוֹלִים בְּיָדְךָ וְתֹאכַל שָׁם, וְאַל תָּנִיף מַגָּל עַל שִׁבּוֹלֵי חֲבֵרְךָ כְּדֵי לָקַחַת.
ס פָּרָשַׁת גְּרוּשָׁה
כד א כִּֽי־יִקַּ֥ח אִ֛ישׁ אִשָּׁ֖ה וּבְעָלָ֑הּ וְהָיָ֞ה אִם־לֹ֧א תִמְצָא־חֵ֣ן בְּעֵינָ֗יו כִּי־מָ֤צָא בָהּ֙ עֶרְוַ֣ת דָּבָ֔ר וְכָ֨תַב לָ֜הּ סֵ֤פֶר כְּרִיתֻת֙ וְנָתַ֣ן בְּיָדָ֔הּ וְשִׁלְּחָ֖הּ מִבֵּיתֽוֹ׃ אִם אִישׁ יִשָּׂא אִשָּׁה וְיִהְיֶה בַּעֲלָהּ, אִם הִיא לֹא תִּמְצָא חֵן בְּעֵינָיו כִּי הוּא יִרְאֶה בָּהּ דָּבָר מְגֻנֶּה, הוּא יִכְתּוֹב לָהּ מִכְתַּב גֵּרוּשִׁין, יִתֵּן בְּיָדָהּ, וְיוֹצִיא אוֹתָהּ מִבֵּיתוֹ.
ב וְיָצְאָ֖ה מִבֵּית֑וֹ וְהָלְכָ֖ה וְהָיְתָ֥ה לְאִישׁ־אַחֵֽר׃ הִיא תֵּצֵא מִבֵּיתוֹ וְתֵלֵךְ וְתִנָּשֵׂא לְאִישׁ אַחֵר.
ג וּשְׂנֵאָהּ֮ הָאִ֣ישׁ הָאַחֲרוֹן֒ וְכָ֨תַב לָ֜הּ סֵ֤פֶר כְּרִיתֻת֙ וְנָתַ֣ן בְּיָדָ֔הּ וְשִׁלְּחָ֖הּ מִבֵּית֑וֹ א֣וֹ כִ֤י יָמוּת֙ הָאִ֣ישׁ הָאַחֲר֔וֹן אֲשֶׁר־לְקָחָ֥הּ ל֖וֹ לְאִשָּֽׁה׃ אִם הָאִישׁ הָאַחֲרוֹן יִשְׂנָא אוֹתָהּ וְיִכְתּוֹב לָהּ מִכְתַּב גֵּרוּשִׁין וְיִתֵּן בְּיָדָהּ וְיוֹצִיא אוֹתָהּ מִבֵּיתוֹ, אוֹ אִם יָמוּת הָאִישׁ הָאַחֲרוֹן שֶׁלָּקַח אוֹתָהּ לוֹ לְאִשָּׁה.
ד לֹא־יוּכַ֣ל בַּעְלָ֣הּ הָרִאשׁ֣וֹן אֲשֶֽׁר־שִׁ֠לְּחָ֠הּ לָשׁ֨וּב לְקַחְתָּ֜הּ לִהְי֧וֹת ל֣וֹ לְאִשָּׁ֗ה אַחֲרֵי֙ אֲשֶׁ֣ר הֻטַּמָּ֔אָה כִּֽי־תוֹעֵבָ֥ה הִ֖וא לִפְנֵ֣י יְהֹוָ֑ה וְלֹ֤א תַחֲטִיא֙ אֶת־הָאָ֔רֶץ אֲשֶׁר֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ נֹתֵ֥ן לְךָ֖ נַחֲלָֽה׃ בַּעְלָהּ הָרִאשׁוֹן שֶׁגֵּרֵשׁ אוֹתָהּ לֹא יוּכַל לָשֵׂאת אוֹתָהּ שׁוּב שֶׁתִּהְיֶה לוֹ לְאִשָּׁה אַחֲרֵי שֶׁטִּמְּאוּ אוֹתָהּ, כִּי זֶה תּוֹעֵבָה לִפְנֵי ה', וְאַל תַּכְתִּים בְּחֵטְא אֶת הָאָרֶץ שֶׁה' אֱלֹהֶיךָ נוֹתֵן לְךָ לְנַחֲלָה.

ששי[עריכה]

ס פָּרָשַׁת אִסּוּרֵי צַעַר
ה כִּֽי־יִקַּ֥ח אִישׁ֙ אִשָּׁ֣ה חֲדָשָׁ֔ה לֹ֤א יֵצֵא֙ בַּצָּבָ֔א וְלֹא־יַעֲבֹ֥ר עָלָ֖יו לְכׇל־דָּבָ֑ר נָקִ֞י יִהְיֶ֤ה לְבֵיתוֹ֙ שָׁנָ֣ה אֶחָ֔ת וְשִׂמַּ֖ח אֶת־אִשְׁתּ֥וֹ אֲשֶׁר־לָקָֽח׃ כְּשֶׁאִישׁ יִשָּׂא אִשָּׁה חֲדָשָׁה, הוּא לֹא יֵצֵא לַצָּבָא כְּדֵי שֶׁלֹּא יִקְרֶה לוֹ שׁוּם דָּבָר, אֶלָּא יִהְיֶה פָּנוּי לְבֵיתוֹ שָׁנָה אַחַת וִישַׂמֵּחַ אֶת אִשְׁתּוֹ שֶׁנָּשָׂא.
ו לֹא־יַחֲבֹ֥ל רֵחַ֖יִם וָרָ֑כֶב כִּי־נֶ֖פֶשׁ ה֥וּא חֹבֵֽל׃ אָדָם לֹא יְמַשְׁכֵּן רֵחַיִם אוֹ אֶת חֶלְקָם הָעֶלְיוֹן, כִּי בְּכָךְ הוּא מְמַשְׁכֵּן אֶת נֶפֶשׁ הַלֹּוֶה – בָּרָעָב.
ס
ז כִּי־יִמָּצֵ֣א אִ֗ישׁ גֹּנֵ֨ב נֶ֤פֶשׁ מֵאֶחָיו֙ מִבְּנֵ֣י יִשְׂרָאֵ֔ל וְהִתְעַמֶּר־בּ֖וֹ וּמְכָר֑וֹ וּמֵת֙ הַגַּנָּ֣ב הַה֔וּא וּבִֽעַרְתָּ֥ הָרָ֖ע מִקִּרְבֶּֽךָ׃ אִם יִתָּפֵשׂ אִישׁ שֶׁגָּנַב אָדָם מֵאֶחָיו מִבְּנֵי יִשְׂרָאֵל וְסָחַר בּוֹ וְמָכַר אוֹתוֹ, הַגַּנָּב הַהוּא יָמוּת, וּתְבַעֵר אֶת הָרָע מִתּוֹכְךָ.
ס פָּרָשַׁת הַצָּרַעַת
ח הִשָּׁ֧מֶר בְּנֶֽגַע־הַצָּרַ֛עַת לִשְׁמֹ֥ר מְאֹ֖ד וְלַעֲשׂ֑וֹת כְּכֹל֩ אֲשֶׁר־יוֹר֨וּ אֶתְכֶ֜ם הַכֹּהֲנִ֧ים הַלְוִיִּ֛ם כַּאֲשֶׁ֥ר צִוִּיתִ֖ם תִּשְׁמְר֥וּ לַעֲשֽׂוֹת׃ הִזָּהֵר בְּנֶגַע הַצָּרַעַת לְהַקְפִּיד מְאוֹד וְלַעֲשׂוֹת –אֶת כָּל מַה שֶּׁיּוֹרוּ לָכֶם הַכֹּהֲנִים הַלְוִיִּים כְּמוֹ שֶׁצִּוִּיתִי אוֹתָם, הַקְפִּידוּ לְקַיֵּם.
ט זָכ֕וֹר אֵ֧ת אֲשֶׁר־עָשָׂ֛ה יְהֹוָ֥ה אֱלֹהֶ֖יךָ לְמִרְיָ֑ם בַּדֶּ֖רֶךְ בְּצֵאתְכֶ֥ם מִמִּצְרָֽיִם׃ זְכוֹר אֶת הַצָּרַעַת שֶׁה' אֱלֹהֶיךָ עָשָׂה לְמִרְיָם בִּגְלַל חֲשָׁד שָׁוְא, בַּדֶּרֶךְ בְּצֵאתְכֶם מִמִּצְרַיִם.
ס פָּרָשַׁת הַמַּשְׁכּוֹן
י כִּֽי־תַשֶּׁ֥ה בְרֵֽעֲךָ֖ מַשַּׁ֣את מְא֑וּמָה לֹא־תָבֹ֥א אֶל־בֵּית֖וֹ לַעֲבֹ֥ט עֲבֹטֽוֹ׃ אִם תִּגְבֶּה מֵחֲבֵרְךָ חוֹב כָּלְשֶׁהוּ, אַל תָּבוֹא לְבֵיתוֹ לִגְבּוֹת מַשְׁכּוֹן.
יא בַּח֖וּץ תַּעֲמֹ֑ד וְהָאִ֗ישׁ אֲשֶׁ֤ר אַתָּה֙ נֹשֶׁ֣ה ב֔וֹ יוֹצִ֥יא אֵלֶ֛יךָ אֶֽת־הַעֲב֖וֹט הַחֽוּצָה׃ תַּעֲמוֹד בַּחוּץ, וְהָאִישׁ שֶׁאַתָּה גּוֹבֶה מִמֶּנּוּ יוֹצִיא אֵלֶיךָ אֶת הַמַּשְׁכּוֹן הַחוּצָה.
יב וְאִם־אִ֥ישׁ עָנִ֖י ה֑וּא לֹ֥א תִשְׁכַּ֖ב בַּעֲבֹטֽוֹ׃ וְאִם הוּא אִישׁ עָנִי אַל תִּשְׁכַּב לִישׁוֹן כְּשֶׁמַּשְׁכּוֹנוֹ אֶצְלְךָ.
יג הָשֵׁב֩ תָּשִׁ֨יב ל֤וֹ אֶֽת־הַעֲבוֹט֙ כְּב֣וֹא הַשֶּׁ֔מֶשׁ וְשָׁכַ֥ב בְּשַׂלְמָת֖וֹ וּבֵֽרְכֶ֑ךָּ וּלְךָ֙ תִּהְיֶ֣ה צְדָקָ֔ה לִפְנֵ֖י יְהֹוָ֥ה אֱלֹהֶֽיךָ׃ הַחֲזֵר לוֹ אֶת הַמַּשְׁכּוֹן בִּשְׁקִיעַת הַשֶּׁמֶשׁ, וְהוּא יִישַׁן עִם שְׂמִיכָתוֹ וִיבָרֵךְ אוֹתְךָ, וּלְךָ תִּהְיֶה זְכוּת לִפְנֵי ה' אֱלֹהֶיךָ.

שביעי[עריכה]

ס פָּרָשַׁת בְּיוֹמוֹ תִתֵּן שְׂכָרוֹ
יד לֹא־תַעֲשֹׁ֥ק שָׂכִ֖יר עָנִ֣י וְאֶבְי֑וֹן מֵאַחֶ֕יךָ א֧וֹ מִגֵּרְךָ֛ אֲשֶׁ֥ר בְּאַרְצְךָ֖ בִּשְׁעָרֶֽיךָ׃ אַל תִּגְזוֹל שָׂכִיר עָנִי וְאֶבְיוֹן מֵאַחֶיךָ אוֹ מִגֵּרְךָ שֶׁבְּאַרְצְךָ בְּעָרֶיךָ.
טו בְּיוֹמוֹ֩ תִתֵּ֨ן שְׂכָר֜וֹ וְֽלֹא־תָב֧וֹא עָלָ֣יו הַשֶּׁ֗מֶשׁ כִּ֤י עָנִי֙ ה֔וּא וְאֵלָ֕יו ה֥וּא נֹשֵׂ֖א אֶת־נַפְשׁ֑וֹ וְלֹֽא־יִקְרָ֤א עָלֶ֙יךָ֙ אֶל־יְהֹוָ֔ה וְהָיָ֥ה בְךָ֖ חֵֽטְא׃ תִּתֵּן לוֹ אֶת שְׂכָרוֹ בְּיוֹם עֲבוֹדָתוֹ, שֶׁלֹּא תִּשְׁקַע הַשֶּׁמֶשׁ לִפְנֵי כֵן, כִּי הוּא עָנִי וְנַפְשׁוֹ מִשְׁתּוֹקֶקֶת לִשְׂכָרוֹ, וְכָךְ לֹא יִתְפַּלֵּל לַה' עַל הַצַּעַר שֶׁגָּרַמְתָּ לוֹ וְיִהְיֶה בְּךָ חֵטְא.
ס פָּרָשַׁת רַחֲמֵי יָתוֹם וְאַלְמָנָה
טז לֹֽא־יוּמְת֤וּ אָבוֹת֙ עַל־בָּנִ֔ים וּבָנִ֖ים לֹא־יוּמְת֣וּ עַל־אָב֑וֹת אִ֥ישׁ בְּחֶטְא֖וֹ יוּמָֽתוּ׃ אָבוֹת לֹא יוּמְתוּ בִּגְלַל חֵטְא בְּנֵיהֶם, וּבָנִים לֹא יוּמְתוּ בִּגְלַל חֵטְא אֲבוֹתָם, אָדָם יוּמַת רַק בְּחֶטְאוֹ.
ס
יז לֹ֣א תַטֶּ֔ה מִשְׁפַּ֖ט גֵּ֣ר יָת֑וֹם וְלֹ֣א תַחֲבֹ֔ל בֶּ֖גֶד אַלְמָנָֽה׃ אַל תְּעַוֵּת פְּסַק דִּין שֶׁל מִשְׁפַּט גֵּר אוֹ יָתוֹם, וְאַל תִּקַּח לְמַשְׁכּוֹן בֶּגֶד שֶׁל אַלְמָנָה.
יח וְזָכַרְתָּ֗ כִּ֣י עֶ֤בֶד הָיִ֙יתָ֙ בְּמִצְרַ֔יִם וַֽיִּפְדְּךָ֛ יְהֹוָ֥ה אֱלֹהֶ֖יךָ מִשָּׁ֑ם עַל־כֵּ֞ן אָנֹכִ֤י מְצַוְּךָ֙ לַעֲשׂ֔וֹת אֶת־הַדָּבָ֖ר הַזֶּֽה׃ זְכוֹר שֶׂהָיִיתָ עֶבֶד בְּמִצְרַיִם וַה' אֱלֹהֶיךָ פָּדָה אוֹתְךָ מִשָּׁם, לָכֵן אֲנִי מְצַוֶּה אוֹתְךָ לַעֲשׂוֹת אֶת הַדָּבָר הַזֶּה.
ס
יט כִּ֣י תִקְצֹר֩ קְצִֽירְךָ֨ בְשָׂדֶ֜ךָ וְשָֽׁכַחְתָּ֧ עֹ֣מֶר בַּשָּׂדֶ֗ה לֹ֤א תָשׁוּב֙ לְקַחְתּ֔וֹ לַגֵּ֛ר לַיָּת֥וֹם וְלָאַלְמָנָ֖ה יִהְיֶ֑ה לְמַ֤עַן יְבָרֶכְךָ֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ בְּכֹ֖ל מַעֲשֵׂ֥ה יָדֶֽיךָ׃ אִם תִּקְצוֹר אֶת קְצִירְךָ בְּשָׂדְךָ וְתִשְׁכַּח אֲלֻמָּה בַּשָּׂדֶה אַל תַּחֲזוֹר לָקַחַת אוֹתָהּ, אֶלָּא הִיא תִּהְיֶה לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה, כְּדֵי שֶׁה' אֱלֹהֶיךָ יְבָרֵךְ אוֹתְךָ בְּכָל מַעֲשֵׂה יָדֶיךָ.
ס
כ כִּ֤י תַחְבֹּט֙ זֵֽיתְךָ֔ לֹ֥א תְפַאֵ֖ר אַחֲרֶ֑יךָ לַגֵּ֛ר לַיָּת֥וֹם וְלָאַלְמָנָ֖ה יִהְיֶֽה׃ כְּשֶׁתַּחְבּוֹט בְּעֵץ הַזַּיִת שֶׁלְּךָ לְהַפִּיל אֶת פֵּרוֹתָיו, אַל תִּקַּח אֶת הַזֵּתִים הַגְּדֵלִים בַּחֵלֶק הָעֶלְיוֹן, הֵם יִהְיוּ לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה.
כא כִּ֤י תִבְצֹר֙ כַּרְמְךָ֔ לֹ֥א תְעוֹלֵ֖ל אַחֲרֶ֑יךָ לַגֵּ֛ר לַיָּת֥וֹם וְלָאַלְמָנָ֖ה יִהְיֶֽה׃ כְּשֶׁתִּבְצוֹר אֶת כַּרְמְךָ אַל תִּבְצוֹר אֶשְׁכּוֹלוֹת דַּקִּים, הֵם יִהְיוּ לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה.
כב וְזָ֣כַרְתָּ֔ כִּי־עֶ֥בֶד הָיִ֖יתָ בְּאֶ֣רֶץ מִצְרָ֑יִם עַל־כֵּ֞ן אָנֹכִ֤י מְצַוְּךָ֙ לַעֲשׂ֔וֹת אֶת־הַדָּבָ֖ר הַזֶּֽה׃ זְכוֹר שֶׁהָיִיתָ עֶבֶד בְּאֶרֶץ מִצְרַיִם לָכֵן אֲנִי מְצַוֶּה אוֹתְךָ לַעֲשׂוֹת אֶת הַדָּבָר הַזֶּה.
ס פָּרָשַׁת מַלְקוֹת
כה א כִּֽי־יִהְיֶ֥ה רִיב֙ בֵּ֣ין אֲנָשִׁ֔ים וְנִגְּשׁ֥וּ אֶל־הַמִּשְׁפָּ֖ט וּשְׁפָט֑וּם וְהִצְדִּ֙יקוּ֙ אֶת־הַצַּדִּ֔יק וְהִרְשִׁ֖יעוּ אֶת־הָרָשָֽׁע׃ אִם יִהְיֶה רִיב בֵּין אֲנָשִׁים הֵם יִגְּשׁוּ לַמִּשְׁפָּט וְיִשְׁפְּטוּ אוֹתָם, וְיַצְדִּיקוּ אֶת הַצַּדִּיק וְיַרְשִׁיעוּ אֶת הָרָשָׁע.
ב וְהָיָ֛ה אִם־בִּ֥ן הַכּ֖וֹת הָרָשָׁ֑ע וְהִפִּיל֤וֹ הַשֹּׁפֵט֙ וְהִכָּ֣הוּ לְפָנָ֔יו כְּדֵ֥י רִשְׁעָת֖וֹ בְּמִסְפָּֽר׃ אִם הָרָשָׁע יִתְחַיֵּב מַלְקוֹת, הַשּׁוֹפֵט יַפִּילוֹ וְיַכֶּה אוֹתוֹ לְפָנָיו בְּמִסְפַּר הַמַּלְקוֹת שֶׁיִּתְחַיֵּב בָּהֵן כְּפִי רִשְׁעָתוֹ.
ג אַרְבָּעִ֥ים יַכֶּ֖נּוּ לֹ֣א יֹסִ֑יף פֶּן־יֹסִ֨יף לְהַכֹּת֤וֹ עַל־אֵ֙לֶּה֙ מַכָּ֣ה רַבָּ֔ה וְנִקְלָ֥ה אָחִ֖יךָ לְעֵינֶֽיךָ׃ הוּא יַכֶּה אוֹתוֹ אַרְבָּעִים מַכּוֹת וְלֹא יוֹסִיף, שֶׁלֹּא יוֹסִיף לְהַכּוֹתוֹ הַכָּאָה רַבָּה יוֹתֵר מִזֶּה וְאָחִיךָ יִתְבַּזֶּה לְעֵינֶיךָ.
ד לֹא־תַחְסֹ֥ם שׁ֖וֹר בְּדִישֽׁוֹ׃ אַל תַּחְסוֹם פֶּה שֶׁל שׁוֹר בִּזְמָן שֶׁדָּשׁ תְּבוּאָה כְּדֵי שֶׁלֹּא יֹאכַל.
ס פָּרָשַׁת יִבּוּם
ה כִּֽי־יֵשְׁב֨וּ אַחִ֜ים יַחְדָּ֗ו וּמֵ֨ת אַחַ֤ד מֵהֶם֙ וּבֵ֣ן אֵֽין־ל֔וֹ לֹֽא־תִהְיֶ֧ה אֵֽשֶׁת־הַמֵּ֛ת הַח֖וּצָה לְאִ֣ישׁ זָ֑ר יְבָמָהּ֙ יָבֹ֣א עָלֶ֔יהָ וּלְקָחָ֥הּ ל֛וֹ לְאִשָּׁ֖ה וְיִבְּמָֽהּ׃ אִם אַחִים יָגוּרוּ יַחַד וְאֶחָד מֵהֶם יָמוּת וְאֵין לוֹ בֵּן, אֵשֶׁת הַמֵּת לֹא תִּנָּשֵׂא לְאִישׁ מִחוּץ לַמִּשְׁפָּחָה, אֲחִי בַּעֲלָהּ יָבוֹא אֵלֶיהָ וְיִקַּח אוֹתָהּ לוֹ לְאִשָּׁה וִייַבֵּם אוֹתָהּ.
ו וְהָיָ֗ה הַבְּכוֹר֙ אֲשֶׁ֣ר תֵּלֵ֔ד יָק֕וּם עַל־שֵׁ֥ם אָחִ֖יו הַמֵּ֑ת וְלֹֽא־יִמָּחֶ֥ה שְׁמ֖וֹ מִיִּשְׂרָאֵֽל׃ הַבְּכוֹר שֶׁתֵּלֵד יָקוּם בִּמְקוֹם אָחִיו הַמֵּת וְכָךְ שְׁמוֹ לֹא יִמָּחֶה מִיִּשְׂרָאֵל.
ז וְאִם־לֹ֤א יַחְפֹּץ֙ הָאִ֔ישׁ לָקַ֖חַת אֶת־יְבִמְתּ֑וֹ וְעָלְתָה֩ יְבִמְתּ֨וֹ הַשַּׁ֜עְרָה אֶל־הַזְּקֵנִ֗ים וְאָֽמְרָה֙ מֵאֵ֨ן יְבָמִ֜י לְהָקִ֨ים לְאָחִ֥יו שֵׁם֙ בְּיִשְׂרָאֵ֔ל לֹ֥א אָבָ֖ה יַבְּמִֽי׃ וְאִם הָאִישׁ לֹא יִרְצֶה לָשֵׂאת אֶת גִּיסָתוֹ, גִּיסָתוֹ תַּעֲלֶה לְבֵית הַדִּין לַזְּקֵנִים, וְתֹאמַר: גִּיסִי מְסָרֵב לְהָקִים לְאָחִיו שֵׁם בְּעַם יִשְׂרָאֵל – הוּא לֹא הִסְכִּים לְיַבֵּם אוֹתִי.
ח וְקָֽרְאוּ־ל֥וֹ זִקְנֵי־עִיר֖וֹ וְדִבְּר֣וּ אֵלָ֑יו וְעָמַ֣ד וְאָמַ֔ר לֹ֥א חָפַ֖צְתִּי לְקַחְתָּֽהּ׃ זִקְנֵי עִירוֹ יִקְרְאוּ לוֹ וִידַבְּרוּ אֵלָיו, הוּא יַעֲמוֹד וְיֹאמַר: אֵינֶנִּי רוֹצֶה לָשֵׂאת אוֹתָהּ.
ט וְנִגְּשָׁ֨ה יְבִמְתּ֣וֹ אֵלָיו֮ לְעֵינֵ֣י הַזְּקֵנִים֒ וְחָלְצָ֤ה נַעֲלוֹ֙ מֵעַ֣ל רַגְל֔וֹ וְיָרְקָ֖ה בְּפָנָ֑יו וְעָֽנְתָה֙ וְאָ֣מְרָ֔ה כָּ֚כָה יֵעָשֶׂ֣ה לָאִ֔ישׁ אֲשֶׁ֥ר לֹא־יִבְנֶ֖ה אֶת־בֵּ֥ית אָחִֽיו׃ גִּיסָתוֹ תִּגַּשׁ אֵלָיו לְעֵינֵי הַזְּקֵנִים, תַּחֲלוֹץ אֶת נַעֲלוֹ מֵעַל רַגְלוֹ, תִּירַק לְפָנָיו לְבַיְּשׁוֹ, וְתַעֲנֶה וְתֹאמַר: כָּכָה יֵעָשֶׂה לָאִישׁ שֶׁלֹּא יִבְנֶה אֶת בֵּית אָחִיו.
י וְנִקְרָ֥א שְׁמ֖וֹ בְּיִשְׂרָאֵ֑ל בֵּ֖ית חֲל֥וּץ הַנָּֽעַל׃ וִיכַנּוּ בְּיִשְׂרָאֵל אֶת שְׁמוֹ: מִשְׁפַּחַת חֲלוּץ הַנַּעַל.
ס פָּרָשַׁת הַבּוֹשֶׁת
יא כִּֽי־יִנָּצ֨וּ אֲנָשִׁ֤ים יַחְדָּו֙ אִ֣ישׁ וְאָחִ֔יו וְקָֽרְבָה֙ אֵ֣שֶׁת הָֽאֶחָ֔ד לְהַצִּ֥יל אֶת־אִישָׁ֖הּ מִיַּ֣ד מַכֵּ֑הוּ וְשָׁלְחָ֣ה יָדָ֔הּ וְהֶחֱזִ֖יקָה בִּמְבֻשָֽׁיו׃ אִם אֲנָשִׁים יֵאָבְקוּ אֶחָד בַּשֵּׁנִי, וְאֵשֶׁת אֶחָד מֵהֶם תִּגַּשׁ לְהַצִּיל אֶת בַּעֲלָהּ מִיַּד מַכֵּהוּ וְתִשְׁלַח אֶת יָדָהּ וְתַחֲזִיק בְּאֵבַר הַבּוּשָׁה שֶׁלּוֹ – בְּבֵצָיו.
יב וְקַצֹּתָ֖ה אֶת־כַּפָּ֑הּ לֹ֥א תָח֖וֹס עֵינֶֽךָ׃ קְצוֹץ אֶת כַּף יָדָהּ אוֹ שֶׁהִיא תְּשַׁלֵּם אֶת פִּדְיוֹנָהּ, אַל תְּרַחֵם.
ס פָּרָשַׁת מִדּוֹת וּמִשְׁקוֹלוֹת
יג לֹֽא־יִהְיֶ֥ה לְךָ֛ בְּכִֽיסְךָ֖ אֶ֣בֶן וָאָ֑בֶן גְּדוֹלָ֖ה וּקְטַנָּֽה׃ לֹא יִהְיֶה לְךָ בְּכִיסְךָ שְׁתֵּי מִשְׁקוֹלוֹת שׁוֹנוֹת – גְּדוֹלָה וּקְטַנָּה, כְּדֵי לְרַמּוֹת.
יד לֹא־יִהְיֶ֥ה לְךָ֛ בְּבֵיתְךָ֖ אֵיפָ֣ה וְאֵיפָ֑ה גְּדוֹלָ֖ה וּקְטַנָּֽה׃ לֹא יִהְיֶה לְךָ בְּבֵיתְךָ שְׁתֵּי מִדּוֹת שׁוֹנוֹת לִמְדִידַת נֶפַח – גְּדוֹלָה וּקְטַנָּה, כְּדֵי לְרַמּוֹת.
טו אֶ֣בֶן שְׁלֵמָ֤ה וָצֶ֙דֶק֙ יִֽהְיֶה־לָּ֔ךְ אֵיפָ֧ה שְׁלֵמָ֛ה וָצֶ֖דֶק יִֽהְיֶה־לָּ֑ךְ לְמַ֙עַן֙ יַאֲרִ֣יכוּ יָמֶ֔יךָ עַ֚ל הָֽאֲדָמָ֔ה אֲשֶׁר־יְהֹוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֥ן לָֽךְ׃ אֶלָּא מִשְׁקוֹלֶת שְׁלֵמָה שֶׁל צֶדֶק יִהְיֶה לְךָ, מִדָּה שְׁלֵמָה שֶׁל צֶדֶק יִהְיֶה לְךָ, וּבִשְׂכַר זֶה יְמֵי חַיֶּיךָ יִתְאָרְכוּ עַל הָאֲדָמָה שֶׁה' אֱלֹהֶיךָ נוֹתֵן לְךָ.
טז כִּ֧י תוֹעֲבַ֛ת יְהֹוָ֥ה אֱלֹהֶ֖יךָ כׇּל־עֹ֣שֵׂה אֵ֑לֶּה כֹּ֖ל עֹ֥שֵׂה עָֽוֶל׃ כִּי כָּל מִי שֶׁעוֹשֶׂה רַמָּאוּיוֹת כָּאֵלֶּה הוּא תּוֹעֵבָה בְּעֵינֵי ה' אֱלֹהֶיךָ – כָּל מִי שֶׁעוֹשֶׂה עָוֶל.

מפטיר[עריכה]

פ פָּרָשַׁת עֲמָלֵק
יז זָכ֕וֹר אֵ֛ת אֲשֶׁר־עָשָׂ֥ה לְךָ֖ עֲמָלֵ֑ק בַּדֶּ֖רֶךְ בְּצֵאתְכֶ֥ם מִמִּצְרָֽיִם׃ זְכוֹר אֶת מַה שֶּׁעֲמָלֵק עָשָׂה לְךָ בַּדֶּרֶךְ כְּשֶׁיְּצָאתֶם מִמִּצְרַיִם.
יח אֲשֶׁ֨ר קָֽרְךָ֜ בַּדֶּ֗רֶךְ וַיְזַנֵּ֤ב בְּךָ֙ כׇּל־הַנֶּחֱשָׁלִ֣ים אַֽחֲרֶ֔יךָ וְאַתָּ֖ה עָיֵ֣ף וְיָגֵ֑עַ וְלֹ֥א יָרֵ֖א אֱלֹהִֽים׃ שֶׁפָּגַשׁ אוֹתְךָ בַּדֶּרֶךְ וְהָרַג בָּאַחֲרוֹנִים שֶׁבְּךָ – אֶת כָּל הַחַלָּשִׁים שֶׁמְּאַחֲרִים אַחֲרֶיךָ, וְאַתָּה הָיִיתָ עָיֵף וְיָגֵעַ, וְהוּא לֹא פָּחַד מֵאֱלֹהִים.
יט וְהָיָ֡ה בְּהָנִ֣יחַ יְהֹוָ֣ה אֱלֹהֶ֣יךָ ׀ לְ֠ךָ֠ מִכׇּל־אֹ֨יְבֶ֜יךָ מִסָּבִ֗יב בָּאָ֙רֶץ֙ אֲשֶׁ֣ר יְהֹוָה־אֱ֠לֹהֶ֠יךָ נֹתֵ֨ן לְךָ֤ נַחֲלָה֙ לְרִשְׁתָּ֔הּ תִּמְחֶה֙ אֶת־זֵ֣כֶר עֲמָלֵ֔ק מִתַּ֖חַת הַשָּׁמָ֑יִם לֹ֖א תִּשְׁכָּֽח׃ כְּשֶׁה' אֱלֹהֶיךָ יִתֵּן לְךָ מְנוּחָה מִכָּל אוֹיְבֶיךָ מִסָּבִיב בָּאָרֶץ שֶׁה' אֱלֹהֶיךָ נוֹתֵן לְךָ נַחֲלָה לִכְבּוֹשׁ אוֹתָהּ – תַּשְׁמִיד כָּל זֵכֶר שֶׁל עֲמָלֵק מִתַּחַת לַשָּׁמַיִם, אַל תִּשְׁכָּח!
