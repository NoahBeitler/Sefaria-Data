ראשון[עריכה]

פ פָּרָשַׁת שְׂכַר פִּינְחָס
י וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יא פִּֽינְחָ֨ס‏[1] בֶּן־אֶלְעָזָ֜ר בֶּן־אַהֲרֹ֣ן הַכֹּהֵ֗ן הֵשִׁ֤יב אֶת־חֲמָתִי֙ מֵעַ֣ל בְּנֵֽי־יִשְׂרָאֵ֔ל בְּקַנְא֥וֹ אֶת־קִנְאָתִ֖י בְּתוֹכָ֑ם וְלֹא־כִלִּ֥יתִי אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל בְּקִנְאָתִֽי׃ פִּינְחָס בֶּן אֶלְעָזָר בֶּן אַהֲרֹן הַכֹּהֵן הִשְׁקִיט אֶת כַּעֲסִי עַל בְּנֵי יִשְׂרָאֵל בְּכָךְ שֶׁנָּקַם אֶת נִקְמָתִי בְּתוֹכָם, וְלָכֵן לֹא הִשְׁמַדְתִּי אֶת בְּנֵי יִשְׂרָאֵל בְּכַעֲסִי.
יב לָכֵ֖ן אֱמֹ֑ר הִנְנִ֨י נֹתֵ֥ן ל֛וֹ אֶת־בְּרִיתִ֖י שָׁלֽוֹם‏[2]׃ לָכֵן אֱמוֹר לוֹ שֶׁאֲנִי מִתְחַיֵּב לְהַעֲנִיק לוֹ שָׁלוֹם.
יג וְהָ֤יְתָה לּוֹ֙ וּלְזַרְע֣וֹ אַחֲרָ֔יו בְּרִ֖ית כְּהֻנַּ֣ת עוֹלָ֑ם תַּ֗חַת אֲשֶׁ֤ר קִנֵּא֙ לֵֽאלֹהָ֔יו וַיְכַפֵּ֖ר עַל־בְּנֵ֥י יִשְׂרָאֵֽל׃ וְתִהְיֶה לוֹ וּלְצֶאֱצָאָיו אַחֲרָיו הִתְחַיְּבוּת מִמֶּנִּי שֶׁיִּהְיוּ כֹּהֲנִים לְעוֹלָם, כְּשָׂכָר עַל שֶׁנָּקַם עֲבוּר אֱלֹהָיו וְכִפֵּר עַל בְּנֵי יִשְׂרָאֵל.
יד וְשֵׁם֩ אִ֨ישׁ יִשְׂרָאֵ֜ל הַמֻּכֶּ֗ה אֲשֶׁ֤ר הֻכָּה֙ אֶת־הַמִּדְיָנִ֔ית זִמְרִ֖י בֶּן־סָל֑וּא נְשִׂ֥יא בֵֽית־אָ֖ב לַשִּׁמְעֹנִֽי׃ וְשֵׁם הָאִישׁ מִיִּשְׂרָאֵל הָהָרוּג שֶׁנֶּהֱרַג עִם הַמִּדְיָנִית: זִמְרִי בֶּן סָלוּא, נְשִׂיא עֵדָה בְּשֵׁבֶט שִׁמְעוֹן.
טו וְשֵׁ֨ם הָֽאִשָּׁ֧ה הַמֻּכָּ֛ה הַמִּדְיָנִ֖ית כׇּזְבִּ֣י בַת־צ֑וּר רֹ֣אשׁ אֻמּ֥וֹת בֵּֽית־אָ֛ב בְּמִדְיָ֖ן הֽוּא׃ וְשֵׁם הָאִשָּׁה הַמִּדְיָנִית הַהֲרוּגָה: כָּזְבִּי בַּת צוּר, שֶׁהָיָה מַנְהִיג שֶׁל מִסְפַּר שֶׁל שְׁבָטִים בְּמִדְיָן.
פ
טז וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יז צָר֖וֹר אֶת־הַמִּדְיָנִ֑ים וְהִכִּיתֶ֖ם אוֹתָֽם׃ הִלָּחֲמוּ בַּמִּדְיָנִים וְהִרְגּוּ אוֹתָם.
יח כִּ֣י צֹרְרִ֥ים הֵם֙ לָכֶ֔ם בְּנִכְלֵיהֶ֛ם אֲשֶׁר־נִכְּל֥וּ לָכֶ֖ם עַל־דְּבַר־פְּע֑וֹר וְעַל־דְּבַ֞ר כׇּזְבִּ֨י בַת־נְשִׂ֤יא מִדְיָן֙ אֲחֹתָ֔ם הַמֻּכָּ֥ה בְיוֹם־הַמַּגֵּפָ֖ה עַל־דְּבַר־פְּעֽוֹר׃ כִּי הֵם נִלְחָמִים בָּכֶם בִּמְזִמּוֹתֵיהֶם נֶגְדְּכֶם בְּעִנְיַן פְּעוֹר, וּבְעִנְיַן אֲחוֹתָם כָּזְבִּי בַּת נְשִׂיא מִדְיָן שֶׁנֶהֶרְגָה בְּיוֹם הַמַּגֵּפָה שֶׁהָיְתָה עוֹנֶשׁ עַל עִנְיַן פְּעוֹר.
פָּרָשַׁת הַפְּקוּדִים
כו א וַיְהִ֖י אַחֲרֵ֣י הַמַּגֵּפָ֑ה


וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה וְאֶ֧ל אֶלְעָזָ֛ר בֶּן־אַהֲרֹ֥ן הַכֹּהֵ֖ן לֵאמֹֽר׃

אַחֲרֵי הַמַּגֵּפָה, ה' אָמַר לְמֹשֶׁה וּלְאֶלְעָזָר בֶּן אַהֲרֹן הַכֹּהֵן:
ב שְׂא֞וּ אֶת־רֹ֣אשׁ ׀ כׇּל־עֲדַ֣ת בְּנֵי־יִשְׂרָאֵ֗ל מִבֶּ֨ן עֶשְׂרִ֥ים שָׁנָ֛ה וָמַ֖עְלָה לְבֵ֣ית אֲבֹתָ֑ם כׇּל־יֹצֵ֥א צָבָ֖א בְּיִשְׂרָאֵֽל׃ סִפְרוּ אֶת כָּל הַצִּבּוּר שֶׁל בְּנֵי יִשְׂרָאֵל מִגִּיל עֶשְׂרִים שָׁנָה וָמַעְלָה לְפִי מִשְׁפְּחוֹתֵיהֶם, כָּל מִתְגַּיֵּס לַצָּבָא בְּיִשְׂרָאֵל.
ג וַיְדַבֵּ֨ר מֹשֶׁ֜ה וְאֶלְעָזָ֧ר הַכֹּהֵ֛ן אֹתָ֖ם בְּעַֽרְבֹ֣ת מוֹאָ֑ב עַל־יַרְדֵּ֥ן יְרֵח֖וֹ לֵאמֹֽר׃ מֹשֶׁה וְאֶלְעָזָר הַכֹּהֵן דִּבְּרוּ אֶל יִשְׂרָאֵל בַּמִּישׁוֹרִים שֶׁל מוֹאָב עַל יַד יַרְדֵּן יְרִיחוֹ וְאָמְרוּ:
ד מִבֶּ֛ן עֶשְׂרִ֥ים שָׁנָ֖ה וָמָ֑עְלָה כַּאֲשֶׁר֩ צִוָּ֨ה יְהֹוָ֤ה אֶת־מֹשֶׁה֙ וּבְנֵ֣י יִשְׂרָאֵ֔ל הַיֹּצְאִ֖ים מֵאֶ֥רֶץ מִצְרָֽיִם׃ מִגִּיל עֶשְׂרִים שָׁנָה וָמָעְלָה יִסָּפְרוּ, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה וּבְנֵי יִשְׂרָאֵל אַחֲרֵי שֶׁיָּצְאוּ מֵאֶרֶץ מִצְרַיִם (בְּפָרָשַׁת בְּמִדְבַּר).

שני[עריכה]

ה רְאוּבֵ֖ן בְּכ֣וֹר יִשְׂרָאֵ֑ל בְּנֵ֣י רְאוּבֵ֗ן חֲנוֹךְ֙ מִשְׁפַּ֣חַת הַחֲנֹכִ֔י לְפַלּ֕וּא מִשְׁפַּ֖חַת הַפַּלֻּאִֽי׃ רְאוּבֵן – בְּכוֹר יִשְׂרָאֵל, בְּנֵי רְאוּבֵן: חֲנוֹךְ – אֲבִי מִשְׁפַּחַת הַחֲנוֹכִי, לְפַלּוּא – מִשְׁפַּחַת הַפַּלּוּאִי.
ו לְחֶצְרֹ֕ן מִשְׁפַּ֖חַת הַֽחֶצְרוֹנִ֑י לְכַרְמִ֕י מִשְׁפַּ֖חַת הַכַּרְמִֽי׃ לְחֶצְרוֹן מִשְׁפַּחַת הַחֶצְרוֹנִי, לְכַרְמִי מִשְׁפַּחַת הַכַּרְמִי.
ז אֵ֖לֶּה מִשְׁפְּחֹ֣ת הָראוּבֵנִ֑י וַיִּהְי֣וּ פְקֻדֵיהֶ֗ם שְׁלֹשָׁ֤ה וְאַרְבָּעִים֙ אֶ֔לֶף וּשְׁבַ֥ע מֵא֖וֹת וּשְׁלֹשִֽׁים׃ אֵלֶּה מִשְׁפְּחוֹת רְאוּבֵן, מִסְפַּר אַנְשֵׁיהֶם הָיָה אַרְבָּעִים וּשְׁלוֹשָׁה אֶלֶף וּשְׁבַע מֵאוֹת וּשְׁלוֹשִׁים.
ח וּבְנֵ֥י פַלּ֖וּא אֱלִיאָֽב׃ וּבְנֵי פַּלּוּא: אֱלִיאָב.
ט וּבְנֵ֣י אֱלִיאָ֔ב נְמוּאֵ֖ל וְדָתָ֣ן וַאֲבִירָ֑ם הֽוּא־דָתָ֨ן וַאֲבִירָ֜ם קרואי קְרִיאֵ֣י הָעֵדָ֗ה אֲשֶׁ֨ר הִצּ֜וּ עַל־מֹשֶׁ֤ה וְעַֽל־אַהֲרֹן֙ בַּעֲדַת־קֹ֔רַח בְּהַצֹּתָ֖ם עַל־יְהֹוָֽה׃ וּבְנֵי אֱלִיאָב: נְמוּאֵל וְדָתָן וַאֲבִירָם, הֵם דָּתָן וַאֲבִירָם הַמּוּכָּרִים, נְצִיגֵי הַצִּבּוּר בַּאֲסֵיפוֹת הַהַנְהָגָה, שֶׁהִמְרִידוּ אֶת הָעָם נֶגֶד מֹשֶׁה וְאַהֲרֹן בַּקָּהָל שֶׁל קוֹרַח כְּשֶׁהִמְרִידוּ לָרִיב עִם ה'.
י וַתִּפְתַּ֨ח הָאָ֜רֶץ אֶת־פִּ֗יהָ וַתִּבְלַ֥ע אֹתָ֛ם וְאֶת־קֹ֖רַח בְּמ֣וֹת הָעֵדָ֑ה בַּאֲכֹ֣ל הָאֵ֗שׁ אֵ֣ת חֲמִשִּׁ֤ים וּמָאתַ֙יִם֙ אִ֔ישׁ וַיִּהְי֖וּ לְנֵֽס׃ הָאָרֶץ פָּתְחָה אֶת פִּיהָ וּבָלְעָה אוֹתָם וְאֶת קוֹרַח, בִּזְמַן מוֹת הַצִּבּוּר שֶׁל קוֹרַח, כְּשֶׁהָאֵשׁ שָׂרְפָה אֶת מָאתַיִם וַחֲמִשִּׁים הָאֲנָשִׁים, וְהֵם הָפְכוּ לְשֵׁם דָּבָר.
יא וּבְנֵי־קֹ֖רַח לֹא־מֵֽתוּ׃ וּבְנֵי קוֹרַח לֹא מֵתוּ.
ס
יב בְּנֵ֣י שִׁמְעוֹן֮ לְמִשְׁפְּחֹתָם֒ לִנְמוּאֵ֗ל מִשְׁפַּ֙חַת֙ הַנְּמ֣וּאֵלִ֔י לְיָמִ֕ין מִשְׁפַּ֖חַת הַיָּמִינִ֑י לְיָכִ֕ין מִשְׁפַּ֖חַת הַיָּכִינִֽי׃ בְּנֵי שִׁמְעוֹן לְמִשְׁפְּחוֹתֵיהֶם: לִנְמוּאֵל מִשְׁפַּחַת הַנְּמוּאֵלִי, לְיָמִין מִשְׁפַּחַת הַיָּמִינִי, לְיָכִין מִשְׁפַּחַת הַיָּכִינִי.
יג לְזֶ֕רַח מִשְׁפַּ֖חַת הַזַּרְחִ֑י לְשָׁא֕וּל מִשְׁפַּ֖חַת הַשָּׁאוּלִֽי׃ לְזֶרַח מִשְׁפַּחַת הַזַּרְחִי, לְשָׁאוּל מִשְׁפַּחַת הַשָּׁאוּלִי.
יד אֵ֖לֶּה מִשְׁפְּחֹ֣ת הַשִּׁמְעֹנִ֑י שְׁנַ֧יִם וְעֶשְׂרִ֛ים אֶ֖לֶף וּמָאתָֽיִם׃ אֵלֶּה מִשְׁפְּחוֹת הַשִּׁמְעוֹנִי, וּמִסְפָּרָם עֶשְׂרִים וּשְׁנַיִם אֶלֶף וּמָאתָיִם.

טו בְּנֵ֣י גָד֮ לְמִשְׁפְּחֹתָם֒ לִצְפ֗וֹן מִשְׁפַּ֙חַת֙ הַצְּפוֹנִ֔י לְחַגִּ֕י מִשְׁפַּ֖חַת הַֽחַגִּ֑י לְשׁוּנִ֕י מִשְׁפַּ֖חַת הַשּׁוּנִֽי׃ בְּנֵי גָּד לְמִשְׁפְּחוֹתֵיהֶם: לִצְפוֹן מִשְׁפַּחַת הַצְּפוֹנִי, לְחַגִּי מִשְׁפַּחַת הַחַגִּי, לְשׁוּנִי מִשְׁפַּחַת הַשּׁוּנִי.
טז לְאׇזְנִ֕י מִשְׁפַּ֖חַת הָאׇזְנִ֑י לְעֵרִ֕י מִשְׁפַּ֖חַת הָעֵרִֽי׃ לְאָזְנִי מִשְׁפַּחַת הָאָזְנִי, לְעֵרִי מִשְׁפַּחַת הָעֵרִי.
יז לַאֲר֕וֹד מִשְׁפַּ֖חַת הָאֲרוֹדִ֑י לְאַ֨רְאֵלִ֔י מִשְׁפַּ֖חַת הָאַרְאֵלִֽי׃ לַאֲרוֹד מִשְׁפַּחַת הָאֲרוֹדִי, לְאַרְאֵלִי מִשְׁפַּחַת הָאַרְאֵלִי.
יח אֵ֛לֶּה מִשְׁפְּחֹ֥ת בְּנֵֽי־גָ֖ד לִפְקֻדֵיהֶ֑ם אַרְבָּעִ֥ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת׃ אֵלֶּה מִשְׁפְּחוֹת בְּנֵי גָּד שֶׁנִּסְפְּרוּ – אַרְבָּעִים אֶלֶף וַחֲמֵשׁ מֵאוֹת.

יט בְּנֵ֥י יְהוּדָ֖ה עֵ֣ר וְאוֹנָ֑ן וַיָּ֥מׇת עֵ֛ר וְאוֹנָ֖ן בְּאֶ֥רֶץ כְּנָֽעַן׃ בְּנֵי יְהוּדָה: עֵר וְאוֹנָן, עֵר וְאוֹנָן מֵתוּ בְּאֶרֶץ כְּנַעַן.
כ וַיִּהְי֣וּ בְנֵי־יְהוּדָה֮ לְמִשְׁפְּחֹתָם֒ לְשֵׁלָ֗ה מִשְׁפַּ֙חַת֙ הַשֵּׁ֣לָנִ֔י לְפֶ֕רֶץ מִשְׁפַּ֖חַת הַפַּרְצִ֑י לְזֶ֕רַח מִשְׁפַּ֖חַת הַזַּרְחִֽי׃ בְּנֵי יְהוּדָה לְמִשְׁפְּחוֹתֵיהֶם הָיוּ: לְשֵׁלָה מִשְׁפַּחַת הַשֵּׁלָנִי, לְפֶרֶץ מִשְׁפַּחַת הַפַּרְצִי, לְזֶרַח מִשְׁפַּחַת הַזַּרְחִי.
כא וַיִּהְי֣וּ בְנֵי־פֶ֔רֶץ לְחֶצְרֹ֕ן מִשְׁפַּ֖חַת הַֽחֶצְרֹנִ֑י לְחָמ֕וּל מִשְׁפַּ֖חַת הֶחָמוּלִֽי׃ בְּנֵי פֶּרֶץ הָיוּ: לְחֶצְרוֹן מִשְׁפַּחַת הַחֶצְרוֹנִי, לְחָמוּל מִשְׁפַּחַת הֶחָמוּלִי.
כב אֵ֛לֶּה מִשְׁפְּחֹ֥ת יְהוּדָ֖ה לִפְקֻדֵיהֶ֑ם שִׁשָּׁ֧ה וְשִׁבְעִ֛ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת׃ אֵלֶּה מִשְׁפְּחוֹת יְהוּדָה שֶׁנִּסְפְּרוּ – שִׁבְעִים וְשִׁשָּׁה אֶלֶף וַחֲמֵשׁ מֵאוֹת.
ס
כג בְּנֵ֤י יִשָּׂשכָר֙ לְמִשְׁפְּחֹתָ֔ם תּוֹלָ֕ע מִשְׁפַּ֖חַת הַתּוֹלָעִ֑י לְפֻוָ֕ה מִשְׁפַּ֖חַת הַפּוּנִֽי׃ בְּנֵי יִשָּׂשכָר לְמִשְׁפְּחוֹתֵיהֶם: תּוֹלָע מִשְׁפַּחַת הַתּוֹלָעִי, לְפֻוָה מִשְׁפַּחַת הַפּוּנִי.
כד לְיָשׁ֕וּב מִשְׁפַּ֖חַת הַיָּשֻׁבִ֑י לְשִׁמְרֹ֕ן מִשְׁפַּ֖חַת הַשִּׁמְרֹנִֽי׃ לְיָשׁוּב מִשְׁפַּחַת הַיָּשׁוּבִי, לְשִׁמְרוֹן מִשְׁפַּחַת הַשִּׁמְרוֹנִי.
כה אֵ֛לֶּה מִשְׁפְּחֹ֥ת יִשָּׂשכָ֖ר לִפְקֻדֵיהֶ֑ם אַרְבָּעָ֧ה וְשִׁשִּׁ֛ים אֶ֖לֶף וּשְׁלֹ֥שׁ מֵאֽוֹת׃ אֵלֶּה מִשְׁפְּחוֹת יִשָּׂשכָר שֶׁנִּסְפְּרוּ – שִׁשִּׁים וְאַרְבָּעָה אֶלֶף וּשְׁלוֹשׁ מֵאוֹת.

כו בְּנֵ֣י זְבוּלֻן֮ לְמִשְׁפְּחֹתָם֒ לְסֶ֗רֶד מִשְׁפַּ֙חַת֙ הַסַּרְדִּ֔י לְאֵל֕וֹן מִשְׁפַּ֖חַת הָאֵלֹנִ֑י לְיַ֨חְלְאֵ֔ל מִשְׁפַּ֖חַת הַיַּחְלְאֵלִֽי׃ בְּנֵי זְבוּלוּן לְמִשְׁפְּחוֹתֵיהֶם: לְסֶרֶד מִשְׁפַּחַת הַסַּרְדִּי, לְאֵלוֹן מִשְׁפַּחַת הָאֵלוֹנִי, לְיַחְלְאֵל מִשְׁפַּחַת הַיַּחְלְאֵלִי.
כז אֵ֛לֶּה מִשְׁפְּחֹ֥ת הַזְּבוּלֹנִ֖י לִפְקֻדֵיהֶ֑ם שִׁשִּׁ֥ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת׃ אֵלֶּה מִשְׁפְּחוֹת הַזְּבוּלוֹנִי שֶׁנִּסְפְּרוּ – שִׁשִּׁים אֶלֶף וַחֲמֵשׁ מֵאוֹת.

כח בְּנֵ֥י יוֹסֵ֖ף לְמִשְׁפְּחֹתָ֑ם מְנַשֶּׁ֖ה וְאֶפְרָֽיִם׃ בְּנֵי יוֹסֵף לְמִשְׁפְּחוֹתֵיהֶם: מְנַשֶּׁה וְאֶפְרַיִם.
כט בְּנֵ֣י מְנַשֶּׁ֗ה לְמָכִיר֙ מִשְׁפַּ֣חַת הַמָּכִירִ֔י וּמָכִ֖יר הוֹלִ֣יד אֶת־גִּלְעָ֑ד לְגִלְעָ֕ד מִשְׁפַּ֖חַת הַגִּלְעָדִֽי׃ בְּנֵי מְנַשֶּׁה: לְמָכִיר מִשְׁפַּחַת הַמָּכִירִי וּמָכִיר הוֹלִיד אֶת גִּלְעָד, לְגִלְעָד מִשְׁפַּחַת הַגִּלְעָדִי.
ל אֵ֚לֶּה בְּנֵ֣י גִלְעָ֔ד אִיעֶ֕זֶר מִשְׁפַּ֖חַת הָאִֽיעֶזְרִ֑י לְחֵ֕לֶק מִשְׁפַּ֖חַת הַֽחֶלְקִֽי׃ אֵלֶּה בְּנֵי גִּלְעָד: אִיעֶזֶר מִשְׁפַּחַת הָאִיעֶזְרִי, לְחֵלֶק מִשְׁפַּחַת הַחֶלְקִי.
לא וְאַ֨שְׂרִיאֵ֔ל מִשְׁפַּ֖חַת הָאַשְׂרִֽאֵלִ֑י וְשֶׁ֕כֶם מִשְׁפַּ֖חַת הַשִּׁכְמִֽי׃ וְאַשְׂרִיאֵל מִשְׁפַּחַת הָאַשְׂרִאֵלִי, וְשֶׁכֶם מִשְׁפַּחַת הַשִּׁכְמִי.
לב וּשְׁמִידָ֕ע מִשְׁפַּ֖חַת הַשְּׁמִידָעִ֑י וְחֵ֕פֶר מִשְׁפַּ֖חַת הַֽחֶפְרִֽי׃ וּשְׁמִידָע מִשְׁפַּחַת הַשְּׁמִידָעִי, וְחֵפֶר מִשְׁפַּחַת הַחֶפְרִי.
לג וּצְלׇפְחָ֣ד בֶּן־חֵ֗פֶר לֹא־הָ֥יוּ ל֛וֹ בָּנִ֖ים כִּ֣י אִם־בָּנ֑וֹת וְשֵׁם֙ בְּנ֣וֹת צְלׇפְחָ֔ד מַחְלָ֣ה וְנֹעָ֔ה חׇגְלָ֥ה מִלְכָּ֖ה וְתִרְצָֽה׃ וּצְלָפְחָד בֶּן חֵפֶר לֹא הָיוּ לוֹ בָּנִים אֶלָּא בָּנוֹת, וְשֵׁם בְּנוֹת צְלָפְחָד: מַחְלָה וְנוֹעָה חָגְלָה מִלְכָּה וְתִרְצָה.
לד אֵ֖לֶּה מִשְׁפְּחֹ֣ת מְנַשֶּׁ֑ה וּפְקֻ֣דֵיהֶ֔ם שְׁנַ֧יִם וַחֲמִשִּׁ֛ים אֶ֖לֶף וּשְׁבַ֥ע מֵאֽוֹת׃ אֵלֶּה מִשְׁפְּחוֹת מְנַשֶּׁה, וּמִסְפָּרָם חֲמִשִּׁים וּשְׁנַיִם אֶלֶף וּשְׁבַע מֵאוֹת.

לה אֵ֣לֶּה בְנֵי־אֶפְרַ֘יִם֮ לְמִשְׁפְּחֹתָם֒ לְשׁוּתֶ֗לַח מִשְׁפַּ֙חַת֙ הַשֻּׁ֣תַלְחִ֔י לְבֶ֕כֶר מִשְׁפַּ֖חַת הַבַּכְרִ֑י לְתַ֕חַן מִשְׁפַּ֖חַת הַֽתַּחֲנִֽי׃ אֵלֶּה בְּנֵי אֶפְרַיִם לְמִשְׁפְּחוֹתֵיהֶם: לְשׁוּתֶלַח מִשְׁפַּחַת הַשּׁוּתַלְחִי, לְבֶכֶר מִשְׁפַּחַת הַבַּכְרִי, לְתַחַן מִשְׁפַּחַת הַתַּחֲנִי.
לו וְאֵ֖לֶּה בְּנֵ֣י שׁוּתָ֑לַח לְעֵרָ֕ן מִשְׁפַּ֖חַת הָעֵרָנִֽי׃ וְאֵלֶּה בְּנֵי שׁוּתֶלַח: לְעֵרָן מִשְׁפַּחַת הָעֵרָנִי.
לז אֵ֣לֶּה מִשְׁפְּחֹ֤ת בְּנֵי־אֶפְרַ֙יִם֙ לִפְקֻ֣דֵיהֶ֔ם שְׁנַ֧יִם וּשְׁלֹשִׁ֛ים אֶ֖לֶף וַחֲמֵ֣שׁ מֵא֑וֹת אֵ֥לֶּה בְנֵי־יוֹסֵ֖ף לְמִשְׁפְּחֹתָֽם׃ אֵלֶּה מִשְׁפְּחוֹת בְּנֵי אֶפְרַיִם שֶׁנִּסְפְּרוּ – שְׁלוֹשִׁים וּשְׁנַיִם אֶלֶף וַחֲמֵשׁ מֵאוֹת. אֵלֶּה בְּנֵי יוֹסֵף לְמִשְׁפְּחוֹתֵיהֶם.

לח בְּנֵ֣י בִנְיָמִן֮ לְמִשְׁפְּחֹתָם֒ לְבֶ֗לַע מִשְׁפַּ֙חַת֙ הַבַּלְעִ֔י לְאַשְׁבֵּ֕ל מִשְׁפַּ֖חַת הָֽאַשְׁבֵּלִ֑י לַאֲחִירָ֕ם מִשְׁפַּ֖חַת הָאֲחִירָמִֽי׃ בְּנֵי בִּנְיָמִין לְמִשְׁפְּחוֹתֵיהֶם: לְבֶלַע מִשְׁפַּחַת הַבַּלְעִי, לְאַשְׁבֵּל מִשְׁפַּחַת הָאַשְׁבֵּלִי, לַאֲחִירָם מִשְׁפַּחַת הָאֲחִירָמִי.
לט לִשְׁפוּפָ֕ם מִשְׁפַּ֖חַת הַשּׁוּפָמִ֑י לְחוּפָ֕ם מִשְׁפַּ֖חַת הַחוּפָמִֽי׃ לִשְׁפוּפָם מִשְׁפַּחַת הַשּׁוּפָמִי, לְחוּפָם מִשְׁפַּחַת הַחוּפָמִי.
מ וַיִּהְי֥וּ בְנֵי־בֶ֖לַע אַ֣רְדְּ וְנַעֲמָ֑ן מִשְׁפַּ֙חַת֙ הָֽאַרְדִּ֔י לְנַֽעֲמָ֔ן מִשְׁפַּ֖חַת הַֽנַּעֲמִֽי׃ בְּנֵי בֶּלַע הָיוּ אַרְדְּ וְנַעֲמָן: לְאַרְדְּ מִשְׁפַּחַת הָאַרְדִּי, לְנַעֲמָן מִשְׁפַּחַת הַנַּעֲמִי.
מא אֵ֥לֶּה בְנֵי־בִנְיָמִ֖ן לְמִשְׁפְּחֹתָ֑ם וּפְקֻ֣דֵיהֶ֔ם חֲמִשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וְשֵׁ֥שׁ מֵאֽוֹת׃ אֵלֶּה בְּנֵי בִּנְיָמִין לְמִשְׁפְּחוֹתֵיהֶם, וּמִסְפָּרָם אַרְבָּעִים וַחֲמִשָּׁה אֶלֶף וְשֵׁשׁ מֵאוֹת.

מב אֵ֤לֶּה בְנֵי־דָן֙ לְמִשְׁפְּחֹתָ֔ם לְשׁוּחָ֕ם מִשְׁפַּ֖חַת הַשּׁוּחָמִ֑י אֵ֛לֶּה מִשְׁפְּחֹ֥ת דָּ֖ן לְמִשְׁפְּחֹתָֽם׃ אֵלֶּה בְּנֵי דָּן לְמִשְׁפְּחוֹתֵיהֶם: לְשׁוּחָם מִשְׁפַּחַת הַשּׁוּחָמִי, אֵלֶּה מִשְׁפְּחוֹת דָּן לְמִשְׁפְּחוֹתֵיהֶם.
מג כׇּל־מִשְׁפְּחֹ֥ת הַשּׁוּחָמִ֖י לִפְקֻדֵיהֶ֑ם אַרְבָּעָ֧ה וְשִׁשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ כָּל מִשְׁפְּחֹת הַשּׁוּחָמִי שֶׁנִּסְפְּרוּ – שִׁשִּׁים וְאַרְבָּעָה אֶלֶף וְאַרְבַּע מֵאוֹת.
ס
מד בְּנֵ֣י אָשֵׁר֮ לְמִשְׁפְּחֹתָם֒ לְיִמְנָ֗ה מִשְׁפַּ֙חַת֙ הַיִּמְנָ֔ה לְיִשְׁוִ֕י מִשְׁפַּ֖חַת הַיִּשְׁוִ֑י לִבְרִיעָ֕ה מִשְׁפַּ֖חַת הַבְּרִיעִֽי׃ בְּנֵי אָשֵׁר לְמִשְׁפְּחוֹתֵיהֶם: לְיִמְנָה מִשְׁפַּחַת הַיִּמְנָה, לְיִשְׁוִי מִשְׁפַּחַת הַיִּשְׁוִי, לִבְרִיעָה מִשְׁפַּחַת הַבְּרִיעִי.
מה לִבְנֵ֣י בְרִיעָ֔ה לְחֶ֕בֶר מִשְׁפַּ֖חַת הַֽחֶבְרִ֑י לְמַ֨לְכִּיאֵ֔ל מִשְׁפַּ֖חַת הַמַּלְכִּיאֵלִֽי׃ לִבְנֵי בְּרִיעָה: לְחֶבֶר מִשְׁפַּחַת הַחֶבְרִי, לְמַלְכִּיאֵל מִשְׁפַּחַת הַמַּלְכִּיאֵלִי.
מו וְשֵׁ֥ם בַּת־אָשֵׁ֖ר שָֽׂרַח׃ וְשֵׁם הַבַּת שֶׁל אָשֵׁר – שֶׂרַח.
מז אֵ֛לֶּה מִשְׁפְּחֹ֥ת בְּנֵי־אָשֵׁ֖ר לִפְקֻדֵיהֶ֑ם שְׁלֹשָׁ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ אֵלֶּה מִשְׁפְּחוֹת בְּנֵי אָשֵׁר שֶׁנִּסְפְּרוּ – חֲמִשִּׁים וּשְׁלוֹשָׁה אֶלֶף וְאַרְבַּע מֵאוֹת.
ס
מח בְּנֵ֤י נַפְתָּלִי֙ לְמִשְׁפְּחֹתָ֔ם לְיַ֨חְצְאֵ֔ל מִשְׁפַּ֖חַת הַיַּחְצְאֵלִ֑י לְגוּנִ֕י מִשְׁפַּ֖חַת הַגּוּנִֽי׃ בְּנֵי נַפְתָּלִי לְמִשְׁפְּחוֹתֵיהֶם: לְיַחְצְאֵל מִשְׁפַּחַת הַיַּחְצְאֵלִי, לְגוּנִי מִשְׁפַּחַת הַגּוּנִי.
מט לְיֵ֕צֶר מִשְׁפַּ֖חַת הַיִּצְרִ֑י לְשִׁלֵּ֕ם מִשְׁפַּ֖חַת הַשִּׁלֵּמִֽי׃ לְיֵצֶר מִשְׁפַּחַת הַיִּצְרִי, לְשִׁלֵּם מִשְׁפַּחַת הַשִּׁלֵּמִי.
נ אֵ֛לֶּה מִשְׁפְּחֹ֥ת נַפְתָּלִ֖י לְמִשְׁפְּחֹתָ֑ם וּפְקֻ֣דֵיהֶ֔ם חֲמִשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ אֵלֶּה מִשְׁפְּחוֹת נַפְתָּלִי לְמִשְׁפְּחוֹתֵיהֶם, וּמִסְפָּרָם אַרְבָּעִים וַחֲמִשָּׁה אֶלֶף וְאַרְבַּע מֵאוֹת.
נא אֵ֗לֶּה פְּקוּדֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל שֵׁשׁ־מֵא֥וֹת אֶ֖לֶף וָאָ֑לֶף שְׁבַ֥ע מֵא֖וֹת וּשְׁלֹשִֽׁים׃ זֶה מִסְפַּר בְּנֵי יִשְׂרָאֵל שֶׁנִּסְפְּרוּ – שֵׁשׁ מֵאוֹת וְאֶחָד אֶלֶף שְׁבַע מֵאוֹת וּשְׁלוֹשִׁים.

שלישי[עריכה]

פ
נג לָאֵ֗לֶּה תֵּחָלֵ֥ק הָאָ֛רֶץ בְּנַחֲלָ֖ה בְּמִסְפַּ֥ר שֵׁמֽוֹת׃ הָאָרֶץ תִּתְחַלֵּק לְנַחֲלָה לָאֲנָשִׁים הָאֵלֶּה שֶׁנִּסְפְּרוּ לְפִי מִסְפַּר הַשֵּׁמוֹת הָאֵלֶּה.
נד לָרַ֗ב תַּרְבֶּה֙ נַחֲלָת֔וֹ וְלַמְעַ֕ט תַּמְעִ֖יט נַחֲלָת֑וֹ אִ֚ישׁ לְפִ֣י פְקֻדָ֔יו יֻתַּ֖ן נַחֲלָתֽוֹ׃ לְשֵׁבֶט גָּדוֹל תַּגְדִּיל אֶת נַחֲלָתוֹ, וּלְשֵׁבֶט קָטָן תַּקְטִין אֶת נַחֲלָתוֹ – לְפִי מִסְפַּר אַנְשֵׁי כָּל שֵׁבֶט כָּךְ תִּנָּתֵן נַחֲלָתוֹ.
נה אַךְ־בְּגוֹרָ֕ל יֵחָלֵ֖ק אֶת־הָאָ֑רֶץ לִשְׁמ֥וֹת מַטּוֹת־אֲבֹתָ֖ם יִנְחָֽלוּ׃ אֲבָל הָאָרֶץ תִּתְחַלֵּק בְּגוֹרָל – לְפִי הַשֵּׁמוֹת שֶׁיַּעֲלוּ בַּגּוֹרָל יִקָּבַע הַמָּקוֹם שֶׁיִּנְחֲלוּ שִׁבְטֵי אֲבוֹתֵיהֶם.
נו עַל־פִּי֙ הַגּוֹרָ֔ל תֵּחָלֵ֖ק נַחֲלָת֑וֹ בֵּ֥ין רַ֖ב לִמְעָֽט׃ עַל פִּי הַגּוֹרָל תִּתְחַלֵּק הַנַּחֲלָה בֵּין מִשְׁפָּחָה גְּדוֹלָה לְמִשְׁפָּחָה קְטַנָּה.
ס
נז וְאֵ֨לֶּה פְקוּדֵ֣י הַלֵּוִי֮ לְמִשְׁפְּחֹתָם֒ לְגֵרְשׁ֗וֹן מִשְׁפַּ֙חַת֙ הַגֵּ֣רְשֻׁנִּ֔י לִקְהָ֕ת מִשְׁפַּ֖חַת הַקְּהָתִ֑י לִמְרָרִ֕י מִשְׁפַּ֖חַת הַמְּרָרִֽי׃ וְאֵלֶּה הַנִּסְפָּרִים מִשֵּׁבֶט הַלֵּוִי לְמִשְׁפְּחוֹתֵיהֶם: לְגֵרְשׁוֹן מִשְׁפַּחַת הַגֵּרְשֻׁנִּי, לִקְהָת מִשְׁפַּחַת הַקְּהָתִי, לִמְרָרִי מִשְׁפַּחַת הַמְּרָרִי.
נח אֵ֣לֶּה ׀ מִשְׁפְּחֹ֣ת לֵוִ֗י מִשְׁפַּ֨חַת הַלִּבְנִ֜י מִשְׁפַּ֤חַת הַֽחֶבְרֹנִי֙ מִשְׁפַּ֤חַת הַמַּחְלִי֙ מִשְׁפַּ֣חַת הַמּוּשִׁ֔י מִשְׁפַּ֖חַת הַקׇּרְחִ֑י וּקְהָ֖ת הוֹלִ֥ד אֶת־עַמְרָֽם׃ אֵלֶּה מִשְׁפְּחוֹת לֵוִי: מִשְׁפַּחַת הַלִּבְנִי, מִשְׁפַּחַת הַחֶבְרוֹנִי, מִשְׁפַּחַת הַמַּחְלִי, מִשְׁפַּחַת הַמּוּשִׁי, מִשְׁפַּחַת הַקָּרְחִי, וּקְהָת הוֹלִיד אֶת עַמְרָם.
נט וְשֵׁ֣ם ׀ אֵ֣שֶׁת עַמְרָ֗ם יוֹכֶ֙בֶד֙ בַּת־לֵוִ֔י אֲשֶׁ֨ר יָלְדָ֥ה אֹתָ֛הּ לְלֵוִ֖י בְּמִצְרָ֑יִם וַתֵּ֣לֶד לְעַמְרָ֗ם אֶֽת־אַהֲרֹן֙ וְאֶת־מֹשֶׁ֔ה וְאֵ֖ת מִרְיָ֥ם אֲחֹתָֽם׃ וְשֵׁם אֵשֶׁת עַמְרָם יוֹכֶבֶד בַּת לֵוִי שֶׁאֵשֶׁת לֵוִי יָלְדָה לְלֵוִי בְּמִצְרַיִם, הִיא יָלְדָה לְעַמְרָם אֶת אַהֲרֹן אֶת מֹשֶׁה וְאֶת מִרְיָם אֲחוֹתָם.
ס וַיִּוָּלֵ֣ד לְאַהֲרֹ֔ן אֶת־נָדָ֖ב וְאֶת־אֲבִיה֑וּא אֶת־אֶלְעָזָ֖ר וְאֶת־אִיתָמָֽר׃ לְאַהֲרֹן נוֹלְדוּ: נָדָב אֲבִיהוּא אֶלְעָזָר וְאִיתָמָר.
סא וַיָּ֥מׇת נָדָ֖ב וַאֲבִיה֑וּא בְּהַקְרִיבָ֥ם אֵשׁ־זָרָ֖ה לִפְנֵ֥י יְהֹוָֽה׃ נָדָב וַאֲבִיהוּא מֵתוּ כְּשֶׁהִקְרִיבוּ לִפְנֵי ה' אֵשׁ שֶׁהוּא לֹא רָצָה בָּהּ.
סב וַיִּהְי֣וּ פְקֻדֵיהֶ֗ם שְׁלֹשָׁ֤ה וְעֶשְׂרִים֙ אֶ֔לֶף כׇּל־זָכָ֖ר מִבֶּן־חֹ֣דֶשׁ וָמָ֑עְלָה כִּ֣י ׀ לֹ֣א הׇתְפָּקְד֗וּ בְּתוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל כִּ֠י לֹא־נִתַּ֤ן לָהֶם֙ נַחֲלָ֔ה בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל׃ מִסְפָּרָם הָיָה עֶשְׂרִים וּשְׁלוֹשָׁה אֶלֶף – כָּל זָכָר מִגִּיל חוֹדֶשׁ וָמַעְלָה, וְלֹא מִגִּיל עֶשְׂרִים כִּי הֵם לֹא נִסְפְּרוּ יַחַד עִם בְּנֵי יִשְׂרָאֵל כִּי לֹא נִתְּנָה לָהֶם נַחֲלָה יַחַד עִם בְּנֵי יִשְׂרָאֵל.
סג אֵ֚לֶּה פְּקוּדֵ֣י מֹשֶׁ֔ה וְאֶלְעָזָ֖ר הַכֹּהֵ֑ן אֲשֶׁ֨ר פָּֽקְד֜וּ אֶת־בְּנֵ֤י יִשְׂרָאֵל֙ בְּעַֽרְבֹ֣ת מוֹאָ֔ב עַ֖ל יַרְדֵּ֥ן יְרֵחֽוֹ׃ אֵלֶּה הָאֲנָשִׁים שֶׁמֹּשֶׁה וְאֶלְעָזָר הַכֹּהֵן סָפְרוּ – כְּשֶׁסָּפְרוּ אֶת בְּנֵי יִשְׂרָאֵל בַּמִּישׁוֹרִים שֶׁל מוֹאָב עַל יַד הַיַּרְדֵּן שֶׁלְּיַד יְרִיחוֹ.
סד וּבְאֵ֙לֶּה֙ לֹא־הָ֣יָה אִ֔ישׁ מִפְּקוּדֵ֣י מֹשֶׁ֔ה וְאַהֲרֹ֖ן הַכֹּהֵ֑ן אֲשֶׁ֥ר פָּקְד֛וּ אֶת־בְּנֵ֥י יִשְׂרָאֵ֖ל בְּמִדְבַּ֥ר סִינָֽי׃ וּבְאֵלֶּה לֹא הָיָה אַף אֶחָד מֵהָאֲנָשִׁים שֶׁמֹּשֶׁה וְאַהֲרֹן הַכֹּהֵן סָפְרוּ כְּשֶׁסָּפְרוּ אֶת בְּנֵי יִשְׂרָאֵל בְּמִדְבַּר סִינַי.
סה כִּֽי־אָמַ֤ר יְהֹוָה֙ לָהֶ֔ם מ֥וֹת יָמֻ֖תוּ בַּמִּדְבָּ֑ר וְלֹא־נוֹתַ֤ר מֵהֶם֙ אִ֔ישׁ כִּ֚י אִם־כָּלֵ֣ב בֶּן־יְפֻנֶּ֔ה וִיהוֹשֻׁ֖עַ בִּן־נֽוּן׃ כִּי ה' אָמַר עֲלֵיהֶם שֶׁהֵם יָמוּתוּ בַּמִּדְבָּר, וְלֹא נִשְׁאַר מֵהֶם אַף אֶחָד חוּץ מִכָּלֵב בֶּן יְפֻנֶּה וִיהוֹשֻׁעַ בִּן נוּן.
ס פָּרָשַׁת בְּנוֹת צְלָפְחָד
כז א וַתִּקְרַ֜בְנָה בְּנ֣וֹת צְלׇפְחָ֗ד בֶּן־חֵ֤פֶר בֶּן־גִּלְעָד֙ בֶּן־מָכִ֣יר בֶּן־מְנַשֶּׁ֔ה לְמִשְׁפְּחֹ֖ת מְנַשֶּׁ֣ה בֶן־יוֹסֵ֑ף וְאֵ֙לֶּה֙ שְׁמ֣וֹת בְּנֹתָ֔יו מַחְלָ֣ה נֹעָ֔ה וְחׇגְלָ֥ה וּמִלְכָּ֖ה וְתִרְצָֽה׃ בְּנוֹת צְלָפְחָד בֶּן חֵפֶר בֶּן גִּלְעָד בֶּן מָכִיר בֶּן מְנַשֶּׁה מִמִּשְׁפְּחוֹת מְנַשֶּׁה בֶּן יוֹסֵף נִגְּשׁוּ, וְאֵלֶּה שְׁמוֹת בְּנוֹתָיו: מַחְלָה נוֹעָה וְחָגְלָה וּמִלְכָּה וְתִרְצָה.
ב וַֽתַּעֲמֹ֜דְנָה לִפְנֵ֣י מֹשֶׁ֗ה וְלִפְנֵי֙ אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְלִפְנֵ֥י הַנְּשִׂיאִ֖ם וְכׇל־הָעֵדָ֑ה פֶּ֥תַח אֹֽהֶל־מוֹעֵ֖ד לֵאמֹֽר׃ הֵן עָמְדוּ לִפְנֵי מֹשֶׁה וְלִפְנֵי אֶלְעָזָר הַכֹּהֵן וְלִפְנֵי הַנְּשִׂיאִים וְכָל הַצִּבּוּר מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת, וְאָמְרוּ:
ג אָבִ֘ינוּ֮ מֵ֣ת בַּמִּדְבָּר֒ וְה֨וּא לֹא־הָיָ֜ה בְּת֣וֹךְ הָעֵדָ֗ה הַנּוֹעָדִ֛ים עַל־יְהֹוָ֖ה בַּעֲדַת־קֹ֑רַח כִּֽי־בְחֶטְא֣וֹ מֵ֔ת וּבָנִ֖ים לֹא־הָ֥יוּ לֽוֹ׃ אָבִינוּ מֵת בַּמִּדְבָּר וְהוּא לֹא הָיָה בַּקְּבוּצָה שֶׁהִתְאַסְּפוּ לְהַפְגִּין נֶגֶד ה' בַּצִּבּוּר שֶׁל קוֹרַח, אֶלָּא מֵת בְּחֶטְאוֹ הַפְּרָטִי, וְלֹא הָיוּ לוֹ בָּנִים.
ד לָ֣מָּה יִגָּרַ֤ע שֵׁם־אָבִ֙ינוּ֙ מִתּ֣וֹךְ מִשְׁפַּחְתּ֔וֹ כִּ֛י אֵ֥ין ל֖וֹ בֵּ֑ן תְּנָה־לָּ֣נוּ אֲחֻזָּ֔ה בְּת֖וֹךְ אֲחֵ֥י אָבִֽינוּ׃ לָמָּה שֶׁיֵּעָלֵם שֵׁם אָבִינוּ מִתּוֹךְ מִשְׁפַּחְתּוֹ רַק מִפְּנֵי שֶׁאֵין לוֹ בֵּן?! תֵּן לָנוּ אֲחֻזָּה יַחַד עִם אֲחֵי אָבִינוּ!
ה וַיַּקְרֵ֥ב מֹשֶׁ֛ה אֶת־מִשְׁפָּטָ֖ן לִפְנֵ֥י יְהֹוָֽה׃ מֹשֶׁה הִגִּישׁ אֶת שְׁאֵלַת בֵּרוּר תְּבִיעָתָן לִפְנֵי ה'.

רביעי[עריכה]

פ פָּרָשַׁת הַנַּחֲלוֹת
ו וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' אָמַר לְמֹשֶׁה:
ז כֵּ֗ן בְּנ֣וֹת צְלׇפְחָד֮ דֹּבְרֹת֒ נָתֹ֨ן תִּתֵּ֤ן לָהֶם֙ אֲחֻזַּ֣ת נַחֲלָ֔ה בְּת֖וֹךְ אֲחֵ֣י אֲבִיהֶ֑ם וְהַֽעֲבַרְתָּ֛ אֶת־נַחֲלַ֥ת אֲבִיהֶ֖ן לָהֶֽן׃ דִּבְרֵי בְּנוֹת צְלָפְחָד נְכוֹנִים – תֵּן לָהֶן אֲחֻזַּת קַרְקַע לְנַחֲלָה יַחַד עִם אֲחֵי אֲבִיהֶן – הַעֲבֵר אֶת נַחֲלַת אֲבִיהֶן לָהֶן.
ח וְאֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל תְּדַבֵּ֣ר לֵאמֹ֑ר אִ֣ישׁ כִּֽי־יָמ֗וּת וּבֵן֙ אֵ֣ין ל֔וֹ וְהַֽעֲבַרְתֶּ֥ם אֶת־נַחֲלָת֖וֹ לְבִתּֽוֹ׃ וְאֶל בְּנֵי יִשְׂרָאֵל תְּדַבֵּר וְתֹאמַר: אִם אִישׁ יָמוּת וְאֵין לוֹ בֵּן – תַּעֲבִירוּ אֶת נַחֲלָתוֹ לְבִתּוֹ.
ט וְאִם־אֵ֥ין ל֖וֹ בַּ֑ת וּנְתַתֶּ֥ם אֶת־נַחֲלָת֖וֹ לְאֶחָֽיו׃ וְאִם אֵין לוֹ בַּת, תִּתְּנוּ אֶת נַחֲלָתוֹ לְאֶחָיו.
י וְאִם־אֵ֥ין ל֖וֹ אַחִ֑ים וּנְתַתֶּ֥ם אֶת־נַחֲלָת֖וֹ לַאֲחֵ֥י אָבִֽיו׃ וְאִם אֵין לוֹ אַחִים, תִּתְּנוּ אֶת נַחֲלָתוֹ לַאֲחֵי אָבִיו.
יא וְאִם־אֵ֣ין אַחִים֮ לְאָבִיו֒ וּנְתַתֶּ֣ם אֶת־נַחֲלָת֗וֹ לִשְׁאֵר֞וֹ הַקָּרֹ֥ב אֵלָ֛יו מִמִּשְׁפַּחְתּ֖וֹ וְיָרַ֣שׁ אֹתָ֑הּ וְֽהָ֨יְתָ֜ה לִבְנֵ֤י יִשְׂרָאֵל֙ לְחֻקַּ֣ת מִשְׁפָּ֔ט כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ וְאִם אֵין לְאָבִיו אַחִים תִּתְּנוּ אֶת נַחֲלָתוֹ לִקְרוֹבוֹ הַקָּרוֹב אֵלָיו בְּיוֹתֵר מִמִּשְׁפַּחְתּוֹ וְהוּא יִירַשׁ אוֹתָהּ, דִּין זֶה יִהְיֶה לִבְנֵי יִשְׂרָאֵל לְחוֹק שֶׁיִּשְׁפְּטוּ כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.
פ פָּרָשַׁת סְמִיכַת יְהוֹשֻׁעַ
יב וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה עֲלֵ֛ה אֶל־הַ֥ר הָעֲבָרִ֖ים הַזֶּ֑ה וּרְאֵה֙ אֶת־הָאָ֔רֶץ אֲשֶׁ֥ר נָתַ֖תִּי לִבְנֵ֥י יִשְׂרָאֵֽל׃ ה' אָמַר לְמֹשֶׁה: עֲלֵה לְהַר 'מַעֲבַר הַיַּרְדֵּן' הַזֶּה, וּרְאֵה אֶת הָאָרֶץ שֶׁנָּתַתִּי לִבְנֵי יִשְׂרָאֵל.
יג וְרָאִ֣יתָה אֹתָ֔הּ וְנֶאֱסַפְתָּ֥ אֶל־עַמֶּ֖יךָ גַּם־אָ֑תָּה כַּאֲשֶׁ֥ר נֶאֱסַ֖ף אַהֲרֹ֥ן אָחִֽיךָ׃ וְאַחֲרֵי שֶׁתִּרְאֶה אוֹתָהּ תֵּאָסֵף גַּם נִשְׁמָתְךָ אֶל מְקוֹם הַנְּשָׁמוֹת כְּמוֹ שֶׁנֶּאֶסְפָה נִשְׁמַת אַהֲרֹן אָחִיךָ.
יד כַּאֲשֶׁר֩ מְרִיתֶ֨ם פִּ֜י בְּמִדְבַּר־צִ֗ן בִּמְרִיבַת֙ הָֽעֵדָ֔ה לְהַקְדִּישֵׁ֥נִי בַמַּ֖יִם לְעֵינֵיהֶ֑ם הֵ֛ם מֵֽי־מְרִיבַ֥ת קָדֵ֖שׁ מִדְבַּר־צִֽן׃ עוֹנֶשׁ עַל שֶׁסֵּרַבְתֶּם לִפְקֻדָּתִי בְּמִדְבַּר צִין בִּמְרִיבַת הַצִּבּוּר, וְהָיָה עֲלֵיכֶם לְהַקְדִּישׁ אֶת שְׁמִי בְּעִנְיַן הַמַּיִם לְעֵינֵיהֶם, אֵלֶּה הֵם מֵי הַמְּרִיבָה שֶׁהָיוּ בְּקָדֵשׁ שֶׁבְּמִדְבַּר צִין.
ס
טו וַיְדַבֵּ֣ר מֹשֶׁ֔ה אֶל־יְהֹוָ֖ה לֵאמֹֽר׃ מֹשֶׁה דִּבֵּר אֶל ה' וְאָמַר:
טז יִפְקֹ֣ד יְהֹוָ֔ה אֱלֹהֵ֥י הָרוּחֹ֖ת לְכׇל־בָּשָׂ֑ר אִ֖ישׁ עַל־הָעֵדָֽה׃ ה' אֱלֹהֵי הַנְּשָׁמוֹת שֶׁל כָּל בָּשָׂר וָדָם, יְמַנֶּה בְּבַקָּשָׁה אִישׁ שֶׁיַּנְהִיג אֶת הַצִּבּוּר.
יז אֲשֶׁר־יֵצֵ֣א לִפְנֵיהֶ֗ם וַאֲשֶׁ֤ר יָבֹא֙ לִפְנֵיהֶ֔ם וַאֲשֶׁ֥ר יוֹצִיאֵ֖ם וַאֲשֶׁ֣ר יְבִיאֵ֑ם וְלֹ֤א תִהְיֶה֙ עֲדַ֣ת יְהֹוָ֔ה כַּצֹּ֕אן אֲשֶׁ֥ר אֵין־לָהֶ֖ם רֹעֶֽה׃ שֶׁיֵּצֵא לִפְנֵיהֶם לַקְּרָב, וְשֶׁיָּבוֹא לִפְנֵיהֶם בְּשׁוּבָם, וְשֶׁיּוֹצִיא וְיָבִיא אוֹתָם לְהַנְהִיגָם, כְּדֵי שֶׁלֹּא יִהְיֶה קְהַל ה' כְּצֹאן שֶׁאֵין לוֹ רוֹעֶה.
יח וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה קַח־לְךָ֙ אֶת־יְהוֹשֻׁ֣עַ בִּן־נ֔וּן אִ֖ישׁ אֲשֶׁר־ר֣וּחַ בּ֑וֹ וְסָמַכְתָּ֥ אֶת־יָדְךָ֖ עָלָֽיו׃ ה' אָמַר לְמֹשֶׁה: קַח לְךָ אֶת יְהוֹשֻׁעַ בִּן נוּן – אִישׁ שֶׁיֵּשׁ בּוֹ רוּחַ נְבוּאָה, וְתָשִׂים אֶת יָדְךָ עַל רֹאשׁוֹ.
יט וְהַֽעֲמַדְתָּ֣ אֹת֗וֹ לִפְנֵי֙ אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְלִפְנֵ֖י כׇּל־הָעֵדָ֑ה וְצִוִּיתָ֥ה אֹת֖וֹ לְעֵינֵיהֶֽם׃ וְתַעֲמִיד אוֹתוֹ לִפְנֵי אֶלְעָזָר הַכֹּהֵן וְלִפְנֵי כָּל הַצִּבּוּר, וּתְמַנֶּה אוֹתוֹ לְעֵינֵיהֶם.
כ וְנָתַתָּ֥ה מֵהֽוֹדְךָ֖ עָלָ֑יו לְמַ֣עַן יִשְׁמְע֔וּ כׇּל־עֲדַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ וְתָשִׂים מִכְּבוֹדְךָ עָלָיו כְּדֵי שֶׁיִּשְׁמְעוּ בְּקוֹלוֹ כָּל הַצִּבּוּר שֶׁל בְּנֵי יִשְׂרָאֵל.
כא וְלִפְנֵ֨י אֶלְעָזָ֤ר הַכֹּהֵן֙ יַעֲמֹ֔ד וְשָׁ֥אַל ל֛וֹ בְּמִשְׁפַּ֥ט הָאוּרִ֖ים לִפְנֵ֣י יְהֹוָ֑ה עַל־פִּ֨יו יֵצְא֜וּ וְעַל־פִּ֣יו יָבֹ֗אוּ ה֛וּא וְכׇל־בְּנֵי־יִשְׂרָאֵ֥ל אִתּ֖וֹ וְכׇל־הָעֵדָֽה׃ וְהוּא יַעֲמוֹד לִפְנֵי אֶלְעָזָר הַכֹּהֵן שֶׁיִּשְׁאַל עֲבוּרוֹ מֵה' אֶת הוֹרָאַת הָאֲבָנִים הַמְּאִירוֹת בַּחוֹשֶׁן שֶׁעוֹנוֹת לִשְׁאֵלוֹת, לְפִי הוֹרָאוֹתָיו יֵצְאוּ לַקְּרָב וּלְפִי הוֹרָאוֹתָיו יָשׁוּבוּ הוּא וְכָל בְּנֵי יִשְׂרָאֵל אִתּוֹ וְכָל הַצִּבּוּר.
כב וַיַּ֣עַשׂ מֹשֶׁ֔ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֹת֑וֹ וַיִּקַּ֣ח אֶת־יְהוֹשֻׁ֗עַ וַיַּֽעֲמִדֵ֙הוּ֙ לִפְנֵי֙ אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְלִפְנֵ֖י כׇּל־הָעֵדָֽה׃ מֹשֶׁה עָשָׂה כְּמוֹ שֶׁה' צִוָּה אוֹתוֹ, הוּא לָקַח אֶת יְהוֹשֻׁעַ וְהֶעֱמִיד אוֹתוֹ לִפְנֵי אֶלְעָזָר הַכֹּהֵן וְלִפְנֵי כָּל הַצִּבּוּר.
כג וַיִּסְמֹ֧ךְ אֶת־יָדָ֛יו עָלָ֖יו וַיְצַוֵּ֑הוּ כַּאֲשֶׁ֛ר דִּבֶּ֥ר יְהֹוָ֖ה בְּיַד־מֹשֶֽׁה׃ הוּא שָׂם עָלָיו אֶת יָדָיו וּמִנָּה אוֹתוֹ, כְּמוֹ שֶׁה' דִּבֵּר אֶל מֹשֶׁה.

חמישי[עריכה]

פ פָּרָשַׁת הַמּוּסָפִים
כח א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב צַ֚ו אֶת־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָמַרְתָּ֖ אֲלֵהֶ֑ם אֶת־קׇרְבָּנִ֨י לַחְמִ֜י לְאִשַּׁ֗י רֵ֚יחַ נִֽיחֹחִ֔י תִּשְׁמְר֕וּ לְהַקְרִ֥יב לִ֖י בְּמוֹעֲדֽוֹ׃ צַוֵּה אֶת בְּנֵי יִשְׂרָאֵל וְתֹאמַר לָהֶם: הַקְפִּידוּ לְהַקְרִיב לְפָנַי מַאֲכָל בִּזְמַנּוֹ – הַקָּרְבָּן הָעוֹלֶה בָּאֵשׁ לְנַחַת רוּחַ לְפָנַי.
ג וְאָמַרְתָּ֣ לָהֶ֔ם זֶ֚ה הָֽאִשֶּׁ֔ה אֲשֶׁ֥ר תַּקְרִ֖יבוּ לַיהֹוָ֑ה כְּבָשִׂ֨ים בְּנֵֽי־שָׁנָ֧ה תְמִימִ֛ם שְׁנַ֥יִם לַיּ֖וֹם עֹלָ֥ה תָמִֽיד׃ וְתֹאמַר לָהֶם: זֶה הַקָּרְבַּן הָעוֹלֶה בָּאֵשׁ שֶׁתַּקְרִיבוּ לַה' – כְּבָשִׂים שֶׁגִּילָם עַד שָׁנָה לְלֹא מוּם, שְׁנַיִם בְּכָל יוֹם לְעוֹלָה תָּמִיד.
ד אֶת־הַכֶּ֥בֶשׂ אֶחָ֖ד תַּעֲשֶׂ֣ה בַבֹּ֑קֶר וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּעֲשֶׂ֖ה בֵּ֥ין הָֽעַרְבָּֽיִם׃ כֶּבֶשׂ אֶחָד תַּקְרִיב בַּבּוֹקֶר, וְאֶת הַכֶּבֶשׂ הַשֵּׁנִי תַּקְרִיב אַחַר הַצָּהֳרַיִם.
ה וַעֲשִׂירִ֧ית הָאֵיפָ֛ה סֹ֖לֶת לְמִנְחָ֑ה בְּלוּלָ֛ה בְּשֶׁ֥מֶן כָּתִ֖ית רְבִיעִ֥ת הַהִֽין׃ וְכַמּוּת שֶׁל עֲשִׂירִית אֵיפָה (כִּ2.16 לִיטֶר) סוֹלֶת לְמִנְחָה מְעֻרְבֶּבֶת בְּשֶׁמֶן כָּתוּשׁ בְּכַמּוּת שֶׁל רְבִיעִית הַהִין (כִּ900 מ"ל).
ו עֹלַ֖ת תָּמִ֑יד הָעֲשֻׂיָה֙ בְּהַ֣ר סִינַ֔י לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַֽיהֹוָֽה׃ עוֹלָה שֶׁתֻּקְרַב תָּמִיד, כְּמוֹ שֶׁהֻקְרְבָה בְּהַר סִינַי, לְנַחַת רוּחַ – קָרְבָּן עוֹלֶה בָּאֵשׁ לַה'.
ז וְנִסְכּוֹ֙ רְבִיעִ֣ת הַהִ֔ין לַכֶּ֖בֶשׂ הָאֶחָ֑ד בַּקֹּ֗דֶשׁ הַסֵּ֛ךְ נֶ֥סֶךְ שֵׁכָ֖ר לַיהֹוָֽה׃ הַיַּיִן שֶׁמּוֹזְגִים אִתָּהּ יִהְיֶה בְּכַמּוּת שֶׁל רְבִיעִית הַהִין (כִּ900 מ"ל) לְכָל כֶּבֶשׂ, יִמְזְגוּ בַּמִּקְדָשׁ עַל הַמִּזְבֵּחַ יַיִן יָשָׁן לַה'.
ח וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּעֲשֶׂ֖ה בֵּ֣ין הָֽעַרְבָּ֑יִם כְּמִנְחַ֨ת הַבֹּ֤קֶר וּכְנִסְכּוֹ֙ תַּעֲשֶׂ֔ה אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַיהֹוָֽה׃ וְאֶת הַכֶּבֶשׂ הַשֵּׁנִי תַּקְרִיב אַחַר הַצָּהֳרַיִם, וְתָבִיא אִתּוֹ מִנְחָה וְיַיִן כְּמוֹ בַּבּוֹקֶר, קָרְבָּן עוֹלֶה בָּאֵשׁ לְנַחַת רוּחַ לַה'.
פ
ט וּבְיוֹם֙ הַשַּׁבָּ֔ת שְׁנֵֽי־כְבָשִׂ֥ים בְּנֵֽי־שָׁנָ֖ה תְּמִימִ֑ם וּשְׁנֵ֣י עֶשְׂרֹנִ֗ים סֹ֧לֶת מִנְחָ֛ה בְּלוּלָ֥ה בַשֶּׁ֖מֶן וְנִסְכּֽוֹ׃ וּבְיוֹם הַשַּׁבָּת תַּקְרִיבוּ שְׁנֵי כְּבָשִׂים שֶׁגִּילָם עַד שָׁנָה לְלֹא מוּם, וּשְׁתֵּי עֲשִׂירִיּוֹת הָאֵיפָה (כְּ⅓4 לִיטֶר) שֶׁל סוֹלֶת מְעֻרְבֶּבֶת בְּשֶׁמֶן לְמִנְחָה, וְהַיַּיִן שֶׁלּוֹ.
י עֹלַ֥ת שַׁבַּ֖ת בְּשַׁבַּתּ֑וֹ עַל־עֹלַ֥ת הַתָּמִ֖יד וְנִסְכָּֽהּ׃ עוֹלָה קְבוּעָה בְּכָל שַׁבָּת, נוֹסָף עַל הָעוֹלָה הַקְּבוּעָה תָּמִיד וּשְׁפִיכַת הַיַּיִן שֶׁלָּהּ.
פ
יא וּבְרָאשֵׁי֙ חׇדְשֵׁיכֶ֔ם תַּקְרִ֥יבוּ עֹלָ֖ה לַיהֹוָ֑ה פָּרִ֨ים בְּנֵֽי־בָקָ֤ר שְׁנַ֙יִם֙ וְאַ֣יִל אֶחָ֔ד כְּבָשִׂ֧ים בְּנֵי־שָׁנָ֛ה שִׁבְעָ֖ה תְּמִימִֽם׃ וּבְרָאשֵׁי חָדְשֵׁיכֶם תַּקְרִיבוּ עוֹלָה לַה' – שְׁנֵי פָּרִים, אַיִל אֶחָד, וְשִׁבְעָה כְּבָשִׂים שֶׁגִּילָם עַד שָׁנָה לְלֹא מוּם.
יב וּשְׁלֹשָׁ֣ה עֶשְׂרֹנִ֗ים סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן לַפָּ֖ר הָאֶחָ֑ד וּשְׁנֵ֣י עֶשְׂרֹנִ֗ים סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן לָאַ֖יִל הָֽאֶחָֽד׃ וּשָׁלוֹשׁ עֲשִׂירִיּוֹת הָאֵיפָה (כְּ6.5 לִיטֶר) סוֹלֶת מְעֻרְבֶּבֶת בְּשֶׁמֶן לְכָל פָּר לְמִנְחָה, וּשְׁתֵּי עֲשִׂירִיּוֹת הָאֵיפָה (כְּ⅓4 לִיטֶר) סוֹלֶת מְעֻרְבֶּבֶת בְּשֶׁמֶן לְכָל אַיִל לְמִנְחָה.
יג וְעִשָּׂרֹ֣ן עִשָּׂר֗וֹן סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן לַכֶּ֖בֶשׂ הָאֶחָ֑ד עֹלָה֙ רֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַיהֹוָֽה׃ וַעֲשִׂירִית הָאֵיפָה (כִּ2.16 לִיטֶר) סוֹלֶת מְעֻרְבֶּבֶת בְּשֶׁמֶן לְכָל כֶּבֶשׂ לְמִנְחָה, עוֹלָה לְנַחַת רוּחַ, קָרְבָּן עוֹלֶה בָּאֵשׁ לַה'.
יד וְנִסְכֵּיהֶ֗ם חֲצִ֣י הַהִין֩ יִהְיֶ֨ה לַפָּ֜ר וּשְׁלִישִׁ֧ת הַהִ֣ין לָאַ֗יִל וּרְבִיעִ֥ת הַהִ֛ין לַכֶּ֖בֶשׂ יָ֑יִן זֹ֣את עֹלַ֥ת חֹ֙דֶשׁ֙ בְּחׇדְשׁ֔וֹ לְחׇדְשֵׁ֖י הַשָּׁנָֽה׃ וְהַיַּיִן לִמְזִיגָה אִתָּם: חֲצִי הִין יַיִן (כְּלִיטֶר וּ800 מ"ל) יִהְיֶה לְכָל פָּר וּשְׁלִישׁ הִין (כְּלִיטֶר וּ200 מ"ל) לְכָל אַיִל וּרְבִיעִית הִין (כִּ900 מ"ל) לְכָל כֶּבֶשׂ, זֹאת עוֹלַת כָּל חוֹדֶשׁ מֵחָדְשֵׁי הַשָּׁנָה.
טו וּשְׂעִ֨יר עִזִּ֥ים אֶחָ֛ד לְחַטָּ֖את לַיהֹוָ֑ה עַל־עֹלַ֧ת הַתָּמִ֛יד יֵעָשֶׂ֖ה וְנִסְכּֽוֹ׃ וְתַיִשׁ אֶחָד לְקָרְבַּן חַטָּאת לַה' יֻקְרַב בְּנוֹסָף לָעוֹלָה שֶׁל קָרְבַּן הַתָּמִיד, עִם הַיַּיִן שֶׁאִתּוֹ.

שישי[עריכה]

ס
טז וּבַחֹ֣דֶשׁ הָרִאשׁ֗וֹן בְּאַרְבָּעָ֥ה עָשָׂ֛ר י֖וֹם לַחֹ֑דֶשׁ פֶּ֖סַח לַיהֹוָֽה׃ וּבַחוֹדֶשׁ הָרִאשׁוֹן בַּיּוֹם הָאַרְבָּעָה עָשָׂר לַחוֹדֶשׁ – הַקְרָבַת קָרְבַּן פֶּסַח לַה'.
יז וּבַחֲמִשָּׁ֨ה עָשָׂ֥ר י֛וֹם לַחֹ֥דֶשׁ הַזֶּ֖ה חָ֑ג שִׁבְעַ֣ת יָמִ֔ים מַצּ֖וֹת יֵאָכֵֽל׃ וּבַיּוֹם הַחֲמִשָּׁה עָשָׂר לַחוֹדֶשׁ הַזֶּה – חָג, שִׁבְעָה יָמִים יֹאכְלוּ מַצּוֹת.
יח בַּיּ֥וֹם הָרִאשׁ֖וֹן מִקְרָא־קֹ֑דֶשׁ כׇּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַעֲשֽׂוּ׃ בַּיּוֹם הָרִאשׁוֹן מְאוֹרָע קָדוֹשׁ – אַל תַּעֲשׂוּ שׁוּם מְלָאכָה שֶׁל עֲבוֹדָה.
יט וְהִקְרַבְתֶּ֨ם אִשֶּׁ֤ה עֹלָה֙ לַֽיהֹוָ֔ה פָּרִ֧ים בְּנֵי־בָקָ֛ר שְׁנַ֖יִם וְאַ֣יִל אֶחָ֑ד וְשִׁבְעָ֤ה כְבָשִׂים֙ בְּנֵ֣י שָׁנָ֔ה תְּמִימִ֖ם יִהְי֥וּ לָכֶֽם׃ וְתַקְרִיבוּ לָאֵשׁ קָרְבַּן עוֹלָה לַה' – שְׁנֵי פָּרִים וְאַיִל אֶחָד וְשִׁבְעָה כְּבָשִׂים שֶׁגִּילָם עַד שָׁנָה, לְלֹא מוּם כְּדֵי שֶׁיִּהְיוּ רְצוּיִים בְּעֵינֵי ה' עֲבוּרְכֶם.
כ וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֨ה עֶשְׂרֹנִ֜ים לַפָּ֗ר וּשְׁנֵ֧י עֶשְׂרֹנִ֛ים לָאַ֖יִל תַּעֲשֽׂוּ׃ וּמִנְחוֹתֵיהֶם סוֹלֶת מְעֻרְבֶּבֶת בְּשֶׁמֶן שָׁלוֹשׁ עֲשִׂירִיּוֹת הָאֵיפָה (כְּ6.5 לִיטֶר) לְכָל פָּר וּשְׁתֵּי עֲשִׂירִיּוֹת הָאֵיפָה (כְּ⅓4 לִיטֶר) תַּעֲשׂוּ לְכָל אַיִל.
כא עִשָּׂר֤וֹן עִשָּׂרוֹן֙ תַּעֲשֶׂ֔ה לַכֶּ֖בֶשׂ הָאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים׃ עֲשִׂירִית אֵיפָה אַחַת (כִּ2.16 לִיטֶר) תַּעֲשֶׂה לְכָל כֶּבֶשׂ לְשִׁבְעַת הַכְּבָשִׂים.
כב וּשְׂעִ֥יר חַטָּ֖את אֶחָ֑ד לְכַפֵּ֖ר עֲלֵיכֶֽם׃ וְתַיִשׁ אֶחָד לְחַטָּאת לְכַפֵּר עֲלֵיכֶם.
כג מִלְּבַד֙ עֹלַ֣ת הַבֹּ֔קֶר אֲשֶׁ֖ר לְעֹלַ֣ת הַתָּמִ֑יד תַּעֲשׂ֖וּ אֶת־אֵֽלֶּה׃ תַּעֲשׂוּ אֶת אֵלֶּה נוֹסָף עַל עוֹלַת הַבּוֹקֶר – הָעוֹלָה שֶׁל קָרְבַּן הַתָּמִיד.
כד כָּאֵ֜לֶּה תַּעֲשׂ֤וּ לַיּוֹם֙ שִׁבְעַ֣ת יָמִ֔ים לֶ֛חֶם אִשֵּׁ֥ה רֵֽיחַ־נִיחֹ֖חַ לַיהֹוָ֑ה עַל־עוֹלַ֧ת הַתָּמִ֛יד יֵעָשֶׂ֖ה וְנִסְכּֽוֹ׃ כָּךְ תַּעֲשׂוּ בְּכָל יוֹם בְּמֶשֶׁךְ שִׁבְעָה יָמִים, קָרְבָּן עוֹלֶה בָּאֵשׁ כְּמַאֲכָל לְנַחַת רוּחַ לַה', הוּא יֻקְרַב בְּנוֹסָף לָעוֹלָה שֶׁל קָרְבַּן הַתָּמִיד עִם שְׁפִיכַת הַיַּיִן שֶׁאִתּוֹ.
כה וּבַיּוֹם֙ הַשְּׁבִיעִ֔י מִקְרָא־קֹ֖דֶשׁ יִהְיֶ֣ה לָכֶ֑ם כׇּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַעֲשֽׂוּ׃ וּבַיּוֹם הַשְּׁבִיעִי יִהְיֶה לָכֶם מְאוֹרָע קָדוֹשׁ, אַל תַּעֲשׂוּ שׁוּם מְלָאכָה שֶׁל עֲבוֹדָה.

כו וּבְי֣וֹם הַבִּכּוּרִ֗ים בְּהַקְרִ֨יבְכֶ֜ם מִנְחָ֤ה חֲדָשָׁה֙ לַֽיהֹוָ֔ה בְּשָׁבֻעֹ֖תֵיכֶ֑ם מִֽקְרָא־קֹ֙דֶשׁ֙ יִהְיֶ֣ה לָכֶ֔ם כׇּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַעֲשֽׂוּ׃ וּבְיוֹם הַבִּכּוּרִים כְּשֶׁתַּקְרִיבוּ מִנְחָה חֲדָשָׁה לַה' בְּסִיּוּם סְפִירַתְכֶם אֶת הַשָּׁבוּעוֹת יִהְיֶה לָכֶם מְאוֹרָע קָדוֹשׁ – אַל תַּעֲשׂוּ שׁוּם מְלָאכָה שֶׁל עֲבוֹדָה.
כז וְהִקְרַבְתֶּ֨ם עוֹלָ֜ה לְרֵ֤יחַ נִיחֹ֙חַ֙ לַֽיהֹוָ֔ה פָּרִ֧ים בְּנֵי־בָקָ֛ר שְׁנַ֖יִם אַ֣יִל אֶחָ֑ד שִׁבְעָ֥ה כְבָשִׂ֖ים בְּנֵ֥י שָׁנָֽה׃ וְתַקְרִיבוּ עוֹלָה לְנַחַת רוּחַ לַה': שְׁנֵי פָּרִים בְּנֵי בָּקָר אַיִל אֶחָד וְשִׁבְעָה כְּבָשִׂים שֶׁגִּילָם עַד שָׁנָה.
כח וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֤ה עֶשְׂרֹנִים֙ לַפָּ֣ר הָֽאֶחָ֔ד שְׁנֵי֙ עֶשְׂרֹנִ֔ים לָאַ֖יִל הָאֶחָֽד׃ וּמִנְחוֹתֵיהֶם: סוֹלֶת מְעֻרְבֶּבֶת בְּשֶׁמֶן, שָׁלוֹשׁ עֲשִׂירִיּוֹת הָאֵיפָה (כְּ6.5 לִיטֶר) לְכָל פָּר, וּשְׁתֵּי עֲשִׂירִיּוֹת הָאֵיפָה (כְּ⅓4 לִיטֶר) לְכָל אַיִל.
כט עִשָּׂרוֹן֙ עִשָּׂר֔וֹן לַכֶּ֖בֶשׂ הָאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים׃ עֲשִׂירִית אֵיפָה (כִּ2.16 לִיטֶר) לְכָל כֶּבֶשׂ לְשִׁבְעַת הַכְּבָשִׂים.
ל שְׂעִ֥יר עִזִּ֖ים אֶחָ֑ד לְכַפֵּ֖ר עֲלֵיכֶֽם׃ תַּיִשׁ אֶחָד לְכַפֵּר עֲלֵיכֶם.
לא מִלְּבַ֞ד עֹלַ֧ת הַתָּמִ֛יד וּמִנְחָת֖וֹ תַּעֲשׂ֑וּ תְּמִימִ֥ם יִהְיוּ־לָכֶ֖ם וְנִסְכֵּיהֶֽם׃ כָּךְ תַּעֲשׂוּ בְּנוֹסָף לָעוֹלָה שֶׁל קָרְבַּן הַתָּמִיד וְהַיַּיִן שֶׁאִתָּהּ, הֵם יִהְיוּ לְלֹא מוּם כְּדֵי שֶׁיִּהְיוּ רְצוּיִים בְּעֵינֵי ה' עֲבוּרְכֶם, וְעִם הַיַּיִן שֶׁאִתָּם.

כט א וּבַחֹ֨דֶשׁ הַשְּׁבִיעִ֜י בְּאֶחָ֣ד לַחֹ֗דֶשׁ מִֽקְרָא־קֹ֙דֶשׁ֙ יִהְיֶ֣ה לָכֶ֔ם כׇּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֣א תַעֲשׂ֑וּ י֥וֹם תְּרוּעָ֖ה יִהְיֶ֥ה לָכֶֽם׃ וּבַחוֹדֶשׁ הַשְּׁבִיעִי בָּרִאשׁוֹן לַחוֹדֶשׁ יִהְיֶה לָכֶם מְאוֹרָע קָדוֹשׁ – אַל תַּעֲשׂוּ שׁוּם מְלָאכָה שֶׁל עֲבוֹדָה, יִהְיֶה לָכֶם יוֹם שֶׁל תְּקִיעַת תְּרוּעוֹת.
ב וַעֲשִׂיתֶ֨ם עֹלָ֜ה לְרֵ֤יחַ נִיחֹ֙חַ֙ לַֽיהֹוָ֔ה פַּ֧ר בֶּן־בָּקָ֛ר אֶחָ֖ד אַ֣יִל אֶחָ֑ד כְּבָשִׂ֧ים בְּנֵי־שָׁנָ֛ה שִׁבְעָ֖ה תְּמִימִֽם׃ וְתַעֲשׂוּ עוֹלָה לְנַחַת רוּחַ לַה' – פַּר אֶחָד אַיִל אֶחָד וְשִׁבְעָה כְּבָשִׂים שֶׁגִּילָם עַד שָׁנָה לְלֹא מוּם.
ג וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֤ה עֶשְׂרֹנִים֙ לַפָּ֔ר שְׁנֵ֥י עֶשְׂרֹנִ֖ים לָאָֽיִל׃ וּמִנְחוֹתֵיהֶם: סוֹלֶת מְעֻרְבֶּבֶת בְּשֶׁמֶן, שָׁלוֹשׁ עֲשִׂירִיּוֹת הָאֵיפָה (כְּ6.5 לִיטֶר) לְכָל פָּר שְׁתֵּי עֲשִׂירִיּוֹת הָאֵיפָה (כְּ⅓4 לִיטֶר) לְכָל אַיִל.
ד וְעִשָּׂר֣וֹן אֶחָ֔ד לַכֶּ֖בֶשׂ הָאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים׃ וַעֲשִׂירִית אֵיפָה (כִּ2.16 לִיטֶר) לְכָל כֶּבֶשׂ מִשִּׁבְעַת הַכְּבָשִׂים.
ה וּשְׂעִיר־עִזִּ֥ים אֶחָ֖ד חַטָּ֑את לְכַפֵּ֖ר עֲלֵיכֶֽם׃ וְתַיִשׁ אֶחָד חַטָּאת לְכַפֵּר עֲלֵיכֶם.
ו מִלְּבַד֩ עֹלַ֨ת הַחֹ֜דֶשׁ וּמִנְחָתָ֗הּ וְעֹלַ֤ת הַתָּמִיד֙ וּמִנְחָתָ֔הּ וְנִסְכֵּיהֶ֖ם כְּמִשְׁפָּטָ֑ם לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַיהֹוָֽה׃ נוֹסָף עַל עוֹלַת רֹאשׁ חוֹדֶשׁ וּמִנְחָתָהּ וְהָעוֹלָה שֶׁל קָרְבַּן הַתָּמִיד וּמִנְחָתָהּ וּשְׁפִיכַת הַיַּיִן שֶׁאִתָּם כְּדִינֵיהֶם, קָרְבָּן עוֹלֶה בָּאֵשׁ לְנַחַת רוּחַ לַה'.

ז וּבֶעָשׂוֹר֩ לַחֹ֨דֶשׁ הַשְּׁבִיעִ֜י הַזֶּ֗ה מִֽקְרָא־קֹ֙דֶשׁ֙ יִהְיֶ֣ה לָכֶ֔ם וְעִנִּיתֶ֖ם אֶת־נַפְשֹׁתֵיכֶ֑ם כׇּל־מְלָאכָ֖ה לֹ֥א תַעֲשֽׂוּ׃ וּבָעֲשִׂירִי לַחוֹדֶשׁ הַשְּׁבִיעִי הַזֶּה יִהְיֶה לָכֶם מְאוֹרָע קָדוֹשׁ, וּתְעַנּוּ אֶת נַפְשׁוֹתֵיכֶם בְּצוֹם וְאַל תַּעֲשׂוּ שׁוּם מְלָאכָה.
ח וְהִקְרַבְתֶּ֨ם עֹלָ֤ה לַֽיהֹוָה֙ רֵ֣יחַ נִיחֹ֔חַ פַּ֧ר בֶּן־בָּקָ֛ר אֶחָ֖ד אַ֣יִל אֶחָ֑ד כְּבָשִׂ֤ים בְּנֵֽי־שָׁנָה֙ שִׁבְעָ֔ה תְּמִימִ֖ם יִהְי֥וּ לָכֶֽם׃ וְתַקְרִיבוּ עוֹלָה לַה' לְנַחַת רוּחַ: פַּר אֶחָד אַיִל אֶחָד וְשִׁבְעָה כְּבָשִׂים שֶׁגִּילָם עַד שָׁנָה לְלֹא מוּם כְּדֵי שֶׁיִּהְיוּ רְצוּיִים בְּעֵינֵי ה' עֲבוּרְכֶם.
ט וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֤ה עֶשְׂרֹנִים֙ לַפָּ֔ר שְׁנֵי֙ עֶשְׂרֹנִ֔ים לָאַ֖יִל הָאֶחָֽד׃ וּמִנְחוֹתֵיהֶם סוֹלֶת מְעֻרְבֶּבֶת בְּשֶׁמֶן שָׁלוֹשׁ עֲשִׂירִיּוֹת הָאֵיפָה (כְּ6.5 לִיטֶר) לְכָל פָּר, וּשְׁתֵּי עֲשִׂירִיּוֹת הָאֵיפָה (כְּ⅓4 לִיטֶר) לְכָל אַיִל.
י עִשָּׂרוֹן֙ עִשָּׂר֔וֹן לַכֶּ֖בֶשׂ הָאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים׃ עֲשִׂירִית אֵיפָה (כִּ2.16 לִיטֶר) לְכָל כֶּבֶשׂ מִשִּׁבְעַת הַכְּבָשִׂים.
יא שְׂעִיר־עִזִּ֥ים אֶחָ֖ד חַטָּ֑את מִלְּבַ֞ד חַטַּ֤את הַכִּפֻּרִים֙ וְעֹלַ֣ת הַתָּמִ֔יד וּמִנְחָתָ֖הּ וְנִסְכֵּיהֶֽם׃ תַּיִשׁ אֶחָד חַטָּאת, נוֹסָף עַל חַטַּאת הַכִּפּוּרִים וְהָעוֹלָה שֶׁל קָרְבַּן הַתָּמִיד וּמִנְחָתָהּ וּשְׁפִיכַת הַיַּיִן שֶׁאִתָּם.

יב וּבַחֲמִשָּׁה֩ עָשָׂ֨ר י֜וֹם לַחֹ֣דֶשׁ הַשְּׁבִיעִ֗י מִֽקְרָא־קֹ֙דֶשׁ֙ יִהְיֶ֣ה לָכֶ֔ם כׇּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֣א תַעֲשׂ֑וּ וְחַגֹּתֶ֥ם חַ֛ג לַיהֹוָ֖ה שִׁבְעַ֥ת יָמִֽים׃ וּבַיּוֹם הַחֲמִשָּׁה עָשָׂר לַחוֹדֶשׁ הַשְּׁבִיעִי יִהְיֶה לָכֶם מְאוֹרָע קָדוֹשׁ – אַל תַּעֲשׂוּ שׁוּם מְלָאכָה שֶׁל עֲבוֹדָה, וְתַחְגְּגוּ חַג לַה' שִׁבְעָה יָמִים.
יג וְהִקְרַבְתֶּ֨ם עֹלָ֜ה אִשֵּׁ֨ה רֵ֤יחַ נִיחֹ֙חַ֙ לַֽיהֹוָ֔ה פָּרִ֧ים בְּנֵי־בָקָ֛ר שְׁלֹשָׁ֥ה עָשָׂ֖ר אֵילִ֣ם שְׁנָ֑יִם כְּבָשִׂ֧ים בְּנֵֽי־שָׁנָ֛ה אַרְבָּעָ֥ה עָשָׂ֖ר תְּמִימִ֥ם יִהְיֽוּ׃ וְתַקְרִיבוּ קָרְבָּן עוֹלָה לָאֵשׁ לְנַחַת רוּחַ לַה': שְׁלוֹשָׁה עָשָׂר פָּרִים, שְׁנֵי אֵילִים, וְאַרְבָּעָה עָשָׂר כְּבָשִׂים שֶׁגִּילָם עַד שָׁנָה, וְיִהְיוּ לְלֹא מוּם.
יד וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֨ה עֶשְׂרֹנִ֜ים לַפָּ֣ר הָֽאֶחָ֗ד לִשְׁלֹשָׁ֤ה עָשָׂר֙ פָּרִ֔ים שְׁנֵ֤י עֶשְׂרֹנִים֙ לָאַ֣יִל הָֽאֶחָ֔ד לִשְׁנֵ֖י הָאֵילִֽם׃ וּמִנְחוֹתֵיהֶם סוֹלֶת מְעֻרְבֶּבֶת בְּשֶׁמֶן: שָׁלוֹשׁ עֲשִׂירִיּוֹת הָאֵיפָה (כְּ6.5 לִיטֶר) לְכָל פָּר מִשְּׁלוֹשָׁה עָשָׂר הַפָּרִים, שְׁתֵּי עֲשִׂירִיּוֹת הָאֵיפָה (כְּ⅓4 לִיטֶר) לְכָל אַיִל מִשְּׁנֵי הָאֵילִים.
טו וְעִשָּׂרוֹׄן֙ עִשָּׂר֔וֹן לַכֶּ֖בֶשׂ הָאֶחָ֑ד לְאַרְבָּעָ֥ה עָשָׂ֖ר כְּבָשִֽׂים׃ עֲשִׂירִית אֵיפָה (כִּ2.16 לִיטֶר) לְכָל כֶּבֶשׂ מֵאַרְבָּעָה עָשָׂר הַכְּבָשִׂים.
טז וּשְׂעִיר־עִזִּ֥ים אֶחָ֖ד חַטָּ֑את מִלְּבַד֙ עֹלַ֣ת הַתָּמִ֔יד מִנְחָתָ֖הּ וְנִסְכָּֽהּ׃ וְתַיִשׁ אֶחָד חַטָּאת בְּנוֹסָף לְעוֹלַת הַתָּמִיד מִנְחָתָהּ וּשְׁפִיכַת הַיַּיִן שֶׁאִתָּהּ.

לה בַּיּוֹם֙ הַשְּׁמִינִ֔י עֲצֶ֖רֶת תִּהְיֶ֣ה לָכֶ֑ם כׇּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַעֲשֽׂוּ׃ בַּיּוֹם הַשְּׁמִינִי תִּהְיֶה לָכֶם הִתְכַּנְּסוּת חֲגִיגִית, אַל תַּעֲשׂוּ שׁוּם מְלָאכָה שֶׁל עֲבוֹדָה.
לו וְהִקְרַבְתֶּ֨ם עֹלָ֜ה אִשֵּׁ֨ה רֵ֤יחַ נִיחֹ֙חַ֙ לַֽיהֹוָ֔ה פַּ֥ר אֶחָ֖ד אַ֣יִל אֶחָ֑ד כְּבָשִׂ֧ים בְּנֵי־שָׁנָ֛ה שִׁבְעָ֖ה תְּמִימִֽם׃ וְתַקְרִיבוּ קָרְבָּן עוֹלָה לָאֵשׁ לְנַחַת רוּחַ לַה', פַּר אֶחָד, אַיִל אֶחָד, וְשִׁבְעָה כְּבָשִׂים שֶׁגִּילָם עַד שָׁנָה לְלֹא מוּם.
לז מִנְחָתָ֣ם וְנִסְכֵּיהֶ֗ם לַפָּ֨ר לָאַ֧יִל וְלַכְּבָשִׂ֛ים בְּמִסְפָּרָ֖ם כַּמִּשְׁפָּֽט׃ מִנְחוֹתֵיהֶם וּשְׁפִיכַת הַיַּיִן שֶׁאִתָּם לַפָּר לָאַיִל וְלַכְּבָשִׂים יִהְיוּ בְּמִסְפָּרָם כַּדִּין.
לח וּשְׂעִ֥יר חַטָּ֖את אֶחָ֑ד מִלְּבַד֙ עֹלַ֣ת הַתָּמִ֔יד וּמִנְחָתָ֖הּ וְנִסְכָּֽהּ׃ וְתַיִשׁ אֶחָד לְחַטָּאת נוֹסָף עַל עוֹלַת הַתָּמִיד וּמִנְחָתָהּ וּשְׁפִיכַת הַיַּיִן שֶׁאִתָּהּ.
לט אֵ֛לֶּה תַּעֲשׂ֥וּ לַיהֹוָ֖ה בְּמוֹעֲדֵיכֶ֑ם לְבַ֨ד מִנִּדְרֵיכֶ֜ם וְנִדְבֹתֵיכֶ֗ם לְעֹלֹֽתֵיכֶם֙ וּלְמִנְחֹ֣תֵיכֶ֔ם וּלְנִסְכֵּיכֶ֖ם וּלְשַׁלְמֵיכֶֽם׃ קָרְבָּנוֹת אֵלֶּה תַּעֲשׂוּ לַה' בְּמוֹעֲדֵיכֶם, בְּנוֹסָף לְעוֹלוֹתֵיכֶם, מִנְחוֹתֵיכֶם, שְׁפִיכַת הַיַּיִן שֶׁלָּכֶם, וְקָרְבְּנוֹת הַשְּׁלָמִים שֶׁלָּכֶם, שֶׁתַּקְרִיבוּ לְנִדְרֵיכֶם וְנִדְבוֹתֵיכֶם.
ל א וַיֹּ֥אמֶר מֹשֶׁ֖ה אֶל־בְּנֵ֣י יִשְׂרָאֵ֑ל כְּכֹ֛ל אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ מֹשֶׁה אָמַר לִבְנֵי יִשְׂרָאֵל אֶת כָּל מַה שֶּׁה' צִוָּה אוֹתוֹ.
