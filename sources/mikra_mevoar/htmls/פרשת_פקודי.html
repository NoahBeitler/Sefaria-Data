ראשון[עריכה]

ס פָּרָשַׁת פְּקוּדֵי הַמִּשְׁכָּן
לח כא אֵ֣לֶּה פְקוּדֵ֤י הַמִּשְׁכָּן֙ מִשְׁכַּ֣ן הָעֵדֻ֔ת אֲשֶׁ֥ר פֻּקַּ֖ד עַל־פִּ֣י מֹשֶׁ֑ה עֲבֹדַת֙ הַלְוִיִּ֔ם בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן־אַהֲרֹ֖ן הַכֹּהֵֽן׃ אֵלֶּה חֶשְׁבּוֹנוֹת הַמִּשְׁכָּן – מִשְׁכַּן לוּחוֹת הַבְּרִית, שֶׁהִצְטַוּוּ עָלָיו בְּאֶמְצָעוּת מֹשֶׁה: עֲבוֹדַת הַלְוִיִּים בְּהַנְהָגַת אִיתָמָר בֶּן אַהֲרֹן הַכֹּהֵן.
כב וּבְצַלְאֵ֛ל בֶּן־אוּרִ֥י בֶן־ח֖וּר לְמַטֵּ֣ה יְהוּדָ֑ה עָשָׂ֕ה אֵ֛ת כׇּל־אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ וּבְצַלְאֵל בֶּן אוּרִי בֶּן חוּר מִשֵּׁבֶט יְהוּדָה עָשָׂה אֶת כָּל מַה שֶּׁה' צִוָּה אֶת מֹשֶׁה.
כג וְאִתּ֗וֹ אׇהֳלִיאָ֞ב בֶּן־אֲחִיסָמָ֛ךְ לְמַטֵּה־דָ֖ן חָרָ֣שׁ וְחֹשֵׁ֑ב וְרֹקֵ֗ם בַּתְּכֵ֙לֶת֙ וּבָֽאַרְגָּמָ֔ן וּבְתוֹלַ֥עַת הַשָּׁנִ֖י וּבַשֵּֽׁשׁ׃ וְאִתּוֹ הָיָה אָהֳלִיאָב בֶּן אֲחִיסָמָךְ מִשֵּׁבֶט דָּן, שֶׁהָיָה אָמָּן וּמְתַכְנֵן וְרוֹקֵם בִּתְכֵלֶת וּבְאַרְגָּמָן וּבַצֶּבַע הָאָדוֹם מִתּוֹלַעַת וּבְפִשְׁתָּן.
ס
כד כׇּל־הַזָּהָ֗ב הֶֽעָשׂוּי֙ לַמְּלָאכָ֔ה בְּכֹ֖ל מְלֶ֣אכֶת הַקֹּ֑דֶשׁ וַיְהִ֣י ׀ זְהַ֣ב הַתְּנוּפָ֗ה תֵּ֤שַׁע וְעֶשְׂרִים֙ כִּכָּ֔ר וּשְׁבַ֨ע מֵא֧וֹת וּשְׁלֹשִׁ֛ים שֶׁ֖קֶל בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ׃ חֶשְׁבּוֹן כָּל הַזָּהָב הֶעָשׂוּי לַמְּלָאכָה בְּכָל מְלֶאכֶת הַקּוֹדֶשׁ: זְהַב הַתְּרוּמָה הָיָה מִשְׁקַל עֶשְׂרִים וְתֵשַׁע כִּכָּרוֹת וּשְׁבַע מֵאוֹת וּשְׁלוֹשִׁים שֶׁקֶל, בְּשִׁקְלֵי הַקּוֹדֶשׁ (כְּ1,237 קִילוֹ).
כה וְכֶ֛סֶף פְּקוּדֵ֥י הָעֵדָ֖ה מְאַ֣ת כִּכָּ֑ר וְאֶ֩לֶף֩ וּשְׁבַ֨ע מֵא֜וֹת וַחֲמִשָּׁ֧ה וְשִׁבְעִ֛ים שֶׁ֖קֶל בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ׃ וְהַכֶּסֶף שֶׁנֶּאֱסַף מִסְּפִירַת הַצִּבּוּר – מֵאָה כִּכָּרוֹת, וְאֶלֶף וּשְׁבַע מֵאוֹת וְשִׁבְעִים וַחֲמִשָּׁה שִׁקְלֵי קוֹדֶשׁ (כְּ4,255 קִילוֹ).
כו בֶּ֚קַע לַגֻּלְגֹּ֔לֶת מַחֲצִ֥ית הַשֶּׁ֖קֶל בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ לְכֹ֨ל הָעֹבֵ֜ר עַל־הַפְּקֻדִ֗ים מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה לְשֵׁשׁ־מֵא֥וֹת אֶ֙לֶף֙ וּשְׁלֹ֣שֶׁת אֲלָפִ֔ים וַחֲמֵ֥שׁ מֵא֖וֹת וַחֲמִשִּֽׁים׃ תְּרוּמַת חֲצִי שֶׁקֶל לַגֻּלְגּוֹלֶת – שֶׁהוּא מַחֲצִית שֶׁקֶל מִשִּׁקְלֵי הַקּוֹדֶשׁ (כְּ7 גְּרַם), תְּרוּמַת כָּל מִי שֶׁנִּסְפַּר מִגִּיל עֶשְׂרִים שָׁנָה וָמַעְלָה – שֵׁשׁ מֵאוֹת אֶלֶף וּשְׁלוֹשֶׁת אֲלָפִים וַחֲמֵשׁ מֵאוֹת וַחֲמִשִּׁים אִישׁ.
כז וַיְהִ֗י מְאַת֙ כִּכַּ֣ר הַכֶּ֔סֶף לָצֶ֗קֶת אֵ֚ת אַדְנֵ֣י הַקֹּ֔דֶשׁ וְאֵ֖ת אַדְנֵ֣י הַפָּרֹ֑כֶת מְאַ֧ת אֲדָנִ֛ים לִמְאַ֥ת הַכִּכָּ֖ר כִּכָּ֥ר לָאָֽדֶן׃ מֵאָה כִּכְּרוֹת הַכֶּסֶף הָיוּ כְּדֵי לָצֶקֶת אֶת בְּסִיסֵי מִשְׁכַּן הַקּוֹדֶשׁ וְאֶת בְּסִיסֵי הַפָּרוֹכֶת – מֵאָה כִּכָּרוֹת עֲבוּר מֵאָה בְּסִיסִים – כִּכָּר לְכָל בָּסִיס.
כח וְאֶת־הָאֶ֜לֶף וּשְׁבַ֤ע הַמֵּאוֹת֙ וַחֲמִשָּׁ֣ה וְשִׁבְעִ֔ים עָשָׂ֥ה וָוִ֖ים לָעַמּוּדִ֑ים וְצִפָּ֥ה רָאשֵׁיהֶ֖ם וְחִשַּׁ֥ק אֹתָֽם׃ וּמִמִּשְׁקַל אֶלֶף וּשְׁבַע מֵאוֹת וְשִׁבְעִים וַחֲמִשָּׁה הַשְּׁקָלִים הַנּוֹתָרִים הוּא עָשָׂה וָוִים לָעַמּוּדִים וְצִפּוּי לְרָאשֵׁי הָעַמּוּדִים וְחִשּׁוּקִים עֲבוּרָם.
כט וּנְחֹ֥שֶׁת הַתְּנוּפָ֖ה שִׁבְעִ֣ים כִּכָּ֑ר וְאַלְפַּ֥יִם וְאַרְבַּע־מֵא֖וֹת שָֽׁקֶל׃ וּתְרוּמַת הַנְּחוֹשֶׁת – שִׁבְעִים כִּכָּרוֹת וְאַלְפַּיִם וְאַרְבַּע מֵאוֹת שְׁקָלִים (כְּ2,995 קִילוֹ).
ל וַיַּ֣עַשׂ בָּ֗הּ אֶת־אַדְנֵי֙ פֶּ֚תַח אֹ֣הֶל מוֹעֵ֔ד וְאֵת֙ מִזְבַּ֣ח הַנְּחֹ֔שֶׁת וְאֶת־מִכְבַּ֥ר הַנְּחֹ֖שֶׁת אֲשֶׁר־ל֑וֹ וְאֵ֖ת כׇּל־כְּלֵ֥י הַמִּזְבֵּֽחַ׃ הוּא עָשָׂה בָּהּ אֶת בְּסִיסֵי פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת וְאֶת מִזְבַּח הַנְּחוֹשֶׁת וְאֶת קִשּׁוּט רֶשֶׁת הַנְּחוֹשֶׁת שֶׁלּוֹ וְאֶת כָּל כְּלֵי הַמִּזְבֵּחַ.
לא וְאֶת־אַדְנֵ֤י הֶֽחָצֵר֙ סָבִ֔יב וְאֶת־אַדְנֵ֖י שַׁ֣עַר הֶחָצֵ֑ר וְאֵ֨ת כׇּל־יִתְדֹ֧ת הַמִּשְׁכָּ֛ן וְאֶת־כׇּל־יִתְדֹ֥ת הֶחָצֵ֖ר סָבִֽיב׃ וְאֶת בְּסִיסֵי הֶחָצֵר מִסָּבִיב וְאֶת בְּסִיסֵי שַׁעַר הֶחָצֵר וְאֶת כָּל יִתְדוֹת הַמִּשְׁכָּן וְאֶת כָּל יִתְדוֹת הֶחָצֵר מִסָּבִיב.
לט א וּמִן־הַתְּכֵ֤לֶת וְהָֽאַרְגָּמָן֙ וְתוֹלַ֣עַת הַשָּׁנִ֔י עָשׂ֥וּ בִגְדֵי־שְׂרָ֖ד לְשָׁרֵ֣ת בַּקֹּ֑דֶשׁ וַֽיַּעֲשׂ֞וּ אֶת־בִּגְדֵ֤י הַקֹּ֙דֶשׁ֙ אֲשֶׁ֣ר לְאַהֲרֹ֔ן כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ וּמֵהַתְּכֵלֶת וְהָאַרְגָּמָן וְהַצֶּבַע הָאָדוֹם מֵהַתּוֹלַעַת עָשׂוּ כִּסּוּיִים לִכְלֵי הַשָּׁרֵת בַּקּוֹדֶשׁ, הֵם עָשׂוּ אֶת בִּגְדֵי הַקּוֹדֶשׁ שֶׁל אַהֲרֹן כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

שני (חמישי במחוברין)[עריכה]

פ פָּרָשַׁת הָאֵפוֹד
ב וַיַּ֖עַשׂ אֶת־הָאֵפֹ֑ד זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ מׇשְׁזָֽר׃ הוּא עָשָׂה אֶת הָאֵפוֹד מִזָּהָב תְּכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת וּפִשְׁתָּן שָׁזוּר.
ג וַֽיְרַקְּע֞וּ אֶת־פַּחֵ֣י הַזָּהָב֮ וְקִצֵּ֣ץ פְּתִילִם֒ לַעֲשׂ֗וֹת בְּת֤וֹךְ הַתְּכֵ֙לֶת֙ וּבְת֣וֹךְ הָֽאַרְגָּמָ֔ן וּבְת֛וֹךְ תּוֹלַ֥עַת הַשָּׁנִ֖י וּבְת֣וֹךְ הַשֵּׁ֑שׁ מַעֲשֵׂ֖ה חֹשֵֽׁב׃ הֵם רִדְּדוּ אֶת הַזָּהָב לְפַחִים דַּקִּים, וְהוּא קִצֵּץ מֵהֶם פְּתִילִים לְהַשְׁחִיל בְּחוּטֵי הַתְּכֵלֶת וּבְחוּטֵי הָאַרְגָּמָן וּבְחוּטֵי הַצֶּבַע הָאָדוֹם מֵהַתּוֹלַעַת וּבְחוּטֵי הַפִּשְׁתָּן, בִּמְלֶאכֶת אֳמָנוּת.
ד כְּתֵפֹ֥ת עָֽשׂוּ־ל֖וֹ חֹבְרֹ֑ת עַל־שְׁנֵ֥י קצוותו קְצוֹתָ֖יו חֻבָּֽר׃ עָשׂוּ לָאֵפוֹד כְּתֵפִיּוֹת מִתְחַבְּרוֹת, וְחִבְּרוּ אוֹתָן לִשְׁנֵי קְצוֹתָיו.
ה וְחֵ֨שֶׁב אֲפֻדָּת֜וֹ אֲשֶׁ֣ר עָלָ֗יו מִמֶּ֣נּוּ הוּא֮ כְּמַעֲשֵׂ֒הוּ֒ זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מׇשְׁזָ֑ר כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ וַחֲגוֹרַת אֲפֻדָּתוֹ שֶׁעָלָיו הִיא חֵלֶק מִמֶּנּוּ, וַעֲשׂוּיָה כָּמוֹהוּ – מִזָּהָב תְּכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת וּפִשְׁתָּן שָׁזוּר, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

ו וַֽיַּעֲשׂוּ֙ אֶת־אַבְנֵ֣י הַשֹּׁ֔הַם מֻֽסַבֹּ֖ת מִשְׁבְּצֹ֣ת זָהָ֑ב מְפֻתָּחֹת֙ פִּתּוּחֵ֣י חוֹתָ֔ם עַל־שְׁמ֖וֹת בְּנֵ֥י יִשְׂרָאֵֽל׃ הֵם עָשׂוּ אֶת אַבְנֵי הַשּׁוֹהַם מֻקָּפוֹת בְּמִשְׁבְּצוֹת זָהָב, מְגֻלָּפוֹת בְּאוֹתִיּוֹת חֲקוּקוֹת כְּמוֹ בְּחוֹתֶמֶת, בִּשְׁמוֹת בְּנֵי יִשְׂרָאֵל.
ז וַיָּ֣שֶׂם אֹתָ֗ם עַ֚ל כִּתְפֹ֣ת הָאֵפֹ֔ד אַבְנֵ֥י זִכָּר֖וֹן לִבְנֵ֣י יִשְׂרָאֵ֑ל כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ הוּא שָׂם אוֹתָן עַל כְּתֵפִיּוֹת הָאֵפוֹד – אַבְנֵי זִכָּרוֹן לִבְנֵי יִשְׂרָאֵל, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.
פ פָּרָשַׁת הַחֹשֶׁן
ח וַיַּ֧עַשׂ אֶת־הַחֹ֛שֶׁן מַעֲשֵׂ֥ה חֹשֵׁ֖ב כְּמַעֲשֵׂ֣ה אֵפֹ֑ד זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ מׇשְׁזָֽר׃ הוּא עָשָׂה אֶת הַחוֹשֶׁן בִּמְלֶאכֶת אֳמָנוּת כְּמוֹ בִּמְלֶאכֶת הָאֵפוֹד – מִזָּהָב, תְּכֵלֶת, אַרְגָּמָן, צֶבַע אָדוֹם מִתּוֹלַעַת, וּפִשְׁתָּן שָׁזוּר.
ט רָב֧וּעַ הָיָ֛ה כָּפ֖וּל עָשׂ֣וּ אֶת־הַחֹ֑שֶׁן זֶ֧רֶת אׇרְכּ֛וֹ וְזֶ֥רֶת רׇחְבּ֖וֹ כָּפֽוּל׃ עָשׂוּ אֶת הַחוֹשֶׁן רִבּוּעַ מְקֻפָּל לִשְׁנַיִם, חֲצִי אַמָּה (קָרוֹב לְ23.5 ס"מ) אָרְכּוֹ וַחֲצִי אַמָּה רָחְבּוֹ כְּשֶׁהוּא מְקֻפָּל.
י וַיְמַ֨לְאוּ־ב֔וֹ אַרְבָּעָ֖ה ט֣וּרֵי אָ֑בֶן ט֗וּר אֹ֤דֶם פִּטְדָה֙ וּבָרֶ֔קֶת הַטּ֖וּר הָאֶחָֽד׃ וּמִלְּאוּ בּוֹ אַרְבָּעָה טוּרֵי אֶבֶן – הַטּוּר הָרִאשׁוֹן: אוֹדֶם פִּטְדָה וּבָרֶקֶת.
יא וְהַטּ֖וּר הַשֵּׁנִ֑י נֹ֥פֶךְ סַפִּ֖יר וְיָהֲלֹֽם׃ וְהַטּוּר הַשֵּׁנִי: נוֹפֶךְ סַפִּיר וְיָהֲלוֹם.
יב וְהַטּ֖וּר הַשְּׁלִישִׁ֑י לֶ֥שֶׁם שְׁב֖וֹ וְאַחְלָֽמָה׃ וְהַטּוּר הַשְּׁלִישִׁי: לֶשֶׁם שְׁבוֹ וְאַחְלָמָה.
יג וְהַטּוּר֙ הָֽרְבִיעִ֔י תַּרְשִׁ֥ישׁ שֹׁ֖הַם וְיָשְׁפֵ֑ה מֽוּסַבֹּ֛ת מִשְׁבְּצֹ֥ת זָהָ֖ב בְּמִלֻּאֹתָֽם׃ וְהַטּוּר הָרְבִיעִי: תַּרְשִׁישׁ שׁוֹהַם וְיָשְׁפֵה, מֻקָּפוֹת בְּמִשְׁבְּצוֹת זָהָב שֶׁמְּמֻלָּאוֹת בְּאוֹתָן הָאֲבָנִים.
יד וְ֠הָאֲבָנִ֠ים עַל־שְׁמֹ֨ת בְּנֵי־יִשְׂרָאֵ֥ל הֵ֛נָּה שְׁתֵּ֥ים עֶשְׂרֵ֖ה עַל־שְׁמֹתָ֑ם פִּתּוּחֵ֤י חֹתָם֙ אִ֣ישׁ עַל־שְׁמ֔וֹ לִשְׁנֵ֥ים עָשָׂ֖ר שָֽׁבֶט׃ וְהָאֲבָנִים הֵן עַל שְׁמוֹת בְּנֵי יִשְׂרָאֵל – שְׁתֵּים עֶשְׂרֵה אֲבָנִים בִּשְׁמוֹתָם, אוֹתִיּוֹת חֲקוּקוֹת כְּמוֹ בְּחוֹתֶמֶת, לְכָל שֵׁבֶט יֵשׁ אֶבֶן עַל שְׁמוֹ – לִשְׁנֵים עָשָׂר הַשְּׁבָטִים.
טו וַיַּעֲשׂ֧וּ עַל־הַחֹ֛שֶׁן שַׁרְשְׁרֹ֥ת גַּבְלֻ֖ת מַעֲשֵׂ֣ה עֲבֹ֑ת זָהָ֖ב טָהֽוֹר׃ עָשׂוּ עַל הַחוֹשֶׁן בְּקָצֵהוּ שַׁרְשְׁרוֹת שְׁזוּרוֹת כְּחֶבֶל מִזָּהָב טָהוֹר.
טז וַֽיַּעֲשׂ֗וּ שְׁתֵּי֙ מִשְׁבְּצֹ֣ת זָהָ֔ב וּשְׁתֵּ֖י טַבְּעֹ֣ת זָהָ֑ב וַֽיִּתְּנ֗וּ אֶת־שְׁתֵּי֙ הַטַּבָּעֹ֔ת עַל־שְׁנֵ֖י קְצ֥וֹת הַחֹֽשֶׁן׃ עָשׂוּ שְׁתֵּי מִשְׁבְּצוֹת זָהָב וּשְׁתֵּי טַבָּעוֹת זָהָב, וְשָׂמוּ אֶת שְׁתֵּי הַטַּבָּעוֹת עַל שְׁנֵי קְצוֹת הַחוֹשֶׁן.
יז וַֽיִּתְּנ֗וּ שְׁתֵּי֙ הָעֲבֹתֹ֣ת הַזָּהָ֔ב עַל־שְׁתֵּ֖י הַטַּבָּעֹ֑ת עַל־קְצ֖וֹת הַחֹֽשֶׁן׃ שָׂמוּ אֶת שְׁנֵי הַשְּׂרוֹכִים הָעֲשׂוּיִים מִזָּהָב עַל שְׁתֵּי הַטַּבָּעוֹת עַל קְצוֹת הַחוֹשֶׁן.
יח וְאֵ֨ת שְׁתֵּ֤י קְצוֹת֙ שְׁתֵּ֣י הָֽעֲבֹתֹ֔ת נָתְנ֖וּ עַל־שְׁתֵּ֣י הַֽמִּשְׁבְּצֹ֑ת וַֽיִּתְּנֻ֛ם עַל־כִּתְפֹ֥ת הָאֵפֹ֖ד אֶל־מ֥וּל פָּנָֽיו׃ וְאֶת שְׁנֵי קְצוֹת שְׁנֵי הַשְּׂרוֹכִים שָׂמוּ עַל שְׁתֵּי הַמִּשְׁבְּצוֹת, וְשָׂמוּ אוֹתָם עַל כְּתֵפִיּוֹת הָאֵפוֹד כְּלַפֵּי חוּץ.
יט וַֽיַּעֲשׂ֗וּ שְׁתֵּי֙ טַבְּעֹ֣ת זָהָ֔ב וַיָּשִׂ֕ימוּ עַל־שְׁנֵ֖י קְצ֣וֹת הַחֹ֑שֶׁן עַל־שְׂפָת֕וֹ אֲשֶׁ֛ר אֶל־עֵ֥בֶר הָאֵפֹ֖ד בָּֽיְתָה׃ עָשׂוּ שְׁתֵּי טַבְּעוֹת זָהָב וְשָׂמוּ עַל שְׁנֵי קְצוֹת הַחוֹשֶׁן, עַל שְׂפָתוֹ הַפְּנִימִית שֶׁקְּרוֹבָה לָאֵפוֹד.
כ וַֽיַּעֲשׂוּ֮ שְׁתֵּ֣י טַבְּעֹ֣ת זָהָב֒ וַֽיִּתְּנֻ֡ם עַל־שְׁתֵּי֩ כִתְפֹ֨ת הָאֵפֹ֤ד מִלְּמַ֙טָּה֙ מִמּ֣וּל פָּנָ֔יו לְעֻמַּ֖ת מַחְבַּרְתּ֑וֹ מִמַּ֕עַל לְחֵ֖שֶׁב הָאֵפֹֽד׃ עָשׂוּ שְׁתֵּי טַבְּעוֹת זָהָב וְשָׂמוּ אוֹתָן עַל שְׁתֵּי כְּתֵפִיּוֹת הָאֵפוֹד מִלְּמַטָּה כְּלַפֵּי חוּץ לְיַד מְקוֹם חִבּוּרָם לָאֵפוֹד, מֵעַל חֲגוֹרַת הָאֵפוֹד.
כא וַיִּרְכְּס֣וּ אֶת־הַחֹ֡שֶׁן מִטַּבְּעֹתָיו֩ אֶל־טַבְּעֹ֨ת הָאֵפֹ֜ד בִּפְתִ֣יל תְּכֵ֗לֶת לִֽהְיֹת֙ עַל־חֵ֣שֶׁב הָאֵפֹ֔ד וְלֹֽא־יִזַּ֣ח הַחֹ֔שֶׁן מֵעַ֖ל הָאֵפֹ֑ד כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ חִבְּרוּ אֶת הַחוֹשֶׁן מִטַּבְּעוֹתָיו אֶל טַבְּעוֹת הָאֵפוֹד בִּפְתִיל תְּכֵלֶת כְּדֵי שֶׁיִּהְיֶה עַל חֲגוֹרַת הָאֵפוֹד, שֶׁלֹּא יִתְנַתֵּק הַחוֹשֶׁן מֵעַל הָאֵפוֹד, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

שלישי (ששי במחוברין)[עריכה]

פ פָּרָשַׁת הַמְּעִיל
כב וַיַּ֛עַשׂ אֶת־מְעִ֥יל הָאֵפֹ֖ד מַעֲשֵׂ֣ה אֹרֵ֑ג כְּלִ֖יל תְּכֵֽלֶת׃ הוּא עָשָׂה אֶת מְעִיל הָאֵפוֹד בִּמְלֶאכֶת אֲרִיגָה, כֻּלּוֹ מִצֶּמֶר צָבוּעַ בִּתְכֵלֶת.
כג וּפִֽי־הַמְּעִ֥יל בְּתוֹכ֖וֹ כְּפִ֣י תַחְרָ֑א שָׂפָ֥ה לְפִ֛יו סָבִ֖יב לֹ֥א יִקָּרֵֽעַ׃ וּבְפֶתַח הַמְּעִיל הַבַּד מְקֻפָּל לְתוֹכוֹ כְּמוֹ צַוָּארוֹן הַשִּׁרְיוֹן, יֵשׁ שָׂפָה מִסָּבִיב לְפִתְחוֹ כְּדֵי שֶׁלֹּא יִקָּרַע.
כד וַֽיַּעֲשׂוּ֙ עַל־שׁוּלֵ֣י הַמְּעִ֔יל רִמּוֹנֵ֕י תְּכֵ֥לֶת וְאַרְגָּמָ֖ן וְתוֹלַ֣עַת שָׁנִ֑י מׇשְׁזָֽר׃ עָשׂוּ עַל שׁוּלֵי הַמְּעִיל רִמּוֹנֵי תְּכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת, שְׁזוּרִים.
כה וַיַּעֲשׂ֥וּ פַעֲמֹנֵ֖י זָהָ֣ב טָה֑וֹר וַיִּתְּנ֨וּ אֶת־הַפַּֽעֲמֹנִ֜ים בְּת֣וֹךְ הָרִמֹּנִ֗ים עַל־שׁוּלֵ֤י הַמְּעִיל֙ סָבִ֔יב בְּת֖וֹךְ הָרִמֹּנִֽים׃ עָשׂוּ פַּעֲמוֹנִים מִזָּהָב טָהוֹר, וְשָׂמוּ אֶת הַפַּעֲמוֹנִים בְּתוֹךְ הָרִמּוֹנִים עַל שׁוּלֵי הַמְּעִיל מִסָּבִיב בְּתוֹךְ הָרִמּוֹנִים.
כו פַּעֲמֹ֤ן וְרִמֹּן֙ פַּעֲמֹ֣ן וְרִמֹּ֔ן עַל־שׁוּלֵ֥י הַמְּעִ֖יל סָבִ֑יב לְשָׁרֵ֕ת כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ פַּעֲמוֹן וְרִמּוֹן פַּעֲמוֹן וְרִמּוֹן עַל שׁוּלֵי הַמְּעִיל מִסָּבִיב, כְּדֵי לְשָׁרֵת, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.
ס פָּרָשַׁת בִּגְדֵי הַכֹּהֲנִים
כז וַֽיַּעֲשׂ֛וּ אֶת־הַכׇּתְנֹ֥ת שֵׁ֖שׁ מַעֲשֵׂ֣ה אֹרֵ֑ג לְאַהֲרֹ֖ן וּלְבָנָֽיו׃ עָשׂוּ אֶת הַכֻּתּוֹנוֹת לְאַהֲרֹן וּלְבָנָיו מִפִּשְׁתָּן, בִּמְלֶאכֶת אֲרִיגָה.
כח וְאֵת֙ הַמִּצְנֶ֣פֶת שֵׁ֔שׁ וְאֶת־פַּאֲרֵ֥י הַמִּגְבָּעֹ֖ת שֵׁ֑שׁ וְאֶת־מִכְנְסֵ֥י הַבָּ֖ד שֵׁ֥שׁ מׇשְׁזָֽר׃ וְאֶת הַמִּצְנֶפֶת מִפִּשְׁתָּן וְאֶת הַמִּגְבָּעוֹת הַמְּפוֹאָרוֹת מִפִּשְׁתָּן וְאֶת מִכְנְסֵי הַבָּד מִפִּשְׁתָּן שָׁזוּר.
כט וְֽאֶת־הָאַבְנֵ֞ט שֵׁ֣שׁ מׇשְׁזָ֗ר וּתְכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י מַעֲשֵׂ֣ה רֹקֵ֑ם כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ וְאֶת הָאַבְנֵט מִפִּשְׁתָּן שָׁזוּר וּתְכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת, בִּמְלֶאכֶת רִקְמָה, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

ל וַֽיַּעֲשׂ֛וּ אֶת־צִ֥יץ נֵֽזֶר־הַקֹּ֖דֶשׁ זָהָ֣ב טָה֑וֹר וַיִּכְתְּב֣וּ עָלָ֗יו מִכְתַּב֙ פִּתּוּחֵ֣י חוֹתָ֔ם קֹ֖דֶשׁ לַיהֹוָֽה׃ עָשׂוּ אֶת הַצִּיץ – כֶּתֶר הַקּוֹדֶשׁ, מִזָּהָב טָהוֹר, וְכָתְבוּ עָלָיו כְּתָב חָקוּק כְּמוֹ בְּחוֹתֶמֶת – קוֹדֶשׁ לַה'.
לא וַיִּתְּנ֤וּ עָלָיו֙ פְּתִ֣יל תְּכֵ֔לֶת לָתֵ֥ת עַל־הַמִּצְנֶ֖פֶת מִלְמָ֑עְלָה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ חִבְּרוּ לוֹ פְּתִיל תְּכֵלֶת שֶׁיִּהְיֶה עַל הַמִּצְנֶפֶת מִלְּמַעְלָה, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

רביעי[עריכה]

פ פָּרָשַׁת הֲבָאַת הַמִּשְׁכָּן
לג וַיָּבִ֤יאוּ אֶת־הַמִּשְׁכָּן֙ אֶל־מֹשֶׁ֔ה אֶת־הָאֹ֖הֶל וְאֶת־כׇּל־כֵּלָ֑יו קְרָסָ֣יו קְרָשָׁ֔יו בְּרִיחָ֖ו וְעַמֻּדָ֥יו וַאֲדָנָֽיו׃ הֵבִיאוּ אֶת הַמִּשְׁכָּן לְמֹשֶׁה – אֶת הָאוֹהֶל וְאֶת כָּל כֵּלָיו, קְרָסָיו קְרָשָׁיו בְּרִיחָיו וְעַמּוּדָיו וּבְסִיסָיו.
לד וְאֶת־מִכְסֵ֞ה עוֹרֹ֤ת הָֽאֵילִם֙ הַמְאׇדָּמִ֔ים וְאֶת־מִכְסֵ֖ה עֹרֹ֣ת הַתְּחָשִׁ֑ים וְאֵ֖ת פָּרֹ֥כֶת הַמָּסָֽךְ׃ וְאֶת כִּסּוּי עוֹרוֹת הָאַיִל שֶׁצְּבוּעִים בְּאָדוֹם, וְאֶת כִּסּוּי עוֹרוֹת הַחַיָּה תַּחַשׁ וְאֶת הַפָּרוֹכֶת.
לה אֶת־אֲר֥וֹן הָעֵדֻ֖ת וְאֶת־בַּדָּ֑יו וְאֵ֖ת הַכַּפֹּֽרֶת׃ אֶת אֲרוֹן לוּחוֹת הַבְּרִית וְאֶת הַמּוֹטוֹת לִנְשִׂיאָתוֹ, וְאֶת הַכִּסּוּי.
לו אֶת־הַשֻּׁלְחָן֙ אֶת־כׇּל־כֵּלָ֔יו וְאֵ֖ת לֶ֥חֶם הַפָּנִֽים׃ אֶת הַשֻּׁלְחָן אֶת כָּל כֵּלָיו וְאֶת הַלֶּחֶם הַמְּכֻבָּד.
לז אֶת־הַמְּנֹרָ֨ה הַטְּהֹרָ֜ה אֶת־נֵרֹתֶ֗יהָ נֵרֹ֛ת הַמַּֽעֲרָכָ֖ה וְאֶת־כׇּל־כֵּלֶ֑יהָ וְאֵ֖ת שֶׁ֥מֶן הַמָּאֽוֹר׃ אֶת הַמְּנוֹרָה הַטְּהוֹרָה אֶת נֵרוֹתֶיהָ – הַנֵּרוֹת הָעֲרוּכִים, וְאֶת כָּל כֵּלֶיהָ וְאֶת שֶׁמֶן הַתְּאוּרָה.
לח וְאֵת֙ מִזְבַּ֣ח הַזָּהָ֔ב וְאֵת֙ שֶׁ֣מֶן הַמִּשְׁחָ֔ה וְאֵ֖ת קְטֹ֣רֶת הַסַּמִּ֑ים וְאֵ֕ת מָסַ֖ךְ פֶּ֥תַח הָאֹֽהֶל׃ וְאֶת מִזְבַּח הַזָּהָב וְאֶת הַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ וְאֶת קְטוֹרֶת הַבְּשָׂמִים, וְאֶת מָסַךְ פֶּתַח הָאוֹהֶל.
לט אֵ֣ת ׀ מִזְבַּ֣ח הַנְּחֹ֗שֶׁת וְאֶת־מִכְבַּ֤ר הַנְּחֹ֙שֶׁת֙ אֲשֶׁר־ל֔וֹ אֶת־בַּדָּ֖יו וְאֶת־כׇּל־כֵּלָ֑יו אֶת־הַכִּיֹּ֖ר וְאֶת־כַּנּֽוֹ׃ אֶת מִזְבַּח הַנְּחוֹשֶׁת וְאֶת קִשּׁוּט רֶשֶׁת הַנְּחוֹשֶׁת שֶׁלּוֹ, אֶת הַמּוֹטוֹת לִנְשִׂיאָתוֹ, וְאֶת כָּל כֵּלָיו, אֶת הַכִּיּוֹר וְאֶת בְּסִיסוֹ.
מ אֵת֩ קַלְעֵ֨י הֶחָצֵ֜ר אֶת־עַמֻּדֶ֣יהָ וְאֶת־אֲדָנֶ֗יהָ וְאֶת־הַמָּסָךְ֙ לְשַׁ֣עַר הֶֽחָצֵ֔ר אֶת־מֵיתָרָ֖יו וִיתֵדֹתֶ֑יהָ וְאֵ֗ת כׇּל־כְּלֵ֛י עֲבֹדַ֥ת הַמִּשְׁכָּ֖ן לְאֹ֥הֶל מוֹעֵֽד׃ אֶת הַגְּדֵרוֹת הַקְּלוּעוֹת שֶׁל הֶחָצֵר, אֶת עַמּוּדֵי הֶחָצֵר וְאֶת בְּסִיסֵי הֶחָצֵר, וְאֶת הַמָּסָךְ לְשַׁעַר הֶחָצֵר, אֶת מֵיתָרָיו וִיתֵדוֹתֵיהָ וְאֶת כָּל כְּלֵי עֲבוֹדַת הַמִּשְׁכָּן שֶׁל אוֹהֶל הַהִתְכַּנְּסוּת.
מא אֶת־בִּגְדֵ֥י הַשְּׂרָ֖ד לְשָׁרֵ֣ת בַּקֹּ֑דֶשׁ אֶת־בִּגְדֵ֤י הַקֹּ֙דֶשׁ֙ לְאַהֲרֹ֣ן הַכֹּהֵ֔ן וְאֶת־בִּגְדֵ֥י בָנָ֖יו לְכַהֵֽן׃ אֶת כִּסּוּיֵי הַכֵּלִים שֶׁמְּשָׁרְתִים בָּהֶם בַּקּוֹדֶשׁ, אֶת בִּגְדֵי הַקּוֹדֶשׁ שֶׁל אַהֲרֹן הַכֹּהֵן וְאֶת בִּגְדֵי בָּנָיו שֶׁהֵם לוֹבְשִׁים כְּדֵי לְכַהֵן.
מב כְּכֹ֛ל אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶׁ֑ה כֵּ֤ן עָשׂוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֵ֖ת כׇּל־הָעֲבֹדָֽה׃ כְּמוֹ כָּל מַה שֶּׁה' צִוָּה אֶת מֹשֶׁה – כָּךְ בְּנֵי יִשְׂרָאֵל עָשׂוּ אֶת כָּל הָעֲבוֹדָה.
מג וַיַּ֨רְא מֹשֶׁ֜ה אֶת־כׇּל־הַמְּלָאכָ֗ה וְהִנֵּה֙ עָשׂ֣וּ אֹתָ֔הּ כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה כֵּ֣ן עָשׂ֑וּ וַיְבָ֥רֶךְ אֹתָ֖ם מֹשֶֽׁה׃ מֹשֶׁה רָאָה אֶת כָּל הַמְּלָאכָה וְהִנֵּה עָשׂוּ אוֹתָהּ – כְּמוֹ שֶׁה' צִוָּה כָּךְ עָשׂוּ, וּמֹשֶׁה בֵּרֵךְ אוֹתָם.

חמישי (שביעי במחוברין)[עריכה]

פ פָּרָשַׁת צִוּוּי הֲקָמַת הַמִּשְׁכַּן
מ א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב בְּיוֹם־הַחֹ֥דֶשׁ הָרִאשׁ֖וֹן בְּאֶחָ֣ד לַחֹ֑דֶשׁ תָּקִ֕ים אֶת־מִשְׁכַּ֖ן אֹ֥הֶל מוֹעֵֽד׃ בַּחוֹדֶשׁ הָרִאשׁוֹן בַּיּוֹם הָרִאשׁוֹן לַחוֹדֶשׁ תָּקִים אֶת מִשְׁכַּן אוֹהֶל הַהִתְכַּנְּסוּת.
ג וְשַׂמְתָּ֣ שָׁ֔ם אֵ֖ת אֲר֣וֹן הָעֵד֑וּת וְסַכֹּתָ֥ עַל־הָאָרֹ֖ן אֶת־הַפָּרֹֽכֶת׃ תָּשִׂים שָׁם אֶת אֲרוֹן לוּחוֹת הַבְּרִית, וּתְסַכֵּךְ עַל הָאָרוֹן בַּפָּרוֹכֶת.
ד וְהֵבֵאתָ֙ אֶת־הַשֻּׁלְחָ֔ן וְעָרַכְתָּ֖ אֶת־עֶרְכּ֑וֹ וְהֵבֵאתָ֙ אֶת־הַמְּנֹרָ֔ה וְהַעֲלֵיתָ֖ אֶת־נֵרֹתֶֽיהָ׃ תָּבִיא אֶת הַשֻּׁלְחָן וְתַעֲרוֹךְ אֶת עֲרִיכָתוֹ, וְתָבִיא אֶת הַמְּנוֹרָה וְתַדְלִיק אֶת נֵרוֹתֶיהָ.
ה וְנָתַתָּ֞ה אֶת־מִזְבַּ֤ח הַזָּהָב֙ לִקְטֹ֔רֶת לִפְנֵ֖י אֲר֣וֹן הָעֵדֻ֑ת וְשַׂמְתָּ֛ אֶת־מָסַ֥ךְ הַפֶּ֖תַח לַמִּשְׁכָּֽן׃ תָּשִׂים אֶת מִזְבַּח הַזָּהָב לִקְטוֹרֶת לִפְנֵי אֲרוֹן לוּחוֹת הַבְּרִית, וְתָשִׂים אֶת מָסַךְ הַפֶּתַח לַמִּשְׁכָּן.
ו וְנָ֣תַתָּ֔ה אֵ֖ת מִזְבַּ֣ח הָעֹלָ֑ה לִפְנֵ֕י פֶּ֖תַח מִשְׁכַּ֥ן אֹֽהֶל־מוֹעֵֽד׃ תָּשִׂים אֶת מִזְבַּח הָעוֹלָה לִפְנֵי פֶּתַח מִשְׁכַּן אוֹהֶל הַהִתְכַּנְּסוּת.
ז וְנָֽתַתָּ֙ אֶת־הַכִּיֹּ֔ר בֵּֽין־אֹ֥הֶל מוֹעֵ֖ד וּבֵ֣ין הַמִּזְבֵּ֑חַ וְנָתַתָּ֥ שָׁ֖ם מָֽיִם׃ תָּשִׂים אֶת הַכִּיּוֹר בֵּין אוֹהֶל הַהִתְכַּנְּסוּת וּבֵין הַמִּזְבֵּחַ וְתָשִׂים שָׁם מַיִם.
ח וְשַׂמְתָּ֥ אֶת־הֶחָצֵ֖ר סָבִ֑יב וְנָ֣תַתָּ֔ אֶת־מָסַ֖ךְ שַׁ֥עַר הֶחָצֵֽר׃ תָּקִים אֶת הֶחָצֵר מִסָּבִיב, וְתַעֲמִיד אֶת מָסַךְ שַׁעַר הֶחָצֵר.
ט וְלָקַחְתָּ֙ אֶת־שֶׁ֣מֶן הַמִּשְׁחָ֔ה וּמָשַׁחְתָּ֥ אֶת־הַמִּשְׁכָּ֖ן וְאֶת־כׇּל־אֲשֶׁר־בּ֑וֹ וְקִדַּשְׁתָּ֥ אֹת֛וֹ וְאֶת־כׇּל־כֵּלָ֖יו וְהָ֥יָה קֹֽדֶשׁ׃ תִּקַּח אֶת הַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ וְתִמְרַח עַל הַמִּשְׁכָּן וְעַל כָּל מַה שֶּׁבּוֹ, וּתְקַדֵּשׁ אוֹתוֹ וְאֶת כָּל כֵּלָיו וְהוּא יִהְיֶה קָדוֹשׁ.
י וּמָשַׁחְתָּ֛ אֶת־מִזְבַּ֥ח הָעֹלָ֖ה וְאֶת־כׇּל־כֵּלָ֑יו וְקִדַּשְׁתָּ֙ אֶת־הַמִּזְבֵּ֔חַ וְהָיָ֥ה הַמִּזְבֵּ֖חַ קֹ֥דֶשׁ קׇֽדָשִֽׁים׃ תִּמְרַח אֶת הַשֶּׁמֶן עַל מִזְבַּח הָעוֹלָה וְעַל כָּל כֵּלָיו, וּתְקַדֵּשׁ אֶת הַמִּזְבֵּחַ וְהַמִּזְבֵּחַ יִהְיֶה קוֹדֶשׁ קָדָשִׁים.
יא וּמָשַׁחְתָּ֥ אֶת־הַכִּיֹּ֖ר וְאֶת־כַּנּ֑וֹ וְקִדַּשְׁתָּ֖ אֹתֽוֹ׃ תִּמְרַח עַל הַכִּיּוֹר וְעַל בְּסִיסוֹ וּתְקַדֵּשׁ אוֹתוֹ.
יב וְהִקְרַבְתָּ֤ אֶֽת־אַהֲרֹן֙ וְאֶת־בָּנָ֔יו אֶל־פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְרָחַצְתָּ֥ אֹתָ֖ם בַּמָּֽיִם׃ תְּקָרֵב אֶת אַהֲרֹן וְאֶת בָּנָיו לְפֶתַח אוֹהֶל הַהִתְכַּנְּסוּת, וְתַטְבִּיל אוֹתָם בְּמַיִם.
יג וְהִלְבַּשְׁתָּ֙ אֶֽת־אַהֲרֹ֔ן אֵ֖ת בִּגְדֵ֣י הַקֹּ֑דֶשׁ וּמָשַׁחְתָּ֥ אֹת֛וֹ וְקִדַּשְׁתָּ֥ אֹת֖וֹ וְכִהֵ֥ן לִֽי׃ תַּלְבִּישׁ אֶת אַהֲרֹן אֶת בִּגְדֵי הַקּוֹדֶשׁ, וְתִמְרַח עָלָיו אֶת הַשֶּׁמֶן וּתְקַדֵּשׁ אוֹתוֹ וְהוּא יְכַהֵן לִי.
יד וְאֶת־בָּנָ֖יו תַּקְרִ֑יב וְהִלְבַּשְׁתָּ֥ אֹתָ֖ם כֻּתֳּנֹֽת׃ וּתְקָרֵב אֶת בָּנָיו וְתַלְבִּישׁ אוֹתָם כֻּתּוֹנוֹת.
טו וּמָשַׁחְתָּ֣ אֹתָ֗ם כַּאֲשֶׁ֤ר מָשַׁ֙חְתָּ֙ אֶת־אֲבִיהֶ֔ם וְכִהֲנ֖וּ לִ֑י וְ֠הָיְתָ֠ה לִהְיֹ֨ת לָהֶ֧ם מׇשְׁחָתָ֛ם לִכְהֻנַּ֥ת עוֹלָ֖ם לְדֹרֹתָֽם׃ וְתִמְרַח עֲלֵיהֶם כְּמוֹ שֶׁמָּרַחְתָּ עַל אֲבִיהֶם וְהֵם יְכַהֲנוּ לִי, כְּדֵי שֶׁמְּרִיחָתָם תַּכְשִׁיר אוֹתָם שֶׁתִּהְיֶה לָהֶם כְּהֻנַּת עוֹלָם לְדוֹרוֹת.
טז וַיַּ֖עַשׂ מֹשֶׁ֑ה כְּ֠כֹ֠ל אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֹת֖וֹ כֵּ֥ן עָשָֽׂה׃ מֹשֶׁה בִּצֵּעַ – כְּכָל מַה שֶּׁה' צִוָּה אוֹתוֹ, כָּךְ הוּא עָשָׂה.

ס פָּרָשַׁת הֲקָמַת הַמִּשְׁכַּן
יז וַיְהִ֞י בַּחֹ֧דֶשׁ הָרִאשׁ֛וֹן בַּשָּׁנָ֥ה הַשֵּׁנִ֖ית בְּאֶחָ֣ד לַחֹ֑דֶשׁ הוּקַ֖ם הַמִּשְׁכָּֽן׃ בַּחוֹדֶשׁ הָרִאשׁוֹן בַּשָּׁנָה הַשְּׁנִיָּה בַּיּוֹם הָרִאשׁוֹן לַחוֹדֶשׁ הוּקַם הַמִּשְׁכָּן.
יח וַיָּ֨קֶם מֹשֶׁ֜ה אֶת־הַמִּשְׁכָּ֗ן וַיִּתֵּן֙ אֶת־אֲדָנָ֔יו וַיָּ֙שֶׂם֙ אֶת־קְרָשָׁ֔יו וַיִּתֵּ֖ן אֶת־בְּרִיחָ֑יו וַיָּ֖קֶם אֶת־עַמּוּדָֽיו׃ מֹשֶׁה הֵקִים אֶת הַמִּשְׁכָּן הוּא הֵנִיחַ אֶת בְּסִיסָיו, שָׂם אֶת קְרָשָׁיו, שָׂם אֶת בְּרִיחָיו, וְהֵקִים אֶת עַמּוּדָיו.
יט וַיִּפְרֹ֤שׂ אֶת־הָאֹ֙הֶל֙ עַל־הַמִּשְׁכָּ֔ן וַיָּ֜שֶׂם אֶת־מִכְסֵ֥ה הָאֹ֛הֶל עָלָ֖יו מִלְמָ֑עְלָה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ הוּא פָּרַשׂ אֶת יְרִיעוֹת הָאוֹהֶל עַל הַמִּשְׁכָּן וְשָׂם אֶת כִּסּוּי הָאוֹהֶל עָלָיו מִלְּמַעְלָה, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

כ וַיִּקַּ֞ח וַיִּתֵּ֤ן אֶת־הָעֵדֻת֙ אֶל־הָ֣אָרֹ֔ן וַיָּ֥שֶׂם אֶת־הַבַּדִּ֖ים עַל־הָאָרֹ֑ן וַיִּתֵּ֧ן אֶת־הַכַּפֹּ֛רֶת עַל־הָאָרֹ֖ן מִלְמָֽעְלָה׃ הוּא לָקַח אֶת לוּחוֹת הַבְּרִית וְהִכְנִיס אוֹתָן לְתוֹךְ הָאָרוֹן, שָׂם אֶת הַמּוֹטוֹת עַל הָאָרוֹן, וְשָׂם אֶת הַכִּסּוּי עַל הָאָרוֹן מִלְּמַעְלָה.
כא וַיָּבֵ֣א אֶת־הָאָרֹן֮ אֶל־הַמִּשְׁכָּן֒ וַיָּ֗שֶׂם אֵ֚ת פָּרֹ֣כֶת הַמָּסָ֔ךְ וַיָּ֕סֶךְ עַ֖ל אֲר֣וֹן הָעֵד֑וּת כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ הוּא הֵבִיא אֶת הָאָרוֹן לַמִּשְׁכָּן, שָׂם אֶת מָסָךְ הַפָּרוֹכֶת וְסִכֵּךְ בּוֹ עַל אֲרוֹן לוּחוֹת הַבְּרִית, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

כג וַיַּעֲרֹ֥ךְ עָלָ֛יו עֵ֥רֶךְ לֶ֖חֶם לִפְנֵ֣י יְהֹוָ֑ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ הוּא עָרַךְ עָלָיו עֲרִיכַת לֶחֶם לִפְנֵי ה', כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

לד וַיְכַ֥ס הֶעָנָ֖ן אֶת־אֹ֣הֶל מוֹעֵ֑ד וּכְב֣וֹד יְהֹוָ֔ה מָלֵ֖א אֶת־הַמִּשְׁכָּֽן׃ הֶעָנָן כִּסָּה אֶת אוֹהֶל הַהִתְכַּנְּסוּת, וּכְבוֹד ה' מִלֵּא אֶת הַמִּשְׁכָּן.
לה וְלֹא־יָכֹ֣ל מֹשֶׁ֗ה לָבוֹא֙ אֶל־אֹ֣הֶל מוֹעֵ֔ד כִּֽי־שָׁכַ֥ן עָלָ֖יו הֶעָנָ֑ן וּכְב֣וֹד יְהֹוָ֔ה מָלֵ֖א אֶת־הַמִּשְׁכָּֽן׃ מֹשֶׁה לֹא הָיָה יָכוֹל לָבוֹא לְאוֹהֶל הַהִתְכַּנְּסוּת כִּי הֶעָנָן שָׁכַן עָלָיו, וּכְבוֹד ה' מִלֵּא אֶת הַמִּשְׁכָּן.
לו וּבְהֵעָל֤וֹת הֶֽעָנָן֙ מֵעַ֣ל הַמִּשְׁכָּ֔ן יִסְע֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל בְּכֹ֖ל מַסְעֵיהֶֽם׃ וּכְשֶַׁהֶעָנָן עוֹלֶה מֵעַל הַמִּשְׁכָּן בְּנֵי יִשְׂרָאֵל נוֹסְעִים, כָּךְ הָיָה בְּכָל מַסְעוֹתֵיהֶם.
לז וְאִם־לֹ֥א יֵעָלֶ֖ה הֶעָנָ֑ן וְלֹ֣א יִסְע֔וּ עַד־י֖וֹם הֵעָלֹתֽוֹ׃ וְאִם הֶעָנָן לֹא עוֹלֶה – לֹא נוֹסְעִים, עַד לְיוֹם עֲלִיָּתוֹ.
לח כִּי֩ עֲנַ֨ן יְהֹוָ֤ה עַֽל־הַמִּשְׁכָּן֙ יוֹמָ֔ם וְאֵ֕שׁ תִּהְיֶ֥ה לַ֖יְלָה בּ֑וֹ לְעֵינֵ֥י כׇל־בֵּֽית־יִשְׂרָאֵ֖ל בְּכׇל־מַסְעֵיהֶֽם׃ כִּי עֲנַן ה' נִמְצָא עַל הַמִּשְׁכָּן בַּיּוֹם, וְאֵשׁ נִמְצֵאת בּוֹ בַּלַּיְלָה, לְעֵינֵי כָּל בֵּית יִשְׂרָאֵל בְּכָל מַסְעוֹתֵיהֶם.
