ראשון[עריכה]

פ פָּרָשַׁת הַמַּסָּעוֹת
לג א אֵ֜לֶּה מַסְעֵ֣י בְנֵֽי־יִשְׂרָאֵ֗ל אֲשֶׁ֥ר יָצְא֛וּ מֵאֶ֥רֶץ מִצְרַ֖יִם לְצִבְאֹתָ֑ם בְּיַד־מֹשֶׁ֖ה וְאַהֲרֹֽן׃ אֵלֶּה הֵם מַסְּעוֹת בְּנֵי יִשְׂרָאֵל שֶׁיָּצְאוּ עִם כָּל קְהִלּוֹתֵיהֶם מֵאֶרֶץ מִצְרַיִם, מוּנְהָגִים בִּידֵי מֹשֶׁה וְאַהֲרֹן.
ב וַיִּכְתֹּ֨ב מֹשֶׁ֜ה אֶת־מוֹצָאֵיהֶ֛ם לְמַסְעֵיהֶ֖ם עַל־פִּ֣י יְהֹוָ֑ה וְאֵ֥לֶּה מַסְעֵיהֶ֖ם לְמוֹצָאֵיהֶֽם׃ מֹשֶׁה כָּתַב אֶת שְׁמוֹת הַמְּקוֹמוֹת שֶׁמִּשָּׁם יָצְאוּ לְמַסְּעוֹתֵיהֶם עַל פִּי צִוּוּי ה', וְאֵלֶּה הֵם מַסְּעוֹתֵיהֶם עַל פִּי מְקוֹמוֹת הַמּוֹצָא:
ג וַיִּסְע֤וּ מֵֽרַעְמְסֵס֙ בַּחֹ֣דֶשׁ הָֽרִאשׁ֔וֹן בַּחֲמִשָּׁ֥ה עָשָׂ֛ר י֖וֹם לַחֹ֣דֶשׁ הָרִאשׁ֑וֹן מִֽמׇּחֳרַ֣ת הַפֶּ֗סַח יָצְא֤וּ בְנֵֽי־יִשְׂרָאֵל֙ בְּיָ֣ד רָמָ֔ה לְעֵינֵ֖י כׇּל־מִצְרָֽיִם׃ הֵם נָסְעוּ מֵרַעְמְסֵס בַּחוֹדֶשׁ הָרִאשׁוֹן – בַּיּוֹם הַחֲמִשָּׁה עָשָׂר לַחוֹדֶשׁ הָרִאשׁוֹן, בְּנֵי יִשְׂרָאֵל יָצְאוּ לְמָחֳרַת שְׁחִיטַת קָרְבַּן הַפֶּסַח, בְּיָד מוּרֶמֶת בְּגַאֲוָה לְעֵינֵי כָּל מִצְרַיִם.
ד וּמִצְרַ֣יִם מְקַבְּרִ֗ים אֵת֩ אֲשֶׁ֨ר הִכָּ֧ה יְהֹוָ֛ה בָּהֶ֖ם כׇּל־בְּכ֑וֹר וּבֵאלֹ֣הֵיהֶ֔ם עָשָׂ֥ה יְהֹוָ֖ה שְׁפָטִֽים׃ וּמִצְרַיִם קוֹבְרִים אֶת כָּל הַבְּכוֹרוֹת שֶׁה' הָרַג בָּהֶם, וַה' הֶעֱנִישׁ אֶת אֱלִילֵיהֶם.
ה וַיִּסְע֥וּ בְנֵֽי־יִשְׂרָאֵ֖ל מֵרַעְמְסֵ֑ס וַֽיַּחֲנ֖וּ בְּסֻכֹּֽת׃ בְּנֵי יִשְׂרָאֵל נָסְעוּ מֵרַעְמְסֵס וְחָנוּ בְּסֻכּוֹת.
ו וַיִּסְע֖וּ מִסֻּכֹּ֑ת וַיַּחֲנ֣וּ בְאֵתָ֔ם אֲשֶׁ֖ר בִּקְצֵ֥ה הַמִּדְבָּֽר׃ הֵם נָסְעוּ מִסֻּכּוֹת וְחָנוּ בְּאֵתָם שֶׁבִּקְצֵה הַמִּדְבָּר.
ז וַיִּסְעוּ֙ מֵֽאֵתָ֔ם וַיָּ֙שׇׁב֙ עַל־פִּ֣י הַחִירֹ֔ת אֲשֶׁ֥ר עַל־פְּנֵ֖י בַּ֣עַל צְפ֑וֹן וַֽיַּחֲנ֖וּ לִפְנֵ֥י מִגְדֹּֽל׃ הֵם נָסְעוּ מֵאֵתָם וְחָזְרוּ אֶל לְיַד 'פִּי הַחִירוֹת' שֶׁלִּפְנֵי 'בַּעַל צְפוֹן', וְחָנוּ לִפְנֵי מִגְדּוֹל.
ח וַיִּסְעוּ֙ מִפְּנֵ֣י הַֽחִירֹ֔ת וַיַּֽעַבְר֥וּ בְתוֹךְ־הַיָּ֖ם הַמִּדְבָּ֑רָה וַיֵּ֨לְכ֜וּ דֶּ֣רֶךְ שְׁלֹ֤שֶׁת יָמִים֙ בְּמִדְבַּ֣ר אֵתָ֔ם וַֽיַּחֲנ֖וּ בְּמָרָֽה׃ הֵם נָסְעוּ מִ'פִּי הַחִירוֹת' וְעָבְרוּ בְּתוֹךְ הַיָּם לַמִּדְבָּר, וְהָלְכוּ דֶּרֶךְ שֶׁל שְׁלוֹשָׁה יָמִים בְּמִדְבַּר אֵתָם וְחָנוּ בְּמָרָה.
ט וַיִּסְעוּ֙ מִמָּרָ֔ה וַיָּבֹ֖אוּ אֵילִ֑מָה וּ֠בְאֵילִ֠ם שְׁתֵּ֣ים עֶשְׂרֵ֞ה עֵינֹ֥ת מַ֛יִם וְשִׁבְעִ֥ים תְּמָרִ֖ים וַיַּחֲנוּ־שָֽׁם׃ הֵם נָסְעוּ מִמָּרָה וּבָאוּ לְאֵילִים, וּבְאֵילִים הָיוּ שְׁתֵּים עֶשְׂרֵה מַעְיָנוֹת מַיִם וְשִׁבְעִים עֲצֵי תָּמָר, וְהֵם חָנוּ שָׁם.
י וַיִּסְע֖וּ מֵאֵילִ֑ם וַֽיַּחֲנ֖וּ עַל־יַם־סֽוּף׃ הֵם נָסְעוּ מֵאֵילִים וְחָנוּ עַל יַד יַם סוּף.

שני[עריכה]

יא וַיִּסְע֖וּ מִיַּם־ס֑וּף וַֽיַּחֲנ֖וּ בְּמִדְבַּר־סִֽין׃ הֵם נָסְעוּ מִיַּם סוּף וְחָנוּ בְּמִדְבַּר סִין.
יב וַיִּסְע֖וּ מִמִּדְבַּר־סִ֑ין וַֽיַּחֲנ֖וּ בְּדׇפְקָֽה׃ הֵם נָסְעוּ מִמִּדְבַּר סִין וְחָנוּ בְּדָפְקָה.
יג וַיִּסְע֖וּ מִדׇּפְקָ֑ה וַֽיַּחֲנ֖וּ בְּאָלֽוּשׁ׃ הֵם נָסְעוּ מִדָּפְקָה וְחָנוּ בְּאָלוּשׁ.
יד וַיִּסְע֖וּ מֵאָל֑וּשׁ וַֽיַּחֲנוּ֙ בִּרְפִידִ֔ם וְלֹא־הָ֨יָה שָׁ֥ם מַ֛יִם לָעָ֖ם לִשְׁתּֽוֹת׃ הֵם נָסְעוּ מֵאָלוּשׁ וְחָנוּ בִּרְפִידִים, וְלֹא הָיָה שָׁם מַיִם לָעָם לִשְׁתּוֹת.
טו וַיִּסְע֖וּ מֵרְפִידִ֑ם וַֽיַּחֲנ֖וּ בְּמִדְבַּ֥ר סִינָֽי׃ הֵם נָסְעוּ מֵרְפִידִים וְחָנוּ בְּמִדְבַּר סִינַי.
טז וַיִּסְע֖וּ מִמִּדְבַּ֣ר סִינָ֑י וַֽיַּחֲנ֖וּ בְּקִבְרֹ֥ת הַֽתַּאֲוָֽה׃ הֵם נָסְעוּ מִמִּדְבַּר סִינַי וְחָנוּ בְּקִבְרוֹת הַתַּאֲוָה.
יז וַיִּסְע֖וּ מִקִּבְרֹ֣ת הַֽתַּאֲוָ֑ה וַֽיַּחֲנ֖וּ בַּחֲצֵרֹֽת׃ הֵם נָסְעוּ מִקִּבְרוֹת הַתַּאֲוָה וְחָנוּ בַּחֲצֵרוֹת.
יח וַיִּסְע֖וּ מֵחֲצֵרֹ֑ת וַֽיַּחֲנ֖וּ בְּרִתְמָֽה׃ הֵם נָסְעוּ מֵחֲצֵרוֹת וְחָנוּ בְּרִתְמָה.
יט וַיִּסְע֖וּ מֵרִתְמָ֑ה וַֽיַּחֲנ֖וּ בְּרִמֹּ֥ן פָּֽרֶץ׃ הֵם נָסְעוּ מֵרִתְמָה וְחָנוּ בְּרִמּוֹן פֶּרֶץ.
כ וַיִּסְע֖וּ מֵרִמֹּ֣ן פָּ֑רֶץ וַֽיַּחֲנ֖וּ בְּלִבְנָֽה׃ הֵם נָסְעוּ מֵרִמּוֹן פֶּרֶץ וְחָנוּ בְּלִבְנָה.
כא וַיִּסְע֖וּ מִלִּבְנָ֑ה וַֽיַּחֲנ֖וּ בְּרִסָּֽה׃ הֵם נָסְעוּ מִלִּבְנָה וְחָנוּ בְּרִסָּה.
כב וַיִּסְע֖וּ מֵרִסָּ֑ה וַֽיַּחֲנ֖וּ בִּקְהֵלָֽתָה׃ הֵם נָסְעוּ מֵרִסָּה וְחָנוּ בִּקְהֵלָתָה.
כג וַיִּסְע֖וּ מִקְּהֵלָ֑תָה וַֽיַּחֲנ֖וּ בְּהַר־שָֽׁפֶר׃ הֵם נָסְעוּ מִקְּהֵלָתָה וְחָנוּ בְּהַר שֶׁפֶר.
כד וַיִּסְע֖וּ מֵֽהַר־שָׁ֑פֶר וַֽיַּחֲנ֖וּ בַּחֲרָדָֽה׃ הֵם נָסְעוּ מֵהַר שֶׁפֶר וְחָנוּ בַּחֲרָדָה.
כה וַיִּסְע֖וּ מֵחֲרָדָ֑ה וַֽיַּחֲנ֖וּ בְּמַקְהֵלֹֽת׃ הֵם נָסְעוּ מֵחֲרָדָה וְחָנוּ בְּמַקְהֵלוֹת.
כו וַיִּסְע֖וּ מִמַּקְהֵלֹ֑ת וַֽיַּחֲנ֖וּ בְּתָֽחַת׃ הֵם נָסְעוּ מִמַּקְהֵלוֹת וְחָנוּ בְּתָחַת.
כז וַיִּסְע֖וּ מִתָּ֑חַת וַֽיַּחֲנ֖וּ בְּתָֽרַח׃ הֵם נָסְעוּ מִתָּחַת וְחָנוּ בְּתָרַח.
כח וַיִּסְע֖וּ מִתָּ֑רַח וַֽיַּחֲנ֖וּ בְּמִתְקָֽה׃ הֵם נָסְעוּ מִתָּרַח וְחָנוּ בְּמִתְקָה.
כט וַיִּסְע֖וּ מִמִּתְקָ֑ה וַֽיַּחֲנ֖וּ בְּחַשְׁמֹנָֽה׃ הֵם נָסְעוּ מִמִּתְקָה וְחָנוּ בְּחַשְׁמוֹנָה.
ל וַיִּסְע֖וּ מֵֽחַשְׁמֹנָ֑ה וַֽיַּחֲנ֖וּ בְּמֹסֵרֽוֹת׃ הֵם נָסְעוּ מֵחַשְׁמוֹנָה וְחָנוּ בְּמוֹסֵרוֹת.
לא וַיִּסְע֖וּ מִמֹּסֵר֑וֹת וַֽיַּחֲנ֖וּ בִּבְנֵ֥י יַעֲקָֽן׃ הֵם נָסְעוּ מִמּוֹסֵרוֹת וְחָנוּ בִּבְנֵי יַעֲקָן.
לב וַיִּסְע֖וּ מִבְּנֵ֣י יַעֲקָ֑ן וַֽיַּחֲנ֖וּ בְּחֹ֥ר הַגִּדְגָּֽד׃ הֵם נָסְעוּ מִבְּנֵי יַעֲקָן וְחָנוּ בְּחוֹר הַגִּדְגָּד.
לג וַיִּסְע֖וּ מֵחֹ֣ר הַגִּדְגָּ֑ד וַֽיַּחֲנ֖וּ בְּיׇטְבָֽתָה׃ הֵם נָסְעוּ מֵחוֹר הַגִּדְגָּד וְחָנוּ בְּיָטְבָתָה.
לד וַיִּסְע֖וּ מִיׇּטְבָ֑תָה וַֽיַּחֲנ֖וּ בְּעַבְרֹנָֽה׃ הֵם נָסְעוּ מִיָּטְבָתָה וְחָנוּ בְּעַבְרוֹנָה.
לה וַיִּסְע֖וּ מֵֽעַבְרֹנָ֑ה וַֽיַּחֲנ֖וּ בְּעֶצְיֹ֥ן גָּֽבֶר׃ הֵם נָסְעוּ מֵעַבְרוֹנָה וְחָנוּ בְּעֶצְיוֹן גֶּבֶר.
לו וַיִּסְע֖וּ מֵעֶצְיֹ֣ן גָּ֑בֶר וַיַּחֲנ֥וּ בְמִדְבַּר־צִ֖ן הִ֥וא קָדֵֽשׁ׃ הֵם נָסְעוּ מֵעֶצְיוֹן גֶּבֶר וְחָנוּ בְּמִדְבַּר צִין הִיא קָדֵשׁ.
לז וַיִּסְע֖וּ מִקָּדֵ֑שׁ וַֽיַּחֲנוּ֙ בְּהֹ֣ר הָהָ֔ר בִּקְצֵ֖ה אֶ֥רֶץ אֱדֽוֹם׃ הֵם נָסְעוּ מִקָּדֵשׁ וְחָנוּ בְּהוֹר-הָהָר – בִּקְצֵה אֶרֶץ אֱדוֹם.
לח וַיַּ֩עַל֩ אַהֲרֹ֨ן הַכֹּהֵ֜ן אֶל־הֹ֥ר הָהָ֛ר עַל־פִּ֥י יְהֹוָ֖ה וַיָּ֣מׇת שָׁ֑ם בִּשְׁנַ֣ת הָֽאַרְבָּעִ֗ים לְצֵ֤את בְּנֵֽי־יִשְׂרָאֵל֙ מֵאֶ֣רֶץ מִצְרַ֔יִם בַּחֹ֥דֶשׁ הַחֲמִישִׁ֖י בְּאֶחָ֥ד לַחֹֽדֶשׁ׃ אַהֲרֹן הַכֹּהֵן עָלָה לְהוֹר-הָהָר עַל פִּי צִוּוּי ה', הוּא מֵת שָׁם בִּשְׁנַת הָאַרְבָּעִים לְצֵאת בְּנֵי יִשְׂרָאֵל מֵאֶרֶץ מִצְרַיִם, בַּחוֹדֶשׁ הַחֲמִישִׁי בַּיּוֹם הָרִאשׁוֹן לַחוֹדֶשׁ.
לט וְאַהֲרֹ֔ן בֶּן־שָׁלֹ֧שׁ וְעֶשְׂרִ֛ים וּמְאַ֖ת שָׁנָ֑ה בְּמֹת֖וֹ בְּהֹ֥ר הָהָֽר׃ וְאַהֲרֹן הָיָה בֶּן מֵאָה עֶשְׂרִים וְשָׁלוֹשׁ שָׁנִים כְּשֶׁמֵּת בְּהוֹר-הָהָר.
ס
מ וַיִּשְׁמַ֗ע הַֽכְּנַעֲנִי֙ מֶ֣לֶךְ עֲרָ֔ד וְהֽוּא־יֹשֵׁ֥ב בַּנֶּ֖גֶב בְּאֶ֣רֶץ כְּנָ֑עַן בְּבֹ֖א בְּנֵ֥י יִשְׂרָאֵֽל׃ מֶלֶךְ עֲרָד הַכְּנַעֲנִי שֶׁגָּר בַּנֶּגֶב בְּאֶרֶץ כְּנַעַן שָׁמַע שֶׁבְּנֵי יִשְׂרָאֵל הִגִּיעוּ.
מא וַיִּסְע֖וּ מֵהֹ֣ר הָהָ֑ר וַֽיַּחֲנ֖וּ בְּצַלְמֹנָֽה׃ הֵם נָסְעוּ מֵהוֹר-הָהָר וְחָנוּ בְּצַלְמוֹנָה.
מב וַיִּסְע֖וּ מִצַּלְמֹנָ֑ה וַֽיַּחֲנ֖וּ בְּפוּנֹֽן׃ הֵם נָסְעוּ מִצַּלְמוֹנָה וְחָנוּ בְּפוּנוֹן.
מג וַיִּסְע֖וּ מִפּוּנֹ֑ן וַֽיַּחֲנ֖וּ בְּאֹבֹֽת׃ הֵם נָסְעוּ מִפּוּנוֹן וְחָנוּ בְּאוֹבוֹת.
מד וַיִּסְע֖וּ מֵאֹבֹ֑ת וַֽיַּחֲנ֛וּ בְּעִיֵּ֥י הָעֲבָרִ֖ים בִּגְב֥וּל מוֹאָֽב׃ הֵם נָסְעוּ מֵאוֹבוֹת וְחָנוּ בְּחָרְבוֹת הַמַּעֲבָרִים בַּגְּבוּל עִם מוֹאָב.
מה וַיִּסְע֖וּ מֵעִיִּ֑ים וַֽיַּחֲנ֖וּ בְּדִיבֹ֥ן גָּֽד׃ הֵם נָסְעוּ מֵהָחֳרָבוֹת וְחָנוּ בְּדִיבוֹן גָּד.
מו וַיִּסְע֖וּ מִדִּיבֹ֣ן גָּ֑ד וַֽיַּחֲנ֖וּ בְּעַלְמֹ֥ן דִּבְלָתָֽיְמָה׃ הֵם נָסְעוּ מִדִּיבוֹן גָּד וְחָנוּ בְּעַלְמוֹן דִּבְלָתַיְמָה.
מז וַיִּסְע֖וּ מֵעַלְמֹ֣ן דִּבְלָתָ֑יְמָה וַֽיַּחֲנ֛וּ בְּהָרֵ֥י הָעֲבָרִ֖ים לִפְנֵ֥י נְבֽוֹ׃ הֵם נָסְעוּ מֵעַלְמוֹן דִּבְלָתַיְמָה וְחָנוּ בְּהָרֵי 'מַעֲבַר הַיַּרְדֵּן' לִפְנֵי נְבוֹ.
מח וַיִּסְע֖וּ מֵהָרֵ֣י הָעֲבָרִ֑ים וַֽיַּחֲנוּ֙ בְּעַֽרְבֹ֣ת מוֹאָ֔ב עַ֖ל יַרְדֵּ֥ן יְרֵחֽוֹ׃ הֵם נָסְעוּ מֵהָרֵי 'מַעֲבַר הַיַּרְדֵּן' וְחָנוּ בַּמִּישׁוֹרִים שֶׁל מוֹאָב עַל יַד הַיַּרְדֵּן שֶׁמּוּל יְרִיחוֹ.
מט וַיַּחֲנ֤וּ עַל־הַיַּרְדֵּן֙ מִבֵּ֣ית הַיְשִׁמֹ֔ת עַ֖ד אָבֵ֣ל הַשִּׁטִּ֑ים בְּעַֽרְבֹ֖ת מוֹאָֽב׃ וְהֵם חָנוּ עַל יַד הַיַּרְדֵּן – מִ'בֵּית הַיְּשִׁימוֹת' עַד מִישׁוֹר עֲצֵי הַשִּׁטָּה הַנִּקְרָא 'אָבֵל', בַּמִּישׁוֹרִים שֶׁל מוֹאָב.

שלישי (חמישי במחוברין)[עריכה]

ס פָּרָשַׁת הוֹרָשַׁת הָאָרֶץ
נ וַיְדַבֵּ֧ר יְהֹוָ֛ה אֶל־מֹשֶׁ֖ה בְּעַֽרְבֹ֣ת מוֹאָ֑ב עַל־יַרְדֵּ֥ן יְרֵח֖וֹ לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה בַּמִּישׁוֹרִים שֶׁל מוֹאָב עַל יַד הַיַּרְדֵּן שֶׁמּוּל יְרִיחוֹ וְאָמַר:
נא דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָמַרְתָּ֖ אֲלֵהֶ֑ם כִּ֥י אַתֶּ֛ם עֹבְרִ֥ים אֶת־הַיַּרְדֵּ֖ן אֶל־אֶ֥רֶץ כְּנָֽעַן׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְתֹאמַר לָהֶם: כְּשֶׁתַּעַבְרוּ אֶת הַיַּרְדֵּן לְאֶרֶץ כְּנַעַן,
נב וְה֨וֹרַשְׁתֶּ֜ם אֶת־כׇּל־יֹשְׁבֵ֤י הָאָ֙רֶץ֙ מִפְּנֵיכֶ֔ם וְאִ֨בַּדְתֶּ֔ם אֵ֖ת כׇּל־מַשְׂכִּיֹּתָ֑ם וְאֵ֨ת כׇּל־צַלְמֵ֤י מַסֵּֽכֹתָם֙ תְּאַבֵּ֔דוּ וְאֵ֥ת כׇּל־בָּמוֹתָ֖ם תַּשְׁמִֽידוּ׃ תַּשְׁמִידוּ אֶת כָּל יוֹשְׁבֵי הָאָרֶץ מִפְּנֵיכֶם, וּתְאַבְּדוּ אֶת כָּל מְקוֹמוֹת הַהִשְׁתַּחֲוָיָה שֶׁלָּהֶם, וּתְאַבְּדוּ אֶת כָּל הַפְּסָלִים הַיְּצוּקִים שֶׁלָּהֶם, וְתַשְׁמִידוּ אֶת כָּל הַמִּזְבְּחוֹת שֶׁלָּהֶם.
נג וְהוֹרַשְׁתֶּ֥ם אֶת־הָאָ֖רֶץ וִֽישַׁבְתֶּם־בָּ֑הּ כִּ֥י לָכֶ֛ם נָתַ֥תִּי אֶת־הָאָ֖רֶץ לָרֶ֥שֶׁת אֹתָֽהּ׃ כִּבְשׁוּ אֶת הָאָרֶץ וְתָגוּרוּ בָּהּ, כִּי לָכֶם נָתַתִּי אֶת הָאָרֶץ לְנַחֲלָה.
נד וְהִתְנַחַלְתֶּם֩ אֶת־הָאָ֨רֶץ בְּגוֹרָ֜ל לְמִשְׁפְּחֹֽתֵיכֶ֗ם לָרַ֞ב תַּרְבּ֤וּ אֶת־נַחֲלָתוֹ֙ וְלַמְעַט֙ תַּמְעִ֣יט אֶת־נַחֲלָת֔וֹ אֶל֩ אֲשֶׁר־יֵ֨צֵא ל֥וֹ שָׁ֛מָּה הַגּוֹרָ֖ל ל֣וֹ יִהְיֶ֑ה לְמַטּ֥וֹת אֲבֹתֵיכֶ֖ם תִּתְנֶחָֽלוּ׃ וּתְחַלְּקוּ אֶת הָאָרֶץ בְּגוֹרָל לְנַחֲלוֹת לְמִשְׁפְּחוֹתֵיכֶם – לְמִשְׁפָּחָה גְּדוֹלָה תִּתְּנוּ נַחֲלָה גְּדוֹלָה וּלְמִשְׁפָּחָה קְטַנָּה תִּתְּנוּ נַחֲלָה קְטַנָּה, הַמָּקוֹם שֶׁיֵּצֵא לָהּ שָׁם הַגּוֹרָל – שֶׁלָּהּ יִהְיֶה, לְפִי שִׁבְטֵי אֲבוֹתֵיכֶם תִּנְחֲלוּ.
נה וְאִם־לֹ֨א תוֹרִ֜ישׁוּ אֶת־יֹשְׁבֵ֣י הָאָ֘רֶץ֮ מִפְּנֵיכֶם֒ וְהָיָה֙ אֲשֶׁ֣ר תּוֹתִ֣ירוּ מֵהֶ֔ם לְשִׂכִּים֙ בְּעֵ֣ינֵיכֶ֔ם וְלִצְנִינִ֖ם בְּצִדֵּיכֶ֑ם וְצָרְר֣וּ אֶתְכֶ֔ם עַל־הָאָ֕רֶץ אֲשֶׁ֥ר אַתֶּ֖ם יֹשְׁבִ֥ים בָּֽהּ׃ וְאִם לֹא תְּסַלְּקוּ אֶת יוֹשְׁבֵי הָאָרֶץ מִפְּנֵיכֶם, אֵלֶּה שֶׁתַּשְׁאִירוּ מֵהֶם יִהְיוּ כְּמוֹ סַכִּינִים בְּעֵינֵיכֶם וּכְמוֹ קוֹצִים סְבִיבְכֶם, וְיַעֲשׂוּ לָכֶם צָרוֹת בָּאָרֶץ שֶׁאַתֶּם יוֹשְׁבִים בָּהּ.
נו וְהָיָ֗ה כַּאֲשֶׁ֥ר דִּמִּ֛יתִי לַעֲשׂ֥וֹת לָהֶ֖ם אֶֽעֱשֶׂ֥ה לָכֶֽם׃ וּכְמוֹ שֶׁחָשַׁבְתִּי לַעֲשׂוֹת לָהֶם כָּךְ אֶעֱשֶׂה לָכֶם.
פ פָּרָשַׁת הַגְּבוּלוֹת
לד א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב צַ֞ו אֶת־בְּנֵ֤י יִשְׂרָאֵל֙ וְאָמַרְתָּ֣ אֲלֵהֶ֔ם כִּֽי־אַתֶּ֥ם בָּאִ֖ים אֶל־הָאָ֣רֶץ כְּנָ֑עַן זֹ֣את הָאָ֗רֶץ אֲשֶׁ֨ר תִּפֹּ֤ל לָכֶם֙ בְּֽנַחֲלָ֔ה אֶ֥רֶץ כְּנַ֖עַן לִגְבֻלֹתֶֽיהָ׃ צַוֵּה אֶת בְּנֵי יִשְׂרָאֵל וֶאֱמוֹר לָהֶם: כְּשֶׁתָּבוֹאוּ לָאָרֶץ – אֶרֶץ כְּנַעַן, זֹאת הָאָרֶץ שֶׁתִּהְיֶה לָכֶם לְנַחֲלָה – אֶרֶץ כְּנַעַן לִגְבוּלוֹתֶיהָ.
ג וְהָיָ֨ה לָכֶ֧ם פְּאַת־נֶ֛גֶב מִמִּדְבַּר־צִ֖ן עַל־יְדֵ֣י אֱד֑וֹם וְהָיָ֤ה לָכֶם֙ גְּב֣וּל נֶ֔גֶב מִקְצֵ֥ה יָם־הַמֶּ֖לַח קֵֽדְמָה׃ הַגְּבוּל הַדְּרוֹמִי יִהְיֶה לָכֶם מִמִּדְבַּר צִין שֶׁעַל יַד אֱדוֹם, וְיִהְיֶה לָכֶם קְצֵה גְּבוּל הַדָּרוֹם מִקְצֵה יָם הַמֶּלַח בְּצַד מִזְרָח.
ד וְנָסַ֣ב לָכֶם֩ הַגְּב֨וּל מִנֶּ֜גֶב לְמַעֲלֵ֤ה עַקְרַבִּים֙ וְעָ֣בַר צִ֔נָה והיה וְהָיוּ֙ תּֽוֹצְאֹתָ֔יו מִנֶּ֖גֶב לְקָדֵ֣שׁ בַּרְנֵ֑עַ וְיָצָ֥א חֲצַר־אַדָּ֖ר וְעָבַ֥ר עַצְמֹֽנָה׃ הַגְּבוּל הַדְּרוֹמִי יַקִּיף לָכֶם לְמַעֲלֵה עַקְרַבִּים וְיַעֲבוֹר לְצִין, וְיִהְיֶה קָצֵהוּ דְּרוֹמִית לְקָדֵשׁ בַּרְנֵעַ, וְיַמְשִׁיךְ לַחֲצַר אַדָּר וְיַעֲבוֹר לְעַצְמוֹן.
ה וְנָסַ֧ב הַגְּב֛וּל מֵעַצְמ֖וֹן נַ֣חְלָה מִצְרָ֑יִם וְהָי֥וּ תוֹצְאֹתָ֖יו הַיָּֽמָּה׃ הַגְּבוּל יַקִּיף מֵעַצְמוֹן וְיַמְשִׁיךְ אֶל נַחַל מִצְרַיִם, וְקָצֵהוּ יִהְיֶה בַּיָּם.
ו וּגְב֣וּל יָ֔ם וְהָיָ֥ה לָכֶ֛ם הַיָּ֥ם הַגָּד֖וֹל וּגְב֑וּל זֶֽה־יִהְיֶ֥ה לָכֶ֖ם גְּב֥וּל יָֽם׃ וּגְבוּל מַעֲרָב – הַיָּם הַגָּדוֹל יִהְיֶה לָכֶם לִגְבוּל, זֶה יִהְיֶה לָכֶם גְּבוּל מַעֲרָב.
ז וְזֶֽה־יִהְיֶ֥ה לָכֶ֖ם גְּב֣וּל צָפ֑וֹן מִן־הַיָּם֙ הַגָּדֹ֔ל תְּתָא֥וּ לָכֶ֖ם הֹ֥ר הָהָֽר׃ וְזֶה יִהְיֶה לָכֶם גְּבוּל הַצָּפוֹן – מֵהַיָּם הַגָּדוֹל תְּכַוְּנוּ לָכֶם אֶת הַגְּבוּל אֶל הַפִּסְגָּה הַגְּבוֹהָה שֶׁל הָרֵי אַמְנוֹן.
ח מֵהֹ֣ר הָהָ֔ר תְּתָא֖וּ לְבֹ֣א חֲמָ֑ת וְהָי֛וּ תּוֹצְאֹ֥ת הַגְּבֻ֖ל צְדָֽדָה׃ מֵהַפִּסְגָּה הַגְּבוֹהָה תְּכַוְּנוּ אֶת הַגְּבוּל לְכִוּוּן אַנְטְיוֹכְיָא, וְיִהְיֶה קְצֵה הַגְּבוּל בִּצְדָד.
ט וְיָצָ֤א הַגְּבֻל֙ זִפְרֹ֔נָה וְהָי֥וּ תוֹצְאֹתָ֖יו חֲצַ֣ר עֵינָ֑ן זֶֽה־יִהְיֶ֥ה לָכֶ֖ם גְּב֥וּל צָפֽוֹן׃ הַגְּבוּל יֵצֵא לְזִפְרוֹן וְיִהְיֶה קָצֵהוּ בַּחֲצַר עֵינָן, זֶה יִהְיֶה לָכֶם גְּבוּל צָפוֹן.
י וְהִתְאַוִּיתֶ֥ם לָכֶ֖ם לִגְב֣וּל קֵ֑דְמָה מֵחֲצַ֥ר עֵינָ֖ן שְׁפָֽמָה׃ וּתְכַוְּנוּ לָכֶם אֶת גְּבוּל הַמִּזְרָח מֵחֲצַר עֵינָן אֶל שְׁפָם.
יא וְיָרַ֨ד הַגְּבֻ֧ל מִשְּׁפָ֛ם הָרִבְלָ֖ה מִקֶּ֣דֶם לָעָ֑יִן וְיָרַ֣ד הַגְּבֻ֔ל וּמָחָ֛ה עַל־כֶּ֥תֶף יָם־כִּנֶּ֖רֶת קֵֽדְמָה׃ הַגְּבוּל יֵרֵד מִשְּׁפָם לְרִבְלָה, מִזְרָחִית לַמַּעְיָן, הַגְּבוּל יֵרֵד וְיַגִּיעַ לַפִּנָּה הַמִּזְרָחִית שֶׁל יָם כִּנֶּרֶת.
יב וְיָרַ֤ד הַגְּבוּל֙ הַיַּרְדֵּ֔נָה וְהָי֥וּ תוֹצְאֹתָ֖יו יָ֣ם הַמֶּ֑לַח זֹאת֩ תִּהְיֶ֨ה לָכֶ֥ם הָאָ֛רֶץ לִגְבֻלֹתֶ֖יהָ סָבִֽיב׃ הַגְּבוּל יֵרֵד לַיַּרְדֵּן וְקָצֵהוּ יִהְיֶה בְּיָם הַמֶּלַח. זֹאת תִּהְיֶה לָכֶם הָאָרֶץ לִגְבוּלוֹתֶיהָ מִסָּבִיב.
יג וַיְצַ֣ו מֹשֶׁ֔ה אֶת־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר זֹ֣את הָאָ֗רֶץ אֲשֶׁ֨ר תִּתְנַחֲל֤וּ אֹתָהּ֙ בְּגוֹרָ֔ל אֲשֶׁר֙ צִוָּ֣ה יְהֹוָ֔ה לָתֵ֛ת לְתִשְׁעַ֥ת הַמַּטּ֖וֹת וַחֲצִ֥י הַמַּטֶּֽה׃ מֹשֶׁה צִוָּה אֶת בְּנֵי יִשְׂרָאֵל וְאָמַר: זֹאת הָאָרֶץ שֶׁתְּחַלְּקוּ אוֹתָהּ לְנַחֲלָה בְּגוֹרָל, שֶׁה' צִוָּה לָתֵת לְתִשְׁעָה וָחֵצִי הַשְּׁבָטִים.
יד כִּ֣י לָקְח֞וּ מַטֵּ֨ה בְנֵ֤י הָראוּבֵנִי֙ לְבֵ֣ית אֲבֹתָ֔ם וּמַטֵּ֥ה בְנֵֽי־הַגָּדִ֖י לְבֵ֣ית אֲבֹתָ֑ם וַחֲצִי֙ מַטֵּ֣ה מְנַשֶּׁ֔ה לָקְח֖וּ נַחֲלָתָֽם׃ כִּי שֵׁבֶט בְּנֵי רְאוּבֵן לְמִשְׁפְּחוֹתֵיהֶם וְשֵׁבֶט בְּנֵי גָּד לְמִשְׁפְּחוֹתֵיהֶם, וַחֲצִי שֵׁבֶט מְנַשֶּׁה כְּבָר לָקְחוּ אֶת נַחֲלָתָם.
טו שְׁנֵ֥י הַמַּטּ֖וֹת וַחֲצִ֣י הַמַּטֶּ֑ה לָקְח֣וּ נַחֲלָתָ֗ם מֵעֵ֛בֶר לְיַרְדֵּ֥ן יְרֵח֖וֹ קֵ֥דְמָה מִזְרָֽחָה׃ שְׁנַיִם וָחֵצִי הַשְּׁבָטִים כְּבָר קִבְּלוּ אֶת נַחֲלָתָם בָּעֵבֶר הַמִּזְרָחִי שֶׁל הַיַּרְדֵּן שֶׁמּוּל יְרִיחוֹ.

רביעי (ששי במחוברין)[עריכה]

פ פָּרָשַׁת הַמַּנְחִילִים
טז וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יז אֵ֚לֶּה שְׁמ֣וֹת הָֽאֲנָשִׁ֔ים אֲשֶׁר־יִנְחֲל֥וּ לָכֶ֖ם אֶת־הָאָ֑רֶץ אֶלְעָזָר֙ הַכֹּהֵ֔ן וִיהוֹשֻׁ֖עַ בִּן־נֽוּן׃ אֵלֶּה שְׁמוֹת הָאֲנָשִׁים שֶׁיַּנְחִילוּ לָכֶם אֶת הָאָרֶץ: אֶלְעָזָר הַכֹּהֵן, וִיהוֹשֻׁעַ בִּן נוּן.
יח וְנָשִׂ֥יא אֶחָ֛ד נָשִׂ֥יא אֶחָ֖ד מִמַּטֶּ֑ה תִּקְח֖וּ לִנְחֹ֥ל אֶת־הָאָֽרֶץ׃ וְנָשִׂיא אֶחָד מִכָּל שֵׁבֶט תִּקְחוּ לְהַנְחִיל אֶת הָאָרֶץ.
יט וְאֵ֖לֶּה שְׁמ֣וֹת הָאֲנָשִׁ֑ים לְמַטֵּ֣ה יְהוּדָ֔ה כָּלֵ֖ב בֶּן־יְפֻנֶּֽה׃ וְאֵלֶּה שְׁמוֹת הָאֲנָשִׁים: לְשֵׁבֶט יְהוּדָה – כָּלֵב בֶּן יְפֻנֶּה.
כ וּלְמַטֵּה֙ בְּנֵ֣י שִׁמְע֔וֹן שְׁמוּאֵ֖ל בֶּן־עַמִּיהֽוּד׃ וּלְשֵׁבֶט בְּנֵי שִׁמְעוֹן – שְׁמוּאֵל בֶּן עַמִּיהוּד.
כא לְמַטֵּ֣ה בִנְיָמִ֔ן אֱלִידָ֖ד בֶּן־כִּסְלֽוֹן׃ לְשֵׁבֶט בִּנְיָמִין – אֱלִידָד בֶּן כִּסְלוֹן.
כב וּלְמַטֵּ֥ה בְנֵי־דָ֖ן נָשִׂ֑יא בֻּקִּ֖י בֶּן־יׇגְלִֽי׃ וְהַנָּשִׂיא לְשֵׁבֶט בְּנֵי דָּן – בֻּקִּי בֶּן יָגְלִי.
כג לִבְנֵ֣י יוֹסֵ֔ף לְמַטֵּ֥ה בְנֵֽי־מְנַשֶּׁ֖ה נָשִׂ֑יא חַנִּיאֵ֖ל בֶּן־אֵפֹֽד׃ לִבְנֵי יוֹסֵף: הַנָּשִׂיא לְשֵׁבֶט בְּנֵי מְנַשֶּׁה – חַנִּיאֵל בֶּן אֵפוֹד.
כד וּלְמַטֵּ֥ה בְנֵֽי־אֶפְרַ֖יִם נָשִׂ֑יא קְמוּאֵ֖ל בֶּן־שִׁפְטָֽן׃ וְהַנָּשִׂיא לְשֵׁבֶט בְּנֵי אֶפְרַיִם – קְמוּאֵל בֶּן שִׁפְטָן.
כה וּלְמַטֵּ֥ה בְנֵֽי־זְבוּלֻ֖ן נָשִׂ֑יא אֱלִיצָפָ֖ן בֶּן־פַּרְנָֽךְ׃ וְהַנָּשִׂיא לְשֵׁבֶט בְּנֵי זְבוּלוּן – אֱלִיצָפָן בֶּן פַּרְנָךְ.
כו וּלְמַטֵּ֥ה בְנֵֽי־יִשָּׂשכָ֖ר נָשִׂ֑יא פַּלְטִיאֵ֖ל בֶּן־עַזָּֽן׃ וְהַנָּשִׂיא לְשֵׁבֶט בְּנֵי יִשָּׂשכָר – פַּלְטִיאֵל בֶּן עַזָּן.
כז וּלְמַטֵּ֥ה בְנֵי־אָשֵׁ֖ר נָשִׂ֑יא אֲחִיה֖וּד בֶּן־שְׁלֹמִֽי׃ וְהַנָּשִׂיא לְשֵׁבֶט בְּנֵי אָשֵׁר – אֲחִיהוּד בֶּן שְׁלוֹמִי.
כח וּלְמַטֵּ֥ה בְנֵֽי־נַפְתָּלִ֖י נָשִׂ֑יא פְּדַהְאֵ֖ל בֶּן־עַמִּיהֽוּד׃ וְהַנָּשִׂיא לְשֵׁבֶט בְּנֵי נַפְתָּלִי – פְּדַהְאֵל בֶּן עַמִּיהוּד.
כט אֵ֕לֶּה אֲשֶׁ֖ר צִוָּ֣ה יְהֹוָ֑ה לְנַחֵ֥ל אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל בְּאֶ֥רֶץ כְּנָֽעַן׃ אֵלֶּה הָאֲנָשִׁים שֶׁה' צִוָּה שֶׁיַּנְחִילוּ אֶת בְּנֵי יִשְׂרָאֵל בְּאֶרֶץ כְּנַעַן.

חמישי[עריכה]

פ פָּרָשַׁת עָרֵי הַלְוִיִּים
לה א וַיְדַבֵּ֧ר יְהֹוָ֛ה אֶל־מֹשֶׁ֖ה בְּעַֽרְבֹ֣ת מוֹאָ֑ב עַל־יַרְדֵּ֥ן יְרֵח֖וֹ לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה בַּמִּישׁוֹרִים שֶׁל מוֹאָב עַל יַד הַיַּרְדֵּן שֶׁמּוּל יְרִיחוֹ וְאָמַר:
ב צַו֮ אֶת־בְּנֵ֣י יִשְׂרָאֵל֒ וְנָתְנ֣וּ לַלְוִיִּ֗ם מִֽנַּחֲלַ֛ת אֲחֻזָּתָ֖ם עָרִ֣ים לָשָׁ֑בֶת וּמִגְרָ֗שׁ לֶֽעָרִים֙ סְבִיבֹ֣תֵיהֶ֔ם תִּתְּנ֖וּ לַלְוִיִּֽם׃ צַוֵּה אֶת בְּנֵי יִשְׂרָאֵל שֶׁיִּתְּנוּ לַלְּוִיִּים חֵלֶק מִנַּחֲלָתָם לְעָרִים לִמְגוּרִים, וְתִתְּנוּ לַלְוִיִּים גַּם מִגְרָשׁ סְבִיב כָּל עִיר.
ג וְהָי֧וּ הֶֽעָרִ֛ים לָהֶ֖ם לָשָׁ֑בֶת וּמִגְרְשֵׁיהֶ֗ם יִהְי֤וּ לִבְהֶמְתָּם֙ וְלִרְכֻשָׁ֔ם וּלְכֹ֖ל חַיָּתָֽם׃ הֶעָרִים יִהְיוּ לָהֶם לִמְגוּרִים, וּמִגְרְשֵׁיהֶם יִהְיוּ לְבֶהֱמוֹתֵיהֶם וְלִרְכוּשָׁם וּלְכָל חַיּוֹתֵיהֶם.
ד וּמִגְרְשֵׁי֙ הֶֽעָרִ֔ים אֲשֶׁ֥ר תִּתְּנ֖וּ לַלְוִיִּ֑ם מִקִּ֤יר הָעִיר֙ וָח֔וּצָה אֶ֥לֶף אַמָּ֖ה סָבִֽיב׃ וּמִגְרְשֵׁי הֶעָרִים שֶׁתִּתְּנוּ לַלְוִיִּים לְשִׁמּוּשָׁם – אֶלֶף אַמָּה מִסָּבִיב, מֵחוֹמַת הָעִיר כְּלַפֵּי חוּץ.
ה וּמַדֹּתֶ֞ם מִח֣וּץ לָעִ֗יר אֶת־פְּאַת־קֵ֣דְמָה אַלְפַּ֪יִם בָּֽאַמָּ֟ה וְאֶת־פְּאַת־נֶ֩גֶב֩ אַלְפַּ֨יִם בָּאַמָּ֜ה וְאֶת־פְּאַת־יָ֣ם ׀ אַלְפַּ֣יִם בָּֽאַמָּ֗ה וְאֵ֨ת פְּאַ֥ת צָפ֛וֹן אַלְפַּ֥יִם בָּאַמָּ֖ה וְהָעִ֣יר בַּתָּ֑וֶךְ זֶ֚ה יִהְיֶ֣ה לָהֶ֔ם מִגְרְשֵׁ֖י הֶעָרִֽים׃ וְתִמְדְּדוּ מִחוּץ לָעִיר לְצַד מִזְרַח אַלְפַּיִם אַמָּה, וּלְצַד דָּרוֹם אַלְפַּיִם אַמָּה, וּלְצַד מַעֲרָב אַלְפַּיִם אַמָּה, וּלְצַד צָפוֹן אַלְפַּיִם אַמָּה לְמִגְרָשׁ וּלְחַקְלָאוּת, וְהָעִיר בַּמֶּרְכָּז, זֶה יִהְיֶה לָהֶם מִגְרְשֵׁי הֶעָרִים.
ו וְאֵ֣ת הֶֽעָרִ֗ים אֲשֶׁ֤ר תִּתְּנוּ֙ לַלְוִיִּ֔ם אֵ֚ת שֵׁשׁ־עָרֵ֣י הַמִּקְלָ֔ט אֲשֶׁ֣ר תִּתְּנ֔וּ לָנֻ֥ס שָׁ֖מָּה הָרֹצֵ֑חַ וַעֲלֵיהֶ֣ם תִּתְּנ֔וּ אַרְבָּעִ֥ים וּשְׁתַּ֖יִם עִֽיר׃ הֶעָרִים שֶׁתִּתְּנוּ לַלְוִיִּים – בְּנוֹסַף לְשֵׁשׁ עָרֵי הַמִּקְלָט שֶׁתִּתְּנוּ לָרוֹצְחִים בְּטָעוּת שֶׁיִּבְרְחוּ לְשָׁם, תּוֹסִיפוּ עֲלֵיהֶם וְתִתְּנוּ עוֹד אַרְבָּעִים וּשְׁתַּיִם עָרִים.
ז כׇּל־הֶעָרִ֗ים אֲשֶׁ֤ר תִּתְּנוּ֙ לַלְוִיִּ֔ם אַרְבָּעִ֥ים וּשְׁמֹנֶ֖ה עִ֑יר אֶתְהֶ֖ן וְאֶת־מִגְרְשֵׁיהֶֽן׃ כָּל הֶעָרִים שֶׁתִּתְּנוּ לַלְוִיִּים אַרְבָּעִים וּשְׁמוֹנֶה עָרִים, תִּתְּנוּ אוֹתָן וְאֶת מִגְרְשֵׁיהֶן.
ח וְהֶֽעָרִ֗ים אֲשֶׁ֤ר תִּתְּנוּ֙ מֵאֲחֻזַּ֣ת בְּנֵי־יִשְׂרָאֵ֔ל מֵאֵ֤ת הָרַב֙ תַּרְבּ֔וּ וּמֵאֵ֥ת הַמְעַ֖ט תַּמְעִ֑יטוּ אִ֗ישׁ כְּפִ֤י נַחֲלָתוֹ֙ אֲשֶׁ֣ר יִנְחָ֔לוּ יִתֵּ֥ן מֵעָרָ֖יו לַלְוִיִּֽם׃ וְהֶעָרִים שֶׁתִּתְּנוּ מֵנַחֲלַת בְּנֵי יִשְׂרָאֵל – מִשֵּׁבֶט גָּדוֹל תִּקְחוּ עָרִים רַבּוֹת, וּמִשֵּׁבֶט קָטָן תִּקְחוּ מְעַט – כָּל שֵׁבֶט לְפִי גּוֹדֶל נַחֲלָתוֹ שֶׁיְּקַבֵּל כָּךְ יִתֵּן מֵעָרָיו לַלְוִיִּים.

ששי (שביעי במחוברין)[עריכה]

פ פָּרָשַׁת הָרוֹצֵחַ
ט וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
י דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָמַרְתָּ֖ אֲלֵהֶ֑ם כִּ֥י אַתֶּ֛ם עֹבְרִ֥ים אֶת־הַיַּרְדֵּ֖ן אַ֥רְצָה כְּנָֽעַן׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְתֹאמַר לָהֶם: כְּשֶׁתַּעַבְרוּ אֶת הַיַּרְדֵּן לְאֶרֶץ כְּנַעַן,
יא וְהִקְרִיתֶ֤ם לָכֶם֙ עָרִ֔ים עָרֵ֥י מִקְלָ֖ט תִּהְיֶ֣ינָה לָכֶ֑ם וְנָ֥ס שָׁ֙מָּה֙ רֹצֵ֔חַ מַכֵּה־נֶ֖פֶשׁ בִּשְׁגָגָֽה׃ תָּכִינוּ לָכֶם עָרִים שֶׁתִּהְיֶינָה לָכֶם עָרֵי מִקְלָט – שֶׁיָּנוּס לְשָׁם רוֹצֵחַ שֶׁהָרַג אָדָם בְּטָעוּת.
יב וְהָי֨וּ לָכֶ֧ם הֶעָרִ֛ים לְמִקְלָ֖ט מִגֹּאֵ֑ל וְלֹ֤א יָמוּת֙ הָרֹצֵ֔חַ עַד־עׇמְד֛וֹ לִפְנֵ֥י הָעֵדָ֖ה לַמִּשְׁפָּֽט׃ הֶעָרִים תִּהְיֶינָה לְמִקְלָט מִגּוֹאֵל הַדָּם, וְהָרוֹצֵחַ לֹא יֵהָרֵג עַד שֶׁהוּא יַעֲמוֹד לְמִשְׁפָּט לִפְנֵי הַדַּיָּנִים.
יג וְהֶעָרִ֖ים אֲשֶׁ֣ר תִּתֵּ֑נוּ שֵׁשׁ־עָרֵ֥י מִקְלָ֖ט תִּהְיֶ֥ינָה לָכֶֽם׃ וְהֶעָרִים שֶׁתָּכִינוּ לְכָךְ – יִהְיוּ לָכֶם שֵׁשׁ עָרֵי מִקְלָט.
יד אֵ֣ת ׀ שְׁלֹ֣שׁ הֶעָרִ֗ים תִּתְּנוּ֙ מֵעֵ֣בֶר לַיַּרְדֵּ֔ן וְאֵת֙ שְׁלֹ֣שׁ הֶֽעָרִ֔ים תִּתְּנ֖וּ בְּאֶ֣רֶץ כְּנָ֑עַן עָרֵ֥י מִקְלָ֖ט תִּהְיֶֽינָה׃ אֶת שְׁלוֹשׁ הֶעָרִים הָרִאשׁוֹנוֹת תָּשִׂימוּ מֵעֵבֶר לַיַּרְדֵּן מִמִּזְרָח, וְאֶת שְׁלוֹשׁ הֶעָרִים הָאֲחֵרוֹת תָּשִׂימוּ בְּאֶרֶץ כְּנַעַן, וְהֵן תִּהְיֶינָה עָרֵי מִקְלָט.
טו לִבְנֵ֣י יִשְׂרָאֵ֗ל וְלַגֵּ֤ר וְלַתּוֹשָׁב֙ בְּתוֹכָ֔ם תִּהְיֶ֛ינָה שֵׁשׁ־הֶעָרִ֥ים הָאֵ֖לֶּה לְמִקְלָ֑ט לָנ֣וּס שָׁ֔מָּה כׇּל־מַכֵּה־נֶ֖פֶשׁ בִּשְׁגָגָֽה׃ שֵׁשׁ הֶעָרִים הָאֵלֶּה תִּהְיֶינָה לְמִקְלָט לִבְנֵי יִשְׂרָאֵל וְלַגֵּר וְלַתּוֹשָׁב שֶׁבְּתוֹכָם, שֶׁיִּבְרַח לְשָׁם כָּל מִי שֶׁהָרַג אָדָם בְּטָעוּת.
טז וְאִם־בִּכְלִ֨י בַרְזֶ֧ל ׀ הִכָּ֛הוּ וַיָּמֹ֖ת רֹצֵ֣חַֽ ה֑וּא מ֥וֹת יוּמַ֖ת הָרֹצֵֽחַ׃ וְאִם הוּא הִכָּה אוֹתוֹ בִּכְלִי בַּרְזֶל וְהוּא מֵת – הוּא רוֹצֵחַ בְּמֵזִיד, הָרוֹצֵחַ יֵהָרֵג.
יז וְאִ֡ם בְּאֶ֣בֶן יָד֩ אֲשֶׁר־יָמ֨וּת בָּ֥הּ הִכָּ֛הוּ וַיָּמֹ֖ת רֹצֵ֣חַֽ ה֑וּא מ֥וֹת יוּמַ֖ת הָרֹצֵֽחַ׃ וְאִם הִכָּה אוֹתוֹ בְּאֶבֶן בְּגוֹדֶל שֶׁל כַּף יָד – שֶׁבָּהּ הוּא עָלוּל לָמוּת, וְהוּא אָכֵן מֵת, הוּא רוֹצֵחַ בְּמֵזִיד, הָרוֹצֵחַ יֵהָרֵג.
יח א֡וֹ בִּכְלִ֣י עֵֽץ־יָד֩ אֲשֶׁר־יָמ֨וּת בּ֥וֹ הִכָּ֛הוּ וַיָּמֹ֖ת רֹצֵ֣חַֽ ה֑וּא מ֥וֹת יוּמַ֖ת הָרֹצֵֽחַ׃ אוֹ אִם הִכָּה אוֹתוֹ בִּכְלִי עֵץ בְּגוֹדֶל שֶׁל כַּף יָד – שֶׁבּוֹ הוּא עָלוּל לָמוּת, וְהוּא אָכֵן מֵת, זֶהוּ רוֹצֵחַ בְּמֵזִיד, הָרוֹצֵחַ יֵהָרֵג.
יט גֹּאֵ֣ל הַדָּ֔ם ה֥וּא יָמִ֖ית אֶת־הָרֹצֵ֑חַ בְּפִגְעוֹ־ב֖וֹ ה֥וּא יְמִתֶֽנּוּ׃ גּוֹאֵל הַדָּם הוּא יַהֲרוֹג אֶת הָרוֹצֵחַ, הוּא יַהֲרוֹג אוֹתוֹ כְּשֶׁיִּפְגּוֹשׁ אוֹתוֹ.
כ וְאִם־בְּשִׂנְאָ֖ה יֶהְדֳּפֶ֑נּוּ אֽוֹ־הִשְׁלִ֥יךְ עָלָ֛יו בִּצְדִיָּ֖ה וַיָּמֹֽת׃ וְאִם הוּא דָּחַף אוֹתוֹ בְּשִׂנְאָה, אוֹ זָרַק עָלָיו בְּכַוָּנָה וְהוּא מֵת.
כא א֣וֹ בְאֵיבָ֞ה הִכָּ֤הוּ בְיָדוֹ֙ וַיָּמֹ֔ת מֽוֹת־יוּמַ֥ת הַמַּכֶּ֖ה רֹצֵ֣חַֽ ה֑וּא גֹּאֵ֣ל הַדָּ֗ם יָמִ֛ית אֶת־הָרֹצֵ֖חַ בְּפִגְעוֹ־בֽוֹ׃ אוֹ אִם הוּא הִכָּה אוֹתוֹ בְּיָדוֹ בְּשִׂנְאָה וְהוּא מֵת – הַמַּכֶּה יֵהָרֵג, הוּא רוֹצֵחַ, גּוֹאֵל הַדָּם יַהֲרוֹג אֶת הָרוֹצֵחַ כְּשֶׁהוּא יִפְגּוֹשׁ אוֹתוֹ.
כב וְאִם־בְּפֶ֥תַע בְּלֹא־אֵיבָ֖ה הֲדָפ֑וֹ אוֹ־הִשְׁלִ֥יךְ עָלָ֛יו כׇּל־כְּלִ֖י בְּלֹ֥א צְדִיָּֽה׃ וְאִם הוּא דָּחַף אוֹתוֹ בְּטָעוּת, בְּלִי שִׂנְאָה, אוֹ זָרַק עָלָיו אֵיזֶה כְּלִי בְּלִי כַּוָּנָה.
כג א֣וֹ בְכׇל־אֶ֜בֶן אֲשֶׁר־יָמ֥וּת בָּהּ֙ בְּלֹ֣א רְא֔וֹת וַיַּפֵּ֥ל עָלָ֖יו וַיָּמֹ֑ת וְהוּא֙ לֹא־אוֹיֵ֣ב ל֔וֹ וְלֹ֥א מְבַקֵּ֖שׁ רָעָתֽוֹ׃ אוֹ הָרַג אוֹתוֹ בְּאֶבֶן כָּלְשֶׁהִי שֶׁהוּא עָלוּל לָמוּת מִמֶּנָּה, בְּלִי שֶׁרָאָה אוֹתוֹ, וְהוּא הִפִּיל עָלָיו וְהוּא מֵת, וְהוּא לֹא אוֹיֵב שֶׁלּוֹ וְלֹא מְחַפֵּשׁ אֶת רָעָתוֹ –
כד וְשָֽׁפְטוּ֙ הָֽעֵדָ֔ה בֵּ֚ין הַמַּכֶּ֔ה וּבֵ֖ין גֹּאֵ֣ל הַדָּ֑ם עַ֥ל הַמִּשְׁפָּטִ֖ים הָאֵֽלֶּה׃ הַדַּיָּנִים יִשְׁפְּטוּ בֵּין הַמַּכֶּה וּבֵין גּוֹאֵל הַדָּם עַל פִּי הַמִּשְׁפָּטִים הָאֵלֶּה.
כה וְהִצִּ֨ילוּ הָעֵדָ֜ה אֶת־הָרֹצֵ֗חַ מִיַּד֮ גֹּאֵ֣ל הַדָּם֒ וְהֵשִׁ֤יבוּ אֹתוֹ֙ הָֽעֵדָ֔ה אֶל־עִ֥יר מִקְלָט֖וֹ אֲשֶׁר־נָ֣ס שָׁ֑מָּה וְיָ֣שַׁב בָּ֗הּ עַד־מוֹת֙ הַכֹּהֵ֣ן הַגָּדֹ֔ל אֲשֶׁר־מָשַׁ֥ח אֹת֖וֹ בְּשֶׁ֥מֶן הַקֹּֽדֶשׁ׃ הַדַּיָּנִים יַצִּילוּ אֶת הָרוֹצֵחַ מִיַּד גּוֹאֵל הַדָּם, וְיַחֲזִירוּ אוֹתוֹ לְעִיר מִקְלָטוֹ שֶׁהוּא בָּרַח לְשָׁם, וְהוּא יָגוּר בָּהּ עַד שֶׁהַכֹּהֵן הַגָּדוֹל שֶׁנִּמְשַׁח בְּשֶׁמֶן הַקּוֹדֶשׁ יָמוּת.
כו וְאִם־יָצֹ֥א יֵצֵ֖א הָרֹצֵ֑חַ אֶת־גְּבוּל֙ עִ֣יר מִקְלָט֔וֹ אֲשֶׁ֥ר יָנ֖וּס שָֽׁמָּה׃ וְאִם הָרוֹצֵחַ יֵצֵא מִגְּבוּל עִיר מִקְלָטוֹ שֶׁהוּא בָּרַח לְשָׁם.
כז וּמָצָ֤א אֹתוֹ֙ גֹּאֵ֣ל הַדָּ֔ם מִח֕וּץ לִגְב֖וּל עִ֣יר מִקְלָט֑וֹ וְרָצַ֞ח גֹּאֵ֤ל הַדָּם֙ אֶת־הָ֣רֹצֵ֔חַ אֵ֥ין ל֖וֹ דָּֽם׃ וְגוֹאֵל הַדָּם יִמְצָא אוֹתוֹ מִחוּץ לִגְבוּל עִיר מִקְלָטוֹ – גּוֹאֵל הַדָּם רַשַּׁאי לַהֲרוֹג אֶת הָרוֹצֵחַ, אֵין אִסּוּר רְצִיחָה בַּהֲרִיגָתוֹ.
כח כִּ֣י בְעִ֤יר מִקְלָטוֹ֙ יֵשֵׁ֔ב עַד־מ֖וֹת הַכֹּהֵ֣ן הַגָּדֹ֑ל וְאַחֲרֵ֥י מוֹת֙ הַכֹּהֵ֣ן הַגָּדֹ֔ל יָשׁוּב֙ הָרֹצֵ֔חַ אֶל־אֶ֖רֶץ אֲחֻזָּתֽוֹ׃ לָכֵן הוּא צָרִיךְ לָגוּר בְּעִיר מִקְלָטוֹ עַד שֶׁהַכֹּהֵן הַגָּדוֹל יָמוּת, וְאַחֲרֵי שֶׁהַכֹּהֵן הַגָּדוֹל יָמוּת הָרוֹצֵחַ יַחֲזוֹר לָאֶרֶץ שֶׁקִּבֵּל בְּחֶלְקוֹ.
כט וְהָי֨וּ אֵ֧לֶּה לָכֶ֛ם לְחֻקַּ֥ת מִשְׁפָּ֖ט לְדֹרֹתֵיכֶ֑ם בְּכֹ֖ל מוֹשְׁבֹתֵיכֶֽם׃ דִּינִים אֵלֶּה יִהְיוּ לָכֶם לְחוֹק מִשְׁפָּט לְדוֹרוֹת בְּכָל מְקוֹמוֹת מְגוּרֵיכֶם.
ל כׇּ֨ל־מַכֵּה־נֶ֔פֶשׁ לְפִ֣י עֵדִ֔ים יִרְצַ֖ח אֶת־הָרֹצֵ֑חַ וְעֵ֣ד אֶחָ֔ד לֹא־יַעֲנֶ֥ה בְנֶ֖פֶשׁ לָמֽוּת׃ כָּל הוֹרֵג נֶפֶשׁ – יַהַרְגוּ אֶת הָרוֹצֵחַ עַל פִּי עֵדִים, וְעֵד אֶחָד לֹא יָעִיד נֶגֶד אָדָם כְּדֵי לַהֲרוֹג אוֹתוֹ.
לא וְלֹֽא־תִקְח֥וּ כֹ֙פֶר֙ לְנֶ֣פֶשׁ רֹצֵ֔חַ אֲשֶׁר־ה֥וּא רָשָׁ֖ע לָמ֑וּת כִּי־מ֖וֹת יוּמָֽת׃ וְאַל תִּקְחוּ כֶּסֶף כְּכוֹפֶר-נֶפֶשׁ לְרוֹצֵחַ שֶׁהוּרְשַׁע בְּבֵית דִּין לְהֵהָרֵג, אֶלָּא הוּא יֵהָרֵג.
לב וְלֹא־תִקְח֣וּ כֹ֔פֶר לָנ֖וּס אֶל־עִ֣יר מִקְלָט֑וֹ לָשׁוּב֙ לָשֶׁ֣בֶת בָּאָ֔רֶץ עַד־מ֖וֹת הַכֹּהֵֽן׃ וְאַל תִּקְחוּ כֶּסֶף כְּכוֹפֶר לְמִי שֶׁחַיָּיב לִבְרוֹחַ לְעִיר מִקְלָטוֹ כְּדֵי שֶׁיּוּכַל לַחֲזוֹר וְלָגוּר בְּאַרְצוֹ לִפְנֵי מוֹת הַכֹּהֵן.
לג וְלֹֽא־תַחֲנִ֣יפוּ אֶת־הָאָ֗רֶץ אֲשֶׁ֤ר אַתֶּם֙ בָּ֔הּ כִּ֣י הַדָּ֔ם ה֥וּא יַחֲנִ֖יף אֶת־הָאָ֑רֶץ וְלָאָ֣רֶץ לֹֽא־יְכֻפַּ֗ר לַדָּם֙ אֲשֶׁ֣ר שֻׁפַּךְ־בָּ֔הּ כִּי־אִ֖ם בְּדַ֥ם שֹׁפְכֽוֹ׃ וְכָךְ לֹא תַּרְשִׁיעוּ אֶת הָאָרֶץ שֶׁאַתֶּם בָּהּ, כִּי עֲווֹן רְצִיחָה מַרְשִׁיעַ אֶת הָאָרֶץ, וְלָאָרֶץ לֹא יִתְכַּפֵּר עַל הַדָּם שֶׁל מִי שֶׁנִּרְצַח בָּהּ אֶלָּא רַק בִּשְׁפִיכַת דַּם הָרוֹצֵחַ.
לד וְלֹ֧א תְטַמֵּ֣א אֶת־הָאָ֗רֶץ אֲשֶׁ֤ר אַתֶּם֙ יֹשְׁבִ֣ים בָּ֔הּ אֲשֶׁ֥ר אֲנִ֖י שֹׁכֵ֣ן בְּתוֹכָ֑הּ כִּ֚י אֲנִ֣י יְהֹוָ֔ה שֹׁכֵ֕ן בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל׃ וְלֹא תְּטַמְּאוּ אֶת הָאָרֶץ שֶׁאַתֶּם יוֹשְׁבִים בָּהּ, שֶׁאֲנִי שׁוֹכֵן בְּתוֹכָהּ, שֶׁהֲרֵי אֲנִי ה' שׁוֹכֵן בְּתוֹךְ בְּנֵי יִשְׂרָאֵל.

שביעי[עריכה]

פ פָּרָשַׁת נַחֲלַת צְלָפְחָד
לו א וַֽיִּקְרְב֞וּ רָאשֵׁ֣י הָֽאָב֗וֹת לְמִשְׁפַּ֤חַת בְּנֵֽי־גִלְעָד֙ בֶּן־מָכִ֣יר בֶּן־מְנַשֶּׁ֔ה מִֽמִּשְׁפְּחֹ֖ת בְּנֵ֣י יוֹסֵ֑ף וַֽיְדַבְּר֞וּ לִפְנֵ֤י מֹשֶׁה֙ וְלִפְנֵ֣י הַנְּשִׂאִ֔ים רָאשֵׁ֥י אָב֖וֹת לִבְנֵ֥י יִשְׂרָאֵֽל׃ רָאשֵׁי הַמִּשְׁפָּחוֹת שֶׁל בְּנֵי גִּלְעָד בֶּן מָכִיר בֶּן מְנַשֶּׁה מִמִּשְׁפְּחוֹת בְּנֵי יוֹסֵף נִגְּשׁוּ וְדִבְּרוּ לִפְנֵי מֹשֶׁה וְלִפְנֵי הַנְּשִׂיאִים – רָאשֵׁי הַמִּשְׁפָּחוֹת שֶׁל בְּנֵי יִשְׂרָאֵל.
ב וַיֹּאמְר֗וּ אֶת־אֲדֹנִי֙ צִוָּ֣ה יְהֹוָ֔ה לָתֵ֨ת אֶת־הָאָ֧רֶץ בְּנַחֲלָ֛ה בְּגוֹרָ֖ל לִבְנֵ֣י יִשְׂרָאֵ֑ל וַֽאדֹנִי֙ צֻוָּ֣ה בַֽיהֹוָ֔ה לָתֵ֗ת אֶֽת־נַחֲלַ֛ת צְלׇפְחָ֥ד אָחִ֖ינוּ לִבְנֹתָֽיו׃ הֵם אָמְרוּ: ה' צִוָּה אֶת אֲדוֹנִי לָתֵת אֶת הָאָרֶץ לְנַחֲלָה לִבְנֵי יִשְׂרָאֵל בְּגוֹרָל, וַאֲדֹנִי נִצְטַוָּה בִּדְבַר ה' לָתֵת אֶת נַחֲלַת צְלָפְחָד אָחִינוּ לִבְנוֹתָיו.
ג וְ֠הָי֠וּ לְאֶחָ֞ד מִבְּנֵ֨י שִׁבְטֵ֥י בְנֵֽי־יִשְׂרָאֵל֮ לְנָשִׁים֒ וְנִגְרְעָ֤ה נַחֲלָתָן֙ מִנַּחֲלַ֣ת אֲבֹתֵ֔ינוּ וְנוֹסַ֕ף עַ֚ל נַחֲלַ֣ת הַמַּטֶּ֔ה אֲשֶׁ֥ר תִּהְיֶ֖ינָה לָהֶ֑ם וּמִגֹּרַ֥ל נַחֲלָתֵ֖נוּ יִגָּרֵֽעַ׃ וּכְשֶׁהֵן יִנָּשְׂאוּ לְאֶחָד מִבְּנֵי שִׁבְטֵי בְּנֵי יִשְׂרָאֵל – נַחֲלָתָן תֶּחְסַר מִנַּחֲלַת אֲבוֹתֵינוּ וְתִתְוַסֵּף לְנַחֲלַת הַשֵּׁבֶט שֶׁהֵן יִנָּשְׂאוּ לָהֶם לְנָשִׁים, וְיֵחָסֵר מֵהַחֵלֶק שֶׁקִּבַּלְנוּ לְנַחֲלָה בַּגּוֹרָל.
ד וְאִם־יִהְיֶ֣ה הַיֹּבֵל֮ לִבְנֵ֣י יִשְׂרָאֵל֒ וְנֽוֹסְפָה֙ נַחֲלָתָ֔ן עַ֚ל נַחֲלַ֣ת הַמַּטֶּ֔ה אֲשֶׁ֥ר תִּהְיֶ֖ינָה לָהֶ֑ם וּמִֽנַּחֲלַת֙ מַטֵּ֣ה אֲבֹתֵ֔ינוּ יִגָּרַ֖ע נַחֲלָתָֽן׃ וְגַם כְּשֶׁיִּהְיֶה הַיּוֹבֵל לִבְנֵי יִשְׂרָאֵל, תִּתְוַסֵּף נַחֲלָתָן עַל נַחֲלַת הַשֵּׁבֶט שֶׁהֵן יִנָּשְׂאוּ לָהֶם, וּמִנַּחֲלַת שֵׁבֶט אֲבוֹתֵינוּ תֵּחָסֵר נַחֲלָתָן!
ה וַיְצַ֤ו מֹשֶׁה֙ אֶת־בְּנֵ֣י יִשְׂרָאֵ֔ל עַל־פִּ֥י יְהֹוָ֖ה לֵאמֹ֑ר כֵּ֛ן מַטֵּ֥ה בְנֵֽי־יוֹסֵ֖ף דֹּבְרִֽים׃ מֹשֶׁה צִוָּה אֶת בְּנֵי יִשְׂרָאֵל עַל פִּי דְּבַר ה' וְאָמַר: שֵׁבֶט בְּנֵי יוֹסֵף מְדַבְּרִים דְּבָרִים נְכוֹנִים.
ו זֶ֣ה הַדָּבָ֞ר אֲשֶׁר־צִוָּ֣ה יְהֹוָ֗ה לִבְנ֤וֹת צְלׇפְחָד֙ לֵאמֹ֔ר לַטּ֥וֹב בְּעֵינֵיהֶ֖ם תִּהְיֶ֣ינָה לְנָשִׁ֑ים אַ֗ךְ לְמִשְׁפַּ֛חַת מַטֵּ֥ה אֲבִיהֶ֖ם תִּהְיֶ֥ינָה לְנָשִֽׁים׃ זֶה הַדָּבָר שֶׁה' צִוָּה לוֹמַר לִבְנוֹת צְלָפְחָד: הֵן יִנָּשְׂאוּ לְאִישׁ שֶׁטּוֹב בְּעֵינֵיהֶן, אֲבָל רַק לְאִישׁ מִמִּשְׁפַּחַת שֵׁבֶט אֲבִיהֶן הֵן יִנָּשְׂאוּ.
ז וְלֹֽא־תִסֹּ֤ב נַחֲלָה֙ לִבְנֵ֣י יִשְׂרָאֵ֔ל מִמַּטֶּ֖ה אֶל־מַטֶּ֑ה כִּ֣י אִ֗ישׁ בְּנַחֲלַת֙ מַטֵּ֣ה אֲבֹתָ֔יו יִדְבְּק֖וּ בְּנֵ֥י יִשְׂרָאֵֽל׃ וְכָךְ לֹא תַּעֲבוֹר נַחֲלָה שֶׁל בְּנֵי יִשְׂרָאֵל מִשֵּׁבֶט לְשֵׁבֶט, אֶלָּא בְּנֵי יִשְׂרָאֵל יִדְבְּקוּ כָּל אֶחָד בְּנַחֲלַת שֵׁבֶט אֲבוֹתָיו.
ח וְכׇל־בַּ֞ת יֹרֶ֣שֶׁת נַחֲלָ֗ה מִמַּטּוֹת֮ בְּנֵ֣י יִשְׂרָאֵל֒ לְאֶחָ֗ד מִמִּשְׁפַּ֛חַת מַטֵּ֥ה אָבִ֖יהָ תִּהְיֶ֣ה לְאִשָּׁ֑ה לְמַ֗עַן יִֽירְשׁוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אִ֖ישׁ נַחֲלַ֥ת אֲבֹתָֽיו׃ וְכָל בַּת שֶׁיּוֹרֶשֶׁת נַחֲלָה מִשִּׁבְטֵי בְּנֵי יִשְׂרָאֵל תִּנָּשֵׂא רַק לְאֶחָד מִמִּשְׁפַּחַת שֵׁבֶט אָבִיהָ, כְּדֵי שֶׁשִּׁבְטֵי בְּנֵי יִשְׂרָאֵל יִירְשׁוּ כָּל אֶחָד אֶת נַחֲלַת אֲבוֹתָיו.
ט וְלֹֽא־תִסֹּ֧ב נַחֲלָ֛ה מִמַּטֶּ֖ה לְמַטֶּ֣ה אַחֵ֑ר כִּי־אִישׁ֙ בְּנַ֣חֲלָת֔וֹ יִדְבְּק֕וּ מַטּ֖וֹת בְּנֵ֥י יִשְׂרָאֵֽל׃ וְכָךְ לֹא תַּעֲבוֹר נַחֲלָה מִשֵּׁבֶט לְשֵׁבֶט אַחֵר, אֶלָּא שִׁבְטֵי בְּנֵי יִשְׂרָאֵל יִדְבְּקוּ כָּל אֶחָד בְּנַחֲלָתוֹ.
י כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶׁ֑ה כֵּ֥ן עָשׂ֖וּ בְּנ֥וֹת צְלׇפְחָֽד׃ כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה כָּךְ עָשׂוּ בְּנוֹת צְלָפְחָד.

מפטיר[עריכה]

יא וַתִּהְיֶ֜ינָה מַחְלָ֣ה תִרְצָ֗ה וְחׇגְלָ֧ה וּמִלְכָּ֛ה וְנֹעָ֖ה בְּנ֣וֹת צְלׇפְחָ֑ד לִבְנֵ֥י דֹדֵיהֶ֖ן לְנָשִֽׁים׃ מַחְלָה תִּרְצָה חָגְלָה מִלְכָּה וְנוֹעָה בְּנוֹת צְלָפְחָד נִשְּׂאוּ לִבְנֵי דּוֹדֵיהֶן.
יב מִֽמִּשְׁפְּחֹ֛ת בְּנֵֽי־מְנַשֶּׁ֥ה בֶן־יוֹסֵ֖ף הָי֣וּ לְנָשִׁ֑ים וַתְּהִי֙ נַחֲלָתָ֔ן עַל־מַטֵּ֖ה מִשְׁפַּ֥חַת אֲבִיהֶֽן׃ הֵן נִשְּׂאוּ לַאֲנָשִׁים מִמִּשְׁפְּחוֹת בְּנֵי מְנַשֶּׁה בֶּן יוֹסֵף, וְנַחֲלָתָן נִשְׁאֲרָה אֵצֶל שֵׁבֶט מִשְׁפַּחַת אֲבִיהֶן.
יג אֵ֣לֶּה הַמִּצְוֺ֞ת וְהַמִּשְׁפָּטִ֗ים אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה בְּיַד־מֹשֶׁ֖ה אֶל־בְּנֵ֣י יִשְׂרָאֵ֑ל בְּעַֽרְבֹ֣ת מוֹאָ֔ב עַ֖ל יַרְדֵּ֥ן יְרֵחֽוֹ׃ אֵלֶּה הַמִּצְווֹת וְהַמִּשְׁפָּטִים שֶׁה' צִוָּה אֶת בְּנֵי יִשְׂרָאֵל בְּאֶמְצָעוּת מֹשֶׁה, בַּמִּישׁוֹרִים שֶׁל מוֹאָב עַל יַד הַיַּרְדֵּן שֶׁמּוּל יְרִיחוֹ.
