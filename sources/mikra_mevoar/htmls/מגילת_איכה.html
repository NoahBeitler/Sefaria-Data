קִינָה רִאשׁוֹנָה
א א אֵיכָ֣ה ׀ יָשְׁבָ֣ה בָדָ֗ד הָעִיר֙ רַבָּ֣תִי עָ֔ם הָיְתָ֖ה כְּאַלְמָנָ֑ה רַבָּ֣תִי בַגּוֹיִ֗ם שָׂרָ֙תִי֙ בַּמְּדִינ֔וֹת הָיְתָ֖ה לָמַֽס׃ אֵיךְ הָעִיר שֶׁהָיָה בָּהּ עַם רָב, יָשְׁבָה לְבָד - לְלֹא תּוֹשָׁבֶיהָ, אֵיךְ נִהְיְתָה כְּמוֹ אַלְמָנָה; זֹאת שֶׁהָיְתָה חֲשׁוּבָה בֵּין הָעַמִּים וְהָיְתָה כְּמוֹ שָׂרָה בַּמְּדִינוֹת, אֵיךְ הָפְכָה לִמְשֻׁעְבֶּדֶת.
ב בָּכ֨וֹ תִבְכֶּ֜ה בַּלַּ֗יְלָה וְדִמְעָתָהּ֙ עַ֣ל לֶֽחֱיָ֔הּ אֵֽין־לָ֥הּ מְנַחֵ֖ם מִכׇּל־אֹהֲבֶ֑יהָ כׇּל־רֵעֶ֙יהָ֙ בָּ֣גְדוּ בָ֔הּ הָ֥יוּ לָ֖הּ לְאֹיְבִֽים׃ הִיא בּוֹכָה בַּלַּיְלָה וְדִמְעָתָהּ עַל לֶחֱיָהּ, אֵין לָהּ מְנַחֵם מִכָּל אוֹהֲבֶיהָ, כָּל יְדִידֶיהָ בָּגְדוּ בָּהּ וְהָפְכוּ לָהּ לְאוֹיְבִים.
ג גָּֽלְתָ֨ה יְהוּדָ֤ה מֵעֹ֙נִי֙ וּמֵרֹ֣ב עֲבֹדָ֔ה הִ֚יא יָשְׁבָ֣ה בַגּוֹיִ֔ם לֹ֥א מָצְאָ֖ה מָנ֑וֹחַ כׇּל־רֹדְפֶ֥יהָ הִשִּׂיג֖וּהָ בֵּ֥ין הַמְּצָרִֽים׃ יְהוּדָה גָּלְתָה וְיָשְׁבָה בֵּין הָעַמִּים, הִיא לֹא מָצְאָה מְנוּחָה מֵהַשִּׁפְלוּת וּמֵרוֹב עַבְדּוּת, כָּל הָרוֹדְפִים אַחֲרֶיהָ הִשִּׂיגוּ אוֹתָהּ כְּאִלּוּ הָיְתָה בְּמָקוֹם צָר שֶׁאֵין בּוֹ לְאָן לִבְרוֹחַ.
ד דַּרְכֵ֨י צִיּ֜וֹן אֲבֵל֗וֹת מִבְּלִי֙ בָּאֵ֣י מוֹעֵ֔ד כׇּל־שְׁעָרֶ֙יהָ֙ שֽׁוֹמֵמִ֔ין כֹּהֲנֶ֖יהָ נֶאֱנָחִ֑ים בְּתוּלֹתֶ֥יהָ נּוּג֖וֹת וְהִ֥יא מַר־לָֽהּ׃ דַּרְכֵי צִיּוֹן אֲבֵלוֹת – אֵין בָּאִים לֶחָג, כָּל שְׁעָרֶיהָ שׁוֹמְמִים כּוֹהֲנֶיהָ נֶאֱנָחִים, בְּתוּלוֹתֶיהָ עֲצוּבוֹת, וְהִיא מַר לָהּ.
ה הָי֨וּ צָרֶ֤יהָ לְרֹאשׁ֙ אֹיְבֶ֣יהָ שָׁל֔וּ כִּֽי־יְהֹוָ֥ה הוֹגָ֖הּ עַ֣ל רֹב־פְּשָׁעֶ֑יהָ עוֹלָלֶ֛יהָ הָלְכ֥וּ שְׁבִ֖י לִפְנֵי־צָֽר׃ צוֹרְרֶיהָ הָפְכוּ לַחֲשׁוּבִים, אוֹיְבֶיהָ בְּשַׁלְוָה, כִּי ה' צִעֵר אוֹתָהּ עַל פְּשָׁעֶיהָ הָרַבִּים, יְלָדֶיהָ הָלְכוּ בַּשֶּׁבִי בִּידֵי הַצּוֹרֵר.
ו וַיֵּצֵ֥א מן בת [מִבַּת־]צִיּ֖וֹן כׇּל־הֲדָרָ֑הּ הָי֣וּ שָׂרֶ֗יהָ כְּאַיָּלִים֙ לֹא־מָצְא֣וּ מִרְעֶ֔ה וַיֵּלְכ֥וּ בְלֹא־כֹ֖חַ לִפְנֵ֥י רוֹדֵֽף׃ יָצָא מִבַּת צִיּוֹן כָּל הֶהָדָר שֶׁלָּהּ, שָׂרֶיהָ הָיוּ כְּמוֹ אַיָּלִים שֶׁלֹּא מָצְאוּ מִרְעֶה וּבָרְחוּ בְּלִי כֹּחַ מֵהָרוֹדֵף.
ז זָֽכְרָ֣ה יְרוּשָׁלַ֗͏ִם יְמֵ֤י עׇנְיָהּ֙ וּמְרוּדֶ֔יהָ כֹּ֚ל מַחֲמֻדֶ֔יהָ אֲשֶׁ֥ר הָי֖וּ מִ֣ימֵי קֶ֑דֶם בִּנְפֹ֧ל עַמָּ֣הּ בְּיַד־צָ֗ר וְאֵ֤ין עוֹזֵר֙ לָ֔הּ רָא֣וּהָ צָרִ֔ים שָׂחֲק֖וּ עַ֥ל מִשְׁבַּתֶּֽהָ׃ יְרוּשָׁלַיִם זָכְרָה בִּימֵי שִׁפְלוּתָהּ וְגָלוּתָהּ אֶת כָּל הַדְּבָרִים הַנֶּחְמָדִים שֶׁהָיוּ לָהּ מִימֵי קֶדֶם, רָאוּ אוֹתָהּ צוֹרְרִים כְּשֶׁעַמָּהּ נוֹפֵל בְּיַד הַצּוֹרֵר וְאֵין מִי שֶׁעוֹזֵר לָהּ וְצָחֲקוּ עַל יְרִידָתָהּ מִגְּדֻלָּתָהּ.
ח חֵ֤טְא חָֽטְאָה֙ יְר֣וּשָׁלַ֔͏ִם עַל־כֵּ֖ן לְנִידָ֣ה הָיָ֑תָה כׇּֽל־מְכַבְּדֶ֤יהָ הִזִּיל֙וּהָ֙ כִּי־רָא֣וּ עֶרְוָתָ֔הּ גַּם־הִ֥יא נֶאֶנְחָ֖ה וַתָּ֥שׇׁב אָחֽוֹר׃ יְרוּשָׁלִַם חָטְאָה חֵטְא, לָכֵן נִהְיְתָה כְּמוֹ אִשָּׁה מֻרְחֶקֶת, כָּל מְכַבְּדֶיהָ זִלְזְלוּ בָּהּ כִּי רָאוּ אֶת חֶרְפָּתָהּ, גַּם הִיא נֶאֶנְחָה וְנָסוֹגָה לְאָחוֹר.
ט טֻמְאָתָ֣הּ בְּשׁוּלֶ֗יהָ לֹ֤א זָֽכְרָה֙ אַחֲרִיתָ֔הּ וַתֵּ֣רֶד פְּלָאִ֔ים אֵ֥ין מְנַחֵ֖ם לָ֑הּ רְאֵ֤ה יְהֹוָה֙ אֶת־עׇנְיִ֔י כִּ֥י הִגְדִּ֖יל אוֹיֵֽב׃ חֲטָאֶיהָ נִעֲשׂוּ גְּלוּיִים בְּלֹא שֶׁתִּזְכּוֹר מַה יִּקְרֶה בְּאַחֲרִיתָהּ, הִיא יָרְדָה פְּלָאִים אֵין לָהּ מְנַחֵם, רְאֵה ה' אֶת שִׁפְלוּתִי שֶׁהָאוֹיֵב הִגְזִים.
י יָדוֹ֙ פָּ֣רַשׂ צָ֔ר עַ֖ל כׇּל־מַחֲמַדֶּ֑יהָ כִּֽי־רָאֲתָ֤ה גוֹיִם֙ בָּ֣אוּ מִקְדָּשָׁ֔הּ אֲשֶׁ֣ר צִוִּ֔יתָה לֹא־יָבֹ֥אוּ בַקָּהָ֖ל לָֽךְ׃ הַצּוֹרֵר הֵנִיף אֶת יָדוֹ לָקַחַת אֶת כָּל חֲפָצֶיהָ הַנֶּחְמָדִים, כַּאֲשֶׁר רָאֲתָה שֶׁגּוֹיִם בָּאוּ אֶל מִקְדָּשָׁהּ – גּוֹיִם שֶׁאָסַרְתָּה לְהִתְחַתֵּן אִתָּם.
יא כׇּל־עַמָּ֤הּ נֶאֱנָחִים֙ מְבַקְשִׁ֣ים לֶ֔חֶם נָתְנ֧וּ מחמודיהם מַחֲמַדֵּיהֶ֛ם בְּאֹ֖כֶל לְהָשִׁ֣יב נָ֑פֶשׁ רְאֵ֤ה יְהֹוָה֙ וְֽהַבִּ֔יטָה כִּ֥י הָיִ֖יתִי זוֹלֵלָֽה׃ כָּל עַמָּהּ נֶאֱנָחִים מְבַקְשִׁים לֶחֶם, נָתְנוּ אֶת חֶפְצֵיהֶם הַנֶּחְמָדִים תְּמוּרַת אוֹכֶל כְּדֵי לְהָשִׁיב אֶת נַפְשָׁם, רְאֵה ה' וְהַבֵּט שֶׁהָפַכְתִּי לְזוֹלָה.
יב ל֣וֹא אֲלֵיכֶם֮ כׇּל־עֹ֣בְרֵי דֶ֒רֶךְ֒ הַבִּ֣יטוּ וּרְא֔וּ אִם־יֵ֤שׁ מַכְאוֹב֙ כְּמַכְאֹבִ֔י אֲשֶׁ֥ר עוֹלַ֖ל לִ֑י אֲשֶׁר֙ הוֹגָ֣ה יְהֹוָ֔ה בְּי֖וֹם חֲר֥וֹן אַפּֽוֹ׃ כָּל עוֹבְרֵי דֶּרֶךְ – שֶׁלֹּא תָּבֹא כָּזֹאת צָרָה אֲלֵיכֶם! הַבִּיטוּ וּרְאוּ, הֲאִם יֵשׁ מַכְאוֹב כְּמַכְאוֹבִי שֶׁה' עוֹלֵל לִי – שֶׁצִעֵר אוֹתִי בְּיוֹם כַּעֲסוֹ?
יג מִמָּר֛וֹם שָֽׁלַח־אֵ֥שׁ בְּעַצְמֹתַ֖י וַיִּרְדֶּ֑נָּה פָּרַ֨שׂ רֶ֤שֶׁת לְרַגְלַי֙ הֱשִׁיבַ֣נִי אָח֔וֹר נְתָנַ֙נִי֙ שֹֽׁמֵמָ֔ה כׇּל־הַיּ֖וֹם דָּוָֽה׃ מִמָּרוֹם שָׁלַח אֵשׁ בְּעַצְמוֹתַי וְגָרַף אוֹתָהּ, פָּרַשׂ רֶשֶׁת לְרַגְלַי וְהֵשִׁיב אוֹתִי אֲחוֹרָה, עָשָׂה אוֹתִי שׁוֹמֵמָה וְחַלָּשָׁה כָּל הַיּוֹם.
יד נִשְׂקַד֩ עֹ֨ל פְּשָׁעַ֜י בְּיָד֗וֹ יִשְׂתָּ֥רְג֛וּ עָל֥וּ עַל־צַוָּארִ֖י הִכְשִׁ֣יל כֹּחִ֑י נְתָנַ֣נִי אֲדֹנָ֔י בִּידֵ֖י לֹא־אוּכַ֥ל קֽוּם׃ הִכְבִּיד בְּיָדוֹ אֶת עוֹל עוֹנֶשׁ פְּשָׁעַי, עָלוּ עַל צַוָּארִי כְּעַנְפֵי גֶּפֶן מִתְלַפְּפִים, הִכְשִׁיל כּוֹחִי, ה' שָׂם אוֹתִי בִּידֵי כָּאֵלֶּה שֶׁלֹּא אוּכַל לָקוּם מֵהֶם.
טו סִלָּ֨ה כׇל־אַבִּירַ֤י ׀ אֲדֹנָי֙ בְּקִרְבִּ֔י קָרָ֥א עָלַ֛י מוֹעֵ֖ד לִשְׁבֹּ֣ר בַּחוּרָ֑י גַּ֚ת דָּרַ֣ךְ אֲדֹנָ֔י לִבְתוּלַ֖ת בַּת־יְהוּדָֽה׃ ה' רָמַס אֶת כָּל אַבִּירַי שֶׁבְּקִרְבִּי, הִכְרִיז עַל תַּאֲרִיךְ לְהִתְכַּנְּסוּת הַגּוֹיִם נֶגְדִּי לִשְׁבּוֹר אֶת בַּחוּרַי, ה' דָּרַךְ עַל בְּתוּלַת בַּת יְהוּדָה כְּמוֹ שֶׁדּוֹרְכִים עַל עֲנָבִים בְּגַת.
טז עַל־אֵ֣לֶּה ׀ אֲנִ֣י בוֹכִיָּ֗ה עֵינִ֤י ׀ עֵינִי֙ יֹ֣רְדָה מַּ֔יִם כִּֽי־רָחַ֥ק מִמֶּ֛נִּי מְנַחֵ֖ם מֵשִׁ֣יב נַפְשִׁ֑י הָי֤וּ בָנַי֙ שֽׁוֹמֵמִ֔ים כִּ֥י גָבַ֖ר אוֹיֵֽב׃ עַל אֵלֶּה אֲנִי בּוֹכִיָּה עֵינִי עֵינִי מוֹרִידָה מַיִם, כִּי הִתְרַחֵק מִמֶּנִּי מְנַחֵם מֵשִׁיב נַפְשִׁי, בָּנַי הָיוּ שׁוֹמְמִים כִּי הָאוֹיֵב גָּבַר.
יז פֵּֽרְשָׂ֨ה צִיּ֜וֹן בְּיָדֶ֗יהָ אֵ֤ין מְנַחֵם֙ לָ֔הּ צִוָּ֧ה יְהֹוָ֛ה לְיַעֲקֹ֖ב סְבִיבָ֣יו צָרָ֑יו הָיְתָ֧ה יְרוּשָׁלַ֛͏ִם לְנִדָּ֖ה בֵּינֵיהֶֽם׃ צִיּוֹן פָּרְשָׂה אֶת יָדֶיהָ בִּזְעָקָה, אֵין לָהּ מְנַחֵם, ה' צִוָּה עַל יַעֲקֹב שֶׁצּוֹרְרָיו יַקִּיפוּהוּ סְבִיבָיו, יְרוּשָׁלַיִם הָפְכָה לִשְׂנוּאָה בֵּינֵיהֶם.
יח צַדִּ֥יק ה֛וּא יְהֹוָ֖ה כִּ֣י פִ֣יהוּ מָרִ֑יתִי שִׁמְעוּ־נָ֣א כׇל־הָֽעַמִּ֗ים עמים וּרְאוּ֙ מַכְאֹבִ֔י בְּתוּלֹתַ֥י וּבַחוּרַ֖י הָלְכ֥וּ בַשֶּֽׁבִי׃ צַדִּיק הוּא ה' שֶׁמָּרִיתִי אֶת דְּבָרָיו, שִׁמְעוּ נָא כָּל הָעַמִּים וּרְאוּ אֶת מַכְאוֹבִי, בְּתוּלוֹתַי וּבַחוּרַי הָלְכוּ בַּשֶּׁבִי.
יט קָרָ֤אתִי לַֽמְאַהֲבַי֙ הֵ֣מָּה רִמּ֔וּנִי כֹּהֲנַ֥י וּזְקֵנַ֖י בָּעִ֣יר גָּוָ֑עוּ כִּֽי־בִקְשׁ֥וּ אֹ֙כֶל֙ לָ֔מוֹ וְיָשִׁ֖יבוּ אֶת־נַפְשָֽׁם׃ קָרָאתִי לִמְאַהֲבַי וְהֵם רִמּוּ אוֹתִי, כּוֹהֲנַי וּזְקֵנַי גָּוְעוּ לָמָוֶת בָּעִיר, כִּי חִפְּשׂוּ לְעַצְמָם אוֹכֶל לְהָשִׁיב אֶת נַפְשָׁם, וְלֹא הָיָה לָהֶם.
כ רְאֵ֨ה יְהֹוָ֤ה כִּֽי־צַר־לִי֙ מֵעַ֣י חֳמַרְמָ֔רוּ נֶהְפַּ֤ךְ לִבִּי֙ בְּקִרְבִּ֔י כִּ֥י מָר֖וֹ מָרִ֑יתִי מִח֥וּץ שִׁכְּלָה־חֶ֖רֶב בַּבַּ֥יִת כַּמָּֽוֶת׃ רְאֵה ה' שֶׁצַּר לִי, מֵעַי נִתְכַּוְּצוּ, לִבִּי נֶהְפַּךְ בְּקִרְבִּי כִּי מָרַדְתִּי בְּךָ, מִבַּחוּץ שִׁכְּלָה חֶרֶב וּבַבַּיִת כַּמָּוֶת.
כא שָׁמְע֞וּ כִּ֧י נֶאֱנָחָ֣ה אָ֗נִי אֵ֤ין מְנַחֵם֙ לִ֔י כׇּל־אֹ֨יְבַ֜י שָׁמְע֤וּ רָֽעָתִי֙ שָׂ֔שׂוּ כִּ֥י אַתָּ֖ה עָשִׂ֑יתָ הֵבֵ֥אתָ יוֹם־קָרָ֖אתָ וְיִֽהְי֥וּ כָמֹֽנִי׃ כָּל אוֹיְבַי שָׁמְעוּ שֶׁאֲנִי נֶאֱנַחַת וְאֵין לִי מְנַחֵם, שָׁמְעוּ אֶת רָעָתִי וְשָׂמְחוּ כִּי אַתָּה עָשִׂיתָ, תָּבִיא עֲלֵיהֶם אֶת הַיּוֹם שֶׁקָּרָאתָ, וְיִהְיוּ כָּמוֹנִי.
כב תָּבֹ֨א כׇל־רָעָתָ֤ם לְפָנֶ֙יךָ֙ וְעוֹלֵ֣ל לָ֔מוֹ כַּאֲשֶׁ֥ר עוֹלַ֛לְתָּ לִ֖י עַ֣ל כׇּל־פְּשָׁעָ֑י כִּֽי־רַבּ֥וֹת אַנְחֹתַ֖י וְלִבִּ֥י דַוָּֽי׃ תָּבֹא כָּל רָעָתָם לְפָנֶיךָ וּתְעוֹלֵל לָהֶם כְּמוֹ שֶׁעוֹלַלְתָּ לִי עַל כָּל פְּשָׁעַי, כִּי אַנְחוֹתַי רַבּוֹת וְלִבִּי חַלָּשׁ.
פ קִינָה שְׁנִיָּה
ב א אֵיכָה֩ יָעִ֨יב בְּאַפּ֤וֹ ׀ אֲדֹנָי֙ אֶת־בַּת־צִיּ֔וֹן הִשְׁלִ֤יךְ מִשָּׁמַ֙יִם֙ אֶ֔רֶץ תִּפְאֶ֖רֶת יִשְׂרָאֵ֑ל וְלֹא־זָכַ֥ר הֲדֹם־רַגְלָ֖יו בְּי֥וֹם אַפּֽוֹ׃ אֵיךְ הֶחְשִׁיךְ ה' בְכַעֲסוֹ אֶת מַמְלֶכֶת צִיּוֹן, הִשְׁלִיךְ מִשָּׁמַיִם לָאָרֶץ אֶת תִּפְאֶרֶת יִשְׂרָאֵל, וְלֹא זָכַר בְּיוֹם כַּעֲסוֹ אֶת זְכוּת הָעָם שֶׁהָיָה הַשְּׁרַפְרָף עֲבוּר רַגְלָיו.
ב בִּלַּ֨ע אֲדֹנָ֜י לא וְלֹ֣א חָמַ֗ל אֵ֚ת כׇּל־נְא֣וֹת יַעֲקֹ֔ב הָרַ֧ס בְּעֶבְרָת֛וֹ מִבְצְרֵ֥י בַת־יְהוּדָ֖ה הִגִּ֣יעַ לָאָ֑רֶץ חִלֵּ֥ל מַמְלָכָ֖ה וְשָׂרֶֽיהָ׃ ה' הִשְׁחִית וְלֹא רִחֵם עַל כָּל הַבָּתִים הַיָּפִים שֶׁל יַעֲקֹב, הָרַס בְּכַעֲסוֹ אֶת מִבְצְרֵי קְהִלַּת יְהוּדָה עַד הַיְּסוֹד, בִּטֵּל אֶת מַלְכוּתָהּ וְאֶת שָׂרֶיהָ.
ג גָּדַ֣ע בׇּֽחֳרִי־אַ֗ף כֹּ֚ל קֶ֣רֶן יִשְׂרָאֵ֔ל הֵשִׁ֥יב אָח֛וֹר יְמִינ֖וֹ מִפְּנֵ֣י אוֹיֵ֑ב וַיִּבְעַ֤ר בְּיַעֲקֹב֙ כְּאֵ֣שׁ לֶֽהָבָ֔ה אָכְלָ֖ה סָבִֽיב׃ עָקַר בְּכַעַס אֶת כָּל כְּבוֹד יִשְׂרָאֵל, הֶחֱזִיר אָחוֹרָה אֶת יַד יְמִינוֹ הַמְּגִינָה מֵאוֹיֵב, הוּא בָּעַר בְּיַעֲקֹב כְּמוֹ אֵשׁ לֶהָבָה שֶׁשָּׂרְפָה מִסָּבִיב.
ד דָּרַ֨ךְ קַשְׁתּ֜וֹ כְּאוֹיֵ֗ב נִצָּ֤ב יְמִינוֹ֙ כְּצָ֔ר וַֽיַּהֲרֹ֔ג כֹּ֖ל מַחֲמַדֵּי־עָ֑יִן בְּאֹ֙הֶל֙ בַּת־צִיּ֔וֹן שָׁפַ֥ךְ כָּאֵ֖שׁ חֲמָתֽוֹ׃ דָּרַךְ אֶת קַשְׁתּוֹ כְּמוֹ אוֹיֵב, הֵרִים אֶת יַד יְמִינוֹ כְּמוֹ צוֹרֵר, וְהָרַג אֶת כָּל הָאֲנָשִׁים הַנָּאִים, שָׁפַךְ כָּאֵשׁ אֶת כַּעֲסוֹ בְּמִקְדַּשׁ מַמְלֶכֶת צִיּוֹן.
ה הָיָ֨ה אֲדֹנָ֤י ׀ כְּאוֹיֵב֙ בִּלַּ֣ע יִשְׂרָאֵ֔ל בִּלַּע֙ כׇּל־אַרְמְנוֹתֶ֔יהָ שִׁחֵ֖ת מִבְצָרָ֑יו וַיֶּ֙רֶב֙ בְּבַת־יְהוּדָ֔ה תַּאֲנִיָּ֖ה וַאֲנִיָּֽה׃ ה' הִתְנַהֵג כְּמוֹ אוֹיֵב וְהָרַס אֶת יִשְׂרָאֵל, הָרַס אֶת כָּל אַרְמְנוֹתָיו שִׁחֵת אֶת מִבְצָרָיו, וְהִרְבָּה בְּמַמְלֶכֶת יְהוּדָה אֲבֵילוּת וַאֲנִינוּת.
ו וַיַּחְמֹ֤ס כַּגַּן֙ שֻׂכּ֔וֹ שִׁחֵ֖ת מֹעֲד֑וֹ שִׁכַּ֨ח יְהֹוָ֤ה ׀ בְּצִיּוֹן֙ מוֹעֵ֣ד וְשַׁבָּ֔ת וַיִּנְאַ֥ץ בְּזַֽעַם־אַפּ֖וֹ מֶ֥לֶךְ וְכֹהֵֽן׃ גָּזַם אֶת מִקְדָּשׁוֹ כְּמוֹ שֶׁגּוֹזְמִים גַּן, הִשְׁחִית אֶת מְקוֹם הָעֲלִיָּה לָרֶגֶל, ה' הִשְׁכִּיחַ בְּצִיּוֹן מוֹעֵד וְשַׁבָּת, וְקִלֵּל בְּזַעֲמוֹ מֶלֶךְ וְכֹהֵן.
ז זָנַ֨ח אֲדֹנָ֤י ׀ מִזְבְּחוֹ֙ נִאֵ֣ר מִקְדָּשׁ֔וֹ הִסְגִּיר֙ בְּיַד־אוֹיֵ֔ב חוֹמֹ֖ת אַרְמְנוֹתֶ֑יהָ ק֛וֹל נָתְנ֥וּ בְּבֵית־יְהֹוָ֖ה כְּי֥וֹם מוֹעֵֽד׃ ה' עָזַב אֶת מִזְבְּחוֹ, נָטַשׁ אֶת מִקְדָּשׁוֹ, וְהִסְגִּיר בְּיַד אוֹיֵב אֶת חוֹמוֹת אַרְמְנוֹתָיו, הָאוֹיְבִים הֵרִיעוּ בְּקוֹל בְּבֵית ה' כִּבְיוֹם מוֹעֵד.
ח חָשַׁ֨ב יְהֹוָ֤ה ׀ לְהַשְׁחִית֙ חוֹמַ֣ת בַּת־צִיּ֔וֹן נָ֣טָה קָ֔ו לֹא־הֵשִׁ֥יב יָד֖וֹ מִבַּלֵּ֑עַ וַיַּֽאֲבֶל־חֵ֥ל וְחוֹמָ֖ה יַחְדָּ֥ו אֻמְלָֽלוּ׃ ה' תִּכְנֵן לְהַשְׁחִית אֶת חוֹמַת בַּת צִיּוֹן וּכְבָר מָדַד הֵיכָן, הוּא לֹא יַחֲזִיר אֶת יָדוֹ מִלְּהַשְׁחִית, הוּא יַעֲשֶׂה אֶת הַגָּדֵר וְהַחוֹמָה אֲבֵלִים - בְּבַת אַחַת הֵם יֵעָשׂוּ אֻמְלָלִים.
ט טָבְע֤וּ בָאָ֙רֶץ֙ שְׁעָרֶ֔יהָ אִבַּ֥ד וְשִׁבַּ֖ר בְּרִיחֶ֑יהָ מַלְכָּ֨הּ וְשָׂרֶ֤יהָ בַגּוֹיִם֙ אֵ֣ין תּוֹרָ֔ה גַּם־נְבִיאֶ֕יהָ לֹא־מָצְא֥וּ חָז֖וֹן מֵיְהֹוָֽה׃ אַבְנֵי שְׁעָרֶיהָ שָׁקְעוּ בָּאָרֶץ, ה' אִבֵּד וְשִׁבֵּר אֶת מַנְעוּלֶיהָ, מַלְכָּהּ וְשָׂרֶיהָ נִמְצָאִים בַּגּוֹיִם בְּלֹא יְכוֹלֶת לִלְמוֹד תּוֹרָה, גַּם נְבִיאֶיהָ לֹא זָכוּ לִנְבוּאָה מֵה'.
י יֵשְׁב֨וּ לָאָ֤רֶץ יִדְּמוּ֙ זִקְנֵ֣י בַת־צִיּ֔וֹן הֶֽעֱל֤וּ עָפָר֙ עַל־רֹאשָׁ֔ם חָגְר֖וּ שַׂקִּ֑ים הוֹרִ֤ידוּ לָאָ֙רֶץ֙ רֹאשָׁ֔ן בְּתוּלֹ֖ת יְרוּשָׁלָֽ͏ִם׃ זִקְנֵי בַּת צִיּוֹן יוֹשְׁבִים בַּאֲבֵילוּת עַל הָאָרֶץ בִּשְׁתִיקָה, הֶעֱלוּ עָפָר עַל רֹאשָׁם חָגְרוּ שַׂקִּים, בְּתוּלוֹת יְרוּשָׁלַיִם הוֹרִידוּ לָאָרֶץ אֶת רֹאשָׁן.
יא כָּל֨וּ בַדְּמָע֤וֹת עֵינַי֙ חֳמַרְמְר֣וּ מֵעַ֔י נִשְׁפַּ֤ךְ לָאָ֙רֶץ֙ כְּבֵדִ֔י עַל־שֶׁ֖בֶר בַּת־עַמִּ֑י בֵּֽעָטֵ֤ף עוֹלֵל֙ וְיוֹנֵ֔ק בִּרְחֹב֖וֹת קִרְיָֽה׃ עֵינַי אֵינָן רוֹאוֹת מֵרוֹב דְּמָעוֹת, מֵעַי הִתְכַּוְּצוּ, כְּבֵדִי נִשְׁפַּךְ לָאָרֶץ מֵרוֹב צַּעַר עַל אָסוֹן בַּת עַמִּי, כְּשֶׁיְּלָדִים וְתִנּוֹקוֹת מִתְיַפְּחִים בִּרְחוֹבוֹת עִיר.
יב לְאִמֹּתָם֙ יֹֽאמְר֔וּ אַיֵּ֖ה דָּגָ֣ן וָיָ֑יִן בְּהִֽתְעַטְּפָ֤ם כֶּֽחָלָל֙ בִּרְחֹב֣וֹת עִ֔יר בְּהִשְׁתַּפֵּ֣ךְ נַפְשָׁ֔ם אֶל־חֵ֖יק אִמֹּתָֽם׃ אוֹמְרִים לְאִמָּם: אֵיפֹה יֵשׁ לֶחֶם וְיַיִן? מִתְיַפְּחִים כְּמוֹ חָלָל בִּרְחוֹבוֹת הָעִיר כְּשֶׁנַּפְשָׁם גּוֹוַעַת בְּחֵיק אִמָּם.
יג מָֽה־אֲעִידֵ֞ךְ מָ֣ה אֲדַמֶּה־לָּ֗ךְ הַבַּת֙ יְר֣וּשָׁלַ֔͏ִם מָ֤ה אַשְׁוֶה־לָּךְ֙ וַאֲנַֽחֲמֵ֔ךְ בְּתוּלַ֖ת בַּת־צִיּ֑וֹן כִּֽי־גָד֥וֹל כַּיָּ֛ם שִׁבְרֵ֖ךְ מִ֥י יִרְפָּא־לָֽךְ׃ כֵּיצַד אֲסַפֵּר, לְמָה אַשְׁוֶה אֶת שֶׁקָּרָה לָךְ, מַמְלֶכֶת יְרוּשָׁלַיִם? לְמָה אַשְׁוֶה אֶת שֶׁקָּרָה לָךְ לְנַחֵם אוֹתָךְ מַמְלֶכֶת צִיּוֹן, הֲרֵי אֲסוֹנֵךְ גָּדוֹל כַּיָּם, מִי יְרַפֵּא אוֹתָךְ?
יד נְבִיאַ֗יִךְ חָ֤זוּ לָךְ֙ שָׁ֣וְא וְתָפֵ֔ל וְלֹֽא־גִלּ֥וּ עַל־עֲוֺנֵ֖ךְ לְהָשִׁ֣יב שביתך שְׁבוּתֵ֑ךְ וַיֶּ֣חֱזוּ לָ֔ךְ מַשְׂא֥וֹת שָׁ֖וְא וּמַדּוּחִֽים׃ נְבִיאֵי הַשֶּׁקֶר חָזוּ לָךְ דִּבְרֵי שֶׁקֶר חַסְרֵי טַעַם, וְלֹא גִּלּוּ אֶת עֲוֹנֵךְ לְהַחְזִירֵךְ בִּתְשׁוּבָה, הֵם צָפוּ לָךְ תַּחֲזִיּוֹת שָׁוְא לַהֲסִירֵךְ מִדֶּרֶךְ הַיָּשָׁר.
טו סָֽפְק֨וּ עָלַ֤יִךְ כַּפַּ֙יִם֙ כׇּל־עֹ֣בְרֵי דֶ֔רֶךְ שָֽׁרְקוּ֙ וַיָּנִ֣עוּ רֹאשָׁ֔ם עַל־בַּ֖ת יְרוּשָׁלָ֑͏ִם הֲזֹ֣את הָעִ֗יר שֶׁיֹּֽאמְרוּ֙ כְּלִ֣ילַת יֹ֔פִי מָשׂ֖וֹשׂ לְכׇל־הָאָֽרֶץ׃ מָחֲאוּ כַּפַּיִם עָלַיִךְ בְּצַעַר כָּל עוֹבְרֵי דֶּרֶךְ, שָׁרְקוּ בְּהִשְׁתָּאוּת וְהֵנִיעוּ רֹאשָׁם מִצַּד אֶל צַד עַל מַמְלֶכֶת יְרוּשָׁלַיִם, וְאָמְרוּ: הֲזֹאת הָעִיר שֶׁיֹּאמְרוּ עָלֶיהָ כְּלִילַת יוֹפִי, שִׂמְחָה לְכָל הָאָרֶץ?
טז פָּצ֨וּ עָלַ֤יִךְ פִּיהֶם֙ כׇּל־אֹ֣יְבַ֔יִךְ שָֽׁרְקוּ֙ וַיַּֽחַרְקוּ־שֵׁ֔ן אָמְר֖וּ בִּלָּ֑עְנוּ אַ֣ךְ זֶ֥ה הַיּ֛וֹם שֶׁקִּוִּינֻ֖הוּ מָצָ֥אנוּ רָאִֽינוּ׃ פָּתְחוּ עָלַיִךְ אֶת פִּיהֶם כָּל אוֹיְבַיִךְ, שָׁרְקוּ בְּצַהֲלָה וְשָׁלְפוּ שִׁנַּיִם וְאָמְרוּ: בָּלַעְנוּ אוֹתוֹ; אָכֵן זֶה הַיּוֹם שֶׁקִּוִּינוּ לוֹ, מָצָאנוּ רָאִינוּ!
יז עָשָׂ֨ה יְהֹוָ֜ה אֲשֶׁ֣ר זָמָ֗ם בִּצַּ֤ע אֶמְרָתוֹ֙ אֲשֶׁ֣ר צִוָּ֣ה מִֽימֵי־קֶ֔דֶם הָרַ֖ס וְלֹ֣א חָמָ֑ל וַיְשַׂמַּ֤ח עָלַ֙יִךְ֙ אוֹיֵ֔ב הֵרִ֖ים קֶ֥רֶן צָרָֽיִךְ׃ ה' עָשָׂה אֶת מַה שֶּׁזָּמָם, בִּצֵּעַ אֶת דְּבָרוֹ שֶׁצִּוָּה מִימֵי קֶדֶם, הָרַס וְלֹא חָמַל, שִׂמַּח אֶת אוֹיְבַיִךְ עַל מַפָּלָתֵּךְ, וְהֵרִים אֶת כְּבוֹד עוֹשֵׂי צָרוֹתַיִךְ.
יח צָעַ֥ק לִבָּ֖ם אֶל־אֲדֹנָ֑י חוֹמַ֣ת בַּת־צִ֠יּ֠וֹן הוֹרִ֨ידִי כַנַּ֤חַל דִּמְעָה֙ יוֹמָ֣ם וָלַ֔יְלָה אַֽל־תִּתְּנִ֤י פוּגַת֙ לָ֔ךְ אַל־תִּדֹּ֖ם בַּת־עֵינֵֽךְ׃ לִבָּם שֶׁל יִשְׂרָאֵל צָעַק אֶל ה'; חוֹמַת בַּת צִיּוֹן הוֹרִידִי כַּנַּחַל דִּמְעָה יוֹמָם וָלַיְלָה, אַל יָפוּג צַעֲרֵךְ, אַל יַפְסִיק גַּלְגַּל עֵינֵךְ מִלְּהַזִּיל דִּמְעָה.
יט ק֣וּמִי ׀ רֹ֣נִּי בליל בַלַּ֗יְלָה לְרֹאשׁ֙ אַשְׁמֻר֔וֹת שִׁפְכִ֤י כַמַּ֙יִם֙ לִבֵּ֔ךְ נֹ֖כַח פְּנֵ֣י אֲדֹנָ֑י שְׂאִ֧י אֵלָ֣יו כַּפַּ֗יִךְ עַל־נֶ֙פֶשׁ֙ עֽוֹלָלַ֔יִךְ הָעֲטוּפִ֥ים בְּרָעָ֖ב בְּרֹ֥אשׁ כׇּל־חוּצֽוֹת׃ קוּמִי וְקוֹנְנִי בַּלַּיְלָה בִּתְחִלַּת אַשְׁמוּרוֹת הַבּוֹקֶר, שִׁפְכִי כַּמַּיִם לִבֵּךְ לְנוֹכַח פְּנֵי ה', שְׂאִי אֵלָיו כַּפַּיִךְ בִּתְפִלָּה עַל נֶפֶשׁ יְלָדַיִךְ שֶׁמִּתְיַפְּחִים בְּרָעָב בְּרֹאשׁ כָּל חוּצוֹת.
כ רְאֵ֤ה יְהֹוָה֙ וְֽהַבִּ֔יטָה לְמִ֖י עוֹלַ֣לְתָּ כֹּ֑ה אִם־תֹּאכַ֨לְנָה נָשִׁ֤ים פִּרְיָם֙ עֹלְﬞלֵ֣י טִפֻּחִ֔ים אִם־יֵהָרֵ֛ג בְּמִקְדַּ֥שׁ אֲדֹנָ֖י כֹּהֵ֥ן וְנָבִֽיא׃ רְאֵה ה' וְהַבֵּט לְמִי עוֹלַלְתָּ כָּךְ! הֲאִם רָאוּי שֶׁנָּשִׁים תֹּאכַלְנָה אֶת פִּרְיָם – אֶת יַלְדֵיהֶן הַמְּטֻפָּחִים? הֲאִם יֵהָרֵג בְּמִקְדַּשׁ ה' כֹּהֵן וְנָבִיא?!
כא שָׁכְב֨וּ לָאָ֤רֶץ חוּצוֹת֙ נַ֣עַר וְזָקֵ֔ן בְּתוּלֹתַ֥י וּבַחוּרַ֖י נָפְל֣וּ בֶחָ֑רֶב הָרַ֙גְתָּ֙ בְּי֣וֹם אַפֶּ֔ךָ טָבַ֖חְתָּ לֹ֥א חָמָֽלְתָּ׃ נַעַר וְזָקֵן שָׁכְבוּ עַל הָאָרֶץ בַּחוּצוֹת, בְּתוּלוֹתַי וּבַחוּרַי נָפְלוּ בָּחֶרֶב, הָרַגְתָּ בְּיוֹם כַּעַסְךָ, טָבַחְתָּ וְלֹא רִחַמְתָּ.
כב תִּקְרָא֩ כְי֨וֹם מוֹעֵ֤ד מְגוּרַי֙ מִסָּבִ֔יב וְלֹ֥א הָיָ֛ה בְּי֥וֹם אַף־יְהֹוָ֖ה פָּלִ֣יט וְשָׂרִ֑יד אֲשֶׁר־טִפַּ֥חְתִּי וְרִבִּ֖יתִי אֹיְבִ֥י כִלָּֽם׃ קָרָאתָ לַלּוֹחֲמִים בִּי מִסָּבִיב כְּמוֹ בְּיוֹם מוֹעֵד, וְלֹא הָיָה בְּיוֹם כַּעַס ה' פָּלִיט וְשָׂרִיד, אֶת אֵלּוּ שֶּׁטִּפַּחְתִּי וְגִדַּלְתִּי – אוֹיְבִי הִשְׁמִידָם.
פ קִינָה שְׁלִישִׁית
ג א אֲנִ֤י הַגֶּ֙בֶר֙ רָאָ֣ה עֳנִ֔י בְּשֵׁ֖בֶט עֶבְרָתֽוֹ׃ אֲנִי עַם יִשְׂרָאֵל, רָאִיתִי אֶת עִנּוּיֵי מַקְלוֹ שֶׁל כַּעַס ה'.
ב אוֹתִ֥י נָהַ֛ג וַיֹּלַ֖ךְ חֹ֥שֶׁךְ וְלֹא־אֽוֹר׃ אוֹתִי הִנְהִיג וְהוֹלִיךְ בְּחוֹשֶׁךְ וְלֹא בְּאוֹר.
ג אַ֣ךְ בִּ֥י יָשֻׁ֛ב יַהֲפֹ֥ךְ יָד֖וֹ כׇּל־הַיּֽוֹם׃ רַק בִּי חָזַר בּוֹ ה' וְהָפַך אֶת יָדוֹ לְהַכּוֹתִי כָּל הַיּוֹם.
ד בִּלָּ֤ה בְשָׂרִי֙ וְעוֹרִ֔י שִׁבַּ֖ר עַצְמוֹתָֽי׃ קִלְקֵל אֶת בְּשָׂרִי וְעוֹרִי מֵרוֹב מַכּוֹת, שָׁבַר אֶת עַצְמוֹתַי.
ה בָּנָ֥ה עָלַ֛י וַיַּקַּ֖ף רֹ֥אשׁ וּתְלָאָֽה׃ בָּנָה עָלַי מַעֲרֶכֶת עִנּוּיִם, הִקִּיף אוֹתִי בִּמְרִירוּת וְצָרוֹת.
ו בְּמַחֲשַׁכִּ֥ים הוֹשִׁיבַ֖נִי כְּמֵתֵ֥י עוֹלָֽם׃ הוֹשִׁיב אוֹתִי בַּחוֹשֶׁךְ, כְּמוֹ מֵתִים שֶׁעָבְרוּ מֵהָעוֹלָם.
ז גָּדַ֧ר בַּעֲדִ֛י וְלֹ֥א אֵצֵ֖א הִכְבִּ֥יד נְחׇשְׁתִּֽי׃ שָׂם סְבִיבִי גָּדֵר שֶׁלֹּא אוּכַל לָצֵאת, וְשַׁלְשְׁלוֹת נְחוֹשֶׁת כְּבֵדוֹת.
ח גַּ֣ם כִּ֤י אֶזְעַק֙ וַאֲשַׁוֵּ֔עַ שָׂתַ֖ם תְּפִלָּתִֽי׃ גַּם כְּשֶׁאֶזְעַק וַאֲשַׁוֵּעַ, הוּא סָתַם שַׁעֲרֵי שָׁמַים מִבְּעַד לִתְפִלָּתִי.
ט גָּדַ֤ר דְּרָכַי֙ בְּגָזִ֔ית נְתִיבֹתַ֖י עִוָּֽה׃ שָׂם גֶּדֶר אֲבָנִים מְסֻתָּתוֹת בִּפְנֵי דְּרָכַי, עִקֵּם אֶת הַנְּתִיבִים שֶׁלִּי.
י דֹּ֣ב אֹרֵ֥ב הוּא֙ לִ֔י אריה אֲרִ֖י בְּמִסְתָּרִֽים׃ הוּא אוֹרֵב לִי כְּמוֹ דֹּב, כְּמוֹ אַרְיֵה שֶׁמַּמְתִּין לִי בְּמִסְתּוֹר.
יא דְּרָכַ֥י סוֹרֵ֛ר וַֽיְפַשְּׁחֵ֖נִי שָׂמַ֥נִי שֹׁמֵֽם׃ מִלֵּא בְּקוֹצִים אֶת דְּרָכַי וַאֲנִי קוֹפֵץ לְהַקִּיפָם, שָׂם אוֹתִי שׁוֹמֵם.
יב דָּרַ֤ךְ קַשְׁתּוֹ֙ וַיַּצִּיבֵ֔נִי כַּמַּטָּרָ֖א לַחֵֽץ׃ דָּרַךְ קַשְׁתּוֹ וְהִצִּיב אוֹתִי כַּמַּטָּרָה לַחֵץ.
יג הֵבִיא֙ בְּכִלְיֹתָ֔י בְּנֵ֖י אַשְׁפָּתֽוֹ׃ יָרָה בְּכִלְיוֹתַי אֶת הַחִצִּים שֶׁבְּאַשְׁפָּתוֹ.
יד הָיִ֤יתִי שְּׂחֹק֙ לְכׇל־עַמִּ֔י נְגִינָתָ֖ם כׇּל־הַיּֽוֹם׃ הָיִיתִי שְׂחוֹק לְכָל עַמִּי, הָיִיתִי נוֹשֵׂא לְשִׁירֵיהֶם כָּל הַיּוֹם.
טו הִשְׂבִּיעַ֥נִי בַמְּרוֹרִ֖ים הִרְוַ֥נִי לַעֲנָֽה׃ הִשְׂבִּיעַ אוֹתִי בַּמְּרוֹרִים, הִרְוָה אוֹתִי בְּרַעַל.
טז וַיַּגְרֵ֤ס בֶּֽחָצָץ֙ שִׁנָּ֔י הִכְפִּישַׁ֖נִי בָּאֵֽפֶר׃ גָּרַם לִי לִלְעוֹס חָצָץ בְּשִׁנַּי, שָׁטַח אוֹתִי בָּאֵפֶר.
יז וַתִּזְנַ֧ח מִשָּׁל֛וֹם נַפְשִׁ֖י נָשִׁ֥יתִי טוֹבָֽה׃ הַשָּׁלוֹם זָנַח אֶת נַפְשִׁי, הַטּוֹבָה נֶעֶקְרָה מִמֶּנִּי.
יח וָאֹמַר֙ אָבַ֣ד נִצְחִ֔י וְתוֹחַלְתִּ֖י מֵיְהֹוָֽה׃ אָמַרְתִּי בְּצַעֲרִי אָבַד קִיּוּמִי לָנֶצַח, אָבְדָה תִּקְוָתִי מֵה'.
יט זְכׇר־עׇנְיִ֥י וּמְרוּדִ֖י לַעֲנָ֥ה וָרֹֽאשׁ׃ זְכוֹר אֶת שִׁפְלוּתִי וְאֶת צַעֲרִי – רַעַל וּמְרִירוּת.
כ זָכ֣וֹר תִּזְכּ֔וֹר ותשיח וְתָשׁ֥וֹחַ עָלַ֖י נַפְשִֽׁי׃ זְכוֹר זֹאת, נַפְשִׁי חָשָׁה מֻשְׁפֶּלֶת.
כא זֹ֛את אָשִׁ֥יב אֶל־לִבִּ֖י עַל־כֵּ֥ן אוֹחִֽיל׃ אֲבָל אֶת זֹאת אָשִׁיב אֶל לִבִּי לָכֵן אֲנִי מְקַוֶּה.
כב חַֽסְדֵ֤י יְהֹוָה֙ כִּ֣י לֹא־תָ֔מְנוּ כִּ֥י לֹא־כָל֖וּ רַחֲמָֽיו׃ שֶׁחַסְדֵי ה' לֹא תַּמּוּ – שֶׁלֹּא נִגְמְרוּ רַחֲמָיו.
כג חֲדָשִׁים֙ לַבְּקָרִ֔ים רַבָּ֖ה אֱמוּנָתֶֽךָ׃ חֲסָדִים חֲדָשִׁים בְּכָל בּוֹקֶר, גְּדוֹלָה הָאֱמוּנָה בְּךָ.
כד חֶלְקִ֤י יְהֹוָה֙ אָמְרָ֣ה נַפְשִׁ֔י עַל־כֵּ֖ן אוֹחִ֥יל לֽוֹ׃ אָמַרְתִּי: יְהֵא חֶלְקִי עִם ה', עַל כֵּן אֲקַוֶּה לוֹ.
כה ט֤וֹב יְהֹוָה֙ לְקֹוָ֔ו לְנֶ֖פֶשׁ תִּדְרְשֶֽׁנּוּ׃ טוֹב ה' לְאֵלּוּ שֶׁמְּקַוִּים לוֹ, לְנֶפֶשׁ שֶׁדּוֹרֶשֶׁת אוֹתוֹ.
כו ט֤וֹב וְיָחִיל֙ וְדוּמָ֔ם לִתְשׁוּעַ֖ת יְהֹוָֽה׃ טוֹב הוּא לַמְּקַוֶּה בִּדְמָמָה לִתְשׁוּעַת ה'.
כז ט֣וֹב לַגֶּ֔בֶר כִּֽי־יִשָּׂ֥א עֹ֖ל בִּנְעוּרָֽיו׃ טוֹב לַגֶּבֶר שֶׁיְּקַבֵּל עָלָיו עוֹל בִּנְעוּרָיו.
כח יֵשֵׁ֤ב בָּדָד֙ וְיִדֹּ֔ם כִּ֥י נָטַ֖ל עָלָֽיו׃ יֵשֵׁב לְבָד בִּדְמָמָה כִּי קִבֵּל עָלָיו אֶת עוֹל הַיִּסּוּרִין.
כט יִתֵּ֤ן בֶּֽעָפָר֙ פִּ֔יהוּ אוּלַ֖י יֵ֥שׁ תִּקְוָֽה׃ יַשְׁפִּיל אֶת פִּיו בֶּעָפָר אוּלַי יֵשׁ תִּקְוָה.
ל יִתֵּ֧ן לְמַכֵּ֛הוּ לֶ֖חִי יִשְׂבַּ֥ע בְּחֶרְפָּֽה׃ יַגִּישׁ לֶחִי לְמַכֵּהוּ, יִשְׂבַּע צָרוֹת בְּחֶרְפָּה.
לא כִּ֣י לֹ֥א יִזְנַ֛ח לְעוֹלָ֖ם אֲדֹנָֽי׃ כִּי ה' לֹא יַזְנִיחַ אוֹתוֹ לְעוֹלָם.
לב כִּ֣י אִם־הוֹגָ֔ה וְרִחַ֖ם כְּרֹ֥ב חֲסָדָֽיו׃ כִּי אִם אָדָם נִמְצָא עַכְשָׁיו בְּיָגוֹן, ה' יְרַחֵם עָלָיו כְּרוֹב חֲסָדָיו.
לג כִּ֣י לֹ֤א עִנָּה֙ מִלִּבּ֔וֹ וַיַּגֶּ֖ה בְּנֵי־אִֽישׁ׃ כִּי לֹא מִתּוֹךְ שְׁרִירוּת לֵב ה' עִנָּה וְהֵבִיא יָגוֹן לִבְנֵי אָדָם.
לד לְדַכֵּא֙ תַּ֣חַת רַגְלָ֔יו כֹּ֖ל אֲסִ֥ירֵי אָֽרֶץ׃ לֹא סְתָם הֶחְלִיט לְדַכֵּא תַּחַת רַגְלָיו אֶת כָּל הָאָרֶץ כַּאֲסִירִים.
לה לְהַטּוֹת֙ מִשְׁפַּט־גָּ֔בֶר נֶ֖גֶד פְּנֵ֥י עֶלְיֽוֹן׃ אֵין לִפְנֵי עֶלְיוֹן כַּוָּנָה לְהַטּוֹת מִשְׁפַּט גֶּבֶר.
לו לְעַוֵּ֤ת אָדָם֙ בְּרִיב֔וֹ אֲדֹנָ֖י לֹ֥א רָאָֽה׃ לְעַוֵּת מִשְׁפַּט אָדָם בְּטַעֲנָתוֹ, כְּאִלּוּ ה' לֹא רָאָה.
לז מִ֣י זֶ֤ה אָמַר֙ וַתֶּ֔הִי אֲדֹנָ֖י לֹ֥א צִוָּֽה׃ מִי הוּא זֶה שֶׁאָמַר שֶׁיִהְיֶה דָּבָר וְהוּא אָכֵן הָיָה, כְּשֶׁה' לֹא צִוָּה?!
לח מִפִּ֤י עֶלְיוֹן֙ לֹ֣א תֵצֵ֔א הָרָע֖וֹת וְהַטּֽוֹב׃ מִפִּי עֶלְיוֹן לֹא תֵּצֵא הַחְלָטָה אִם יַעֲשֶׂה אָדָם רָעוֹת אוֹ טוֹב.
לט מַה־יִּתְאוֹנֵן֙ אָדָ֣ם חָ֔י גֶּ֖בֶר עַל־חֲטָאָֽו׃ מַה יִּתְלוֹנֵן אָדָם חַי – גֶּבֶר עַל תּוֹצְאוֹת חֲטָאָיו?
מ נַחְפְּשָׂ֤ה דְרָכֵ֙ינוּ֙ וְֽנַחְקֹ֔רָה וְנָשׁ֖וּבָה עַד־יְהֹוָֽה׃ נְחַפֵּשׂ אֵיפֹה טָעִינוּ בִּדְרָכֵינוּ וְנַחְקוֹר, וְנָשׁוּב אֶל ה'.
מא נִשָּׂ֤א לְבָבֵ֙נוּ֙ אֶל־כַּפָּ֔יִם אֶל־אֵ֖ל בַּשָּׁמָֽיִם׃ נִשָּׂא אֶת לְבָבֵנוּ בִּתְפִלָּה לָאֵל שֶׁבַּשָּׁמַיִם.
מב נַ֤חְנוּ פָשַׁ֙עְנוּ֙ וּמָרִ֔ינוּ אַתָּ֖ה לֹ֥א סָלָֽחְתָּ׃ אֲנַחְנוּ פָּשַׁעְנוּ וּמָרַדְנוּ, אַתָּה לֹא סָלַחְתָּ.
מג סַכּ֤וֹתָה בָאַף֙ וַֽתִּרְדְּפֵ֔נוּ הָרַ֖גְתָּ לֹ֥א חָמָֽלְתָּ׃ כִּסִּיתָ אוֹתָנוּ בְּכַעַס וְרָדַפְתָּ אוֹתָנוּ, הָרַגְתָּ וְלֹא רִחַמְתָּ.
מד סַכּ֤וֹתָה בֶֽעָנָן֙ לָ֔ךְ מֵעֲב֖וֹר תְּפִלָּֽה׃ כִּסִּיתָ אֶת עַצְמְךָ בֶּעָנָן כְּדֵי שֶׁלֹּא תַּעֲבוֹר אֵלֶיךָ תְּפִלָּה.
מה סְחִ֧י וּמָא֛וֹס תְּשִׂימֵ֖נוּ בְּקֶ֥רֶב הָעַמִּֽים׃ שַׂמְתָּ אוֹתָנוּ כְּגוֹלִים וּמְאוּסִים בֵּין הָעַמִּים.
מו פָּצ֥וּ עָלֵ֛ינוּ פִּיהֶ֖ם כׇּל־אֹיְבֵֽינוּ׃ כָּל אוֹיְבֵינוּ פָּתְחוּ עָלֵינוּ אֶת פִּיהֶם.
מז פַּ֧חַד וָפַ֛חַת הָ֥יָה לָ֖נוּ הַשֵּׁ֥את וְהַשָּֽׁבֶר׃ הָיָה לָנוּ פַּחַד וּנְפִילָה, שׁוֹאָה וְשֶׁבֶר.
מח פַּלְגֵי־מַ֙יִם֙ תֵּרַ֣ד עֵינִ֔י עַל־שֶׁ֖בֶר בַּת־עַמִּֽי׃ נְהָרוֹת מַיִם יֵרְדוּ מֵעֵינִי, עַל שֶׁבֶר בַּת עַמִּי.
מט עֵינִ֧י נִגְּרָ֛ה וְלֹ֥א תִדְמֶ֖ה מֵאֵ֥ין הֲפֻגֽוֹת׃ עֵינִי זָלְגָה דְּמָעוֹת וְהִיא לֹא תֶּחְדַּל, לְלֹא הַפְסָקוֹת.
נ עַד־יַשְׁקִ֣יף וְיֵ֔רֶא יְהֹוָ֖ה מִשָּׁמָֽיִם׃ עַד שֶׁה' יַשְׁקִיף וְיִרְאֶה מֵהַשָּׁמַיִם אֶת צַעֲרִי.
נא עֵינִי֙ עֽוֹלְלָ֣ה לְנַפְשִׁ֔י מִכֹּ֖ל בְּנ֥וֹת עִירִֽי׃ עֵינִי הִזִּילָה דִּמְעוֹת צַעַר יוֹתֵר מִכָּל בְּנוֹת עִירִי.
נב צ֥וֹד צָד֛וּנִי כַּצִּפּ֖וֹר אֹיְבַ֥י חִנָּֽם׃ אוֹיְבַי צָדוּ אוֹתִי כְּצִפּוֹר לַשָּׁוְא.
נג צָֽמְת֤וּ בַבּוֹר֙ חַיָּ֔י וַיַּדּוּ־אֶ֖בֶן בִּֽי׃ אָסְרוּ אוֹתִי בְּבוֹר וְיִדּוּ בִּי אֲבָנִים.
נד צָֽפוּ־מַ֥יִם עַל־רֹאשִׁ֖י אָמַ֥רְתִּי נִגְזָֽרְתִּי׃ הֵצִיפוּ מַיִם עַל רֹאשִׁי, חָשַׁבְתִּי שֶׁזֶּה סוֹפִי.
נה קָרָ֤אתִי שִׁמְךָ֙ יְהֹוָ֔ה מִבּ֖וֹר תַּחְתִּיּֽוֹת׃ קָרָאתִי שִׁמְךָ ה' מִבּוֹר תַּחְתִּיּוֹת.
נו קוֹלִ֖י שָׁמָ֑עְתָּ אַל־תַּעְלֵ֧ם אׇזְנְךָ֛ לְרַוְחָתִ֖י לְשַׁוְעָתִֽי׃ קוֹלִי שָׁמַעְתָּ, אַל תַּעְלֵם אֶת אָזְנְךָ מִשַּׁוְעָתִי לַעֲשׂוֹת לִי רְוָחָה.
נז קָרַ֙בְתָּ֙ בְּי֣וֹם אֶקְרָאֶ֔ךָּ אָמַ֖רְתָּ אַל־תִּירָֽא׃ קָרַבְתָּ אֵלַי בְּיוֹם שֶׁקָּרָאתִי לְךָ, אָמַרְתָּ: אַל תִּפְחַד.
נח רַ֧בְתָּ אֲדֹנָ֛י רִיבֵ֥י נַפְשִׁ֖י גָּאַ֥לְתָּ חַיָּֽי׃ נִלְחַמְתָּ ה' אֶת מִלְחֲמוֹתַי, גָּאַלְתָּ אֶת חַיַּי.
נט רָאִ֤יתָה יְהֹוָה֙ עַוָּ֣תָתִ֔י שׇׁפְטָ֖ה מִשְׁפָּטִֽי׃ רָאִיתָה ה' אֶת הָעִוּוּת שֶׁנַּעֲשָׂה לִי, שְׁפוֹט אֶת מִשְׁפָּטִי.
ס רָאִ֙יתָה֙ כׇּל־נִקְמָתָ֔ם כׇּל־מַחְשְׁבֹתָ֖ם לִֽי׃ רָאִיתָה אֶת כָּל מַעֲשֵׂיהֶם שֶׁתִּנְקוֹם, אֶת כָּל מַחְשְׁבוֹתֵיהֶם נֶגְדִּי.
סא שָׁמַ֤עְתָּ חֶרְפָּתָם֙ יְהֹוָ֔ה כׇּל־מַחְשְׁבֹתָ֖ם עָלָֽי׃ שָׁמַעְתָּ אֶת עֶלְבְּנוֹתֵיהֶם ה', אֶת כָּל מַחְשְׁבוֹתֵיהֶם עָלַי.
סב שִׂפְתֵ֤י קָמַי֙ וְהֶגְיוֹנָ֔ם עָלַ֖י כׇּל־הַיּֽוֹם׃ – אֶת דִּבּוּרֵי הַקָּמִים עָלַי וּמְזִמּוֹתֵיהֶם לְרָעָתִי כָּל הַיּוֹם.
סג שִׁבְתָּ֤ם וְקִֽימָתָם֙ הַבִּ֔יטָה אֲנִ֖י מַנְגִּינָתָֽם׃ הַבֵּט, בִּזְמַן שִׁבְתָּם וְקִימָתָם אֲנִי נוֹשֵׂא שִׂיחָתָם.
סד תָּשִׁ֨יב לָהֶ֥ם גְּמ֛וּל יְהֹוָ֖ה כְּמַעֲשֵׂ֥ה יְדֵיהֶֽם׃ ה', תָּשִׁיב לָהֶם גְּמוּל כְּמַעֲשֵׂה יְדֵיהֶם.
סה תִּתֵּ֤ן לָהֶם֙ מְגִנַּת־לֵ֔ב תַּאֲלָֽתְךָ֖ לָהֶֽם׃ תִּתֵּן לָהֶם שִׁבְרוֹן לֵב, אֶת קִלְלָתְךָ עֲלֵיהֶם.
סו תִּרְדֹּ֤ף בְּאַף֙ וְתַשְׁמִידֵ֔ם מִתַּ֖חַת שְׁמֵ֥י יְהֹוָֽה׃ תִּרְדּוֹף אוֹתָם בְּכַעַס וְתַשְׁמִיד אוֹתָם, מִתַּחַת שְׁמֵי ה'.
פ קִינָה רְבִיעִית
ד א אֵיכָה֙ יוּעַ֣ם זָהָ֔ב יִשְׁנֶ֖א הַכֶּ֣תֶם הַטּ֑וֹב תִּשְׁתַּפֵּ֙כְנָה֙ אַבְנֵי־קֹ֔דֶשׁ בְּרֹ֖אשׁ כׇּל־חוּצֽוֹת׃ אֵיךְ הִתְעַמְעֵם בְּרַק הַזָּהָב, נִשְׁתַּנָּה צֶבַע הַתַּכְשִׁיט הַיָּפֶה, נִתְפַּזְּרוּ אַבְנֵי קוֹדֶשׁ בְּרֹאשׁ כָּל חוּצוֹת.
ב בְּנֵ֤י צִיּוֹן֙ הַיְקָרִ֔ים הַמְסֻלָּאִ֖ים בַּפָּ֑ז אֵיכָ֤ה נֶחְשְׁבוּ֙ לְנִבְלֵי־חֶ֔רֶשׂ מַעֲשֵׂ֖ה יְדֵ֥י יוֹצֵֽר׃ בְּנֵי צִיּוֹן הַיְּקָרִים הַשָּׁוִים זָהָב, אֵיךְ נֶחְשְׁבוּ כְּכַדֵּי חֶרֶס שֶׁנִּשְׁבָּר, מַעֲשֵׂה יְדֵי יוֹצֵר.
ג גַּם־תַּנִּים֙ תנין חָ֣לְצוּ שַׁ֔ד הֵינִ֖יקוּ גּוּרֵיהֶ֑ן בַּת־עַמִּ֣י לְאַכְזָ֔ר כי ענים [כַּיְעֵינִ֖ים] בַּמִּדְבָּֽר׃ אֲפִלּוּ תַּנִּים חָלְצוּ שַׁד וְהֵינִיקוּ אֶת גּוּרֵיהֶן, אַךְ בַּת עַמִּי הָפְכָה לְאַכְזָר כַּיְעֵנִים שֶׁבַּמִּדְבָּר.
ד דָּבַ֨ק לְשׁ֥וֹן יוֹנֵ֛ק אֶל־חִכּ֖וֹ בַּצָּמָ֑א עֽוֹלָלִים֙ שָׁ֣אֲלוּ לֶ֔חֶם פֹּרֵ֖שׂ אֵ֥ין לָהֶֽם׃ לְשׁוֹן יוֹנֵק דָּבְקָה אֶל חִכּוֹ בַּצָּמָא, יְלָדִים רַכִּים בִּקְשׁוּ לֶחֶם, וְאֵין מִי שֶׁיִּפְרוֹס לָהֶם.
ה הָאֹֽכְלִים֙ לְמַ֣עֲדַנִּ֔ים נָשַׁ֖מּוּ בַּחוּצ֑וֹת הָאֱמֻנִים֙ עֲלֵ֣י תוֹלָ֔ע חִבְּק֖וּ אַשְׁפַּתּֽוֹת׃ אֵלּוּ שֶׁהָיוּ אוֹכְלִים מַעֲדַנִּים הָפְכוּ לִשְׁמָמָה בַּחוּצוֹת, הָרְגִילִים בְּבִגְדֵי צֶבַע יָקָר נִצְמְדוּ לְאַשְׁפַּתּוֹת.
ו וַיִּגְדַּל֙ עֲוֺ֣ן בַּת־עַמִּ֔י מֵֽחַטַּ֖את סְדֹ֑ם הַֽהֲפוּכָ֣ה כְמוֹ־רָ֔גַע וְלֹא־חָ֥לוּ בָ֖הּ יָדָֽיִם׃ גָּדַל עֲוֹן בַּת עַמִּי יוֹתֵר מֵחַטַּאת סְדוֹם – הָעִיר שֶׁנֶּהֶפְכָה בְּרֶגַע אַךְ יְדֵי זָרִים לֹא שָׁלְטוּ בָּהּ.
ז זַכּ֤וּ נְזִירֶ֙יהָ֙ מִשֶּׁ֔לֶג צַח֖וּ מֵחָלָ֑ב אָ֤דְמוּ עֶ֙צֶם֙ מִפְּנִינִ֔ים סַפִּ֖יר גִּזְרָתָֽם׃ נְזִירֶיהָ שֶׁהָיוּ יוֹתֵר זַכִּים מִשֶּׁלֶג וְצַחִים מֵחָלָב, נַעֲשׂוּ אֲדֻמִּים יוֹתֵר מִפְּנִינִים בְּצוּרַת אֶבֶן סַפִּיר.
ח חָשַׁ֤ךְ מִשְּׁחוֹר֙ תׇּֽאֳרָ֔ם לֹ֥א נִכְּר֖וּ בַּחוּצ֑וֹת צָפַ֤ד עוֹרָם֙ עַל־עַצְמָ֔ם יָבֵ֖שׁ הָיָ֥ה כָעֵֽץ׃ חָשַׁךְ מִשְּׁחוֹר מַרְאָם, עַד שֶׁלֹּא הִכִּירוּ אוֹתָם בַּחוּצוֹת, נִדְבַּק עוֹרָם עַל עַצְמוֹתָם, נַעֲשָׂה יָבֵשׁ כָּעֵץ.
ט טוֹבִ֤ים הָיוּ֙ חַלְלֵי־חֶ֔רֶב מֵֽחַלְלֵ֖י רָעָ֑ב שֶׁ֣הֵ֤ם יָזֻ֙בוּ֙ מְדֻקָּרִ֔ים מִתְּנוּבֹ֖ת שָׂדָֽי׃ עֲדִיפִים הָיוּ חַלְלֵי חֶרֶב מֵחַלְלֵי רָעָב, שֶׁהֵם הָיוּ דְּקוּרִים וְזָבֵי דָּם, אַךְ בִּטְנָם מְלֵאָה מִתְּנוּבוֹת שָׂדֶה.
י יְדֵ֗י נָשִׁים֙ רַחֲמָ֣נִיּ֔וֹת בִּשְּׁל֖וּ יַלְדֵיהֶ֑ן הָי֤וּ לְבָרוֹת֙ לָ֔מוֹ בְּשֶׁ֖בֶר בַּת־עַמִּֽי׃ יְדֵי נָשִׁים רַחֲמָנִיּוֹת בִּשְּׁלוּ אֶת יַלְדֵיהֶן, הָיוּ לָהֶם לְמַאֲכָל בִּגְלַל הַשֶּׁבֶר שֶׁל בַּת עַמִּי.
יא כִּלָּ֤ה יְהֹוָה֙ אֶת־חֲמָת֔וֹ שָׁפַ֖ךְ חֲר֣וֹן אַפּ֑וֹ וַיַּצֶּת־אֵ֣שׁ בְּצִיּ֔וֹן וַתֹּ֖אכַל יְסֹדֹתֶֽיהָ׃ ה' כִּלָּה בָּהֶם אֶת חֲמָתוֹ וְשָׁפַךְ אֶת חֲרוֹן אַפּוֹ, הוּא הִצִּית אֵשׁ בְּצִיּוֹן שֶׁשָׂרְפָה אֶת יְסוֹדוֹתֶיהָ.
יב לֹ֤א הֶאֱמִ֙ינוּ֙ מַלְכֵי־אֶ֔רֶץ וכל כֹּ֖ל יֹשְׁבֵ֣י תֵבֵ֑ל כִּ֤י יָבֹא֙ צַ֣ר וְאוֹיֵ֔ב בְּשַׁעֲרֵ֖י יְרוּשָׁלָֽ͏ִם׃ מַלְכֵי אֶרֶץ וְכָל יוֹשְׁבֵי תֵבֵל לֹא הֶאֱמִינוּ שֶׁיָּבוֹא צוֹרֵר וְאוֹיֵב בְּשַׁעֲרֵי יְרוּשָׁלַיִם,
יג מֵֽחַטֹּ֣אות נְבִיאֶ֔יהָ עֲוֺנֹ֖ת כֹּהֲנֶ֑יהָ הַשֹּׁפְכִ֥ים בְּקִרְבָּ֖הּ דַּ֥ם צַדִּיקִֽים׃ בִּגְלַל חַטֹּאת נְבִיאֶיהָ וַעֲוֹנוֹת כּוֹהֲנֶיהָ, שֶׁשּׁוֹפְכִים בְּתוֹכָהּ דַּם צַדִּיקִים.
יד נָע֤וּ עִוְרִים֙ בַּֽחוּצ֔וֹת נְגֹֽאֲל֖וּ בַּדָּ֑ם בְּלֹ֣א יֽוּכְל֔וּ יִגְּע֖וּ בִּלְבֻשֵׁיהֶֽם׃ עִוְרִים הִסְתּוֹבְבוּ בַּחוּצוֹת מְגוֹאֲלִים בְּדָם, עַד שֶׁלֹּא נִתָּן הָיָה לִנְגּוֹעַ בִּלְבוּשֵׁיהֶם.
טו ס֣וּרוּ טָמֵ֞א קָ֣רְאוּ לָ֗מוֹ ס֤וּרוּ ס֙וּרוּ֙ אַל־תִּגָּ֔עוּ כִּ֥י נָצ֖וּ גַּם־נָ֑עוּ אָֽמְרוּ֙ בַּגּוֹיִ֔ם לֹ֥א יוֹסִ֖פוּ לָגֽוּר׃ קָרְאוּ לָהֶם: 'סוּרוּ טָמֵא!', 'סוּרוּ סוּרוּ אַל תִּגְּעוּ!', הָיוּ רָבִים עִמָּם וְזָזִים בְּחוֹסֶר נוֹחוּת, אָמְרוּ בַּגּוֹיִם: שֶׁלֹּא יוֹסִיפוּ אֵלּוּ לָגוּר אִתָּנוּ.
טז פְּנֵ֤י יְהֹוָה֙ חִלְּקָ֔ם לֹ֥א יוֹסִ֖יף לְהַבִּיטָ֑ם פְּנֵ֤י כֹהֲנִים֙ לֹ֣א נָשָׂ֔אוּ זקנים וּזְקֵנִ֖ים לֹ֥א חָנָֽנוּ׃ ה' הִפְרִיד אוֹתָם מֵעִם פָּנָיו וְלֹא יוֹסִיף לְהַבִּיט בָּהֵם, כִּי לֹא כִּבְּדוּ פְּנֵי כּוֹהֲנִים, וְלֹא רִחֲמוּ עַל זְקֵנִים.
יז עודינה עוֹדֵ֙ינוּ֙ תִּכְלֶ֣ינָה עֵינֵ֔ינוּ אֶל־עֶזְרָתֵ֖נוּ הָ֑בֶל בְּצִפִּיָּתֵ֣נוּ צִפִּ֔ינוּ אֶל־גּ֖וֹי לֹ֥א יוֹשִֽׁעַ׃ בְּעוֹד שֶׁעֵינֵינוּ הָיוּ מִשְׁתּוֹקְקוֹת לְעֶזְרַת מִי שֶׁאֵינוֹ מוֹעִיל – בְּצִפִּיָּתֵנוּ צִפִּינוּ לְעָם שֶׁאֵינוֹ מוֹשִׁיעַ,
יח צָד֣וּ צְעָדֵ֔ינוּ מִלֶּ֖כֶת בִּרְחֹבֹתֵ֑ינוּ קָרַ֥ב קִצֵּ֛נוּ מָלְא֥וּ יָמֵ֖ינוּ כִּי־בָ֥א קִצֵּֽנוּ׃ הֵצֵרוּ אֶת צְעָדֵינוּ מִלָּלֶכֶת בִּרְחוֹבוֹתֵינוּ, קִצֵּינוּ קָרַב, יָמֵינוּ נִתְמַלְּאוּ, כִּי קִצֵּינוּ בָּא.
יט קַלִּ֤ים הָיוּ֙ רֹדְפֵ֔ינוּ מִנִּשְׁרֵ֖י שָׁמָ֑יִם עַל־הֶהָרִ֣ים דְּלָקֻ֔נוּ בַּמִּדְבָּ֖ר אָ֥רְבוּ לָֽנוּ׃ רוֹדְפֵינוּ הָיוּ קַלִּים יוֹתֵר מִנִּשְׁרֵי שָׁמַיִם, רָדְפוּ אַחֲרֵינוּ עַל הֶהָרִים, אָרְבוּ לָנוּ בַּמִּדְבָּר.
כ ר֤וּחַ אַפֵּ֙ינוּ֙ מְשִׁ֣יחַ יְהֹוָ֔ה נִלְכַּ֖ד בִּשְׁחִיתוֹתָ֑ם אֲשֶׁ֣ר אָמַ֔רְנוּ בְּצִלּ֖וֹ נִֽחְיֶ֥ה בַגּוֹיִֽם׃ מַלְכֵּנוּ – רוּחַ אַפֵּינוּ, שֶׁנִּמְשַׁח בְּשֶׁמֶן לִכְבוֹד ה', נִלְכַּד בַּשּׁוּחוֹת שֶׁחָפְרוּ, אוֹתוֹ שֶׁאָמַרְנוּ: בְּצִלּוֹ שֶׁל הַמֶּלֶךְ נִחְיֶה בַּגּוֹיִם כְּמוֹתָם.
כא שִׂ֤ישִׂי וְשִׂמְחִי֙ בַּת־אֱד֔וֹם יושבתי יוֹשֶׁ֖בֶת בְּאֶ֣רֶץ ע֑וּץ גַּם־עָלַ֙יִךְ֙ תַּעֲבׇר־כּ֔וֹס תִּשְׁכְּרִ֖י וְתִתְעָרִֽי׃ שִׂישִׂי וְשִׂמְחִי בַּת אֱדוֹם שֶׁיּוֹשֶׁבֶת בְּאֶרֶץ אֲרָם, גַּם עָלַיִךְ תַּעֲבוֹר כּוֹס תַּרְעֵלָה, תִּשְׁתַּכְּרִי מִמֶּנּוּ וְתָקִיאִי.
כב תַּם־עֲוֺנֵךְ֙ בַּת־צִיּ֔וֹן לֹ֥א יוֹסִ֖יף לְהַגְלוֹתֵ֑ךְ פָּקַ֤ד עֲוֺנֵךְ֙ בַּת־אֱד֔וֹם גִּלָּ֖ה עַל־חַטֹּאתָֽיִךְ׃ תַּם עוֹנֶשׁ עֲוֹנֵךְ בַּת צִיּוֹן, ה' לֹא יוֹסִיף לְהַגְלוֹתֵךְ; ה' זָכַר אֶת עֲוֹנֵךְ בַּת אֱדוֹם, הוּא יַעֲנִישֵׁךְ לְעֵינֵי כֹּל עַל הַחַטָּאִים שֶׁלָּךְ.
פ קִינָה חֲמִישִׁית
ה א זְכֹ֤ר יְהֹוָה֙ מֶֽה־הָ֣יָה לָ֔נוּ הביט הַבִּ֖יטָה וּרְאֵ֥ה אֶת־חֶרְפָּתֵֽנוּ׃ זְכוֹר ה' מַה קָּרָה לָנוּ, הַבֵּט וּרְאֵה אֶת בּוּשָׁתֵנוּ.
ב נַחֲלָתֵ֙נוּ֙ נֶֽהֶפְכָ֣ה לְזָרִ֔ים בָּתֵּ֖ינוּ לְנׇכְרִֽים׃ נַחֲלָתֵנוּ הָפְכָה לִרְכוּשׁ זָרִים, בָּתֵּינוּ לְנָכְרִים.
ג יְתוֹמִ֤ים הָיִ֙ינוּ֙ אין וְאֵ֣ין אָ֔ב אִמֹּתֵ֖ינוּ כְּאַלְמָנֽוֹת׃ הָפַכְנוּ לִיתוֹמִים לְלֹא אָב, אִמּוֹתֵינוּ כְּמוֹ אַלְמָנוֹת.
ד מֵימֵ֙ינוּ֙ בְּכֶ֣סֶף שָׁתִ֔ינוּ עֵצֵ֖ינוּ בִּמְחִ֥יר יָבֹֽאוּ׃ שָׁתִינוּ אֶת מֵימֵינוּ תְּמוּרַת כֶּסֶף, אֶת עֵצֵינוּ נְקַבֵּל רַק בִּמְחִיר מָלֵא.
ה עַ֤ל צַוָּארֵ֙נוּ֙ נִרְדָּ֔פְנוּ יָגַ֖עְנוּ לא וְלֹ֥א הֽוּנַֽח־לָֽנוּ׃ נִרְדָּפְנוּ בְּעוֹל עַל צַוָּארֵנוּ, יָגַעְנוּ וְלֹא הוּנַח לָנוּ.
ו מִצְרַ֙יִם֙ נָתַ֣נּוּ יָ֔ד אַשּׁ֖וּר לִשְׂבֹּ֥עַֽ לָֽחֶם׃ הוֹשַׁטְנוּ יָד לְמִצְרַיִם שֶׁיַּעַזְרוּ לָנוּ, בִּקַּשְׁנוּ מֵאַשּׁוּר לִשְׂבּוֹעַ לֶחֶם.
ז אֲבֹתֵ֤ינוּ חָֽטְאוּ֙ אינם וְאֵינָ֔ם אנחנו וַאֲנַ֖חְנוּ עֲוֺנֹתֵיהֶ֥ם סָבָֽלְנוּ׃ אֲבוֹתֵינוּ חָטְאוּ וְאֵינָם, וַאֲנַחְנוּ סָבַלְנוּ מֵעֲווֹנוֹתֵיהֶם.
ח עֲבָדִים֙ מָ֣שְׁלוּ בָ֔נוּ פֹּרֵ֖ק אֵ֥ין מִיָּדָֽם׃ עֲבָדִים מָשְׁלוּ בָּנוּ, וְאֵין מִי שֶׁיַּצִּיל מִיָּדָם.
ט בְּנַפְשֵׁ֙נוּ֙ נָבִ֣יא לַחְמֵ֔נוּ מִפְּנֵ֖י חֶ֥רֶב הַמִּדְבָּֽר׃ נָבִיא אֶת לַחְמֵנוּ בְּסַכָּנַת נְפָשׁוֹת, בִּגְלַל אוֹחֲזֵי הַחֶרֶב שֶׁבָּאִים מֵהַמִּדְבָּר.
י עוֹרֵ֙נוּ֙ כְּתַנּ֣וּר נִכְמָ֔רוּ מִפְּנֵ֖י זַלְעֲפ֥וֹת רָעָֽב׃ עוֹרֵנוּ נִרְאָה כְּשָׂרוּף בְּתַנּוּר, בִּגְלַל הִתְיַבְּשׁוּת מֵרָעָב.
יא נָשִׁים֙ בְּצִיּ֣וֹן עִנּ֔וּ בְּתֻלֹ֖ת בְּעָרֵ֥י יְהוּדָֽה׃ עִנּוּ נָשִׁים בְּצִיּוֹן, וּבְתוּלוֹת בְּעָרֵי יְהוּדָה.
יב שָׂרִים֙ בְּיָדָ֣ם נִתְל֔וּ פְּנֵ֥י זְקֵנִ֖ים לֹ֥א נֶהְדָּֽרוּ׃ שָׂרִים נִתְלוּ בְּיָדָם, פְּנֵי זְקֵנִים לֹא כֻּבְּדוּ.
יג בַּחוּרִים֙ טְח֣וֹן נָשָׂ֔אוּ וּנְעָרִ֖ים בָּעֵ֥ץ כָּשָֽׁלוּ׃ בַּחוּרִים נָשְׂאוּ רֵחַיִם שֶׁל טְחִינָה, וּנְעָרִים אֲסוּרִים בְּטַבְלַת עֵץ בְּצַוָּרָם כָּשְׁלוּ וְנָפְלוּ.
יד זְקֵנִים֙ מִשַּׁ֣עַר שָׁבָ֔תוּ בַּחוּרִ֖ים מִנְּגִינָתָֽם׃ זְקֵנִים פָּסְקוּ מִלָּשֶׁבֶת בַּשַּׁעַר, בַּחוּרִים פָּסְקוּ מִנְּגִינָתָם.
טו שָׁבַת֙ מְשׂ֣וֹשׂ לִבֵּ֔נוּ נֶהְפַּ֥ךְ לְאֵ֖בֶל מְחֹלֵֽנוּ׃ שִׂמְחַת לִבֵּנוּ הֻשְׁבְּתָה, רִקּוּדֵנוּ הָפַךְ לְאֵבֶל.
טז נָֽפְלָה֙ עֲטֶ֣רֶת רֹאשֵׁ֔נוּ אֽוֹי־נָ֥א לָ֖נוּ כִּ֥י חָטָֽאנוּ׃ הַכֶּתֶר שֶׁעַל רֹאשֵׁנוּ נָפַל, אוֹי לָנוּ שֶׁחָטָאנוּ.
יז עַל־זֶ֗ה הָיָ֤ה דָוֶה֙ לִבֵּ֔נוּ עַל־אֵ֖לֶּה חָשְׁכ֥וּ עֵינֵֽינוּ׃ עַל זֶה הָיָה לִבֵּנוּ עָצוּב, עַל דְּבָרִים אֵלֶּה עֵינֵינוּ חָשְׁכוּ –
יח עַ֤ל הַר־צִיּוֹן֙ שֶׁשָּׁמֵ֔ם שׁוּעָלִ֖ים הִלְּכוּ־בֽוֹ׃ עַל הַר צִיּוֹן שֶׁשָּׁמֵם – שׁוּעָלִים הָלְכוּ בּוֹ.
פ
יט אַתָּ֤ה יְהֹוָה֙ לְעוֹלָ֣ם תֵּשֵׁ֔ב כִּסְאֲךָ֖ לְד֥וֹר וָדֽוֹר׃ אַתָּה ה' תֵּשֵׁב בְּמַלְכוּתְךָ לְעוֹלָם, כִּסְאֲךָ יִתְקַיֵּם לְדוֹר וָדוֹר.
כ לָ֤מָּה לָנֶ֙צַח֙ תִּשְׁכָּחֵ֔נוּ תַּֽעַזְבֵ֖נוּ לְאֹ֥רֶךְ יָמִֽים׃ אָז לָמָּה תִּשְׁכַּח אוֹתָנוּ לָנֶצַח, לָמָּה עֲזַבְתָּ אוֹתָנוּ יָמִים אֲרֻכִּים.
כא הֲשִׁיבֵ֨נוּ יְהֹוָ֤ה ׀ אֵלֶ֙יךָ֙ ונשוב וְֽנָשׁ֔וּבָה חַדֵּ֥שׁ יָמֵ֖ינוּ כְּקֶֽדֶם׃ הֲשִׁיבֵנוּ ה' וְנָשׁוּב אֵלֶיךָ, חַדֵּשׁ אֶת יָמֵינוּ כְּמוֹ בֶּעָבָר.
כב כִּ֚י אִם־מָאֹ֣ס מְאַסְתָּ֔נוּ קָצַ֥פְתָּ עָלֵ֖ינוּ עַד־מְאֹֽד׃ וְכִי נִמְאַסְנוּ עָלֶיךָ? כָּעַסְתָּ עָלֵינוּ עַד מְאוֹד?!
