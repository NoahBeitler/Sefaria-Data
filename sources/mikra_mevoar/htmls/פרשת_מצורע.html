ראשון[עריכה]

פ פָּרָשַׁת טַהֲרַת הַמְּצוֹרָע
יד א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב זֹ֤את תִּֽהְיֶה֙ תּוֹרַ֣ת הַמְּצֹרָ֔ע בְּי֖וֹם טׇהֳרָת֑וֹ וְהוּבָ֖א אֶל־הַכֹּהֵֽן׃ אֵלֶּה יִהְיוּ דִּינֵי הַמְּצוֹרָע בַּיּוֹם שֶׁהוּא נִטְהַר: הוּא יוּבָא לַכֹּהֵן.
ג וְיָצָא֙ הַכֹּהֵ֔ן אֶל־מִח֖וּץ לַֽמַּחֲנֶ֑ה וְרָאָה֙ הַכֹּהֵ֔ן וְהִנֵּ֛ה נִרְפָּ֥א נֶֽגַע־הַצָּרַ֖עַת מִן־הַצָּרֽוּעַ׃ הַכֹּהֵן יֵצֵא אֶל מִחוּץ לַמַּחֲנֶה, וְהַכֹּהֵן יִרְאֶה שֶׁהִנֵּה נֶגַע הַצָּרַעַת נִרְפָּא מֵהַצָּרוּעַ.
ד וְצִוָּה֙ הַכֹּהֵ֔ן וְלָקַ֧ח לַמִּטַּהֵ֛ר שְׁתֵּֽי־צִפֳּרִ֥ים חַיּ֖וֹת טְהֹר֑וֹת וְעֵ֣ץ אֶ֔רֶז וּשְׁנִ֥י תוֹלַ֖עַת וְאֵזֹֽב׃ הַכֹּהֵן יְצַוֶּה שֶׁיִּקְחוּ בִּשְׁבִיל הַנִּטְהַר שְׁתֵּי צִפּוֹרִים חַיּוֹת טְהוֹרוֹת, וְעֵץ אֶרֶז וְחוּט צָבוּעַ בְּצֶבַע אָדוֹם מִתּוֹלַעַת וְאֵזוֹב.
ה וְצִוָּה֙ הַכֹּהֵ֔ן וְשָׁחַ֖ט אֶת־הַצִּפּ֣וֹר הָאֶחָ֑ת אֶל־כְּלִי־חֶ֖רֶשׂ עַל־מַ֥יִם חַיִּֽים׃ הַכֹּהֵן יְצַוֶּה שֶׁיִּשְׁחֲטוּ אֶת אַחַת הַצִּפּוֹרִים, מֵעַל כְּלִי חֶרֶס מָלֵא בְּמֵי מַעְיָן.
ו אֶת־הַצִּפֹּ֤ר הַֽחַיָּה֙ יִקַּ֣ח אֹתָ֔הּ וְאֶת־עֵ֥ץ הָאֶ֛רֶז וְאֶת־שְׁנִ֥י הַתּוֹלַ֖עַת וְאֶת־הָאֵזֹ֑ב וְטָבַ֨ל אוֹתָ֜ם וְאֵ֣ת ׀ הַצִּפֹּ֣ר הַֽחַיָּ֗ה בְּדַם֙ הַצִּפֹּ֣ר הַשְּׁחֻטָ֔ה עַ֖ל הַמַּ֥יִם הַֽחַיִּֽים׃ וְיִקַּח אֶת הַצִּפּוֹר הַחַיָּה וְאֶת עֵץ הָאֶרֶז וְאֶת הַחוּט הַצָּבוּעַ בְּצֶבַע אָדוֹם מִתּוֹלַעַת וְאֶת הָאֵזוֹב, וְיַטְבִּיל אוֹתָם וְאֶת הַצִּפּוֹר הַחַיָּה בְּדַם הַצִּפּוֹר שֶׁנִּשְׁחֲטָה עַל מֵי הַמַּעְיַן.
ז וְהִזָּ֗ה עַ֧ל הַמִּטַּהֵ֛ר מִן־הַצָּרַ֖עַת שֶׁ֣בַע פְּעָמִ֑ים וְטִ֣הֲר֔וֹ וְשִׁלַּ֛ח אֶת־הַצִּפֹּ֥ר הַֽחַיָּ֖ה עַל־פְּנֵ֥י הַשָּׂדֶֽה׃ וְיַזֶּה עַל הַנִּטְהַר מֵהַצָּרַעַת שֶׁבַע פְּעָמִים, וִיטַהֵר אוֹתוֹ, וִישַׁחְרֵר בַּשָּׂדֶה אֶת הַצִּפּוֹר הַחַיָּה.
ח וְכִבֶּס֩ הַמִּטַּהֵ֨ר אֶת־בְּגָדָ֜יו וְגִלַּ֣ח אֶת־כׇּל־שְׂעָר֗וֹ וְרָחַ֤ץ בַּמַּ֙יִם֙ וְטָהֵ֔ר וְאַחַ֖ר יָב֣וֹא אֶל־הַֽמַּחֲנֶ֑ה וְיָשַׁ֛ב מִח֥וּץ לְאׇהֳל֖וֹ שִׁבְעַ֥ת יָמִֽים׃ הַנִּטְהַר יַטְבִּיל אֶת בְּגָדָיו וִיגַלֵּחַ אֶת כָּל שְׂעָרוֹ וְיִטְבּוֹל בְּמַיִם וְיִטָּהֵר וְאַחַר כָּךְ יוּכַל לָבוֹא לַמַּחֲנֶה, אֲבָל יֵשֵׁב עֲדַיִן מִחוּץ לְאָהֳלוֹ שִׁבְעָה יָמִים.
ט וְהָיָה֩ בַיּ֨וֹם הַשְּׁבִיעִ֜י יְגַלַּ֣ח אֶת־כׇּל־שְׂעָר֗וֹ אֶת־רֹאשׁ֤וֹ וְאֶת־זְקָנוֹ֙ וְאֵת֙ גַּבֹּ֣ת עֵינָ֔יו וְאֶת־כׇּל־שְׂעָר֖וֹ יְגַלֵּ֑חַ וְכִבֶּ֣ס אֶת־בְּגָדָ֗יו וְרָחַ֧ץ אֶת־בְּשָׂר֛וֹ בַּמַּ֖יִם וְטָהֵֽר׃ בַּיּוֹם הַשְּׁבִיעִי יְגַלֵּחַ אֶת כָּל שְׂעָרוֹ – אֶת שְׂעַר רֹאשׁוֹ וְאֶת זְקָנוֹ וְאֶת גַּבּוֹת עֵינָיו וְאֶת כָּל שְׁאָר שְׂעָרוֹ, וְיַטְבִּיל אֶת בְּגָדָיו וְיַטְבִּיל אֶת גּוּפוֹ בְּמַיִם וְאָז יִטָּהֵר.
י וּבַיּ֣וֹם הַשְּׁמִינִ֗י יִקַּ֤ח שְׁנֵֽי־כְבָשִׂים֙ תְּמִימִ֔ם וְכַבְשָׂ֥ה אַחַ֛ת בַּת־שְׁנָתָ֖הּ תְּמִימָ֑ה וּשְׁלֹשָׁ֣ה עֶשְׂרֹנִ֗ים סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן וְלֹ֥ג אֶחָ֖ד שָֽׁמֶן׃ וּבַיּוֹם הַשְּׁמִינִי יִקַּח שְׁנֵי כְּבָשִׂים לְלֹא מוּם, וְכִבְשָׂה אַחַת בַּת שָׁנָה לְכָל הַיּוֹתֵר, לְלֹא מוּם, וְכַמּוּת שֶׁל שָׁלוֹשׁ עֲשִׂירִיּוֹת אֵיפָה (כְּ6.5 לִיטֶר) סוֹלֶת עֲבוּר מִנְחָה מְעֻרְבֶּבֶת בְּשֶׁמֶן, וְלוֹג אֶחָד (כִּ300 מ"ל) שֶׁמֶן.
יא וְהֶעֱמִ֞יד הַכֹּהֵ֣ן הַֽמְטַהֵ֗ר אֵ֛ת הָאִ֥ישׁ הַמִּטַּהֵ֖ר וְאֹתָ֑ם לִפְנֵ֣י יְהֹוָ֔ה פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ הַכֹּהֵן הַמְּטַהֵר יַעֲמִיד אֶת הָאִישׁ הַנִּטְהַר וְאֶת הַכְּבָשִׂים לִפְנֵי ה' מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.
יב וְלָקַ֨ח הַכֹּהֵ֜ן אֶת־הַכֶּ֣בֶשׂ הָאֶחָ֗ד וְהִקְרִ֥יב אֹת֛וֹ לְאָשָׁ֖ם וְאֶת־לֹ֣ג הַשָּׁ֑מֶן וְהֵנִ֥יף אֹתָ֛ם תְּנוּפָ֖ה לִפְנֵ֥י יְהֹוָֽה׃ הַכֹּהֵן יִקַּח אֶת אֶחָד מֵהַכְּבָשִׂים וְיַגִּישׁ אוֹתוֹ וְאֶת לוֹג הַשֶּׁמֶן לְקָרְבַּן אָשָׁם וְיָנִיף אוֹתָם לִפְנֵי ה'.

שני[עריכה]

יג וְשָׁחַ֣ט אֶת־הַכֶּ֗בֶשׂ בִּ֠מְק֠וֹם אֲשֶׁ֨ר יִשְׁחַ֧ט אֶת־הַֽחַטָּ֛את וְאֶת־הָעֹלָ֖ה בִּמְק֣וֹם הַקֹּ֑דֶשׁ כִּ֡י כַּ֠חַטָּ֠את הָאָשָׁ֥ם הוּא֙ לַכֹּהֵ֔ן קֹ֥דֶשׁ קׇֽדָשִׁ֖ים הֽוּא׃ הוּא יִשְׁחַט אֶת הַכֶּבֶשׂ בַּמָּקוֹם שֶׁשּׁוֹחֲטִים אֶת הַחַטָּאת וְאֶת הָעוֹלָה – בִּמְקוֹם הַקּוֹדֶשׁ, כִּי כְּמוֹ הַחַטָּאת – כָּךְ גַּם הָאָשָׁם נִתָּן לַכֹּהֵן, הוּא קוֹדֶשׁ קָדָשִׁים.
יד וְלָקַ֣ח הַכֹּהֵן֮ מִדַּ֣ם הָאָשָׁם֒ וְנָתַן֙ הַכֹּהֵ֔ן עַל־תְּנ֛וּךְ אֹ֥זֶן הַמִּטַּהֵ֖ר הַיְמָנִ֑ית וְעַל־בֹּ֤הֶן יָדוֹ֙ הַיְמָנִ֔ית וְעַל־בֹּ֥הֶן רַגְל֖וֹ הַיְמָנִֽית׃ הַכֹּהֵן יִקַּח מִדַּם הָאָשָׁם, וְיָשִׂים עַל תְּנוּךְ הָאוֹזֶן הַיְּמָנִית שֶׁל הַנִּטְהַר וְעַל אֲגֻדָּל יָדוֹ הַיְּמָנִית וְעַל בּוֹהֶן רַגְלוֹ הַיְּמָנִית.
טו וְלָקַ֥ח הַכֹּהֵ֖ן מִלֹּ֣ג הַשָּׁ֑מֶן וְיָצַ֛ק עַל־כַּ֥ף הַכֹּהֵ֖ן הַשְּׂמָאלִֽית׃ הַכֹּהֵן יִקַּח מִלּוֹג הַשֶּׁמֶן וְיִשְׁפּוֹךְ עַל כַּף הַיַּד הַשְּׂמָאלִית שֶׁל הַכֹּהֵן.
טז וְטָבַ֤ל הַכֹּהֵן֙ אֶת־אֶצְבָּע֣וֹ הַיְמָנִ֔ית מִן־הַשֶּׁ֕מֶן אֲשֶׁ֥ר עַל־כַּפּ֖וֹ הַשְּׂמָאלִ֑ית וְהִזָּ֨ה מִן־הַשֶּׁ֧מֶן בְּאֶצְבָּע֛וֹ שֶׁ֥בַע פְּעָמִ֖ים לִפְנֵ֥י יְהֹוָֽה׃ הַכֹּהֵן יַטְבִּיל אֶת אֶצְבָּעוֹ הַיְּמָנִית בַּשֶּׁמֶן שֶׁעַל יָדוֹ הַשְּׂמָאלִית וְיַזֶּה בְּאֶצְבָּעוֹ מֵהַשֶּׁמֶן שֶׁבַע פְּעָמִים לִפְנֵי ה'.
יז וּמִיֶּ֨תֶר הַשֶּׁ֜מֶן אֲשֶׁ֣ר עַל־כַּפּ֗וֹ יִתֵּ֤ן הַכֹּהֵן֙ עַל־תְּנ֞וּךְ אֹ֤זֶן הַמִּטַּהֵר֙ הַיְמָנִ֔ית וְעַל־בֹּ֤הֶן יָדוֹ֙ הַיְמָנִ֔ית וְעַל־בֹּ֥הֶן רַגְל֖וֹ הַיְמָנִ֑ית עַ֖ל דַּ֥ם הָאָשָֽׁם׃ וּמִשְּׁאָר הַשֶּׁמֶן שֶׁעַל יָדוֹ הַכֹּהֵן יָשִׂים עַל תְּנוּךְ הָאוֹזֶן הַיְּמָנִית שֶׁל הַנִּטְהַר וְעַל אֲגֻדָּל יָדוֹ הַיְּמָנִית וְעַל בּוֹהֶן רַגְלוֹ הַיְּמָנִית, בַּמָּקוֹם שֶׁנִּתַּן לִפְנֵי כֵן דַּם הָאָשָׁם.
יח וְהַנּוֹתָ֗ר בַּשֶּׁ֙מֶן֙ אֲשֶׁר֙ עַל־כַּ֣ף הַכֹּהֵ֔ן יִתֵּ֖ן עַל־רֹ֣אשׁ הַמִּטַּהֵ֑ר וְכִפֶּ֥ר עָלָ֛יו הַכֹּהֵ֖ן לִפְנֵ֥י יְהֹוָֽה׃ וְאֶת מַה שֶּׁנִּשְׁאָר מֵהַשֶּׁמֶן שֶׁעַל כַּף הַיַּד שֶׁל הַכֹּהֵן יָשִׂים עַל רֹאשׁ הַנִּטְהַר, וְהַכֹּהֵן יְכַפֵּר עָלָיו לִפְנֵי ה'.
יט וְעָשָׂ֤ה הַכֹּהֵן֙ אֶת־הַ֣חַטָּ֔את וְכִפֶּ֕ר עַל־הַמִּטַּהֵ֖ר מִטֻּמְאָת֑וֹ וְאַחַ֖ר יִשְׁחַ֥ט אֶת־הָעֹלָֽה׃ הַכֹּהֵן יַעֲשֶׂה אֶת הַחַטָּאת וִיכַפֵּר עַל הַנִּטְהַר מִטֻּמְאָתוֹ, וְאַחַר כָּךְ יִשְׁחַט אֶת הָעוֹלָה.
כ וְהֶעֱלָ֧ה הַכֹּהֵ֛ן אֶת־הָעֹלָ֥ה וְאֶת־הַמִּנְחָ֖ה הַמִּזְבֵּ֑חָה וְכִפֶּ֥ר עָלָ֛יו הַכֹּהֵ֖ן וְטָהֵֽר׃ הַכֹּהֵן יַעֲלֶה אֶת הָעוֹלָה וְאֶת הַמִּנְחָה לַמִּזְבֵּחַ, הַכֹּהֵן יְכַפֵּר עָלָיו וְאָז הוּא יִהְיֶה טָהוֹר.

שלישי (חמישי במחוברין)[עריכה]

ס
כא וְאִם־דַּ֣ל ה֗וּא וְאֵ֣ין יָדוֹ֮ מַשֶּׂ֒גֶת֒ וְ֠לָקַ֠ח כֶּ֣בֶשׂ אֶחָ֥ד אָשָׁ֛ם לִתְנוּפָ֖ה לְכַפֵּ֣ר עָלָ֑יו וְעִשָּׂר֨וֹן סֹ֜לֶת אֶחָ֨ד בָּל֥וּל בַּשֶּׁ֛מֶן לְמִנְחָ֖ה וְלֹ֥ג שָֽׁמֶן׃ וְאִם הוּא עָנִי וְאֵין לוֹ מַסְפִּיק כֶּסֶף, יָבִיא כֶּבֶשׂ אֶחָד לְאָשָׁם לַהֲנָפָה כְּדֵי לְכַפֵּר עַל עַצְמוֹ, וַעֲשִׂירִית אֵיפָה (כִּ2.16 לִיטֶר) שֶׁל סוֹלֶת מְעֻרְבֶּבֶת בְּשֶׁמֶן לְמִנְחָה, וְלוֹג (כִּ300 מ"ל) שֶׁמֶן.
כב וּשְׁתֵּ֣י תֹרִ֗ים א֤וֹ שְׁנֵי֙ בְּנֵ֣י יוֹנָ֔ה אֲשֶׁ֥ר תַּשִּׂ֖יג יָד֑וֹ וְהָיָ֤ה אֶחָד֙ חַטָּ֔את וְהָאֶחָ֖ד עֹלָֽה׃ וּשְׁתֵּי תּוֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה – שֶׁיֵּשׁ לוֹ מַסְפִּיק כֶּסֶף לִקְנוֹת, וְיִהְיֶה אֶחָד חַטָּאת, וְאֶחָד עוֹלָה.
כג וְהֵבִ֨יא אֹתָ֜ם בַּיּ֧וֹם הַשְּׁמִינִ֛י לְטׇהֳרָת֖וֹ אֶל־הַכֹּהֵ֑ן אֶל־פֶּ֥תַח אֹֽהֶל־מוֹעֵ֖ד לִפְנֵ֥י יְהֹוָֽה׃ הוּא יָבִיא אוֹתָם בַּיּוֹם הַשְּׁמִינִי לְטַהֲרָתוֹ לַכֹּהֵן, אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת – לִפְנֵי ה'.
כד וְלָקַ֧ח הַכֹּהֵ֛ן אֶת־כֶּ֥בֶשׂ הָאָשָׁ֖ם וְאֶת־לֹ֣ג הַשָּׁ֑מֶן וְהֵנִ֨יף אֹתָ֧ם הַכֹּהֵ֛ן תְּנוּפָ֖ה לִפְנֵ֥י יְהֹוָֽה׃ הַכֹּהֵן יִקַּח אֶת כֶּבֶשׂ הָאָשָׁם וְאֶת לוֹג הַשֶּׁמֶן, וְהַכֹּהֵן יָנִיף אוֹתָם לִפְנֵי ה'.
כה וְשָׁחַט֮ אֶת־כֶּ֣בֶשׂ הָֽאָשָׁם֒ וְלָקַ֤ח הַכֹּהֵן֙ מִדַּ֣ם הָֽאָשָׁ֔ם וְנָתַ֛ן עַל־תְּנ֥וּךְ אֹֽזֶן־הַמִּטַּהֵ֖ר הַיְמָנִ֑ית וְעַל־בֹּ֤הֶן יָדוֹ֙ הַיְמָנִ֔ית וְעַל־בֹּ֥הֶן רַגְל֖וֹ הַיְמָנִֽית׃ הוּא יִשְׁחַט אֶת כֶּבֶשׂ הָאָשָׁם, וְהַכֹּהֵן יִקַּח מִדַּם הָאָשָׁם וְיָשִׂים עַל תְּנוּךְ הָאוֹזֶן הַיְּמָנִית שֶׁל הַנִּטְהַר וְעַל אֲגֻדָּל יָדוֹ הַיְּמָנִית וְעַל בּוֹהֶן רַגְלוֹ הַיְּמָנִית.
כו וּמִן־הַשֶּׁ֖מֶן יִצֹ֣ק הַכֹּהֵ֑ן עַל־כַּ֥ף הַכֹּהֵ֖ן הַשְּׂמָאלִֽית׃ וְהַכֹּהֵן יִשְׁפּוֹךְ מֵהַשֶּׁמֶן עַל כַּף יַד הַכֹּהֵן הַשְּׂמָאלִית.
כז וְהִזָּ֤ה הַכֹּהֵן֙ בְּאֶצְבָּע֣וֹ הַיְמָנִ֔ית מִן־הַשֶּׁ֕מֶן אֲשֶׁ֥ר עַל־כַּפּ֖וֹ הַשְּׂמָאלִ֑ית שֶׁ֥בַע פְּעָמִ֖ים לִפְנֵ֥י יְהֹוָֽה׃ הַכֹּהֵן יַתִּיז בְּאֶצְבָּעוֹ הַיְּמָנִית מֵהַשֶּׁמֶן שֶׁעַל יָדוֹ הַשְּׂמָאלִית שֶׁבַע פְּעָמִים לִפְנֵי ה'.
כח וְנָתַ֨ן הַכֹּהֵ֜ן מִן־הַשֶּׁ֣מֶן ׀ אֲשֶׁ֣ר עַל־כַּפּ֗וֹ עַל־תְּנ֞וּךְ אֹ֤זֶן הַמִּטַּהֵר֙ הַיְמָנִ֔ית וְעַל־בֹּ֤הֶן יָדוֹ֙ הַיְמָנִ֔ית וְעַל־בֹּ֥הֶן רַגְל֖וֹ הַיְמָנִ֑ית עַל־מְק֖וֹם דַּ֥ם הָאָשָֽׁם׃ הַכֹּהֵן יָשִׂים מֵהַשֶּׁמֶן שֶׁעַל יָדוֹ עַל תְּנוּךְ הָאוֹזֶן הַיְּמָנִית שֶׁל הַנִּטְהַר וְעַל אֲגֻדָּל יָדוֹ הַיְּמָנִית וְעַל בּוֹהֶן רַגְלוֹ הַיְּמָנִית, עַל הַמָּקוֹם שֶׁשָּׂם אֶת דַּם הָאָשָׁם.
כט וְהַנּוֹתָ֗ר מִן־הַשֶּׁ֙מֶן֙ אֲשֶׁר֙ עַל־כַּ֣ף הַכֹּהֵ֔ן יִתֵּ֖ן עַל־רֹ֣אשׁ הַמִּטַּהֵ֑ר לְכַפֵּ֥ר עָלָ֖יו לִפְנֵ֥י יְהֹוָֽה׃ וְאֶת מַה שֶּׁנִּשְׁאָר מֵהַשֶּׁמֶן שֶׁעַל כַּף יָדוֹ שֶׁל הַכֹּהֵן יָשִׂים עַל רֹאשׁ הַנִּטְהַר, לְכַפֵּר עָלָיו לִפְנֵי ה'.
ל וְעָשָׂ֤ה אֶת־הָֽאֶחָד֙ מִן־הַתֹּרִ֔ים א֖וֹ מִן־בְּנֵ֣י הַיּוֹנָ֑ה מֵאֲשֶׁ֥ר תַּשִּׂ֖יג יָדֽוֹ׃ וְיַעֲשֶׂה אֶת אֶחָד הָעוֹפוֹת – מֵהַתּוֹרִים אוֹ מִבְּנֵי הַיּוֹנָה – מִמַּה שֶּׁיִּהְיֶה לוֹ מַסְפִּיק כֶּסֶף לִקְנוֹת.
לא אֵ֣ת אֲשֶׁר־תַּשִּׂ֞יג יָד֗וֹ אֶת־הָאֶחָ֥ד חַטָּ֛את וְאֶת־הָאֶחָ֥ד עֹלָ֖ה עַל־הַמִּנְחָ֑ה וְכִפֶּ֧ר הַכֹּהֵ֛ן עַ֥ל הַמִּטַּהֵ֖ר לִפְנֵ֥י יְהֹוָֽה׃ אֶת מַה שֶּׁיִּהְיֶה לוֹ מַסְפִּיק כֶּסֶף לִקְנוֹת – יַעֲשֶׂה אֶת הָאֶחָד חַטָּאת וְאֶת הָאֶחָד עוֹלָה, בְּנוֹסַף עַל הַמִּנְחָה, הַכֹּהֵן יְכַפֵּר עַל הַנִּטְהַר לִפְנֵי ה'.
לב זֹ֣את תּוֹרַ֔ת אֲשֶׁר־בּ֖וֹ נֶ֣גַע צָרָ֑עַת אֲשֶׁ֛ר לֹֽא־תַשִּׂ֥יג יָד֖וֹ בְּטׇהֳרָתֽוֹ׃ אֵלֶּה דִּינֵי הַהִטַּהֲרוּת שֶׁל אָדָם שֶׁמִּתְרַפֵּא מִנֶּגַע צָרַעַת שֶׁאֵין לוֹ מַסְפִּיק כֶּסֶף.

רביעי (ששי במחוברין)[עריכה]

פ פָּרָשַׁת צָרַעַת הַבַּיִת
לג וַיְדַבֵּ֣ר יְהֹוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאֶל אַהֲרֹן וְאָמַר:
לד כִּ֤י תָבֹ֙אוּ֙ אֶל־אֶ֣רֶץ כְּנַ֔עַן אֲשֶׁ֥ר אֲנִ֛י נֹתֵ֥ן לָכֶ֖ם לַאֲחֻזָּ֑ה וְנָתַתִּי֙ נֶ֣גַע צָרַ֔עַת בְּבֵ֖ית אֶ֥רֶץ אֲחֻזַּתְכֶֽם׃ כְּשֶׁתָּבוֹאוּ לְאֶרֶץ כְּנַעַן שֶׁאֲנִי נוֹתֵן לָכֶם לְנַחֲלָה, וְאֶעֱשֶׂה נֶגַע צָרַעַת בְּבָתִּים בְּאֶרֶץ נַחֲלַתְכֶם,
לה וּבָא֙ אֲשֶׁר־ל֣וֹ הַבַּ֔יִת וְהִגִּ֥יד לַכֹּהֵ֖ן לֵאמֹ֑ר כְּנֶ֕גַע נִרְאָ֥ה לִ֖י בַּבָּֽיִת׃ בַּעַל הַבַּיִת יָבוֹא וִיסַפֵּר לַכֹּהֵן וְיֹאמַר: נִרְאָה אֶצְלִי בַּבַּיִת כְּמוֹ נֶגַע.
לו וְצִוָּ֨ה הַכֹּהֵ֜ן וּפִנּ֣וּ אֶת־הַבַּ֗יִת בְּטֶ֨רֶם יָבֹ֤א הַכֹּהֵן֙ לִרְא֣וֹת אֶת־הַנֶּ֔גַע וְלֹ֥א יִטְמָ֖א כׇּל־אֲשֶׁ֣ר בַּבָּ֑יִת וְאַ֥חַר כֵּ֛ן יָבֹ֥א הַכֹּהֵ֖ן לִרְא֥וֹת אֶת־הַבָּֽיִת׃ הַכֹּהֵן יְצַוֶּה שֶׁיְּפַנּוּ אֶת הַבַּיִת לִפְנֵי שֶׁהַכֹּהֵן יָבוֹא לִרְאוֹת אֶת הַנֶּגַע כְּדֵי שֶׁלֹּא יִטְמָא כָּל מַה שֶּׁבַּבַּיִת, וְאַחַר כָּךְ הַכֹּהֵן יָבוֹא לִרְאוֹת אֶת הַבַּיִת.
לז וְרָאָ֣ה אֶת־הַנֶּ֗גַע וְהִנֵּ֤ה הַנֶּ֙גַע֙ בְּקִירֹ֣ת הַבַּ֔יִת שְׁקַֽעֲרוּרֹת֙ יְרַקְרַקֹּ֔ת א֖וֹ אֲדַמְדַּמֹּ֑ת וּמַרְאֵיהֶ֥ן שָׁפָ֖ל מִן־הַקִּֽיר׃ הוּא יִרְאֶה אֶת הַנֶּגַע וְהִנֵּה הַנֶּגַע בְּקִירוֹת הַבַּיִת שְׁקָעִים צְהַבְהַבִּים אוֹ אֲדֻמִּים, וּמַרְאֵיהֶם עָמוֹק מֵהַקִּיר.
לח וְיָצָ֧א הַכֹּהֵ֛ן מִן־הַבַּ֖יִת אֶל־פֶּ֣תַח הַבָּ֑יִת וְהִסְגִּ֥יר אֶת־הַבַּ֖יִת שִׁבְעַ֥ת יָמִֽים׃ הַכֹּהֵן יֵצֵא מֵהַבַּיִת אֶל פֶּתַח הַבַּיִת וְיַסְגִּיר אֶת הַבַּיִת שִׁבְעָה יָמִים.
לט וְשָׁ֥ב הַכֹּהֵ֖ן בַּיּ֣וֹם הַשְּׁבִיעִ֑י וְרָאָ֕ה וְהִנֵּ֛ה פָּשָׂ֥ה הַנֶּ֖גַע בְּקִירֹ֥ת הַבָּֽיִת׃ הַכֹּהֵן יַחֲזוֹר בַּיּוֹם הַשְּׁבִיעִי, וְיִרְאֶה וְהִנֵּה הַנֶּגַע הִתְפַּשֵּׁט בְּקִירוֹת הַבַּיִת.
מ וְצִוָּה֙ הַכֹּהֵ֔ן וְחִלְּצוּ֙ אֶת־הָ֣אֲבָנִ֔ים אֲשֶׁ֥ר בָּהֵ֖ן הַנָּ֑גַע וְהִשְׁלִ֤יכוּ אֶתְהֶן֙ אֶל־מִח֣וּץ לָעִ֔יר אֶל־מָק֖וֹם טָמֵֽא׃ הַכֹּהֵן יְצַוֶּה שֶׁיּוֹצִיאוּ אֶת הָאֲבָנִים שֶׁנִּמְצָא בָּהֶם הַנֶּגַע וְיַשְׁלִיכוּ אוֹתָן אֶל מִחוּץ לָעִיר לְמָקוֹם טָמֵא.
מא וְאֶת־הַבַּ֛יִת יַקְצִ֥עַ מִבַּ֖יִת סָבִ֑יב וְשָׁפְכ֗וּ אֶת־הֶֽעָפָר֙ אֲשֶׁ֣ר הִקְצ֔וּ אֶל־מִח֣וּץ לָעִ֔יר אֶל־מָק֖וֹם טָמֵֽא׃ וְאֶת הַבַּיִת יְקַלֵּף מִבִּפְנִים מִסָּבִיב, וְיִשְׁפְּכוּ אֶת הֶעָפָר שֶׁקִּלְּפוּ אֶל מִחוּץ לָעִיר לְמָקוֹם טָמֵא.
מב וְלָקְחוּ֙ אֲבָנִ֣ים אֲחֵר֔וֹת וְהֵבִ֖יאוּ אֶל־תַּ֣חַת הָאֲבָנִ֑ים וְעָפָ֥ר אַחֵ֛ר יִקַּ֖ח וְטָ֥ח אֶת־הַבָּֽיִת׃ יִקְחוּ אֲבָנִים אֲחֵרוֹת וְיָבִיאוּ בִּמְקוֹם הָאֲבָנִים שֶׁהוֹצִיאוּ, וְיִקְּחוּ עָפָר אַחֵר וִיטַיְּחוּ אֶת הַבַּיִת.
מג וְאִם־יָשׁ֤וּב הַנֶּ֙גַע֙ וּפָרַ֣ח בַּבַּ֔יִת אַחַ֖ר חִלֵּ֣ץ אֶת־הָאֲבָנִ֑ים וְאַחֲרֵ֛י הִקְצ֥וֹת אֶת־הַבַּ֖יִת וְאַחֲרֵ֥י הִטּֽוֹחַ׃ וְאִם הַנֶּגַע יַחֲזוֹר וְיִתְפַּשֵּׁט בַּבַּיִת אַחֲרֵי הוֹצָאַת הָאֲבָנִים וְאַחֲרֵי קִלּוּף הַבַּיִת וְאַחֲרֵי הַטִּיּוּחַ.
מד וּבָא֙ הַכֹּהֵ֔ן וְרָאָ֕ה וְהִנֵּ֛ה פָּשָׂ֥ה הַנֶּ֖גַע בַּבָּ֑יִת צָרַ֨עַת מַמְאֶ֥רֶת הִ֛וא בַּבַּ֖יִת טָמֵ֥א הֽוּא׃ הַכֹּהֵן יָבוֹא וְיִרְאֶה שֶׁהִנֵּה הַנֶּגַע הִתְפַּשֵּׁט בַּבַּיִת – זוֹ צָרַעַת מַכְאִיבָה בַּבַּיִת – הוּא טָמֵא.
מה וְנָתַ֣ץ אֶת־הַבַּ֗יִת אֶת־אֲבָנָיו֙ וְאֶת־עֵצָ֔יו וְאֵ֖ת כׇּל־עֲפַ֣ר הַבָּ֑יִת וְהוֹצִיא֙ אֶל־מִח֣וּץ לָעִ֔יר אֶל־מָק֖וֹם טָמֵֽא׃ הוּא יְנַפֵּץ אֶת הַבַּיִת – אֶת אֲבָנָיו וְאֶת עֵצָיו וְאֶת כָּל עֲפַר הַבַּיִת, וְיוֹצִיא אֶל מִחוּץ לָעִיר לְמָקוֹם טָמֵא.
מו וְהַבָּא֙ אֶל־הַבַּ֔יִת כׇּל־יְמֵ֖י הִסְגִּ֣יר אֹת֑וֹ יִטְמָ֖א עַד־הָעָֽרֶב׃ וְהַנִּכְנָס לַבַּיִת בְּכָל יְמֵי הַהֶסְגֵּר יִהְיֶה טָמֵא עַד הָעֶרֶב.
מז וְהַשֹּׁכֵ֣ב בַּבַּ֔יִת יְכַבֵּ֖ס אֶת־בְּגָדָ֑יו וְהָאֹכֵ֣ל בַּבַּ֔יִת יְכַבֵּ֖ס אֶת־בְּגָדָֽיו׃ וְהַשּׁוֹכֵב בַּבַּיִת יַטְבִּיל אֶת בְּגָדָיו, וְהָאוֹכֵל בַּבַּיִת יַטְבִּיל אֶת בְּגָדָיו.
פָּרָשַׁת טַהֲרַת הַבַּיִת
מח וְאִם־בֹּ֨א יָבֹ֜א הַכֹּהֵ֗ן וְרָאָה֙ וְ֠הִנֵּ֠ה לֹא־פָשָׂ֤ה הַנֶּ֙גַע֙ בַּבַּ֔יִת אַחֲרֵ֖י הִטֹּ֣חַ אֶת־הַבָּ֑יִת וְטִהַ֤ר הַכֹּהֵן֙ אֶת־הַבַּ֔יִת כִּ֥י נִרְפָּ֖א הַנָּֽגַע׃ וְאִם הַכֹּהֵן יָבוֹא וְיִרְאֶה שֶׁהִנֵּה הַנֶּגַע לֹא הִתְפַּשֵּׁט בַּבַּיִת אַחֲרֵי טִיּוּחַ הַבַּיִת, הַכֹּהֵן יְטַהֵר אֶת הַבַּיִת כִּי הַנֶּגַע הִתְרַפֵּא.
מט וְלָקַ֛ח לְחַטֵּ֥א אֶת־הַבַּ֖יִת שְׁתֵּ֣י צִפֳּרִ֑ים וְעֵ֣ץ אֶ֔רֶז וּשְׁנִ֥י תוֹלַ֖עַת וְאֵזֹֽב׃ וְיִקַּח כְּדֵי לְטַהֵר אֶת הַבַּיִת שְׁתֵּי צִפּוֹרִים וְעֵץ אֶרֶז וְחוּט צָבוּעַ בְּצֶבַע אָדוֹם מִתּוֹלַעַת, וְאֵזוֹב.
נ וְשָׁחַ֖ט אֶת־הַצִּפֹּ֣ר הָאֶחָ֑ת אֶל־כְּלִי־חֶ֖רֶשׂ עַל־מַ֥יִם חַיִּֽים׃ וְיִשְׁחַט אֶת אַחַת מֵהַצִּפּוֹרִים כָּךְ שֶׁדָּמָהּ יִשָּׁפֵךְ עַל מַיִם מִמַּעְיָן שֶׁבִּכְלִי חֶרֶס.
נא וְלָקַ֣ח אֶת־עֵֽץ־הָ֠אֶ֠רֶז וְאֶת־הָ֨אֵזֹ֜ב וְאֵ֣ת ׀ שְׁנִ֣י הַתּוֹלַ֗עַת וְאֵת֮ הַצִּפֹּ֣ר הַֽחַיָּה֒ וְטָבַ֣ל אֹתָ֗ם בְּדַם֙ הַצִּפֹּ֣ר הַשְּׁחוּטָ֔ה וּבַמַּ֖יִם הַֽחַיִּ֑ים וְהִזָּ֥ה אֶל־הַבַּ֖יִת שֶׁ֥בַע פְּעָמִֽים׃ וְיִקַּח אֶת עֵץ הָאֶרֶז וְאֶת הָאֵזוֹב וְאֶת הַחוּט הַצָּבוּעַ בְּצֶבַע אָדוֹם מִתּוֹלַעַת, וְאֶת הַצִּפּוֹר הַחַיָּה, וְיַטְבִּיל אוֹתָם בְּדַם הַצִּפּוֹר הַשְּׁחוּטָה וּבַמַּיִם מֵהַמַּעְיַן, וְיַתִּיז עַל הַבַּיִת שֶׁבַע פְּעָמִים.
נב וְחִטֵּ֣א אֶת־הַבַּ֔יִת בְּדַם֙ הַצִּפּ֔וֹר וּבַמַּ֖יִם הַֽחַיִּ֑ים וּבַצִּפֹּ֣ר הַחַיָּ֗ה וּבְעֵ֥ץ הָאֶ֛רֶז וּבָאֵזֹ֖ב וּבִשְׁנִ֥י הַתּוֹלָֽעַת׃ הוּא יְטַהֵר אֶת הַבַּיִת בְּדַם הַצִּפּוֹר, וּבַמַּיִם מֵהַמַּעְיַן, וּבַצִּפּוֹר הַחַיָּה, וּבְעֵץ הָאֶרֶז, וּבָאֵזוֹב, וּבַחוּט הַצָּבוּעַ בְּצֶבַע אָדוֹם מִתּוֹלַעַת.
נג וְשִׁלַּ֞ח אֶת־הַצִּפֹּ֧ר הַֽחַיָּ֛ה אֶל־מִח֥וּץ לָעִ֖יר אֶל־פְּנֵ֣י הַשָּׂדֶ֑ה וְכִפֶּ֥ר עַל־הַבַּ֖יִת וְטָהֵֽר׃ וִישַׁחְרֵר אֶת הַצִּפּוֹר הַחַיָּה מִחוּץ לָעִיר בַּשָּׂדֶה, כָּךְ יְכַפֵּר עַל הַבַּיִת וְהַבַּיִת יִהְיֶה טָהוֹר.

חמישי[עריכה]

נד זֹ֖את הַתּוֹרָ֑ה לְכׇל־נֶ֥גַע הַצָּרַ֖עַת וְלַנָּֽתֶק׃ אֵלֶּה הַדִּינִים לְכָל נִגְעֵי הַצָּרַעַת וּלְנֶגַע נְתִיקַת הַשֵּׂעָר.
נה וּלְצָרַ֥עַת הַבֶּ֖גֶד וְלַבָּֽיִת׃ וּלְצָרַעַת הַבֶּגֶד וּלְצָרַעַת הַבַּיִת.
נו וְלַשְׂאֵ֥ת וְלַסַּפַּ֖חַת וְלַבֶּהָֽרֶת׃ וְלַשּׁוּמָה וְלַגִּדּוּל וּלְנֶגַע הַכֶּתֶם הַבָּהִיר.
נז לְהוֹרֹ֕ת בְּי֥וֹם הַטָּמֵ֖א וּבְי֣וֹם הַטָּהֹ֑ר זֹ֥את תּוֹרַ֖ת הַצָּרָֽעַת׃ עַל פִּיהֶם יֵדַע הַכֹּהֵן לְהוֹרוֹת בְּיוֹם שֶׁיִּרְאֶה טָמֵא וּבְיוֹם שֶׁיִּרְאֶה טָהוֹר, אֵלֶּה דִּינֵי הַצָּרַעַת.
פ פָּרָשַׁת הַזָּב
טו א וַיְדַבֵּ֣ר יְהֹוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאֶל אַהֲרֹן וְאָמַר:
ב דַּבְּרוּ֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וַאֲמַרְתֶּ֖ם אֲלֵהֶ֑ם אִ֣ישׁ אִ֗ישׁ כִּ֤י יִהְיֶה֙ זָ֣ב מִבְּשָׂר֔וֹ זוֹב֖וֹ טָמֵ֥א הֽוּא׃ דַּבְּרוּ אֶל בְּנֵי יִשְׂרָאֵל וְאִמְרוּ לָהֶם: כָּל אִישׁ שֶׁיֵּצֵא מִגּוּפוֹ נוֹזֵל חָרִיג, הַנּוֹזֵל הַהוּא טָמֵא.
ג וְזֹ֛את תִּהְיֶ֥ה טֻמְאָת֖וֹ בְּזוֹב֑וֹ רָ֣ר בְּשָׂר֞וֹ אֶת־זוֹב֗וֹ אֽוֹ־הֶחְתִּ֤ים בְּשָׂרוֹ֙ מִזּוֹב֔וֹ טֻמְאָת֖וֹ הִֽוא׃ וְכָךְ תִּהְיֶה טֻמְאָתוֹ בְּזִיבָתוֹ: כְּשֶׁנּוֹטֵף מִגּוּפוֹ נוֹזֵל זִיבָה, אוֹ שֶׁנִּסְתַּם נֶקֶב גּוּפוֹ מִנּוֹזְלָיו – זוֹ טֻמְאָתוֹ.
ד כׇּל־הַמִּשְׁכָּ֗ב אֲשֶׁ֨ר יִשְׁכַּ֥ב עָלָ֛יו הַזָּ֖ב יִטְמָ֑א וְכׇֽל־הַכְּלִ֛י אֲשֶׁר־יֵשֵׁ֥ב עָלָ֖יו יִטְמָֽא׃ כָּל דָּבָר שֶׁעָשׂוּי לִשְׁכִיבָה שֶׁהַזָּב יִשְׁכַּב עָלָיו – יִטְמָא, וְכָל כְּלִי שֶׁעָשׂוּי לִישִׁיבָה שֶׁיֵּשֵׁב עָלָיו – יִטְמָא.
ה וְאִ֕ישׁ אֲשֶׁ֥ר יִגַּ֖ע בְּמִשְׁכָּב֑וֹ יְכַבֵּ֧ס בְּגָדָ֛יו וְרָחַ֥ץ בַּמַּ֖יִם וְטָמֵ֥א עַד־הָעָֽרֶב׃ וְאִישׁ שֶׁיִּגַּע בְּמִשְׁכָּבוֹ יַטְבִּיל אֶת בְּגָדָיו וְיִטְבּוֹל בְּמַיִם וְיִהְיֶה טָמֵא עַד הָעֶרֶב.
ו וְהַיֹּשֵׁב֙ עַֽל־הַכְּלִ֔י אֲשֶׁר־יֵשֵׁ֥ב עָלָ֖יו הַזָּ֑ב יְכַבֵּ֧ס בְּגָדָ֛יו וְרָחַ֥ץ בַּמַּ֖יִם וְטָמֵ֥א עַד־הָעָֽרֶב׃ וּמִי שֶׁיּוֹשֵׁב עַל כְּלִי שֶׁהַזָּב יָשַׁב עָלָיו יַטְבִּיל אֶת בְּגָדָיו וְיִטְבּוֹל בְּמַיִם וְיִהְיֶה טָמֵא עַד הָעֶרֶב.
ז וְהַנֹּגֵ֖עַ בִּבְשַׂ֣ר הַזָּ֑ב יְכַבֵּ֧ס בְּגָדָ֛יו וְרָחַ֥ץ בַּמַּ֖יִם וְטָמֵ֥א עַד־הָעָֽרֶב׃ וּמִי שֶׁנּוֹגֵעַ בְּגוּף הַזָּב יַטְבִּיל אֶת בְּגָדָיו וְיִטְבּוֹל בְּמַיִם וְיִהְיֶה טָמֵא עַד הָעֶרֶב.
ח וְכִֽי־יָרֹ֥ק הַזָּ֖ב בַּטָּה֑וֹר וְכִבֶּ֧ס בְּגָדָ֛יו וְרָחַ֥ץ בַּמַּ֖יִם וְטָמֵ֥א עַד־הָעָֽרֶב׃ וְאִם הַזָּב יִירַק בְּטָהוֹר הוּא יַטְבִּיל אֶת בְּגָדָיו וְיִטְבּוֹל בְּמַיִם וְיִהְיֶה טָמֵא עַד הָעֶרֶב.
ט וְכׇל־הַמֶּרְכָּ֗ב אֲשֶׁ֨ר יִרְכַּ֥ב עָלָ֛יו הַזָּ֖ב יִטְמָֽא׃ וְכָל דָּבָר שֶׁהַזָּב יִרְכַּב עָלָיו – יִטְמָא.
י וְכׇל־הַנֹּגֵ֗עַ בְּכֹל֙ אֲשֶׁ֣ר יִהְיֶ֣ה תַחְתָּ֔יו יִטְמָ֖א עַד־הָעָ֑רֶב וְהַנּוֹשֵׂ֣א אוֹתָ֔ם יְכַבֵּ֧ס בְּגָדָ֛יו וְרָחַ֥ץ בַּמַּ֖יִם וְטָמֵ֥א עַד־הָעָֽרֶב׃ וְכָל הַנּוֹגֵעַ בְּכָל דָּבָר שֶׁהָיָה מִתַּחַת לַזָּב יִטְמָא עַד הָעֶרֶב, וּמִי שֶׁמֵּרִים אוֹתָם יַטְבִּיל אֶת בְּגָדָיו וְיִטְבּוֹל בְּמַיִם וְיִהְיֶה טָמֵא עַד הָעֶרֶב.
יא וְכֹ֨ל אֲשֶׁ֤ר יִגַּע־בּוֹ֙ הַזָּ֔ב וְיָדָ֖יו לֹא־שָׁטַ֣ף בַּמָּ֑יִם וְכִבֶּ֧ס בְּגָדָ֛יו וְרָחַ֥ץ בַּמַּ֖יִם וְטָמֵ֥א עַד־הָעָֽרֶב׃ וְכָל מִי שֶׁיִּגַּע בּוֹ הַזָּב שֶׁלֹּא שָׁטַף אֶת יָדָיו בְּמַיִם – יַטְבִּיל אֶת בְּגָדָיו וְיִטְבּוֹל בְּמַיִם וְיִהְיֶה טָמֵא עַד הָעֶרֶב.
יב וּכְלִי־חֶ֛רֶשׂ אֲשֶׁר־יִגַּע־בּ֥וֹ הַזָּ֖ב יִשָּׁבֵ֑ר וְכׇ֨ל־כְּלִי־עֵ֔ץ יִשָּׁטֵ֖ף בַּמָּֽיִם׃ וּכְלִי חֶרֶס שֶׁיִּגַּע בּוֹ הַזָּב יִשָּׁבֵר, וְכָל כְּלִי עֵץ יוּטְבַּל בְּמַיִם.
פָּרָשַׁת טַהֲרַת הַזָּב
יג וְכִֽי־יִטְהַ֤ר הַזָּב֙ מִזּוֹב֔וֹ וְסָ֨פַר ל֜וֹ שִׁבְעַ֥ת יָמִ֛ים לְטׇהֳרָת֖וֹ וְכִבֶּ֣ס בְּגָדָ֑יו וְרָחַ֧ץ בְּשָׂר֛וֹ בְּמַ֥יִם חַיִּ֖ים וְטָהֵֽר׃ וּכְשֶׁזָּב יֵרָפֵא מִזִּיבָתוֹ הוּא יִסְפּוֹר שִׁבְעָה יָמִים לְטָהֳרָתוֹ וְיַטְבִּיל אֶת בְּגָדָיו, וְיַטְבִּיל אֶת גּוּפוֹ בְּמַיִם נוֹבְעִים, וְאָז יִהְיֶה טָהוֹר.
יד וּבַיּ֣וֹם הַשְּׁמִינִ֗י יִֽקַּֽח־לוֹ֙ שְׁתֵּ֣י תֹרִ֔ים א֥וֹ שְׁנֵ֖י בְּנֵ֣י יוֹנָ֑ה וּבָ֣א ׀ לִפְנֵ֣י יְהֹוָ֗ה אֶל־פֶּ֙תַח֙ אֹ֣הֶל מוֹעֵ֔ד וּנְתָנָ֖ם אֶל־הַכֹּהֵֽן׃ וּבַיּוֹם הַשְּׁמִינִי יִקַּח שְׁתֵּי תּוֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה, וְיָבוֹא אֶל לִפְנֵי ה' אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת וְיִתֵּן אוֹתָם לַכֹּהֵן.
טו וְעָשָׂ֤ה אֹתָם֙ הַכֹּהֵ֔ן אֶחָ֣ד חַטָּ֔את וְהָאֶחָ֖ד עֹלָ֑ה וְכִפֶּ֨ר עָלָ֧יו הַכֹּהֵ֛ן לִפְנֵ֥י יְהֹוָ֖ה מִזּוֹבֽוֹ׃ הַכֹּהֵן יַעֲשֶׂה אוֹתָם אֶחָד חַטָּאת וְאֶחָד עוֹלָה, וִיכַפֵּר עָלָיו לִפְנֵי ה' מִטֻּמְאַת זִיבָתוֹ.

ששי (שביעי במחוברין)[עריכה]

ס פָּרָשַׁת טוּמְאַת זֶרַע
טז וְאִ֕ישׁ כִּֽי־תֵצֵ֥א מִמֶּ֖נּוּ שִׁכְבַת־זָ֑רַע וְרָחַ֥ץ בַּמַּ֛יִם אֶת־כׇּל־בְּשָׂר֖וֹ וְטָמֵ֥א עַד־הָעָֽרֶב׃ וְאִישׁ שֶׁיֵּצֵא מִמֶּנּוּ נוֹזֵל הַזֶּרַע יַטְבִּיל בְּמַיִם אֶת כָּל גּוּפוֹ וְיִהְיֶה טָמֵא עַד הָעֶרֶב.
יז וְכׇל־בֶּ֣גֶד וְכׇל־ע֔וֹר אֲשֶׁר־יִהְיֶ֥ה עָלָ֖יו שִׁכְבַת־זָ֑רַע וְכֻבַּ֥ס בַּמַּ֖יִם וְטָמֵ֥א עַד־הָעָֽרֶב׃ וְכָל בֶּגֶד וְכָל עוֹר שֶׁיִּהְיֶה עָלָיו נוֹזֵל הַזֶּרַע יוּטְבַּל בְּמַיִם וְיִהְיֶה טָמֵא עַד הָעֶרֶב.
יח וְאִשָּׁ֕ה אֲשֶׁ֨ר יִשְׁכַּ֥ב אִ֛ישׁ אֹתָ֖הּ שִׁכְבַת־זָ֑רַע וְרָחֲצ֣וּ בַמַּ֔יִם וְטָמְא֖וּ עַד־הָעָֽרֶב׃ וְאִשָּׁה שֶׁאִישׁ יִשְׁכַּב אִתָּהּ וְיִתֵּן בָּהּ אֶת נוֹזֵל הַזֶּרַע, הֵם יִטְבְּלוּ בְּמַיִם וְיִהְיוּ טְמֵאִים עַד הָעֶרֶב.
פ פָּרָשַׁת נִדָּה
יט וְאִשָּׁה֙ כִּֽי־תִהְיֶ֣ה זָבָ֔ה דָּ֛ם יִהְיֶ֥ה זֹבָ֖הּ בִּבְשָׂרָ֑הּ שִׁבְעַ֤ת יָמִים֙ תִּהְיֶ֣ה בְנִדָּתָ֔הּ וְכׇל־הַנֹּגֵ֥עַ בָּ֖הּ יִטְמָ֥א עַד־הָעָֽרֶב׃ וְאִשָּׁה כְּשֶׁתִּהְיֶה זָבָה – שֶׁיִּנְזוֹל דָּם מִגּוּפָה, הִיא תִּהְיֶה מְנֻדָּה בְּטֻמְאָתָהּ שִׁבְעָה יָמִים וְכָל מִי שֶׁנּוֹגֵעַ בָּהּ יִטְמָא עַד הָעֶרֶב.
כ וְכֹל֩ אֲשֶׁ֨ר תִּשְׁכַּ֥ב עָלָ֛יו בְּנִדָּתָ֖הּ יִטְמָ֑א וְכֹ֛ל אֲשֶׁר־תֵּשֵׁ֥ב עָלָ֖יו יִטְמָֽא׃ וְכָל מַה שֶּׁתִּשְׁכַּב עָלָיו בְּנִדָּתָהּ יִטְמָא, וְכָל מַה שֶּׁתֵּשֵׁב עָלָיו יִטְמָא.
כא וְכׇל־הַנֹּגֵ֖עַ בְּמִשְׁכָּבָ֑הּ יְכַבֵּ֧ס בְּגָדָ֛יו וְרָחַ֥ץ בַּמַּ֖יִם וְטָמֵ֥א עַד־הָעָֽרֶב׃ וְכָל מִי שֶׁנּוֹגֵעַ בְּמַה שֶּׁשָּׁכְבָה עָלָיו יַטְבִּיל אֶת בְּגָדָיו וְיִטְבּוֹל בְּמַיִם וְיִהְיֶה טָמֵא עַד הָעֶרֶב.
כב וְכׇ֨ל־הַנֹּגֵ֔עַ בְּכׇל־כְּלִ֖י אֲשֶׁר־תֵּשֵׁ֣ב עָלָ֑יו יְכַבֵּ֧ס בְּגָדָ֛יו וְרָחַ֥ץ בַּמַּ֖יִם וְטָמֵ֥א עַד־הָעָֽרֶב׃ וְכָל מִי שֶׁנּוֹגֵעַ בְּכָל כְּלִי שֶׁהִיא תֵּשֵׁב עָלָיו – יַטְבִּיל אֶת בְּגָדָיו וְיִטְבּוֹל בְּמַיִם וְיִהְיֶה טָמֵא עַד הָעֶרֶב.
כג וְאִ֨ם עַֽל־הַמִּשְׁכָּ֜ב ה֗וּא א֧וֹ עַֽל־הַכְּלִ֛י אֲשֶׁר־הִ֥וא יֹשֶֽׁבֶת־עָלָ֖יו בְּנׇגְעוֹ־ב֑וֹ יִטְמָ֖א עַד־הָעָֽרֶב׃ וְאִם הוּא נִשְׁעָן עַל מַה שֶּׁהִיא שׁוֹכֶבֶת עָלָיו אוֹ עַל הַכְּלִי שֶׁהִיא יוֹשֶׁבֶת עָלָיו, אוֹ שֶׁהוּא נוֹגֵעַ בּוֹ, יִטְמָא עַד הָעֶרֶב.
כד וְאִ֡ם שָׁכֹב֩ יִשְׁכַּ֨ב אִ֜ישׁ אֹתָ֗הּ וּתְהִ֤י נִדָּתָהּ֙ עָלָ֔יו וְטָמֵ֖א שִׁבְעַ֣ת יָמִ֑ים וְכׇל־הַמִּשְׁכָּ֛ב אֲשֶׁר־יִשְׁכַּ֥ב עָלָ֖יו יִטְמָֽא׃ וְאִם אִישׁ יִשְׁכַּב אִתָּהּ, טֻמְאַת נִדָּתָהּ תִּהְיֶה עָלָיו וְהוּא יִטְמָא שִׁבְעָה יָמִים, וְכָל דָּבָר שֶׁעָשׂוּי לִשְׁכִיבָה שֶׁהוּא יִשְׁכַּב עָלָיו – יִטְמָא.
ס פָּרָשַׁת זָבָה
כה וְאִשָּׁ֡ה כִּֽי־יָזוּב֩ ז֨וֹב דָּמָ֜הּ יָמִ֣ים רַבִּ֗ים בְּלֹא֙ עֶת־נִדָּתָ֔הּ א֥וֹ כִֽי־תָז֖וּב עַל־נִדָּתָ֑הּ כׇּל־יְמֵ֞י ז֣וֹב טֻמְאָתָ֗הּ כִּימֵ֧י נִדָּתָ֛הּ תִּהְיֶ֖ה טְמֵאָ֥ה הִֽוא׃ וְאִשָּׁה שֶׁדָּמָהּ יֵצֵא בְּמֶשֶׁךְ יָמִים רַבִּים שֶׁלֹּא בִּזְמַן נִדָּתָהּ, אוֹ שֶׁיַּמְשִׁיךְ לָצֵאת יוֹתֵר מִזְּמַן נִדָּתָהּ, בְּכָל הַיָּמִים שֶׁיּוֹצֵא מִמֶּנָּה דַּם טֻמְאָתָהּ הִיא תִּהְיֶה כְּמוֹ בִּימֵי נִדָּתָהּ – הִיא טְמֵאָה.
כו כׇּל־הַמִּשְׁכָּ֞ב אֲשֶׁר־תִּשְׁכַּ֤ב עָלָיו֙ כׇּל־יְמֵ֣י זוֹבָ֔הּ כְּמִשְׁכַּ֥ב נִדָּתָ֖הּ יִֽהְיֶה־לָּ֑הּ וְכׇֽל־הַכְּלִי֙ אֲשֶׁ֣ר תֵּשֵׁ֣ב עָלָ֔יו טָמֵ֣א יִהְיֶ֔ה כְּטֻמְאַ֖ת נִדָּתָֽהּ׃ כָּל דָּבָר שֶׁעָשׂוּי לִשְׁכִיבָה שֶׁהִיא תִּשְׁכַּב עָלָיו בְּכָל הַיָּמִים שֶׁיּוֹצֵא מִמֶּנָּה דָּם – יִהְיֶה טָמֵא כְּמוֹ מִשְׁכַּב נִדָּתָהּ, וְכָל כְּלִי שֶׁתֵּשֵׁב עָלָיו – יִהְיֶה טָמֵא כְּטֻמְאַת נִדָּתָהּ.
כז וְכׇל־הַנּוֹגֵ֥עַ בָּ֖ם יִטְמָ֑א וְכִבֶּ֧ס בְּגָדָ֛יו וְרָחַ֥ץ בַּמַּ֖יִם וְטָמֵ֥א עַד־הָעָֽרֶב׃ וְכָל מִי שֶׁנּוֹגֵעַ בָּהֶם יִטְמָא, וְיַטְבִּיל אֶת בְּגָדָיו וְיִטְבּוֹל בְּמַיִם וְיִהְיֶה טָמֵא עַד הָעֶרֶב.
פָּרָשַׁת טַהֲרַת הַזָּבָה
כח וְאִֽם־טָהֲרָ֖ה מִזּוֹבָ֑הּ וְסָ֥פְרָה לָּ֛הּ שִׁבְעַ֥ת יָמִ֖ים וְאַחַ֥ר תִּטְהָֽר׃ וְאִם נִתְרַפְּאָה מִזִּיבָתָהּ הִיא תִּסְפּוֹר שִׁבְעָה יָמִים וְאַחַר כָּךְ תִּטְהַר.

שביעי[עריכה]

כט וּבַיּ֣וֹם הַשְּׁמִינִ֗י תִּֽקַּֽח־לָהּ֙ שְׁתֵּ֣י תֹרִ֔ים א֥וֹ שְׁנֵ֖י בְּנֵ֣י יוֹנָ֑ה וְהֵבִיאָ֤ה אוֹתָם֙ אֶל־הַכֹּהֵ֔ן אֶל־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וּבַיּוֹם הַשְּׁמִינִי תִּקַּח שְׁתֵּי תּוֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה, וְתָבִיא אוֹתָם לַכֹּהֵן אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.
ל וְעָשָׂ֤ה הַכֹּהֵן֙ אֶת־הָאֶחָ֣ד חַטָּ֔את וְאֶת־הָאֶחָ֖ד עֹלָ֑ה וְכִפֶּ֨ר עָלֶ֤יהָ הַכֹּהֵן֙ לִפְנֵ֣י יְהֹוָ֔ה מִזּ֖וֹב טֻמְאָתָֽהּ׃ הַכֹּהֵן יַעֲשֶׂה אֶת הָאֶחָד חַטָּאת וְאֶת הָאֶחָד עוֹלָה, וּבָזֶה יְכַפֵּר עָלֶיהָ הַכֹּהֵן לִפְנֵי ה' מִטֻּמְאַת זִיבָתָהּ.

מפטיר[עריכה]

לא וְהִזַּרְתֶּ֥ם אֶת־בְּנֵי־יִשְׂרָאֵ֖ל מִטֻּמְאָתָ֑ם וְלֹ֤א יָמֻ֙תוּ֙ בְּטֻמְאָתָ֔ם בְּטַמְּאָ֥ם אֶת־מִשְׁכָּנִ֖י אֲשֶׁ֥ר בְּתוֹכָֽם׃ הַרְחִיקוּ אֶת בְּנֵי יִשְׂרָאֵל מִטֻּמְאָתָם, כְּדֵי שֶׁלֹּא יָמוּתוּ בְּטֻמְאָתָם אִם יְטַמְּאוּ אֶת שְׁכִינָתִי שֶׁבְּתוֹכָם.
לב זֹ֥את תּוֹרַ֖ת הַזָּ֑ב וַאֲשֶׁ֨ר תֵּצֵ֥א מִמֶּ֛נּוּ שִׁכְבַת־זֶ֖רַע לְטׇמְאָה־בָֽהּ׃ אֵלֶּה דִּינֵי הַזָּב וּמִי שֶׁיֵּצֵא מִמֶּנּוּ נוֹזֵל הַזֶּרַע וְיִטָּמֵא בּוֹ,
לג וְהַדָּוָה֙ בְּנִדָּתָ֔הּ וְהַזָּב֙ אֶת־זוֹב֔וֹ לַזָּכָ֖ר וְלַנְּקֵבָ֑ה וּלְאִ֕ישׁ אֲשֶׁ֥ר יִשְׁכַּ֖ב עִם־טְמֵאָֽה׃ וְהַטְּמֵאָה בְּהַרְחָקָתָהּ, וּמִי שֶׁנּוֹזְלִים מִמֶּנּוּ נוֹזְלָיו זָכָר אוֹ נְקֵבָה, וְאִישׁ שֶׁיִּשְׁכַּב עִם אִשָּׁה טְמֵאָה.
