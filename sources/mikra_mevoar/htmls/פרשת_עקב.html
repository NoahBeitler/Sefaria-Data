פ פָּרָשַׁת הַבְּרָכוֹת
ז יב וְהָיָ֣ה ׀ עֵ֣קֶב תִּשְׁמְע֗וּן אֵ֤ת הַמִּשְׁפָּטִים֙ הָאֵ֔לֶּה וּשְׁמַרְתֶּ֥ם וַעֲשִׂיתֶ֖ם אֹתָ֑ם וְשָׁמַר֩ יְהֹוָ֨ה אֱלֹהֶ֜יךָ לְךָ֗ אֶֽת־הַבְּרִית֙ וְאֶת־הַחֶ֔סֶד אֲשֶׁ֥ר נִשְׁבַּ֖ע לַאֲבֹתֶֽיךָ׃ בְּעִקְבוֹת שֶׁתִּשְׁמְעוּ בְּקוֹל הַמִּשְׁפָּטִים הָאֵלֶּה וְתַקְפִּידוּ לַעֲשׂוֹת אוֹתָם, ה' אֱלֹהֶיךָ יְקַיֵּם לְךָ אֶת הַהֶסְכֵּם וְאֶת הַחֲסָדִים כְּפִי שֶׁנִּשְׁבַּע לַאֲבוֹתֶיךָ.
יג וַאֲהֵ֣בְךָ֔ וּבֵרַכְךָ֖ וְהִרְבֶּ֑ךָ וּבֵרַ֣ךְ פְּרִֽי־בִטְנְךָ֣ וּפְרִֽי־אַ֠דְמָתֶ֠ךָ דְּגָ֨נְךָ֜ וְתִירֹֽשְׁךָ֣ וְיִצְהָרֶ֗ךָ שְׁגַר־אֲלָפֶ֙יךָ֙ וְעַשְׁתְּרֹ֣ת צֹאנֶ֔ךָ עַ֚ל הָֽאֲדָמָ֔ה אֲשֶׁר־נִשְׁבַּ֥ע לַאֲבֹתֶ֖יךָ לָ֥תֶת לָֽךְ׃ הוּא יֹאהַב אוֹתְךָ וִיבָרֵךְ וְיַרְבֶּה אוֹתְךָ, וְיָשִׂים בְּרָכָה בִּילָדֶיךָ, פֵּרוֹת אַדְמָתְךָ – הַתְּבוּאָה הַיַּיִן וְהַשֶּׁמֶן שֶׁלְּךָ, וְלָדוֹת שְׁוָרֶיךָ וְעֶדְרֵי צֹאנְךָ, עַל הָאֲדָמָה שֶׁנִּשְׁבַּע לַאֲבוֹתֶיךָ לָתֵת לְךָ.
יד בָּר֥וּךְ תִּֽהְיֶ֖ה מִכׇּל־הָעַמִּ֑ים לֹא־יִהְיֶ֥ה בְךָ֛ עָקָ֥ר וַֽעֲקָרָ֖ה וּבִבְהֶמְתֶּֽךָ׃ תִּהְיֶה מְבוֹרָךְ יוֹתֵר מִכָּל הָעַמִּים, לֹא יִהְיֶה בְּךָ וּבִבְהֶמְתֶּךָ מִי שֶׁאֵינוֹ יָכוֹל לְהוֹלִיד אוֹ שֶׁאֵינָהּ יְכוֹלָה לָלֶדֶת.
טו וְהֵסִ֧יר יְהֹוָ֛ה מִמְּךָ֖ כׇּל־חֹ֑לִי וְכׇל־מַדְוֵי֩ מִצְרַ֨יִם הָרָעִ֜ים אֲשֶׁ֣ר יָדַ֗עְתָּ לֹ֤א יְשִׂימָם֙ בָּ֔ךְ וּנְתָנָ֖ם בְּכׇל־שֹׂנְאֶֽיךָ׃ ה' יָסִיר מִמְּךָ אֶת כָּל הַמַּחֲלוֹת, וְלֹא יָשִׂים בְּךָ אֶת כָּל מַחֲלוֹת מִצְרַיִם הָרָעוֹת שֶׁהִכַּרְתָּ, אֶלָּא יָשִׂים אוֹתָם בְּכָל שׂוֹנְאֶיךָ.
טז וְאָכַלְתָּ֣ אֶת־כׇּל־הָֽעַמִּ֗ים אֲשֶׁ֨ר יְהֹוָ֤ה אֱלֹהֶ֙יךָ֙ נֹתֵ֣ן לָ֔ךְ לֹא־תָח֥וֹס עֵֽינְךָ֖ עֲלֵיהֶ֑ם וְלֹ֤א תַעֲבֹד֙ אֶת־אֱלֹ֣הֵיהֶ֔ם כִּֽי־מוֹקֵ֥שׁ ה֖וּא לָֽךְ׃ אַתָּה תִּכְבּוֹשׁ אֶת כָּל הָעַמִּים שֶׁה' אֱלֹהֶיךָ נוֹתֵן לְךָ, אַל תְּרַחֵם עֲלֵיהֶם, וְאַל תַּעֲבוֹד אֶת אֱלֹהֵיהֶם כִּי הֵם מַלְכּוֹדֶת עֲבוּרְךָ.
ס פָּרָשַׁת הוֹרָשַׁת הַגּוֹיִם
יז כִּ֤י תֹאמַר֙ בִּלְבָ֣בְךָ֔ רַבִּ֛ים הַגּוֹיִ֥ם הָאֵ֖לֶּה מִמֶּ֑נִּי אֵיכָ֥ה אוּכַ֖ל לְהוֹרִישָֽׁם׃ אִם יַעֲלֶה עַל דַּעַתְךָ כָּךְ: הָעַמִּים הָאֵלֶּה רַבִּים יוֹתֵר מִמֶּנִּי, אֵיךְ אוּכַל לִכְבּוֹשׁ אוֹתָם?!
יח לֹ֥א תִירָ֖א מֵהֶ֑ם זָכֹ֣ר תִּזְכֹּ֗ר אֵ֤ת אֲשֶׁר־עָשָׂה֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ לְפַרְעֹ֖ה וּלְכׇל־מִצְרָֽיִם׃ אַל תִּפְחַד מֵהֶם! זְכוֹר אֶת מַה שֶּׁה' אֱלֹהֶיךָ עָשָׂה לְפַרְעֹה וּלְכָל מִצְרַיִם –
יט הַמַּסֹּ֨ת הַגְּדֹלֹ֜ת אֲשֶׁר־רָא֣וּ עֵינֶ֗יךָ וְהָאֹתֹ֤ת וְהַמֹּֽפְתִים֙ וְהַיָּ֤ד הַחֲזָקָה֙ וְהַזְּרֹ֣עַ הַנְּטוּיָ֔ה אֲשֶׁ֥ר הוֹצִֽאֲךָ֖ יְהֹוָ֣ה אֱלֹהֶ֑יךָ כֵּֽן־יַעֲשֶׂ֞ה יְהֹוָ֤ה אֱלֹהֶ֙יךָ֙ לְכׇל־הָ֣עַמִּ֔ים אֲשֶׁר־אַתָּ֥ה יָרֵ֖א מִפְּנֵיהֶֽם׃ אֶת הַנִּסִּים הַגְּדוֹלִים שֶׁעֵינֶיךָ רָאוּ, וְהַסִּימָנִים וְהַהוֹכָחוֹת, וְהַיָּד הַחֲזָקָה וְהַזְּרוֹעַ הַמּוּנֶפֶת לְהַכּוֹת, שֶׁה' אֱלֹהֶיךָ הוֹצִיא אוֹתְךָ בְּעֶזְרָתָם, כָּךְ ה' אֱלֹהֶיךָ יַעֲשֶׂה לְכָל הָעַמִּים שֶׁאַתָּה פּוֹחֵד מֵהֶם.
כ וְגַם֙ אֶת־הַצִּרְעָ֔ה יְשַׁלַּ֛ח יְהֹוָ֥ה אֱלֹהֶ֖יךָ בָּ֑ם עַד־אֲבֹ֗ד הַנִּשְׁאָרִ֛ים וְהַנִּסְתָּרִ֖ים מִפָּנֶֽיךָ׃ וְה' אֱלֹהֶיךָ יְשַׁסֶּה בָּהֶם גַּם אֶת הַצְּרָעוֹת, עַד שֶׁהַנִּשְׁאָרִים וְהַמִּסְתַּתְּרִים מִפָּנֶיךָ יֹאבְדוּ.
כא לֹ֥א תַעֲרֹ֖ץ מִפְּנֵיהֶ֑ם כִּֽי־יְהֹוָ֤ה אֱלֹהֶ֙יךָ֙ בְּקִרְבֶּ֔ךָ אֵ֥ל גָּד֖וֹל וְנוֹרָֽא׃ אַל תִּשָּׁבֵר מִפְּנֵיהֶם, כִּי ה' אֱלֹהֶיךָ אִתְּךָ, הוּא אֵל גָּדוֹל וְנוֹרָא.
כב וְנָשַׁל֩ יְהֹוָ֨ה אֱלֹהֶ֜יךָ אֶת־הַגּוֹיִ֥ם הָאֵ֛ל מִפָּנֶ֖יךָ מְעַ֣ט מְעָ֑ט לֹ֤א תוּכַל֙ כַּלֹּתָ֣ם מַהֵ֔ר פֶּן־תִּרְבֶּ֥ה עָלֶ֖יךָ חַיַּ֥ת הַשָּׂדֶֽה׃ ה' אֱלֹהֶיךָ יָסִיר אֶת הָעַמִּים הָאֵלֶּה מִפָּנֶיךָ מְעַט מְעַט, לֹא תּוּכַל לְהַשְׁמִיד אוֹתָם מַהֵר כְּדֵי שֶׁחַיַּת הַשָּׂדֶה לֹא תִּתְרַבֶּה אֶצְלְךָ כְּשֶׁהָאָרֶץ תִּהְיֶה רֵיקָה מִתּוֹשָׁבִים.
כג וּנְתָנָ֛ם יְהֹוָ֥ה אֱלֹהֶ֖יךָ לְפָנֶ֑יךָ וְהָמָם֙ מְהוּמָ֣ה גְדֹלָ֔ה עַ֖ד הִשָּׁמְדָֽם׃ ה' אֱלֹהֶיךָ יִמְסוֹר אוֹתָם לְפָנֶיךָ, וִיבַלְבֵּל אוֹתָם בִּמְהוּמָה גְּדוֹלָה עַד שֶׁיֻּשְׁמְדוּ.
כד וְנָתַ֤ן מַלְכֵיהֶם֙ בְּיָדֶ֔ךָ וְהַאֲבַדְתָּ֣ אֶת־שְׁמָ֔ם מִתַּ֖חַת הַשָּׁמָ֑יִם לֹֽא־יִתְיַצֵּ֥ב אִישׁ֙ בְּפָנֶ֔יךָ עַ֥ד הִשְׁמִֽדְךָ֖ אֹתָֽם׃ הוּא יִתֵּן אֶת מַלְכֵיהֶם בְּיָדְךָ, וְתַשְׁמִיד אֶת שְׁמָם מִתַּחַת לַשָּׁמַיִם, אִישׁ לֹא יַעֲמוֹד בְּפָנֶיךָ עַד שֶׁתַּשְׁמִיד אוֹתָם.
כה פְּסִילֵ֥י אֱלֹהֵיהֶ֖ם תִּשְׂרְפ֣וּן בָּאֵ֑שׁ לֹֽא־תַחְמֹד֩ כֶּ֨סֶף וְזָהָ֤ב עֲלֵיהֶם֙ וְלָקַחְתָּ֣ לָ֔ךְ פֶּ֚ן תִּוָּקֵ֣שׁ בּ֔וֹ כִּ֧י תוֹעֲבַ֛ת יְהֹוָ֥ה אֱלֹהֶ֖יךָ הֽוּא׃ אֶת פִּסְלֵי אֱלֹהֵיהֶם תִּשְׂרְפוּ בָּאֵשׁ, אַל תַּחְשׁוֹק בַּכֶּסֶף וּבַזָּהָב שֶׁעֲלֵיהֶם לָקַחַת אוֹתוֹ לְעַצְמְךָ כְּדֵי שֶׁלֹּא תִּכָּשֵׁל בּוֹ, כִּי הוּא תּוֹעֲבַת ה' אֱלֹהֶיךָ.
כו וְלֹא־תָבִ֤יא תֽוֹעֵבָה֙ אֶל־בֵּיתֶ֔ךָ וְהָיִ֥יתָ חֵ֖רֶם כָּמֹ֑הוּ שַׁקֵּ֧ץ ׀ תְּשַׁקְּצֶ֛נּוּ וְתַעֵ֥ב ׀ תְּֽתַעֲבֶ֖נּוּ כִּי־חֵ֥רֶם הֽוּא׃ וְאַל תָּבִיא אֱלִיל תּוֹעֵבָה לְבֵיתְךָ שֶׁלֹּא תִּתְחַיֵּב כְּלָיָה כָּמוֹהוּ, אֶלָּא תִּמְאַס אוֹתוֹ וְתִשְׂנָא אוֹתוֹ כִּי חוֹבָה לְהַשְׁמִידוֹ.
פ פָּרָשַׁת זְכִירַת הַשֵּׁם
ח א כׇּל־הַמִּצְוָ֗ה אֲשֶׁ֨ר אָנֹכִ֧י מְצַוְּךָ֛ הַיּ֖וֹם תִּשְׁמְר֣וּן לַעֲשׂ֑וֹת לְמַ֨עַן תִּֽחְי֜וּן וּרְבִיתֶ֗ם וּבָאתֶם֙ וִֽירִשְׁתֶּ֣ם אֶת־הָאָ֔רֶץ אֲשֶׁר־נִשְׁבַּ֥ע יְהֹוָ֖ה לַאֲבֹתֵיכֶֽם׃ הַקְפִּידוּ לְקַיֵּם אֶת כָּל הַמִּצְווֹת שֶׁאֲנִי מְצַוֶּה אֶתְכֶם הַיּוֹם, כְּדֵי שֶׁתִּחְיוּ וְתִתְרַבּוּ וְתָבוֹאוּ וְתִכְבְּשׁוּ אֶת הָאָרֶץ שֶׁה' נִשְׁבַּע לַאֲבוֹתֵיכֶם.
ב וְזָכַרְתָּ֣ אֶת־כׇּל־הַדֶּ֗רֶךְ אֲשֶׁ֨ר הוֹלִֽיכְךָ֜ יְהֹוָ֧ה אֱלֹהֶ֛יךָ זֶ֛ה אַרְבָּעִ֥ים שָׁנָ֖ה בַּמִּדְבָּ֑ר לְמַ֨עַן עַנֹּֽתְךָ֜ לְנַסֹּֽתְךָ֗ לָדַ֜עַת אֶת־אֲשֶׁ֧ר בִּֽלְבָבְךָ֛ הֲתִשְׁמֹ֥ר מִצְוֺתָ֖ו אִם־לֹֽא׃ זְכוֹר אֶת כָּל הַדֶּרֶךְ שֶׁה' אֱלֹהֶיךָ הוֹלִיךְ אוֹתְךָ זֶה אַרְבָּעִים שָׁנָה בַּמִּדְבָּר, כְּדֵי לְהַכְנִיעַ אוֹתְךָ – לִבְחוֹן אוֹתְךָ לָדַעַת אֶת מַה שֶּׁבְּלִבְּךָ – אִם תִּשְׁמוֹר אֶת מִצְווֹתָיו אוֹ לֹא.
ג וַֽיְעַנְּךָ֮ וַיַּרְעִבֶ֒ךָ֒ וַיַּאֲכִֽלְךָ֤ אֶת־הַמָּן֙ אֲשֶׁ֣ר לֹא־יָדַ֔עְתָּ וְלֹ֥א יָדְע֖וּן אֲבֹתֶ֑יךָ לְמַ֣עַן הוֹדִֽיעֲךָ֗ כִּ֠י לֹ֣א עַל־הַלֶּ֤חֶם לְבַדּוֹ֙ יִחְיֶ֣ה הָֽאָדָ֔ם כִּ֛י עַל־כׇּל־מוֹצָ֥א פִֽי־יְהֹוָ֖ה יִחְיֶ֥ה הָאָדָֽם׃ הוּא הִכְנִיעַ אוֹתְךָ וְהִרְעִיב אוֹתְךָ וְהֶאֱכִיל אוֹתְךָ אֶת הַמָּן שֶׁלֹּא הִכַּרְתָּ וַאֲבוֹתֶיךָ לֹא הִכִּירוּ, כְּדֵי לְהוֹדִיעַ לְךָ שֶׁהָאָדָם לֹא יִחְיֶה רַק עַל הַלֶּחֶם, אֶלָּא הָאָדָם יִחְיֶה עִם כָּל דָּבָר שֶׁה' יֹאמַר.
ד שִׂמְלָ֨תְךָ֜ לֹ֤א בָֽלְתָה֙ מֵֽעָלֶ֔יךָ וְרַגְלְךָ֖ לֹ֣א בָצֵ֑קָה זֶ֖ה אַרְבָּעִ֥ים שָׁנָֽה׃ זֶה אַרְבָּעִים שָׁנָה שֶׁבְּגָדֶיךָ לֹא בָּלוּ עַד שֶׁנִּצְרַכְתָּ לְפָשְׁטָם מֵעָלֶיךָ, וְרַגְלְךָ לֹא הִתְנַפְּחָה.
ה וְיָדַעְתָּ֖ עִם־לְבָבֶ֑ךָ כִּ֗י כַּאֲשֶׁ֨ר יְיַסֵּ֥ר אִישׁ֙ אֶת־בְּנ֔וֹ יְהֹוָ֥ה אֱלֹהֶ֖יךָ מְיַסְּרֶֽךָּ׃ דַּע בְּלִבְּךָ, שֶׁכְּמוֹ שֶׁאִישׁ מְיַסֵּר אֶת בְּנוֹ כָּךְ ה' אֱלֹהֶיךָ מְיַסֵּר אוֹתְךָ.
ו וְשָׁ֣מַרְתָּ֔ אֶת־מִצְוֺ֖ת יְהֹוָ֣ה אֱלֹהֶ֑יךָ לָלֶ֥כֶת בִּדְרָכָ֖יו וּלְיִרְאָ֥ה אֹתֽוֹ׃ שְׁמוֹר אֶת מִצְווֹת ה' אֱלֹהֶיךָ – לָלֶכֶת בִּדְרָכָיו וּלְכַבֵּד אוֹתוֹ.
ז כִּ֚י יְהֹוָ֣ה אֱלֹהֶ֔יךָ מְבִֽיאֲךָ֖ אֶל־אֶ֣רֶץ טוֹבָ֑ה אֶ֚רֶץ נַ֣חֲלֵי מָ֔יִם עֲיָנֹת֙ וּתְהֹמֹ֔ת יֹצְאִ֥ים בַּבִּקְעָ֖ה וּבָהָֽר׃ כִּי ה' אֱלֹהֶיךָ מֵבִיא אוֹתְךָ לְאֶרֶץ טוֹבָה – אֶרֶץ נַחֲלֵי מַיִם מַעֲיָנוֹת וּתְהוֹמוֹת, שֶׁנּוֹבְעִים בַּבִּקְעָה וּבָהָר.
ח אֶ֤רֶץ חִטָּה֙ וּשְׂעֹרָ֔ה וְגֶ֥פֶן וּתְאֵנָ֖ה וְרִמּ֑וֹן אֶֽרֶץ־זֵ֥ית שֶׁ֖מֶן וּדְבָֽשׁ׃ אֶרֶץ חִטָּה וּשְׂעוֹרָה וְגֶפֶן וּתְאֵנָה וְרִמּוֹן, אֶרֶץ זֵיתֵי שֶׁמֶן, וּדְבַשׁ תְּמָרִים.
ט אֶ֗רֶץ אֲשֶׁ֨ר לֹ֤א בְמִסְכֵּנֻת֙ תֹּֽאכַל־בָּ֣הּ לֶ֔חֶם לֹֽא־תֶחְסַ֥ר כֹּ֖ל בָּ֑הּ אֶ֚רֶץ אֲשֶׁ֣ר אֲבָנֶ֣יהָ בַרְזֶ֔ל וּמֵהֲרָרֶ֖יהָ תַּחְצֹ֥ב נְחֹֽשֶׁת׃ אֶרֶץ שֶׁלֹּא תֹּאכַל בָּהּ לֶחֶם בַּעֲנִיּוּת – לֹא יֶחְסַר לְךָ בָּהּ שׁוּם דָּבָר, אֶרֶץ שֶׁיֵּשׁ בַּאֲבָנֶיהָ בַּרְזֶל, וּמֵהָרֶיהָ תַּחְצוֹב נְחוֹשֶׁת.
י וְאָכַלְתָּ֖ וְשָׂבָ֑עְתָּ וּבֵֽרַכְתָּ֙ אֶת־יְהֹוָ֣ה אֱלֹהֶ֔יךָ עַל־הָאָ֥רֶץ הַטֹּבָ֖ה אֲשֶׁ֥ר נָֽתַן־לָֽךְ׃ אַתָּה תֹּאכַל וְתִשְׂבַּע, וּתְבָרֵךְ אֶת ה' אֱלֹהֶיךָ עַל הָאָרֶץ הַטּוֹבָה שֶׁנָּתַן לְךָ.

שני[עריכה]

יא הִשָּׁ֣מֶר לְךָ֔ פֶּן־תִּשְׁכַּ֖ח אֶת־יְהֹוָ֣ה אֱלֹהֶ֑יךָ לְבִלְתִּ֨י שְׁמֹ֤ר מִצְוֺתָיו֙ וּמִשְׁפָּטָ֣יו וְחֻקֹּתָ֔יו אֲשֶׁ֛ר אָנֹכִ֥י מְצַוְּךָ֖ הַיּֽוֹם׃ הִזָּהֵר לְךָ שֶׁלֹּא תִּשְׁכַּח אֶת ה' אֱלֹהֶיךָ, בְּכָךְ שֶׁלֹּא תִּשְׁמוֹר אֶת מִצְווֹתָיו וּמִשְׁפָּטָיו וְחֻקָּיו שֶׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם.
יב פֶּן־תֹּאכַ֖ל וְשָׂבָ֑עְתָּ וּבָתִּ֥ים טֹבִ֛ים תִּבְנֶ֖ה וְיָשָֽׁבְתָּ׃ שֶׁלֹּא תֹּאכַל וְתִשְׂבַּע וְתִבְנֶה בָּתִּים טוֹבִים וְתָגוּר בָּהֶם,
יג וּבְקָֽרְךָ֤ וְצֹֽאנְךָ֙ יִרְבְּיֻ֔ן וְכֶ֥סֶף וְזָהָ֖ב יִרְבֶּה־לָּ֑ךְ וְכֹ֥ל אֲשֶׁר־לְךָ֖ יִרְבֶּֽה׃ וּבְקָרְךָ וְצֹאנְךָ יִתְרַבּוּ, וְיִתְרַבֶּה לְךָ כֶּסֶף וְזָהָב, וְכָל מַה שֶּׁיֵּשׁ לְךָ יִתְרַבֶּה.
יד וְרָ֖ם לְבָבֶ֑ךָ וְשָֽׁכַחְתָּ֙ אֶת־יְהֹוָ֣ה אֱלֹהֶ֔יךָ הַמּוֹצִיאֲךָ֛ מֵאֶ֥רֶץ מִצְרַ֖יִם מִבֵּ֥ית עֲבָדִֽים׃ וְתִתְגָּאֶה בְּלִבְּךָ, וְתִשְׁכַּח אֶת ה' אֱלֹהֶיךָ שֶׁהוֹצִיא אוֹתְךָ מֵאֶרֶץ מִצְרַיִם מִבֵּית עֲבָדִים.
טו הַמּוֹלִ֨יכְךָ֜ בַּמִּדְבָּ֣ר ׀ הַגָּדֹ֣ל וְהַנּוֹרָ֗א נָחָ֤שׁ ׀ שָׂרָף֙ וְעַקְרָ֔ב וְצִמָּא֖וֹן אֲשֶׁ֣ר אֵֽין־מָ֑יִם הַמּוֹצִ֤יא לְךָ֙ מַ֔יִם מִצּ֖וּר הַֽחַלָּמִֽישׁ׃ שֶׁהוֹלִיךְ אוֹתְךָ בַּמִּדְבָּר הַגָּדוֹל וְהַנּוֹרָא שֶׁיֵּשׁ בּוֹ נְחָשִׁים אַרְסִיִּים וְעַקְרַבִּים, וְצִמָּאוֹן – שֶׁאֵין בּוֹ מַיִם, שֶׁהוֹצִיא לְךָ מַיִם מֵהַסֶּלַע הַקָּשֶׁה.
טז הַמַּֽאֲכִ֨לְךָ֥ מָן֙ בַּמִּדְבָּ֔ר אֲשֶׁ֥ר לֹא־יָדְע֖וּן אֲבֹתֶ֑יךָ לְמַ֣עַן עַנֹּֽתְךָ֗ וּלְמַ֙עַן֙ נַסֹּתֶ֔ךָ לְהֵיטִֽבְךָ֖ בְּאַחֲרִיתֶֽךָ׃ שֶׁהֶאֱכִיל אוֹתְךָ בַּמִּדְבָּר מָן שֶׁלֹּא הִכִּירוּ אֲבוֹתֶיךָ, כְּדֵי לְהַכְנִיעַ אוֹתְךָ וּכְדֵי לִבְחוֹן אוֹתְךָ, כְּדֵי שֶׁיּוּכַל לְבַסּוֹף לְהֵיטִיב לְךָ.
יז וְאָמַרְתָּ֖ בִּלְבָבֶ֑ךָ כֹּחִי֙ וְעֹ֣צֶם יָדִ֔י עָ֥שָׂה לִ֖י אֶת־הַחַ֥יִל הַזֶּֽה׃ שֶׁלֹּא תֹּאמַר לְעַצְמְךָ: כּוֹחִי וְעָצְמַת יָדִי עָשׂוּ לִי אֶת הָעוֹשֶׁר הַזֶּה.
יח וְזָֽכַרְתָּ֙ אֶת־יְהֹוָ֣ה אֱלֹהֶ֔יךָ כִּ֣י ה֗וּא הַנֹּתֵ֥ן לְךָ֛ כֹּ֖חַ לַעֲשׂ֣וֹת חָ֑יִל לְמַ֨עַן הָקִ֧ים אֶת־בְּרִית֛וֹ אֲשֶׁר־נִשְׁבַּ֥ע לַאֲבֹתֶ֖יךָ כַּיּ֥וֹם הַזֶּֽה׃ אֶלָּא זְכוֹר אֶת ה' אֱלֹהֶיךָ, שֶׁהוּא שֶׁנּוֹתֵן לְךָ כֹּחַ לַעֲשׂוֹת עוֹשֶׁר, כְּדֵי לְקַיֵּים אֶת הַהֶסְכֵּם שֶׁנִשְׁבַּע לַאֲבוֹתֶיךָ, כְּמוֹ שֶׁהוּא מְקַיֵּם כַּיּוֹם.
פ
יט וְהָיָ֗ה אִם־שָׁכֹ֤חַ תִּשְׁכַּח֙ אֶת־יְהֹוָ֣ה אֱלֹהֶ֔יךָ וְהָֽלַכְתָּ֗ אַחֲרֵי֙ אֱלֹהִ֣ים אֲחֵרִ֔ים וַעֲבַדְתָּ֖ם וְהִשְׁתַּחֲוִ֣יתָ לָהֶ֑ם הַעִדֹ֤תִי בָכֶם֙ הַיּ֔וֹם כִּ֥י אָבֹ֖ד תֹּאבֵדֽוּן׃ אִם תִּשְׁכַּח אֶת ה' אֱלֹהֶיךָ, וְתֵלֵךְ אַחֲרֵי אֱלֹהִים אֲחֵרִים וְתַעֲבוֹד אוֹתָם וְתִשְׁתַּחֲוֶה לָהֶם, אֲנִי מֵעִיד בָּכֶם הַיּוֹם שֶׁתֹּאבְדוּ.
כ כַּגּוֹיִ֗ם אֲשֶׁ֤ר יְהֹוָה֙ מַאֲבִ֣יד מִפְּנֵיכֶ֔ם כֵּ֖ן תֹּאבֵד֑וּן עֵ֚קֶב לֹ֣א תִשְׁמְע֔וּן בְּק֖וֹל יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ כְּמוֹ הָעַמִּים שֶׁה' מַשְׁמִיד מִפְּנֵיכֶם כָּךְ תֹּאבְדוּ, כְּתוֹצָאָה מִכָּךְ שֶׁלֹּא תִּשְׁמְעוּ בְּקוֹל ה' אֱלֹהֵיכֶם.
פ פָּרָשַׁת זְכִירַת הָחֲטָאִים
ט א שְׁמַ֣ע יִשְׂרָאֵ֗ל אַתָּ֨ה עֹבֵ֤ר הַיּוֹם֙ אֶת־הַיַּרְדֵּ֔ן לָבֹא֙ לָרֶ֣שֶׁת גּוֹיִ֔ם גְּדֹלִ֥ים וַעֲצֻמִ֖ים מִמֶּ֑ךָּ עָרִ֛ים גְּדֹלֹ֥ת וּבְצֻרֹ֖ת בַּשָּׁמָֽיִם׃ שְׁמַע יִשְׂרָאֵל, אַתָּה עוֹבֵר הַיּוֹם אֶת הַיַּרְדֵּן לָבוֹא לִכְבּוֹשׁ עַמִּים גְּדוֹלִים וַעֲצוּמִים יוֹתֵר מִמְּךָ, עָרִים גְּדוֹלוֹת וּמְבֻצָּרוֹת בְּחוֹמוֹת – כְּאִלּוּ עַד הַשָּׁמַיִם.
ב עַֽם־גָּד֥וֹל וָרָ֖ם בְּנֵ֣י עֲנָקִ֑ים אֲשֶׁ֨ר אַתָּ֤ה יָדַ֙עְתָּ֙ וְאַתָּ֣ה שָׁמַ֔עְתָּ מִ֣י יִתְיַצֵּ֔ב לִפְנֵ֖י בְּנֵ֥י עֲנָֽק׃ עַם גָּדוֹל וְגָבוֹהַּ – בְּנֵי עֲנָקִים, הֲרֵי אַתָּה יָדַעְתָּ וְאַתָּה שָׁמַעְתָּ – וְכִי מִי יָכוֹל לְהִתְיַצֵּב לִפְנֵי בְּנֵי עֲנָק!
ג וְיָדַעְתָּ֣ הַיּ֗וֹם כִּי֩ יְהֹוָ֨ה אֱלֹהֶ֜יךָ הֽוּא־הָעֹבֵ֤ר לְפָנֶ֙יךָ֙ אֵ֣שׁ אֹֽכְלָ֔ה ה֧וּא יַשְׁמִידֵ֛ם וְה֥וּא יַכְנִיעֵ֖ם לְפָנֶ֑יךָ וְהֽוֹרַשְׁתָּ֤ם וְהַֽאֲבַדְתָּם֙ מַהֵ֔ר כַּאֲשֶׁ֛ר דִּבֶּ֥ר יְהֹוָ֖ה לָֽךְ׃ דַּע הַיּוֹם, שֶׁה' אֱלֹהֶיךָ הוּא שֶׁעוֹבֵר לְפָנֶיךָ, כּוֹחוֹ – אֵשׁ שׂוֹרֶפֶת, הוּא יַשְׁמִיד אוֹתָם וְהוּא יַכְנִיעַ אוֹתָם לְפָנֶיךָ, כְּבוֹשׁ אוֹתָם וְאַבֵּד אוֹתָם מַהֵר, כְּמוֹ שֶׁה' אָמַר לְךָ.

שלישי[עריכה]

ד אַל־תֹּאמַ֣ר בִּלְבָבְךָ֗ בַּהֲדֹ֣ף יְהֹוָה֩ אֱלֹהֶ֨יךָ אֹתָ֥ם ׀ מִלְּפָנֶ֘יךָ֮ לֵאמֹר֒ בְּצִדְקָתִי֙ הֱבִיאַ֣נִי יְהֹוָ֔ה לָרֶ֖שֶׁת אֶת־הָאָ֣רֶץ הַזֹּ֑את וּבְרִשְׁעַת֙ הַגּוֹיִ֣ם הָאֵ֔לֶּה יְהֹוָ֖ה מוֹרִישָׁ֥ם מִפָּנֶֽיךָ׃ כְּשֶׁה' אֱלֹהֶיךָ יַהֲדוֹף אוֹתָם מִלְּפָנֶיךָ אַל תַּחְשׁוֹב וְתֹאמַר: "גַּם בִּזְכוּת צִדְקָתִי ה' הֵבִיא אוֹתִי לִכְבּוֹשׁ אֶת הָאָרֶץ הַזֹּאת וְגַם בִּגְלַל רִשְׁעַת הָעַמִּים הָאֵלֶּה ה' מְגָרֵשׁ אוֹתָם מִלְּפָנֶיךָ".
ה לֹ֣א בְצִדְקָתְךָ֗ וּבְיֹ֙שֶׁר֙ לְבָ֣בְךָ֔ אַתָּ֥ה בָ֖א לָרֶ֣שֶׁת אֶת־אַרְצָ֑ם כִּ֞י בְּרִשְׁעַ֣ת ׀ הַגּוֹיִ֣ם הָאֵ֗לֶּה יְהֹוָ֤ה אֱלֹהֶ֙יךָ֙ מוֹרִישָׁ֣ם מִפָּנֶ֔יךָ וּלְמַ֜עַן הָקִ֣ים אֶת־הַדָּבָ֗ר אֲשֶׁ֨ר נִשְׁבַּ֤ע יְהֹוָה֙ לַאֲבֹתֶ֔יךָ לְאַבְרָהָ֥ם לְיִצְחָ֖ק וּֽלְיַעֲקֹֽב׃ לֹא בִּזְכוּת צִדְקָתְךָ וְיוֹשֶׁר לִבְּךָ אַתָּה בָּא לִכְבּוֹשׁ אֶת אַרְצָם, אֶלָּא רַק בִּגְלַל רִשְׁעַת הָעַמִּים הָאֵלֶּה ה' אֱלֹהֶיךָ מְגָרֵשׁ אוֹתָם מִלְּפָנֶיךָ, וּכְדֵי לְקַיֵּם אֶת מַה שֶּׁה' נִשְׁבַּע לַאֲבוֹתֶיךָ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב.
ו וְיָדַעְתָּ֗ כִּ֠י לֹ֤א בְצִדְקָֽתְךָ֙ יְהֹוָ֣ה אֱ֠לֹהֶ֠יךָ נֹתֵ֨ן לְךָ֜ אֶת־הָאָ֧רֶץ הַטּוֹבָ֛ה הַזֹּ֖את לְרִשְׁתָּ֑הּ כִּ֥י עַם־קְשֵׁה־עֹ֖רֶף אָֽתָּה׃ דַּע שֶׁלֹּא בִּזְכוּת צִדְקָתְךָ ה' אֱלֹהֶיךָ נוֹתֵן לְךָ אֶת הָאָרֶץ הַטּוֹבָה הַזֹּאת לִכְבּוֹשׁ אוֹתָהּ, כִּי אַתָּה הֲרֵי עַם עַקְשָׁן וְסַרְבָן.
ז זְכֹר֙ אַל־תִּשְׁכַּ֔ח אֵ֧ת אֲשֶׁר־הִקְצַ֛פְתָּ אֶת־יְהֹוָ֥ה אֱלֹהֶ֖יךָ בַּמִּדְבָּ֑ר לְמִן־הַיּ֞וֹם אֲשֶׁר־יָצָ֣אתָ ׀ מֵאֶ֣רֶץ מִצְרַ֗יִם עַד־בֹּֽאֲכֶם֙ עַד־הַמָּק֣וֹם הַזֶּ֔ה מַמְרִ֥ים הֱיִיתֶ֖ם עִם־יְהֹוָֽה׃ זְכוֹר אַל תִּשְׁכַּח אֶת זֶה שֶׁהִכְעַסְתָּ אֶת ה' אֱלֹהֶיךָ בַּמִּדְבָּר, מֵהַיּוֹם שֶׁיָּצָאתָ מֵאֶרֶץ מִצְרַיִם עַד שֶׁבָּאתֶם עַד הַמָּקוֹם הַזֶּה הֲיִיתֶם מְסָרְבִים לְדִבְרֵי ה'.
ח וּבְחֹרֵ֥ב הִקְצַפְתֶּ֖ם אֶת־יְהֹוָ֑ה וַיִּתְאַנַּ֧ף יְהֹוָ֛ה בָּכֶ֖ם לְהַשְׁמִ֥יד אֶתְכֶֽם׃ וּבְהַר חוֹרֵב הִכְעַסְתֶּם אֶת ה', וַה' כָּעַס עֲלֵיכֶם וְרָצָה לְהַשְׁמִיד אֶתְכֶם.
ט בַּעֲלֹתִ֣י הָהָ֗רָה לָקַ֜חַת לוּחֹ֤ת הָֽאֲבָנִים֙ לוּחֹ֣ת הַבְּרִ֔ית אֲשֶׁר־כָּרַ֥ת יְהֹוָ֖ה עִמָּכֶ֑ם וָאֵשֵׁ֣ב בָּהָ֗ר אַרְבָּעִ֥ים יוֹם֙ וְאַרְבָּעִ֣ים לַ֔יְלָה לֶ֚חֶם לֹ֣א אָכַ֔לְתִּי וּמַ֖יִם לֹ֥א שָׁתִֽיתִי׃ כְּשֶׁעָלִיתִי לָהָר לָקַחַת אֶת לוּחוֹת הָאֲבָנִים – לוּחוֹת הַהֶסְכֵּם שֶׁה' עָשָׂה אִתְּכֶם, יָשַׁבְתִּי בָּהָר אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה, לֹא אָכַלְתִּי לֶחֶם וְלֹא שָׁתִיתִי מַיִם.
י וַיִּתֵּ֨ן יְהֹוָ֜ה אֵלַ֗י אֶת־שְׁנֵי֙ לוּחֹ֣ת הָֽאֲבָנִ֔ים כְּתֻבִ֖ים בְּאֶצְבַּ֣ע אֱלֹהִ֑ים וַעֲלֵיהֶ֗ם כְּֽכׇל־הַדְּבָרִ֡ים אֲשֶׁ֣ר דִּבֶּר֩ יְהֹוָ֨ה עִמָּכֶ֥ם בָּהָ֛ר מִתּ֥וֹךְ הָאֵ֖שׁ בְּי֥וֹם הַקָּהָֽל׃ ה' נָתַן לִי אֶת שְׁנֵי לוּחוֹת הָאֲבָנִים כְּתוּבִים בְּאֶצְבַּע אֱלֹהִים, וַעֲלֵיהֶם כְּתוּבִים כָּל הַדְּבָרִים שֶׁה' דִּבֵּר אִתְּכֶם בָּהָר מִתּוֹךְ הָאֵשׁ בְּיוֹם הִתְכַּנְּסוּת הַקָּהָל.
יא וַיְהִ֗י מִקֵּץ֙ אַרְבָּעִ֣ים י֔וֹם וְאַרְבָּעִ֖ים לָ֑יְלָה נָתַ֨ן יְהֹוָ֜ה אֵלַ֗י אֶת־שְׁנֵ֛י לֻחֹ֥ת הָאֲבָנִ֖ים לֻח֥וֹת הַבְּרִֽית׃ בְּסוֹף אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה, ה' נָתַן לִי אֶת שְׁנֵי לוּחוֹת הָאֲבָנִים – לוּחוֹת הַהֶסְכֵּם.
יב וַיֹּ֨אמֶר יְהֹוָ֜ה אֵלַ֗י ק֣וּם רֵ֤ד מַהֵר֙ מִזֶּ֔ה כִּ֚י שִׁחֵ֣ת עַמְּךָ֔ אֲשֶׁ֥ר הוֹצֵ֖אתָ מִמִּצְרָ֑יִם סָ֣רוּ מַהֵ֗ר מִן־הַדֶּ֙רֶךְ֙ אֲשֶׁ֣ר צִוִּיתִ֔ם עָשׂ֥וּ לָהֶ֖ם מַסֵּכָֽה׃ ה' אָמַר לִי: קוּם רֵד מִכָּאן מַהֵר, כִּי עַמְּךָ שֶׁהוֹצֵאתָ מִמִּצְרַיִם קִלְקֵל, הֵם סָטוּ מַהֵר מֵהַדֶּרֶךְ שֶׁצִּוִּיתִי אוֹתָם, עָשׂוּ לָהֶם פֶּסֶל מִמַּתֶּכֶת.
יג וַיֹּ֥אמֶר יְהֹוָ֖ה אֵלַ֣י לֵאמֹ֑ר רָאִ֙יתִי֙ אֶת־הָעָ֣ם הַזֶּ֔ה וְהִנֵּ֥ה עַם־קְשֵׁה־עֹ֖רֶף הֽוּא׃ ה' אָמַר לִי: רָאִיתִי אֶת הָעָם הַזֶּה וְהִנֵּה הוּא עַם עַקְשָׁן וְסַרְבָן.
יד הֶ֤רֶף מִמֶּ֙נִּי֙ וְאַשְׁמִידֵ֔ם וְאֶמְחֶ֣ה אֶת־שְׁמָ֔ם מִתַּ֖חַת הַשָּׁמָ֑יִם וְאֶֽעֱשֶׂה֙ אֽוֹתְךָ֔ לְגוֹי־עָצ֥וּם וָרָ֖ב מִמֶּֽנּוּ׃ עֲזוֹב אוֹתִי וְאַשְׁמִיד אוֹתָם, וְאֶמְחוֹק אֶת שְׁמָם מִתַּחַת לַשָּׁמַיִם, וְאֶעֱשֶׂה מִמְּךָ עַם עָצוּם וְגָדוֹל יוֹתֵר מֵהֶם!
טו וָאֵ֗פֶן וָֽאֵרֵד֙ מִן־הָהָ֔ר וְהָהָ֖ר בֹּעֵ֣ר בָּאֵ֑שׁ וּשְׁנֵי֙ לוּחֹ֣ת הַבְּרִ֔ית עַ֖ל שְׁתֵּ֥י יָדָֽי׃ פָּנִיתִי וְיָרַדְתִּי מֵהָהָר, וְהָהָר בּוֹעֵר בָּאֵשׁ, וּשְׁנֵי לוּחוֹת הַהֶסְכֵּם עַל שְׁתֵּי יָדַי.
טז וָאֵ֗רֶא וְהִנֵּ֤ה חֲטָאתֶם֙ לַיהֹוָ֣ה אֱלֹֽהֵיכֶ֔ם עֲשִׂיתֶ֣ם לָכֶ֔ם עֵ֖גֶל מַסֵּכָ֑ה סַרְתֶּ֣ם מַהֵ֔ר מִן־הַדֶּ֕רֶךְ אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה אֶתְכֶֽם׃ רָאִיתִי שֶׁהִנֵּה חֲטָאתֶם לַה' אֱלֹהֵיכֶם – עֲשִׂיתֶם לָכֶם עֵגֶל מִמַּתֶּכֶת, סְטִיתֶם מַהֵר מֵהַדֶּרֶךְ שֶׁה' צִוָּה אֶתְכֶם.
יז וָאֶתְפֹּשׂ֙ בִּשְׁנֵ֣י הַלֻּחֹ֔ת וָֽאַשְׁלִכֵ֔ם מֵעַ֖ל שְׁתֵּ֣י יָדָ֑י וָאֲשַׁבְּרֵ֖ם לְעֵינֵיכֶֽם׃ תָּפַשְׂתִּי בִּשְׁנֵי הַלּוּחוֹת וְהִשְׁלַכְתִּי אוֹתָם מֵעַל שְׁתֵּי יָדַי, וְשָׁבַרְתִּי אוֹתָם לְעֵינֵיכֶם.
יח וָֽאֶתְנַפַּל֩ לִפְנֵ֨י יְהֹוָ֜ה כָּרִאשֹׁנָ֗ה אַרְבָּעִ֥ים יוֹם֙ וְאַרְבָּעִ֣ים לַ֔יְלָה לֶ֚חֶם לֹ֣א אָכַ֔לְתִּי וּמַ֖יִם לֹ֣א שָׁתִ֑יתִי עַ֤ל כׇּל־חַטַּאתְכֶם֙ אֲשֶׁ֣ר חֲטָאתֶ֔ם לַעֲשׂ֥וֹת הָרַ֛ע בְּעֵינֵ֥י יְהֹוָ֖ה לְהַכְעִיסֽוֹ׃ הִשְׁתַּטַּחְתִּי לִפְנֵי ה' אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה כְּמוֹ בַּפַּעַם הָרִאשׁוֹנָה, לֹא אָכַלְתִּי לֶחֶם וְלֹא שָׁתִיתִי מַיִם, וּבִקַּשְׁתִּי סְלִיחָה עַל כָּל הַחֵטְא שֶׁלָּכֶם שֶׁחֲטָאתֶם – לַעֲשׂוֹת אֶת הָרַע בְּעֵינֵי ה', לְהַכְעִיסוֹ.
יט כִּ֣י יָגֹ֗רְתִּי מִפְּנֵ֤י הָאַף֙ וְהַ֣חֵמָ֔ה אֲשֶׁ֨ר קָצַ֧ף יְהֹוָ֛ה עֲלֵיכֶ֖ם לְהַשְׁמִ֣יד אֶתְכֶ֑ם וַיִּשְׁמַ֤ע יְהֹוָה֙ אֵלַ֔י גַּ֖ם בַּפַּ֥עַם הַהִֽוא׃ כִּי חָשַׁשְׁתִּי מֵהַכַּעַס וְהָרוֹגֶז שֶׁה' כָּעַס עֲלֵיכֶם וְרָצָה לְהַשְׁמִיד אֶתְכֶם, ה' שָׁמַע אֵלַי גַּם בַּפַּעַם הַהִיא.
כ וּֽבְאַהֲרֹ֗ן הִתְאַנַּ֧ף יְהֹוָ֛ה מְאֹ֖ד לְהַשְׁמִיד֑וֹ וָֽאֶתְפַּלֵּ֛ל גַּם־בְּעַ֥ד אַהֲרֹ֖ן בָּעֵ֥ת הַהִֽוא׃ ה' כָּעַס מְאוֹד עַל אַהֲרֹן וְרָצָה לְהַשְׁמִיד אוֹתוֹ, הִתְפַּלַּלְתִּי בְּאוֹתוֹ זְמָן גַּם עַל אַהֲרֹן.
כא וְֽאֶת־חַטַּאתְכֶ֞ם אֲשֶׁר־עֲשִׂיתֶ֣ם אֶת־הָעֵ֗גֶל לָקַ֘חְתִּי֮ וָאֶשְׂרֹ֣ף אֹת֣וֹ ׀ בָּאֵשׁ֒ וָאֶכֹּ֨ת אֹת֤וֹ טָחוֹן֙ הֵיטֵ֔ב עַ֥ד אֲשֶׁר־דַּ֖ק לְעָפָ֑ר וָֽאַשְׁלִךְ֙ אֶת־עֲפָר֔וֹ אֶל־הַנַּ֖חַל הַיֹּרֵ֥ד מִן־הָהָֽר׃ וְאֶת הַחֵטְא שֶׁלָּכֶם שֶׁעֲשִׂיתֶם – אֶת הָעֵגֶל, לָקַחְתִּי וְשָׂרַפְתִּי אוֹתוֹ בָּאֵשׁ, וְטָחַנְתִּי אוֹתוֹ הֵיטֵב, עַד שֶׁנִּהְיָה דַּק וְנַעֲשָׂה אַבְקָה, וְהִשְׁלַכְתִּי אֶת עֲפָרוֹ לַנַּחַל שֶׁיּוֹרֵד מֵהָהָר.
כב וּבְתַבְעֵרָה֙ וּבְמַסָּ֔ה וּבְקִבְרֹ֖ת הַֽתַּאֲוָ֑ה מַקְצִפִ֥ים הֱיִיתֶ֖ם אֶת־יְהֹוָֽה׃ וּבְתַבְעֵרָה, וּבְמַסָּה וּמְרִיבָה, וּבְקִבְרוֹת הַתַּאֲוָה, הִכְעַסְתֶּם אֶת ה'.
כג וּבִשְׁלֹ֨חַ יְהֹוָ֜ה אֶתְכֶ֗ם מִקָּדֵ֤שׁ בַּרְנֵ֙עַ֙ לֵאמֹ֔ר עֲלוּ֙ וּרְשׁ֣וּ אֶת־הָאָ֔רֶץ אֲשֶׁ֥ר נָתַ֖תִּי לָכֶ֑ם וַתַּמְר֗וּ אֶת־פִּ֤י יְהֹוָה֙ אֱלֹ֣הֵיכֶ֔ם וְלֹ֤א הֶֽאֱמַנְתֶּם֙ ל֔וֹ וְלֹ֥א שְׁמַעְתֶּ֖ם בְּקֹלֽוֹ׃ וּכְשֶׁה' שָׁלַח אֶתְכֶם מִקָּדֵשׁ בַּרְנֵעַ וְאָמַר: עֲלוּ וְכִבְשׁוּ אֶת הָאָרֶץ שֶׁנָּתַתִּי לָכֶם, סֵרַבְתֶּם לְצִוּוּי ה' אֱלֹהֵיכֶם וְלֹא הֶאֱמַנְתֶּם לוֹ וְלֹא שְׁמַעְתֶּם בְּקוֹלוֹ.
כד מַמְרִ֥ים הֱיִיתֶ֖ם עִם־יְהֹוָ֑ה מִיּ֖וֹם דַּעְתִּ֥י אֶתְכֶֽם׃ הֲיִיתֶם מְסָרְבִים לְצִוּוּי ה' מֵהַיּוֹם שֶׁהִכַּרְתִּי אֶתְכֶם.
כה וָֽאֶתְנַפַּ֞ל לִפְנֵ֣י יְהֹוָ֗ה אֵ֣ת אַרְבָּעִ֥ים הַיּ֛וֹם וְאֶת־אַרְבָּעִ֥ים הַלַּ֖יְלָה אֲשֶׁ֣ר הִתְנַפָּ֑לְתִּי כִּֽי־אָמַ֥ר יְהֹוָ֖ה לְהַשְׁמִ֥יד אֶתְכֶֽם׃ הִשְׁתַּטַּחְתִּי לִפְנֵי ה' בְּאוֹתָם אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה שֶׁהִשְׁתַּטַּחְתִּי, כִּי ה' רָצָה לְהַשְׁמִיד אֶתְכֶם.
כו וָאֶתְפַּלֵּ֣ל אֶל־יְהֹוָה֮ וָאֹמַר֒ אֲדֹנָ֣י יֱהֹוִ֗ה אַל־תַּשְׁחֵ֤ת עַמְּךָ֙ וְנַחֲלָ֣תְךָ֔ אֲשֶׁ֥ר פָּדִ֖יתָ בְּגׇדְלֶ֑ךָ אֲשֶׁר־הוֹצֵ֥אתָ מִמִּצְרַ֖יִם בְּיָ֥ד חֲזָקָֽה׃ הִתְפַּלַּלְתִּי אֶל ה' וְאָמַרְתִּי: ה' אֱלֹהִים, אַל תַּהֲרוֹס אֶת עַמְּךָ וְנַחֲלָתְךָ שֶׁגָּאַלְתָּ בְּגַדְלוּתְךָ – שֶׁהוֹצֵאתָ מִמִּצְרַיִם בְּיָד חֲזָקָה.
כז זְכֹר֙ לַעֲבָדֶ֔יךָ לְאַבְרָהָ֥ם לְיִצְחָ֖ק וּֽלְיַעֲקֹ֑ב אַל־תֵּ֗פֶן אֶל־קְשִׁי֙ הָעָ֣ם הַזֶּ֔ה וְאֶל־רִשְׁע֖וֹ וְאֶל־חַטָּאתֽוֹ׃ זְכוֹר אֶת הַשְּׁבוּעָה שֶׁנִּשְׁבַּעְתָּ לְאַבְרָהָם יִצְחָק וְיַעֲקֹב עֲבָדֶיךָ, אַל תִּפְנֶה לְהַבִּיט עַל עַקְשָׁנוּת הָעָם הַזֶּה וְעַל רִשְׁעוּתוֹ וְעַל חֶטְאוֹ.
כח פֶּן־יֹאמְר֗וּ הָאָ֘רֶץ֮ אֲשֶׁ֣ר הוֹצֵאתָ֣נוּ מִשָּׁם֒ מִבְּלִי֙ יְכֹ֣לֶת יְהֹוָ֔ה לַהֲבִיאָ֕ם אֶל־הָאָ֖רֶץ אֲשֶׁר־דִּבֶּ֣ר לָהֶ֑ם וּמִשִּׂנְאָת֣וֹ אוֹתָ֔ם הוֹצִיאָ֖ם לַהֲמִתָ֥ם בַּמִּדְבָּֽר׃ שֶׁלֹּא יֹאמְרוּ תּוֹשְׁבֵי הָאָרֶץ שֶׁהוֹצֵאתָ אוֹתָנוּ מִשָּׁם: ה' הוֹצִיא אוֹתָם לַמִּדְבָּר לַהֲרוֹג אוֹתָם, בִּגְלַל חוֹסֶר יְכוֹלְתּוֹ לְהָבִיא אוֹתָם לָאָרֶץ שֶׁהִבְטִיחַ לָהֶם וּבִגְלַל שִׂנְאָתוֹ אוֹתָם.
כט וְהֵ֥ם עַמְּךָ֖ וְנַחֲלָתֶ֑ךָ אֲשֶׁ֤ר הוֹצֵ֙אתָ֙ בְּכֹחֲךָ֣ הַגָּדֹ֔ל וּבִֽזְרֹעֲךָ֖ הַנְּטוּיָֽה׃ וַהֲרֵי הֵם עַמְּךָ וְנַחֲלָתְךָ שֶׁהוֹצֵאתָ בְּכוֹחֲךָ הַגָּדוֹל וּבִזְרוֹעֲךָ הַמּוּרֶמֶת בְּגַאֲוָה!

רביעי[עריכה]

פ פָּרָשַׁת לִקְחֵי חֵטְא הָעֵגֶל
י א בָּעֵ֨ת הַהִ֜וא אָמַ֧ר יְהֹוָ֣ה אֵלַ֗י פְּסׇל־לְךָ֞ שְׁנֵֽי־לוּחֹ֤ת אֲבָנִים֙ כָּרִ֣אשֹׁנִ֔ים וַעֲלֵ֥ה אֵלַ֖י הָהָ֑רָה וְעָשִׂ֥יתָ לְּךָ֖ אֲר֥וֹן עֵֽץ׃ בַּזְּמַן הַהוּא ה' אָמַר אֵלַי: פְּסוֹל לְךָ שְׁנֵי לוּחוֹת אֲבָנִים כְּמוֹ הָרִאשׁוֹנִים, וַעֲלֵה אֵלַי לָהָר, וַעֲשֵׂה לְךָ אֲרוֹן עֵץ.
ב וְאֶכְתֹּב֙ עַל־הַלֻּחֹ֔ת אֶ֨ת־הַדְּבָרִ֔ים אֲשֶׁ֥ר הָי֛וּ עַל־הַלֻּחֹ֥ת הָרִאשֹׁנִ֖ים אֲשֶׁ֣ר שִׁבַּ֑רְתָּ וְשַׂמְתָּ֖ם בָּאָרֽוֹן׃ וְאֶכְתּוֹב עַל הַלּוּחוֹת אֶת הַדְּבָרִים שֶׁהָיוּ כְּתוּבִים עַל הַלּוּחוֹת הָרִאשׁוֹנִים שֶׁשָּׁבַרְתָּ, וְתָשִׁים אוֹתָם בָּאָרוֹן.
ג וָאַ֤עַשׂ אֲרוֹן֙ עֲצֵ֣י שִׁטִּ֔ים וָאֶפְסֹ֛ל שְׁנֵי־לֻחֹ֥ת אֲבָנִ֖ים כָּרִאשֹׁנִ֑ים וָאַ֣עַל הָהָ֔רָה וּשְׁנֵ֥י הַלֻּחֹ֖ת בְּיָדִֽי׃ עָשִׂיתִי אֲרוֹן מֵעֲצֵי שִׁטָּה, וּפִסַּלְתִּי שְׁנֵי לוּחוֹת אֲבָנִים כְּמוֹ הָרִאשׁוֹנִים, וְעָלִיתִי לָהָר כְּשֶׁשְּׁנֵי הַלּוּחוֹת בְּיָדִי.
ד וַיִּכְתֹּ֨ב עַֽל־הַלֻּחֹ֜ת כַּמִּכְתָּ֣ב הָרִאשׁ֗וֹן אֵ֚ת עֲשֶׂ֣רֶת הַדְּבָרִ֔ים אֲשֶׁ֣ר דִּבֶּר֩ יְהֹוָ֨ה אֲלֵיכֶ֥ם בָּהָ֛ר מִתּ֥וֹךְ הָאֵ֖שׁ בְּי֣וֹם הַקָּהָ֑ל וַיִּתְּנֵ֥ם יְהֹוָ֖ה אֵלָֽי׃ הוּא כָּתַב עַל הַלּוּחוֹת כְּמוֹ הַכְּתִיבָה הָרִאשׁוֹנָה – אֶת עֲשֶׂרֶת הַדִּבְּרוֹת שֶׁה' דִּבֵּר אֲלֵיכֶם בָּהָר מִתּוֹךְ הָאֵשׁ בְּיוֹם הִתְכַּנְּסוּת הַקָּהָל, וַה' נָתַן אוֹתָם לִי.
ה וָאֵ֗פֶן וָֽאֵרֵד֙ מִן־הָהָ֔ר וָֽאָשִׂם֙ אֶת־הַלֻּחֹ֔ת בָּאָר֖וֹן אֲשֶׁ֣ר עָשִׂ֑יתִי וַיִּ֣הְיוּ שָׁ֔ם כַּאֲשֶׁ֥ר צִוַּ֖נִי יְהֹוָֽה׃ פָּנִיתִי וְיָרַדְתִּי מֵהָהָר, וְשַׂמְתִּי אֶת הַלּוּחוֹת בָּאָרוֹן שֶׁעָשִׂיתִי, וְהֵם הָיוּ שָׁם כְּמוֹ שֶׁה' צִוָּה אוֹתִי.
ו וּבְנֵ֣י יִשְׂרָאֵ֗ל נָ֥סְע֛וּ מִבְּאֵרֹ֥ת בְּנֵי־יַעֲקָ֖ן מוֹסֵרָ֑ה שָׁ֣ם מֵ֤ת אַהֲרֹן֙ וַיִּקָּבֵ֣ר שָׁ֔ם וַיְכַהֵ֛ן אֶלְעָזָ֥ר בְּנ֖וֹ תַּחְתָּֽיו׃ וּבְנֵי יִשְׂרָאֵל נָסְעוּ מִבְּאֵרוֹת בְּנֵי יַעֲקָן לְמוֹסֵרָה, שָׁם אַהֲרֹן מֵת וְנִקְבַּר שָׁם, וְאֶלְעָזָר בְּנוֹ כִּהֵן תַּחְתָּיו.
ז מִשָּׁ֥ם נָסְע֖וּ הַגֻּדְגֹּ֑דָה וּמִן־הַגֻּדְגֹּ֣דָה יׇטְבָ֔תָה אֶ֖רֶץ נַ֥חֲלֵי מָֽיִם׃ מִשָּׁם נָסְעוּ לְגֻדְגּוֹד, וּמִגֻּדְגּוֹד לְיׇטְבָת – אֶרֶץ נַחֲלֵי מַיִם.
ח בָּעֵ֣ת הַהִ֗וא הִבְדִּ֤יל יְהֹוָה֙ אֶת־שֵׁ֣בֶט הַלֵּוִ֔י לָשֵׂ֖את אֶת־אֲר֣וֹן בְּרִית־יְהֹוָ֑ה לַעֲמֹד֩ לִפְנֵ֨י יְהֹוָ֤ה לְשָֽׁרְתוֹ֙ וּלְבָרֵ֣ךְ בִּשְׁמ֔וֹ עַ֖ד הַיּ֥וֹם הַזֶּֽה׃ בְּאוֹתוֹ הַזְּמָן – אַחֲרֵי חֵטְא הָעֵגֶל, ה' הִבְדִּיל אֶת שֵׁבֶט הַלֵּוִי שֶׁרַק הוּא יִשָּׂא אֶת אֲרוֹן בְּרִית ה' וְיַעֲמוֹד לִפְנֵי ה' לְשָׁרֵת אוֹתוֹ וִיבָרֵךְ בִּשְׁמוֹ כְּמוֹ שֶׁהֵם עוֹשִׂים עַד הַיּוֹם.
ט עַל־כֵּ֞ן לֹֽא־הָיָ֧ה לְלֵוִ֛י חֵ֥לֶק וְנַחֲלָ֖ה עִם־אֶחָ֑יו יְהֹוָה֙ ה֣וּא נַחֲלָת֔וֹ כַּאֲשֶׁ֥ר דִּבֶּ֛ר יְהֹוָ֥ה אֱלֹהֶ֖יךָ לֽוֹ׃ לָכֵן אֵין לְלֵוִי חֵלֶק וְנַחֲלָה עִם אֶחָיו, עֲבוֹדַת ה' הִיא נַחֲלָתוֹ כְּמוֹ שֶׁה' אֱלֹהֶיךָ דִּבֵּר לְגַבָּיו.
י וְאָנֹכִ֞י עָמַ֣דְתִּי בָהָ֗ר כַּיָּמִים֙ הָרִ֣אשֹׁנִ֔ים אַרְבָּעִ֣ים י֔וֹם וְאַרְבָּעִ֖ים לָ֑יְלָה וַיִּשְׁמַ֨ע יְהֹוָ֜ה אֵלַ֗י גַּ֚ם בַּפַּ֣עַם הַהִ֔וא לֹא־אָבָ֥ה יְהֹוָ֖ה הַשְׁחִיתֶֽךָ׃ וַאֲנִי עָמַדְתִּי בָּהָר כְּמִסְפַּר הַיָּמִים הָרִאשׁוֹנִים – אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה, ה' שָׁמַע אֵלַי גַּם בַּפַּעַם הַהִיא – ה' לֹא רָצָה לְהַשְׁמִיד אוֹתְךָ.
יא וַיֹּ֤אמֶר יְהֹוָה֙ אֵלַ֔י ק֛וּם לֵ֥ךְ לְמַסַּ֖ע לִפְנֵ֣י הָעָ֑ם וְיָבֹ֙אוּ֙ וְיִֽירְשׁ֣וּ אֶת־הָאָ֔רֶץ אֲשֶׁר־נִשְׁבַּ֥עְתִּי לַאֲבֹתָ֖ם לָתֵ֥ת לָהֶֽם׃ ה' אָמַר לִי: קוּם, לֵךְ הַנְהֵג אֶת הָעָם, וְיָבוֹאוּ וְיִכְבְּשׁוּ אֶת הָאָרֶץ שֶׁנִּשְׁבַּעְתִּי לַאֲבוֹתֵיהֶם לָתֵת לָהֶם.

חמישי[עריכה]

פ פָּרָשַׁת אַהֲבַת הַשֵּׁם
יב וְעַתָּה֙ יִשְׂרָאֵ֔ל מָ֚ה יְהֹוָ֣ה אֱלֹהֶ֔יךָ שֹׁאֵ֖ל מֵעִמָּ֑ךְ כִּ֣י אִם־לְ֠יִרְאָ֠ה אֶת־יְהֹוָ֨ה אֱלֹהֶ֜יךָ לָלֶ֤כֶת בְּכׇל־דְּרָכָיו֙ וּלְאַהֲבָ֣ה אֹת֔וֹ וְלַֽעֲבֹד֙ אֶת־יְהֹוָ֣ה אֱלֹהֶ֔יךָ בְּכׇל־לְבָבְךָ֖ וּבְכׇל־נַפְשֶֽׁךָ׃ וְעַכְשָׁיו יִשְׂרָאֵל, מָה ה' אֱלֹהֶיךָ מְבַקֵּשׁ מִמְּךָ, רַק לְכַבֵּד אֶת ה' אֱלֹהֶיךָ לָלֶכֶת בְּכָל דְּרָכָיו וְלֶאֱהוֹב אוֹתוֹ וְלַעֲבוֹד אֶת ה' אֱלֹהֶיךָ בְּכָל לִבְּךָ וּבְכָל רְצוֹנְךָ,
יג לִשְׁמֹ֞ר אֶת־מִצְוֺ֤ת יְהֹוָה֙ וְאֶת־חֻקֹּתָ֔יו אֲשֶׁ֛ר אָנֹכִ֥י מְצַוְּךָ֖ הַיּ֑וֹם לְט֖וֹב לָֽךְ׃ לִשְׁמוֹר אֶת מִצְווֹת ה' וְאֶת חֻקָּיו שֶׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם, וְכָל זֹאת כְּדֵי שֶׁיִּהְיֶה לְךָ טוֹב.
יד הֵ֚ן לַיהֹוָ֣ה אֱלֹהֶ֔יךָ הַשָּׁמַ֖יִם וּשְׁמֵ֣י הַשָּׁמָ֑יִם הָאָ֖רֶץ וְכׇל־אֲשֶׁר־בָּֽהּ׃ הֲרֵי לַה' אֱלֹהֶיךָ שַׁיָּכִים הַשָּׁמַיִם וּמַה שֶּׁמֵּעֲלֵיהֶם, הָאָרֶץ, וְכָל מַה שֶּׁבָּהּ!
טו רַ֧ק בַּאֲבֹתֶ֛יךָ חָשַׁ֥ק יְהֹוָ֖ה לְאַהֲבָ֣ה אוֹתָ֑ם וַיִּבְחַ֞ר בְּזַרְעָ֣ם אַחֲרֵיהֶ֗ם בָּכֶ֛ם מִכׇּל־הָעַמִּ֖ים כַּיּ֥וֹם הַזֶּֽה׃ אֲבָל רַק בַּאֲבוֹתֶיךָ ה' חָשַׁק לֶאֱהוֹב אוֹתָם, וּבָחַר בָּכֶם – בְּצֶאֱצָאֵיהֶם אַחֲרֵיהֶם מִתּוֹךְ כָּל הָעַמִּים, כְּמוֹ שֶׁאַתֶּם רוֹאִים כַּיּוֹם.
טז וּמַלְתֶּ֕ם אֵ֖ת עׇרְלַ֣ת לְבַבְכֶ֑ם וְעׇ֨רְפְּכֶ֔ם לֹ֥א תַקְשׁ֖וּ עֽוֹד׃ הָסִירוּ אֶת אֲטִימוּת לִבְּכֶם – וְאַל תִּתְעַקְּשׁוּ עוֹד.
יז כִּ֚י יְהֹוָ֣ה אֱלֹֽהֵיכֶ֔ם ה֚וּא אֱלֹהֵ֣י הָֽאֱלֹהִ֔ים וַאֲדֹנֵ֖י הָאֲדֹנִ֑ים הָאֵ֨ל הַגָּדֹ֤ל הַגִּבֹּר֙ וְהַנּוֹרָ֔א אֲשֶׁר֙ לֹא־יִשָּׂ֣א פָנִ֔ים וְלֹ֥א יִקַּ֖ח שֹֽׁחַד׃ כִּי ה' אֱלֹהֵיכֶם הוּא אֱלֹהֵי הָאֱלֹהִים וַאֲדוֹנֵי הָאֲדוֹנִים, הוּא הָאֵל הַגָּדוֹל הַגִּבּוֹר וְהַנּוֹרָא שֶׁלֹּא יִתֵּן עֲדִיפוּת שֶׁלֹּא כַּדִּין וְלֹא יִקַּח שׁוֹחַד.
יח עֹשֶׂ֛ה מִשְׁפַּ֥ט יָת֖וֹם וְאַלְמָנָ֑ה וְאֹהֵ֣ב גֵּ֔ר לָ֥תֶת ל֖וֹ לֶ֥חֶם וְשִׂמְלָֽה׃ הוּא עוֹשֶׂה מִשְׁפַּט צֶדֶק עֲבוּר יָתוֹם וְאַלְמָנָה, וְאוֹהֵב גֵּר לָתֵת לוֹ אוֹכֶל וּבְגָדִים.
יט וַאֲהַבְתֶּ֖ם אֶת־הַגֵּ֑ר כִּֽי־גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֥רֶץ מִצְרָֽיִם׃ אֶהֱבוּ אֶת הַגֵּר כִּי גַּם אַתֶּם הֱיִיתֶם תּוֹשָׁבִים זָרִים בְּאֶרֶץ מִצְרַיִם.
כ אֶת־יְהֹוָ֧ה אֱלֹהֶ֛יךָ תִּירָ֖א אֹת֣וֹ תַעֲבֹ֑ד וּב֣וֹ תִדְבָּ֔ק וּבִשְׁמ֖וֹ תִּשָּׁבֵֽעַ׃ כַּבֵּד אֶת ה' אֱלֹהֶיךָ, אוֹתוֹ תַּעֲבוֹד, בּוֹ תִּדְבָּק, וְתִשָּׁבַע רַק בִּשְׁמוֹ.
כא ה֥וּא תְהִלָּתְךָ֖ וְה֣וּא אֱלֹהֶ֑יךָ אֲשֶׁר־עָשָׂ֣ה אִתְּךָ֗ אֶת־הַגְּדֹלֹ֤ת וְאֶת־הַנּֽוֹרָאֹת֙ הָאֵ֔לֶּה אֲשֶׁ֥ר רָא֖וּ עֵינֶֽיךָ׃ הוּא מְקוֹר גַּאֲוַתְךָ וְהוּא אֱלֹהֶיךָ, שֶׁעָשָׂה לְךָ אֶת הַנִּפְלָאוֹת הַגְּדוֹלוֹת וְהַמַּפְחִידוֹת הָאֵלֶּה שֶׁעֵינֶיךָ רָאוּ.
כב בְּשִׁבְעִ֣ים נֶ֔פֶשׁ יָרְד֥וּ אֲבֹתֶ֖יךָ מִצְרָ֑יְמָה וְעַתָּ֗ה שָֽׂמְךָ֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ כְּכוֹכְבֵ֥י הַשָּׁמַ֖יִם לָרֹֽב׃ שִׁבְעִים אֲנָשִׁים הָיוּ אֲבוֹתֶיךָ כְּשֶׁיָּרְדוּ לְמִצְרַיִם, וְעַכְשָׁיו ה' אֱלֹהֶיךָ עָשָׂה אֶתְכֶם רַבִּים כְּכוֹכְבֵי הַשָּׁמַיִם.
יא א וְאָ֣הַבְתָּ֔ אֵ֖ת יְהֹוָ֣ה אֱלֹהֶ֑יךָ וְשָׁמַרְתָּ֣ מִשְׁמַרְתּ֗וֹ וְחֻקֹּתָ֧יו וּמִשְׁפָּטָ֛יו וּמִצְוֺתָ֖יו כׇּל־הַיָּמִֽים׃ אֱהוֹב אֶת ה' אֱלֹהֶיךָ, הַקְפֵּד לְקַיֵּם אֶת חֻקָּיו מִשְׁפָּטָיו וּמִצְווֹתָיו, תָּמִיד.
ב וִֽידַעְתֶּם֮ הַיּוֹם֒ כִּ֣י ׀ לֹ֣א אֶת־בְּנֵיכֶ֗ם אֲשֶׁ֤ר לֹֽא־יָדְעוּ֙ וַאֲשֶׁ֣ר לֹא־רָא֔וּ אֶת־מוּסַ֖ר יְהֹוָ֣ה אֱלֹהֵיכֶ֑ם אֶת־גׇּדְל֕וֹ אֶת־יָדוֹ֙ הַחֲזָקָ֔ה וּזְרֹע֖וֹ הַנְּטוּיָֽה׃ דְּעוּ וְהַפְנִימוּ הַיּוֹם, כִּי לֹא לְצֶאֱצָאֵיכֶם אֲנִי פּוֹנֶה – שֶׁלֹּא הִכִּירוּ וְשֶׁלֹּא רָאוּ אֶת מוּסַר הַהַשְׂכֵּל שֶׁה' אֱלֹהֵיכֶם לִמֵּד אֶתְכֶם, אֶת גְּדֻלָּתוֹ, אֶת יָדוֹ הַחֲזָקָה וּזְרוֹעוֹ שֶׁמּוּנֶפֶת לְהַכּוֹת.
ג וְאֶת־אֹֽתֹתָיו֙ וְאֶֽת־מַעֲשָׂ֔יו אֲשֶׁ֥ר עָשָׂ֖ה בְּת֣וֹךְ מִצְרָ֑יִם לְפַרְעֹ֥ה מֶֽלֶךְ־מִצְרַ֖יִם וּלְכׇל־אַרְצֽוֹ׃ וְאֶת סִימָנָיו, וְאֶת מַעֲשָׂיו שֶׁעָשָׂה בְּמִצְרַיִם לְפַרְעֹה מֶלֶךְ מִצְרַיִם וּלְכָל אַרְצוֹ.
ד וַאֲשֶׁ֣ר עָשָׂה֩ לְחֵ֨יל מִצְרַ֜יִם לְסוּסָ֣יו וּלְרִכְבּ֗וֹ אֲשֶׁ֨ר הֵצִ֜יף אֶת־מֵ֤י יַם־סוּף֙ עַל־פְּנֵיהֶ֔ם בְּרׇדְפָ֖ם אַחֲרֵיכֶ֑ם וַיְאַבְּדֵ֣ם יְהֹוָ֔ה עַ֖ד הַיּ֥וֹם הַזֶּֽה׃ וְאֶת מַה שֶּׁעָשָׂה לְחֵיל מִצְרַיִם – לְסוּסָיו וּלְרִכְבּוֹ, שֶׁהֵצִיף אֶת מֵי יַם סוּף עַל פְּנֵיהֶם, כְּשֶׁרָדְפוּ אַחֲרֵיכֶם, וְה' אִבֵּד אוֹתָם, וְהֵם לֹא חָזְרוּ לְעוֹצְמָתָם עַד הַיּוֹם הַזֶּה.
ה וַאֲשֶׁ֥ר עָשָׂ֛ה לָכֶ֖ם בַּמִּדְבָּ֑ר עַד־בֹּאֲכֶ֖ם עַד־הַמָּק֥וֹם הַזֶּֽה׃ וְאֶת מַה שֶּׁעָשָׂה לָכֶם בַּמִּדְבָּר עַד שֶׁבָּאתֶם לַמָּקוֹם הַזֶּה.
ו וַאֲשֶׁ֨ר עָשָׂ֜ה לְדָתָ֣ן וְלַאֲבִירָ֗ם בְּנֵ֣י אֱלִיאָב֮ בֶּן־רְאוּבֵן֒ אֲשֶׁ֨ר פָּצְתָ֤ה הָאָ֙רֶץ֙ אֶת־פִּ֔יהָ וַתִּבְלָעֵ֥ם וְאֶת־בָּתֵּיהֶ֖ם וְאֶת־אׇהֳלֵיהֶ֑ם וְאֵ֤ת כׇּל־הַיְקוּם֙ אֲשֶׁ֣ר בְּרַגְלֵיהֶ֔ם בְּקֶ֖רֶב כׇּל־יִשְׂרָאֵֽל׃ וְאֶת מַה שֶּׁעָשָׂה לְדָתָן וְלַאֲבִירָם בְּנֵי אֱלִיאָב בֶּן רְאוּבֵן – שֶׁהָאָרֶץ פָּתְחָה אֶת פִּיהָ וּבָלְעָה אוֹתָם וְאֶת בָּתֵּיהֶם וְאֶת אָהֳלֵיהֶם וְאֶת כָּל הַנְּכָסִים שֶׁאִתָּם, בְּמֶרְכַּז כָּל יִשְׂרָאֵל.
ז כִּ֤י עֵֽינֵיכֶם֙ הָֽרֹאֹ֔ת אֶת־כׇּל־מַעֲשֵׂ֥ה יְהֹוָ֖ה הַגָּדֹ֑ל אֲשֶׁ֖ר עָשָֽׂה׃ אֶלָּא עֵינֵיכֶם הֵן שֶׁרָאוּ אֶת כָּל מַעֲשֵׂה ה' הַגָּדוֹל שֶׁעָשָׂה.
ח וּשְׁמַרְתֶּם֙ אֶת־כׇּל־הַמִּצְוָ֔ה אֲשֶׁ֛ר אָנֹכִ֥י מְצַוְּךָ֖ הַיּ֑וֹם לְמַ֣עַן תֶּחֶזְק֗וּ וּבָאתֶם֙ וִֽירִשְׁתֶּ֣ם אֶת־הָאָ֔רֶץ אֲשֶׁ֥ר אַתֶּ֛ם עֹבְרִ֥ים שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ הַקְפִּידוּ לְקַיֵּם אֶת כָּל הַמִּצְווֹת שֶׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם, כְּדֵי שֶׁתִּתְחַזְּקוּ וְתָבוֹאוּ וְתִכְבְּשׁוּ אֶת הָאָרֶץ שֶׁאַתֶּם עוֹבְרִים לְשָׁם לִכְבּוֹשׁ אוֹתָהּ.
ט וּלְמַ֨עַן תַּאֲרִ֤יכוּ יָמִים֙ עַל־הָ֣אֲדָמָ֔ה אֲשֶׁר֩ נִשְׁבַּ֨ע יְהֹוָ֧ה לַאֲבֹתֵיכֶ֛ם לָתֵ֥ת לָהֶ֖ם וּלְזַרְעָ֑ם אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָֽשׁ׃ וּכְדֵי שֶׁתִּתְקַיְּמוּ זְמַן רָב עַל הָאֲדָמָה שֶׁה' נִשְׁבַּע לַאֲבוֹתֵיכֶם לָתֵת לָהֶם וּלְצֶאֱצָאֵיהֶם – אֶרֶץ שׁוֹפַעַת חָלָב וּדְבָשׁ.

ששי[עריכה]

ס פָּרָשַׁת שֶׁבַח הָאָרֶץ
י כִּ֣י הָאָ֗רֶץ אֲשֶׁ֨ר אַתָּ֤ה בָא־שָׁ֙מָּה֙ לְרִשְׁתָּ֔הּ לֹ֣א כְאֶ֤רֶץ מִצְרַ֙יִם֙ הִ֔וא אֲשֶׁ֥ר יְצָאתֶ֖ם מִשָּׁ֑ם אֲשֶׁ֤ר תִּזְרַע֙ אֶֽת־זַרְעֲךָ֔ וְהִשְׁקִ֥יתָ בְרַגְלְךָ֖ כְּגַ֥ן הַיָּרָֽק׃ כִּי הָאָרֶץ שֶׁאַתָּה בָּא לְשָׁם לִכְבּוֹשׁ אוֹתָהּ, הִיא לֹא כְּאֶרֶץ מִצְרַיִם שֶׁיְּצָאתֶם מִשָּׁם שֶׁהָיִיתָ זוֹרֵעַ אֶת זַרְעֲךָ וּמַשְׁקֶה בְּעַצְמְךָ כְּמוֹ שֶׁמַּשְׁקִים גַּן יָרָק.
יא וְהָאָ֗רֶץ אֲשֶׁ֨ר אַתֶּ֜ם עֹבְרִ֥ים שָׁ֙מָּה֙ לְרִשְׁתָּ֔הּ אֶ֥רֶץ הָרִ֖ים וּבְקָעֹ֑ת לִמְטַ֥ר הַשָּׁמַ֖יִם תִּשְׁתֶּה־מָּֽיִם׃ הָאָרֶץ שֶׁאַתֶּם עוֹבְרִים לְשָׁם לִכְבּוֹשׁ אוֹתָהּ הִיא אֶרֶץ הָרִים וּבְקָעוֹת, הִיא תִּשְׁתֶּה מַיִם מִמְּטַר הַשָּׁמַיִם.
יב אֶ֕רֶץ אֲשֶׁר־יְהֹוָ֥ה אֱלֹהֶ֖יךָ דֹּרֵ֣שׁ אֹתָ֑הּ תָּמִ֗יד עֵינֵ֨י יְהֹוָ֤ה אֱלֹהֶ֙יךָ֙ בָּ֔הּ מֵֽרֵשִׁית֙ הַשָּׁנָ֔ה וְעַ֖ד אַחֲרִ֥ית שָׁנָֽה׃ אֶרֶץ שֶׁה' אֱלֹהֶיךָ דּוֹרֵשׁ בִּשְׁלוֹמָהּ, עֵינֵי ה' אֱלֹהֶיךָ מַשְׁגִּיחוֹת וּמַבִּיטוֹת בָּהּ תָּמִיד, מִתְּחִלַּת הַשָּׁנָה וְעַד סוֹף הַשָּׁנָה.
ס פָּרָשַׁת וְהָיָה אִם שָׁמֹעַ
יג וְהָיָ֗ה אִם־שָׁמֹ֤עַ תִּשְׁמְעוּ֙ אֶל־מִצְוֺתַ֔י אֲשֶׁ֧ר אָנֹכִ֛י מְצַוֶּ֥ה אֶתְכֶ֖ם הַיּ֑וֹם לְאַהֲבָ֞ה אֶת־יְהֹוָ֤ה אֱלֹֽהֵיכֶם֙ וּלְעׇבְד֔וֹ בְּכׇל־לְבַבְכֶ֖ם וּבְכׇל־נַפְשְׁכֶֽם׃ אִם תִּשְׁמְעוּ לְמִצְווֹתַי שֶׁאֲנִי מְצַוֶּה אֶתְכֶם הַיּוֹם – לֶאֱהוֹב אֶת ה' אֱלֹהֵיכֶם, וְלַעֲבוֹד אוֹתוֹ בְּכָל לִבְּכֶם וּבְכָל רְצוֹנְכֶם,
יד וְנָתַתִּ֧י מְטַֽר־אַרְצְכֶ֛ם בְּעִתּ֖וֹ יוֹרֶ֣ה וּמַלְק֑וֹשׁ וְאָסַפְתָּ֣ דְגָנֶ֔ךָ וְתִירֹֽשְׁךָ֖ וְיִצְהָרֶֽךָ׃ אֶתֵּן אֶת הַגֶּשֶׁם בְּאַרְצְכֶם בִּזְמַנּוֹ מֵהַגֶּשֶׁם הָרִאשׁוֹן בַּחוֹרֶף וְעַד לָאַחֲרוֹן, וְתֶאֱסוֹף אֶת תְּבוּאָתְךָ וְיֵינְךָ וְשַׁמְנְךָ.
טו וְנָתַתִּ֛י עֵ֥שֶׂב בְּשָׂדְךָ֖ לִבְהֶמְתֶּ֑ךָ וְאָכַלְתָּ֖ וְשָׂבָֽעְתָּ׃ אֶתֵּן עֵשֶׂב בִּשְׂדוֹתֶיךָ לִבְהֶמְתְּךָ, וְתֹאכַל וְתִשְׂבַּע.
טז הִשָּֽׁמְר֣וּ לָכֶ֔ם פֶּ֥ן יִפְתֶּ֖ה לְבַבְכֶ֑ם וְסַרְתֶּ֗ם וַעֲבַדְתֶּם֙ אֱלֹהִ֣ים אֲחֵרִ֔ים וְהִשְׁתַּחֲוִיתֶ֖ם לָהֶֽם׃ הִזָּהֲרוּ לָכֶם שֶׁלִּבְּכֶם לֹא יִתְפַּתֶּה, וְתִסְטוּ מֵהַדֶּרֶךְ הַנְּכוֹנָה וְתַעַבְדוּ אֱלֹהִים אֲחֵרִים וְתִשְׁתַּחֲווּ לָהֶם.
יז וְחָרָ֨ה אַף־יְהֹוָ֜ה בָּכֶ֗ם וְעָצַ֤ר אֶת־הַשָּׁמַ֙יִם֙ וְלֹֽא־יִהְיֶ֣ה מָטָ֔ר וְהָ֣אֲדָמָ֔ה לֹ֥א תִתֵּ֖ן אֶת־יְבוּלָ֑הּ וַאֲבַדְתֶּ֣ם מְהֵרָ֗ה מֵעַל֙ הָאָ֣רֶץ הַטֹּבָ֔ה אֲשֶׁ֥ר יְהֹוָ֖ה נֹתֵ֥ן לָכֶֽם׃ ה' יִכְעַס עֲלֵיכֶם, וְיַעֲצוֹר אֶת הַשָּׁמַיִם וְלֹא יֵרֵד גֶּשֶׁם, וְהָאֲדָמָה לֹא תִּתֵּן אֶת יְבוּלָהּ, וְתֹאבְדוּ בִּמְהִירוּת מֵעַל הָאָרֶץ הַטּוֹבָה שֶׁה' נוֹתֵן לָכֶם.
יח וְשַׂמְתֶּם֙ אֶת־דְּבָרַ֣י אֵ֔לֶּה עַל־לְבַבְכֶ֖ם וְעַֽל־נַפְשְׁכֶ֑ם וּקְשַׁרְתֶּ֨ם אֹתָ֤ם לְאוֹת֙ עַל־יֶדְכֶ֔ם וְהָי֥וּ לְטוֹטָפֹ֖ת בֵּ֥ין עֵינֵיכֶֽם׃ הַפְנִימוּ אֶת דְּבָרַי אֵלֶּה בְּלִבְּכֶם וּבִרְצוֹנְכֶם, קִשְׁרוּ אוֹתָם לְתֵעוּד עַל הַיַּד שֶׁלָּכֶם, וְיִהְיוּ לְתַכְשִׁיט בֵּין עֵינֵיכֶם.
יט וְלִמַּדְתֶּ֥ם אֹתָ֛ם אֶת־בְּנֵיכֶ֖ם לְדַבֵּ֣ר בָּ֑ם בְּשִׁבְתְּךָ֤ בְּבֵיתֶ֙ךָ֙ וּבְלֶכְתְּךָ֣ בַדֶּ֔רֶךְ וּֽבְשׇׁכְבְּךָ֖ וּבְקוּמֶֽךָ׃ לַמְּדוּ אוֹתָם אֶת בְּנֵיכֶם לְדַבֵּר בָּהֶם, כְּשֶׁאַתָּה יוֹשֵׁב בְּבֵיתְךָ וּכְשֶׁאַתָּה הוֹלֵךְ בַּדֶּרֶךְ, כְּשֶׁאַתָּה שׁוֹכֵב וּכְשֶׁאַתָּה קָם.
כ וּכְתַבְתָּ֛ם עַל־מְזוּז֥וֹת בֵּיתֶ֖ךָ וּבִשְׁעָרֶֽיךָ׃ כְּתוֹב אוֹתָם עַל מְזוּזוֹת בֵּיתְךָ וּבִשְׁעָרֶיךָ.
כא לְמַ֨עַן יִרְבּ֤וּ יְמֵיכֶם֙ וִימֵ֣י בְנֵיכֶ֔ם עַ֚ל הָֽאֲדָמָ֔ה אֲשֶׁ֨ר נִשְׁבַּ֧ע יְהֹוָ֛ה לַאֲבֹתֵיכֶ֖ם לָתֵ֣ת לָהֶ֑ם כִּימֵ֥י הַשָּׁמַ֖יִם עַל־הָאָֽרֶץ׃‏[1] כְּדֵי שֶׁחַיֵּיכֶם וְחַיֵּי בְּנֵיכֶם יִתְאָרְכוּ עַל הָאֲדָמָה שֶׁה' נִשְׁבַּע לַאֲבוֹתֵיכֶם לָתֵת לָהֶם, כְּמֶשֶׁךְ שְׁנוֹת הַשָּׁמַיִם מֵעַל הָאָרֶץ.

שביעי ומפטיר[עריכה]

ס פָּרָשַׁת הַיְּרֻשָּׁה הַשְּׁלֵמָה
כב כִּי֩ אִם־שָׁמֹ֨ר תִּשְׁמְר֜וּן אֶת־כׇּל־הַמִּצְוָ֣ה הַזֹּ֗את אֲשֶׁ֧ר אָנֹכִ֛י מְצַוֶּ֥ה אֶתְכֶ֖ם לַעֲשֹׂתָ֑הּ לְאַהֲבָ֞ה אֶת־יְהֹוָ֧ה אֱלֹהֵיכֶ֛ם לָלֶ֥כֶת בְּכׇל־דְּרָכָ֖יו וּלְדׇבְקָה־בֽוֹ׃ כִּי אִם תַּקְפִּידוּ לְקַיֵּם אֶת כָּל הַמִּצְווֹת הָאֵלֶּה שֶׁאֲנִי מְצַוֶּה אֶתְכֶם לַעֲשׂוֹת אוֹתָן – לֶאֱהוֹב אֶת ה' אֱלֹהֵיכֶם לָלֶכֶת בְּכָל דְּרָכָיו וְלִדְבּוֹק בּוֹ.
כג וְהוֹרִ֧ישׁ יְהֹוָ֛ה אֶת־כׇּל־הַגּוֹיִ֥ם הָאֵ֖לֶּה מִלִּפְנֵיכֶ֑ם וִֽירִשְׁתֶּ֣ם גּוֹיִ֔ם גְּדֹלִ֥ים וַעֲצֻמִ֖ים מִכֶּֽם׃ ה' יְגָרֵשׁ אֶת כָּל הָעַמִּים הָאֵלֶּה מִלִּפְנֵיכֶם, וְתִכְבְּשׁוּ עַמִּים גְּדוֹלִים וַעֲצוּמִים יוֹתֵר מִכֶּם.
כד כׇּל־הַמָּק֗וֹם אֲשֶׁ֨ר תִּדְרֹ֧ךְ כַּֽף־רַגְלְכֶ֛ם בּ֖וֹ לָכֶ֣ם יִהְיֶ֑ה מִן־הַמִּדְבָּ֨ר וְהַלְּבָנ֜וֹן מִן־הַנָּהָ֣ר נְהַר־פְּרָ֗ת וְעַד֙ הַיָּ֣ם הָאַֽחֲר֔וֹן יִהְיֶ֖ה גְּבֻלְכֶֽם׃ כָּל מָקוֹם שֶׁכַּף רַגְלְכֶם תִּדְרוֹךְ בּוֹ יִהְיֶה שֶׁלָּכֶם, גְּבוּלְכֶם יִהְיֶה מֵהַמִּדְבָּר וְהַלְּבָנוֹן וּמֵהַנָּהָר – נְהַר פְּרָת, וְעַד הַיָּם הַמַּעֲרָבִי.
כה לֹא־יִתְיַצֵּ֥ב אִ֖ישׁ בִּפְנֵיכֶ֑ם פַּחְדְּכֶ֨ם וּמוֹרַֽאֲכֶ֜ם יִתֵּ֣ן ׀ יְהֹוָ֣ה אֱלֹֽהֵיכֶ֗ם עַל־פְּנֵ֤י כׇל־הָאָ֙רֶץ֙ אֲשֶׁ֣ר תִּדְרְכוּ־בָ֔הּ כַּאֲשֶׁ֖ר דִּבֶּ֥ר לָכֶֽם׃ אִישׁ לֹא יַעֲמוֹד בִּפְנֵיכֶם, ה' אֱלֹהֵיכֶם יַשְׁרֶה פַּחַד וְכָבוֹד מִכֶּם בְּכָל הָאָרֶץ שֶׁתִּדְרְכוּ בָּהּ כְּמוֹ שֶׁהִבְטִיחַ לָכֶם.
