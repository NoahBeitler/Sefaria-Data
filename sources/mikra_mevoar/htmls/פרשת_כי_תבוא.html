ראשון[עריכה]

פ פָּרָשַׁת הַבִּכּוּרִים
כו א וְהָיָה֙ כִּֽי־תָב֣וֹא אֶל־הָאָ֔רֶץ אֲשֶׁר֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ נֹתֵ֥ן לְךָ֖ נַחֲלָ֑ה וִֽירִשְׁתָּ֖הּ וְיָשַׁ֥בְתָּ בָּֽהּ׃ כְּשֶׁתָּבוֹא לָאָרֶץ שֶׁה' אֱלֹהֶיךָ נוֹתֵן לְךָ לִירוּשָׁה וְתִכְבּוֹשׁ אוֹתָהּ וְתִתְיַשֵּׁב בָּהּ.
ב וְלָקַחְתָּ֞ מֵרֵאשִׁ֣ית ׀ כׇּל־פְּרִ֣י הָאֲדָמָ֗ה אֲשֶׁ֨ר תָּבִ֧יא מֵֽאַרְצְךָ֛ אֲשֶׁ֨ר יְהֹוָ֧ה אֱלֹהֶ֛יךָ נֹתֵ֥ן לָ֖ךְ וְשַׂמְתָּ֣ בַטֶּ֑נֶא וְהָֽלַכְתָּ֙ אֶל־הַמָּק֔וֹם אֲשֶׁ֤ר יִבְחַר֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ לְשַׁכֵּ֥ן שְׁמ֖וֹ שָֽׁם׃ תִּקַּח מֵהָרִאשׁוֹנִים בְּכָל פֵּרוֹת הָאֲדָמָה שֶׁתָּבִיא מֵאַרְצְךָ שֶׁה' אֱלֹהֶיךָ נוֹתֵן לְךָ וְתָשִׂים בַּסַּל, וְתֵלֵךְ לַמָּקוֹם שֶׁה' אֱלֹהֶיךָ יִבְחַר לְהַשְׁרוֹת שָׁם אֶת שְׁכִינָתוֹ.
ג וּבָאתָ֙ אֶל־הַכֹּהֵ֔ן אֲשֶׁ֥ר יִהְיֶ֖ה בַּיָּמִ֣ים הָהֵ֑ם וְאָמַרְתָּ֣ אֵלָ֗יו הִגַּ֤דְתִּי הַיּוֹם֙ לַיהֹוָ֣ה אֱלֹהֶ֔יךָ כִּי־בָ֙אתִי֙ אֶל־הָאָ֔רֶץ אֲשֶׁ֨ר נִשְׁבַּ֧ע יְהֹוָ֛ה לַאֲבֹתֵ֖ינוּ לָ֥תֶת לָֽנוּ׃ תָּבוֹא לַכֹּהֵן שֶׁיִּהְיֶה בְּאוֹתָם הַיָּמִים וְתֹאמַר לוֹ: אֲנִי מַכְרִיז הַיּוֹם לִפְנֵי ה' אֱלֹהֶיךָ שֶׁבָּאתִי לָאָרֶץ שֶׁה' נִשְׁבַּע לַאֲבוֹתֵינוּ לָתֵת לָנוּ.
ד וְלָקַ֧ח הַכֹּהֵ֛ן הַטֶּ֖נֶא מִיָּדֶ֑ךָ וְהִ֨נִּיח֔וֹ לִפְנֵ֕י מִזְבַּ֖ח יְהֹוָ֥ה אֱלֹהֶֽיךָ׃ הַכֹּהֵן יִקַּח מִיָּדְךָ אֶת הַסָּל וְיַנִּיחוֹ לִפְנֵי הַמִּזְבֵּחַ שֶׁל ה' אֱלֹהֶיךָ.
ה וְעָנִ֨יתָ וְאָמַרְתָּ֜ לִפְנֵ֣י ׀ יְהֹוָ֣ה אֱלֹהֶ֗יךָ אֲרַמִּי֙ אֹבֵ֣ד אָבִ֔י וַיֵּ֣רֶד מִצְרַ֔יְמָה וַיָּ֥גׇר שָׁ֖ם בִּמְתֵ֣י מְעָ֑ט וַֽיְהִי־שָׁ֕ם לְג֥וֹי גָּד֖וֹל עָצ֥וּם וָרָֽב׃ תַּכְרִיז וְתֹאמַר לִפְנֵי ה' אֱלֹהֶיךָ: לָבָן הָאֲרַמִּי רָצָה לְהַשְׁמִיד אֶת יַעֲקֹב אָבִינוּ, וְיַעֲקֹב יָרַד לְמִצְרַיִם וְגָר שָׁם עִם מִשְׁפָּחָה קְטַנָּה, וְנַעֲשָׂה שָׁם עָם גָּדוֹל עָצוּם וּמְרֻבֶּה.
ו וַיָּרֵ֧עוּ אֹתָ֛נוּ הַמִּצְרִ֖ים וַיְעַנּ֑וּנוּ וַיִּתְּנ֥וּ עָלֵ֖ינוּ עֲבֹדָ֥ה קָשָֽׁה׃ הַמִּצְרִים הֵרֵעוּ לָנוּ וְהִשְׁפִּילוּ אוֹתָנוּ וְנָתְנוּ לָנוּ עֲבוֹדָה קָשָׁה.
ז וַנִּצְעַ֕ק אֶל־יְהֹוָ֖ה אֱלֹהֵ֣י אֲבֹתֵ֑ינוּ וַיִּשְׁמַ֤ע יְהֹוָה֙ אֶת־קֹלֵ֔נוּ וַיַּ֧רְא אֶת־עׇנְיֵ֛נוּ וְאֶת־עֲמָלֵ֖נוּ וְאֶֽת־לַחֲצֵֽנוּ׃ צָעַקְנוּ אֶל ה' אֱלֹהֵי אֲבוֹתֵינוּ, ה' שָׁמַע אֶת קוֹלֵנוּ וְרָאָה אֶת הַשְׁפָּלָתֵינוּ וְאֶת עֲמָלֵנוּ וְאֶת לַחֲצֵנוּ.
ח וַיּוֹצִאֵ֤נוּ יְהֹוָה֙ מִמִּצְרַ֔יִם בְּיָ֤ד חֲזָקָה֙ וּבִזְרֹ֣עַ נְטוּיָ֔ה וּבְמֹרָ֖א גָּדֹ֑ל וּבְאֹת֖וֹת וּבְמֹפְתִֽים׃ ה' הוֹצִיא אוֹתָנוּ מִמִּצְרַיִם בְּיָד חֲזָקָה, בִּזְרוֹעַ מוּנֶפֶת לְהַכּוֹת, בְּהַפְחָדוֹת גְּדוֹלוֹת, בְּסִימָנִים לְכוֹחוֹ, וּבְנִסִּים.
ט וַיְבִאֵ֖נוּ אֶל־הַמָּק֣וֹם הַזֶּ֑ה וַיִּתֶּן־לָ֙נוּ֙ אֶת־הָאָ֣רֶץ הַזֹּ֔את אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָֽשׁ׃ וְהֵבִיא אוֹתָנוּ לַמָּקוֹם הַזֶּה וְנָתַן לָנוּ אֶת הָאָרֶץ הַזֹּאת – אֶרֶץ שׁוֹפַעַת חָלָב וּדְבָשׁ.
י וְעַתָּ֗ה הִנֵּ֤ה הֵבֵ֙אתִי֙ אֶת־רֵאשִׁית֙ פְּרִ֣י הָאֲדָמָ֔ה אֲשֶׁר־נָתַ֥תָּה לִּ֖י יְהֹוָ֑ה וְהִנַּחְתּ֗וֹ לִפְנֵי֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ וְהִֽשְׁתַּחֲוִ֔יתָ לִפְנֵ֖י יְהֹוָ֥ה אֱלֹהֶֽיךָ׃ וְעַכְשָׁיו ה', הִנֵּה הֵבֵאתִי אֶת הָרִאשׁוֹנִים בְּפֵרוֹת הָאֲדָמָה שֶׁנָּתַתָּה לִי. תַּנִּיחַ אוֹתוֹ לִפְנֵי ה' אֱלֹהֶיךָ וְתִשְׁתַּחֲוֶה לִפְנֵי ה' אֱלֹהֶיךָ.
יא וְשָׂמַחְתָּ֣ בְכׇל־הַטּ֗וֹב אֲשֶׁ֧ר נָֽתַן־לְךָ֛ יְהֹוָ֥ה אֱלֹהֶ֖יךָ וּלְבֵיתֶ֑ךָ אַתָּה֙ וְהַלֵּוִ֔י וְהַגֵּ֖ר אֲשֶׁ֥ר בְּקִרְבֶּֽךָ׃ וְתִשְׂמַח בְּכָל הַטּוֹב שֶׁה' אֱלֹהֶיךָ נָתַן לְךָ וּלְמִשְׁפַּחְתְּךָ, אַתָּה עִם הַלֵּוִי וְעִם הַגֵּר שֶׁאֶצְלְךָ.

שני[עריכה]

ס פָּרָשַׁת וִדּוּי מַעַשְׂרוֹת
יב כִּ֣י תְכַלֶּ֞ה לַ֠עְשֵׂ֠ר אֶת־כׇּל־מַעְשַׂ֧ר תְּבוּאָתְךָ֛ בַּשָּׁנָ֥ה הַשְּׁלִישִׁ֖ת שְׁנַ֣ת הַֽמַּעֲשֵׂ֑ר וְנָתַתָּ֣ה לַלֵּוִ֗י לַגֵּר֙ לַיָּת֣וֹם וְלָֽאַלְמָנָ֔ה וְאָכְל֥וּ בִשְׁעָרֶ֖יךָ וְשָׂבֵֽעוּ׃ כְּשֶׁתַּשְׁלִים לְעַשֵּׂר אֶת כָּל הַמַּעַשְׂרוֹת שֶׁל תְּבוּאָתְךָ בַּשָּׁנָה הַשְּׁלִישִׁית שֶׁהִיא שְׁנַת מַעֲשֵׂר עָנִי, וְתִתֵּן לַלֵּוִי לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה, וְיֹאכְלוּ בְּעָרֶיךָ וְיִשְׂבְּעוּ –
יג וְאָמַרְתָּ֡ לִפְנֵי֩ יְהֹוָ֨ה אֱלֹהֶ֜יךָ בִּעַ֧רְתִּי הַקֹּ֣דֶשׁ מִן־הַבַּ֗יִת וְגַ֨ם נְתַתִּ֤יו לַלֵּוִי֙ וְלַגֵּר֙ לַיָּת֣וֹם וְלָאַלְמָנָ֔ה כְּכׇל־מִצְוָתְךָ֖ אֲשֶׁ֣ר צִוִּיתָ֑נִי לֹֽא־עָבַ֥רְתִּי מִמִּצְוֺתֶ֖יךָ וְלֹ֥א שָׁכָֽחְתִּי׃ תַּכְרִיז לִפְנֵי ה' אֱלֹהֶיךָ: הִפְרַשְׁתִּי אֶת הַקּוֹדֶשׁ מֵהַבַּיִת וְגַם נָתַתִּי אוֹתוֹ לַלֵּוִי לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה כְּכָל הַצִּוּוּי שֶׁצִּוִּיתָ אוֹתִי, לֹא עָבַרְתִּי עַל מִצְווֹתֶיךָ וְלֹא שָׁכַחְתִּי.
יד לֹא־אָכַ֨לְתִּי בְאֹנִ֜י מִמֶּ֗נּוּ וְלֹא־בִעַ֤רְתִּי מִמֶּ֙נּוּ֙ בְּטָמֵ֔א וְלֹא־נָתַ֥תִּי מִמֶּ֖נּוּ לְמֵ֑ת שָׁמַ֗עְתִּי בְּקוֹל֙ יְהֹוָ֣ה אֱלֹהָ֔י עָשִׂ֕יתִי כְּכֹ֖ל אֲשֶׁ֥ר צִוִּיתָֽנִי׃ לֹא אָכַלְתִּי מִמֶּנּוּ בִּשְׁעַת צַעֲרִי בְּיוֹם מוֹת קְרוֹבִי וְלֹא הִפְרַשְׁתִּי אוֹתוֹ בִּהְיוֹתִי טָמֵא וְלֹא נָתַתִּי מִמֶּנּוּ לְצָרְכֵי מֵת, שָׁמַעְתִּי בְּקוֹל ה' אֱלֹהַי עָשִׂיתִי כְּכָל מַה שֶּׁצִּוִּיתָ אוֹתִי.
טו הַשְׁקִ֩יפָה֩ מִמְּע֨וֹן קׇדְשְׁךָ֜ מִן־הַשָּׁמַ֗יִם וּבָרֵ֤ךְ אֶֽת־עַמְּךָ֙ אֶת־יִשְׂרָאֵ֔ל וְאֵת֙ הָאֲדָמָ֔ה אֲשֶׁ֥ר נָתַ֖תָּה לָ֑נוּ כַּאֲשֶׁ֤ר נִשְׁבַּ֙עְתָּ֙ לַאֲבֹתֵ֔ינוּ אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָֽשׁ׃ הַבֵּט מִמְּקוֹם מִשְׁכַּן קָדְשְׁךָ – מֵהַשָּׁמַיִם, וּבָרֵךְ אֶת עַמְּךָ – אֶת יִשְׂרָאֵל, וְאֶת הָאֲדָמָה שֶׁנָּתַתָּה לָנוּ, כְּמוֹ שֶׁנִּשְׁבַּעְתָּ לַאֲבוֹתֵינוּ – אֶרֶץ שׁוֹפַעַת חָלָב וּדְבָשׁ.

שלישי[עריכה]

ס פָּרָשַׁת קִדּוּשׁ עַם סְגֻלָּה
טז הַיּ֣וֹם הַזֶּ֗ה יְהֹוָ֨ה אֱלֹהֶ֜יךָ מְצַוְּךָ֧ לַעֲשׂ֛וֹת אֶת־הַחֻקִּ֥ים הָאֵ֖לֶּה וְאֶת־הַמִּשְׁפָּטִ֑ים וְשָׁמַרְתָּ֤ וְעָשִׂ֙יתָ֙ אוֹתָ֔ם בְּכׇל־לְבָבְךָ֖ וּבְכׇל־נַפְשֶֽׁךָ׃ הַיּוֹם הַזֶּה ה' אֱלֹהֶיךָ מְצַוֶּה אוֹתְךָ לַעֲשׂוֹת אֶת הַחֻקִּים הָאֵלֶּה וְאֶת הַמִּשְׁפָּטִים, הַקְפֵּד וַעֲשֵׂה אוֹתָם בְּכָל לִבְּךָ וּבְכָל רְצוֹנְךָ.
יז אֶת־יְהֹוָ֥ה הֶאֱמַ֖רְתָּ הַיּ֑וֹם לִהְיוֹת֩ לְךָ֨ לֵֽאלֹהִ֜ים וְלָלֶ֣כֶת בִּדְרָכָ֗יו וְלִשְׁמֹ֨ר חֻקָּ֧יו וּמִצְוֺתָ֛יו וּמִשְׁפָּטָ֖יו וְלִשְׁמֹ֥עַ בְּקֹלֽוֹ׃ הִכְתַּרְתָּ הַיּוֹם אֶת ה' שֶׁיִּהְיֶה לְךָ לֵאלֹהִים – לָלֶכֶת בִּדְרָכָיו לְהַקְפִּיד לְקַיֵּם אֶת חֻקָּיו מִצְווֹתָיו וּמִשְׁפָּטָיו וְלִשְׁמוֹעַ בְּקוֹלוֹ.
יח וַֽיהֹוָ֞ה הֶאֱמִֽירְךָ֣ הַיּ֗וֹם לִהְי֥וֹת לוֹ֙ לְעַ֣ם סְגֻלָּ֔ה כַּאֲשֶׁ֖ר דִּבֶּר־לָ֑ךְ וְלִשְׁמֹ֖ר כׇּל־מִצְוֺתָֽיו׃ וַה' הִכְתִּיר אוֹתְךָ הַיּוֹם שֶׁתִּהְיֶה עֲבוּרוֹ הָעָם הַמְּיוּחָד וְהַמְּקוֹרָב אֵלָיו כְּמוֹ שֶׁהִבְטִיחַ לְךָ – הָעָם שֶׁיִּשְׁמוֹר אֶת כָּל מִצְווֹתָיו.
יט וּֽלְתִתְּךָ֣ עֶלְי֗וֹן עַ֤ל כׇּל־הַגּוֹיִם֙ אֲשֶׁ֣ר עָשָׂ֔ה לִתְהִלָּ֖ה וּלְשֵׁ֣ם וּלְתִפְאָ֑רֶת וְלִֽהְיֹתְךָ֧ עַם־קָדֹ֛שׁ לַיהֹוָ֥ה אֱלֹהֶ֖יךָ כַּאֲשֶׁ֥ר דִּבֵּֽר׃ וְהוּא שָׂם אוֹתְךָ עֶלְיוֹן עַל כָּל הָעַמִּים שֶׁעָשָׂה, שֶׁיְּהַלְלוּ אוֹתְךָ וְתִתְפַּרְסֵם לְתִפְאֶרֶת, וְשֶׁתִּהְיֶה עַם קָדוֹשׁ לַה' אֱלֹהֶיךָ כְּמוֹ שֶׁדִּבֵּר.

רביעי[עריכה]

פ פָּרָשַׁת מִזְבַּח הַר עֵיבָל
כז א וַיְצַ֤ו מֹשֶׁה֙ וְזִקְנֵ֣י יִשְׂרָאֵ֔ל אֶת־הָעָ֖ם לֵאמֹ֑ר שָׁמֹר֙ אֶת־כׇּל־הַמִּצְוָ֔ה אֲשֶׁ֧ר אָנֹכִ֛י מְצַוֶּ֥ה אֶתְכֶ֖ם הַיּֽוֹם׃ מֹשֶׁה וְזִקְנֵי יִשְׂרָאֵל צִוּוּ אֶת הָעָם וְאָמְרוּ: הַקְפִּידוּ עַל כָּל הַמִּצְווֹת שֶׁאֲנִי מְצַוֶּה אֶתְכֶם הַיּוֹם.
ב וְהָיָ֗ה בַּיּוֹם֮ אֲשֶׁ֣ר תַּעַבְר֣וּ אֶת־הַיַּרְדֵּן֒ אֶל־הָאָ֕רֶץ אֲשֶׁר־יְהֹוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֣ן לָ֑ךְ וַהֲקֵמֹתָ֤ לְךָ֙ אֲבָנִ֣ים גְּדֹל֔וֹת וְשַׂדְתָּ֥ אֹתָ֖ם בַּשִּֽׂיד׃ בַּיּוֹם שֶׁתַּעַבְרוּ אֶת הַיַּרְדֵּן לָאָרֶץ שֶׁה' אֱלֹהֶיךָ נוֹתֵן לְךָ, תַּעֲמִיד אֲבָנִים גְּדוֹלוֹת וּתְסַיֵּד אוֹתָם בְּסִיד.
ג וְכָתַבְתָּ֣ עֲלֵיהֶ֗ן אֶֽת־כׇּל־דִּבְרֵ֛י הַתּוֹרָ֥ה הַזֹּ֖את בְּעׇבְרֶ֑ךָ לְמַ֡עַן אֲשֶׁר֩ תָּבֹ֨א אֶל־הָאָ֜רֶץ אֲֽשֶׁר־יְהֹוָ֥ה אֱלֹהֶ֣יךָ ׀ נֹתֵ֣ן לְךָ֗ אֶ֣רֶץ זָבַ֤ת חָלָב֙ וּדְבַ֔שׁ כַּאֲשֶׁ֥ר דִּבֶּ֛ר יְהֹוָ֥ה אֱלֹהֵֽי־אֲבֹתֶ֖יךָ לָֽךְ׃ וְתִכְתוֹב עֲלֵיהֶן אֶת כָּל דִּבְרֵי הַתּוֹרָה הַזֹּאת כְּשֶׁתַּעֲבוֹר לָאָרֶץ, כְּדֵי שֶׁתָּבוֹא לָאָרֶץ שֶׁה' אֱלֹהֶיךָ נוֹתֵן לְךָ – אֶרֶץ שׁוֹפַעַת חָלָב וּדְבַשׁ, כְּמוֹ שֶׁה' אֱלֹהֵי אֲבוֹתֶיךָ דִּבֵּר אֵלֶיךָ.
ד וְהָיָה֮ בְּעׇבְרְכֶ֣ם אֶת־הַיַּרְדֵּן֒ תָּקִ֜ימוּ אֶת־הָאֲבָנִ֣ים הָאֵ֗לֶּה אֲשֶׁ֨ר אָנֹכִ֜י מְצַוֶּ֥ה אֶתְכֶ֛ם הַיּ֖וֹם בְּהַ֣ר עֵיבָ֑ל וְשַׂדְתָּ֥ אוֹתָ֖ם בַּשִּֽׂיד׃ כְּשֶׁתַּעַבְרוּ אֶת הַיַּרְדֵּן, תַּעֲמִידוּ בְּהַר עֵיבָל אֶת הָאֲבָנִים הָאֵלֶּה שֶׁאֲנִי מְצַוֶּה אֶתְכֶם הַיּוֹם, וּתְסַיֵּד אוֹתָם בְּסִיד.
ה וּבָנִ֤יתָ שָּׁם֙ מִזְבֵּ֔חַ לַיהֹוָ֖ה אֱלֹהֶ֑יךָ מִזְבַּ֣ח אֲבָנִ֔ים לֹא־תָנִ֥יף עֲלֵיהֶ֖ם בַּרְזֶֽל׃ וְתִבְנֶה שָׁם מִזְבֵּחַ לַה' אֱלֹהֶיךָ – מִזְבַּח אֲבָנִים, אַל תָּנִיף עֲלֵיהֶם בַּרְזֶל.
ו אֲבָנִ֤ים שְׁלֵמוֹת֙ תִּבְנֶ֔ה אֶת־מִזְבַּ֖ח יְהֹוָ֣ה אֱלֹהֶ֑יךָ וְהַעֲלִ֤יתָ עָלָיו֙ עוֹלֹ֔ת לַיהֹוָ֖ה אֱלֹהֶֽיךָ׃ תִּבְנֶה אֶת מִזְבַּח ה' אֱלֹהֶיךָ מֵאֲבָנִים שְׁלֵמוֹת, וְתַקְרִיב עָלָיו עוֹלוֹת לַה' אֱלֹהֶיךָ.
ז וְזָבַחְתָּ֥ שְׁלָמִ֖ים וְאָכַ֣לְתָּ שָּׁ֑ם וְשָׂ֣מַחְתָּ֔ לִפְנֵ֖י יְהֹוָ֥ה אֱלֹהֶֽיךָ׃ תִּשְׁחַט שְׁלָמִים וְתֹאכַל שָׁם, וְתִשְׂמַח לִפְנֵי ה' אֱלֹהֶיךָ.
ח וְכָתַבְתָּ֣ עַל־הָאֲבָנִ֗ים אֶֽת־כׇּל־דִּבְרֵ֛י הַתּוֹרָ֥ה הַזֹּ֖את בַּאֵ֥ר הֵיטֵֽב׃ תִּכְתוֹב עַל הָאֲבָנִים אֶת כָּל דִּבְרֵי הַתּוֹרָה הַזֹּאת מְבוֹאָרִים הֵיטֵב.
ס
ט וַיְדַבֵּ֤ר מֹשֶׁה֙ וְהַכֹּהֲנִ֣ים הַלְוִיִּ֔ם אֶ֥ל כׇּל־יִשְׂרָאֵ֖ל לֵאמֹ֑ר הַסְכֵּ֤ת ׀ וּשְׁמַע֙ יִשְׂרָאֵ֔ל הַיּ֤וֹם הַזֶּה֙ נִהְיֵ֣יתָֽ לְעָ֔ם לַיהֹוָ֖ה אֱלֹהֶֽיךָ׃ מֹשֶׁה וְהַכֹּהֲנִים הַלְוִיִּים דִּבְּרוּ אֶל כָּל יִשְׂרָאֵל וְאָמְרוּ: יִשְׂרָאֵל, הַקְשֵׁב וּשְׁמַע! הַיּוֹם הַזֶּה נַעֲשֵׂיתָ עָם לַה' אֱלֹהֶיךָ.
י וְשָׁ֣מַעְתָּ֔ בְּק֖וֹל יְהֹוָ֣ה אֱלֹהֶ֑יךָ וְעָשִׂ֤יתָ אֶת־מִצְוֺתָו֙ וְאֶת־חֻקָּ֔יו אֲשֶׁ֛ר אָנֹכִ֥י מְצַוְּךָ֖ הַיּֽוֹם׃ שְׁמַע בְּקוֹל ה' אֱלֹהֶיךָ וַעֲשֵׂה אֶת מִצְווֹתָיו וְאֶת חֻקָּיו שֶׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם.

חמישי[עריכה]

ס
יא וַיְצַ֤ו מֹשֶׁה֙ אֶת־הָעָ֔ם בַּיּ֥וֹם הַה֖וּא לֵאמֹֽר׃ מֹשֶׁה צִוָּה אֶת הָעָם בְּאוֹתוֹ יוֹם וְאָמַר:
יב אֵ֠לֶּה יַֽעַמְד֞וּ לְבָרֵ֤ךְ אֶת־הָעָם֙ עַל־הַ֣ר גְּרִזִ֔ים בְּעׇבְרְכֶ֖ם אֶת־הַיַּרְדֵּ֑ן שִׁמְעוֹן֙ וְלֵוִ֣י וִֽיהוּדָ֔ה וְיִשָּׂשכָ֖ר וְיוֹסֵ֥ף וּבִנְיָמִֽן׃ אֵלֶּה הַשְּׁבָטִים שֶׁיַּעַמְדוּ לִשְׁמוֹעַ אֶת בִּרְכַּת הָעָם עַל הַר גְּרִיזִים כְּשֶׁתַּעַבְרוּ אֶת הַיַּרְדֵּן: שִׁמְעוֹן וְלֵוִי וִיהוּדָה וְיִשָּׂשכָר וְיוֹסֵף וּבִנְיָמִין.
יג וְאֵ֛לֶּה יַֽעַמְד֥וּ עַל־הַקְּלָלָ֖ה בְּהַ֣ר עֵיבָ֑ל רְאוּבֵן֙ גָּ֣ד וְאָשֵׁ֔ר וּזְבוּלֻ֖ן דָּ֥ן וְנַפְתָּלִֽי׃ וְאֵלֶּה יַעַמְדוּ לִשְׁמוֹעַ אֶת הַקְּלָלָה בְּהַר עֵיבָל: רְאוּבֵן גָּד וְאָשֵׁר וּזְבוּלוּן דָּן וְנַפְתָּלִי.
יד וְעָנ֣וּ הַלְוִיִּ֗ם וְאָ֥מְר֛וּ אֶל־כׇּל־אִ֥ישׁ יִשְׂרָאֵ֖ל ק֥וֹל רָֽם׃ וְהַלְוִיִּים יַכְרִיזוּ וְיֹאמְרוּ אֶל כָּל אַנְשֵׁי יִשְׂרָאֵל בְּקוֹל רָם:
ס
טו אָר֣וּר הָאִ֡ישׁ אֲשֶׁ֣ר יַעֲשֶׂה֩ פֶ֨סֶל וּמַסֵּכָ֜ה תּוֹעֲבַ֣ת יְהֹוָ֗ה מַעֲשֵׂ֛ה יְדֵ֥י חָרָ֖שׁ וְשָׂ֣ם בַּסָּ֑תֶר וְעָנ֧וּ כׇל־הָעָ֛ם וְאָמְר֖וּ אָמֵֽן׃ אָרוּר הָאִישׁ שֶׁיַּעֲשֶׂה תּוֹעֵבָה בְּעֵינֵי ה' – פֶּסֶל אוֹ דְּמוּת יְצוּקָה מַעֲשֵׂה יְדֵי אֻמָּן, וְיָשִׂים אוֹתוֹ בַּסֵּתֶר, וְיַעֲנוּ כָּל הָעָם וְיֹאמְרוּ: אָמֵן.
ס
טז אָר֕וּר מַקְלֶ֥ה אָבִ֖יו וְאִמּ֑וֹ וְאָמַ֥ר כׇּל־הָעָ֖ם אָמֵֽן׃ אָרוּר מְבַזֶּה אָבִיו וְאִמּוֹ, וְיֹאמַר כָּל הָעָם: אָמֵן.
ס
יז אָר֕וּר מַסִּ֖יג גְּב֣וּל רֵעֵ֑הוּ וְאָמַ֥ר כׇּל־הָעָ֖ם אָמֵֽן׃ אָרוּר מִי שֶׁמֵּזִיז לְאָחוֹר אֶת גְּבוּל הָאֲחֻזָּה שֶׁל חֲבֵרוֹ, וְיֹאמַר כָּל הָעָם: אָמֵן.

יח אָר֕וּר מַשְׁגֶּ֥ה עִוֵּ֖ר בַּדָּ֑רֶךְ וְאָמַ֥ר כׇּל־הָעָ֖ם אָמֵֽן׃ אָרוּר מַטְעֶה עִוֵּר בַּדֶּרֶךְ, וְיֹאמַר כָּל הָעָם: אָמֵן.

פ פָּרָשַׁת הַבְּרָכוֹת
כח א וְהָיָ֗ה אִם־שָׁמ֤וֹעַ תִּשְׁמַע֙ בְּקוֹל֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ לִשְׁמֹ֤ר לַעֲשׂוֹת֙ אֶת־כׇּל־מִצְוֺתָ֔יו אֲשֶׁ֛ר אָנֹכִ֥י מְצַוְּךָ֖ הַיּ֑וֹם וּנְתָ֨נְךָ֜ יְהֹוָ֤ה אֱלֹהֶ֙יךָ֙ עֶלְי֔וֹן עַ֖ל כׇּל־גּוֹיֵ֥י הָאָֽרֶץ׃ אִם תִּשְׁמַע בְּקוֹל ה' אֱלֹהֶיךָ וְתַקְפִּיד לְקַיֵּם אֶת כָּל מִצְווֹתָיו שֶׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם, ה' אֱלֹהֶיךָ יַעֲשֶׂה שֶׁתִּהְיֶה עֶלְיוֹן מֵעַל כָּל עַמֵּי הָעוֹלָם.
ב וּבָ֧אוּ עָלֶ֛יךָ כׇּל־הַבְּרָכ֥וֹת הָאֵ֖לֶּה וְהִשִּׂיגֻ֑ךָ כִּ֣י תִשְׁמַ֔ע בְּק֖וֹל יְהֹוָ֥ה אֱלֹהֶֽיךָ׃ כָּל הַבְּרָכוֹת הָאֵלֶּה יָבוֹאוּ עָלֶיךָ וְיַגִּיעוּ אֵלֶיךָ כִּי תִּשְׁמַע בְּקוֹל ה' אֱלֹהֶיךָ.
ג בָּר֥וּךְ אַתָּ֖ה בָּעִ֑יר וּבָר֥וּךְ אַתָּ֖ה בַּשָּׂדֶֽה׃ בָּרוּךְ אַתָּה בָּעִיר וּבָרוּךְ אַתָּה בַּשָּׂדֶה.
ד בָּר֧וּךְ פְּרִֽי־בִטְנְךָ֛ וּפְרִ֥י אַדְמָתְךָ֖ וּפְרִ֣י בְהֶמְתֶּ֑ךָ שְׁגַ֥ר אֲלָפֶ֖יךָ וְעַשְׁתְּר֥וֹת צֹאנֶֽךָ׃ בְּרוּכִים יְלִידֵי בִּטְנְךָ וּפֵרוֹת אַדְמָתְךָ וּוְלָדוֹת בֶּהֱמוֹתֶיךָ – וְלָדוֹת שְׁוָרֶיךָ וְעֶדְרֵי צֹאנְךָ.
ה בָּר֥וּךְ טַנְאֲךָ֖ וּמִשְׁאַרְתֶּֽךָ׃ בָּרוּךְ תִּהְיֶה בְּפֵרוֹת הַסַּל שֶׁלְּךָ וּבַבָּצֵק שֶׁלְּךָ.
ו בָּר֥וּךְ אַתָּ֖ה בְּבֹאֶ֑ךָ וּבָר֥וּךְ אַתָּ֖ה בְּצֵאתֶֽךָ׃ בָּרוּךְ אַתָּה בִּכְנִיסָתְךָ וּבָרוּךְ אַתָּה בְּצֵאתְךָ.

ששי[עריכה]

ז יִתֵּ֨ן יְהֹוָ֤ה אֶת־אֹיְבֶ֙יךָ֙ הַקָּמִ֣ים עָלֶ֔יךָ נִגָּפִ֖ים לְפָנֶ֑יךָ בְּדֶ֤רֶךְ אֶחָד֙ יֵצְא֣וּ אֵלֶ֔יךָ וּבְשִׁבְעָ֥ה דְרָכִ֖ים יָנ֥וּסוּ לְפָנֶֽיךָ׃ ה' יַעֲשֶׂה שֶׁאוֹיְבֶיךָ הָעוֹמְדִים נֶגְדְּךָ יֻבְסוּ לְפָנֶיךָ, הֵם יֵצְאוּ אֵלֶיךָ בְּדֶרֶךְ אַחַת, וְיִבְרְחוּ וְיִתְפַּזְּרוּ מִלְּפָנֶיךָ בְּשִׁבְעָה דְּרָכִים.
ח יְצַ֨ו יְהֹוָ֤ה אִתְּךָ֙ אֶת־הַבְּרָכָ֔ה בַּאֲסָמֶ֕יךָ וּבְכֹ֖ל מִשְׁלַ֣ח יָדֶ֑ךָ וּבֵ֣רַכְךָ֔ בָּאָ֕רֶץ אֲשֶׁר־יְהֹוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֥ן לָֽךְ׃ ה' יְצַוֶּה אֶת הַבְּרָכָה שֶׁתִּהְיֶה אִתְּךָ בַּאֲסָמֶיךָ וּבְכָל מַעֲשֵׂי יָדֶךָ, וִיבָרֵךְ אוֹתְךָ בָּאָרֶץ שֶׁה' אֱלֹהֶיךָ נוֹתֵן לְךָ.
ט יְקִֽימְךָ֨ יְהֹוָ֥ה לוֹ֙ לְעַ֣ם קָד֔וֹשׁ כַּאֲשֶׁ֖ר נִֽשְׁבַּֽע־לָ֑ךְ כִּ֣י תִשְׁמֹ֗ר אֶת־מִצְוֺת֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ וְהָלַכְתָּ֖ בִּדְרָכָֽיו׃ ה' יַעֲשֶׂה אוֹתְךָ לָעָם הַקָּדוֹשׁ שֶׁלּוֹ כְּמוֹ שֶׁנִּשְׁבַּע לְךָ, כִּי תִּשְׁמוֹר אֶת מִצְווֹת ה' אֱלֹהֶיךָ וְתֵלֵךְ בִּדְרָכָיו.
י וְרָאוּ֙ כׇּל־עַמֵּ֣י הָאָ֔רֶץ כִּ֛י שֵׁ֥ם יְהֹוָ֖ה נִקְרָ֣א עָלֶ֑יךָ וְיָֽרְא֖וּ מִמֶּֽךָּ׃ כָּל עַמֵּי הָעוֹלָם יִרְאוּ שֶׁשֵּׁם ה' נִקְרָא עָלֶיךָ, וִיכַבְּדוּ אוֹתְךָ.
יא וְהוֹתִרְךָ֤ יְהֹוָה֙ לְטוֹבָ֔ה בִּפְרִ֧י בִטְנְךָ֛ וּבִפְרִ֥י בְהֶמְתְּךָ֖ וּבִפְרִ֣י אַדְמָתֶ֑ךָ עַ֚ל הָאֲדָמָ֔ה אֲשֶׁ֨ר נִשְׁבַּ֧ע יְהֹוָ֛ה לַאֲבֹתֶ֖יךָ לָ֥תֶת לָֽךְ׃ ה' יוֹסִיף לְךָ טוֹבָה – בִּילָדֶיךָ בִּוְלָדוֹת בֶּהֱמוֹתֶיךָ וּבְפֵרוֹת אַדְמָתְךָ, עַל הָאֲדָמָה שֶׁה' נִשְׁבַּע לַאֲבוֹתֶיךָ לָתֵת לְךָ.
יב יִפְתַּ֣ח יְהֹוָ֣ה ׀ לְ֠ךָ֠ אֶת־אוֹצָר֨וֹ הַטּ֜וֹב אֶת־הַשָּׁמַ֗יִם לָתֵ֤ת מְטַֽר־אַרְצְךָ֙ בְּעִתּ֔וֹ וּלְבָרֵ֕ךְ אֵ֖ת כׇּל־מַעֲשֵׂ֣ה יָדֶ֑ךָ וְהִלְוִ֙יתָ֙ גּוֹיִ֣ם רַבִּ֔ים וְאַתָּ֖ה לֹ֥א תִלְוֶֽה׃ ה' יִפְתַּח לְךָ אֶת הַמַּחְסָן שֶׁל טוּבוֹ – אֶת הַשָּׁמַיִם, כְּדֵי לְהוֹרִיד אֶת הַגֶּשֶׁם לְאַרְצְךָ בִּזְמַנּוֹ, וּלְבָרֵךְ אֶת כָּל מַעֲשֵׂה יָדְךָ, תַּלְוֶה לְעַמִּים רַבִּים וְאַתָּה לֹא תִּזְדַּקֵּק לְהַלְוָאָה.
יג וּנְתָֽנְךָ֨ יְהֹוָ֤ה לְרֹאשׁ֙ וְלֹ֣א לְזָנָ֔ב וְהָיִ֙יתָ֙ רַ֣ק לְמַ֔עְלָה וְלֹ֥א תִהְיֶ֖ה לְמָ֑טָּה כִּֽי־תִשְׁמַ֞ע אֶל־מִצְוֺ֣ת ׀ יְהֹוָ֣ה אֱלֹהֶ֗יךָ אֲשֶׁ֨ר אָנֹכִ֧י מְצַוְּךָ֛ הַיּ֖וֹם לִשְׁמֹ֥ר וְלַעֲשֽׂוֹת׃ ה' יַעֲשֶׂה שֶׁתִּהְיֶה שׁוֹלֵט וְלֹא נִשְׁלָט, וְתִהְיֶה רַק לְמַעְלָה בַּחֲשִׁיבוּת וְלֹא תִּהְיֶה לְמַטָּה, כִּי אַתָּה תְּצַיֵּת לְמִצְווֹת ה' אֱלֹהֶיךָ שֶׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם לְהַקְפִּיד לְקַיֵּם,
יד וְלֹ֣א תָס֗וּר מִכׇּל־הַדְּבָרִים֙ אֲשֶׁ֨ר אָנֹכִ֜י מְצַוֶּ֥ה אֶתְכֶ֛ם הַיּ֖וֹם יָמִ֣ין וּשְׂמֹ֑אול לָלֶ֗כֶת אַחֲרֵ֛י אֱלֹהִ֥ים אֲחֵרִ֖ים לְעׇבְדָֽם׃ וְלֹא תָּסוּר מִכָּל הַדְּבָרִים שֶׁאֲנִי מְצַוֶּה אֶתְכֶם הַיּוֹם לְיָמִין אוֹ לִשְׂמֹאל לָלֶכֶת אַחֲרֵי אֱלֹהִים אֲחֵרִים לַעֲבוֹד אוֹתָם.
פ פָּרָשַׁת הַקְּלָלוֹת
טו וְהָיָ֗ה אִם־לֹ֤א תִשְׁמַע֙ בְּקוֹל֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ לִשְׁמֹ֤ר לַעֲשׂוֹת֙ אֶת־כׇּל־מִצְוֺתָ֣יו וְחֻקֹּתָ֔יו אֲשֶׁ֛ר אָנֹכִ֥י מְצַוְּךָ֖ הַיּ֑וֹם וּבָ֧אוּ עָלֶ֛יךָ כׇּל־הַקְּלָל֥וֹת הָאֵ֖לֶּה וְהִשִּׂיגֽוּךָ׃ וְאִם לֹא תִּשְׁמַע בְּקוֹל ה' אֱלֹהֶיךָ לְהַקְפִּיד לְקַיֵּם אֶת כָּל הַמִּצְווֹת וְהַחֻקִּים שֶׁלּוֹ שֶׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם, כָּל הַקְּלָלוֹת הָאֵלֶּה יָבוֹאוּ עָלֶיךָ וְיַגִּיעוּ אֵלֶיךָ.
טז אָר֥וּר אַתָּ֖ה בָּעִ֑יר וְאָר֥וּר אַתָּ֖ה בַּשָּׂדֶֽה׃ אָרוּר אַתָּה בָּעִיר, וְאָרוּר אַתָּה בַּשָּׂדֶה.
יז אָר֥וּר טַנְאֲךָ֖ וּמִשְׁאַרְתֶּֽךָ׃ אָרוּר תִּהְיֶה בְּפֵרוֹת סַלְּךָ וּבַבָּצֵק שֶׁלְּךָ.
יח אָר֥וּר פְּרִֽי־בִטְנְךָ֖ וּפְרִ֣י אַדְמָתֶ֑ךָ שְׁגַ֥ר אֲלָפֶ֖יךָ וְעַשְׁתְּרֹ֥ת צֹאנֶֽךָ׃ אָרוּרִים יְלִידֵי בִּטְנְךָ וּפֵרוֹת אַדְמָתְךָ, וְלָדוֹת שְׁוָרֶיךָ וְעֶדְרֵי צֹאנְךָ.
יט אָר֥וּר אַתָּ֖ה בְּבֹאֶ֑ךָ וְאָר֥וּר אַתָּ֖ה בְּצֵאתֶֽךָ׃ אָרוּר אַתָּה בִכְנִיסָתְךָ, וְאָרוּר אַתָּה בְּצֵאתְךָ.
כ יְשַׁלַּ֣ח יְהֹוָ֣ה ׀ בְּ֠ךָ֠ אֶת־הַמְּאֵרָ֤ה אֶת־הַמְּהוּמָה֙ וְאֶת־הַמִּגְעֶ֔רֶת בְּכׇל־מִשְׁלַ֥ח יָדְךָ֖ אֲשֶׁ֣ר תַּעֲשֶׂ֑ה עַ֣ד הִשָּׁמֶדְךָ֤ וְעַד־אֲבׇדְךָ֙ מַהֵ֔ר מִפְּנֵ֛י רֹ֥עַ מַֽעֲלָלֶ֖יךָ אֲשֶׁ֥ר עֲזַבְתָּֽנִי׃ ה' יִשְׁלַח עָלֶיךָ אֶת הַקְּלָלָה אֶת הַבֶּהָלָה וְאֶת הַגְּעָרָה בְּכָל מַעֲשֵׂי יָדְךָ שֶׁתַּעֲשֶׂה, עַד שֶׁתֻּשְׁמַד וְעַד שֶׁתֹּאבַד מַהֵר, בִּגְלַל מַעֲשֶׂיךָ הָרָעִים – שֶׁעֲזַבְתָּ אוֹתִי.
כא יַדְבֵּ֧ק יְהֹוָ֛ה בְּךָ֖ אֶת־הַדָּ֑בֶר עַ֚ד כַּלֹּת֣וֹ אֹֽתְךָ֔ מֵעַל֙ הָאֲדָמָ֔ה אֲשֶׁר־אַתָּ֥ה בָא־שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ ה' יַדְבִּיק אוֹתְךָ בְּדֶבֶר עַד שֶׁיַּשְׁמִיד אוֹתְךָ מֵעַל הָאֲדָמָה שֶׁאַתָּה בָּא לְשָׁם לִכְבּוֹשׁ אוֹתָהּ.
כב יַכְּכָ֣ה יְ֠הֹוָ֠ה בַּשַּׁחֶ֨פֶת וּבַקַּדַּ֜חַת וּבַדַּלֶּ֗קֶת וּבַֽחַרְחֻר֙ וּבַחֶ֔רֶב וּבַשִּׁדָּפ֖וֹן וּבַיֵּרָק֑וֹן וּרְדָפ֖וּךָ עַ֥ד אׇבְדֶֽךָ׃ ה' יַכֶּה אוֹתְךָ בְּמַחֲלַת הַשַּׁחֶפֶת וְהַקַּדַּחַת וְהַדַּלֶּקֶת וּבְחוֹם גָּבוֹהַּ וּבְמִלְחָמָה וּבְיִבּוּשׁ וּבְרִקְבוֹן הַתְּבוּאָה, וְצָרוֹת אֵלֶּה יִרְדְּפוּ אוֹתְךָ עַד שֶׁתֻּשְׁמַד.
כג וְהָי֥וּ שָׁמֶ֛יךָ אֲשֶׁ֥ר עַל־רֹאשְׁךָ֖ נְחֹ֑שֶׁת וְהָאָ֥רֶץ אֲשֶׁר־תַּחְתֶּ֖יךָ בַּרְזֶֽל׃ הַשָּׁמַיִם מֵעַל רֹאשְׁךָ יִהְיוּ יְבֵשִׁים מִגֶּשֶׁם כִּנְחוֹשֶׁת, וְהָאָרֶץ תַּחְתֶּיךָ תִּהְיֶה יְבֵשָׁה כְּבַרְזֶל.
כד יִתֵּ֧ן יְהֹוָ֛ה אֶת־מְטַ֥ר אַרְצְךָ֖ אָבָ֣ק וְעָפָ֑ר מִן־הַשָּׁמַ֙יִם֙ יֵרֵ֣ד עָלֶ֔יךָ עַ֖ד הִשָּׁמְדָֽךְ׃ ה' יַהֲפוֹךְ אֶת מְטַר אַרְצְךָ לְאָבָק וְעָפָר שֶׁיֵּרֵד עָלֶיךָ מֵהַשָּׁמַיִם עַד שֶׁתֻּשְׁמַד.
כה יִתֶּנְךָ֨ יְהֹוָ֥ה ׀ נִגָּף֮ לִפְנֵ֣י אֹיְבֶ֒יךָ֒ בְּדֶ֤רֶךְ אֶחָד֙ תֵּצֵ֣א אֵלָ֔יו וּבְשִׁבְעָ֥ה דְרָכִ֖ים תָּנ֣וּס לְפָנָ֑יו וְהָיִ֣יתָ לְזַֽעֲוָ֔ה לְכֹ֖ל מַמְלְכ֥וֹת הָאָֽרֶץ׃ ה' יַעֲשֶׂה שֶׁתּוּבַס לִפְנֵי אוֹיְבֶיךָ, תֵּצֵא אֵלָיו בְּדֶרֶךְ אַחַת וְתִבְרַח מִפָּנָיו וְתִתְפַּזֵּר מִפָּנָיו בְּשִׁבְעָה דְּרָכִים, תִּהְיֶה דֻּגְמָה לִזְוָעָה לְכָל מַמְלְכוֹת הָעוֹלָם.
כו וְהָיְתָ֤ה נִבְלָֽתְךָ֙ לְמַֽאֲכָ֔ל לְכׇל־ע֥וֹף הַשָּׁמַ֖יִם וּלְבֶהֱמַ֣ת הָאָ֑רֶץ וְאֵ֖ין מַחֲרִֽיד׃ נִבְלָתְךָ תִּהְיֶה לְמַאֲכָל לְכָל עוֹפוֹת הַשָּׁמַיִם וּלְבֶהֱמוֹת הָאָרֶץ, וְאֵין מִי שֶׁיַּפְחִיד וִיגָרֵשׁ אוֹתָם.
כז יַכְּכָ֨ה יְהֹוָ֜ה בִּשְׁחִ֤ין מִצְרַ֙יִם֙ ובעפלים וּבַטְּחֹרִ֔ים וּבַגָּרָ֖ב וּבֶחָ֑רֶס אֲשֶׁ֥ר לֹא־תוּכַ֖ל לְהֵרָפֵֽא׃ ה' יַכֶּה אוֹתְךָ בִּשְׁחִין מִצְרַיִם וּבְמַחֲלַת הַטְּחוֹרִים וּבַאֲבַעְבּוּעוֹת וּבְגָרֶדֶת, שֶׁלֹּא תּוּכַל לְהִתְרַפֵּא.
כח יַכְּכָ֣ה יְהֹוָ֔ה בְּשִׁגָּע֖וֹן וּבְעִוָּר֑וֹן וּבְתִמְה֖וֹן לֵבָֽב׃ ה' יַכֶּה אוֹתְךָ בְּשִׁגָּעוֹן בְּעִוָּרוֹן וּבְהֶלֶם.
כט וְהָיִ֜יתָ מְמַשֵּׁ֣שׁ בַּֽצׇּהֳרַ֗יִם כַּאֲשֶׁ֨ר יְמַשֵּׁ֤שׁ הַֽעִוֵּר֙ בָּאֲפֵלָ֔ה וְלֹ֥א תַצְלִ֖יחַ אֶת־דְּרָכֶ֑יךָ וְהָיִ֜יתָ אַ֣ךְ עָשׁ֧וּק וְגָז֛וּל כׇּל־הַיָּמִ֖ים וְאֵ֥ין מוֹשִֽׁיעַ׃ תִּהְיֶה מְמַשֵּׁשׁ בַּצָּהֳרַיִם כְּמוֹ שֶׁהָעִוֵּר מְמַשֵּׁשׁ בְּלִי לִרְאוֹת, וְלֹא תַּצְלִיחַ בִּדְרָכֶיךָ, וְתִהְיֶה כָּל הַזְּמָן רַק עָשׁוּק וְגָזוּל וְאֵין מוֹשִׁיעַ.
ל אִשָּׁ֣ה תְאָרֵ֗שׂ וְאִ֤ישׁ אַחֵר֙ ישגלנה יִשְׁכָּבֶ֔נָּה בַּ֥יִת תִּבְנֶ֖ה וְלֹא־תֵשֵׁ֣ב בּ֑וֹ כֶּ֥רֶם תִּטַּ֖ע וְלֹ֥א תְחַלְּלֶֽנּוּ׃ תְּאָרֵס אִשָּׁה וְאִישׁ אַחֵר יִשְׁכַּב אִתָּהּ, תִּבְנֶה בַּיִת וְלֹא תֵּשֵׁב בּוֹ, תִּטַּע כֶּרֶם וְלֹא תִּפְדֶּה אוֹתוֹ בַּשָּׁנָה הָרְבִיעִית כְּדֵי לֶאֱכוֹל מִמֶּנּוּ.
לא שׁוֹרְךָ֞ טָב֣וּחַ לְעֵינֶ֗יךָ וְלֹ֣א תֹאכַל֮ מִמֶּ֒נּוּ֒ חֲמֹֽרְךָ֙ גָּז֣וּל מִלְּפָנֶ֔יךָ וְלֹ֥א יָשׁ֖וּב לָ֑ךְ צֹֽאנְךָ֙ נְתֻנ֣וֹת לְאֹיְבֶ֔יךָ וְאֵ֥ין לְךָ֖ מוֹשִֽׁיעַ׃ שׁוֹרְךָ שָׁחוּט לְעֵינֶיךָ וְלֹא תֹּאכַל מִמֶּנּוּ, חֲמוֹרְךָ גָּזוּל מִלְּפָנֶיךָ וְהוּא לֹא יַחֲזוֹר אֵלֶיךָ, צֹאנְךָ נָתוּן לְאוֹיְבֶיךָ וְאֵין לְךָ מוֹשִׁיעַ.
לב בָּנֶ֨יךָ וּבְנֹתֶ֜יךָ נְתֻנִ֨ים לְעַ֤ם אַחֵר֙ וְעֵינֶ֣יךָ רֹא֔וֹת וְכָל֥וֹת אֲלֵיהֶ֖ם כׇּל־הַיּ֑וֹם וְאֵ֥ין לְאֵ֖ל יָדֶֽךָ׃ בָּנֶיךָ וּבְנוֹתֶיךָ שְׁבוּיִים אֵצֶל עַם אַחֵר, וְעֵינֶיךָ רוֹאוֹת וּמְצַפּוֹת אֲלֵיהֶם כָּל הַיּוֹם וְאֵין בְּכוֹחֲךָ לְהַצִּילָם.
לג פְּרִ֤י אַדְמָֽתְךָ֙ וְכׇל־יְגִ֣יעֲךָ֔ יֹאכַ֥ל עַ֖ם אֲשֶׁ֣ר לֹא־יָדָ֑עְתָּ וְהָיִ֗יתָ רַ֛ק עָשׁ֥וּק וְרָצ֖וּץ כׇּל־הַיָּמִֽים׃ עַם שֶׁלֹּא הִכַּרְתָּ יֹאכַל אֶת פְּרִי אַדְמָתְךָ וְאֶת כָּל מַה שֶּׁטָּרַחְתָּ לְהַרְוִיח, וְתִהְיֶה רַק עָשׁוּק וְשָׁבוּר כָּל הַיָּמִים.
לד וְהָיִ֖יתָ מְשֻׁגָּ֑ע מִמַּרְאֵ֥ה עֵינֶ֖יךָ אֲשֶׁ֥ר תִּרְאֶֽה׃ תִּהְיֶה מְשֻׁגָּע מִמַּרְאֵה עֵינֶיךָ שֶׁתִּרְאֶה.
לה יַכְּכָ֨ה יְהֹוָ֜ה בִּשְׁחִ֣ין רָ֗ע עַל־הַבִּרְכַּ֙יִם֙ וְעַל־הַשֹּׁקַ֔יִם אֲשֶׁ֥ר לֹא־תוּכַ֖ל לְהֵרָפֵ֑א מִכַּ֥ף רַגְלְךָ֖ וְעַ֥ד קׇדְקֳדֶֽךָ׃ ה' יַכֶּה אוֹתְךָ בִּשְׁחִין רָע עַל הַבִּרְכַּיִם וְעַל הַשּׁוֹקַיִם שֶׁלֹּא תּוּכַל לְהִתְרַפֵּא, מִכַּף רַגְלְךָ וְעַד קָדְקָדְךָ.
לו יוֹלֵ֨ךְ יְהֹוָ֜ה אֹתְךָ֗ וְאֶֽת־מַלְכְּךָ֙ אֲשֶׁ֣ר תָּקִ֣ים עָלֶ֔יךָ אֶל־גּ֕וֹי אֲשֶׁ֥ר לֹא־יָדַ֖עְתָּ אַתָּ֣ה וַאֲבֹתֶ֑יךָ וְעָבַ֥דְתָּ שָּׁ֛ם אֱלֹהִ֥ים אֲחֵרִ֖ים עֵ֥ץ וָאָֽבֶן׃ ה' יוֹלִיךְ אוֹתְךָ וְאֶת מַלְכְּךָ שֶׁתָּקִים עָלֶיךָ אֶל עַם שֶׁלֹּא הִכַּרְתָּ אַתָּה וַאֲבוֹתֶיךָ, וְתַעֲבוֹד שָׁם עוֹבְדֵי אֱלֹהִים אֲחֵרִים שֶׁעֲשׂוּיִים מֵעֵץ וְאֶבֶן.
לז וְהָיִ֣יתָ לְשַׁמָּ֔ה לְמָשָׁ֖ל וְלִשְׁנִינָ֑ה בְּכֹל֙ הָֽעַמִּ֔ים אֲשֶׁר־יְנַהֶגְךָ֥ יְהֹוָ֖ה שָֽׁמָּה׃ וְתִהְיֶה לִפְלִיאָה לְדוּגְמָה וּלְשִׂיחָה בְּכָל הָעַמִּים שֶׁה' יוֹבִיל אוֹתְךָ לְשָׁם.
לח זֶ֥רַע רַ֖ב תּוֹצִ֣יא הַשָּׂדֶ֑ה וּמְעַ֣ט תֶּאֱסֹ֔ף כִּ֥י יַחְסְלֶ֖נּוּ הָאַרְבֶּֽה׃ הַשָּׂדֶה תַּצְמִיחַ הַרְבֵּה תְּבוּאָה, וְאַתָּה תֶּאֱסוֹף מְעַט כִּי הָאַרְבֶּה יְחַסֵּל אוֹתָהּ.
לט כְּרָמִ֥ים תִּטַּ֖ע וְעָבָ֑דְתָּ וְיַ֤יִן לֹֽא־תִשְׁתֶּה֙ וְלֹ֣א תֶאֱגֹ֔ר כִּ֥י תֹאכְלֶ֖נּוּ הַתֹּלָֽעַת׃ תִּטַּע כְּרָמִים וְתַעֲבוֹד, וְלֹא תִּשְׁתֶּה וְלֹא תֶּאֱגוֹר יַיִן כִּי הַתּוֹלַעַת תֹּאכַל אוֹתוֹ.
מ זֵיתִ֛ים יִהְי֥וּ לְךָ֖ בְּכׇל־גְּבוּלֶ֑ךָ וְשֶׁ֙מֶן֙ לֹ֣א תָס֔וּךְ כִּ֥י יִשַּׁ֖ל זֵיתֶֽךָ׃ יִהְיוּ לְךָ זֵיתִים בְּכָל גְּבוּלְךָ, וְלֹא תִּמְרַח שֶׁמֶן עַל גּוּפְךָ כִּי זֵיתֵיךָ יִנְשְׁרוּ.
מא בָּנִ֥ים וּבָנ֖וֹת תּוֹלִ֑יד וְלֹא־יִהְי֣וּ לָ֔ךְ כִּ֥י יֵלְכ֖וּ בַּשֶּֽׁבִי׃ תּוֹלִיד בָּנִים וּבָנוֹת, וְהֵם לֹא יִהְיוּ שֶׁלְּךָ כִּי יֵלְכוּ בַּשֶּׁבִי.
מב כׇּל־עֵצְךָ֖ וּפְרִ֣י אַדְמָתֶ֑ךָ יְיָרֵ֖שׁ הַצְּלָצַֽל׃ הַחֲגָבִים יְכַלּוּ אֶת כָּל עֵצֶיךָ וּפֵרוֹת אַדְמָתְךָ.
מג הַגֵּר֙ אֲשֶׁ֣ר בְּקִרְבְּךָ֔ יַעֲלֶ֥ה עָלֶ֖יךָ מַ֣עְלָה מָּ֑עְלָה וְאַתָּ֥ה תֵרֵ֖ד מַ֥טָּה מָּֽטָּה׃ הַגּוֹי שֶׁגָּר בְּאַרְצְךָ יַעֲלֶה עָלֶיךָ מַעְלָה מַעְלָה בַּחֲשִׁיבוּת, וְאַתָּה תֵּרֵד מַטָּה מַטָּה.
מד ה֣וּא יַלְוְךָ֔ וְאַתָּ֖ה לֹ֣א תַלְוֶ֑נּוּ ה֚וּא יִהְיֶ֣ה לְרֹ֔אשׁ וְאַתָּ֖ה תִּֽהְיֶ֥ה לְזָנָֽב׃ הוּא יַלְוֶה לְךָ וְאַתָּה לֹא תַּלְוֶה לוֹ, הוּא יִהְיֶה הַשּׁוֹלֵט וְאַתָּה תִּהְיֶה הַנִּשְׁלָט.
מה וּבָ֨אוּ עָלֶ֜יךָ כׇּל־הַקְּלָל֣וֹת הָאֵ֗לֶּה וּרְדָפ֙וּךָ֙ וְהִשִּׂיג֔וּךָ עַ֖ד הִשָּׁמְדָ֑ךְ כִּי־לֹ֣א שָׁמַ֗עְתָּ בְּקוֹל֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ לִשְׁמֹ֛ר מִצְוֺתָ֥יו וְחֻקֹּתָ֖יו אֲשֶׁ֥ר צִוָּֽךְ׃ וְיָבוֹאוּ עָלֶיךָ כָּל הַקְּלָלוֹת הָאֵלֶּה וְיִרְדְּפוּ אַחֲרֶיךָ וְיַשִּׂיגוּ אוֹתְךָ עַד שֶׁתֻּשְׁמַד, כִּי לֹא שָׁמַעְתָּ בְּקוֹל ה' אֱלֹהֶיךָ לִשְׁמוֹר אֶת הַמִּצְווֹת וְהַחֻקִּים שֶׁלּוֹ שֶׁצִּוָּה אוֹתְךָ.
מו וְהָי֣וּ בְךָ֔ לְא֖וֹת וּלְמוֹפֵ֑ת וּֽבְזַרְעֲךָ֖ עַד־עוֹלָֽם׃ וְהֵם יִהְיוּ בְּךָ וּבְצֶאֱצָאֶיךָ עַד עוֹלָם לְסִימָן וּלְהוֹכָחָה,
מז תַּ֗חַת אֲשֶׁ֤ר לֹא־עָבַ֙דְתָּ֙ אֶת־יְהֹוָ֣ה אֱלֹהֶ֔יךָ בְּשִׂמְחָ֖ה וּבְט֣וּב לֵבָ֑ב מֵרֹ֖ב כֹּֽל׃ כְּעוֹנֶשׁ עַל כָּךְ שֶׁלֹּא עָבַדְתָּ אֶת ה' אֱלֹהֶיךָ בְּשִׂמְחָה וּבְטוּב לֵב כְּשֶׁהָיָה לְךָ הַרְבֵּה מֵהַכֹּל.
מח וְעָבַדְתָּ֣ אֶת־אֹיְבֶ֗יךָ אֲשֶׁ֨ר יְשַׁלְּחֶ֤נּוּ יְהֹוָה֙ בָּ֔ךְ בְּרָעָ֧ב וּבְצָמָ֛א וּבְעֵירֹ֖ם וּבְחֹ֣סֶר כֹּ֑ל וְנָתַ֞ן עֹ֤ל בַּרְזֶל֙ עַל־צַוָּארֶ֔ךָ עַ֥ד הִשְׁמִיד֖וֹ אֹתָֽךְ׃ תִּהְיֶה מְשֻׁעְבָּד לְאוֹיְבֶיךָ שֶׁה' יָבִיא עָלֶיךָ, בְּרָעָב וּבְצָמָא וּבְמַחְסוֹר בִּבְגָדִים וּבְחוֹסֶר כֹּל, וְיִתֵּן עוֹל בַּרְזֶל עַל צַוָּארְךָ עַד שֶׁיַּשְׁמִיד אוֹתְךָ.
מט יִשָּׂ֣א יְהֹוָה֩ עָלֶ֨יךָ גּ֤וֹי מֵרָחֹק֙ מִקְצֵ֣ה הָאָ֔רֶץ כַּאֲשֶׁ֥ר יִדְאֶ֖ה הַנָּ֑שֶׁר גּ֕וֹי אֲשֶׁ֥ר לֹא־תִשְׁמַ֖ע לְשֹׁנֽוֹ׃ ה' יַעֲלֶה עָלֶיךָ עַם מִמֶּרְחָק מִקְצֵה הָעוֹלָם בִּמְהִירוּת כְּמוֹ שֶׁנֶּשֶׁר עָף, עַם שֶׁלֹּא תָּבִין אֶת שְׂפָתוֹ.
נ גּ֖וֹי עַ֣ז פָּנִ֑ים אֲשֶׁ֨ר לֹא־יִשָּׂ֤א פָנִים֙ לְזָקֵ֔ן וְנַ֖עַר לֹ֥א יָחֹֽן׃ עַם חָצוּף, שֶׁלֹּא יְכַבֵּד זָקֵן וְלֹא יְרַחֵם עַל נַעַר.
נא וְ֠אָכַ֠ל פְּרִ֨י בְהֶמְתְּךָ֥ וּפְרִֽי־אַדְמָתְךָ֮ עַ֣ד הִשָּׁמְדָךְ֒ אֲשֶׁ֨ר לֹא־יַשְׁאִ֜יר לְךָ֗ דָּגָן֙ תִּיר֣וֹשׁ וְיִצְהָ֔ר שְׁגַ֥ר אֲלָפֶ֖יךָ וְעַשְׁתְּרֹ֣ת צֹאנֶ֑ךָ עַ֥ד הַאֲבִיד֖וֹ אֹתָֽךְ׃ וְהוּא יֹאכַל אֶת וְלָדוֹת בֶּהֱמוֹתֶיךָ וּפֵרוֹת אַדְמָתְךָ עַד שֶׁתֻּשְׁמַד – שֶׁלֹּא יַשְׁאִיר לְךָ תְּבוּאָה יַיִן וְשֶׁמֶן, וְלָדוֹת שְׁוָרֶיךָ וְעֶדְרֵי צֹאנְךָ, עַד שֶׁיַּשְׁמִיד אוֹתְךָ.
נב וְהֵצַ֨ר לְךָ֜ בְּכׇל־שְׁעָרֶ֗יךָ עַ֣ד רֶ֤דֶת חֹמֹתֶ֙יךָ֙ הַגְּבֹהֹ֣ת וְהַבְּצֻר֔וֹת אֲשֶׁ֥ר אַתָּ֛ה בֹּטֵ֥חַ בָּהֵ֖ן בְּכׇל־אַרְצֶ֑ךָ וְהֵצַ֤ר לְךָ֙ בְּכׇל־שְׁעָרֶ֔יךָ בְּכׇ֨ל־אַרְצְךָ֔ אֲשֶׁ֥ר נָתַ֛ן יְהֹוָ֥ה אֱלֹהֶ֖יךָ לָֽךְ׃ הוּא יַעֲשֶׂה עָלֶיךָ מָצוֹר בְּכָל עָרֶיךָ בְּכָל אַרְצְךָ עַד לִנְפִילַת חוֹמוֹתֶיךָ הַגְּבוֹהוֹת וְהַבְּצוּרוֹת שֶׁאַתָּה בּוֹטֵחַ בָּהֵן, הוּא יַעֲשֶׂה עָלֶיךָ מָצוֹר בְּכָל עָרֶיךָ בְּכָל אַרְצְךָ שֶׁה' אֱלֹהֶיךָ נָתַן לְךָ.
נג וְאָכַלְתָּ֣ פְרִֽי־בִטְנְךָ֗ בְּשַׂ֤ר בָּנֶ֙יךָ֙ וּבְנֹתֶ֔יךָ אֲשֶׁ֥ר נָתַן־לְךָ֖ יְהֹוָ֣ה אֱלֹהֶ֑יךָ בְּמָצוֹר֙ וּבְמָצ֔וֹק אֲשֶׁר־יָצִ֥יק לְךָ֖ אֹיְבֶֽךָ׃ וְתֹאכַל אֶת פֵּרוֹת בִּטְנְךָ – בְּשַׂר הַבָּנִים וְהַבָּנוֹת שֶׁלְּךָ שֶׁה' אֱלֹהֶיךָ נָתַן לְךָ – תֹּאכַל אוֹתָם בְּמָצוֹר וּבִמְצוּקָה שֶׁאוֹיִבְךָ יָצִיק לְךָ.
נד הָאִישׁ֙ הָרַ֣ךְ בְּךָ֔ וְהֶעָנֹ֖ג מְאֹ֑ד תֵּרַ֨ע עֵינ֤וֹ בְאָחִיו֙ וּבְאֵ֣שֶׁת חֵיק֔וֹ וּבְיֶ֥תֶר בָּנָ֖יו אֲשֶׁ֥ר יוֹתִֽיר׃ הָאִישׁ הָרַךְ בְּיוֹתֵר שֶׁאֶצְלְךָ וְהַמְּפֻנָּק מְאוֹד, יִתְאַכְזֵר מִלָּתֵת לְאָחִיו וּלְאִשְׁתּוֹ הַיְּשֵׁנָה בְּחֵיקוֹ וְלִשְׁאָר בָּנָיו שֶׁנִּשְׁאֲרוּ,
נה מִתֵּ֣ת ׀ לְאַחַ֣ד מֵהֶ֗ם מִבְּשַׂ֤ר בָּנָיו֙ אֲשֶׁ֣ר יֹאכֵ֔ל מִבְּלִ֥י הִשְׁאִֽיר־ל֖וֹ כֹּ֑ל בְּמָצוֹר֙ וּבְמָצ֔וֹק אֲשֶׁ֨ר יָצִ֥יק לְךָ֛ אֹיִבְךָ֖ בְּכׇל־שְׁעָרֶֽיךָ׃ מִלָּתֵת לְאֶחָד מֵהֶם מִבְּשַׂר בָּנָיו שֶׁיֹּאכַל, כִּי לֹא נִשְׁאַר לוֹ כְּלוּם, בְּמָצוֹר וּבִמְצוּקָה שֶׁאוֹיִבְךָ יָצִיק לְךָ בְּכָל עָרֶיךָ.
נו הָרַכָּ֨ה בְךָ֜ וְהָעֲנֻגָּ֗ה אֲשֶׁ֨ר לֹֽא־נִסְּתָ֤ה כַף־רַגְלָהּ֙ הַצֵּ֣ג עַל־הָאָ֔רֶץ מֵהִתְעַנֵּ֖ג וּמֵרֹ֑ךְ תֵּרַ֤ע עֵינָהּ֙ בְּאִ֣ישׁ חֵיקָ֔הּ וּבִבְנָ֖הּ וּבְבִתָּֽהּ׃ הָרַכָּה בְּיוֹתֵר שֶׁאֶצְלְךָ וְהַמְּפֻנֶּקֶת – שֶׁלֹּא נִסְּתָה לְהָנִיחַ אֶת כַּף רַגְלָהּ יְחֵפָה עַל הָאָרֶץ מֵרוֹב פִּנּוּק וְרַכּוּת, תִּתְקַמְצֵן מִלָּתֵת לְבַעֲלָהּ שֶׁיָּשֵׁן בְּחֵיקָהּ וְלִבְנָהּ וּלְבִתָּהּ,
נז וּֽבְשִׁלְיָתָ֞הּ הַיּוֹצֵ֣ת ׀ מִבֵּ֣ין רַגְלֶ֗יהָ וּבְבָנֶ֙יהָ֙ אֲשֶׁ֣ר תֵּלֵ֔ד כִּֽי־תֹאכְלֵ֥ם בְּחֹסֶר־כֹּ֖ל בַּסָּ֑תֶר בְּמָצוֹר֙ וּבְמָצ֔וֹק אֲשֶׁ֨ר יָצִ֥יק לְךָ֛ אֹיִבְךָ֖ בִּשְׁעָרֶֽיךָ׃ וּלְתִנּוֹקָהּ הַיּוֹצֵא מִבֵּין רַגְלֶיהָ וּלְבָנֶיהָ שֶׁתֵּלֵד, אֶלָּא תֹּאכַל אוֹתָם בַּסֵּתֶר מֵחוֹסֶר כֹּל, בְּמָצוֹר וּבִמְצוּקָה שֶׁאוֹיִבְךָ יָצִיק לְךָ בְּעָרֶיךָ.
נח אִם־לֹ֨א תִשְׁמֹ֜ר לַעֲשׂ֗וֹת אֶת־כׇּל־דִּבְרֵי֙ הַתּוֹרָ֣ה הַזֹּ֔את הַכְּתֻבִ֖ים בַּסֵּ֣פֶר הַזֶּ֑ה לְ֠יִרְאָ֠ה אֶת־הַשֵּׁ֞ם הַנִּכְבָּ֤ד וְהַנּוֹרָא֙ הַזֶּ֔ה אֵ֖ת יְהֹוָ֥ה אֱלֹהֶֽיךָ׃ אִם לֹא תַּקְפִּיד לְקַיֵּם אֶת כָּל דִּבְרֵי הַתּוֹרָה הַזֹּאת הַכְּתוּבִים בַּסֵּפֶר הַזֶּה, לְכַבֵּד אֶת הַשֵּׁם הַנִּכְבָּד וְהַנּוֹרָא הַזֶּה – אֶת שְׁמוֹ שֶׁל ה' אֱלֹהֶיךָ.
נט וְהִפְלָ֤א יְהֹוָה֙ אֶת־מַכֹּ֣תְךָ֔ וְאֵ֖ת מַכּ֣וֹת זַרְעֶ֑ךָ מַכּ֤וֹת גְּדֹלֹת֙ וְנֶ֣אֱמָנ֔וֹת וׇחֳלָיִ֖ם רָעִ֥ים וְנֶאֱמָנִֽים׃ ה' יְחַזֵּק מְאוֹד אֶת הַמַּכּוֹת בְּךָ וּבִילָדֶיךָ – מַכּוֹת גְּדוֹלוֹת וּמְמוּשָּׁכוֹת וּמַחֲלוֹת רָעוֹת וּמְמוּשָּׁכוֹת.
ס וְהֵשִׁ֣יב בְּךָ֗ אֵ֚ת כׇּל־מַדְוֵ֣ה מִצְרַ֔יִם אֲשֶׁ֥ר יָגֹ֖רְתָּ מִפְּנֵיהֶ֑ם וְדָבְק֖וּ בָּֽךְ׃ וְיָבִיא עָלֶיךָ שׁוּב אֶת כָּל מַכּוֹת מִצְרַיִם שֶׁפָּחַדְתָּ מִפְּנֵיהֶם, וְיִדְבְּקוּ בְּךָ.
סא גַּ֤ם כׇּל־חֳלִי֙ וְכׇל־מַכָּ֔ה אֲשֶׁר֙ לֹ֣א כָת֔וּב בְּסֵ֖פֶר הַתּוֹרָ֣ה הַזֹּ֑את יַעְלֵ֤ם יְהֹוָה֙ עָלֶ֔יךָ עַ֖ד הִשָּׁמְדָֽךְ׃ גַּם כָּל מַחֲלָה וְכָל מַכָּה שֶׁלֹּא כְּתוּבִים בְּסֵפֶר הַתּוֹרָה הַזֹּאת, ה' יָבִיא אוֹתָם עָלֶיךָ עַד שֶׁתֻּשְׁמַד.
סב וְנִשְׁאַרְתֶּם֙ בִּמְתֵ֣י מְעָ֔ט תַּ֚חַת אֲשֶׁ֣ר הֱיִיתֶ֔ם כְּכוֹכְבֵ֥י הַשָּׁמַ֖יִם לָרֹ֑ב כִּֽי־לֹ֣א שָׁמַ֔עְתָּ בְּק֖וֹל יְהֹוָ֥ה אֱלֹהֶֽיךָ׃ וְתִשָּׁאֲרוּ אֲנָשִׁים מְעַטִּים, בִּמְקוֹם שֶׁהֲיִיתֶם רַבִּים כְּכוֹכְבֵי הַשָּׁמַיִם, כִּי לֹא שָׁמַעְתָּ בְּקוֹל ה' אֱלֹהֶיךָ.
סג וְ֠הָיָ֠ה כַּאֲשֶׁר־שָׂ֨שׂ יְהֹוָ֜ה עֲלֵיכֶ֗ם לְהֵיטִ֣יב אֶתְכֶם֮ וּלְהַרְבּ֣וֹת אֶתְכֶם֒ כֵּ֣ן יָשִׂ֤ישׂ יְהֹוָה֙ עֲלֵיכֶ֔ם לְהַאֲבִ֥יד אֶתְכֶ֖ם וּלְהַשְׁמִ֣יד אֶתְכֶ֑ם וְנִסַּחְתֶּם֙ מֵעַ֣ל הָאֲדָמָ֔ה אֲשֶׁר־אַתָּ֥ה בָא־שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ כְּמוֹ שֶׁה' שָׂמַח לְהֵיטִיב לָכֶם וּלְהַרְבּוֹת אֶתְכֶם, כָּךְ ה' יְשַׂמֵּחַ אֶת אוֹיְבֵיכֶם לְחַסֵּל אֶתְכֶם וּלְהַשְׁמִיד אֶתְכֶם, וְתֵעָקְרוּ מֵעַל הָאֲדָמָה שֶׁאַתָּה בָּא לְשָׁם לִכְבּוֹשׁ אוֹתָהּ.
סד וֶהֱפִֽיצְךָ֤ יְהֹוָה֙ בְּכׇל־הָ֣עַמִּ֔ים מִקְצֵ֥ה הָאָ֖רֶץ וְעַד־קְצֵ֣ה הָאָ֑רֶץ וְעָבַ֨דְתָּ שָּׁ֜ם אֱלֹהִ֣ים אֲחֵרִ֗ים אֲשֶׁ֧ר לֹא־יָדַ֛עְתָּ אַתָּ֥ה וַאֲבֹתֶ֖יךָ עֵ֥ץ וָאָֽבֶן׃ ה' יְפַזֵּר אוֹתְךָ בְּכָל הָעַמִּים מִקְצֵה הָעוֹלָם וְעַד קְצֵה הָעוֹלָם, וְתַעֲבוֹד שָׁם עוֹבְדֵי אֱלֹהִים אֲחֵרִים עֲשׂוּיִים מֵעֵץ וְאֶבֶן שֶׁלֹּא הִכַּרְתָּ אַתָּה וַאֲבוֹתֶיךָ.
סה וּבַגּוֹיִ֤ם הָהֵם֙ לֹ֣א תַרְגִּ֔יעַ וְלֹא־יִהְיֶ֥ה מָנ֖וֹחַ לְכַף־רַגְלֶ֑ךָ וְנָתַן֩ יְהֹוָ֨ה לְךָ֥ שָׁם֙ לֵ֣ב רַגָּ֔ז וְכִלְי֥וֹן עֵינַ֖יִם וְדַאֲב֥וֹן נָֽפֶשׁ׃ וּבָעַמִּים הָהֵם לֹא יִהְיֶה לְךָ מַרְגּוֹעַ וְלֹא תִּהְיֶה מְנוּחָה לְכַף רַגְלְךָ, ה' יִתֵּן לְךָ שָׁם עַצְבָּנוּת וְצִפִּיָּה וְאַכְזָבָה.
סו וְהָי֣וּ חַיֶּ֔יךָ תְּלֻאִ֥ים לְךָ֖ מִנֶּ֑גֶד וּפָֽחַדְתָּ֙ לַ֣יְלָה וְיוֹמָ֔ם וְלֹ֥א תַאֲמִ֖ין בְּחַיֶּֽיךָ׃ יְמֵי חַיֶּיךָ יִהְיוּ מוּטָּלִים בְּסָפֵק, תִּפְחַד בַּלַּיְלָה וּבַיּוֹם וְלֹא יִהְיֶה לְךָ בִּטָּחוֹן בְּהֶמְשֵׁךְ חַיֶּיךָ.
סז בַּבֹּ֤קֶר תֹּאמַר֙ מִֽי־יִתֵּ֣ן עֶ֔רֶב וּבָעֶ֥רֶב תֹּאמַ֖ר מִֽי־יִתֵּ֣ן בֹּ֑קֶר מִפַּ֤חַד לְבָֽבְךָ֙ אֲשֶׁ֣ר תִּפְחָ֔ד וּמִמַּרְאֵ֥ה עֵינֶ֖יךָ אֲשֶׁ֥ר תִּרְאֶֽה׃ בַּבּוֹקֶר תֹּאמַר הַלְוַאי שֶׁכְּבָר יַגִּיע עֶרֶב וּבָעֶרֶב תֹּאמַר הַלְוַאי שֶׁכְּבָר יַגִּיע בּוֹקֶר, מִפַּחַד לִבְּךָ שֶׁתִּפְחָד וּמִמַּרְאֵה עֵינֶיךָ שֶׁתִּרְאֶה.
סח וֶהֱשִֽׁיבְךָ֨ יְהֹוָ֥ה ׀ מִצְרַ֘יִם֮ בׇּאֳנִיּוֹת֒ בַּדֶּ֙רֶךְ֙ אֲשֶׁ֣ר אָמַ֣רְתִּֽי לְךָ֔ לֹא־תֹסִ֥יף ע֖וֹד לִרְאֹתָ֑הּ וְהִתְמַכַּרְתֶּ֨ם שָׁ֧ם לְאֹיְבֶ֛יךָ לַעֲבָדִ֥ים וְלִשְׁפָח֖וֹת וְאֵ֥ין קֹנֶֽה׃ ה' יַחֲזִיר אוֹתְךָ לְמִצְרַיִם בָּאֳנִיּוֹת, בַּדֶּרֶךְ שֶׁאָמַרְתִּי לְךָ שֶׁלֹּא תִּרְאֶה אוֹתָהּ עוֹד שׁוּב, וְתַצִּיעוּ אֶת עַצְמְכֶם שָׁם לִמְכִירָה לַעֲבָדִים וְלִשְׁפָחוֹת לְאוֹיְבֶיךָ וְאֵין קוֹנֶה.
ס
סט אֵ֩לֶּה֩ דִבְרֵ֨י הַבְּרִ֜ית אֲֽשֶׁר־צִוָּ֧ה יְהֹוָ֣ה אֶת־מֹשֶׁ֗ה לִכְרֹ֛ת אֶת־בְּנֵ֥י יִשְׂרָאֵ֖ל בְּאֶ֣רֶץ מוֹאָ֑ב מִלְּבַ֣ד הַבְּרִ֔ית אֲשֶׁר־כָּרַ֥ת אִתָּ֖ם בְּחֹרֵֽב׃ אֵלֶּה דִּבְרֵי הַבְּרִית שֶׁה' צִוָּה אֶת מֹשֶׁה לִכְרוֹת עִם בְּנֵי יִשְׂרָאֵל בְּאֶרֶץ מוֹאָב, בְּנוֹסָף לַבְּרִית שֶׁכָּרַת אִתָּם בְּחוֹרֵב.

שביעי[עריכה]

פ פָּרָשַׁת הַזְהָרַת הַבְּרִית
כט א וַיִּקְרָ֥א מֹשֶׁ֛ה אֶל־כׇּל־יִשְׂרָאֵ֖ל וַיֹּ֣אמֶר אֲלֵהֶ֑ם אַתֶּ֣ם רְאִיתֶ֗ם אֵ֣ת כׇּל־אֲשֶׁר֩ עָשָׂ֨ה יְהֹוָ֤ה לְעֵֽינֵיכֶם֙ בְּאֶ֣רֶץ מִצְרַ֔יִם לְפַרְעֹ֥ה וּלְכׇל־עֲבָדָ֖יו וּלְכׇל־אַרְצֽוֹ׃ מֹשֶׁה קָרָא לְכָל יִשְׂרָאֵל וְאָמַר לָהֶם: אַתֶּם רְאִיתֶם אֶת כָּל מַה שֶּׁה' עָשָׂה לְעֵינֵיכֶם בְּאֶרֶץ מִצְרַיִם לְפַרְעֹה וּלְכָל עֲבָדָיו וּלְכָל אַרְצוֹ.
ב הַמַּסּוֹת֙ הַגְּדֹלֹ֔ת אֲשֶׁ֥ר רָא֖וּ עֵינֶ֑יךָ הָאֹתֹ֧ת וְהַמֹּפְתִ֛ים הַגְּדֹלִ֖ים הָהֵֽם׃ הַנִּסִּים הַגְּדוֹלִים שֶׁעֵינֶיךָ רָאוּ, הַסִּימָנִים וְהַהוֹכָחוֹת הַגְּדוֹלִים הָהֵם.
ג וְלֹא־נָתַן֩ יְהֹוָ֨ה לָכֶ֥ם לֵב֙ לָדַ֔עַת וְעֵינַ֥יִם לִרְא֖וֹת וְאׇזְנַ֣יִם לִשְׁמֹ֑עַ עַ֖ד הַיּ֥וֹם הַזֶּֽה׃ וַה' לֹא נָתַן לָכֶם לֵב לָדַעַת וְעֵינַיִם לִרְאוֹת וְאָזְנַיִם לִשְׁמוֹעַ לְהַכִּיר בְּגַדְלוּת ה' עַד לַיּוֹם הַזֶּה.
ד וָאוֹלֵ֥ךְ אֶתְכֶ֛ם אַרְבָּעִ֥ים שָׁנָ֖ה בַּמִּדְבָּ֑ר לֹֽא־בָל֤וּ שַׂלְמֹֽתֵיכֶם֙ מֵעֲלֵיכֶ֔ם וְנַעַלְךָ֥ לֹֽא־בָלְתָ֖ה מֵעַ֥ל רַגְלֶֽךָ׃ הוֹלַכְתִּי אֶתְכֶם אַרְבָּעִים שָׁנָה בַּמִּדְבָּר, בִּגְדֵיכֶם שֶׁעֲלֵיכֶם לֹא בָּלוּ וְנַעַלְךָ שֶׁעַל רַגְלְךָ לֹא בָּלְתָה.
ה לֶ֚חֶם לֹ֣א אֲכַלְתֶּ֔ם וְיַ֥יִן וְשֵׁכָ֖ר לֹ֣א שְׁתִיתֶ֑ם לְמַ֙עַן֙ תֵּֽדְע֔וּ כִּ֛י אֲנִ֥י יְהֹוָ֖ה אֱלֹהֵיכֶֽם׃ לֹא אֲכַלְתֶּם לֶחֶם וְלֹא שְׁתִיתֶם יַיִן וּמַשְׁקֶה מְשַׁכֵּר, כְּדֵי שֶׁתֵּדְעוּ שֶׁאֲנִי ה' אֱלֹהֵיכֶם.

מפטיר[עריכה]

ו וַתָּבֹ֖אוּ אֶל־הַמָּק֣וֹם הַזֶּ֑ה וַיֵּצֵ֣א סִיחֹ֣ן מֶֽלֶךְ־חֶ֠שְׁבּ֠וֹן וְע֨וֹג מֶלֶךְ־הַבָּשָׁ֧ן לִקְרָאתֵ֛נוּ לַמִּלְחָמָ֖ה וַנַּכֵּֽם׃ בָּאתֶם לַמָּקוֹם הַזֶּה, וְסִיחוֹן מֶלֶךְ חֶשְׁבּוֹן וְעוֹג מֶלֶךְ הַבָּשָׁן יָצְאוּ לִקְרָאתֵנוּ לַמִּלְחָמָה וְהִכִּינוּ אוֹתָם.
ז וַנִּקַּח֙ אֶת־אַרְצָ֔ם וַנִּתְּנָ֣הּ לְנַחֲלָ֔ה לָראוּבֵנִ֖י וְלַגָּדִ֑י וְלַחֲצִ֖י שֵׁ֥בֶט הַֽמְנַשִּֽׁי׃ לָקַחְנוּ אֶת אַרְצָם, וְנָתַנּוּ אוֹתָהּ לְנַחֲלָה לָרְאוּבֵנִי וְלַגָּדִי וְלַחֲצִי שֵׁבֶט מְנַשֶּׁה.
ח וּשְׁמַרְתֶּ֗ם אֶת־דִּבְרֵי֙ הַבְּרִ֣ית הַזֹּ֔את וַעֲשִׂיתֶ֖ם אֹתָ֑ם לְמַ֣עַן תַּשְׂכִּ֔ילוּ אֵ֖ת כׇּל־אֲשֶׁ֥ר תַּעֲשֽׂוּן׃ הַקְפִּידוּ עַל דִּבְרֵי הַבְּרִית הַזֹּאת וַעֲשׂוּ אוֹתָם, כְּדֵי שֶׁתַּצְלִיחוּ בְּכָל מַה שֶּׁתַּעֲשׂוּ.
