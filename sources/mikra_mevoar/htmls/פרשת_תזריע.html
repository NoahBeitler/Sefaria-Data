ראשון[עריכה]

פ פָּרָשַׁת הַיּוֹלֶדֶת
יב א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב דַּבֵּ֞ר אֶל־בְּנֵ֤י יִשְׂרָאֵל֙ לֵאמֹ֔ר אִשָּׁה֙ כִּ֣י תַזְרִ֔יעַ וְיָלְדָ֖ה זָכָ֑ר וְטָֽמְאָה֙ שִׁבְעַ֣ת יָמִ֔ים כִּימֵ֛י נִדַּ֥ת דְּוֺתָ֖הּ תִּטְמָֽא׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וֶאֱמוֹר: כְּשֶׁאִשָּׁה תִּהְיֶה בְּהֵרָיוֹן וְתֵלֵד זָכָר הִיא תִּהְיֶה טְמֵאָה שִׁבְעָה יָמִים כְּמוֹ בִּימֵי טֻמְאַת הַנִּדָּה.
ג וּבַיּ֖וֹם הַשְּׁמִינִ֑י יִמּ֖וֹל בְּשַׂ֥ר עׇרְלָתֽוֹ׃ וּבַיּוֹם הַשְּׁמִינִי יָמוּלוּ אֶת בְּשַׂר עָרְלַת הַבֵּן.
ד וּשְׁלֹשִׁ֥ים יוֹם֙ וּשְׁלֹ֣שֶׁת יָמִ֔ים תֵּשֵׁ֖ב בִּדְמֵ֣י טׇהֳרָ֑הֿ בְּכׇל־קֹ֣דֶשׁ לֹֽא־תִגָּ֗ע וְאֶל־הַמִּקְדָּשׁ֙ לֹ֣א תָבֹ֔א עַד־מְלֹ֖את יְמֵ֥י טׇהֳרָֽהּ׃ שְׁלוֹשִׁים וּשְׁלוֹשָׁה יָמִים דַּם הַנִּדָּה שֶׁלָּהּ יִהְיֶה טָהוֹר, אָסוּר לָהּ לַגַּעַת בְּשׁוּם קוֹדֶשׁ וְלֹא לְהִכָּנֵס לַמִּקְדָּשׁ עַד שֶׁיִּגָּמְרוּ יְמֵי הַטַּהֲרָה.
ה וְאִם־נְקֵבָ֣ה תֵלֵ֔ד וְטָמְאָ֥ה שְׁבֻעַ֖יִם כְּנִדָּתָ֑הּ וְשִׁשִּׁ֥ים יוֹם֙ וְשֵׁ֣שֶׁת יָמִ֔ים תֵּשֵׁ֖ב עַל־דְּמֵ֥י טׇהֳרָֽהֿ׃ וְאִם הִיא תֵּלֵד נְקֵבָה, שְׁבוּעַיִם הִיא תִּהְיֶה טְמֵאָה כְּמוֹ בְּנִדָּתָהּ, וְשִׁשִּׁים וְשִׁשָּׁה יָמִים דַּם הַנִּדָּה שֶׁלָּהּ יִהְיֶה טָהוֹר.
ו וּבִמְלֹ֣את ׀ יְמֵ֣י טׇהֳרָ֗הּ לְבֵן֮ א֣וֹ לְבַת֒ תָּבִ֞יא כֶּ֤בֶשׂ בֶּן־שְׁנָתוֹ֙ לְעֹלָ֔ה וּבֶן־יוֹנָ֥ה אוֹ־תֹ֖ר לְחַטָּ֑את אֶל־פֶּ֥תַח אֹֽהֶל־מוֹעֵ֖ד אֶל־הַכֹּהֵֽן׃ וּכְשֶׁיִּגָּמְרוּ יְמֵי הַטַּהֲרָה שֶׁל לֵידַת בֵּן אוֹ בַּת, הִיא תָּבִיא לַכֹּהֵן אֶל מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת, כֶּבֶשׂ שֶׁגִּילוֹ עַד שָׁנָה לְעוֹלָה, וּבֶן יוֹנָה אוֹ תּוֹר לְחַטָּאת.
ז וְהִקְרִיב֞וֹ לִפְנֵ֤י יְהֹוָה֙ וְכִפֶּ֣ר עָלֶ֔יהָ וְטָהֲרָ֖הֿ מִמְּקֹ֣ר דָּמֶ֑יהָ זֹ֤את תּוֹרַת֙ הַיֹּלֶ֔דֶת לַזָּכָ֖ר א֥וֹ לַנְּקֵבָֽה׃ הַכֹּהֵן יַגִּישׁ אוֹתוֹ אֶל לִפְנֵי ה' וִיכַפֵּר עָלֶיהָ, וְהִיא תִּהְיֶה טְהוֹרָה מֵהַטֻּמְאָה שֶׁבָּאָה מִמְּקוֹר הַדָּם שֶׁלָּהּ, אֵלֶּה דִּינֵי יוֹלֶדֶת זָכָר אוֹ נְקֵבָה.
ח וְאִם־לֹ֨א תִמְצָ֣א יָדָהּ֮ דֵּ֣י שֶׂה֒ וְלָקְחָ֣ה שְׁתֵּֽי־תֹרִ֗ים א֤וֹ שְׁנֵי֙ בְּנֵ֣י יוֹנָ֔ה אֶחָ֥ד לְעֹלָ֖ה וְאֶחָ֣ד לְחַטָּ֑את וְכִפֶּ֥ר עָלֶ֛יהָ הַכֹּהֵ֖ן וְטָהֵֽרָה׃ וְאִם לֹא יִהְיֶה לָהּ מַסְפִּיק כֶּסֶף לְטָלֶה, הִיא תִּקַּח שְׁתֵּי תּוֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה אֶחָד לְעוֹלָה וְאֶחָד לְחַטָּאת, וְהַכֹּהֵן יְכַפֵּר עָלֶיהָ וְהִיא תִּהְיֶה טְהוֹרָה.
פ פָּרָשַׁת צָרַעַת הָעוֹר
יג א וַיְדַבֵּ֣ר יְהֹוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאֶל אַהֲרֹן וְאָמַר:
ב אָדָ֗ם כִּֽי־יִהְיֶ֤ה בְעוֹר־בְּשָׂרוֹ֙ שְׂאֵ֤ת אֽוֹ־סַפַּ֙חַת֙ א֣וֹ בַהֶ֔רֶת וְהָיָ֥ה בְעוֹר־בְּשָׂר֖וֹ לְנֶ֣גַע צָרָ֑עַת וְהוּבָא֙ אֶל־אַהֲרֹ֣ן הַכֹּהֵ֔ן א֛וֹ אֶל־אַחַ֥ד מִבָּנָ֖יו הַכֹּהֲנִֽים׃ אָדָם שֶׁיִּהְיֶה בְּעוֹרוֹ שׁוּמָה אוֹ גִּדּוּל אוֹ כֶּתֶם בָּהִיר, זֶה נֶגַע צָרַעַת בְּעוֹרוֹ, וְצָרִיךְ לַהֲבִיאוֹ אֶל אַהֲרֹן הַכֹּהֵן אוֹ אֶל אֶחָד מִבָּנָיו הַכֹּהֲנִים.
ג וְרָאָ֣ה הַכֹּהֵ֣ן אֶת־הַנֶּ֣גַע בְּעֽוֹר־הַ֠בָּשָׂ֠ר וְשֵׂעָ֨ר בַּנֶּ֜גַע הָפַ֣ךְ ׀ לָבָ֗ן וּמַרְאֵ֤ה הַנֶּ֙גַע֙ עָמֹק֙ מֵע֣וֹר בְּשָׂר֔וֹ נֶ֥גַע צָרַ֖עַת ה֑וּא וְרָאָ֥הוּ הַכֹּהֵ֖ן וְטִמֵּ֥א אֹתֽוֹ׃ הַכֹּהֵן יִרְאֶה אֶת הַנֶּגַע שֶׁבְּעוֹר הַבָּשָׂר וְאִם הַשֵּׂעָר שֶׁבַּנֶּגַע הָפַךְ לְלָבָן וְצֶבַע הַנֶּגַע עָמוֹק מֵהָעוֹר – זֶהוּ נֶגַע צָרַעַת, הַכֹּהֵן יִרְאֶה אוֹתוֹ וִיטַמֵּא אוֹתוֹ.
ד וְאִם־בַּהֶ֩רֶת֩ לְבָנָ֨ה הִ֜וא בְּע֣וֹר בְּשָׂר֗וֹ וְעָמֹק֙ אֵין־מַרְאֶ֣הָ מִן־הָע֔וֹר וּשְׂעָרָ֖הֿ לֹא־הָפַ֣ךְ לָבָ֑ן וְהִסְגִּ֧יר הַכֹּהֵ֛ן אֶת־הַנֶּ֖גַע שִׁבְעַ֥ת יָמִֽים׃ וְאִם הַכֶּתֶם בְּעוֹרוֹ לָבָן בָּהִיר אֲבָל צִבְעוֹ אֵינוֹ עָמוֹק מֵהָעוֹר וְהַשֵּׂעָר שֶׁבּוֹ לֹא הָפַךְ לְלָבָן, הַכֹּהֵן יַסְגִּיר אֶת הַנֶּגַע שִׁבְעָה יָמִים.
ה וְרָאָ֣הוּ הַכֹּהֵן֮ בַּיּ֣וֹם הַשְּׁבִיעִי֒ וְהִנֵּ֤ה הַנֶּ֙גַע֙ עָמַ֣ד בְּעֵינָ֔יו לֹֽא־פָשָׂ֥ה הַנֶּ֖גַע בָּע֑וֹר וְהִסְגִּיר֧וֹ הַכֹּהֵ֛ן שִׁבְעַ֥ת יָמִ֖ים שֵׁנִֽית׃ הַכֹּהֵן יִרְאֶה אוֹתוֹ בַּיּוֹם הַשְּׁבִיעִי, וְהִנֵּה אִם מַרְאֶה הַנֶּגַע לֹא הִשְׁתַּנָּה – הַנֶּגַע לֹא הִתְפַּשֵּׁט בָּעוֹר, הַכֹּהֵן יַסְגִּיר אוֹתוֹ שִׁבְעָה יָמִים שֵׁנִית.

שני[עריכה]

ו וְרָאָה֩ הַכֹּהֵ֨ן אֹת֜וֹ בַּיּ֣וֹם הַשְּׁבִיעִי֮ שֵׁנִית֒ וְהִנֵּה֙ כֵּהָ֣ה הַנֶּ֔גַע וְלֹא־פָשָׂ֥ה הַנֶּ֖גַע בָּע֑וֹר וְטִהֲר֤וֹ הַכֹּהֵן֙ מִסְפַּ֣חַת הִ֔וא וְכִבֶּ֥ס בְּגָדָ֖יו וְטָהֵֽר׃ הַכֹּהֵן יִרְאֶה אוֹתוֹ שֵׁנִית בַּיּוֹם הַשְּׁבִיעִי, וְהִנֵּה אִם הַנֶּגַע נֶחְלַשׁ וְהַנֶּגַע לֹא הִתְפַּשֵּׁט בָּעוֹר, הַכֹּהֵן יְטַהֵר אוֹתוֹ – זֶהוּ גִּדּוּל חָרִיג, הוּא יַטְבִּיל אֶת בְּגָדָיו וְיִהְיֶה טָהוֹר.
ז וְאִם־פָּשֹׂ֨ה תִפְשֶׂ֤ה הַמִּסְפַּ֙חַת֙ בָּע֔וֹר אַחֲרֵ֧י הֵרָאֹת֛וֹ אֶל־הַכֹּהֵ֖ן לְטׇהֳרָת֑וֹ וְנִרְאָ֥ה שֵׁנִ֖ית אֶל־הַכֹּהֵֽן׃ וְאִם הַגִּדּוּל יִתְפַּשֵּׁט בָּעוֹר אַחֲרֵי שֶׁהֶרְאוּ אוֹתוֹ לַכֹּהֵן וְהוּא טִהֵר אוֹתוֹ, יַרְאוּ אוֹתוֹ שֵׁנִית לַכֹּהֵן.
ח וְרָאָה֙ הַכֹּהֵ֔ן וְהִנֵּ֛ה פָּשְׂתָ֥ה הַמִּסְפַּ֖חַת בָּע֑וֹר וְטִמְּא֥וֹ הַכֹּהֵ֖ן צָרַ֥עַת הִֽוא׃ הַכֹּהֵן יִרְאֶה וְהִנֵּה הַגִּדּוּל הִתְפַּשֵּׁט בָּעוֹר, הַכֹּהֵן יְטַמֵּא אוֹתוֹ – זוֹ צָרַעַת.
פ פָּרָשַׁת הַמִּחְיָה
ט נֶ֣גַע צָרַ֔עַת כִּ֥י תִהְיֶ֖ה בְּאָדָ֑ם וְהוּבָ֖א אֶל־הַכֹּהֵֽן׃ אִם יִהְיֶה נֶגַע צָרַעַת בְּאָדָם, הוּא יוּבָא לַכֹּהֵן.
י וְרָאָ֣ה הַכֹּהֵ֗ן וְהִנֵּ֤ה שְׂאֵת־לְבָנָה֙ בָּע֔וֹר וְהִ֕יא הָפְכָ֖ה שֵׂעָ֣ר לָבָ֑ן וּמִֽחְיַ֛ת בָּשָׂ֥ר חַ֖י בַּשְׂאֵֽת׃ הַכֹּהֵן יִרְאֶה וְהִנֵּה יֵשׁ שׁוּמָה לְבָנָה בָּעוֹר וְהִיא הָפְכָה אֶת הַשֵּׂעָר לְלָבָן וְיֵשׁ בַּשּׁוּמָה בָּשָׂר חַי שֶׁמַּבְרִיא.
יא צָרַ֨עַת נוֹשֶׁ֤נֶת הִוא֙ בְּע֣וֹר בְּשָׂר֔וֹ וְטִמְּא֖וֹ הַכֹּהֵ֑ן לֹ֣א יַסְגִּרֶ֔נּוּ כִּ֥י טָמֵ֖א הֽוּא׃ זוֹ צָרַעַת יְשָׁנָה בְּעוֹרוֹ – הַכֹּהֵן יְטַמֵּא אוֹתוֹ, הוּא לֹא יַסְגִּיר אוֹתוֹ כִּי הוּא טָמֵא.
יב וְאִם־פָּר֨וֹחַ תִּפְרַ֤ח הַצָּרַ֙עַת֙ בָּע֔וֹר וְכִסְּתָ֣ה הַצָּרַ֗עַת אֵ֚ת כׇּל־ע֣וֹר הַנֶּ֔גַע מֵרֹאשׁ֖וֹ וְעַד־רַגְלָ֑יו לְכׇל־מַרְאֵ֖ה עֵינֵ֥י הַכֹּהֵֽן׃ וְאִם הַצָּרַעַת תִּפְרַח וְתִתְפַּשֵּׁט בָּעוֹר וְהַצָּרַעַת תְּכַסֶּה אֶת כָּל עוֹר הַגּוּף שֶׁבּוֹ הַנֶּגַע – מֵרֹאשׁוֹ וְעַד רַגְלָיו, בְּכָל מָקוֹם שֶׁעֵינֵי הַכֹּהֵן רוֹאוֹת,
יג וְרָאָ֣ה הַכֹּהֵ֗ן וְהִנֵּ֨ה כִסְּתָ֤ה הַצָּרַ֙עַת֙ אֶת־כׇּל־בְּשָׂר֔וֹ וְטִהַ֖ר אֶת־הַנָּ֑גַע כֻּלּ֛וֹ הָפַ֥ךְ לָבָ֖ן טָה֥וֹר הֽוּא׃ הַכֹּהֵן יִרְאֶה שֶׁהַצָּרַעַת כִּסְּתָה אֶת כָּל גּוּפוֹ וִיטַהֵר אֶת הַנֶּגַע, אִם כֻּלּוֹ הָפַךְ לָבָן – הוּא טָהוֹר.
יד וּבְי֨וֹם הֵרָא֥וֹת בּ֛וֹ בָּשָׂ֥ר חַ֖י יִטְמָֽא׃ וּבַיּוֹם שֶׁיֵּרָאֶה בּוֹ בָּשָׂר חַי – הוּא יִהְיֶה טָמֵא.
טו וְרָאָ֧ה הַכֹּהֵ֛ן אֶת־הַבָּשָׂ֥ר הַחַ֖י וְטִמְּא֑וֹ הַבָּשָׂ֥ר הַחַ֛י טָמֵ֥א ה֖וּא צָרַ֥עַת הֽוּא׃ הַכֹּהֵן יִרְאֶה אֶת הַבָּשָׂר הַחַי וִיטַמֵּא אוֹתוֹ, הַבָּשָׂר הַחַי הוּא טָמֵא, הוּא צָרַעַת.
טז א֣וֹ כִ֥י יָשׁ֛וּב הַבָּשָׂ֥ר הַחַ֖י וְנֶהְפַּ֣ךְ לְלָבָ֑ן וּבָ֖א אֶל־הַכֹּהֵֽן׃ וְאִם הַבָּשָׂר הַחַי יַחֲזוֹר וְיֵהָפֵךְ לְלָבָן, הוּא יָבוֹא לַכֹּהֵן.
יז וְרָאָ֙הוּ֙ הַכֹּהֵ֔ן וְהִנֵּ֛ה נֶהְפַּ֥ךְ הַנֶּ֖גַע לְלָבָ֑ן וְטִהַ֧ר הַכֹּהֵ֛ן אֶת־הַנֶּ֖גַע טָה֥וֹר הֽוּא׃ הַכֹּהֵן יִרְאֶה אוֹתוֹ וְהִנֵּה הַנֶּגַע נֶהְפַּךְ לְלָבָן, הַכֹּהֵן יְטַהֵר אֶת הַנֶּגַע – הוּא טָהוֹר.

שלישי[עריכה]

פ פָּרָשַׁת צָרַעַת הַשְּׁחִין וְהַמִּכְוָה
יח וּבָשָׂ֕ר כִּֽי־יִהְיֶ֥ה בֽוֹ־בְעֹר֖וֹ שְׁחִ֑ין וְנִרְפָּֽא׃ אִם יִהְיֶה בְּעוֹר הַבָּשָׂר שְׁחִין, וְהוּא יִתְרַפֵּא,
יט וְהָיָ֞ה בִּמְק֤וֹם הַשְּׁחִין֙ שְׂאֵ֣ת לְבָנָ֔ה א֥וֹ בַהֶ֖רֶת לְבָנָ֣ה אֲדַמְדָּ֑מֶת וְנִרְאָ֖ה אֶל־הַכֹּהֵֽן׃ וְתִהְיֶה בִּמְקוֹם הַשְּׁחִין שׁוּמָה לְבָנָה אוֹ כֶּתֶם בָּהִיר לָבָן מְעוֹרָב בְּאָדוֹם – יַרְאוּ אוֹתוֹ לַכֹּהֵן.
כ וְרָאָ֣ה הַכֹּהֵ֗ן וְהִנֵּ֤ה מַרְאֶ֙הָ֙ שָׁפָ֣ל מִן־הָע֔וֹר וּשְׂעָרָ֖הּ הָפַ֣ךְ לָבָ֑ן וְטִמְּא֧וֹ הַכֹּהֵ֛ן נֶֽגַע־צָרַ֥עַת הִ֖וא בַּשְּׁחִ֥ין פָּרָֽחָה׃ הַכֹּהֵן יִרְאֶה שֶׁצִּבְעוֹ עָמוֹק מֵהָעוֹר וְהַשֵּׂעָר שֶׁבּוֹ הָפַךְ לְלָבָן, הַכֹּהֵן יְטַמֵּא אוֹתוֹ – זֶהוּ נֶגַע צָרַעַת שֶׁפָּרַח בַּשְּׁחִין.
כא וְאִ֣ם ׀ יִרְאֶ֣נָּה הַכֹּהֵ֗ן וְהִנֵּ֤ה אֵֽין־בָּהּ֙ שֵׂעָ֣ר לָבָ֔ן וּשְׁפָלָ֥הֿ אֵינֶ֛נָּה מִן־הָע֖וֹר וְהִ֣יא כֵהָ֑ה וְהִסְגִּיר֥וֹ הַכֹּהֵ֖ן שִׁבְעַ֥ת יָמִֽים׃ וְאִם הַכֹּהֵן יִרְאֶה אוֹתוֹ וְהִנֵּה אֵין בּוֹ שֵׂעָר לָבָן וְצִבְעוֹ אֵינוֹ עָמוֹק מֵהָעוֹר וְהוּא דָּהוּי, הַכֹּהֵן יַסְגִּיר אוֹתוֹ שִׁבְעָה יָמִים.
כב וְאִם־פָּשֹׂ֥ה תִפְשֶׂ֖ה בָּע֑וֹר וְטִמֵּ֧א הַכֹּהֵ֛ן אֹת֖וֹ נֶ֥גַע הִֽוא׃ וְאִם הוּא יִתְפַּשֵּׁט בָּעוֹר, הַכֹּהֵן יְטַמֵּא אוֹתוֹ, זֶהוּ נֶגַע.
כג וְאִם־תַּחְתֶּ֜יהָ תַּעֲמֹ֤ד הַבַּהֶ֙רֶת֙ לֹ֣א פָשָׂ֔תָה צָרֶ֥בֶת הַשְּׁחִ֖ין הִ֑וא וְטִהֲר֖וֹ הַכֹּהֵֽן׃ וְאִם הַכֶּתֶם יַעֲמוֹד בִּמְקוֹמוֹ וְלֹא יִתְפַּשֵּׁט, זוֹ רַק צְרִיבָה מֵהַשְּׁחִין, הַכֹּהֵן יְטַהֵר אוֹתוֹ.

רביעי (שני במחוברין)[עריכה]

ס
כד א֣וֹ בָשָׂ֔ר כִּֽי־יִהְיֶ֥ה בְעֹר֖וֹ מִכְוַת־אֵ֑שׁ וְֽהָיְתָ֞ה מִֽחְיַ֣ת הַמִּכְוָ֗ה בַּהֶ֛רֶת לְבָנָ֥ה אֲדַמְדֶּ֖מֶת א֥וֹ לְבָנָֽה׃ אוֹ אִם יִהְיֶה בְּעוֹר הַבָּשָׂר כְּוִיָּה מֵאֵשׁ, וְיִהְיֶה בַּמָּקוֹם שֶׁל הַכְּוִיָּה שֶׁמִּתְרַפֵּאת כֶּתֶם בָּהִיר לָבָן מְעוֹרָב בְּאָדוֹם אוֹ רַק לָבָן.
כה וְרָאָ֣ה אֹתָ֣הּ הַכֹּהֵ֡ן וְהִנֵּ֣ה נֶהְפַּךְ֩ שֵׂעָ֨ר לָבָ֜ן בַּבַּהֶ֗רֶת וּמַרְאֶ֙הָ֙ עָמֹ֣ק מִן־הָע֔וֹר צָרַ֣עַת הִ֔וא בַּמִּכְוָ֖ה פָּרָ֑חָה וְטִמֵּ֤א אֹתוֹ֙ הַכֹּהֵ֔ן נֶ֥גַע צָרַ֖עַת הִֽוא׃ הַכֹּהֵן יִרְאֶה אוֹתוֹ וְהִנֵּה הַשֵּׂעָר שֶׁבַּכֶּתֶם נֶהְפַּךְ לְלָבָן וְצִבְעוֹ עָמוֹק מֵהָעוֹר – זוֹ צָרַעַת שֶׁפָּרְחָה בַּכְּוִיָּה, הַכֹּהֵן יְטַמֵּא אוֹתוֹ, זֶה נֶגַע צָרַעַת.
כו וְאִ֣ם ׀ יִרְאֶ֣נָּה הַכֹּהֵ֗ן וְהִנֵּ֤ה אֵֽין־בַּבַּהֶ֙רֶת֙ שֵׂעָ֣ר לָבָ֔ן וּשְׁפָלָ֥הֿ אֵינֶ֛נָּה מִן־הָע֖וֹר וְהִ֣וא כֵהָ֑ה וְהִסְגִּיר֥וֹ הַכֹּהֵ֖ן שִׁבְעַ֥ת יָמִֽים׃ וְאִם הַכֹּהֵן יִרְאֶה אוֹתוֹ וְהִנֵּה אֵין בַּכֶּתֶם שֵׂעָר לָבָן וְצִבְעוֹ אֵינוֹ עָמוֹק מֵהָעוֹר וְהוּא דָּהוּי, הַכֹּהֵן יַסְגִּיר אוֹתוֹ שִׁבְעָה יָמִים.
כז וְרָאָ֥הוּ הַכֹּהֵ֖ן בַּיּ֣וֹם הַשְּׁבִיעִ֑י אִם־פָּשֹׂ֤ה תִפְשֶׂה֙ בָּע֔וֹר וְטִמֵּ֤א הַכֹּהֵן֙ אֹת֔וֹ נֶ֥גַע צָרַ֖עַת הִֽוא׃ הַכֹּהֵן יִרְאֶה אוֹתוֹ בַּיּוֹם הַשְּׁבִיעִי, אִם הוּא יִתְפַּשֵּׁט בָּעוֹר – הַכֹּהֵן יְטַמֵּא אוֹתוֹ, זֶה נֶגַע צָרַעַת.
כח וְאִם־תַּחְתֶּ֩יהָ֩ תַעֲמֹ֨ד הַבַּהֶ֜רֶת לֹא־פָשְׂתָ֤ה בָעוֹר֙ וְהִ֣וא כֵהָ֔ה שְׂאֵ֥ת הַמִּכְוָ֖ה הִ֑וא וְטִֽהֲרוֹ֙ הַכֹּהֵ֔ן כִּֽי־צָרֶ֥בֶת הַמִּכְוָ֖ה הִֽוא׃ וְאִם הַכֶּתֶם יַעֲמוֹד בִּמְקוֹמוֹ וְלֹא יִתְפַּשֵּׁט בָּעוֹר, וְהוּא דָּהוּי, זֶה שׁוּמָה שֶׁל כְּוִיָּה, הַכֹּהֵן יְטַהֵר אוֹתוֹ כִּי זֶה צְרִיבָה שֶׁל כְּוִיָּה.

חמישי[עריכה]

פ פָּרָשַׁת צָרַעַת הַשֵּׂעַר
כט וְאִישׁ֙ א֣וֹ אִשָּׁ֔ה כִּֽי־יִהְיֶ֥ה ב֖וֹ נָ֑גַע בְּרֹ֖אשׁ א֥וֹ בְזָקָֽן׃ אִם יִהְיֶה בְּאִישׁ אוֹ אִשָּׁה נֶגַע, בָּרֹאשׁ אוֹ בַּזָּקָן.
ל וְרָאָ֨ה הַכֹּהֵ֜ן אֶת־הַנֶּ֗גַע וְהִנֵּ֤ה מַרְאֵ֙הוּ֙ עָמֹ֣ק מִן־הָע֔וֹר וּב֛וֹ שֵׂעָ֥ר צָהֹ֖ב דָּ֑ק וְטִמֵּ֨א אֹת֤וֹ הַכֹּהֵן֙ נֶ֣תֶק ה֔וּא צָרַ֧עַת הָרֹ֛אשׁ א֥וֹ הַזָּקָ֖ן הֽוּא׃ הַכֹּהֵן יִרְאֶה אֶת הַנֶּגַע וְהִנֵּה צִבְעוֹ עָמוֹק מֵהָעוֹר וְיֵשׁ בּוֹ שֵׂעָר צָהוֹב דָּק, הַכֹּהֵן יְטַמֵּא אוֹתוֹ – זֶה נֶגַע שֶׁל נִתּוּק שֵׂעָר, זֶה צָרַעַת הָרֹאשׁ אוֹ הַזָּקָן.
לא וְכִֽי־יִרְאֶ֨ה הַכֹּהֵ֜ן אֶת־נֶ֣גַע הַנֶּ֗תֶק וְהִנֵּ֤ה אֵין־מַרְאֵ֙הוּ֙ עָמֹ֣ק מִן־הָע֔וֹר וְשֵׂעָ֥ר שָׁחֹ֖ר אֵ֣ין בּ֑וֹ וְהִסְגִּ֧יר הַכֹּהֵ֛ן אֶת־נֶ֥גַע הַנֶּ֖תֶק שִׁבְעַ֥ת יָמִֽים׃ וְאִם הַכֹּהֵן יִרְאֶה אֶת נֶגַע נִתּוּק הַשֵּׂעָר, וְהִנֵּה צִבְעוֹ אֵינוֹ עָמוֹק מֵהָעוֹר וְאֵין בּוֹ שֵׂעָר שָׁחוֹר, הַכֹּהֵן יַסְגִּיר אֶת נֶגַע נִתּוּק הַשֵּׂעָר שִׁבְעָה יָמִים.
לב וְרָאָ֨ה הַכֹּהֵ֣ן אֶת־הַנֶּ֘גַע֮ בַּיּ֣וֹם הַשְּׁבִיעִי֒ וְהִנֵּה֙ לֹא־פָשָׂ֣ה הַנֶּ֔תֶק וְלֹא־הָ֥יָה ב֖וֹ שֵׂעָ֣ר צָהֹ֑ב וּמַרְאֵ֣ה הַנֶּ֔תֶק אֵ֥ין עָמֹ֖ק מִן־הָעֽוֹר׃ הַכֹּהֵן יִרְאֶה אֶת הַנֶּגַע בַּיּוֹם הַשְּׁבִיעִי, וְהִנֵּה נֶגַע נִתּוּק הַשֵּׂעָר לֹא הִתְפַּשֵּׁט וְאֵין בּוֹ שֵׂעָר צָהוֹב וְצִבְעוֹ שֶׁל נֶגַע נִתּוּק הַשֵּׂעָר אֵינוֹ עָמוֹק מֵהָעוֹר.
לג וְהִ֨תְגַּלָּ֔ח וְאֶת־הַנֶּ֖תֶק לֹ֣א יְגַלֵּ֑חַ וְהִסְגִּ֨יר הַכֹּהֵ֧ן אֶת־הַנֶּ֛תֶק שִׁבְעַ֥ת יָמִ֖ים שֵׁנִֽית׃ הוּא יְגַלֵּחַ מִסָּבִיב לַנֶּגַע, אֲבָל אֶת הַשֵּׂעָר שֶׁצָּמוּד לְנֶגַע נִתּוּק הַשֵּׂעָר לֹא יְגַלֵּחַ, הַכֹּהֵן יַסְגִּיר שֵׁנִית שִׁבְעָה יָמִים אֶת נֶגַע נִתּוּק הַשֵּׂעָר.
לד וְרָאָה֩ הַכֹּהֵ֨ן אֶת־הַנֶּ֜תֶק בַּיּ֣וֹם הַשְּׁבִיעִ֗י וְ֠הִנֵּ֠ה לֹא־פָשָׂ֤ה הַנֶּ֙תֶק֙ בָּע֔וֹר וּמַרְאֵ֕הוּ אֵינֶ֥נּוּ עָמֹ֖ק מִן־הָע֑וֹר וְטִהַ֤ר אֹתוֹ֙ הַכֹּהֵ֔ן וְכִבֶּ֥ס בְּגָדָ֖יו וְטָהֵֽר׃ הַכֹּהֵן יִרְאֶה אֶת נֶגַע נִתּוּק הַשֵּׂעָר בַּיּוֹם הַשְּׁבִיעִי, וְהִנֵּה נֶגַע נִתּוּק הַשֵּׂעָר לֹא הִתְפַּשֵּׁט בָּעוֹר, וְצִבְעוֹ אֵינוֹ עָמוֹק מֵהָעוֹר, הַכֹּהֵן יְטַהֵר אוֹתוֹ, וְהוּא יַטְבִּיל אֶת בְּגָדָיו וְיִהְיֶה טָהוֹר.
לה וְאִם־פָּשֹׂ֥ה יִפְשֶׂ֛ה הַנֶּ֖תֶק בָּע֑וֹר אַחֲרֵ֖י טׇהֳרָתֽוֹ׃ וְאִם מַרְאֵה נֶגַע נִתּוּק הַשֵּׂעָר יִתְפַּשֵּׁט בָּעוֹר אַחֲרֵי שֶׁנִּטְהַר.
לו וְרָאָ֙הוּ֙ הַכֹּהֵ֔ן וְהִנֵּ֛ה פָּשָׂ֥ה הַנֶּ֖תֶק בָּע֑וֹר לֹֽא־יְבַקֵּ֧ר הַכֹּהֵ֛ן לַשֵּׂעָ֥ר הַצָּהֹ֖ב טָמֵ֥א הֽוּא׃ הַכֹּהֵן יִרְאֶה אוֹתוֹ וְהִנֵּה נֶגַע נִתּוּק הַשֵּׂעָר הִתְפַּשֵּׁט בָּעוֹר, הַכֹּהֵן לֹא יְחַפֵּשׂ גַּם שֵׂעָר צָהוֹב – זֶה טָמֵא.
לז וְאִם־בְּעֵינָיו֩ עָמַ֨ד הַנֶּ֜תֶק וְשֵׂעָ֨ר שָׁחֹ֧ר צָֽמַח־בּ֛וֹ נִרְפָּ֥א הַנֶּ֖תֶק טָה֣וֹר ה֑וּא וְטִהֲר֖וֹ הַכֹּהֵֽן׃ וְאִם מַרְאֵה נֶגַע נִתּוּק הַשֵּׂעָר לֹא הִשְׁתַּנָּה וְצָמַח בּוֹ שֵׂעָר שָׁחוֹר – נֶגַע נִתּוּק הַשֵּׂעָר נִרְפָּא, הוּא טָהוֹר, הַכֹּהֵן יְטַהֵר אוֹתוֹ.
פ פָּרָשַׁת הַבֶּהָרוֹת
לח וְאִישׁ֙ אֽוֹ־אִשָּׁ֔ה כִּֽי־יִהְיֶ֥ה בְעוֹר־בְּשָׂרָ֖ם בֶּהָרֹ֑ת בֶּהָרֹ֖ת לְבָנֹֽת׃ וְאִישׁ אוֹ אִשָּׁה אִם יִהְיוּ בְּעוֹרָם כְּתָמִים בְּהִירִים לְבָנִים.
לט וְרָאָ֣ה הַכֹּהֵ֗ן וְהִנֵּ֧ה בְעוֹר־בְּשָׂרָ֛ם בֶּהָרֹ֖ת כֵּה֣וֹת לְבָנֹ֑ת בֹּ֥הַק ה֛וּא פָּרַ֥ח בָּע֖וֹר טָה֥וֹר הֽוּא׃ הַכֹּהֵן יִרְאֶה וְהִנֵּה יֵשׁ בְּעוֹרָם כְּתָמִים בְּהִירִים דְּהוּיִים לְבָנִים, זוֹ לַבְקָנוּת שֶׁפָּרְחָה בָּעוֹר – זֶה טָהוֹר.

ששי (שלישי במחוברין)[עריכה]

פ פָּרָשַׁת קַרַחַת וְגַבַּחַת
מ וְאִ֕ישׁ כִּ֥י יִמָּרֵ֖ט רֹאשׁ֑וֹ קֵרֵ֥חַ ה֖וּא טָה֥וֹר הֽוּא׃ וְאִישׁ שֶׁשֵּׂעַר רֹאשׁוֹ יִתָּלֵשׁ – הוּא קֵרֵחַ, הוּא טָהוֹר.
מא וְאִם֙ מִפְּאַ֣ת פָּנָ֔יו יִמָּרֵ֖ט רֹאשׁ֑וֹ גִּבֵּ֥חַ ה֖וּא טָה֥וֹר הֽוּא׃ וְאִם שֵׂעַר רֹאשׁוֹ יִתָּלֵשׁ מֵהַצַּד שֶׁמּוּל פָּנָיו – הוּא גִּבֵּחַ, הוּא טָהוֹר.
מב וְכִֽי־יִהְיֶ֤ה בַקָּרַ֙חַת֙ א֣וֹ בַגַּבַּ֔חַת נֶ֖גַע לָבָ֣ן אֲדַמְדָּ֑ם צָרַ֤עַת פֹּרַ֙חַת֙ הִ֔וא בְּקָרַחְתּ֖וֹ א֥וֹ בְגַבַּחְתּֽוֹ׃ וְאִם יִהְיֶה בַּקָּרַחַת אוֹ בַּגַּבַּחַת נֶגַע לָבָן מְעוֹרָב בְּאָדוֹם, זוֹ צָרַעַת שֶׁפּוֹרַחַת בְּקָרַחְתּוֹ אוֹ בְּגַבַּחְתּוֹ.
מג וְרָאָ֨ה אֹת֜וֹ הַכֹּהֵ֗ן וְהִנֵּ֤ה שְׂאֵת־הַנֶּ֙גַע֙ לְבָנָ֣ה אֲדַמְדֶּ֔מֶת בְּקָרַחְתּ֖וֹ א֣וֹ בְגַבַּחְתּ֑וֹ כְּמַרְאֵ֥ה צָרַ֖עַת ע֥וֹר בָּשָֽׂר׃ הַכֹּהֵן יִרְאֶה אוֹתוֹ וְהִנֵּה הַשּׁוּמָה שֶׁל הַנֶּגַע לְבָנָה מְעוֹרֶבֶת בְּאָדוֹם – בְּקָרַחְתּוֹ אוֹ בְּגַבַּחְתּוֹ, דִּינוֹ כְּמוֹ מַרְאֶה שֶׁל צָרַעַת עוֹר.
מד אִישׁ־צָר֥וּעַ ה֖וּא טָמֵ֣א ה֑וּא טַמֵּ֧א יְטַמְּאֶ֛נּוּ הַכֹּהֵ֖ן בְּרֹאשׁ֥וֹ נִגְעֽוֹ׃ הוּא אִישׁ צָרוּעַ – הוּא טָמֵא, הַכֹּהֵן יְטַמֵּא אוֹתוֹ כִּי יֵשׁ בְּרֹאשׁוֹ נֶגַע.
מה וְהַצָּר֜וּעַ אֲשֶׁר־בּ֣וֹ הַנֶּ֗גַע בְּגָדָ֞יו יִהְי֤וּ פְרֻמִים֙ וְרֹאשׁוֹ֙ יִהְיֶ֣ה פָר֔וּעַ וְעַל־שָׂפָ֖ם יַעְטֶ֑ה וְטָמֵ֥א ׀ טָמֵ֖א יִקְרָֽא׃ וְהַצָּרוּעַ שֶׁיֵּשׁ בּוֹ נֶגַע – בְּגָדָיו יִהְיוּ פְּרוּמִים, וְרֹאשׁוֹ יִהְיֶה גָּלוּי – בְּלִי כּוֹבַע מְכֻבָּד, וְהוּא יַעֲטוֹף אֶת רֹאשׁוֹ וְאֶת שְׂפָתָיו בְּבִגְדּוֹ, וְיַכְרִיז: זְהִירוּת! אֲנִי טָמֵא!
מו כׇּל־יְמֵ֞י אֲשֶׁ֨ר הַנֶּ֥גַע בּ֛וֹ יִטְמָ֖א טָמֵ֣א ה֑וּא בָּדָ֣ד יֵשֵׁ֔ב מִח֥וּץ לַֽמַּחֲנֶ֖ה מוֹשָׁבֽוֹ׃ כָּל הַיָּמִים שֶׁהַנֶּגַע בּוֹ הוּא יִהְיֶה טָמֵא, הוּא יֵשֵׁב בּוֹדֵד – הוּא יָגוּר מִחוּץ לַמַּחֲנֶה.
ס פָּרָשַׁת צָרַעַת הַבֶּגֶד
מז וְהַבֶּ֕גֶד כִּֽי־יִהְיֶ֥ה ב֖וֹ נֶ֣גַע צָרָ֑עַת בְּבֶ֣גֶד צֶ֔מֶר א֖וֹ בְּבֶ֥גֶד פִּשְׁתִּֽים׃ וּבֶגֶד, אִם יִהְיֶה בּוֹ נֶגַע צָרָעַת – בְּבֶגֶד צֶמֶר אוֹ בְּבֶגֶד פִּשְׁתָּן,
מח א֤וֹ בִֽשְׁתִי֙ א֣וֹ בְעֵ֔רֶב לַפִּשְׁתִּ֖ים וְלַצָּ֑מֶר א֣וֹ בְע֔וֹר א֖וֹ בְּכׇל־מְלֶ֥אכֶת עֽוֹר׃ בֵּין בִּשְׁתִי בֵּין בְּעֵרֶב, בֵּין בְּפִשְׁתָּן בֵּין בְּצֶמֶר, בֵּין בְּעוֹר בֵּין בְּכָל דָּבָר שֶׁעָשׂוּי מֵעוֹר,
מט וְהָיָ֨ה הַנֶּ֜גַע יְרַקְרַ֣ק ׀ א֣וֹ אֲדַמְדָּ֗ם בַּבֶּ֩גֶד֩ א֨וֹ בָע֜וֹר אֽוֹ־בַשְּׁתִ֤י אוֹ־בָעֵ֙רֶב֙ א֣וֹ בְכׇל־כְּלִי־ע֔וֹר נֶ֥גַע צָרַ֖עַת ה֑וּא וְהׇרְאָ֖ה אֶת־הַכֹּהֵֽן׃ וְהַנֶּגַע יִהְיֶה צְהַבְהַב אוֹ אָדוֹם, בַּבֶּגֶד אוֹ בָּעוֹר, אוֹ בַּשְּׁתִי אוֹ בָּעֵרֶב, אוֹ בְּכָל כְּלִי עוֹר – זֶהוּ נֶגַע צָרַעַת, וְיַרְאוּ אוֹתוֹ לַכֹּהֵן.
נ וְרָאָ֥ה הַכֹּהֵ֖ן אֶת־הַנָּ֑גַע וְהִסְגִּ֥יר אֶת־הַנֶּ֖גַע שִׁבְעַ֥ת יָמִֽים׃ הַכֹּהֵן יִרְאֶה אֶת הַנֶּגַע, וְיַסְגִּיר אֶת הַנֶּגַע שִׁבְעָה יָמִים.
נא וְרָאָ֨ה אֶת־הַנֶּ֜גַע בַּיּ֣וֹם הַשְּׁבִיעִ֗י כִּֽי־פָשָׂ֤ה הַנֶּ֙גַע֙ בַּ֠בֶּ֠גֶד אֽוֹ־בַשְּׁתִ֤י אֽוֹ־בָעֵ֙רֶב֙ א֣וֹ בָע֔וֹר לְכֹ֛ל אֲשֶׁר־יֵעָשֶׂ֥ה הָע֖וֹר לִמְלָאכָ֑ה צָרַ֧עַת מַמְאֶ֛רֶת הַנֶּ֖גַע טָמֵ֥א הֽוּא׃ הוּא יִרְאֶה אֶת הַנֶּגַע בַּיּוֹם הַשְּׁבִיעִי, אִם הַנֶּגַע הִתְפַּשֵּׁט בַּבֶּגֶד אוֹ בַּשְּׁתִי אוֹ בָּעֵרֶב אוֹ בָּעוֹר – בְּכָל דָּבָר שֶׁעָשׂוּי מֵעוֹר, הַנֶּגַע הוּא צָרַעַת מַכְאִיבָה, הוּא טָמֵא.
נב וְשָׂרַ֨ף אֶת־הַבֶּ֜גֶד א֥וֹ אֶֽת־הַשְּׁתִ֣י ׀ א֣וֹ אֶת־הָעֵ֗רֶב בַּצֶּ֙מֶר֙ א֣וֹ בַפִּשְׁתִּ֔ים א֚וֹ אֶת־כׇּל־כְּלִ֣י הָע֔וֹר אֲשֶׁר־יִהְיֶ֥ה ב֖וֹ הַנָּ֑גַע כִּֽי־צָרַ֤עַת מַמְאֶ֙רֶת֙ הִ֔וא בָּאֵ֖שׁ תִּשָּׂרֵֽף׃ הוּא יִשְׂרוֹף אֶת הַבֶּגֶד אוֹ אֶת הַשְּׁתִי אוֹ אֶת הָעֵרֶב הֶעָשׂוּי מִצֶּמֶר אוֹ מִפִּשְׁתָּן, אוֹ כָּל כְּלִי עוֹר שֶׁיִּהְיֶה בּוֹ הַנֶּגַע, כִּי זוֹ צָרַעַת מַכְאִיבָה – הִיא תִּשָּׂרֵף בָּאֵשׁ.
נג וְאִם֮ יִרְאֶ֣ה הַכֹּהֵן֒ וְהִנֵּה֙ לֹא־פָשָׂ֣ה הַנֶּ֔גַע בַּבֶּ֕גֶד א֥וֹ בַשְּׁתִ֖י א֣וֹ בָעֵ֑רֶב א֖וֹ בְּכׇל־כְּלִי־עֽוֹר׃ וְאִם הַכֹּהֵן יִרְאֶה וְהִנֵּה הַנֶּגַע לֹא הִתְפַּשֵּׁט בַּבֶּגֶד אוֹ בַּשְּׁתִי אוֹ בָּעֵרֶב אוֹ בְּכָל כְּלִי עוֹר.
נד וְצִוָּה֙ הַכֹּהֵ֔ן וְכִ֨בְּס֔וּ אֵ֥ת אֲשֶׁר־בּ֖וֹ הַנָּ֑גַע וְהִסְגִּיר֥וֹ שִׁבְעַת־יָמִ֖ים שֵׁנִֽית׃ הַכֹּהֵן יְצַוֶּה שֶׁיְּכַבְּסוּ אֶת הַדָּבָר שֶׁבּוֹ הַנֶּגַע וְיַסְגִּירוּ אוֹתוֹ שֵׁנִית שִׁבְעָה יָמִים.

שביעי (רביעי במחוברין)[עריכה]

נה וְרָאָ֨ה הַכֹּהֵ֜ן אַחֲרֵ֣י ׀ הֻכַּבֵּ֣ס אֶת־הַנֶּ֗גַע וְ֠הִנֵּ֠ה לֹֽא־הָפַ֨ךְ הַנֶּ֤גַע אֶת־עֵינוֹ֙ וְהַנֶּ֣גַע לֹֽא־פָשָׂ֔ה טָמֵ֣א ה֔וּא בָּאֵ֖שׁ תִּשְׂרְפֶ֑נּוּ פְּחֶ֣תֶת הִ֔וא בְּקָרַחְתּ֖וֹ א֥וֹ בְגַבַּחְתּֽוֹ׃ הַכֹּהֵן יִרְאֶה אַחֲרֵי שֶׁכִּבְּסוּ אֶת הַנֶּגַע, וְהִנֵּה הַנֶּגַע לֹא שִׁנָּה אֶת צִבְעוֹ וְהַנֶּגַע לֹא הִתְפַּשֵּׁט, הוּא טָמֵא – תִּשְׂרוֹף אוֹתוֹ בָּאֵשׁ, זֶהוּ חֶסְרוֹן בְּמָקוֹם שָׁחוּק אוֹ לֹא שָׁחוּק.
נו וְאִם֮ רָאָ֣ה הַכֹּהֵן֒ וְהִנֵּה֙ כֵּהָ֣ה הַנֶּ֔גַע אַחֲרֵ֖י הֻכַּבֵּ֣ס אֹת֑וֹ וְקָרַ֣ע אֹת֗וֹ מִן־הַבֶּ֙גֶד֙ א֣וֹ מִן־הָע֔וֹר א֥וֹ מִן־הַשְּׁתִ֖י א֥וֹ מִן־הָעֵֽרֶב׃ וְאִם הַכֹּהֵן יִרְאֶה וְהִנֵּה הַנֶּגַע נֶחֱלַשׁ אַחֲרֵי שֶׁכִּבְּסוּ אוֹתוֹ, הוּא יִקְרַע אוֹתוֹ מֵהַבֶּגֶד אוֹ מֵהָעוֹר אוֹ מֵהַשְּׁתִי אוֹ מֵהָעֵרֶב.

מפטיר[עריכה]

נז וְאִם־תֵּרָאֶ֨ה ע֜וֹד בַּ֠בֶּ֠גֶד אֽוֹ־בַשְּׁתִ֤י אֽוֹ־בָעֵ֙רֶב֙ א֣וֹ בְכׇל־כְּלִי־ע֔וֹר פֹּרַ֖חַת הִ֑וא בָּאֵ֣שׁ תִּשְׂרְפֶ֔נּוּ אֵ֥ת אֲשֶׁר־בּ֖וֹ הַנָּֽגַע׃ וְאִם הִיא תֵּרָאֶה שׁוּב בַּבֶּגֶד אוֹ בַּשְּׁתִי אוֹ בָּעֵרֶב אוֹ בְּכָל כְּלִי עוֹר, זוֹ צָרַעַת מִתְפַּשֶּׁטֶת – תִּשְׂרוֹף בָּאֵשׁ אֶת הַדָּבָר שֶׁבּוֹ הַנֶּגַע.
נח וְהַבֶּ֡גֶד אֽוֹ־הַשְּׁתִ֨י אוֹ־הָעֵ֜רֶב אֽוֹ־כׇל־כְּלִ֤י הָעוֹר֙ אֲשֶׁ֣ר תְּכַבֵּ֔ס וְסָ֥ר מֵהֶ֖ם הַנָּ֑גַע וְכֻבַּ֥ס שֵׁנִ֖ית וְטָהֵֽר׃ וְהַבֶּגֶד אוֹ הַשְּׁתִי אוֹ הָעֵרֶב אוֹ כָּל כְּלִי עוֹר, שֶׁיְּכֻבַּס וְיֵרֵד מֵהֶם הַנֶּגַע – הוּא יְכֻבַּס שֵׁנִית וְיִהְיֶה טָהוֹר.
נט זֹ֠את תּוֹרַ֨ת נֶֽגַע־צָרַ֜עַת בֶּ֥גֶד הַצֶּ֣מֶר ׀ א֣וֹ הַפִּשְׁתִּ֗ים א֤וֹ הַשְּׁתִי֙ א֣וֹ הָעֵ֔רֶב א֖וֹ כׇּל־כְּלִי־ע֑וֹר לְטַהֲר֖וֹ א֥וֹ לְטַמְּאֽוֹ׃ אֵלֶּה דִּינֵי נֶגַע צָרַעַת שֶׁל בֶּגֶד מִצֶּמֶר אוֹ מִפִּשְׁתָּן אוֹ שְׁתִי אוֹ עֵרֶב אוֹ כָּל כְּלִי עוֹר, אִם לְטַהֵר אוֹ לְטַמֵּא אוֹתוֹ.
