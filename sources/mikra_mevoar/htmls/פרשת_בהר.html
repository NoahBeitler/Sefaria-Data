ראשון[עריכה]

פ פָּרָשַׁת הַשְּׁמִטָּה
כה א וַיְדַבֵּ֤ר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה בְּהַ֥ר סִינַ֖י לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה בְּהַר סִינַי וְאָמַר:
ב דַּבֵּ֞ר אֶל־בְּנֵ֤י יִשְׂרָאֵל֙ וְאָמַרְתָּ֣ אֲלֵהֶ֔ם כִּ֤י תָבֹ֙אוּ֙ אֶל־הָאָ֔רֶץ אֲשֶׁ֥ר אֲנִ֖י נֹתֵ֣ן לָכֶ֑ם וְשָׁבְתָ֣ה הָאָ֔רֶץ שַׁבָּ֖ת לַיהֹוָֽה׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וֶאֱמוֹר לָהֶם: כְּשֶׁתָּבוֹאוּ לָאָרֶץ שֶׁאֲנִי נוֹתֵן לָכֶם, הָאָרֶץ תִּשְׁבּוֹת לִכְבוֹד ה'.
ג שֵׁ֤שׁ שָׁנִים֙ תִּזְרַ֣ע שָׂדֶ֔ךָ וְשֵׁ֥שׁ שָׁנִ֖ים תִּזְמֹ֣ר כַּרְמֶ֑ךָ וְאָסַפְתָּ֖ אֶת־תְּבוּאָתָֽהּ׃ שֵׁשׁ שָׁנִים תִּזְרַע אֶת שָׂדְךָ, וְתִגְזוֹם אֶת הָעֲנָפִים בְּכַרְמְךָ, וְתֶאֱסוֹף אֶת תְּבוּאַת הָאָרֶץ.
ד וּבַשָּׁנָ֣ה הַשְּׁבִיעִ֗ת שַׁבַּ֤ת שַׁבָּתוֹן֙ יִהְיֶ֣ה לָאָ֔רֶץ שַׁבָּ֖ת לַיהֹוָ֑ה שָֽׂדְךָ֙ לֹ֣א תִזְרָ֔ע וְכַרְמְךָ֖ לֹ֥א תִזְמֹֽר׃ וּבַשָּׁנָה הַשְּׁבִיעִית תִּהְיֶה שַׁבַּת מְנוּחָה לָאָרֶץ, שַׁבָּת לִכְבוֹד ה' – אַל תִּזְרַע אֶת שָׂדְךָ וְאַל תִּגְזוֹם אֶת עַנְפֵי כַּרְמְךָ.
ה אֵ֣ת סְפִ֤יחַ קְצִֽירְךָ֙ לֹ֣א תִקְצ֔וֹר וְאֶת־עִנְּבֵ֥י נְזִירֶ֖ךָ לֹ֣א תִבְצֹ֑ר שְׁנַ֥ת שַׁבָּת֖וֹן יִהְיֶ֥ה לָאָֽרֶץ׃ אַל תִּקְצוֹר אֲפִלּוּ אֶת הַזְּרָעִים שֶׁנָּפְלוּ בַּקְּצִירָה וְצָמְחוּ, וְאַל תִּבְצוֹר אֶת הָעֲנָבִים שֶׁפָּרַשְׁתָּ מֵעֲבוֹדָתָם, שְׁנַת מְנוּחָה תִּהְיֶה לָאָרֶץ.
ו וְ֠הָיְתָ֠ה שַׁבַּ֨ת הָאָ֤רֶץ לָכֶם֙ לְאׇכְלָ֔ה לְךָ֖ וּלְעַבְדְּךָ֣ וְלַאֲמָתֶ֑ךָ וְלִשְׂכִֽירְךָ֙ וּלְתוֹשָׁ֣בְךָ֔ הַגָּרִ֖ים עִמָּֽךְ׃ תְּבוּאַת שַׁבַּת הָאָרֶץ תִּהְיֶה מְיוּעֶדֶת לַאֲכִילָה, לְךָ וּלְעַבְדְּךָ וּלְשִׁפְחָתְךָ וְלַפּוֹעֵל שֶׁלְּךָ וּלְגֵר הַתּוֹשָׁב שֶׁגָּרִים אִתְּךָ.
ז וְלִ֨בְהֶמְתְּךָ֔ וְלַֽחַיָּ֖ה אֲשֶׁ֣ר בְּאַרְצֶ֑ךָ תִּהְיֶ֥ה כׇל־תְּבוּאָתָ֖הּ לֶאֱכֹֽל׃ וְגַם לִבְהֶמְתְּךָ וְלַחַיָּה שֶׁבְּאַרְצְךָ כָּל תְּבוּאַת אוֹתָהּ שָׁנָה תִּהְיֶה מְיוּעֶדֶת לַאֲכִילָה.
ס פָּרָשַׁת הַיּוֹבֵל
ח וְסָפַרְתָּ֣ לְךָ֗ שֶׁ֚בַע שַׁבְּתֹ֣ת שָׁנִ֔ים שֶׁ֥בַע שָׁנִ֖ים שֶׁ֣בַע פְּעָמִ֑ים וְהָי֣וּ לְךָ֗ יְמֵי֙ שֶׁ֚בַע שַׁבְּתֹ֣ת הַשָּׁנִ֔ים תֵּ֥שַׁע וְאַרְבָּעִ֖ים שָׁנָֽה׃ סְפוֹר לְךָ שֶׁבַע שְׁנוֹת מְנוּחָה – שֶׁבַע שָׁנִים שֶׁבַע פְּעָמִים, וְיִהְיֶה סְכוּם תְּקוּפַת שֶׁבַע שְׁנוֹת הַמְּנוּחָה – אַרְבָּעִים וְתֵשַׁע שָׁנִים.
ט וְהַֽעֲבַרְתָּ֞ שׁוֹפַ֤ר תְּרוּעָה֙ בַּחֹ֣דֶשׁ הַשְּׁבִעִ֔י בֶּעָשׂ֖וֹר לַחֹ֑דֶשׁ בְּיוֹם֙ הַכִּפֻּרִ֔ים תַּעֲבִ֥ירוּ שׁוֹפָ֖ר בְּכׇל־אַרְצְכֶֽם׃ תַּכְרִיז בִּתְרוּעַת שׁוֹפָר בַּחוֹדֶשׁ הַשְּׁבִיעִי בַּעֲשִׂירִי לַחוֹדֶשׁ – בְּיוֹם הַכִּפּוּרִים תַּכְרִיזוּ בְּשׁוֹפָר בְּכָל אַרְצְכֶם.
י וְקִדַּשְׁתֶּ֗ם אֵ֣ת שְׁנַ֤ת הַחֲמִשִּׁים֙ שָׁנָ֔ה וּקְרָאתֶ֥ם דְּר֛וֹר בָּאָ֖רֶץ לְכׇל־יֹשְׁבֶ֑יהָ יוֹבֵ֥ל הִוא֙ תִּהְיֶ֣ה לָכֶ֔ם וְשַׁבְתֶּ֗ם אִ֚ישׁ אֶל־אֲחֻזָּת֔וֹ וְאִ֥ישׁ אֶל־מִשְׁפַּחְתּ֖וֹ תָּשֻֽׁבוּ׃ קַדְּשׁוּ אֶת שְׁנַת הַחֲמִשִּׁים וְתַכְרִיזוּ עַל חוֹפֶשׁ בָּאָרֶץ לְכָל תּוֹשָׁבֶיהָ, זוֹ תִּהְיֶה לָכֶם שְׁנַת הַיּוֹבֵל – כָּל אִישׁ יַחֲזוֹר אֶל שְׂדֵה אֲחֻזָּתוֹ וְכָל עֶבֶד יַחֲזוֹר אֶל מִשְׁפַּחְתּוֹ.
יא יוֹבֵ֣ל הִ֗וא שְׁנַ֛ת הַחֲמִשִּׁ֥ים שָׁנָ֖ה תִּהְיֶ֣ה לָכֶ֑ם לֹ֣א תִזְרָ֔עוּ וְלֹ֤א תִקְצְרוּ֙ אֶת־סְפִיחֶ֔יהָ וְלֹ֥א תִבְצְר֖וּ אֶת־נְזִרֶֽיהָ׃ זוֹ יוֹבֵל – שְׁנַת הַחֲמִשִּׁים שֶׁתִּהְיֶה לָכֶם, אַל תִּזְרְעוּ בָּהּ וְאַל תִּקְצְרוּ אֲפִלּוּ אֶת הַזְּרָעִים שֶׁנָּפְלוּ בַּקְּצִירָה וְצָמְחוּ, וְאַל תִּבְצְרוּ אֶת הָעֲנָבִים שֶׁפְּרַשְׁתֶּם מֵעֲבוֹדָתָם.
יב כִּ֚י יוֹבֵ֣ל הִ֔וא קֹ֖דֶשׁ תִּהְיֶ֣ה לָכֶ֑ם מִ֨ן־הַשָּׂדֶ֔ה תֹּאכְל֖וּ אֶת־תְּבוּאָתָֽהּ׃ כִּי הִיא יוֹבֵל, הִיא תִּהְיֶה לָכֶם קוֹדֶשׁ, תֹּאכְלוּ אֶת הַתְּבוּאָה יְשִׁירוֹת מֵהַשָּׂדֶה בְּלִי לֶאֱגוֹר.
יג בִּשְׁנַ֥ת הַיּוֹבֵ֖ל הַזֹּ֑את תָּשֻׁ֕בוּ אִ֖ישׁ אֶל־אֲחֻזָּתֽוֹ׃ בִּשְׁנַת הַיּוֹבֵל הַזֹּאת תָּשׁוּבוּ כָּל אִישׁ לַאֲחֻזָּתוֹ.

שני[עריכה]

יד וְכִֽי־תִמְכְּר֤וּ מִמְכָּר֙ לַעֲמִיתֶ֔ךָ א֥וֹ קָנֹ֖ה מִיַּ֣ד עֲמִיתֶ֑ךָ אַל־תּוֹנ֖וּ אִ֥ישׁ אֶת־אָחִֽיו׃ כְּשֶׁתִּמְכּוֹר לַחֲבֵרְךָ אוֹ שֶׁתִּקְנֶה מִיַּד חֲבֵרְךָ אַל תְּרַמּוּ אֶחָד אֶת הַשֵּׁנִי.
טו בְּמִסְפַּ֤ר שָׁנִים֙ אַחַ֣ר הַיּוֹבֵ֔ל תִּקְנֶ֖ה מֵאֵ֣ת עֲמִיתֶ֑ךָ בְּמִסְפַּ֥ר שְׁנֵֽי־תְבוּאֹ֖ת יִמְכׇּר־לָֽךְ׃ תִּקְנֶה מֵחֲבֵרְךָ וּתְשַׁלֵּם לְפִי מִסְפַּר הַשָּׁנִים שֶׁעָבְרוּ מֵאַחֲרֵי הַיּוֹבֵל – הוּא יִמְכּוֹר לְךָ לְפִי מִסְפַּר הַשָּׁנִים שֶׁתֹּאכַל אֶת הַתְּבוּאָה.
טז לְפִ֣י ׀ רֹ֣ב הַשָּׁנִ֗ים תַּרְבֶּה֙ מִקְנָת֔וֹ וּלְפִי֙ מְעֹ֣ט הַשָּׁנִ֔ים תַּמְעִ֖יט מִקְנָת֑וֹ כִּ֚י מִסְפַּ֣ר תְּבוּאֹ֔ת ה֥וּא מֹכֵ֖ר לָֽךְ׃ לְפִי רִבּוּי הַשָּׁנִים תַּעֲלֶה אֶת הַמְּחִיר וּלְפִי מִעוּט הַשָּׁנִים תּוֹרִיד אֶת הַמְּחִיר, כִּי הוּא מוֹכֵר לְךָ מִסְפַּר שֶׁל שְׁנוֹת תְּבוּאָה.
יז וְלֹ֤א תוֹנוּ֙ אִ֣ישׁ אֶת־עֲמִית֔וֹ וְיָרֵ֖אתָ מֵֽאֱלֹהֶ֑יךָ כִּ֛י אֲנִ֥י יְהֹוָ֖ה אֱלֹהֵיכֶֽם׃ וְאַל תְּרַמּוּ אֶחָד אֶת הַשֵּׁנִי וּתְכַבֵּד אֶת אֱלֹהֶיךָ, כִּי אֲנִי ה' אֱלֹהֵיכֶם.
יח וַעֲשִׂיתֶם֙ אֶת־חֻקֹּתַ֔י וְאֶת־מִשְׁפָּטַ֥י תִּשְׁמְר֖וּ וַעֲשִׂיתֶ֣ם אֹתָ֑ם וִֽישַׁבְתֶּ֥ם עַל־הָאָ֖רֶץ לָבֶֽטַח׃ תַּעֲשׂוּ אֶת חֻקַּי וְתִשְׁמְרוּ אֶת מִשְׁפָּטַי וְתַעֲשׂוּ אוֹתָם, וְאָז תֵּשְׁבוּ בָּאָרֶץ בְּבִטָּחוֹן.

שלישי (שני במחוברין)[עריכה]

יט וְנָתְנָ֤ה הָאָ֙רֶץ֙ פִּרְיָ֔הּ וַאֲכַלְתֶּ֖ם לָשֹׂ֑בַע וִֽישַׁבְתֶּ֥ם לָבֶ֖טַח עָלֶֽיהָ׃ הָאָרֶץ תִּתֵּן אֶת פֵּרוֹתֶיהָ וְתֹאכְלוּ לָשׂוֹבַע, וְתֵשְׁבוּ עָלֶיהָ בְּבִטָּחוֹן.
כ וְכִ֣י תֹאמְר֔וּ מַה־נֹּאכַ֖ל בַּשָּׁנָ֣ה הַשְּׁבִיעִ֑ת הֵ֚ן לֹ֣א נִזְרָ֔ע וְלֹ֥א נֶאֱסֹ֖ף אֶת־תְּבוּאָתֵֽנוּ׃ וְאִם תֹּאמְרוּ מַה נֹּאכַל בַּשָּׁנָה הַשְּׁבִיעִית? הֲרֵי לֹא נִזְרַע וְלֹא נֶאֱסוֹף אֶת תְּבוּאָתֵנוּ!
כא וְצִוִּ֤יתִי אֶת־בִּרְכָתִי֙ לָכֶ֔ם בַּשָּׁנָ֖ה הַשִּׁשִּׁ֑ית וְעָשָׂת֙ אֶת־הַתְּבוּאָ֔ה לִשְׁלֹ֖שׁ הַשָּׁנִֽים׃ הַתְּשׁוּבָה: אֲצַוֶּה אֶת בִּרְכָתִי שֶׁתָּבוֹא אֲלֵיכֶם בַּשָּׁנָה הַשִּׁשִּׁית שֶׁתַּצְמִיחַ תְּבוּאָה לִשְׁלוֹשׁ הַשָּׁנִים הַבָּאוֹת.
כב וּזְרַעְתֶּ֗ם אֵ֚ת הַשָּׁנָ֣ה הַשְּׁמִינִ֔ת וַאֲכַלְתֶּ֖ם מִן־הַתְּבוּאָ֣ה יָשָׁ֑ן עַ֣ד ׀ הַשָּׁנָ֣ה הַתְּשִׁיעִ֗ת עַד־בּוֹא֙ תְּב֣וּאָתָ֔הּ תֹּאכְל֖וּ יָשָֽׁן׃ וְתִזְרְעוּ בַּשָּׁנָה הַשְּׁמִינִית וְתֹאכְלוּ מֵהַתְּבוּאָה הַיְּשָׁנָה עַד הַשָּׁנָה הַתְּשִׁיעִית – עַד הַגָּעַת תְּבוּאַת אוֹתָהּ שָׁנָה תֹּאכְלוּ מֵהַתְּבוּאָה הַיְּשָׁנָה.
כג וְהָאָ֗רֶץ לֹ֤א תִמָּכֵר֙ לִצְמִתֻ֔ת כִּי־לִ֖י הָאָ֑רֶץ כִּֽי־גֵרִ֧ים וְתוֹשָׁבִ֛ים אַתֶּ֖ם עִמָּדִֽי׃ וְהָאָרֶץ לֹא תִּמָּכֵר לְתָמִיד כִּי הָאָרֶץ הִיא שֶׁלִּי, הֲרֵי אַתֶּם גֵּרִים וְתוֹשָׁבִים אֶצְלִי.
כד וּבְכֹ֖ל אֶ֣רֶץ אֲחֻזַּתְכֶ֑ם גְּאֻלָּ֖ה תִּתְּנ֥וּ לָאָֽרֶץ׃ וּבְכָל אֶרֶץ אֲחֻזַּתְכֶם תִּפְדּוּ אֶת הָאֲדָמָה בְּכָךְ שֶׁתַּחֲזִירוּ אוֹתָהּ לִבְעָלֶיהָ.

רביעי[עריכה]

ס פָּרָשַׁת גְּאֻלַּת הַקַּרְקָעוֹת
כה כִּֽי־יָמ֣וּךְ אָחִ֔יךָ וּמָכַ֖ר מֵאֲחֻזָּת֑וֹ וּבָ֤א גֹֽאֲלוֹ֙ הַקָּרֹ֣ב אֵלָ֔יו וְגָאַ֕ל אֵ֖ת מִמְכַּ֥ר אָחִֽיו׃ אִם אָחִיךָ יִהְיֶה עָנִי וְיִמְכּוֹר מֵאֲחֻזָּתוֹ, יָבוֹא קְרוֹבוֹ הַקָּרוֹב אֵלָיו בְּיוֹתֵר וְיִפְדֶּה אֶת הַשָּׂדֶה שֶׁאָחִיו מָכַר.
כו וְאִ֕ישׁ כִּ֛י לֹ֥א יִֽהְיֶה־לּ֖וֹ גֹּאֵ֑ל וְהִשִּׂ֣יגָה יָד֔וֹ וּמָצָ֖א כְּדֵ֥י גְאֻלָּתֽוֹ׃ וְאִישׁ שֶׁלֹּא הָיָה לוֹ קָרוֹב לִפְדּוֹת, וְהוּא הִרְוִיחַ וְהִשִּׂיג מַסְפִּיק כֶּסֶף כְּדֵי לִפְדּוֹת אֶת שָׂדֵהוּ,
כז וְחִשַּׁב֙ אֶת־שְׁנֵ֣י מִמְכָּר֔וֹ וְהֵשִׁיב֙ אֶת־הָ֣עֹדֵ֔ף לָאִ֖ישׁ אֲשֶׁ֣ר מָֽכַר־ל֑וֹ וְשָׁ֖ב לַאֲחֻזָּתֽוֹ׃ הוּא יְחַשֵּׁב אֶת הַשָּׁנִים שֶׁעָבְרוּ מֵאָז הַמְּכִירָה וְיַחֲזִיר אֶת הַהֶפְרֵשׁ לַקּוֹנֶה שֶׁהוּא מָכַר לוֹ, וְהוּא יַחֲזוֹר לַאֲחֻזָּתוֹ.
כח וְאִ֨ם לֹֽא־מָצְאָ֜ה יָד֗וֹ דֵּי֮ הָשִׁ֣יב לוֹ֒ וְהָיָ֣ה מִמְכָּר֗וֹ בְּיַד֙ הַקֹּנֶ֣ה אֹת֔וֹ עַ֖ד שְׁנַ֣ת הַיּוֹבֵ֑ל וְיָצָא֙ בַּיֹּבֵ֔ל וְשָׁ֖ב לַאֲחֻזָּתֽוֹ׃ וְאִם לֹא יַשִּׂיג מַסְפִּיק כֶּסֶף לְהַחֲזִיר לַקּוֹנֶה, הַשָּׂדֶה הַנִּמְכֶּרֶת תִּהְיֶה בִּידֵי מִי שֶׁקָּנָה אוֹתָהּ עַד שְׁנַת הַיּוֹבֵל, וְהִיא תַּחֲזוֹר בַּיּוֹבֵל וְהוּא יַחֲזוֹר לַאֲחֻזָּתוֹ.

חמישי (שלישי במחוברין)[עריכה]

ס
כט וְאִ֗ישׁ כִּֽי־יִמְכֹּ֤ר בֵּית־מוֹשַׁב֙ עִ֣יר חוֹמָ֔ה וְהָיְתָה֙ גְּאֻלָּת֔וֹ עַד־תֹּ֖ם שְׁנַ֣ת מִמְכָּר֑וֹ יָמִ֖ים תִּהְיֶ֥ה גְאֻלָּתֽוֹ׃ וְאִם אִישׁ יִמְכּוֹר בֵּית מְגוּרִים בְּעִיר מֻקֶּפֶת חוֹמָה, הוּא יוּכַל לִפְדּוֹת אוֹתוֹ עַד סוֹף שְׁנַת מְכִירָתוֹ – תְּקוּפַת פִּדְיוֹנוֹ תִּהְיֶה שָׁנָה.
ל וְאִ֣ם לֹֽא־יִגָּאֵ֗ל עַד־מְלֹ֣את לוֹ֮ שָׁנָ֣ה תְמִימָה֒ וְ֠קָ֠ם הַבַּ֨יִת אֲשֶׁר־בָּעִ֜יר אֲשֶׁר־ל֣וֹ לא חֹמָ֗ה לַצְּמִיתֻ֛ת לַקֹּנֶ֥ה אֹת֖וֹ לְדֹרֹתָ֑יו לֹ֥א יֵצֵ֖א בַּיֹּבֵֽל׃ וְאִם הוּא לֹא יִפָּדֶה עַד שֶׁתַּעֲבוֹר לוֹ שָׁנָה שְׁלֵמָה, הַבַּיִת שֶׁבָּעִיר שֶׁיֵּשׁ לָהּ חוֹמָה יִהְיֶה שַׁיָּךְ לַקּוֹנֶה אוֹתוֹ וּלְיוֹרְשָׁיו לְתָמִיד – הוּא לֹא יֵצֵא בַּיּוֹבֵל.
לא וּבָתֵּ֣י הַחֲצֵרִ֗ים אֲשֶׁ֨ר אֵין־לָהֶ֤ם חֹמָה֙ סָבִ֔יב עַל־שְׂדֵ֥ה הָאָ֖רֶץ יֵחָשֵׁ֑ב גְּאֻלָּה֙ תִּהְיֶה־לּ֔וֹ וּבַיֹּבֵ֖ל יֵצֵֽא׃ וּבָתֵּי הַכְּפָרִים שֶׁאֵין לָהֶם חוֹמָה מִסָּבִיב, יֵחָשְׁבוּ כְּמוֹ שְׂדֵה הָאָרֶץ – אֶפְשָׁר לִפְדּוֹת אוֹתָם תָּמִיד, וְהֵם יַחְזְרוּ בַּיּוֹבֵל.
לב וְעָרֵי֙ הַלְוִיִּ֔ם בָּתֵּ֖י עָרֵ֣י אֲחֻזָּתָ֑ם גְּאֻלַּ֥ת עוֹלָ֖ם תִּהְיֶ֥ה לַלְוִיִּֽם׃ וְעָרֵי הַלְוִיִּים – הַבָּתִּים שֶׁבְּעָרֵי נַחֲלָתָם, הַלְוִיִּים יוּכְלוּ לִפְדּוֹתָם תָּמִיד.
לג וַאֲשֶׁ֤ר יִגְאַל֙ מִן־הַלְוִיִּ֔ם וְיָצָ֧א מִמְכַּר־בַּ֛יִת וְעִ֥יר אֲחֻזָּת֖וֹ בַּיֹּבֵ֑ל כִּ֣י בָתֵּ֞י עָרֵ֣י הַלְוִיִּ֗ם הִ֚וא אֲחֻזָּתָ֔ם בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל׃ וּכְשֶׁאָדָם יִקְנֶה מֵהַלְוִיִּים, מְכִירַת הַבַּיִת אוֹ הָעִיר שֶׁבְּחֶלְקוֹ תִּתְבַּטֵּל בַּיּוֹבֵל, כִּי הַבָּתִּים שֶׁבְּעָרֵי הַלְוִיִּים הֵם חֶלְקָם בֵּין בְּנֵי יִשְׂרָאֵל.
לד וּֽשְׂדֵ֛ה מִגְרַ֥שׁ עָרֵיהֶ֖ם לֹ֣א יִמָּכֵ֑ר כִּֽי־אֲחֻזַּ֥ת עוֹלָ֛ם ה֖וּא לָהֶֽם׃ וּשְׂדֵה מִגְרַשׁ עָרֵיהֶם לֹא יִמָּכֵר לְתָמִיד, כִּי הוּא עֲבוּרָם אֲחֻזָּה לְעוֹלָם.
ס פָּרָשַׁת רִבִּית
לה וְכִֽי־יָמ֣וּךְ אָחִ֔יךָ וּמָ֥טָה יָד֖וֹ עִמָּ֑ךְ וְהֶֽחֱזַ֣קְתָּ בּ֔וֹ גֵּ֧ר וְתוֹשָׁ֛ב וָחַ֖י עִמָּֽךְ׃ וְאִם אָחִיךָ שֶׁאִתְּךָ יִהְיֶה עָנִי וּפַרְנָסָתוֹ תִּתְמוֹטֵט – תְּמוֹךְ בּוֹ, וְגַם בְּבֶן הַמִּיעוּטִים שֶׁאִתְּךָ – כְּדֵי שֶׁיּוּכַל לִחְיוֹת.
לו אַל־תִּקַּ֤ח מֵֽאִתּוֹ֙ נֶ֣שֶׁךְ וְתַרְבִּ֔ית וְיָרֵ֖אתָ מֵֽאֱלֹהֶ֑יךָ וְחֵ֥י אָחִ֖יךָ עִמָּֽךְ׃ אַל תִּקַּח מִמֶּנּוּ רִבִּית – כַּבֵּד אֶת אֱלֹהֶיךָ, וְאָחִיךָ יִחְיֶה אִתְּךָ.
לז אֶ֨ת־כַּסְפְּךָ֔ לֹֽא־תִתֵּ֥ן ל֖וֹ בְּנֶ֑שֶׁךְ וּבְמַרְבִּ֖ית לֹא־תִתֵּ֥ן אׇכְלֶֽךָ׃ אַל תַּלְוֶה לוֹ אֶת כַּסְפְּךָ בִּתְמוּרָה לְתוֹסֶפֶת, וְאַל תַּלְוֶה לוֹ אוֹכֶל בְּרִבִּית.
לח אֲנִ֗י יְהֹוָה֙ אֱלֹ֣הֵיכֶ֔ם אֲשֶׁר־הוֹצֵ֥אתִי אֶתְכֶ֖ם מֵאֶ֣רֶץ מִצְרָ֑יִם לָתֵ֤ת לָכֶם֙ אֶת־אֶ֣רֶץ כְּנַ֔עַן לִהְי֥וֹת לָכֶ֖ם לֵאלֹהִֽים׃ אֲנִי ה' אֱלֹהֵיכֶם שֶׁהוֹצֵאתִי אֶתְכֶם מֵאֶרֶץ מִצְרַיִם לָתֵת לָכֶם אֶת אֶרֶץ כְּנַעַן לִהְיוֹת לָכֶם לֵאלֹהִים.

ששי (רביעי במחוברין)[עריכה]

ס פָּרָשַׁת פִּדְיוֹן הַנִּמְכָּרִים
לט וְכִֽי־יָמ֥וּךְ אָחִ֛יךָ עִמָּ֖ךְ וְנִמְכַּר־לָ֑ךְ לֹא־תַעֲבֹ֥ד בּ֖וֹ עֲבֹ֥דַת עָֽבֶד׃ וְאִם אָחִיךָ שֶׁאִתְּךָ יִהְיֶה עָנִי וְיִמָּכֵר לְךָ, אַל תַּעֲבִיד אוֹתוֹ בַּעֲבוֹדָה שֶׁל עֶבֶד.
מ כְּשָׂכִ֥יר כְּתוֹשָׁ֖ב יִהְיֶ֣ה עִמָּ֑ךְ עַד־שְׁנַ֥ת הַיֹּבֵ֖ל יַעֲבֹ֥ד עִמָּֽךְ׃ הוּא יִהְיֶה אֶצְלְךָ כְּמוֹ פּוֹעֵל שָׂכִיר וְתוֹשָׁב זְמַנִּי, וְיַעֲבוֹד אֶצְלְךָ רַק עַד שְׁנַת הַיּוֹבֵל.
מא וְיָצָא֙ מֵֽעִמָּ֔ךְ ה֖וּא וּבָנָ֣יו עִמּ֑וֹ וְשָׁב֙ אֶל־מִשְׁפַּחְתּ֔וֹ וְאֶל־אֲחֻזַּ֥ת אֲבֹתָ֖יו יָשֽׁוּב׃ הוּא יֵצֵא מֵאֶצְלְךָ בַּיּוֹבֵל, הוּא וּבָנָיו אִתּוֹ, וְיַחֲזוֹר אֶל מִשְׁפַּחְתּוֹ וְאֶל אֲחֻזַּת אֲבוֹתָיו.
מב כִּֽי־עֲבָדַ֣י הֵ֔ם אֲשֶׁר־הוֹצֵ֥אתִי אֹתָ֖ם מֵאֶ֣רֶץ מִצְרָ֑יִם לֹ֥א יִמָּכְר֖וּ מִמְכֶּ֥רֶת עָֽבֶד׃ כִּי הֵם עֲבָדַי שֶׁהוֹצֵאתִי אוֹתָם מֵאֶרֶץ מִצְרַיִם, הֵם לֹא יִמָּכְרוּ כְּמוֹ שֶׁמּוֹכְרִים עֶבֶד.
מג לֹא־תִרְדֶּ֥ה ב֖וֹ בְּפָ֑רֶךְ וְיָרֵ֖אתָ מֵאֱלֹהֶֽיךָ׃ אַל תְּשַׁעְבֵּד אוֹתוֹ בַּעֲבוֹדָה מְפָרֶכֶת, כַּבֵּד אֶת אֱלֹהֶיךָ.
מד וְעַבְדְּךָ֥ וַאֲמָתְךָ֖ אֲשֶׁ֣ר יִהְיוּ־לָ֑ךְ מֵאֵ֣ת הַגּוֹיִ֗ם אֲשֶׁר֙ סְבִיבֹ֣תֵיכֶ֔ם מֵהֶ֥ם תִּקְנ֖וּ עֶ֥בֶד וְאָמָֽה׃ וְעַבְדְּךָ וְשִׁפְחָתְךָ שֶׁיִּהְיוּ לְךָ, יִהְיוּ מֵהָעַמִּים שֶׁסְּבִיבוֹתֵיכֶם – רַק מֵהֶם תִּקְנוּ עֶבֶד וְשִׁפְחָה.
מה וְ֠גַ֠ם מִבְּנֵ֨י הַתּוֹשָׁבִ֜ים הַגָּרִ֤ים עִמָּכֶם֙ מֵהֶ֣ם תִּקְנ֔וּ וּמִמִּשְׁפַּחְתָּם֙ אֲשֶׁ֣ר עִמָּכֶ֔ם אֲשֶׁ֥ר הוֹלִ֖ידוּ בְּאַרְצְכֶ֑ם וְהָי֥וּ לָכֶ֖ם לַֽאֲחֻזָּֽה׃ וְגַם מִבְּנֵי הַתּוֹשָׁבִים הַגּוֹיִם שֶׁגָּרִים אִתְּכֶם מֵהֶם תִּקְנוּ וּמִמִּשְׁפַּחְתָּם שֶׁאִתְּכֶם שֶׁהוֹלִידוּ בְּאַרְצְכֶם, וְהֵם יִהְיוּ לָכֶם לִירֻשָּׁה כְּמוֹ אֲחֻזָּה.
מו וְהִתְנַחַלְתֶּ֨ם אֹתָ֜ם לִבְנֵיכֶ֤ם אַחֲרֵיכֶם֙ לָרֶ֣שֶׁת אֲחֻזָּ֔ה לְעֹלָ֖ם בָּהֶ֣ם תַּעֲבֹ֑דוּ וּבְאַ֨חֵיכֶ֤ם בְּנֵֽי־יִשְׂרָאֵל֙ אִ֣ישׁ בְּאָחִ֔יו לֹא־תִרְדֶּ֥ה ב֖וֹ בְּפָֽרֶךְ׃ תּוֹרִישׁוּ אוֹתָם לִבְנֵיכֶם אַחֲרֵיכֶם שֶׁיִּירְשׁוּ אוֹתָם כְּמוֹ אֲחֻזָּה, תַּעַבְדוּ בָּהֶם לְעוֹלָם אַתֶּם וְצֶאֱצָאֵיכֶם, וּבְאַחֵיכֶם בְּנֵי יִשְׂרָאֵל – אִישׁ בְּאָחִיו – אַל תְּשַׁעְבֵּד אוֹתוֹ בַּעֲבוֹדָה מְפָרֶכֶת.

שביעי[עריכה]

ס
מז וְכִ֣י תַשִּׂ֗יג יַ֣ד גֵּ֤ר וְתוֹשָׁב֙ עִמָּ֔ךְ וּמָ֥ךְ אָחִ֖יךָ עִמּ֑וֹ וְנִמְכַּ֗ר לְגֵ֤ר תּוֹשָׁב֙ עִמָּ֔ךְ א֥וֹ לְעֵ֖קֶר מִשְׁפַּ֥חַת גֵּֽר׃ וְאִם בֶּן מִיעוּטִים שֶׁגָּר אֶצְלְךָ יַעֲשִׁיר, וְאָחִיךָ שֶׁגָּר לְיָדוֹ יִהְיֶה עָנִי וְיִמָּכֵר לְבֶן הַמִּיעוּטִים שֶׁאֶצְלְךָ אוֹ לְאֶחָד מִמִּשְׁפַּחְתּוֹ בְּחוּץ לָאָרֶץ.
מח אַחֲרֵ֣י נִמְכַּ֔ר גְּאֻלָּ֖ה תִּהְיֶה־לּ֑וֹ אֶחָ֥ד מֵאֶחָ֖יו יִגְאָלֶֽנּוּ׃ אַחֲרֵי שֶׁנִּמְכַּר אֶפְשָׁר לִפְדּוֹתוֹ – אֶחָד מֵאֶחָיו יִפְדֶּה אוֹתוֹ.
מט אוֹ־דֹד֞וֹ א֤וֹ בֶן־דֹּדוֹ֙ יִגְאָלֶ֔נּוּ אֽוֹ־מִשְּׁאֵ֧ר בְּשָׂר֛וֹ מִמִּשְׁפַּחְתּ֖וֹ יִגְאָלֶ֑נּוּ אֽוֹ־הִשִּׂ֥יגָה יָד֖וֹ וְנִגְאָֽל׃ אוֹ דּוֹדוֹ אוֹ בֶּן דּוֹדוֹ יִפְדֶּה אוֹתוֹ, אוֹ קָרוֹב אֵלָיו מִמִּשְׁפַּחְתּוֹ יִפְדֶּה אוֹתוֹ, אוֹ שֶׁיַּרְוִיחַ כֶּסֶף וְיִפְדֶּה אֶת עַצְמוֹ.
נ וְחִשַּׁב֙ עִם־קֹנֵ֔הוּ מִשְּׁנַת֙ הִמָּ֣כְרוֹ ל֔וֹ עַ֖ד שְׁנַ֣ת הַיֹּבֵ֑ל וְהָיָ֞ה כֶּ֤סֶף מִמְכָּרוֹ֙ בְּמִסְפַּ֣ר שָׁנִ֔ים כִּימֵ֥י שָׂכִ֖יר יִהְיֶ֥ה עִמּֽוֹ׃ וִיחַשֵּׁב עִם מִי שֶׁקָּנָה אוֹתוֹ אֶת הַזְּמָן מֵהַשָּׁנָה שֶׁנִּמְכַּר לוֹ עַד לִשְׁנַת הַיּוֹבֵל, וִיחַלֵּק אֶת סְכוּם הַכֶּסֶף שֶׁקִּבֵּל בְּמִסְפַּר הַשָּׁנִים, כְּמוֹ שֶׁמְּחַשְּׁבִים מְחִיר יְמֵי פּוֹעֵל שָׂכִיר כָּךְ הוּא יְחַשֵּׁב אִתּוֹ.
נא אִם־ע֥וֹד רַבּ֖וֹת בַּשָּׁנִ֑ים לְפִיהֶן֙ יָשִׁ֣יב גְּאֻלָּת֔וֹ מִכֶּ֖סֶף מִקְנָתֽוֹ׃ אִם נוֹתְרוּ עוֹד שָׁנִים רַבּוֹת, לְפִי זֶה הוּא יַחֲזִיר מְחִיר גָּבוֹהַּ לְפִדְיוֹנוֹ, לְפִי מְחִיר קְנִיָּתוֹ.
נב וְאִם־מְעַ֞ט נִשְׁאַ֧ר בַּשָּׁנִ֛ים עַד־שְׁנַ֥ת הַיֹּבֵ֖ל וְחִשַּׁב־ל֑וֹ כְּפִ֣י שָׁנָ֔יו יָשִׁ֖יב אֶת־גְּאֻלָּתֽוֹ׃ וְאִם נִשְׁאֲרוּ שָׁנִים מְעַטּוֹת עַד שְׁנַת הַיּוֹבֵל, כָּךְ יְחַשֵּׁב לוֹ, הוּא יַחֲזִיר אֶת הַכֶּסֶף לְפִדְיוֹנוֹ לְפִי מִסְפַּר שְׁנוֹתָיו.
נג כִּשְׂכִ֥יר שָׁנָ֛ה בְּשָׁנָ֖ה יִהְיֶ֣ה עִמּ֑וֹ לֹֽא־יִרְדֶּ֥נּֽוּ בְּפֶ֖רֶךְ לְעֵינֶֽיךָ׃ הוּא יִהְיֶה אִתּוֹ כָּל שָׁנָה כְּמוֹ שָׂכִיר לְשָׁנָה – לֹא יְשַׁעְבֵּד אוֹתוֹ בַּעֲבוֹדָה מְפָרֶכֶת לְעֵינֶיךָ.
נד וְאִם־לֹ֥א יִגָּאֵ֖ל בְּאֵ֑לֶּה וְיָצָא֙ בִּשְׁנַ֣ת הַיֹּבֵ֔ל ה֖וּא וּבָנָ֥יו עִמּֽוֹ׃ וְאִם לֹא יִפָּדֶה בְּאַחַת מֵהָאֶפְשָׁרוּיוֹת הָאֵלֶּה הוּא יֵצֵא בִּשְׁנַת הַיּוֹבֵל הוּא וּבָנָיו אִתּוֹ.

מפטיר[עריכה]

נה כִּֽי־לִ֤י בְנֵֽי־יִשְׂרָאֵל֙ עֲבָדִ֔ים עֲבָדַ֣י הֵ֔ם אֲשֶׁר־הוֹצֵ֥אתִי אוֹתָ֖ם מֵאֶ֣רֶץ מִצְרָ֑יִם אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיכֶֽם׃ כִּי בְּנֵי יִשְׂרָאֵל הֵם הָעֲבָדִים שֶׁלִּי – הֵם עֲבָדַי שֶׁהוֹצֵאתִי אוֹתָם מֵאֶרֶץ מִצְרַיִם, אֲנִי ה' אֱלֹהֵיכֶם.
כו א לֹֽא־תַעֲשׂ֨וּ לָכֶ֜ם אֱלִילִ֗ם וּפֶ֤סֶל וּמַצֵּבָה֙ לֹֽא־תָקִ֣ימוּ לָכֶ֔ם וְאֶ֣בֶן מַשְׂכִּ֗ית לֹ֤א תִתְּנוּ֙ בְּאַרְצְכֶ֔ם לְהִֽשְׁתַּחֲוֺ֖ת עָלֶ֑יהָ כִּ֛י אֲנִ֥י יְהֹוָ֖ה אֱלֹהֵיכֶֽם׃ אַל תַּעֲשׂוּ לָכֶם אֱלִילִים, וְאַל תָּקִימוּ לָכֶם פֶּסֶל וּמַצֵּבָה, וְאַל תָּשִׂימוּ בְּאַרְצְכֶם אֶבֶן מְקֻשֶּׁטֶת כְּדֵי לְהִשְׁתַּחֲווֹת עָלֶיהָ, כִּי אֲנִי ה' אֱלֹהֵיכֶם.
ב אֶת־שַׁבְּתֹתַ֣י תִּשְׁמֹ֔רוּ וּמִקְדָּשִׁ֖י תִּירָ֑אוּ אֲנִ֖י יְהֹוָֽה׃ תִּשְׁמְרוּ אֶת שַׁבְּתוֹתַי וּתְכַבְּדוּ אֶת מִקְדָּשִׁי, אֲנִי ה'.
