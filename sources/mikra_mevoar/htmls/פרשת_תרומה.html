ראשון[עריכה]

פ פָּרָשַׁת הַתְּרוּמָה
כה א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְיִקְחוּ־לִ֖י תְּרוּמָ֑ה מֵאֵ֤ת כׇּל־אִישׁ֙ אֲשֶׁ֣ר יִדְּבֶ֣נּוּ לִבּ֔וֹ תִּקְח֖וּ אֶת־תְּרוּמָתִֽי׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל שֶׁיִּקְחוּ לִי תְּרוּמָה, אֶת הַתְּרוּמָה עֲבוּרִי תִּקְחוּ מִכָּל אִישׁ שֶׁיִּתְנַדֵּב בְּטוּב לִבּוֹ.
ג וְזֹאת֙ הַתְּרוּמָ֔ה אֲשֶׁ֥ר תִּקְח֖וּ מֵאִתָּ֑ם זָהָ֥ב וָכֶ֖סֶף וּנְחֹֽשֶׁת׃ וְזֹאת הַתְּרוּמָה שֶׁתִּקְחוּ מֵאִתָּם – זָהָב וְכֶסֶף וּנְחוֹשֶׁת.
ד וּתְכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ וְעִזִּֽים׃ וְצֶמֶר צָבוּעַ בִּתְכֵלֶת וְאַרְגָּמָן מִנּוֹזֵל שֶׁל חִלָּזוֹן וּבְצֶבַע אָדוֹם מִתּוֹלַעַת, וּפִשְׁתָּן וּשְׂעַר עִזִּים.
ה וְעֹרֹ֨ת אֵילִ֧ם מְאׇדָּמִ֛ים וְעֹרֹ֥ת תְּחָשִׁ֖ים וַעֲצֵ֥י שִׁטִּֽים׃ וְעוֹרוֹת אֵילִים צְבוּעִים בְּאָדוֹם וְעוֹרוֹת שֶׁל הַחַיָּה תַּחַשׁ וַעֲצֵי שִׁטָּה.
ו שֶׁ֖מֶן לַמָּאֹ֑ר בְּשָׂמִים֙ לְשֶׁ֣מֶן הַמִּשְׁחָ֔ה וְלִקְטֹ֖רֶת הַסַּמִּֽים׃ שֶׁמֶן לַתְּאוּרָה בְּשָׂמִים לַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ וְלִקְטוֹרֶת הַבְּשָׂמִים.
ז אַבְנֵי־שֹׁ֕הַם וְאַבְנֵ֖י מִלֻּאִ֑ים לָאֵפֹ֖ד וְלַחֹֽשֶׁן׃ אַבְנֵי שׁוֹהַם וְאַבְנֵי חֵן לְשִׁבּוּץ, לָאֵפוֹד וְלַחוֹשֶׁן.
ח וְעָ֥שׂוּ לִ֖י מִקְדָּ֑שׁ וְשָׁכַנְתִּ֖י בְּתוֹכָֽם׃ וְיַעֲשׂוּ לִי מִקְדָּשׁ וְאֶשְׁכּוֹן בְּתוֹכָם.
ט כְּכֹ֗ל אֲשֶׁ֤ר אֲנִי֙ מַרְאֶ֣ה אוֹתְךָ֔ אֵ֚ת תַּבְנִ֣ית הַמִּשְׁכָּ֔ן וְאֵ֖ת תַּבְנִ֣ית כׇּל־כֵּלָ֑יו וְכֵ֖ן תַּעֲשֽׂוּ׃ כְּכָל מַה שֶּׁאֲנִי מַרְאֶה לְךָ אֶת מִבְנֵה הַמִּשְׁכָּן וְאֶת מִבְנֵה כָּל כֵּלָיו – כָּךְ תַּעֲשׂוּ.
ס פָּרָשַׁת הָאֲרוֹן
י וְעָשׂ֥וּ אֲר֖וֹן עֲצֵ֣י שִׁטִּ֑ים אַמָּתַ֨יִם וָחֵ֜צִי אׇרְכּ֗וֹ וְאַמָּ֤ה וָחֵ֙צִי֙ רׇחְבּ֔וֹ וְאַמָּ֥ה וָחֵ֖צִי קֹמָתֽוֹ׃ יַעֲשׂוּ אֲרוֹן מֵעֲצֵי שִׁטָּה, אָרְכּוֹ שְׁתַּיִם וָחֵצִי אַמּוֹת (כְּ120 ס"מ) רָחְבּוֹ אַמָּה וָחֵצִי (כְּ70 ס"מ) וְגָבְהוֹ אַמָּה וָחֵצִי.
יא וְצִפִּיתָ֤ אֹתוֹ֙ זָהָ֣ב טָה֔וֹר מִבַּ֥יִת וּמִח֖וּץ תְּצַפֶּ֑נּוּ וְעָשִׂ֧יתָ עָלָ֛יו זֵ֥ר זָהָ֖ב סָבִֽיב׃ וּתְצַפֶּה אוֹתוֹ בְּזָהָב טָהוֹר מִבִּפְנִים וּמִבַּחוּץ, וְתַעֲשֶׂה עָלָיו מִסָּבִיב זֵר מִזָּהָב.
יב וְיָצַ֣קְתָּ לּ֗וֹ אַרְבַּע֙ טַבְּעֹ֣ת זָהָ֔ב וְנָ֣תַתָּ֔ה עַ֖ל אַרְבַּ֣ע פַּעֲמֹתָ֑יו וּשְׁתֵּ֣י טַבָּעֹ֗ת עַל־צַלְעוֹ֙ הָֽאֶחָ֔ת וּשְׁתֵּי֙ טַבָּעֹ֔ת עַל־צַלְע֖וֹ הַשֵּׁנִֽית׃ וְתַתִּיךְ וְתִיצוֹק עֲבוּרוֹ אַרְבַּע טַבָּעוֹת מִזָּהָב, וְתָשִׂים עַל אַרְבַּע זָוִיּוֹתָיו – שְׁתֵּי טַבָּעוֹת עַל צִדּוֹ הָאֶחָד וּשְׁתֵּי טַבָּעוֹת עַל צִדּוֹ הַשֵּׁנִי.
יג וְעָשִׂ֥יתָ בַדֵּ֖י עֲצֵ֣י שִׁטִּ֑ים וְצִפִּיתָ֥ אֹתָ֖ם זָהָֽב׃ עֲשֵׂה מוֹטוֹת מֵעֲצֵי שִׁטָּה וּתְצַפֶּה אוֹתָם בְּזָהָב.
יד וְהֵֽבֵאתָ֤ אֶת־הַבַּדִּים֙ בַּטַּבָּעֹ֔ת עַ֖ל צַלְעֹ֣ת הָאָרֹ֑ן לָשֵׂ֥את אֶת־הָאָרֹ֖ן בָּהֶֽם׃ וְתַכְנִיס אֶת הַמּוֹטוֹת בַּטַּבָּעוֹת שֶׁעַל דְּפָנוֹת הָאָרוֹן כְּדֵי לְהָרִים בָּהֶם אֶת הָאָרוֹן.
טו בְּטַבְּעֹת֙ הָאָרֹ֔ן יִהְי֖וּ הַבַּדִּ֑ים לֹ֥א יָסֻ֖רוּ מִמֶּֽנּוּ׃ הַמּוֹטוֹת יִהְיוּ בְּטַבְּעוֹת הָאָרוֹן וְלֹא יָסוּרוּ מִמֶּנּוּ.
טז וְנָתַתָּ֖ אֶל־הָאָרֹ֑ן אֵ֚ת הָעֵדֻ֔ת אֲשֶׁ֥ר אֶתֵּ֖ן אֵלֶֽיךָ׃ וְתָשִׂים בָּאָרוֹן אֶת לוּחוֹת הַבְּרִית, שֶׁאֶתֵּן לְךָ.

שני[עריכה]

יז וְעָשִׂ֥יתָ כַפֹּ֖רֶת זָהָ֣ב טָה֑וֹר אַמָּתַ֤יִם וָחֵ֙צִי֙ אׇרְכָּ֔הּ וְאַמָּ֥ה וָחֵ֖צִי רׇחְבָּֽהּ׃ עֲשֵׂה כִּסּוּי מִזָּהָב טָהוֹר, אָרְכּוֹ שְׁתַּיִם וָחֵצִי אַמּוֹת וְרָחְבּוֹ אַמָּה וָחֵצִי.
יח וְעָשִׂ֛יתָ שְׁנַ֥יִם כְּרֻבִ֖ים זָהָ֑ב מִקְשָׁה֙ תַּעֲשֶׂ֣ה אֹתָ֔ם מִשְּׁנֵ֖י קְצ֥וֹת הַכַּפֹּֽרֶת׃ וְתַעֲשֶׂה שְׁתֵּי צוּרוֹת שֶׁל מַלְאָכִים עֲשׂוּיוֹת זָהָב, תַּעֲשֶׂה אוֹתָן מִגּוּשׁ אֶחָד, מִשְּׁנֵי קְצוֹת הַכִּסּוּי.
יט וַ֠עֲשֵׂ֠ה כְּר֨וּב אֶחָ֤ד מִקָּצָה֙ מִזֶּ֔ה וּכְרוּב־אֶחָ֥ד מִקָּצָ֖ה מִזֶּ֑ה מִן־הַכַּפֹּ֛רֶת תַּעֲשׂ֥וּ אֶת־הַכְּרֻבִ֖ים עַל־שְׁנֵ֥י קְצוֹתָֽיו׃ וַעֲשֵׂה צוּרַת מַלְאָךְ אַחַת בַּקָּצֶה הַזֶּה, וְצוּרַת מַלְאָךְ אַחַת בַּקָּצֶה הַשֵּׁנִי, אֶת צוּרוֹת הַמַּלְאָכִים תַּעֲשׂוּ מִגּוּף הַכִּסּוּי עַל שְׁנֵי קְצוֹתָיו.
כ וְהָי֣וּ הַכְּרֻבִים֩ פֹּרְשֵׂ֨י כְנָפַ֜יִם לְמַ֗עְלָה סֹכְכִ֤ים בְּכַנְפֵיהֶם֙ עַל־הַכַּפֹּ֔רֶת וּפְנֵיהֶ֖ם אִ֣ישׁ אֶל־אָחִ֑יו אֶ֨ל־הַכַּפֹּ֔רֶת יִהְי֖וּ פְּנֵ֥י הַכְּרֻבִֽים׃ וְיִהְיוּ צוּרוֹת הַמַּלְאָכִים פּוֹרְשׂוֹת כְּנָפַיִם לְמַעְלָה מְסַכְּכוֹת בְּכַנְפֵיהֶן עַל הַכִּסּוּי וּפְנֵיהֶן אַחַת לַשְּׁנִיָּה, פְּנֵי צוּרוֹת הַמַּלְאָכִים יִהְיוּ לְכִוּוּן מֶרְכַּז הַכִּסּוּי.
כא וְנָתַתָּ֧ אֶת־הַכַּפֹּ֛רֶת עַל־הָאָרֹ֖ן מִלְמָ֑עְלָה וְאֶל־הָ֣אָרֹ֔ן תִּתֵּן֙ אֶת־הָ֣עֵדֻ֔ת אֲשֶׁ֥ר אֶתֵּ֖ן אֵלֶֽיךָ׃ תָּשִׂים אֶת הַכִּסּוּי עַל הָאָרוֹן מִלְּמַעְלָה, וְתָשִׂים בָּאָרוֹן אֶת לוּחוֹת הַבְּרִית שֶׁאֶתֵּן לְךָ.
כב וְנוֹעַדְתִּ֣י לְךָ֮ שָׁם֒ וְדִבַּרְתִּ֨י אִתְּךָ֜ מֵעַ֣ל הַכַּפֹּ֗רֶת מִבֵּין֙ שְׁנֵ֣י הַכְּרֻבִ֔ים אֲשֶׁ֖ר עַל־אֲר֣וֹן הָעֵדֻ֑ת אֵ֣ת כׇּל־אֲשֶׁ֧ר אֲצַוֶּ֛ה אוֹתְךָ֖ אֶל־בְּנֵ֥י יִשְׂרָאֵֽל׃ וְאֶפָּגֵשׁ אִתְּךָ שָׁם, וַאֲדַבֵּר אִתְּךָ מֵעַל הַכִּסּוּי מִבֵּין שְׁתֵּי צוּרוֹת הַמַּלְאָכִים שֶׁעַל אֲרוֹן לוּחוֹת הַבְּרִית, אֶת כָּל מַה שֶּׁאֲצַוֶּה אוֹתְךָ אֶל בְּנֵי יִשְׂרָאֵל.
פ פָּרָשַׁת הַשֻּׁלְחָן
כג וְעָשִׂ֥יתָ שֻׁלְחָ֖ן עֲצֵ֣י שִׁטִּ֑ים אַמָּתַ֤יִם אׇרְכּוֹ֙ וְאַמָּ֣ה רׇחְבּ֔וֹ וְאַמָּ֥ה וָחֵ֖צִי קֹמָתֽוֹ׃ עֲשֵׂה שֻׁלְחָן מֵעֲצֵי שִׁטָּה, אָרְכּוֹ שְׁתֵּי אַמּוֹת רָחְבּוֹ אַמָּה וְגָבְהוֹ אַמָּה וָחֵצִי.
כד וְצִפִּיתָ֥ אֹת֖וֹ זָהָ֣ב טָה֑וֹר וְעָשִׂ֥יתָ לּ֛וֹ זֵ֥ר זָהָ֖ב סָבִֽיב׃ וּתְצַפֶּה אוֹתוֹ בְּזָהָב טָהוֹר וְתַעֲשֶׂה לוֹ זֵר זָהָב מִסָּבִיב.
כה וְעָשִׂ֨יתָ לּ֥וֹ מִסְגֶּ֛רֶת טֹ֖פַח סָבִ֑יב וְעָשִׂ֧יתָ זֵר־זָהָ֛ב לְמִסְגַּרְתּ֖וֹ סָבִֽיב׃ וְתַעֲשֶׂה לוֹ מִסְגֶּרֶת מִסָּבִיב בְּגוֹבַהּ טֶפַח, וְתַעֲשֶׂה זֵר מִזָּהָב לְמִסְגַּרְתּוֹ מִסָּבִיב.
כו וְעָשִׂ֣יתָ לּ֔וֹ אַרְבַּ֖ע טַבְּעֹ֣ת זָהָ֑ב וְנָתַתָּ֙ אֶת־הַטַּבָּעֹ֔ת עַ֚ל אַרְבַּ֣ע הַפֵּאֹ֔ת אֲשֶׁ֖ר לְאַרְבַּ֥ע רַגְלָֽיו׃ וְתַעֲשֶׂה לוֹ אַרְבַּע טַבְּעוֹת זָהָב, וְתָשִׂים אֶת הַטַּבָּעוֹת עַל אַרְבַּע הַפִּנּוֹת שֶׁלְּיַד אַרְבַּע רַגְלָיו.
כז לְעֻמַּת֙ הַמִּסְגֶּ֔רֶת תִּהְיֶ֖יןָ הַטַּבָּעֹ֑ת לְבָתִּ֣ים לְבַדִּ֔ים לָשֵׂ֖את אֶת־הַשֻּׁלְחָֽן׃ הַטַּבָּעוֹת תִּהְיֶינָה מוּל הַמִּסְגֶּרֶת, כְּבָתִּים לְמוֹטוֹת לְהָרִים אֶת הַשֻּׁלְחָן.
כח וְעָשִׂ֤יתָ אֶת־הַבַּדִּים֙ עֲצֵ֣י שִׁטִּ֔ים וְצִפִּיתָ֥ אֹתָ֖ם זָהָ֑ב וְנִשָּׂא־בָ֖ם אֶת־הַשֻּׁלְחָֽן׃ וְתַעֲשֶׂה אֶת הַמּוֹטוֹת מֵעֲצֵי שִׁטָּה וּתְצַפֶּה אוֹתָם בְּזָהָב, וְיָרִימוּ בָּהֶם אֶת הַשֻּׁלְחָן.
כט וְעָשִׂ֨יתָ קְּעָרֹתָ֜יו וְכַפֹּתָ֗יו וּקְשׂוֹתָיו֙ וּמְנַקִּיֹּתָ֔יו אֲשֶׁ֥ר יֻסַּ֖ךְ בָּהֵ֑ן זָהָ֥ב טָה֖וֹר תַּעֲשֶׂ֥ה אֹתָֽם׃ וְתַעֲשֶׂה אֶת קְעָרוֹתָיו וְכַפּוֹתָיו, וְאֶת הַמַּקְלוֹת שֶׁהַלֶּחֶם מוּנַח עֲלֵיהֶם וְהָעַמּוּדִים שֶׁתּוֹמְכִים מֵהַצְּדָדִים – שֶׁסּוֹכְכִים עַל הַשֻּׁלְחָן, מִזָּהָב טָהוֹר תַּעֲשֶׂה אוֹתָם.
ל וְנָתַתָּ֧ עַֽל־הַשֻּׁלְחָ֛ן לֶ֥חֶם פָּנִ֖ים לְפָנַ֥י תָּמִֽיד׃ וְתָשִׂים עַל הַשֻּׁלְחָן לֶחֶם מְכֻבָּד לִכְבוֹדִי תָּמִיד.
פ פָּרָשַׁת הַמְּנוֹרָה
לא וְעָשִׂ֥יתָ מְנֹרַ֖ת זָהָ֣ב טָה֑וֹר מִקְשָׁ֞ה תֵּעָשֶׂ֤ה‏[1] הַמְּנוֹרָה֙ יְרֵכָ֣הּ וְקָנָ֔הּ גְּבִיעֶ֛יהָ כַּפְתֹּרֶ֥יהָ וּפְרָחֶ֖יהָ מִמֶּ֥נָּה יִהְיֽוּ׃ עֲשֵׂה מְנוֹרָה מִזָּהָב טָהוֹר, הַמְּנוֹרָה – הַבָּסִיס שֶׁלָּהּ וְהַקָּנֶה הַמֶּרְכָּזִי שֶׁלָּהּ יֵעָשׂוּ מִגּוּשׁ אֶחָד, גְּבִיעֶיהָ כַּפְתּוֹרֶיהָ וּפְרָחֶיהָ יִהְיוּ כְּחֵלֶק מִגּוּפָהּ.
לב וְשִׁשָּׁ֣ה קָנִ֔ים יֹצְאִ֖ים מִצִּדֶּ֑יהָ שְׁלֹשָׁ֣ה ׀ קְנֵ֣י מְנֹרָ֗ה מִצִּדָּהּ֙ הָאֶחָ֔ד וּשְׁלֹשָׁה֙ קְנֵ֣י מְנֹרָ֔ה מִצִּדָּ֖הּ הַשֵּׁנִֽי׃ וְשִׁשָּׁה קָנִים יוֹצְאִים מִצִּדֶּיהָ, שְׁלוֹשָׁה קְנֵי מְנוֹרָה מִצַּד אֶחָד וּשְׁלוֹשָׁה קְנֵי מְנוֹרָה מִצַּד שֵׁנִי.
לג שְׁלֹשָׁ֣ה גְ֠בִעִ֠ים מְֽשֻׁקָּדִ֞ים בַּקָּנֶ֣ה הָאֶחָד֮ כַּפְתֹּ֣ר וָפֶ֒רַח֒ וּשְׁלֹשָׁ֣ה גְבִעִ֗ים מְשֻׁקָּדִ֛ים בַּקָּנֶ֥ה הָאֶחָ֖ד כַּפְתֹּ֣ר וָפָ֑רַח כֵּ֚ן לְשֵׁ֣שֶׁת הַקָּנִ֔ים הַיֹּצְאִ֖ים מִן־הַמְּנֹרָֽה׃ בְּקָנֶה אֶחָד יִהְיוּ שְׁלוֹשָׁה תַּבְלִיטִים שֶׁל גְּבִיעִים וְכַפְתּוֹר וּפֶרַח, וּבְקָנֶה אַחֵר יִהְיוּ שְׁלוֹשָׁה תַּבְלִיטִים שֶׁל גְּבִיעִים וְכַפְתּוֹר וּפֶרַח, כָּךְ יִהְיֶה בְּשֵׁשֶׁת הַקָּנִים שֶׁיּוֹצְאִים מֵהַקָּנֶה הַמֶּרְכָּזִי שֶׁל הַמְּנוֹרָה.
לד וּבַמְּנֹרָ֖ה אַרְבָּעָ֣ה גְבִעִ֑ים מְשֻׁ֨קָּדִ֔ים כַּפְתֹּרֶ֖יהָ וּפְרָחֶֽיהָ׃ וּבַקָּנֶה הַמֶּרְכָּזִי שֶׁל הַמְּנוֹרָה אַרְבָּעָה גְּבִיעִים, וְתַבְלִיטֵי כַּפְתּוֹרִים וּפְרָחִים.
לה וְכַפְתֹּ֡ר תַּ֩חַת֩ שְׁנֵ֨י הַקָּנִ֜ים מִמֶּ֗נָּה וְכַפְתֹּר֙ תַּ֣חַת שְׁנֵ֤י הַקָּנִים֙ מִמֶּ֔נָּה וְכַפְתֹּ֕ר תַּחַת־שְׁנֵ֥י הַקָּנִ֖ים מִמֶּ֑נָּה לְשֵׁ֙שֶׁת֙ הַקָּנִ֔ים הַיֹּצְאִ֖ים מִן־הַמְּנֹרָֽה׃ וְכַפְתּוֹר תַּחַת שְׁנֵי קָנִים שֶׁיּוֹצְאִים מִגּוּף הַמְּנוֹרָה וְכַפְתּוֹר תַּחַת שְׁנֵי קָנִים נוֹסָפִים שֶׁיּוֹצְאִים מִמֶּנָּה, וְכַפְתּוֹר תַּחַת שְׁנֵי קָנִים נוֹסָפִים שֶׁיּוֹצְאִים מִמֶּנָּה – כָּךְ לְשֵׁשֶׁת הַקָּנִים שֶׁיּוֹצְאִים מִגּוּף הַמְּנוֹרָה.
לו כַּפְתֹּרֵיהֶ֥ם וּקְנֹתָ֖ם מִמֶּ֣נָּה יִהְי֑וּ כֻּלָּ֛הּ מִקְשָׁ֥ה אַחַ֖ת זָהָ֥ב טָהֽוֹר׃ הַכַּפְתּוֹרִים וְהַקָּנִים יִהְיוּ חֵלֶק מִמֶּנָּה – כֻּלָּהּ עֲשׂוּיָה מִגּוּשׁ אֶחָד שֶׁל זָהָב טָהוֹר.
לז וְעָשִׂ֥יתָ אֶת־נֵרֹתֶ֖יהָ שִׁבְעָ֑ה וְהֶֽעֱלָה֙ אֶת־נֵ֣רֹתֶ֔יהָ וְהֵאִ֖יר עַל־עֵ֥בֶר פָּנֶֽיהָ׃ וְתַעֲשֶׂה אֶת נֵרוֹתֶיהָ שִׁבְעָה, וְיַדְלִיקוּ אֶת נֵרוֹתֶיהָ שֶׁיִּהְיוּ מְאִירִים אֶל מוּל פָּנֶיהָ.
לח וּמַלְקָחֶ֥יהָ וּמַחְתֹּתֶ֖יהָ זָהָ֥ב טָהֽוֹר׃ וְהַמֶּלְקָחַיִם וְהַיָּעִים שֶׁמְּנַקִּים אוֹתָהּ בָּהֶם יִהְיוּ מִזָּהָב טָהוֹר.
לט כִּכָּ֛ר זָהָ֥ב טָה֖וֹר יַעֲשֶׂ֣ה אֹתָ֑הּ אֵ֥ת כׇּל־הַכֵּלִ֖ים הָאֵֽלֶּה׃ מִכִּכָּר זָהָב טָהוֹר יַעֲשׂוּ אוֹתָהּ וְאֶת כָּל הַכֵּלִים הָאֵלֶּה.
מ וּרְאֵ֖ה וַעֲשֵׂ֑ה בְּתַ֨בְנִיתָ֔ם אֲשֶׁר־אַתָּ֥ה מׇרְאֶ֖ה בָּהָֽר׃ וּרְאֵה וַעֲשֵׂה בְּצוּרָתָם שֶׁהֶרְאוּ לְךָ בָּהָר.

שלישי[עריכה]

ס פָּרָשַׁת הַיְרִיעוֹת
כו א וְאֶת־הַמִּשְׁכָּ֥ן תַּעֲשֶׂ֖ה עֶ֣שֶׂר יְרִיעֹ֑ת שֵׁ֣שׁ מׇשְׁזָ֗ר וּתְכֵ֤לֶת וְאַרְגָּמָן֙ וְתֹלַ֣עַת שָׁנִ֔י כְּרֻבִ֛ים מַעֲשֵׂ֥ה חֹשֵׁ֖ב תַּעֲשֶׂ֥ה אֹתָֽם׃ וְאֶת כִּסּוּי הַמִּשְׁכָּן תַּעֲשֶׂה מֵעֶשֶׂר יְרִיעוֹת עֲשׂוּיוֹת מִפִּשְׁתָּן שָׁזוּר וּתְכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת, תַּעֲשֶׂה אוֹתָם בִּמְלֶאכֶת אֳמָנוּת שֶׁיֵּרָאוּ בָּהֶם צוּרוֹת מַלְאָכִים.
ב אֹ֣רֶךְ ׀ הַיְרִיעָ֣ה הָֽאַחַ֗ת שְׁמֹנֶ֤ה וְעֶשְׂרִים֙ בָּֽאַמָּ֔ה וְרֹ֙חַב֙ אַרְבַּ֣ע בָּאַמָּ֔ה הַיְרִיעָ֖ה הָאֶחָ֑ת מִדָּ֥ה אַחַ֖ת לְכׇל־הַיְרִיעֹֽת׃ אוֹרֶךְ הַיְּרִיעָה הָאַחַת עֶשְׂרִים וּשְׁמוֹנֶה אַמּוֹת וְרוֹחַב אַרְבַּע אַמּוֹת – מִדָּה אַחַת לְכָל הַיְּרִיעוֹת.
ג חֲמֵ֣שׁ הַיְרִיעֹ֗ת תִּֽהְיֶ֙יןָ֙ חֹֽבְרֹ֔ת אִשָּׁ֖ה אֶל־אֲחֹתָ֑הּ וְחָמֵ֤שׁ יְרִיעֹת֙ חֹֽבְרֹ֔ת אִשָּׁ֖ה אֶל־אֲחֹתָֽהּ׃ חֲמֵשׁ יְרִיעוֹת תִּהְיֶינָה מְחֻבָּרוֹת יְרִיעָה לַחֲבֶרְתָּהּ, וְחָמֵשׁ יְרִיעוֹת אֲחֵרוֹת תִּהְיֶינָה מְחֻבָּרוֹת יְרִיעָה לַחֲבֶרְתָּהּ.
ד וְעָשִׂ֜יתָ לֻֽלְאֹ֣ת תְּכֵ֗לֶת עַ֣ל שְׂפַ֤ת הַיְרִיעָה֙ הָאֶחָ֔ת מִקָּצָ֖ה בַּחֹבָ֑רֶת וְכֵ֤ן תַּעֲשֶׂה֙ בִּשְׂפַ֣ת הַיְרִיעָ֔ה הַקִּ֣יצוֹנָ֔ה בַּמַּחְבֶּ֖רֶת הַשֵּׁנִֽית׃ וְתַעֲשֶׂה לוּלָאוֹת מִתְּכֵלֶת עַל שְׂפַת הַיְּרִיעָה הָאַחַת בַּקָּצֶה בַּמָּקוֹם שֶׁל הַחִבּוּר, וְכָךְ תַּעֲשֶׂה גַּם בִּשְׂפַת הַיְּרִיעָה הַקִּיצוֹנָה בַּצַּד הַשֵּׁנִי שֶׁל הַחִבּוּר.
ה חֲמִשִּׁ֣ים לֻֽלָאֹ֗ת תַּעֲשֶׂה֮ בַּיְרִיעָ֣ה הָאֶחָת֒ וַחֲמִשִּׁ֣ים לֻֽלָאֹ֗ת תַּעֲשֶׂה֙ בִּקְצֵ֣ה הַיְרִיעָ֔ה אֲשֶׁ֖ר בַּמַּחְבֶּ֣רֶת הַשֵּׁנִ֑ית מַקְבִּילֹת֙ הַלֻּ֣לָאֹ֔ת אִשָּׁ֖ה אֶל־אֲחֹתָֽהּ׃ חֲמִשִּׁים לוּלָאוֹת תַּעֲשֶׂה בַּיְּרִיעָה הָאַחַת וַחֲמִשִּׁים לוּלָאוֹת תַּעֲשֶׂה בִּקְצֵה הַיְּרִיעָה שֶׁבַּצַּד הַשֵּׁנִי שֶׁל הַחִבּוּר, הַלּוּלָאוֹת תִּהְיֶינָה מַקְבִּילוֹת – לוּלָאָה מוּל חֲבֶרְתָּהּ.
ו וְעָשִׂ֕יתָ חֲמִשִּׁ֖ים קַרְסֵ֣י זָהָ֑ב וְחִבַּרְתָּ֨ אֶת־הַיְרִיעֹ֜ת אִשָּׁ֤ה אֶל־אֲחֹתָהּ֙ בַּקְּרָסִ֔ים וְהָיָ֥ה הַמִּשְׁכָּ֖ן אֶחָֽד׃ וְתַעֲשֶׂה חֲמִשִּׁים קַרְסֵי זָהָב, וּתְחַבֵּר אֶת הַיְּרִיעוֹת יְרִיעָה לַחֲבֶרְתָּהּ בַּקְּרָסִים וְכָךְ הַמִּשְׁכָּן יִהְיֶה רֶצֶף אֶחָד.
ז וְעָשִׂ֙יתָ֙ יְרִיעֹ֣ת עִזִּ֔ים לְאֹ֖הֶל עַל־הַמִּשְׁכָּ֑ן עַשְׁתֵּי־עֶשְׂרֵ֥ה יְרִיעֹ֖ת תַּעֲשֶׂ֥ה אֹתָֽם׃ וַעֲשֵׂה יְרִיעוֹת מִשְּׂעַר עִזִּים לְאוֹהֶל עַל הַמִּשְׁכָּן, אַחַת עֶשְׂרֵה יְרִיעוֹת תַּעֲשֶׂה אוֹתָם.
ח אֹ֣רֶךְ ׀ הַיְרִיעָ֣ה הָֽאַחַ֗ת שְׁלֹשִׁים֙ בָּֽאַמָּ֔ה וְרֹ֙חַב֙ אַרְבַּ֣ע בָּאַמָּ֔ה הַיְרִיעָ֖ה הָאֶחָ֑ת מִדָּ֣ה אַחַ֔ת לְעַשְׁתֵּ֥י עֶשְׂרֵ֖ה יְרִיעֹֽת׃ אוֹרֶךְ כָּל יְרִיעָה שְׁלוֹשִׁים אַמָּה וְרוֹחַב כָּל יְרִיעָה אַרְבַּע אַמּוֹת – מִדָּה אַחַת לְאַחַת עֶשְׂרֵה הַיְּרִיעוֹת.
ט וְחִבַּרְתָּ֞ אֶת־חֲמֵ֤שׁ הַיְרִיעֹת֙ לְבָ֔ד וְאֶת־שֵׁ֥שׁ הַיְרִיעֹ֖ת לְבָ֑ד וְכָפַלְתָּ֙ אֶת־הַיְרִיעָ֣ה הַשִּׁשִּׁ֔ית אֶל־מ֖וּל פְּנֵ֥י הָאֹֽהֶל׃ וּתְחַבֵּר חֲמֵשׁ יְרִיעוֹת לְבָד וְשֵׁשׁ יְרִיעוֹת לְבָד, וּתְקַפֵּל אֶת הַיְּרִיעָה הַשִּׁשִּׁית שֶׁתִּשְׁתַּלְשֵׁל מוּל פֶּתַח הָאוֹהֶל.
י וְעָשִׂ֜יתָ חֲמִשִּׁ֣ים לֻֽלָאֹ֗ת עַ֣ל שְׂפַ֤ת הַיְרִיעָה֙ הָֽאֶחָ֔ת הַקִּיצֹנָ֖ה בַּחֹבָ֑רֶת וַחֲמִשִּׁ֣ים לֻֽלָאֹ֗ת עַ֚ל שְׂפַ֣ת הַיְרִיעָ֔ה הַחֹבֶ֖רֶת הַשֵּׁנִֽית׃ וְתַעֲשֶׂה חֲמִשִּׁים לוּלָאוֹת עַל שְׂפַת הַיְּרִיעָה הַקִּיצוֹנָה בַּמָּקוֹם שֶׁל הַחִבּוּר, וַחֲמִשִּׁים לוּלָאוֹת עַל שְׂפַת הַיְּרִיעָה בַּצַּד הַשֵּׁנִי שֶׁל הַחִבּוּר.
יא וְעָשִׂ֛יתָ קַרְסֵ֥י נְחֹ֖שֶׁת חֲמִשִּׁ֑ים וְהֵבֵאתָ֤ אֶת־הַקְּרָסִים֙ בַּלֻּ֣לָאֹ֔ת וְחִבַּרְתָּ֥ אֶת־הָאֹ֖הֶל וְהָיָ֥ה אֶחָֽד׃ וְתַעֲשֶׂה חֲמִשִּׁים קַרְסֵי נְחוֹשֶׁת, וְתַעֲבִיר אֶת הַקְּרָסִים בַּלּוּלָאוֹת וּתְחַבֵּר אֶת הָאוֹהֶל וְהוּא יִהְיֶה רֶצֶף אֶחָד.
יב וְסֶ֙רַח֙ הָעֹדֵ֔ף בִּירִיעֹ֖ת הָאֹ֑הֶל חֲצִ֤י הַיְרִיעָה֙ הָעֹדֶ֔פֶת תִּסְרַ֕ח עַ֖ל אֲחֹרֵ֥י הַמִּשְׁכָּֽן׃ וּמַה שֶּׁעוֹדֵף בִּירִיעוֹת הָאוֹהֶל יָתֵר עַל יְרִיעוֹת הַמִּשְׁכָּן – אוֹתָהּ חֲצִי יְרִיעָה הָעוֹדֶפֶת תִּמָּשֵׁךְ עַל הַקַּרְקַע מֵאֲחוֹרֵי הַמִּשְׁכָּן.
יג וְהָאַמָּ֨ה מִזֶּ֜ה וְהָאַמָּ֤ה מִזֶּה֙ בָּעֹדֵ֔ף בְּאֹ֖רֶךְ יְרִיעֹ֣ת הָאֹ֑הֶל יִהְיֶ֨ה סָר֜וּחַ עַל־צִדֵּ֧י הַמִּשְׁכָּ֛ן מִזֶּ֥ה וּמִזֶּ֖ה לְכַסֹּתֽוֹ׃ וְהָאַמָּה מִכָּאן וְהָאַמָּה מִכָּאן בָּעוֹדֵף בְּאוֹרֶךְ יְרִיעוֹת הָאוֹהֶל – שֶׁהֵן אֲרוּכּוֹת מִיְּרִיעוֹת הַמִּשְׁכָּן, יִשְׁתַּלְשְׁלוּ עַל צִדֵּי הַמִּשְׁכָּן מִכָּאן וּמִכָּאן לְכַסּוֹתוֹ.
יד וְעָשִׂ֤יתָ מִכְסֶה֙ לָאֹ֔הֶל עֹרֹ֥ת אֵילִ֖ם מְאׇדָּמִ֑ים וּמִכְסֵ֛ה עֹרֹ֥ת תְּחָשִׁ֖ים מִלְמָֽעְלָה׃ וְתַעֲשֶׂה כִּסּוּי לָאוֹהֶל מֵעוֹרוֹת אֵילִים צְבוּעִים בְּאָדוֹם וְכִסּוּי מֵעוֹרוֹת הַחַיָּה תַּחַשׁ מִלְּמַעְלָה.

רביעי[עריכה]

פ פָּרָשַׁת הַקְּרָשִׁים
טו וְעָשִׂ֥יתָ אֶת־הַקְּרָשִׁ֖ים לַמִּשְׁכָּ֑ן עֲצֵ֥י שִׁטִּ֖ים עֹמְדִֽים׃ עֲשֵׂה קְרָשִׁים עוֹמְדִים לַמִּשְׁכָּן מֵעֲצֵי שִׁטָּה.
טז עֶ֥שֶׂר אַמּ֖וֹת אֹ֣רֶךְ הַקָּ֑רֶשׁ וְאַמָּה֙ וַחֲצִ֣י הָֽאַמָּ֔ה רֹ֖חַב הַקֶּ֥רֶשׁ הָאֶחָֽד׃ עֶשֶׂר אַמּוֹת אוֹרֶךְ הַקֶּרֶשׁ, וְאַמָּה וָחֵצִי רוֹחַב כָּל קֶרֶשׁ.
יז שְׁתֵּ֣י יָד֗וֹת לַקֶּ֙רֶשׁ֙ הָאֶחָ֔ד מְשֻׁ֨לָּבֹ֔ת אִשָּׁ֖ה אֶל־אֲחֹתָ֑הּ כֵּ֣ן תַּעֲשֶׂ֔ה לְכֹ֖ל קַרְשֵׁ֥י הַמִּשְׁכָּֽן׃ וְתַעֲשֵׂה שְׁתֵּי יָדִיּוֹת בּוֹלְטוֹת לְכָל קֶרֶשׁ, מְכֻוָּנוֹת אַחַת מוּל אַחַת, כָּךְ תַּעֲשֶׂה לְכָל קַרְשֵׁי הַמִּשְׁכָּן.
יח וְעָשִׂ֥יתָ אֶת־הַקְּרָשִׁ֖ים לַמִּשְׁכָּ֑ן עֶשְׂרִ֣ים קֶ֔רֶשׁ לִפְאַ֖ת נֶ֥גְבָּה תֵימָֽנָה׃ וְתַעֲשֶׂה אֶת הַקְּרָשִׁים לַמִּשְׁכָּן עֶשְׂרִים קְרָשִׁים לְצַד הַנֶּגֶב – לַדָּרוֹם.
יט וְאַרְבָּעִים֙ אַדְנֵי־כֶ֔סֶף תַּעֲשֶׂ֕ה תַּ֖חַת עֶשְׂרִ֣ים הַקָּ֑רֶשׁ שְׁנֵ֨י אֲדָנִ֜ים תַּֽחַת־הַקֶּ֤רֶשׁ הָאֶחָד֙ לִשְׁתֵּ֣י יְדֹתָ֔יו וּשְׁנֵ֧י אֲדָנִ֛ים תַּֽחַת־הַקֶּ֥רֶשׁ הָאֶחָ֖ד לִשְׁתֵּ֥י יְדֹתָֽיו׃ וְאַרְבָּעִים בְּסִיסִים מִכֶּסֶף תַּעֲשֶׂה מִתַּחַת לְעֶשְׂרִים הַקְּרָשִׁים, שְׁנֵי בְּסִיסִים מִתַּחַת לְכָל קֶרֶשׁ – לִשְׁתֵּי יָדִיּוֹתָיו.
כ וּלְצֶ֧לַע הַמִּשְׁכָּ֛ן הַשֵּׁנִ֖ית לִפְאַ֣ת צָפ֑וֹן עֶשְׂרִ֖ים קָֽרֶשׁ׃ וְלַדּוֹפֶן הַשְּׁנִיָּה שֶׁל הַמִּשְׁכָּן – לְצַד צָפוֹן, תַּעֲשֶׂה עֶשְׂרִים קְרָשִׁים.
כא וְאַרְבָּעִ֥ים אַדְנֵיהֶ֖ם כָּ֑סֶף שְׁנֵ֣י אֲדָנִ֗ים תַּ֚חַת הַקֶּ֣רֶשׁ הָֽאֶחָ֔ד וּשְׁנֵ֣י אֲדָנִ֔ים תַּ֖חַת הַקֶּ֥רֶשׁ הָאֶחָֽד׃ וְאַרְבָּעִים בְּסִיסֵיהֶם מִכֶּסֶף – שְׁנֵי בְּסִיסִים מִתַּחַת לְכָל קֶרֶשׁ.
כב וּֽלְיַרְכְּתֵ֥י הַמִּשְׁכָּ֖ן יָ֑מָּה תַּעֲשֶׂ֖ה שִׁשָּׁ֥ה קְרָשִֽׁים׃ וּבִקְצֵה הַמִּשְׁכָּן לְצַד מַעֲרָב תַּעֲשֶׂה שִׁשָּׁה קְרָשִׁים.
כג וּשְׁנֵ֤י קְרָשִׁים֙ תַּעֲשֶׂ֔ה לִמְקֻצְעֹ֖ת הַמִּשְׁכָּ֑ן בַּיַּרְכָתָֽיִם׃ וּשְׁנֵי קְרָשִׁים תַּעֲשֶׂה לְפִנּוֹת הַמִּשְׁכָּן שֶׁבַּקָּצֶה.
כד וְיִֽהְי֣וּ תֹֽאֲמִם֮ מִלְּמַ֒טָּה֒ וְיַחְדָּ֗ו יִהְי֤וּ תַמִּים֙ עַל־רֹאשׁ֔וֹ אֶל־הַטַּבַּ֖עַת הָאֶחָ֑ת כֵּ֚ן יִהְיֶ֣ה לִשְׁנֵיהֶ֔ם לִשְׁנֵ֥י הַמִּקְצֹעֹ֖ת יִהְיֽוּ׃ וְהַקְּרָשִׁים יִהְיוּ מַקְבִּילִים מִלְּמַטָּה, וְיִהְיוּ מֻצְמָדִים יַחַד עַל רָאשֵׁיהֶם בְּטַבַּעַת אַחַת, וְכָךְ יִהְיֶה גַּם לִשְׁנֵי הַקְּרָשִׁים שֶׁבַּפִּנּוֹת.
כה וְהָיוּ֙ שְׁמֹנָ֣ה קְרָשִׁ֔ים וְאַדְנֵיהֶ֣ם כֶּ֔סֶף שִׁשָּׁ֥ה עָשָׂ֖ר אֲדָנִ֑ים שְׁנֵ֣י אֲדָנִ֗ים תַּ֚חַת הַקֶּ֣רֶשׁ הָאֶחָ֔ד וּשְׁנֵ֣י אֲדָנִ֔ים תַּ֖חַת הַקֶּ֥רֶשׁ הָאֶחָֽד׃ יַחַד הֵם יִהְיוּ שְׁמוֹנָה קְרָשִׁים בַּמַּעֲרָב, וּבְסִיסֵיהֶם מִכֶּסֶף – שִׁשָּׁה עָשָׂר בְּסִיסִים, שְׁנֵי בְּסִיסִים מִתַּחַת לְקֶרֶשׁ אֶחָד וּשְׁנֵי בְּסִיסִים מִתַּחַת לְקֶרֶשׁ אַחֵר.
כו וְעָשִׂ֥יתָ בְרִיחִ֖ם עֲצֵ֣י שִׁטִּ֑ים חֲמִשָּׁ֕ה לְקַרְשֵׁ֥י צֶֽלַע־הַמִּשְׁכָּ֖ן הָאֶחָֽד׃ וְתַעֲשֶׂה חֲמִשָּׁה בְּרִיחִים מֵעֲצֵי שִׁטָּה לְקַרְשֵׁי דּוֹפֶן הַמִּשְׁכָּן הָאַחַת.
כז וַחֲמִשָּׁ֣ה בְרִיחִ֔ם לְקַרְשֵׁ֥י צֶֽלַע־הַמִּשְׁכָּ֖ן הַשֵּׁנִ֑ית וַחֲמִשָּׁ֣ה בְרִיחִ֗ם לְקַרְשֵׁי֙ צֶ֣לַע הַמִּשְׁכָּ֔ן לַיַּרְכָתַ֖יִם יָֽמָּה׃ וַחֲמִשָּׁה בְּרִיחִים לְקַרְשֵׁי דּוֹפֶן הַמִּשְׁכָּן הַשְּׁנִיָּה, וַחֲמִשָּׁה בְּרִיחִים לְקַרְשֵׁי דּוֹפֶן הַמִּשְׁכָּן לַאֲחוֹרָיו בַּמַּעֲרָב.
כח וְהַבְּרִ֥יחַ הַתִּיכֹ֖ן בְּת֣וֹךְ הַקְּרָשִׁ֑ים מַבְרִ֕חַ מִן־הַקָּצֶ֖ה אֶל־הַקָּצֶֽה׃ וְהַבָּרִיחַ הָאֶמְצָעִי בְּתוֹךְ הַקְּרָשִׁים עוֹבֵר מֵהַקָּצֶה אֶל הַקָּצֶה.
כט וְֽאֶת־הַקְּרָשִׁ֞ים תְּצַפֶּ֣ה זָהָ֗ב וְאֶת־טַבְּעֹֽתֵיהֶם֙ תַּעֲשֶׂ֣ה זָהָ֔ב בָּתִּ֖ים לַבְּרִיחִ֑ם וְצִפִּיתָ֥ אֶת־הַבְּרִיחִ֖ם זָהָֽב׃ וְאֶת הַקְּרָשִׁים תְּצַפֶּה בְּזָהָב וְאֶת טַבְּעוֹתֵיהֶם תַּעֲשֶׂה מִזָּהָב כְּבָתִּים לַבְּרִיחִים, וּתְצַפֶּה אֶת הַבְּרִיחִים בְּזָהָב.
ל וַהֲקֵמֹתָ֖ אֶת־הַמִּשְׁכָּ֑ן כְּמִ֨שְׁפָּט֔וֹ אֲשֶׁ֥ר הׇרְאֵ֖יתָ בָּהָֽר׃ תָּקִים אֶת הַמִּשְׁכָּן כְּהִלְכָתוֹ, כְּמוֹ שֶׁהֶרְאֵיתִי לְךָ בָּהָר.

חמישי[עריכה]

ס פָּרָשַׁת הַפָּרוֹכֶת
לא וְעָשִׂ֣יתָ פָרֹ֗כֶת תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מׇשְׁזָ֑ר מַעֲשֵׂ֥ה חֹשֵׁ֛ב יַעֲשֶׂ֥ה אֹתָ֖הּ כְּרֻבִֽים׃ תַּעֲשֶׂה פָּרוֹכֶת מִתְּכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת וּפִשְׁתָּן שָׁזוּר, יַעֲשׂוּ אוֹתָהּ בִּמְלֶאכֶת אֳמָנוּת, שֶׁיֵּרָאוּ בָּהּ צוּרוֹת מַלְאָכִים.
לב וְנָתַתָּ֣ה אֹתָ֗הּ עַל־אַרְבָּעָה֙ עַמּוּדֵ֣י שִׁטִּ֔ים מְצֻפִּ֣ים זָהָ֔ב וָוֵיהֶ֖ם זָהָ֑ב עַל־אַרְבָּעָ֖ה אַדְנֵי־כָֽסֶף׃ וְתָשִׂים אוֹתָהּ עַל אַרְבָּעָה עַמּוּדִים מֵעֲצֵי שִׁטָּה מְצֻפִּים זָהָב וְגַם וָוֵיהֶם מִזָּהָב, עַל אַרְבָּעָה בְּסִיסֵי כֶּסֶף.
לג וְנָתַתָּ֣ה אֶת־הַפָּרֹ֘כֶת֮ תַּ֣חַת הַקְּרָסִים֒ וְהֵבֵאתָ֥ שָׁ֙מָּה֙ מִבֵּ֣ית לַפָּרֹ֔כֶת אֵ֖ת אֲר֣וֹן הָעֵד֑וּת וְהִבְדִּילָ֤ה הַפָּרֹ֙כֶת֙ לָכֶ֔ם בֵּ֣ין הַקֹּ֔דֶשׁ וּבֵ֖ין קֹ֥דֶשׁ הַקֳּדָשִֽׁים׃ וְתָשִׂים אֶת הַפָּרוֹכֶת מִתַּחַת לַקְּרָסִים וְתָבִיא לְשָׁם – פְּנִימָה מֵעֵבֶר לַפָּרוֹכֶת, אֶת אֲרוֹן לוּחוֹת הַבְּרִית, וְהַפָּרוֹכֶת תַּבְדִּיל לָכֶם בֵּין הַקּוֹדֶשׁ וּבֵין קוֹדֶשׁ הַקֳּדָשִׁים.
לד וְנָתַתָּ֙ אֶת־הַכַּפֹּ֔רֶת עַ֖ל אֲר֣וֹן הָעֵדֻ֑ת בְּקֹ֖דֶשׁ הַקֳּדָשִֽׁים׃ וְתָשִׂים אֶת הַכִּסּוּי עַל אֲרוֹן לוּחוֹת הַבְּרִית, בְּקוֹדֶשׁ הַקֳּדָשִׁים.
לה וְשַׂמְתָּ֤ אֶת־הַשֻּׁלְחָן֙ מִח֣וּץ לַפָּרֹ֔כֶת וְאֶת־הַמְּנֹרָה֙ נֹ֣כַח הַשֻּׁלְחָ֔ן עַ֛ל צֶ֥לַע הַמִּשְׁכָּ֖ן תֵּימָ֑נָה וְהַ֨שֻּׁלְחָ֔ן תִּתֵּ֖ן עַל־צֶ֥לַע צָפֽוֹן׃ וְתָשִׂים אֶת הַשֻּׁלְחָן מֵעֵבֶר לַפָּרוֹכֶת מִבַּחוּץ, וְאֶת הַמְּנוֹרָה מוּל הַשֻּׁלְחָן עַל יַד דּוֹפֶן הַמִּשְׁכָּן הַדְּרוֹמִית, וְאֶת הַשֻּׁלְחָן תָּשִׂים עַל יַד הַדּוֹפֶן הַצְּפוֹנִית.
לו וְעָשִׂ֤יתָ מָסָךְ֙ לְפֶ֣תַח הָאֹ֔הֶל תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מׇשְׁזָ֑ר מַעֲשֵׂ֖ה רֹקֵֽם׃ וְתַעֲשֶׂה מָסָךְ לְפֶתַח הָאוֹהֶל מִתְּכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת וּפִשְׁתָּן שָׁזוּר, בִּמְלֶאכֶת רִקְמָה.
לז וְעָשִׂ֣יתָ לַמָּסָ֗ךְ חֲמִשָּׁה֙ עַמּוּדֵ֣י שִׁטִּ֔ים וְצִפִּיתָ֤ אֹתָם֙ זָהָ֔ב וָוֵיהֶ֖ם זָהָ֑ב וְיָצַקְתָּ֣ לָהֶ֔ם חֲמִשָּׁ֖ה אַדְנֵ֥י נְחֹֽשֶׁת׃ וְתַעֲשֶׂה לַמָּסָךְ חֲמִשָּׁה עַמּוּדֵי שִׁטָּה וּתְצַפֶּה אוֹתָם בְּזָהָב, וָוֵיהֶם זָהָב, וְתַתִּיךְ וְתִיצוֹק בִּשְׁבִילָם חֲמִשָּׁה בְּסִיסֵי נְחוֹשֶׁת.

ששי[עריכה]

ס פָּרָשַׁת הַמִּזְבֵּחַ
כז א וְעָשִׂ֥יתָ אֶת־הַמִּזְבֵּ֖חַ עֲצֵ֣י שִׁטִּ֑ים חָמֵשׁ֩ אַמּ֨וֹת אֹ֜רֶךְ וְחָמֵ֧שׁ אַמּ֣וֹת רֹ֗חַב רָב֤וּעַ יִהְיֶה֙ הַמִּזְבֵּ֔חַ וְשָׁלֹ֥שׁ אַמּ֖וֹת קֹמָתֽוֹ׃ וְתַעֲשֶׂה אֶת הַמִּזְבֵּחַ מֵעֲצֵי שִׁטָּה, הַמִּזְבֵּחַ יִהְיֶה בְּצוּרַת רִבּוּעַ – חָמֵשׁ אַמּוֹת אוֹרֶךְ וְחָמֵשׁ אַמּוֹת רוֹחַב, וְשָׁלוֹשׁ אַמּוֹת גָּבְהוֹ.
ב וְעָשִׂ֣יתָ קַרְנֹתָ֗יו עַ֚ל אַרְבַּ֣ע פִּנֹּתָ֔יו מִמֶּ֖נּוּ תִּהְיֶ֣יןָ קַרְנֹתָ֑יו וְצִפִּיתָ֥ אֹת֖וֹ נְחֹֽשֶׁת׃ וְתַעֲשֶׂה אֶת קַרְנוֹתָיו עַל אַרְבַּע פִּנּוֹתָיו, קַרְנוֹתָיו תִּהְיֶינָה חֵלֶק מִמֶּנּוּ, וּתְצַפֶּה אוֹתוֹ בִּנְחוֹשֶׁת.
ג וְעָשִׂ֤יתָ סִּֽירֹתָיו֙ לְדַשְּׁנ֔וֹ וְיָעָיו֙ וּמִזְרְקֹתָ֔יו וּמִזְלְגֹתָ֖יו וּמַחְתֹּתָ֑יו לְכׇל־כֵּלָ֖יו תַּעֲשֶׂ֥ה נְחֹֽשֶׁת׃ וְתַעֲשֶׂה אֶת הַסִּירִים שֶׁמּוֹצִיאִים בָּהֶם אֶת הָאֵפֶר, וְאֶת הַמַּגְרְפוֹת לְנִקּוּי הָאֵפֶר, וְאֶת הַכּוֹסוֹת לְקַבָּלַת הַדָּם וּזְרִיקָתוֹ, וְאֶת הַקִּלְּשׁוֹנִים לַהֲפִיכַת הַבָּשָׂר, וְהַיָּעִים לִלְקִיחַת הַגֶּחָלִים – אֶת כָּל כְּלֵי הַמִּזְבֵּחַ תַּעֲשֶׂה מִנְּחוֹשֶׁת.
ד וְעָשִׂ֤יתָ לּוֹ֙ מִכְבָּ֔ר מַעֲשֵׂ֖ה רֶ֣שֶׁת נְחֹ֑שֶׁת וְעָשִׂ֣יתָ עַל־הָרֶ֗שֶׁת אַרְבַּע֙ טַבְּעֹ֣ת נְחֹ֔שֶׁת עַ֖ל אַרְבַּ֥ע קְצוֹתָֽיו׃ וְתַעֲשֶׂה לוֹ קִישׁוּט בְּצוּרַת רֶשֶׁת מִנְּחוֹשֶׁת, וְתַעֲשֶׂה עַל הָרֶשֶׁת אַרְבַּע טַבְּעוֹת נְחוֹשֶׁת עַל אַרְבַּע קְצוֹתָיו.
ה וְנָתַתָּ֣ה אֹתָ֗הּ תַּ֛חַת כַּרְכֹּ֥ב הַמִּזְבֵּ֖חַ מִלְּמָ֑טָּה וְהָיְתָ֣ה הָרֶ֔שֶׁת עַ֖ד חֲצִ֥י הַמִּזְבֵּֽחַ׃ וְתָשִׂים אוֹתָהּ מִתַּחַת לַבְּלִיטָה שֶׁסְּבִיב הַמִּזְבֵּחַ, מִלְּמַטָּה, וְהָרֶשֶׁת תִּהְיֶה עַד חֲצִי הַמִּזְבֵּחַ.
ו וְעָשִׂ֤יתָ בַדִּים֙ לַמִּזְבֵּ֔חַ בַּדֵּ֖י עֲצֵ֣י שִׁטִּ֑ים וְצִפִּיתָ֥ אֹתָ֖ם נְחֹֽשֶׁת׃ וְתַעֲשֶׂה מוֹטוֹת לַמִּזְבֵּחַ – מוֹטוֹת מֵעֲצֵי שִׁטָּה, וּתְצַפֶּה אוֹתָם בִּנְחוֹשֶׁת.
ז וְהוּבָ֥א אֶת־בַּדָּ֖יו בַּטַּבָּעֹ֑ת וְהָי֣וּ הַבַּדִּ֗ים עַל־שְׁתֵּ֛י צַלְעֹ֥ת הַמִּזְבֵּ֖חַ בִּשְׂאֵ֥ת אֹתֽוֹ׃ הַמּוֹטוֹת יוּכְנְסוּ בַּטַּבָּעוֹת – הַמּוֹטוֹת יִהְיוּ עַל שְׁתֵּי דַּפְנוֹת הַמִּזְבֵּחַ כְּשֶׁיָּרִימוּ אוֹתוֹ.
ח נְב֥וּב לֻחֹ֖ת תַּעֲשֶׂ֣ה אֹת֑וֹ כַּאֲשֶׁ֨ר הֶרְאָ֥ה אֹתְךָ֛ בָּהָ֖ר כֵּ֥ן יַעֲשֽׂוּ׃ עֲשֵׂה אוֹתוֹ חָלוּל מֻקָּף בְּלוּחוֹת, כְּמוֹ שֶׁה' הֶרְאָה לְךָ בָּהָר כָּךְ יַעֲשׂוּ.

שביעי[עריכה]

ס פָּרָשַׁת הֶחָצֵר
ט וְעָשִׂ֕יתָ אֵ֖ת חֲצַ֣ר הַמִּשְׁכָּ֑ן לִפְאַ֣ת נֶֽגֶב־תֵּ֠ימָ֠נָה קְלָעִ֨ים לֶחָצֵ֜ר שֵׁ֣שׁ מׇשְׁזָ֗ר מֵאָ֤ה בָֽאַמָּה֙ אֹ֔רֶךְ לַפֵּאָ֖ה הָאֶחָֽת׃ עֲשֵׂה אֶת חֲצַר הַמִּשְׁכָּן, לְצַד נֶגֶב לְדָרוֹם תַּעֲשֶׂה לֶחָצֵר גָּדֵר קְלוּעָה מִפִּשְׁתָּן שָׁזוּר, מֵאָה אַמָּה אוֹרֶךְ צַד אֶחָד.
י וְעַמֻּדָ֣יו עֶשְׂרִ֔ים וְאַדְנֵיהֶ֥ם עֶשְׂרִ֖ים נְחֹ֑שֶׁת וָוֵ֧י הָעַמֻּדִ֛ים וַחֲשֻׁקֵיהֶ֖ם כָּֽסֶף׃ וְעַמּוּדָיו עֶשְׂרִים וּבְסִיסֵיהֶם עֶשְׂרִים מִנְּחוֹשֶׁת, וָוֵי הָעַמּוּדִים וְהַטַּבָּעוֹת הַמַּקִּיפוֹת אוֹתָם לְקִשּׁוּט, מִכֶּסֶף.
יא וְכֵ֨ן לִפְאַ֤ת צָפוֹן֙ בָּאֹ֔רֶךְ קְלָעִ֖ים מֵ֣אָה אֹ֑רֶךְ וְעַמֻּדָ֣ו עֶשְׂרִ֗ים וְאַדְנֵיהֶ֤ם עֶשְׂרִים֙ נְחֹ֔שֶׁת וָוֵ֧י הָֽעַמֻּדִ֛ים וַחֲשֻׁקֵיהֶ֖ם כָּֽסֶף׃ וְכֵן לְאוֹרֶךְ צַד צָפוֹן, תַּעֲשֶׂה גָּדֵר קְלוּעָה בְּאוֹרֶךְ מֵאָה אַמּוֹת, וְעַמּוּדָיו עֶשְׂרִים וּבְסִיסֵיהֶם עֶשְׂרִים מִנְּחוֹשֶׁת, וָוֵי הָעַמּוּדִים וְהַטַּבָּעוֹת הַמַּקִּיפוֹת אוֹתָם לְקִשּׁוּט, מִכֶּסֶף.
יב וְרֹ֤חַב הֶֽחָצֵר֙ לִפְאַת־יָ֔ם קְלָעִ֖ים חֲמִשִּׁ֣ים אַמָּ֑ה עַמֻּדֵיהֶ֣ם עֲשָׂרָ֔ה וְאַדְנֵיהֶ֖ם עֲשָׂרָֽה׃ וְרוֹחַב הֶחָצֵר לְצַד מַעֲרָב, גָּדֵר קְלוּעָה בְּאוֹרֶךְ חֲמִשִּׁים אַמָּה, עַמּוּדֵיהֶם עֲשָׂרָה וּבְסִיסֵיהֶם עֲשָׂרָה.
יג וְרֹ֣חַב הֶֽחָצֵ֗ר לִפְאַ֛ת קֵ֥דְמָה מִזְרָ֖חָה חֲמִשִּׁ֥ים אַמָּֽה׃ וְרוֹחַב הֶחָצֵר לְצַד מִזְרָח, חֲמִשִּׁים אַמָּה.
יד וַחֲמֵ֨שׁ עֶשְׂרֵ֥ה אַמָּ֛ה קְלָעִ֖ים לַכָּתֵ֑ף עַמֻּדֵיהֶ֣ם שְׁלֹשָׁ֔ה וְאַדְנֵיהֶ֖ם שְׁלֹשָֽׁה׃ וַחֲמֵשׁ עֶשְׂרֵה אַמָּה גָּדֵר קְלוּעָה לְצַד-הַפֶּתַח, עַמּוּדֵיהֶם שְׁלוֹשָׁה וּבְסִיסֵיהֶם שְׁלוֹשָׁה.
טו וְלַכָּתֵף֙ הַשֵּׁנִ֔ית חֲמֵ֥שׁ עֶשְׂרֵ֖ה קְלָעִ֑ים עַמֻּדֵיהֶ֣ם שְׁלֹשָׁ֔ה וְאַדְנֵיהֶ֖ם שְׁלֹשָֽׁה׃ וּלְצַד-הַפֶּתַח הַשֵּׁנִי חֲמֵשׁ עֶשְׂרֵה אַמּוֹת גָּדֵר קְלוּעָה, עַמּוּדֵיהֶם שְׁלוֹשָׁה וּבְסִיסֵיהֶם שְׁלוֹשָׁה.
טז וּלְשַׁ֨עַר הֶֽחָצֵ֜ר מָסָ֣ךְ ׀ עֶשְׂרִ֣ים אַמָּ֗ה תְּכֵ֨לֶת וְאַרְגָּמָ֜ן וְתוֹלַ֧עַת שָׁנִ֛י וְשֵׁ֥שׁ מׇשְׁזָ֖ר מַעֲשֵׂ֣ה רֹקֵ֑ם עַמֻּֽדֵיהֶם֙ אַרְבָּעָ֔ה וְאַדְנֵיהֶ֖ם אַרְבָּעָֽה׃ וּלְשַׁעַר הֶחָצֵר מָסָךְ עֶשְׂרִים אַמָּה עָשׂוּי מִתְּכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת וּפִשְׁתָּן שָׁזוּר, בִּמְלֶאכֶת רִקְמָה. עַמּוּדֵי הַשַּׁעַר וְהַמָּסָךְ אַרְבָּעָה, וּבְסִיסֵיהֶם אַרְבָּעָה.

מפטיר[עריכה]

יז כׇּל־עַמּוּדֵ֨י הֶֽחָצֵ֤ר סָבִיב֙ מְחֻשָּׁקִ֣ים כֶּ֔סֶף וָוֵיהֶ֖ם כָּ֑סֶף וְאַדְנֵיהֶ֖ם נְחֹֽשֶׁת׃ כָּל עַמּוּדֵי הֶחָצֵר מִסָּבִיב מֻקָּפִים בְּטַבְּעוֹת כֶּסֶף וְגַם וָוֵיהֶם מִכֶּסֶף, וּבְסִיסֵיהֶם מִנְּחוֹשֶׁת.
יח אֹ֣רֶךְ הֶֽחָצֵר֩ מֵאָ֨ה בָֽאַמָּ֜ה וְרֹ֣חַב ׀ חֲמִשִּׁ֣ים בַּחֲמִשִּׁ֗ים וְקֹמָ֛ה חָמֵ֥שׁ אַמּ֖וֹת שֵׁ֣שׁ מׇשְׁזָ֑ר וְאַדְנֵיהֶ֖ם נְחֹֽשֶׁת׃ אוֹרֶךְ הֶחָצֵר מֵאָה אַמָּה, וּלְכָל אָרְכָּהּ – רוֹחַב חֲמִשִּׁים אַמָּה, וְגוֹבַהּ הַגָּדֵר חָמֵשׁ אַמּוֹת פִּשְׁתָּן שָׁזוּר, וּבְסִיסֵי עַמּוּדֶיהָ מִנְּחוֹשֶׁת.
יט לְכֹל֙ כְּלֵ֣י הַמִּשְׁכָּ֔ן בְּכֹ֖ל עֲבֹדָת֑וֹ וְכׇל־יְתֵדֹתָ֛יו וְכׇל־יִתְדֹ֥ת הֶחָצֵ֖ר נְחֹֽשֶׁת׃ כָּל כְּלֵי הַמִּשְׁכָּן לְכָל עֲבוֹדוֹתָיו, וְכָל יְתֵדוֹתָיו וְכָל יִתְדוֹת הֶחָצֵר – מִנְּחוֹשֶׁת.
