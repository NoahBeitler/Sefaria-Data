ראשון[עריכה]

פ פָּרָשַׁת הֲפָרַת נְדָרִים
ב וַיְדַבֵּ֤ר מֹשֶׁה֙ אֶל־רָאשֵׁ֣י הַמַּטּ֔וֹת לִבְנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר זֶ֣ה הַדָּבָ֔ר אֲשֶׁ֖ר צִוָּ֥ה יְהֹוָֽה׃ מֹשֶׁה דִּבֵּר אֶל רָאשֵׁי שִׁבְטֵי בְּנֵי יִשְׂרָאֵל וְאָמַר: זֶה הַדָּבָר שֶׁה' צִוָּה:
ג אִישׁ֩ כִּֽי־יִדֹּ֨ר נֶ֜דֶר לַֽיהֹוָ֗ה אֽוֹ־הִשָּׁ֤בַע שְׁבֻעָה֙ לֶאְסֹ֤ר אִסָּר֙ עַל־נַפְשׁ֔וֹ לֹ֥א יַחֵ֖ל דְּבָר֑וֹ כְּכׇל־הַיֹּצֵ֥א מִפִּ֖יו יַעֲשֶֽׂה׃ אִם אִישׁ יִדּוֹר נֶדֶר לַה', אוֹ יִשָּׁבַע שְׁבוּעָה לֶאֱסוֹר עַל עַצְמוֹ אִסּוּר – הוּא לֹא יָפֵר וִיחַלֵּל אֶת דְּבָרוֹ, אֶלָּא יַעֲשֶׂה כְּכָל מַה שֶּׁיָּצָא מִפִּיו.
ד וְאִשָּׁ֕ה כִּֽי־תִדֹּ֥ר נֶ֖דֶר לַיהֹוָ֑ה וְאָסְרָ֥ה אִסָּ֛ר בְּבֵ֥ית אָבִ֖יהָ בִּנְעֻרֶֽיהָ׃ וְאִם אִשָּׁה תִּדּוֹר נֶדֶר לַה' בִּנְעוּרֶיהָ אוֹ תֶּאֱסוֹר עַל עַצְמָהּ אִסּוּר כְּשֶׁהִיא בְּבֵית אָבִיהָ.
ה וְשָׁמַ֨ע אָבִ֜יהָ אֶת־נִדְרָ֗הּ וֶֽאֱסָרָהּ֙ אֲשֶׁ֣ר אָֽסְרָ֣ה עַל־נַפְשָׁ֔הּ וְהֶחֱרִ֥ישׁ לָ֖הּ אָבִ֑יהָ וְקָ֙מוּ֙ כׇּל־נְדָרֶ֔יהָ וְכׇל־אִסָּ֛ר אֲשֶׁר־אָסְרָ֥ה עַל־נַפְשָׁ֖הּ יָקֽוּם׃ וְאָבִיהָ יִשְׁמַע אֶת הַנֶּדֶר שֶׁלָּהּ אוֹ אֶת הָאִסּוּר שֶׁאָסְרָה עַל עַצְמָהּ וְאָבִיהָ יִשְׁתּוֹק לָהּ – כָּל נְדָרֶיהָ קַיָּמִים וְכָל אִסּוּר שֶׁאָסְרָה עַל עַצְמָהּ קַיָּם.
ו וְאִם־הֵנִ֨יא אָבִ֣יהָ אֹתָהּ֮ בְּי֣וֹם שׇׁמְעוֹ֒ כׇּל־נְדָרֶ֗יהָ וֶֽאֱסָרֶ֛יהָ אֲשֶׁר־אָסְרָ֥ה עַל־נַפְשָׁ֖הּ לֹ֣א יָק֑וּם וַֽיהֹוָה֙ יִֽסְלַח־לָ֔הּ כִּי־הֵנִ֥יא אָבִ֖יהָ אֹתָֽהּ׃ וְאִם אָבִיהָ יְבַטֵּל לָהּ בַּיּוֹם שֶׁיִּשְׁמַע, כָּל נְדָרֶיהָ וְאִסּוּרֶיהָ שֶׁאָסְרָה עַל נַפְשָׁהּ לֹא יִתְקַיְּמוּ, וַה' יִסְלַח לָהּ אִם עָבְרָה עַל הַנֶּדֶר כִּי אָבִיהָ בִּטֵּל לָהּ.
ז וְאִם־הָי֤וֹ תִֽהְיֶה֙ לְאִ֔ישׁ וּנְדָרֶ֖יהָ עָלֶ֑יהָ א֚וֹ מִבְטָ֣א שְׂפָתֶ֔יהָ אֲשֶׁ֥ר אָסְרָ֖ה עַל־נַפְשָֽׁהּ׃ וְאִם הִיא תִּנָּשֵׂא לְאִישׁ אַחֲרֵי שֶׁנָּדְרָה אִסּוּר עַל עַצְמָהּ אוֹ אָסְרָה עַל עַצְמָהּ בִּשְׂפָתֶיהָ בִּשְׁבוּעָה.
ח וְשָׁמַ֥ע אִישָׁ֛הּ בְּי֥וֹם שׇׁמְע֖וֹ וְהֶחֱרִ֣ישׁ לָ֑הּ וְקָ֣מוּ נְדָרֶ֗יהָ וֶֽאֱסָרֶ֛הָ אֲשֶׁר־אָסְרָ֥ה עַל־נַפְשָׁ֖הּ יָקֻֽמוּ׃ וּבַעֲלָהּ יִשְׁמַע וְיִשְׁתּוֹק לָהּ בַּיּוֹם שֶׁיִּשְׁמַע, אָז נְדָרֶיהָ קַיָּמִים, וְאִסּוּרֶיהָ שֶׁאָסְרָה עַל נַפְשָׁהּ קַיָּמִים.
ט וְ֠אִ֠ם בְּי֨וֹם שְׁמֹ֣עַ אִישָׁהּ֮ יָנִ֣יא אוֹתָהּ֒ וְהֵפֵ֗ר אֶת־נִדְרָהּ֙ אֲשֶׁ֣ר עָלֶ֔יהָ וְאֵת֙ מִבְטָ֣א שְׂפָתֶ֔יהָ אֲשֶׁ֥ר אָסְרָ֖ה עַל־נַפְשָׁ֑הּ וַיהֹוָ֖ה יִֽסְלַֽח־לָֽהּ׃ וְאִם בַּיּוֹם שֶׁבַּעֲלָהּ יִשְׁמַע הוּא יְבַטֵּל לָהּ, הוּא יָפֵר אֶת נִדְרָהּ שֶׁעָלֶיהָ וְאֶת הַשְּׁבוּעָה שֶׁאָסְרָה בִּשְׂפָתֶיהָ עַל עַצְמָהּ, וַה' יִסְלַח לָהּ אִם עָבְרָה עַל הַנֶּדֶר.
י וְנֵ֥דֶר אַלְמָנָ֖ה וּגְרוּשָׁ֑ה כֹּ֛ל אֲשֶׁר־אָסְרָ֥ה עַל־נַפְשָׁ֖הּ יָק֥וּם עָלֶֽיהָ׃ וְנֶדֶר שֶׁל אַלְמָנָה וּגְרוּשָׁה – כָּל מַה שֶּׁתֶּאֱסוֹר עַל עַצְמָהּ יָחוּל עָלֶיהָ.
יא וְאִם־בֵּ֥ית אִישָׁ֖הּ נָדָ֑רָה אֽוֹ־אָסְרָ֥ה אִסָּ֛ר עַל־נַפְשָׁ֖הּ בִּשְׁבֻעָֽה׃ וְאִם הִיא תִּנְדּוֹר בְּבֵית בַּעֲלָהּ, אוֹ תֶּאֱסוֹר עַל עַצְמָהּ אִסּוּר בִּשְׁבוּעָה.
יב וְשָׁמַ֤ע אִישָׁהּ֙ וְהֶחֱרִ֣שׁ לָ֔הּ לֹ֥א הֵנִ֖יא אֹתָ֑הּ וְקָ֙מוּ֙ כׇּל־נְדָרֶ֔יהָ וְכׇל־אִסָּ֛ר אֲשֶׁר־אָסְרָ֥ה עַל־נַפְשָׁ֖הּ יָקֽוּם׃ וּבַעֲלָהּ יִשְׁמַע וְיִשְׁתּוֹק לָהּ וְלֹא יְבַטֵּל לָהּ, כָּל נְדָרֶיהָ קַיָּמִים וְכָל אִסּוּר שֶׁאָסְרָה עַל עַצְמָה קַיָּם.
יג וְאִם־הָפֵר֩ יָפֵ֨ר אֹתָ֥ם ׀ אִישָׁהּ֮ בְּי֣וֹם שׇׁמְעוֹ֒ כׇּל־מוֹצָ֨א שְׂפָתֶ֧יהָ לִנְדָרֶ֛יהָ וּלְאִסַּ֥ר נַפְשָׁ֖הּ לֹ֣א יָק֑וּם אִישָׁ֣הּ הֲפֵרָ֔ם וַיהֹוָ֖ה יִֽסְלַֽח־לָֽהּ׃ וְאִם בַּעֲלָהּ יָפֵר אוֹתָם בַּיּוֹם שֶׁיִּשְׁמַע, כָּל מַה שֶּׁנָּדְרָה בִּשְׂפָתֶיהָ וְאָסְרָה עַל עַצְמָה לֹא יִתְקַיֵּם, בַּעֲלָהּ הֵפֵר אוֹתָם וַה' יִסְלַח לָהּ אִם עָבְרָה עַל הַנֶּדֶר.
יד כׇּל־נֵ֛דֶר וְכׇל־שְׁבֻעַ֥ת אִסָּ֖ר לְעַנֹּ֣ת נָ֑פֶשׁ אִישָׁ֥הּ יְקִימֶ֖נּוּ וְאִישָׁ֥הּ יְפֵרֶֽנּוּ׃ כָּל נֵדֶר וְכָל שְׁבוּעַת אִסּוּר עַל עַצְמָהּ שֶׁגּוֹרְמִים לְעִנּוּי – בַּעֲלָהּ יְקַיֵּם וּבַעֲלָהּ יָפֵר.
טו וְאִם־הַחֲרֵשׁ֩ יַחֲרִ֨ישׁ לָ֥הּ אִישָׁהּ֮ מִיּ֣וֹם אֶל־יוֹם֒ וְהֵקִים֙ אֶת־כׇּל־נְדָרֶ֔יהָ א֥וֹ אֶת־כׇּל־אֱסָרֶ֖יהָ אֲשֶׁ֣ר עָלֶ֑יהָ הֵקִ֣ים אֹתָ֔ם כִּי־הֶחֱרִ֥שׁ לָ֖הּ בְּי֥וֹם שׇׁמְעֽוֹ׃ וְאִם בַּעֲלָהּ יִשְׁתּוֹק מֵהַיּוֹם שֶׁשָּׁמַע עַד לַיּוֹם הַבָּא, הוּא יְקַיֵּם בָּזֶה אֶת כָּל נְדָרֶיהָ אוֹ אֶת כָּל אִסּוּרֶיהָ שֶׁעָלֶיהָ – הוּא מְקַיֵּם אוֹתָם כִּי שָׁתַק לָהּ בַּיּוֹם שֶׁשָּׁמַע.
טז וְאִם־הָפֵ֥ר יָפֵ֛ר אֹתָ֖ם אַחֲרֵ֣י שׇׁמְע֑וֹ וְנָשָׂ֖א אֶת־עֲוֺנָֽהּ׃ וְאִם יָפֵר אוֹתָם אַחֲרֵי הַיּוֹם שֶׁשָּׁמַע וְלָכֵן עָבְרָה עַל נִדְרָהּ, הוּא שֶׁיְּקַבֵּל אֶת הָעוֹנֶשׁ עַל עֲוֹנָהּ.
יז אֵ֣לֶּה הַֽחֻקִּ֗ים אֲשֶׁ֨ר צִוָּ֤ה יְהֹוָה֙ אֶת־מֹשֶׁ֔ה בֵּ֥ין אִ֖ישׁ לְאִשְׁתּ֑וֹ בֵּֽין־אָ֣ב לְבִתּ֔וֹ בִּנְעֻרֶ֖יהָ בֵּ֥ית אָבִֽיהָ׃ אֵלֶּה הַחֻקִּים שֶׁה' צִוָּה אֶת מֹשֶׁה, בֵּין אִישׁ לְאִשְׁתּוֹ, בֵּין אָב לְבִתּוֹ בִּנְעוּרֶיהָ כְּשֶׁנִּמְצֵאת בְּבֵית אָבִיהָ.

שני[עריכה]

פ פָּרָשַׁת הַנְּקָמָה בְּמִדְיָן
לא א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב נְקֹ֗ם נִקְמַת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל מֵאֵ֖ת הַמִּדְיָנִ֑ים אַחַ֖ר תֵּאָסֵ֥ף אֶל־עַמֶּֽיךָ׃ נְקוֹם אֶת נִקְמַת בְּנֵי יִשְׂרָאֵל בַּמִּדְיָנִים, וְאַחַר כָּךְ תֵּאָסֵף נִשְׁמָתְךָ אֶל מְקוֹם הַנְּשָׁמוֹת.
ג וַיְדַבֵּ֤ר מֹשֶׁה֙ אֶל־הָעָ֣ם לֵאמֹ֔ר הֵחָלְצ֧וּ מֵאִתְּכֶ֛ם אֲנָשִׁ֖ים לַצָּבָ֑א וְיִהְיוּ֙ עַל־מִדְיָ֔ן לָתֵ֥ת נִקְמַת־יְהֹוָ֖ה בְּמִדְיָֽן׃ מֹשֶׁה דִּבֵּר אֶל הָעָם וְאָמַר: גַּיְּסוּ מֵאִתְּכֶם אֲנָשִׁים לַצָּבָא, וְיִלָּחֲמוּ בְּמִדְיָן כְּדֵי לִנְקוֹם אֶת נִקְמַת ה' בְּמִדְיָן.
ד אֶ֚לֶף לַמַּטֶּ֔ה אֶ֖לֶף לַמַּטֶּ֑ה לְכֹל֙ מַטּ֣וֹת יִשְׂרָאֵ֔ל תִּשְׁלְח֖וּ לַצָּבָֽא׃ תִּשְׁלְחוּ לַצָּבָא אֶלֶף אִישׁ מִכָּל שֵׁבֶט, מִכָּל שִׁבְטֵי יִשְׂרָאֵל.
ה וַיִּמָּֽסְרוּ֙ מֵאַלְפֵ֣י יִשְׂרָאֵ֔ל אֶ֖לֶף לַמַּטֶּ֑ה שְׁנֵים־עָשָׂ֥ר אֶ֖לֶף חֲלוּצֵ֥י צָבָֽא׃ מֵאַלְפֵי בְּנֵי יִשְׂרָאֵל נִשְׁלְחוּ אֶלֶף אִישׁ מִכָּל שֵׁבֶט – שְׁנֵים עָשָׂר אֶלֶף מְגֻיָּסִים לַצָּבָא.
ו וַיִּשְׁלַ֨ח אֹתָ֥ם מֹשֶׁ֛ה אֶ֥לֶף לַמַּטֶּ֖ה לַצָּבָ֑א אֹ֠תָ֠ם וְאֶת־פִּ֨ינְחָ֜ס בֶּן־אֶלְעָזָ֤ר הַכֹּהֵן֙ לַצָּבָ֔א וּכְלֵ֥י הַקֹּ֛דֶשׁ וַחֲצֹצְר֥וֹת הַתְּרוּעָ֖ה בְּיָדֽוֹ׃ מֹשֶׁה שָׁלַח אוֹתָם לַצָּבָא אֶלֶף אִישׁ מִכָּל שֵׁבֶט, אוֹתָם וְאֶת פִּינְחָס בֶּן אֶלְעָזָר הַכֹּהֵן, וּכְלֵי הַקּוֹדֶשׁ מֵהַמִּשְׁכָּן וַחֲצוֹצְרוֹת הַתְּרוּעָה בְּיָדוֹ.
ז וַֽיִּצְבְּאוּ֙ עַל־מִדְיָ֔ן כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶׁ֑ה וַיַּֽהַרְג֖וּ כׇּל־זָכָֽר׃ הֵם נִלְחֲמוּ נֶגֶד מִדְיָן כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה, וְהֵם הָרְגוּ כָּל זָכָר.
ח וְאֶת־מַלְכֵ֨י מִדְיָ֜ן הָרְג֣וּ עַל־חַלְלֵיהֶ֗ם אֶת־אֱוִ֤י וְאֶת־רֶ֙קֶם֙ וְאֶת־צ֤וּר וְאֶת־חוּר֙ וְאֶת־רֶ֔בַע חֲמֵ֖שֶׁת מַלְכֵ֣י מִדְיָ֑ן וְאֵת֙ בִּלְעָ֣ם בֶּן־בְּע֔וֹר הָרְג֖וּ בֶּחָֽרֶב׃ וְהֵם הָרְגוּ אֶת מַלְכֵי מִדְיָן – אֶת אֱוִי וְאֶת רֶקֶם וְאֶת צוּר וְאֶת חוּר וְאֶת רֶבַע – חֲמֵשֶׁת מַלְכֵי מִדְיָן, בְּנוֹסָף לִשְׁאָר הֲרוּגֵיהֶם בַּקְּרָב, וְאֶת בִּלְעָם בֶּן בְּעוֹר הָרְגוּ בְּחֶרֶב.
ט וַיִּשְׁבּ֧וּ בְנֵי־יִשְׂרָאֵ֛ל אֶת־נְשֵׁ֥י מִדְיָ֖ן וְאֶת־טַפָּ֑ם וְאֵ֨ת כׇּל־בְּהֶמְתָּ֧ם וְאֶת־כׇּל־מִקְנֵהֶ֛ם וְאֶת־כׇּל־חֵילָ֖ם בָּזָֽזוּ׃ בְּנֵי יִשְׂרָאֵל לָקְחוּ בַּשֶּׁבִי אֶת נְשׁוֹת מִדְיָן וְאֶת יַלְדֵיהֶן, וּבָזְזוּ אֶת כָּל בְּהֵמוֹתֵיהֶם וְאֶת כָּל עֶדְרֵיהֶם וְאֶת כָּל נִכְסֵיהֶם.
י וְאֵ֤ת כׇּל־עָרֵיהֶם֙ בְּמ֣וֹשְׁבֹתָ֔ם וְאֵ֖ת כׇּל־טִֽירֹתָ֑ם שָׂרְפ֖וּ בָּאֵֽשׁ׃ וְשָׂרְפוּ בָּאֵשׁ אֶת כָּל הֶעָרִים שֶׁהֵם הִתְיַשְּׁבוּ בָּהֶן וְאֶת כָּל מִבְצְרֵיהֶם.
יא וַיִּקְחוּ֙ אֶת־כׇּל־הַשָּׁלָ֔ל וְאֵ֖ת כׇּל־הַמַּלְק֑וֹחַ בָּאָדָ֖ם וּבַבְּהֵמָֽה׃ וְלָקְחוּ אֶת כָּל הַשָּׁלָל וְאֶת כָּל הָאֲנָשִׁים וְהַבְּהֵמוֹת שֶׁלָּקְחוּ.
יב וַיָּבִ֡אוּ אֶל־מֹשֶׁה֩ וְאֶל־אֶלְעָזָ֨ר הַכֹּהֵ֜ן וְאֶל־עֲדַ֣ת בְּנֵֽי־יִשְׂרָאֵ֗ל אֶת־הַשְּׁבִ֧י וְאֶת־הַמַּלְק֛וֹחַ וְאֶת־הַשָּׁלָ֖ל אֶל־הַֽמַּחֲנֶ֑ה אֶל־עַֽרְבֹ֣ת מוֹאָ֔ב אֲשֶׁ֖ר עַל־יַרְדֵּ֥ן יְרֵחֽוֹ׃ הֵם הֵבִיאוּ לְמֹשֶׁה וּלְאֶלְעָזָר הַכֹּהֵן וְלַצִּבּוּר שֶׁל בְּנֵי יִשְׂרָאֵל אֶת הַשְּׁבוּיִים וְאֶת הַבְּהֵמוֹת שֶׁלָּקְחוּ וְאֶת הַשָּׁלָל לַמַּחֲנֶה, לְמִישׁוֹרֵי מוֹאָב שֶׁלְּיַד הַיַּרְדֵּן מוּל יְרִיחוֹ.

שלישי (שני במחובר)[עריכה]

ס
יג וַיֵּ֨צְא֜וּ מֹשֶׁ֨ה וְאֶלְעָזָ֧ר הַכֹּהֵ֛ן וְכׇל־נְשִׂיאֵ֥י הָעֵדָ֖ה לִקְרָאתָ֑ם אֶל־מִח֖וּץ לַֽמַּחֲנֶֽה׃ מֹשֶׁה וְאֶלְעָזָר הַכֹּהֵן וְכָל נְשִׂיאֵי הַצִּבּוּר יָצְאוּ לִקְרָאתָם אֶל מִחוּץ לַמַּחֲנֶה.
יד וַיִּקְצֹ֣ף מֹשֶׁ֔ה עַ֖ל פְּקוּדֵ֣י הֶחָ֑יִל שָׂרֵ֤י הָאֲלָפִים֙ וְשָׂרֵ֣י הַמֵּא֔וֹת הַבָּאִ֖ים מִצְּבָ֥א הַמִּלְחָמָֽה׃ מֹשֶׁה כָּעַס עַל מְפַקְּדֵי הַצָּבָא שָׂרֵי הָאֲלָפִים וְשָׂרֵי הַמֵּאוֹת שֶׁבָּאוּ מִצְּבָא הַמִּלְחָמָה.
טו וַיֹּ֥אמֶר אֲלֵיהֶ֖ם מֹשֶׁ֑ה הַֽחִיִּיתֶ֖ם כׇּל־נְקֵבָֽה׃ מֹשֶׁה אָמַר לָהֶם: הַאִם הִשְׁאַרְתֶּם אֶת הַנָּשִׁים בַּחַיִּים?!
טז הֵ֣ן הֵ֜נָּה הָי֨וּ לִבְנֵ֤י יִשְׂרָאֵל֙ בִּדְבַ֣ר בִּלְעָ֔ם לִמְסׇר־מַ֥עַל בַּיהֹוָ֖ה עַל־דְּבַר־פְּע֑וֹר וַתְּהִ֥י הַמַּגֵּפָ֖ה בַּעֲדַ֥ת יְהֹוָֽה׃ הֲרֵי הֵן אֵלֶּה שֶׁהִכְשִׁילוּ אֶת בְּנֵי יִשְׂרָאֵל בַּעֲצַת בִּלְעָם, לִסְטוֹת וְלִמְרוֹד בַּה' בְּעִנְיַן פְּעוֹר, וְהָיְתָה אָז הַמַּגֵּפָה בַּצִּבּוּר שֶׁל ה'!
יז וְעַתָּ֕ה הִרְג֥וּ כׇל־זָכָ֖ר בַּטָּ֑ף וְכׇל־אִשָּׁ֗ה יֹדַ֥עַת אִ֛ישׁ לְמִשְׁכַּ֥ב זָכָ֖ר הֲרֹֽגוּ׃ וְעַכְשָׁיו הִרְגוּ אֶת כָּל הַיְּלָדִים הַזְּכָרִים, וְגַם כָּל אִשָּׁה שֶׁשָּׁכְבָה עִם אִישׁ תַּהַרְגוּ!
יח וְכֹל֙ הַטַּ֣ף בַּנָּשִׁ֔ים אֲשֶׁ֥ר לֹא־יָדְע֖וּ מִשְׁכַּ֣ב זָכָ֑ר הַחֲי֖וּ לָכֶֽם׃ וּתְחַיּוּ לָכֶם אֶת כָּל הַיְּלָדוֹת שֶׁלֹּא שָׁכְבוּ עִם זָכָר.
יט וְאַתֶּ֗ם חֲנ֛וּ מִח֥וּץ לַֽמַּחֲנֶ֖ה שִׁבְעַ֣ת יָמִ֑ים כֹּל֩ הֹרֵ֨ג נֶ֜פֶשׁ וְכֹ֣ל ׀ נֹגֵ֣עַ בֶּֽחָלָ֗ל תִּֽתְחַטְּא֞וּ בַּיּ֤וֹם הַשְּׁלִישִׁי֙ וּבַיּ֣וֹם הַשְּׁבִיעִ֔י אַתֶּ֖ם וּשְׁבִיכֶֽם׃ וְאַתֶּם חֲנוּ מִחוּץ לַמַּחֲנֶה שִׁבְעָה יָמִים, כָּל מִי שֶׁהָרַג אָדָם וְכָל מִי שֶׁנָּגַע בְּמֵת – תִּטָּהֲרוּ בַּיּוֹם הַשְּׁלִישִׁי וּבַיּוֹם הַשְּׁבִיעִי, אַתֶּם וּשְׁבוּיֵיכֶם.
כ וְכׇל־בֶּ֧גֶד וְכׇל־כְּלִי־ע֛וֹר וְכׇל־מַעֲשֵׂ֥ה עִזִּ֖ים וְכׇל־כְּלִי־עֵ֑ץ תִּתְחַטָּֽאוּ׃ וּתְטַהֲרוּ כָּל בֶּגֶד וְכָל כְּלִי עוֹר וְכָל דָּבָר עָשׂוּי מִשֵּׂעַר עִזִּים וְכָל כְּלִי עֵץ.
ס
כא וַיֹּ֨אמֶר אֶלְעָזָ֤ר הַכֹּהֵן֙ אֶל־אַנְשֵׁ֣י הַצָּבָ֔א הַבָּאִ֖ים לַמִּלְחָמָ֑ה זֹ֚את חֻקַּ֣ת הַתּוֹרָ֔ה אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ אֶלְעָזָר הַכֹּהֵן אָמַר לַלּוֹחֲמִים שֶׁבָּאוּ לַמִּלְחָמָה: זֶהוּ חוֹק הַתּוֹרָה שֶׁה' צִוָּה אֶת מֹשֶׁה:
כב אַ֥ךְ אֶת־הַזָּהָ֖ב וְאֶת־הַכָּ֑סֶף אֶֽת־הַנְּחֹ֙שֶׁת֙ אֶת־הַבַּרְזֶ֔ל אֶֽת־הַבְּדִ֖יל וְאֶת־הָעֹפָֽרֶת׃ רַק אֶת הַזָּהָב וְאֶת הַכֶּסֶף אֶת הַנְּחוֹשֶׁת אֶת הַבַּרְזֶל אֶת הַבְּדִיל וְאֶת הָעוֹפֶרֶת –
כג כׇּל־דָּבָ֞ר אֲשֶׁר־יָבֹ֣א בָאֵ֗שׁ תַּעֲבִ֤ירוּ בָאֵשׁ֙ וְטָהֵ֔ר אַ֕ךְ בְּמֵ֥י נִדָּ֖ה יִתְחַטָּ֑א וְכֹ֨ל אֲשֶׁ֧ר לֹֽא־יָבֹ֛א בָּאֵ֖שׁ תַּעֲבִ֥ירוּ בַמָּֽיִם׃ כָּל דָּבָר שֶׁאֶפְשָׁר לְהַעֲבִירוֹ בָּאֵשׁ תַּעֲבִירוּ בָּאֵשׁ וְיִהְיֶה טָהוֹר, אֲבָל צָרִיךְ לְטַהֵר אוֹתוֹ גַּם בְּמֵי הֲזָאָה, וְכָל דָּבָר שֶׁאִי אֶפְשָׁר לְהַעֲבִירוֹ בָּאֵשׁ תַּטְבִּילוּ רַק בְּמַיִם.
כד וְכִבַּסְתֶּ֧ם בִּגְדֵיכֶ֛ם בַּיּ֥וֹם הַשְּׁבִיעִ֖י וּטְהַרְתֶּ֑ם וְאַחַ֖ר תָּבֹ֥אוּ אֶל־הַֽמַּחֲנֶֽה׃ וְתַטְבִּילוּ אֶת בִּגְדֵיכֶם בַּיּוֹם הַשְּׁבִיעִי וְתִהְיוּ טְהוֹרִים, וְאַחַר כָּךְ תָּבוֹאוּ לַמַּחֲנֶה.

רביעי[עריכה]

ס
כה וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' אָמַר לְמֹשֶׁה:
כו שָׂ֗א אֵ֣ת רֹ֤אשׁ מַלְק֙וֹחַ֙ הַשְּׁבִ֔י בָּאָדָ֖ם וּבַבְּהֵמָ֑ה אַתָּה֙ וְאֶלְעָזָ֣ר הַכֹּהֵ֔ן וְרָאשֵׁ֖י אֲב֥וֹת הָעֵדָֽה׃ סְפוֹר, אַתָּה וְאֶלְעָזָר הַכֹּהֵן וְרָאשֵׁי אֲבוֹת הַצִּבּוּר, אֶת הַלְּקוּחִים בַּשֶּׁבִי מֵהָאָדָם וּמֵהַבְּהֵמָה.
כז וְחָצִ֙יתָ֙ אֶת־הַמַּלְק֔וֹחַ בֵּ֚ין תֹּפְשֵׂ֣י הַמִּלְחָמָ֔ה הַיֹּצְאִ֖ים לַצָּבָ֑א וּבֵ֖ין כׇּל־הָעֵדָֽה׃ וּתְחַלֵּק אֶת הַלְּקוּחִים בַּשֶּׁבִי בְּשָׁוֶה – בֵּין הַמִּשְׁתַּתְּפִים בַּמִּלְחָמָה שֶׁיָּצְאוּ לַצָּבָא וּבֵין שְׁאָר הַצִּבּוּר.
כח וַהֲרֵמֹתָ֨ מֶ֜כֶס לַֽיהֹוָ֗ה מֵאֵ֞ת אַנְשֵׁ֤י הַמִּלְחָמָה֙ הַיֹּצְאִ֣ים לַצָּבָ֔א אֶחָ֣ד נֶ֔פֶשׁ מֵחֲמֵ֖שׁ הַמֵּא֑וֹת מִן־הָאָדָם֙ וּמִן־הַבָּקָ֔ר וּמִן־הַחֲמֹרִ֖ים וּמִן־הַצֹּֽאן׃ וְתַפְרִישׁ מֶכֶס לַה' מֵחֶלְקָם שֶׁל אַנְשֵׁי הַמִּלְחָמָה שֶׁיָּצְאוּ לַצָּבָא, נֶפֶשׁ אַחַת מִכָּל חֲמֵשׁ מֵאוֹת – מֵהָאָדָם מֵהַבָּקָר מֵהַחֲמוֹרִים וּמֵהַצֹּאן.
כט מִמַּֽחֲצִיתָ֖ם תִּקָּ֑חוּ וְנָתַתָּ֛ה לְאֶלְעָזָ֥ר הַכֹּהֵ֖ן תְּרוּמַ֥ת יְהֹוָֽה׃ תִּקְחוּ מֵהַחֵצִי שֶׁל מִשְׁתַּתְּפֵי הַמִּלְחָמָה וְתִתֵּן אוֹתָם לְאֶלְעָזָר הַכֹּהֵן כִּתְרוּמָה לַה'.
ל וּמִמַּחֲצִ֨ת בְּנֵֽי־יִשְׂרָאֵ֜ל תִּקַּ֣ח ׀ אֶחָ֣ד ׀ אָחֻ֣ז מִן־הַחֲמִשִּׁ֗ים מִן־הָאָדָ֧ם מִן־הַבָּקָ֛ר מִן־הַחֲמֹרִ֥ים וּמִן־הַצֹּ֖אן מִכׇּל־הַבְּהֵמָ֑ה וְנָתַתָּ֤ה אֹתָם֙ לַלְוִיִּ֔ם שֹׁמְרֵ֕י מִשְׁמֶ֖רֶת מִשְׁכַּ֥ן יְהֹוָֽה׃ וּמֵהַחֵצִי שֶׁל בְּנֵי יִשְׂרָאֵל תִּקַּח חֵלֶק אֶחָד מִכָּל חֲמִשִּׁים, מֵהָאָדָם מֵהַבָּקָר מֵהַחֲמוֹרִים וּמֵהַצֹּאן, מִכָּל הַבְּהֵמוֹת, וְתִתֵּן אוֹתָם לַלְוִיִּים הַשּׁוֹמְרִים בְּמִשְׁמַר מִשְׁכַּן ה'.
לא וַיַּ֣עַשׂ מֹשֶׁ֔ה וְאֶלְעָזָ֖ר הַכֹּהֵ֑ן כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ מֹשֶׁה וְאֶלְעָזָר הַכֹּהֵן עָשׂוּ כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.
לב וַיְהִי֙ הַמַּלְק֔וֹחַ יֶ֣תֶר הַבָּ֔ז אֲשֶׁ֥ר בָּזְז֖וּ עַ֣ם הַצָּבָ֑א צֹ֗אן שֵׁשׁ־מֵא֥וֹת אֶ֛לֶף וְשִׁבְעִ֥ים אֶ֖לֶף וַחֲמֵ֥שֶׁת אֲלָפִֽים׃ וְאֵלֶּה הַלְּקוּחִים בַּשֶּׁבִי שֶׁהוּבְאוּ לַחֲלוּקָה –הַנִּשְׁאָר מִמַּה שֶּׁהַלּוֹחֲמִים בָּזְזוּ וְאָכְלוּ בַּמִּלְחָמָה: צֹאן – שֵׁשׁ מֵאוֹת שִׁבְעִים וַחֲמִשָּׁה אֶלֶף.
לג וּבָקָ֕ר שְׁנַ֥יִם וְשִׁבְעִ֖ים אָֽלֶף׃ וּבָקָר – שִׁבְעִים וּשְׁנַיִם אֶלֶף.
לד וַחֲמֹרִ֕ים אֶחָ֥ד וְשִׁשִּׁ֖ים אָֽלֶף׃ וַחֲמוֹרִים – שִׁשִּׁים וְאֶחָד אֶלֶף.
לה וְנֶ֣פֶשׁ אָדָ֔ם מִ֨ן־הַנָּשִׁ֔ים אֲשֶׁ֥ר לֹֽא־יָדְע֖וּ מִשְׁכַּ֣ב זָכָ֑ר כׇּל־נֶ֕פֶשׁ שְׁנַ֥יִם וּשְׁלֹשִׁ֖ים אָֽלֶף׃ וְנֶפֶשׁ אָדָם מֵהַנָּשִׁים שֶׁלֹּא שָׁכְבוּ עִם זָכָר – כָּל הַנְּפָשׁוֹת – שְׁלוֹשִׁים וּשְׁנַיִם אֶלֶף.
לו וַתְּהִי֙ הַֽמֶּחֱצָ֔ה חֵ֕לֶק הַיֹּצְאִ֖ים בַּצָּבָ֑א מִסְפַּ֣ר הַצֹּ֗אן שְׁלֹשׁ־מֵא֥וֹת אֶ֙לֶף֙ וּשְׁלֹשִׁ֣ים אֶ֔לֶף וְשִׁבְעַ֥ת אֲלָפִ֖ים וַחֲמֵ֥שׁ מֵאֽוֹת׃ הַחֵצִי – הַחֵלֶק שֶׁל הַיּוֹצְאִים בַּצָּבָא הָיָה כָּךְ: מִסְפַּר הַצֹּאן – שְׁלוֹשׁ מֵאוֹת וּשְׁלוֹשִׁים וְשִׁבְעָה אֶלֶף וַחֲמֵשׁ מֵאוֹת.
לז וַיְהִ֛י הַמֶּ֥כֶס לַֽיהֹוָ֖ה מִן־הַצֹּ֑אן שֵׁ֥שׁ מֵא֖וֹת חָמֵ֥שׁ וְשִׁבְעִֽים׃ הַמֶּכֶס לַה' מֵהַצֹּאן הָיָה שֵׁשׁ מֵאוֹת וְשִׁבְעִים וְחָמֵשׁ.
לח וְהַ֨בָּקָ֔ר שִׁשָּׁ֥ה וּשְׁלֹשִׁ֖ים אָ֑לֶף וּמִכְסָ֥ם לַיהֹוָ֖ה שְׁנַ֥יִם וְשִׁבְעִֽים׃ וְהַבָּקָר – שְׁלוֹשִׁים וְשִׁשָּׁה אֶלֶף, וְהַמֶּכֶס מֵהֶם לַה' – שִׁבְעִים וּשְׁנַיִם.
לט וַחֲמֹרִ֕ים שְׁלֹשִׁ֥ים אֶ֖לֶף וַחֲמֵ֣שׁ מֵא֑וֹת וּמִכְסָ֥ם לַֽיהֹוָ֖ה אֶחָ֥ד וְשִׁשִּֽׁים׃ וַחֲמוֹרִים – שְׁלוֹשִׁים אֶלֶף וַחֲמֵשׁ מֵאוֹת, וְהַמֶּכֶס מֵהֶם לַה' – שִׁשִּׁים וְאֶחָד.
מ וְנֶ֣פֶשׁ אָדָ֔ם שִׁשָּׁ֥ה עָשָׂ֖ר אָ֑לֶף וּמִכְסָם֙ לַֽיהֹוָ֔ה שְׁנַ֥יִם וּשְׁלֹשִׁ֖ים נָֽפֶשׁ׃ וְנֶפֶשׁ אָדָם – שִׁשָּׁה עָשָׂר אֶלֶף, וְהַמֶּכֶס מֵהֶם לַה' – שְׁלוֹשִׁים וּשְׁנַיִם נֶפֶשׁ.
מא וַיִּתֵּ֣ן מֹשֶׁ֗ה אֶת־מֶ֙כֶס֙ תְּרוּמַ֣ת יְהֹוָ֔ה לְאֶלְעָזָ֖ר הַכֹּהֵ֑ן כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ מֹשֶׁה נָתַן אֶת הַמֶּכֶס שֶׁהִפְרִישׁוּ לַה' לְאֶלְעָזָר הַכֹּהֵן כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.

חמישי[עריכה]

מב וּמִֽמַּחֲצִ֖ית בְּנֵ֣י יִשְׂרָאֵ֑ל אֲשֶׁר֙ חָצָ֣ה מֹשֶׁ֔ה מִן־הָאֲנָשִׁ֖ים הַצֹּבְאִֽים׃ וּמֵהַחֵצִי שֶׁל בְּנֵי יִשְׂרָאֵל שֶׁמֹּשֶׁה חָצָה מֵהַחֵלֶק שֶׁל הַלּוֹחֲמִים –
מג וַתְּהִ֛י מֶחֱצַ֥ת הָעֵדָ֖ה מִן־הַצֹּ֑אן שְׁלֹשׁ־מֵא֥וֹת אֶ֙לֶף֙ וּשְׁלֹשִׁ֣ים אֶ֔לֶף שִׁבְעַ֥ת אֲלָפִ֖ים וַחֲמֵ֥שׁ מֵאֽוֹת׃ בַּחֵצִי שֶׁקִּבְּלוּ הַצִּבּוּר הָיָה: מֵהַצֹּאן – שְׁלוֹשׁ מֵאוֹת וּשְׁלוֹשִׁים וְשִׁבְעָה אֶלֶף וַחֲמֵשׁ מֵאוֹת.
מד וּבָקָ֕ר שִׁשָּׁ֥ה וּשְׁלֹשִׁ֖ים אָֽלֶף׃ וּבָקָר – שְׁלוֹשִׁים וְשִׁשָּׁה אֶלֶף.
מה וַחֲמֹרִ֕ים שְׁלֹשִׁ֥ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת׃ וַחֲמוֹרִים – שְׁלוֹשִׁים אֶלֶף וַחֲמֵשׁ מֵאוֹת.
מו וְנֶ֣פֶשׁ אָדָ֔ם שִׁשָּׁ֥ה עָשָׂ֖ר אָֽלֶף׃ וְנֶפֶשׁ אָדָם – שִׁשָּׁה עָשָׂר אֶלֶף.
מז וַיִּקַּ֨ח מֹשֶׁ֜ה מִמַּחֲצִ֣ת בְּנֵֽי־יִשְׂרָאֵ֗ל אֶת־הָֽאָחֻז֙ אֶחָ֣ד מִן־הַחֲמִשִּׁ֔ים מִן־הָאָדָ֖ם וּמִן־הַבְּהֵמָ֑ה וַיִּתֵּ֨ן אֹתָ֜ם לַלְוִיִּ֗ם שֹֽׁמְרֵי֙ מִשְׁמֶ֙רֶת֙ מִשְׁכַּ֣ן יְהֹוָ֔ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ מֹשֶׁה לָקַח מֵהַחֵצִי שֶׁל בְּנֵי יִשְׂרָאֵל חֵלֶק אֶחָד מִכָּל חֲמִשִּׁים מֵהָאָדָם וּמֵהַבְּהֵמָה וְנָתַן אוֹתָם לַלְוִיִּים שֶׁשּׁוֹמְרִים בְּמִשְׁמַר מִשְׁכַּן ה', כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.
מח וַֽיִּקְרְבוּ֙ אֶל־מֹשֶׁ֔ה הַפְּקֻדִ֕ים אֲשֶׁ֖ר לְאַלְפֵ֣י הַצָּבָ֑א שָׂרֵ֥י הָאֲלָפִ֖ים וְשָׂרֵ֥י הַמֵּאֽוֹת׃ הַמְּפַקְּדִים שֶׁל אַלְפֵי הַלּוֹחֲמִים – שָׂרֵי הָאֲלָפִים וְשָׂרֵי הַמֵּאוֹת, נִגְּשׁוּ לְמֹשֶׁה.
מט וַיֹּֽאמְרוּ֙ אֶל־מֹשֶׁ֔ה עֲבָדֶ֣יךָ נָֽשְׂא֗וּ אֶת־רֹ֛אשׁ אַנְשֵׁ֥י הַמִּלְחָמָ֖ה אֲשֶׁ֣ר בְּיָדֵ֑נוּ וְלֹא־נִפְקַ֥ד מִמֶּ֖נּוּ אִֽישׁ׃ וְאָמְרוּ לְמֹשֶׁה: עֲבָדֶיךָ סָפְרוּ אֶת מִסְפַּר הַלּוֹחֲמִים שֶׁבְּפִקּוּדֵינוּ, וְלֹא נֶחְסַר מֵאִתָּנוּ אַף אֶחָד.
נ וַנַּקְרֵ֞ב אֶת־קׇרְבַּ֣ן יְהֹוָ֗ה אִישׁ֩ אֲשֶׁ֨ר מָצָ֤א כְלִֽי־זָהָב֙ אֶצְעָדָ֣ה וְצָמִ֔יד טַבַּ֖עַת עָגִ֣יל וְכוּמָ֑ז לְכַפֵּ֥ר עַל־נַפְשֹׁתֵ֖ינוּ לִפְנֵ֥י יְהֹוָֽה׃ לָכֵן הֵבֵאנוּ קָרְבַּן לַה', כָּל אֶחָד מֵאִתָּנוּ שֶׁמָּצָא תַּכְשִׁיט שֶׁל זָהָב – אֶצְעָדָה לָרֶגֶל, צָמִיד, טַבַּעַת, עָגִיל, אוֹ מָחוֹךְ, הֵבֵאנוּ כְּדֵי לְכַפֵּר עַל נַפְשׁוֹתֵינוּ לִפְנֵי ה'.
נא וַיִּקַּ֨ח מֹשֶׁ֜ה וְאֶלְעָזָ֧ר הַכֹּהֵ֛ן אֶת־הַזָּהָ֖ב מֵֽאִתָּ֑ם כֹּ֖ל כְּלִ֥י מַעֲשֶֽׂה׃ מֹשֶׁה וְאֶלְעָזָר הַכֹּהֵן לָקְחוּ מֵהֶם אֶת הַזָּהָב – כָּל כְּלִי יָצוּק.
נב וַיְהִ֣י ׀ כׇּל־זְהַ֣ב הַתְּרוּמָ֗ה אֲשֶׁ֤ר הֵרִ֙ימוּ֙ לַֽיהֹוָ֔ה שִׁשָּׁ֨ה עָשָׂ֥ר אֶ֛לֶף שְׁבַע־מֵא֥וֹת וַחֲמִשִּׁ֖ים שָׁ֑קֶל מֵאֵת֙ שָׂרֵ֣י הָֽאֲלָפִ֔ים וּמֵאֵ֖ת שָׂרֵ֥י הַמֵּאֽוֹת׃ כָּל הַזָּהָב שֶׁהִפְרִישׁוּ תְּרוּמָה לַה' הָיָה בְּמִשְׁקַל שִׁשָּׁה עָשָׂר אֶלֶף וּשְׁבַע מֵאוֹת וַחֲמִשִּׁים שֶׁקֶל (כְּ236 קִילוֹ) – שֶׁהִתְקַבֵּל מִשָּׂרֵי הָאֲלָפִים וּמִשָּׂרֵי הַמֵּאוֹת.
נג אַנְשֵׁי֙ הַצָּבָ֔א בָּזְז֖וּ אִ֥ישׁ לֽוֹ׃ הַלּוֹחֲמִים בָּזְזוּ בַּמִּלְחָמָה כָּל אֶחָד לְעַצְמוֹ.
נד וַיִּקַּ֨ח מֹשֶׁ֜ה וְאֶלְעָזָ֤ר הַכֹּהֵן֙ אֶת־הַזָּהָ֔ב מֵאֵ֛ת שָׂרֵ֥י הָאֲלָפִ֖ים וְהַמֵּא֑וֹת וַיָּבִ֤אוּ אֹתוֹ֙ אֶל־אֹ֣הֶל מוֹעֵ֔ד זִכָּר֥וֹן לִבְנֵֽי־יִשְׂרָאֵ֖ל לִפְנֵ֥י יְהֹוָֽה׃ מֹשֶׁה וְאֶלְעָזָר הַכֹּהֵן לָקְחוּ אֶת הַזָּהָב מִשָּׂרֵי הָאֲלָפִים וְשָׂרֵי הַמֵּאוֹת, וְהֵבִיאוּ אוֹתוֹ לְאוֹהֶל הַהִתְכַּנְּסוּת שֶׁיִּהְיֶה זִכָּרוֹן לִבְנֵי יִשְׂרָאֵל לִפְנֵי ה'.

ששי (שלישי במחובר)[עריכה]

פ פָּרָשַׁת בְּנֵי גָד וּבְנֵי רְאוּבֵן
לב א וּמִקְנֶ֣ה ׀ רַ֗ב הָיָ֞ה לִבְנֵ֧י רְאוּבֵ֛ן וְלִבְנֵי־גָ֖ד עָצ֣וּם מְאֹ֑ד וַיִּרְא֞וּ אֶת־אֶ֤רֶץ יַעְזֵר֙ וְאֶת־אֶ֣רֶץ גִּלְעָ֔ד וְהִנֵּ֥ה הַמָּק֖וֹם מְק֥וֹם מִקְנֶֽה׃ לִבְנֵי רְאוּבֵן וְלִבְנֵי גָּד הָיָה עֵדֶר בְּהֵמוֹת גָּדוֹל וְעָצוּם מְאוֹד, הֵם רָאוּ אֶת אֶרֶץ יַעְזֵר וְאֶת אֶרֶץ גִּלְעָד וְהִנֵּה הַמָּקוֹם מָקוֹם טוֹב לְמִרְעֶה.
ב וַיָּבֹ֥אוּ בְנֵֽי־גָ֖ד וּבְנֵ֣י רְאוּבֵ֑ן וַיֹּאמְר֤וּ אֶל־מֹשֶׁה֙ וְאֶל־אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְאֶל־נְשִׂיאֵ֥י הָעֵדָ֖ה לֵאמֹֽר׃ בְּנֵי גָּד וּבְנֵי רְאוּבֵן בָּאוּ וְאָמְרוּ לְמֹשֶׁה וּלְאֶלְעָזָר הַכֹּהֵן וְלִנְשִׂיאֵי הַצִּבּוּר:
ג עֲטָר֤וֹת וְדִיבֹן֙ וְיַעְזֵ֣ר וְנִמְרָ֔ה וְחֶשְׁבּ֖וֹן וְאֶלְעָלֵ֑ה וּשְׂבָ֥ם וּנְב֖וֹ וּבְעֹֽן׃ עֲטָרוֹת וְדִיבוֹן וְיַעְזֵר וְנִמְרָה וְחֶשְׁבּוֹן וְאֶלְעָלֵה וּשְׂבָם וּנְבוֹ וּבְעוֹן –
ד הָאָ֗רֶץ אֲשֶׁ֨ר הִכָּ֤ה יְהֹוָה֙ לִפְנֵי֙ עֲדַ֣ת יִשְׂרָאֵ֔ל אֶ֥רֶץ מִקְנֶ֖ה הִ֑וא וְלַֽעֲבָדֶ֖יךָ מִקְנֶֽה׃ הָאָרֶץ שֶׁה' הִכָּה לִפְנֵי הַצִּבּוּר שֶׁל יִשְׂרָאֵל הִיא אֶרֶץ טוֹבָה לְמִרְעֶה, וְלַעֲבָדֶיךָ יֵשׁ עֶדְרֵי בְּהֵמוֹת.
ס
ה וַיֹּאמְר֗וּ אִם־מָצָ֤אנוּ חֵן֙ בְּעֵינֶ֔יךָ יֻתַּ֞ן אֶת־הָאָ֧רֶץ הַזֹּ֛את לַעֲבָדֶ֖יךָ לַאֲחֻזָּ֑ה אַל־תַּעֲבִרֵ֖נוּ אֶת־הַיַּרְדֵּֽן׃ הֵם אָמְרוּ: אִם מָצָאנוּ חֵן בְּעֵינֶיךָ הָאָרֶץ הַזֹּאת תִּנָּתֵן לַעֲבָדֶיךָ לַאֲחֻזָּתֵנוּ, אַל תַּעֲבִיר אוֹתָנוּ אֶת הַיַּרְדֵּן.
ו וַיֹּ֣אמֶר מֹשֶׁ֔ה לִבְנֵי־גָ֖ד וְלִבְנֵ֣י רְאוּבֵ֑ן הַאַֽחֵיכֶ֗ם יָבֹ֙אוּ֙ לַמִּלְחָמָ֔ה וְאַתֶּ֖ם תֵּ֥שְׁבוּ פֹֽה׃ מֹשֶׁה אָמַר לִבְנֵי גָּד וְלִבְנֵי רְאוּבֵן: הַאִם אֲחֵיכֶם יֵלְכוּ לַמִּלְחָמָה וְאַתֶּם תֵּשְׁבוּ פֹּה?!
ז וְלָ֣מָּה תנואון תְנִיא֔וּן אֶת־לֵ֖ב בְּנֵ֣י יִשְׂרָאֵ֑ל מֵֽעֲבֹר֙ אֶל־הָאָ֔רֶץ אֲשֶׁר־נָתַ֥ן לָהֶ֖ם יְהֹוָֽה׃ לָמָּה אַתֶּם מַסִּיתִים אֶת בְּנֵי יִשְׂרָאֵל שֶׁלֹּא לַעֲבוֹר לָאָרֶץ שֶׁה' נָתַן לָהֶם?!
ח כֹּ֥ה עָשׂ֖וּ אֲבֹתֵיכֶ֑ם בְּשׇׁלְחִ֥י אֹתָ֛ם מִקָּדֵ֥שׁ בַּרְנֵ֖עַ לִרְא֥וֹת אֶת־הָאָֽרֶץ׃ כָּךְ עָשׂוּ אֲבוֹתֵיכֶם כְּשֶׁשָּׁלַחְתִּי אוֹתָם מִקָּדֵשׁ בַּרְנֵעַ לִרְאוֹת אֶת הָאָרֶץ.
ט וַֽיַּעֲל֞וּ עַד־נַ֣חַל אֶשְׁכּ֗וֹל וַיִּרְאוּ֙ אֶת־הָאָ֔רֶץ וַיָּנִ֕יאוּ אֶת־לֵ֖ב בְּנֵ֣י יִשְׂרָאֵ֑ל לְבִלְתִּי־בֹא֙ אֶל־הָאָ֔רֶץ אֲשֶׁר־נָתַ֥ן לָהֶ֖ם יְהֹוָֽה׃ הֵם עָלוּ עַד נַחַל אֶשְׁכּוֹל וְרָאוּ אֶת הָאָרֶץ וְהִסִּיתוּ אֶת בְּנֵי יִשְׂרָאֵל לֹא לַעֲבוֹר לָאָרֶץ שֶׁה' נָתַן לָהֶם.
י וַיִּֽחַר־אַ֥ף יְהֹוָ֖ה בַּיּ֣וֹם הַה֑וּא וַיִּשָּׁבַ֖ע לֵאמֹֽר׃ ה' כָּעַס בְּאוֹתוֹ יוֹם וְנִשְׁבַּע וְאָמַר:
יא אִם־יִרְא֨וּ הָאֲנָשִׁ֜ים הָעֹלִ֣ים מִמִּצְרַ֗יִם מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה אֵ֚ת הָאֲדָמָ֔ה אֲשֶׁ֥ר נִשְׁבַּ֛עְתִּי לְאַבְרָהָ֥ם לְיִצְחָ֖ק וּֽלְיַעֲקֹ֑ב כִּ֥י לֹא־מִלְא֖וּ אַחֲרָֽי׃ הָאֲנָשִׁים שֶׁיָּצְאוּ מִמִּצְרַיִם מִגִּיל עֶשְׂרִים שָׁנָה וָמַעְלָה לֹא יִרְאוּ אֶת הָאֲדָמָה שֶׁנִּשְׁבַּעְתִּי לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב, כִּי הֵם לֹא קִיְּמוּ אֶת מִצְוָתִי.
יב בִּלְתִּ֞י כָּלֵ֤ב בֶּן־יְפֻנֶּה֙ הַקְּנִזִּ֔י וִיהוֹשֻׁ֖עַ בִּן־נ֑וּן כִּ֥י מִלְא֖וּ אַחֲרֵ֥י יְהֹוָֽה׃ חוּץ מִכָּלֵב בֶּן יְפֻנֶּה מִמִּשְׁפַּחַת קְנַז, וִיהוֹשֻׁעַ בִּן נוּן, כִּי הֵם קִיְּמוּ אֶת מִצְוַת ה'.
יג וַיִּֽחַר־אַ֤ף יְהֹוָה֙ בְּיִשְׂרָאֵ֔ל וַיְנִעֵם֙ בַּמִּדְבָּ֔ר אַרְבָּעִ֖ים שָׁנָ֑ה עַד־תֹּם֙ כׇּל־הַדּ֔וֹר הָעֹשֶׂ֥ה הָרַ֖ע בְּעֵינֵ֥י יְהֹוָֽה׃ ה' כָּעַס עַל יִשְׂרָאֵל וְטִלְטֵל אוֹתָם בַּמִּדְבָּר אַרְבָּעִים שָׁנָה עַד שֶׁכָּל הַדּוֹר שֶׁעָשָׂה אֶת הָרַע בְּעֵינֵי ה' מֵתוּ.
יד וְהִנֵּ֣ה קַמְתֶּ֗ם תַּ֚חַת אֲבֹ֣תֵיכֶ֔ם תַּרְבּ֖וּת אֲנָשִׁ֣ים חַטָּאִ֑ים לִסְפּ֣וֹת ע֗וֹד עַ֛ל חֲר֥וֹן אַף־יְהֹוָ֖ה אֶל־יִשְׂרָאֵֽל׃ וְהִנֵּה קַמְתֶּם בִּמְקוֹם אֲבוֹתֵיכֶם – מְחוּנָכִים כַּאֲנָשִׁים חוֹטְאִים, לְהוֹסִיף עוֹד עַל הַכַּעַס שֶׁל ה' עַל יִשְׂרָאֵל.
טו כִּ֤י תְשׁוּבֻן֙ מֵֽאַחֲרָ֔יו וְיָסַ֣ף ע֔וֹד לְהַנִּיח֖וֹ בַּמִּדְבָּ֑ר וְשִֽׁחַתֶּ֖ם לְכׇל־הָעָ֥ם הַזֶּֽה׃ שֶׁכֵּן אַתֶּם סָרִים מִלְּקַיֵּם אֶת מִצְוָתוֹ, וְהוּא יַמְשִׁיךְ עוֹד לְהַשְׁאִיר אֶת הָעָם בַּמִּדְבָּר, וּתְקַלְקְלוּ לְכָל הָעָם הַזֶּה!
ס
טז וַיִּגְּשׁ֤וּ אֵלָיו֙ וַיֹּ֣אמְר֔וּ גִּדְרֹ֥ת צֹ֛אן נִבְנֶ֥ה לְמִקְנֵ֖נוּ פֹּ֑ה וְעָרִ֖ים לְטַפֵּֽנוּ׃ הֵם נִגְּשׁוּ אֵלָיו וְאָמְרוּ: נִבְנֶה פֹּה דִּירִים שֶׁל צֹאן לִבְהֵמוֹתֵנוּ, וְעָרִים לִילָדֵנוּ.
יז וַאֲנַ֜חְנוּ נֵחָלֵ֣ץ חֻשִׁ֗ים לִפְנֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל עַ֛ד אֲשֶׁ֥ר אִם־הֲבִֽיאֹנֻ֖ם אֶל־מְקוֹמָ֑ם וְיָשַׁ֤ב טַפֵּ֙נוּ֙ בְּעָרֵ֣י הַמִּבְצָ֔ר מִפְּנֵ֖י יֹשְׁבֵ֥י הָאָֽרֶץ׃ וַאֲנַחְנוּ נִתְגַּיֵּס וְנִסְתָּעֵר לִפְנֵי בְּנֵי יִשְׂרָאֵל, עַד שֶׁנָּבִיא אוֹתָם לִמְקוֹמָם, וִילָדֵינוּ יָגוּרוּ בְּעָרֵי הַמִּבְצָר הַמֻּגָּנִים מִפְּנֵי תּוֹשְׁבֵי הָאָרֶץ.
יח לֹ֥א נָשׁ֖וּב אֶל־בָּתֵּ֑ינוּ עַ֗ד הִתְנַחֵל֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אִ֖ישׁ נַחֲלָתֽוֹ׃ לֹא נַחֲזוֹר לְבָתֵּינוּ עַד שֶׁבְּנֵי יִשְׂרָאֵל יְקַבְּלוּ כָּל אֶחָד אֶת נַחֲלָתוֹ.
יט כִּ֣י לֹ֤א נִנְחַל֙ אִתָּ֔ם מֵעֵ֥בֶר לַיַּרְדֵּ֖ן וָהָ֑לְאָה כִּ֣י בָ֤אָה נַחֲלָתֵ֙נוּ֙ אֵלֵ֔ינוּ מֵעֵ֥בֶר הַיַּרְדֵּ֖ן מִזְרָֽחָה׃ כִּי לֹא נִתְחַלֵּק בְּנַחֲלָה אִתָּם מֵעֵבֶר לַיַּרְדֵּן, כִּי כְּבָר קִבַּלְנוּ אֶת נַחֲלָתֵנוּ בְּעֵבֶר הַיַּרְדֵּן הַמִּזְרָחִי.

שביעי (רביעי במחובר)[עריכה]

פ
כ וַיֹּ֤אמֶר אֲלֵיהֶם֙ מֹשֶׁ֔ה אִֽם־תַּעֲשׂ֖וּן אֶת־הַדָּבָ֣ר הַזֶּ֑ה אִם־תֵּחָ֥לְצ֛וּ לִפְנֵ֥י יְהֹוָ֖ה לַמִּלְחָמָֽה׃ מֹשֶׁה אָמַר לָהֶם: אִם תַּעֲשׂוּ אֶת הַדָּבָר הַזֶּה – אִם תִּתְגַּיְּסוּ לִפְנֵי עַם ה' לַמִּלְחָמָה.
כא וְעָבַ֨ר לָכֶ֧ם כׇּל־חָל֛וּץ אֶת־הַיַּרְדֵּ֖ן לִפְנֵ֣י יְהֹוָ֑ה עַ֧ד הוֹרִישׁ֛וֹ אֶת־אֹיְבָ֖יו מִפָּנָֽיו׃ וְכָל הַלּוֹחֲמִים שֶׁלָּכֶם יַעַבְרוּ אֶת הַיַּרְדֵּן לִפְנֵי עַם ה' עַד שֶׁה' יַשְׁמִיד אֶת אוֹיְבָיו מִפָּנָיו.
כב וְנִכְבְּשָׁ֨ה הָאָ֜רֶץ לִפְנֵ֤י יְהֹוָה֙ וְאַחַ֣ר תָּשֻׁ֔בוּ וִהְיִיתֶ֧ם נְקִיִּ֛ם מֵיְהֹוָ֖ה וּמִיִּשְׂרָאֵ֑ל וְ֠הָיְתָ֠ה הָאָ֨רֶץ הַזֹּ֥את לָכֶ֛ם לַאֲחֻזָּ֖ה לִפְנֵ֥י יְהֹוָֽה׃ וְהָאָרֶץ תִּכָּבֵשׁ לִפְנֵי עַם ה' וְאַחַר כָּךְ תַּחְזְרוּ, תִּהְיוּ זַכָּאִים בְּעֵינֵי ה' וּבְעֵינֵי יִשְׂרָאֵל וְהָאָרֶץ הַזֹּאת תִּהְיֶה לָכֶם לְחֶלְקְכֶם בַּנַּחֲלָה לִפְנֵי ה'.
כג וְאִם־לֹ֤א תַעֲשׂוּן֙ כֵּ֔ן הִנֵּ֥ה חֲטָאתֶ֖ם לַיהֹוָ֑ה וּדְעוּ֙ חַטַּאתְכֶ֔ם אֲשֶׁ֥ר תִּמְצָ֖א אֶתְכֶֽם׃ וְאִם לֹא תַּעֲשׂוּ כָּךְ – הִנֵּה חֲטָאתֶם לַה', וּדְעוּ שֶׁיַּגִּיעַ אֲלֵיכֶם הָעוֹנֶשׁ שֶׁמַּגִּיעַ לָכֶם.
כד בְּנֽוּ־לָכֶ֤ם עָרִים֙ לְטַפְּכֶ֔ם וּגְדֵרֹ֖ת לְצֹנַאֲכֶ֑ם וְהַיֹּצֵ֥א מִפִּיכֶ֖ם תַּעֲשֽׂוּ׃ בְּנוּ לָכֶם עָרִים לְיַלְדֵיכֶם וְדִירִים לְצֹאנְכֶם, וְתַעֲשׂוּ אֶת מַה שֶּׁאֲמַרְתֶּם.
כה וַיֹּ֤אמֶר בְּנֵי־גָד֙ וּבְנֵ֣י רְאוּבֵ֔ן אֶל־מֹשֶׁ֖ה לֵאמֹ֑ר עֲבָדֶ֣יךָ יַעֲשׂ֔וּ כַּאֲשֶׁ֥ר אֲדֹנִ֖י מְצַוֶּֽה׃ בְּנֵי גָּד וּבְנֵי רְאוּבֵן אָמְרוּ לְמֹשֶׁה: עֲבָדֶיךָ יַעֲשׂוּ אֶת מַה שֶּׁאֲדוֹנִי מְצַוֶּה.
כו טַפֵּ֣נוּ נָשֵׁ֔ינוּ מִקְנֵ֖נוּ וְכׇל־בְּהֶמְתֵּ֑נוּ יִֽהְיוּ־שָׁ֖ם בְּעָרֵ֥י הַגִּלְעָֽד׃ יְלָדֵינוּ, נָשֵׁינוּ, רְכוּשֵׁנוּ, וְכָל בְּהֵמוֹתֵנוּ יִהְיוּ שָׁם בְּעָרֵי הַגִּלְעָד.
כז וַעֲבָדֶ֨יךָ יַֽעַבְר֜וּ כׇּל־חֲל֥וּץ צָבָ֛א לִפְנֵ֥י יְהֹוָ֖ה לַמִּלְחָמָ֑ה כַּאֲשֶׁ֥ר אֲדֹנִ֖י דֹּבֵֽר׃ וַעֲבָדֶיךָ – כָּל לוֹחֵם בַּצָּבָא, יַעַבְרוּ לִפְנֵי עַם ה' לַמִּלְחָמָה, כְּמוֹ שֶׁאֲדוֹנִי אוֹמֵר.
כח וַיְצַ֤ו לָהֶם֙ מֹשֶׁ֔ה אֵ֚ת אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְאֵ֖ת יְהוֹשֻׁ֣עַ בִּן־נ֑וּן וְאֶת־רָאשֵׁ֛י אֲב֥וֹת הַמַּטּ֖וֹת לִבְנֵ֥י יִשְׂרָאֵֽל׃ מֹשֶׁה מִנָּה עֲלֵיהֶם אֶת אֶלְעָזָר הַכֹּהֵן וְאֶת יְהוֹשֻׁעַ בִּן נוּן וְאֶת רָאשֵׁי הַמִּשְׁפָּחוֹת שֶׁל שִׁבְטֵי בְּנֵי יִשְׂרָאֵל.
כט וַיֹּ֨אמֶר מֹשֶׁ֜ה אֲלֵהֶ֗ם אִם־יַעַבְר֣וּ בְנֵי־גָ֣ד וּבְנֵי־רְאוּבֵ֣ן ׀ אִ֠תְּכֶ֠ם אֶֽת־הַיַּרְדֵּ֞ן כׇּל־חָל֤וּץ לַמִּלְחָמָה֙ לִפְנֵ֣י יְהֹוָ֔ה וְנִכְבְּשָׁ֥ה הָאָ֖רֶץ לִפְנֵיכֶ֑ם וּנְתַתֶּ֥ם לָהֶ֛ם אֶת־אֶ֥רֶץ הַגִּלְעָ֖ד לַאֲחֻזָּֽה׃ וְאָמַר לָהֶם מֹשֶׁה: אִם כָּל לוֹחֲמֵי בְּנֵי גָּד וּבְנֵי רְאוּבֵן יַעַבְרוּ אִתְּכֶם אֶת הַיַּרְדֵּן לַמִּלְחָמָה לִפְנֵי עַם ה', וְהָאָרֶץ תִּכָּבֵשׁ לִפְנֵיכֶם – תְּנוּ לָהֶם אֶת אֶרֶץ הַגִּלְעָד לְחֶלְקָם בַּנַּחֲלָה.
ל וְאִם־לֹ֧א יַֽעַבְר֛וּ חֲלוּצִ֖ים אִתְּכֶ֑ם וְנֹֽאחֲז֥וּ בְתֹכְכֶ֖ם בְּאֶ֥רֶץ כְּנָֽעַן׃ וְאִם לֹא יַעַבְרוּ אִתְּכֶם כְּלוֹחֲמִים, הֵם יְקַבְּלוּ נַחֲלָה יַחַד אִתְּכֶם בְּאֶרֶץ כְּנַעַן.
לא וַיַּֽעֲנ֧וּ בְנֵי־גָ֛ד וּבְנֵ֥י רְאוּבֵ֖ן לֵאמֹ֑ר אֵת֩ אֲשֶׁ֨ר דִּבֶּ֧ר יְהֹוָ֛ה אֶל־עֲבָדֶ֖יךָ כֵּ֥ן נַעֲשֶֽׂה׃ בְּנֵי גָּד וּבְנֵי רְאוּבֵן עָנוּ: אֶת מַה שֶּׁה' דִּבֵּר אֶל עֲבָדֶיךָ – כָּךְ נַעֲשֶׂה.
לב נַ֣חְנוּ נַעֲבֹ֧ר חֲלוּצִ֛ים לִפְנֵ֥י יְהֹוָ֖ה אֶ֣רֶץ כְּנָ֑עַן וְאִתָּ֙נוּ֙ אֲחֻזַּ֣ת נַחֲלָתֵ֔נוּ מֵעֵ֖בֶר לַיַּרְדֵּֽן׃ אֲנַחְנוּ נַעֲבוֹר כְּלוֹחֲמִים לִפְנֵי עַם ה' אֶל אֶרֶץ כְּנַעַן, וּנְקַבֵּל אֶת חֶלְקֵנוּ בַּנַּחֲלָה מֵעֵבֶר לַיַּרְדֵּן.
לג וַיִּתֵּ֣ן לָהֶ֣ם ׀ מֹשֶׁ֡ה לִבְנֵי־גָד֩ וְלִבְנֵ֨י רְאוּבֵ֜ן וְלַחֲצִ֣י ׀ שֵׁ֣בֶט ׀ מְנַשֶּׁ֣ה בֶן־יוֹסֵ֗ף אֶת־מַמְלֶ֙כֶת֙ סִיחֹן֙ מֶ֣לֶךְ הָֽאֱמֹרִ֔י וְאֶ֨ת־מַמְלֶ֔כֶת ע֖וֹג מֶ֣לֶךְ הַבָּשָׁ֑ן הָאָ֗רֶץ לְעָרֶ֙יהָ֙ בִּגְבֻלֹ֔ת עָרֵ֥י הָאָ֖רֶץ סָבִֽיב׃ מֹשֶׁה נָתַן לָהֶם – לִבְנֵי גָּד וְלִבְנֵי רְאוּבֵן וְלַחֲצִי שֵׁבֶט מְנַשֶּׁה בֶּן יוֹסֵף, אֶת מַמְלֶכֶת סִיחוֹן מֶלֶךְ הָאֱמוֹרִי וְאֶת מַמְלֶכֶת עוֹג מֶלֶךְ הַבָּשָׁן, וְחִלֵּק בֵּינֵיהֶם אֶת הָאָרֶץ וְאֶת עָרֶיהָ וְצִיֵּן אֶת הַגְּבוּלוֹת, וְאֶת כָּל עָרֵי הָאָרֶץ שֶׁמִּסָּבִיב.
לד וַיִּבְנ֣וּ בְנֵי־גָ֔ד אֶת־דִּיבֹ֖ן וְאֶת־עֲטָרֹ֑ת וְאֵ֖ת עֲרֹעֵֽר׃ בְּנֵי גָּד בָּנוּ אֶת דִּיבוֹן וְאֶת עֲטָרוֹת וְאֶת עֲרוֹעֵר.
לה וְאֶת־עַטְרֹ֥ת שׁוֹפָ֛ן וְאֶת־יַעְזֵ֖ר וְיׇגְבְּהָֽה׃ וְאֶת עַטְרוֹת שׁוֹפָן וְאֶת יַעְזֵר וְהָרָמָה שֶׁלָּהּ.
לו וְאֶת־בֵּ֥ית נִמְרָ֖ה וְאֶת־בֵּ֣ית הָרָ֑ן עָרֵ֥י מִבְצָ֖ר וְגִדְרֹ֥ת צֹֽאן׃ וְאֶת בֵּית נִמְרָה וְאֶת בֵּית הָרָן – כֻּלָּן עָרֵי מִבְצָר וְדִירִים שֶׁל צֹאן.
לז וּבְנֵ֤י רְאוּבֵן֙ בָּנ֔וּ אֶת־חֶשְׁבּ֖וֹן וְאֶת־אֶלְעָלֵ֑א וְאֵ֖ת קִרְיָתָֽיִם׃ וּבְנֵי רְאוּבֵן בָּנוּ אֶת חֶשְׁבּוֹן וְאֶת אֶלְעָלֵא וְאֶת קִרְיָתַיִם.
לח וְאֶת־נְב֞וֹ וְאֶת־בַּ֧עַל מְע֛וֹן מֽוּסַבֹּ֥ת שֵׁ֖ם וְאֶת־שִׂבְמָ֑ה וַיִּקְרְא֣וּ בְשֵׁמֹ֔ת אֶת־שְׁמ֥וֹת הֶעָרִ֖ים אֲשֶׁ֥ר בָּנֽוּ׃ וְאֶת נְבוֹ וְאֶת בַּעַל מְעוֹן – וְשִׁנּוּ אֶת שְׁמוֹתֵיהֶם שֶׁנִּקְרְאוּ עַל שֵׁם אֱלִילִים, וְאֶת שִׂבְמָה, וְנָתְנוּ שֵׁמוֹת חֲדָשִׁים לֶעָרִים שֶׁבָּנוּ.
לט וַיֵּ֨לְכ֜וּ בְּנֵ֨י מָכִ֧יר בֶּן־מְנַשֶּׁ֛ה גִּלְעָ֖דָה וַֽיִּלְכְּדֻ֑הָ וַיּ֖וֹרֶשׁ אֶת־הָאֱמֹרִ֥י אֲשֶׁר־בָּֽהּ׃ בְּנֵי מָכִיר בֶּן מְנַשֶּׁה הָלְכוּ לַגִּלְעָד וְכָבְשׁוּ אוֹתוֹ, וְהִשְׁמִידוּ אֶת הָאֱמוֹרִי שֶׁבּוֹ.

מפטיר[עריכה]

מ וַיִּתֵּ֤ן מֹשֶׁה֙ אֶת־הַגִּלְעָ֔ד לְמָכִ֖יר בֶּן־מְנַשֶּׁ֑ה וַיֵּ֖שֶׁב בָּֽהּ׃ מֹשֶׁה נָתַן אֶת הַגִּלְעָד לְמָכִיר בֶּן מְנַשֶּׁה וְהוּא גָּר בָּהּ.
מא וְיָאִ֤יר בֶּן־מְנַשֶּׁה֙ הָלַ֔ךְ וַיִּלְכֹּ֖ד אֶת־חַוֺּתֵיהֶ֑ם וַיִּקְרָ֥א אֶתְהֶ֖ן חַוֺּ֥ת יָאִֽיר׃ וְיָאִיר בֶּן מְנַשֶּׁה הָלַךְ וְכָבַשׁ אֶת הַחַוּוֹת שֶׁסְּבִיבָם וְקָרָא לָהֵן חַוּוֹת יָאִיר.
מב וְנֹ֣בַח הָלַ֔ךְ וַיִּלְכֹּ֥ד אֶת־קְנָ֖ת וְאֶת־בְּנֹתֶ֑יהָ וַיִּקְרָ֧א לָ֦הֿ נֹ֖בַח בִּשְׁמֽוֹ׃ וְנוֹבַח הָלַךְ וְכָבַשׁ אֶת קְנָת וְאֶת כְּפָרֶיהָ, וְקָרָא לָהּ נוֹבַח בִּשְׁמוֹ.
