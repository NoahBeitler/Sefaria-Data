ראשון[עריכה]

פ פָּרָשַׁת צַוָּאַת מֹשֶׁה
לא א וַיֵּ֖לֶךְ מֹשֶׁ֑ה וַיְדַבֵּ֛ר אֶת־הַדְּבָרִ֥ים הָאֵ֖לֶּה אֶל־כׇּל־יִשְׂרָאֵֽל׃ מֹשֶׁה הָלַךְ וְדִבֵּר אֶת הַדְּבָרִים הָאֵלֶּה אֶל כָּל יִשְׂרָאֵל.
ב וַיֹּ֣אמֶר אֲלֵהֶ֗ם בֶּן־מֵאָה֩ וְעֶשְׂרִ֨ים שָׁנָ֤ה אָנֹכִי֙ הַיּ֔וֹם לֹא־אוּכַ֥ל ע֖וֹד לָצֵ֣את וְלָב֑וֹא וַֽיהֹוָה֙ אָמַ֣ר אֵלַ֔י לֹ֥א תַעֲבֹ֖ר אֶת־הַיַּרְדֵּ֥ן הַזֶּֽה׃ אָמַר לָהֶם: אֲנִי הַיּוֹם בֶּן מֵאָה וְעֶשְׂרִים שָׁנָה לֹא אוּכַל עוֹד לְהַנְהִיגְכֶם בַּיְּצִיאָה לַמִּלְחָמָה וּבַחֲזָרָה, וַה' אָמַר אֵלַי: אַתָּה לֹא תַּעֲבוֹר אֶת הַיַּרְדֵּן הַזֶּה.
ג יְהֹוָ֨ה אֱלֹהֶ֜יךָ ה֣וּא ׀ עֹבֵ֣ר לְפָנֶ֗יךָ הֽוּא־יַשְׁמִ֞יד אֶת־הַגּוֹיִ֥ם הָאֵ֛לֶּה מִלְּפָנֶ֖יךָ וִירִשְׁתָּ֑ם יְהוֹשֻׁ֗עַ ה֚וּא עֹבֵ֣ר לְפָנֶ֔יךָ כַּאֲשֶׁ֖ר דִּבֶּ֥ר יְהֹוָֽה׃ הֲרֵי ה' אֱלֹהֶיךָ הוּא זֶה שֶׁמַּנְהִיג אֶתְכֶם, הוּא יַשְׁמִיד אֶת הָעַמִּים הָאֵלֶּה מִלְּפָנֶיךָ וְתִכְבּוֹשׁ אוֹתָם, וִיהוֹשֻׁעַ הוּא שֶׁיַּנְהִיג אֶתְכֶם בִּמְקוֹמִי כְּמוֹ שֶׁה' דִּבֵּר.

שני[עריכה]

ד וְעָשָׂ֤ה יְהֹוָה֙ לָהֶ֔ם כַּאֲשֶׁ֣ר עָשָׂ֗ה לְסִיח֥וֹן וּלְע֛וֹג מַלְכֵ֥י הָאֱמֹרִ֖י וּלְאַרְצָ֑ם אֲשֶׁ֥ר הִשְׁמִ֖יד אֹתָֽם׃ ה' יַעֲשֶׂה לָהֶם כְּמוֹ שֶׁעָשָׂה לְסִיחוֹן וּלְעוֹג מַלְכֵי הָאֱמוֹרִי וּלְאַרְצָם, שֶׁהִשְׁמִיד אוֹתָם.
ה וּנְתָנָ֥ם יְהֹוָ֖ה לִפְנֵיכֶ֑ם וַעֲשִׂיתֶ֣ם לָהֶ֔ם כְּכׇ֨ל־הַמִּצְוָ֔ה אֲשֶׁ֥ר צִוִּ֖יתִי אֶתְכֶֽם׃ ה' נָתַן אוֹתָם בְּיֶדְכֶם – עֲשׂוּ לָהֶם כְּכָל הַמִּצְוָה שֶׁצִּוִּיתִי אֶתְכֶם.
ו חִזְק֣וּ וְאִמְצ֔וּ אַל־תִּֽירְא֥וּ וְאַל־תַּעַרְצ֖וּ מִפְּנֵיהֶ֑ם כִּ֣י ׀ יְהֹוָ֣ה אֱלֹהֶ֗יךָ ה֚וּא הַהֹלֵ֣ךְ עִמָּ֔ךְ לֹ֥א יַרְפְּךָ֖ וְלֹ֥א יַעַזְבֶֽךָּ׃ חִזְקוּ וְאִמְצוּ אַל תִּפְחֲדוּ וְאַל תִּשָּׁבְרוּ מִפְּנֵיהֶם כִּי ה' אֱלֹהֶיךָ הוּא הַהוֹלֵךְ אִתְּךָ, הוּא לֹא יַעֲזוֹב וְלֹא יִנְטוֹשׁ אוֹתְךָ.

שלישי (חמישי במחוברין)[עריכה]

ס
ז וַיִּקְרָ֨א מֹשֶׁ֜ה לִיהוֹשֻׁ֗עַ וַיֹּ֨אמֶר אֵלָ֜יו לְעֵינֵ֣י כׇל־יִשְׂרָאֵל֮ חֲזַ֣ק וֶאֱמָץ֒ כִּ֣י אַתָּ֗ה תָּבוֹא֙ אֶת־הָעָ֣ם הַזֶּ֔ה אֶל־הָאָ֕רֶץ אֲשֶׁ֨ר נִשְׁבַּ֧ע יְהֹוָ֛ה לַאֲבֹתָ֖ם לָתֵ֣ת לָהֶ֑ם וְאַתָּ֖ה תַּנְחִילֶ֥נָּה אוֹתָֽם׃ מֹשֶׁה קָרָא לִיהוֹשֻׁעַ וְאָמַר לוֹ לְעֵינֵי כָּל יִשְׂרָאֵל: חֲזַק וֶאֱמָץ! כִּי אַתָּה תָּבוֹא עִם הָעָם הַזֶּה לָאָרֶץ שֶׁה' נִשְׁבַּע לַאֲבוֹתֵיהֶם לָתֵת לָהֶם וְאַתָּה תַּנְחִיל לָהֶם אוֹתָהּ.
ח וַיהֹוָ֞ה ה֣וּא ׀ הַהֹלֵ֣ךְ לְפָנֶ֗יךָ ה֚וּא יִהְיֶ֣ה עִמָּ֔ךְ לֹ֥א יַרְפְּךָ֖ וְלֹ֣א יַעַזְבֶ֑ךָּ לֹ֥א תִירָ֖א וְלֹ֥א תֵחָֽת׃ וַה' הוּא הַמַּנְהִיג אוֹתְךָ, הוּא יִהְיֶה אִתְּךָ, הוּא לֹא יַעֲזוֹב וְלֹא יִנְטוֹשׁ אוֹתְךָ, אַל תְּפַחֵד וְאַל תִּשָּׁבֵר.
פָּרָשַׁת מִצְוַת הַקְהֵל
ט וַיִּכְתֹּ֣ב מֹשֶׁה֮ אֶת־הַתּוֹרָ֣ה הַזֹּאת֒ וַֽיִּתְּנָ֗הּ אֶל־הַכֹּֽהֲנִים֙ בְּנֵ֣י לֵוִ֔י הַנֹּ֣שְׂאִ֔ים אֶת־אֲר֖וֹן בְּרִ֣ית יְהֹוָ֑ה וְאֶל־כׇּל־זִקְנֵ֖י יִשְׂרָאֵֽל׃ מֹשֶׁה כָּתַב אֶת הַתּוֹרָה הַזֹּאת, וּנָתַן אוֹתָהּ לַכֹּהֲנִים בְּנֵי לֵוִי שֶׁנּוֹשְׂאִים אֶת אֲרוֹן בְּרִית ה', וּלְכָל זִקְנֵי יִשְׂרָאֵל.

רביעי[עריכה]

י וַיְצַ֥ו מֹשֶׁ֖ה אוֹתָ֣ם לֵאמֹ֑ר מִקֵּ֣ץ ׀ שֶׁ֣בַע שָׁנִ֗ים בְּמֹעֵ֛ד שְׁנַ֥ת הַשְּׁמִטָּ֖ה בְּחַ֥ג הַסֻּכּֽוֹת׃ מֹשֶׁה צִוָּה אוֹתָם וְאָמַר: בְּסוֹף שֶׁבַע שָׁנִים בִּזְמַן שְׁנַת הַשְּׁמִטָּה בְּחַג הַסֻּכּוֹת,
יא בְּב֣וֹא כׇל־יִשְׂרָאֵ֗ל לֵֽרָאוֹת֙ אֶת־פְּנֵי֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ בַּמָּק֖וֹם אֲשֶׁ֣ר יִבְחָ֑ר תִּקְרָ֞א אֶת־הַתּוֹרָ֥ה הַזֹּ֛את נֶ֥גֶד כׇּל־יִשְׂרָאֵ֖ל בְּאׇזְנֵיהֶֽם׃ כְּשֶׁכָּל יִשְׂרָאֵל יָבוֹאוּ לְהֵרָאוֹת לִפְנֵי ה' אֱלֹהֶיךָ בַּמָּקוֹם שֶׁיִּבְחַר, תִּקְרָא אֶת הַתּוֹרָה הַזֹּאת מוּל כָּל יִשְׂרָאֵל בְּאָזְנֵיהֶם.
יב הַקְהֵ֣ל אֶת־הָעָ֗ם הָֽאֲנָשִׁ֤ים וְהַנָּשִׁים֙ וְהַטַּ֔ף וְגֵרְךָ֖ אֲשֶׁ֣ר בִּשְׁעָרֶ֑יךָ לְמַ֨עַן יִשְׁמְע֜וּ וּלְמַ֣עַן יִלְמְד֗וּ וְיָֽרְאוּ֙ אֶת־יְהֹוָ֣ה אֱלֹהֵיכֶ֔ם וְשָֽׁמְר֣וּ לַעֲשׂ֔וֹת אֶת־כׇּל־דִּבְרֵ֖י הַתּוֹרָ֥ה הַזֹּֽאת׃ הַקְהֵל אֶת הָעָם – הָאֲנָשִׁים וְהַנָּשִׁים וְהַיְּלָדִים וְהַגֵּרִים שֶׁבְּעָרֶיךָ, כְּדֵי שֶׁיִּשְׁמְעוּ וּכְדֵי שֶׁיִּלְמְדוּ וִיכַבְּדוּ אֶת ה' אֱלֹהֵיכֶם, וְיַקְפִּידוּ לְקַיֵּם אֶת כָּל דִּבְרֵי הַתּוֹרָה הַזֹּאת.
יג וּבְנֵיהֶ֞ם אֲשֶׁ֣ר לֹא־יָדְע֗וּ יִשְׁמְעוּ֙ וְלָ֣מְד֔וּ לְיִרְאָ֖ה אֶת־יְהֹוָ֣ה אֱלֹהֵיכֶ֑ם כׇּל־הַיָּמִ֗ים אֲשֶׁ֨ר אַתֶּ֤ם חַיִּים֙ עַל־הָ֣אֲדָמָ֔ה אֲשֶׁ֨ר אַתֶּ֜ם עֹבְרִ֧ים אֶת־הַיַּרְדֵּ֛ן שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ וּבְנֵיהֶם שֶׁלֹּא יָדְעוּ, יִשְׁמְעוּ וְיִלְמְדוּ לְכַבֵּד אֶת ה' אֱלֹהֵיכֶם, כָּל הַיָּמִים שֶׁאַתֶּם חַיִּים עַל הָאֲדָמָה שֶׁאַתֶּם עוֹבְרִים אֶת הַיַּרְדֵּן לְשָׁם לִכְבּוֹשׁ אוֹתָהּ.

חמישי (ששי במחוברין)[עריכה]

פ פָּרָשַׁת נְבוּאַת הַגָּלוּת
יד וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה הֵ֣ן קָרְב֣וּ יָמֶ֘יךָ֮ לָמוּת֒ קְרָ֣א אֶת־יְהוֹשֻׁ֗עַ וְהִֽתְיַצְּב֛וּ בְּאֹ֥הֶל מוֹעֵ֖ד וַאֲצַוֶּ֑נּוּ וַיֵּ֤לֶךְ מֹשֶׁה֙ וִיהוֹשֻׁ֔עַ וַיִּֽתְיַצְּב֖וּ בְּאֹ֥הֶל מוֹעֵֽד׃ ה' אָמַר לְמֹשֶׁה: הֲרֵי זְמַן מוֹתְךָ מִתְקָרֵב, קְרָא לִיהוֹשֻׁעַ וְתִתְיַצְּבוּ בְּאוֹהֶל הַהִתְכַּנְּסוּת וַאֲצַוֶּה אוֹתוֹ, מֹשֶׁה וִיהוֹשֻׁעַ הָלְכוּ וְהִתְיַצְּבוּ בְּאוֹהֶל הַהִתְכַּנְּסוּת.
טו וַיֵּרָ֧א יְהֹוָ֛ה בָּאֹ֖הֶל בְּעַמּ֣וּד עָנָ֑ן וַֽיַּעֲמֹ֛ד עַמּ֥וּד הֶעָנָ֖ן עַל־פֶּ֥תַח הָאֹֽהֶל׃ כְּבוֹד ה' הִתְגַּלָּה בָּאוֹהֶל בְּעַמּוּד עָנָן, וְעַמּוּד הֶעָנָן עָמַד מֵעַל פֶּתַח הָאוֹהֶל.
טז וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה הִנְּךָ֥ שֹׁכֵ֖ב עִם־אֲבֹתֶ֑יךָ וְקָם֩ הָעָ֨ם הַזֶּ֜ה וְזָנָ֣ה ׀ אַחֲרֵ֣י ׀ אֱלֹהֵ֣י נֵכַר־הָאָ֗רֶץ אֲשֶׁ֨ר ה֤וּא בָא־שָׁ֙מָּה֙ בְּקִרְבּ֔וֹ וַעֲזָבַ֕נִי וְהֵפֵר֙ אֶת־בְּרִיתִ֔י אֲשֶׁ֥ר כָּרַ֖תִּי אִתּֽוֹ׃ ה' אָמַר לְמֹשֶׁה: הִנֵּה אַתָּה מֵת וּמִצְטָרֵף לַאֲבוֹתֶיךָ, וְהָעָם הַזֶּה יָקוּם וְיִטְעֶה אַחֲרֵי אֱלִילֵי גּוֹיֵי הָאָרֶץ שֶׁהוּא בָּא לְשָׁם בְּתוֹכָם, וְהוּא יַעֲזוֹב אוֹתִי וְיָפֵר אֶת הַהֶסְכֵּם שֶׁלִּי שֶׁעָשִׂיתִי אִתּוֹ.
יז וְחָרָ֣ה אַפִּ֣י ב֣וֹ בַיּוֹם־הַ֠ה֠וּא וַעֲזַבְתִּ֞ים וְהִסְתַּרְתִּ֨י פָנַ֤י מֵהֶם֙ וְהָיָ֣ה לֶאֱכֹ֔ל וּמְצָאֻ֛הוּ רָע֥וֹת רַבּ֖וֹת וְצָר֑וֹת וְאָמַר֙ בַּיּ֣וֹם הַה֔וּא הֲלֹ֗א עַ֣ל כִּי־אֵ֤ין אֱלֹהַי֙ בְּקִרְבִּ֔י מְצָא֖וּנִי הָרָע֥וֹת הָאֵֽלֶּה׃ וְאֶכְעַס עָלָיו בְּאוֹתוֹ זְמָן וְאֶעֱזוֹב אוֹתָם וְאַסְתִּיר אֶת פָּנַי מֵהֶם וְהַגּוֹיִם יִכְבְּשׁוּ אוֹתָם, וְיִקְרוּ לוֹ רָעוֹת רַבּוֹת וְצָרוֹת, וְהוּא יֹאמַר בְּאוֹתוֹ זְמָן: הֲרֵי בִּגְלַל שֶׁאֱלֹהַי אֵינוֹ שׁוֹכֵן בְּתוֹכִי קָרוּ לִי הָרָעוֹת הָאֵלֶּה.
יח וְאָנֹכִ֗י הַסְתֵּ֨ר אַסְתִּ֤יר פָּנַי֙ בַּיּ֣וֹם הַה֔וּא עַ֥ל כׇּל־הָרָעָ֖ה אֲשֶׁ֣ר עָשָׂ֑ה כִּ֣י פָנָ֔ה אֶל־אֱלֹהִ֖ים אֲחֵרִֽים׃ וַאֲנִי אַסְתִּיר אֶת פָּנַי בְּאוֹתוֹ זְמָן, בִּגְלַל כָּל הָרָעָה שֶׁעָשָׂה, שֶׁפָּנָה אֶל אֱלֹהִים אֲחֵרִים.
יט וְעַתָּ֗ה כִּתְב֤וּ לָכֶם֙ אֶת־הַשִּׁירָ֣ה הַזֹּ֔את וְלַמְּדָ֥הּ אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל שִׂימָ֣הּ בְּפִיהֶ֑ם לְמַ֨עַן תִּֽהְיֶה־לִּ֜י הַשִּׁירָ֥ה הַזֹּ֛את לְעֵ֖ד בִּבְנֵ֥י יִשְׂרָאֵֽל׃ וְעַכְשָׁיו, כִּתְבוּ לָכֶם אֶת הַשִּׁירָה הַזֹּאת, וְלַמֵּד אוֹתָהּ אֶת בְּנֵי יִשְׂרָאֵל – שִׂים אוֹתָהּ בְּפִיהֶם, כְּדֵי שֶׁהַשִּׁירָה הַזֹּאת תְּשַׁמֵּשׁ לִי כְּעֵד שֶׁיָּעִיד בִּפְנֵי בְּנֵי יִשְׂרָאֵל.

ששי (שביעי במחוברין)[עריכה]

כ כִּֽי־אֲבִיאֶ֜נּוּ אֶֽל־הָאֲדָמָ֣ה ׀ אֲשֶׁר־נִשְׁבַּ֣עְתִּי לַאֲבֹתָ֗יו זָבַ֤ת חָלָב֙ וּדְבַ֔שׁ וְאָכַ֥ל וְשָׂבַ֖ע וְדָשֵׁ֑ן וּפָנָ֞ה אֶל־אֱלֹהִ֤ים אֲחֵרִים֙ וַעֲבָד֔וּם וְנִ֣אֲצ֔וּנִי וְהֵפֵ֖ר אֶת־בְּרִיתִֽי׃ כְּשֶׁאֲבִיא אוֹתוֹ לָאֲדָמָה שֶׁנִּשְׁבַּעְתִּי לַאֲבוֹתָיו – אֶרֶץ שׁוֹפַעַת חָלָב וּדְבַשׁ, וְיֹאכַל וְיִשְׂבַּע וְיִתְפַּנֵּק, וְיִפְנֶה אֶל אֱלֹהִים אֲחֵרִים וְיַעַבְדוּ אוֹתָם, וְיַרְגִּיזוּ אוֹתִי, וְיָפֵרוּ אֶת הַהֶסְכֵּם אִתִּי.
כא וְ֠הָיָ֠ה כִּֽי־תִמְצֶ֨אןָ אֹת֜וֹ רָע֣וֹת רַבּוֹת֮ וְצָרוֹת֒ וְ֠עָנְתָ֠ה הַשִּׁירָ֨ה הַזֹּ֤את לְפָנָיו֙ לְעֵ֔ד כִּ֛י לֹ֥א תִשָּׁכַ֖ח מִפִּ֣י זַרְע֑וֹ כִּ֧י יָדַ֣עְתִּי אֶת־יִצְר֗וֹ אֲשֶׁ֨ר ה֤וּא עֹשֶׂה֙ הַיּ֔וֹם בְּטֶ֣רֶם אֲבִיאֶ֔נּוּ אֶל־הָאָ֖רֶץ אֲשֶׁ֥ר נִשְׁבַּֽעְתִּי׃ כְּשֶׁיִּקְרוּ לוֹ רָעוֹת רַבּוֹת וְצָרוֹת, תַּעֲנֶה אָז הַשִּׁירָה הַזֹּאת לְפָנָיו וְתָעִיד שֶׁהִזְהַרְתִּי אוֹתָם, שֶׁכֵּן הִיא לֹא תִּשָּׁכַח מִפִּי צֶאֱצָאָיו; כִּי יָדַעְתִּי אֶת יִצְרוֹ הָרַע וְאֶת מַה שֶּׁהוּא עוֹשֶׂה כְּבָר הַיּוֹם – לִפְנֵי שֶׁהֱבֵאתִי אוֹתוֹ לָאָרֶץ שֶׁנִּשְׁבַּעְתִּי.
כב וַיִּכְתֹּ֥ב מֹשֶׁ֛ה אֶת־הַשִּׁירָ֥ה הַזֹּ֖את בַּיּ֣וֹם הַה֑וּא וַֽיְלַמְּדָ֖הּ אֶת־בְּנֵ֥י יִשְׂרָאֵֽל׃ מֹשֶׁה כָּתַב אֶת הַשִּׁירָה הַזֹּאת בַּיּוֹם הַהוּא, וְלִמֵּד אוֹתָהּ אֶת בְּנֵי יִשְׂרָאֵל.
כג וַיְצַ֞ו אֶת־יְהוֹשֻׁ֣עַ בִּן־נ֗וּן וַיֹּ֘אמֶר֮ חֲזַ֣ק וֶאֱמָץ֒ כִּ֣י אַתָּ֗ה תָּבִיא֙ אֶת־בְּנֵ֣י יִשְׂרָאֵ֔ל אֶל־הָאָ֖רֶץ אֲשֶׁר־נִשְׁבַּ֣עְתִּי לָהֶ֑ם וְאָנֹכִ֖י אֶהְיֶ֥ה עִמָּֽךְ׃ הוּא צִוָּה אֶת יְהוֹשֻׁעַ בִּן נוּן, וְאָמַר: חֲזַק וֶאֱמָץ! כִּי אַתָּה תָּבִיא אֶת בְּנֵי יִשְׂרָאֵל לָאָרֶץ שֶׁנִּשְׁבַּעְתִּי לָהֶם וַאֲנִי אֶהְיֶה אִתְּךָ.
כד וַיְהִ֣י ׀ כְּכַלּ֣וֹת מֹשֶׁ֗ה לִכְתֹּ֛ב אֶת־דִּבְרֵ֥י הַתּוֹרָֽה־הַזֹּ֖את עַל־סֵ֑פֶר עַ֖ד תֻּמָּֽם׃ כְּשֶׁמֹּשֶׁה גָּמַר לִכְתּוֹב אֶת דִּבְרֵי הַתּוֹרָה הַזֹּאת עַל סֵפֶר עַד סוֹפָם,

שביעי[עריכה]

כה וַיְצַ֤ו מֹשֶׁה֙ אֶת־הַלְוִיִּ֔ם נֹ֥שְׂאֵ֛י אֲר֥וֹן בְּרִית־יְהֹוָ֖ה לֵאמֹֽר׃ מֹשֶׁה צִוָּה אֶת הַלְוִיִּים נוֹשְׂאֵי אֲרוֹן בְּרִית ה' וְאָמַר:
כו לָקֹ֗חַ אֵ֣ת סֵ֤פֶר הַתּוֹרָה֙ הַזֶּ֔ה וְשַׂמְתֶּ֣ם אֹת֔וֹ מִצַּ֛ד אֲר֥וֹן בְּרִית־יְהֹוָ֖ה אֱלֹהֵיכֶ֑ם וְהָיָה־שָׁ֥ם בְּךָ֖ לְעֵֽד׃ קְחוּ אֶת סֵפֶר הַתּוֹרָה הַזֶּה וְשִׂימוּ אוֹתוֹ בְּצַד אֲרוֹן בְּרִית ה' אֱלֹהֵיכֶם, וְיִהְיֶה שָׁם אֶצְלְכֶם לְעֵד שֶׁהִזְהַרְתִּי אֶתְכֶם.
כז כִּ֣י אָנֹכִ֤י יָדַ֙עְתִּי֙ אֶֽת־מֶרְיְךָ֔ וְאֶֽת־עׇרְפְּךָ֖ הַקָּשֶׁ֑ה הֵ֣ן בְּעוֹדֶ֩נִּי֩ חַ֨י עִמָּכֶ֜ם הַיּ֗וֹם מַמְרִ֤ים הֱיִתֶם֙ עִם־יְהֹוָ֔ה וְאַ֖ף כִּי־אַחֲרֵ֥י מוֹתִֽי׃ כִּי אֲנִי יָדַעְתִּי אֶת סַרְבָנוּתְכֶם וְאֶת עַקְשָׁנוּתְכֶם, הֲרֵי גַּם הַיּוֹם כְּשֶׁאֲנִי עֲדַיִן חַי אִתְּכֶם הֲיִיתֶם מוֹרְדִים בְּצִוּוּי ה', וְכָל שֶׁכֵּן שֶׁיִּהְיֶה כָּךְ אַחֲרֵי מוֹתִי.

מפטיר[עריכה]

כח הַקְהִ֧ילוּ אֵלַ֛י אֶת־כׇּל־זִקְנֵ֥י שִׁבְטֵיכֶ֖ם וְשֹׁטְרֵיכֶ֑ם וַאֲדַבְּרָ֣ה בְאׇזְנֵיהֶ֗ם אֵ֚ת הַדְּבָרִ֣ים הָאֵ֔לֶּה וְאָעִ֣ידָה בָּ֔ם אֶת־הַשָּׁמַ֖יִם וְאֶת־הָאָֽרֶץ׃ אִסְפוּ אֵלַי אֶת כָּל זִקְנֵי שִׁבְטֵיכֶם וְשׁוֹטְרֵיכֶם, וַאֲדַבֵּר בְּאָזְנֵיהֶם אֶת הַדְּבָרִים הָאֵלֶּה, וְאַתְרֶה בָּהֶם בִּפְנֵי הָעֵדִים – הַשָּׁמַיִם וְהָאָרֶץ.
כט כִּ֣י יָדַ֗עְתִּי אַחֲרֵ֤י מוֹתִי֙ כִּֽי־הַשְׁחֵ֣ת תַּשְׁחִת֔וּן וְסַרְתֶּ֣ם מִן־הַדֶּ֔רֶךְ אֲשֶׁ֥ר צִוִּ֖יתִי אֶתְכֶ֑ם וְקָרָ֨את אֶתְכֶ֤ם הָרָעָה֙ בְּאַחֲרִ֣ית הַיָּמִ֔ים כִּֽי־תַעֲשׂ֤וּ אֶת־הָרַע֙ בְּעֵינֵ֣י יְהֹוָ֔ה לְהַכְעִיס֖וֹ בְּמַעֲשֵׂ֥ה יְדֵיכֶֽם׃ כִּי יָדַעְתִּי שֶׁאַחֲרֵי מוֹתִי תַּשְׁחִיתוּ וְתִסְטוּ מֵהַדֶּרֶךְ שֶׁצִּוִּיתִי אֶתְכֶם, וְהָרָעָה שֶׁאָמַרְתִּי תִּקְרֶה לָכֶם לְאַחַר זְמָן, כְּשֶׁתַּעֲשׂוּ אֶת הָרַע בְּעֵינֵי ה' לְהַכְעִיס אוֹתוֹ בְּמַעֲשֵׂה יְדֵיכֶם.
ל וַיְדַבֵּ֣ר מֹשֶׁ֗ה בְּאׇזְנֵי֙ כׇּל־קְהַ֣ל יִשְׂרָאֵ֔ל אֶת־דִּבְרֵ֥י הַשִּׁירָ֖ה הַזֹּ֑את עַ֖ד תֻּמָּֽם׃ מֹשֶׁה דִּבֵּר בְּאָזְנֵי כָּל קְהַל יִשְׂרָאֵל אֶת דִּבְרֵי הַשִּׁירָה הַזֹּאת עַד סוֹפָם.
