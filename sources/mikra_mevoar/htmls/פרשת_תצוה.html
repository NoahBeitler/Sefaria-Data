ראשון[עריכה]

ס פָּרָשַׁת הַשֶּׁמֶן לַמָּאוֹר
כז כ וְאַתָּ֞ה תְּצַוֶּ֣ה ׀ אֶת־בְּנֵ֣י יִשְׂרָאֵ֗ל וְיִקְח֨וּ אֵלֶ֜יךָ שֶׁ֣מֶן זַ֥יִת זָ֛ךְ כָּתִ֖ית לַמָּא֑וֹר לְהַעֲלֹ֥ת נֵ֖ר תָּמִֽיד׃ וְאַתָּה תְּצַוֶּה אֶת בְּנֵי יִשְׂרָאֵל שֶׁיָּבִיאוּ אֵלֶיךָ שֶׁמֶן זַיִת זָךְ כָּתוּשׁ לַתְּאוּרָה – לְהַדְלִיק נֵר קָבוּעַ.
כא בְּאֹ֣הֶל מוֹעֵד֩ מִח֨וּץ לַפָּרֹ֜כֶת אֲשֶׁ֣ר עַל־הָעֵדֻ֗ת יַעֲרֹךְ֩ אֹת֨וֹ אַהֲרֹ֧ן וּבָנָ֛יו מֵעֶ֥רֶב עַד־בֹּ֖קֶר לִפְנֵ֣י יְהֹוָ֑ה חֻקַּ֤ת עוֹלָם֙ לְדֹ֣רֹתָ֔ם מֵאֵ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ אַהֲרֹן וּבָנָיו יַעַרְכוּ אוֹתוֹ בְּאוֹהֶל הַהִתְכַּנְּסוּת מֵעֵבֶר לַפָּרוֹכֶת שֶׁעַל יַד אֲרוֹן לוּחוֹת הַבְּרִית מִבַּחוּץ, שֶׁיִּדְלוֹק מֵהָעֶרֶב עַד הַבּוֹקֶר לִפְנֵי ה', חוֹק לְעוֹלָם לְדוֹרוֹת שֶׁיִּנָּתֵן מִבְּנֵי יִשְׂרָאֵל.
ס פָּרָשַׁת בִּגְדֵי הַכְּהֻנָּה
כח א וְאַתָּ֡ה הַקְרֵ֣ב אֵלֶ֩יךָ֩ אֶת־אַהֲרֹ֨ן אָחִ֜יךָ וְאֶת־בָּנָ֣יו אִתּ֗וֹ מִתּ֛וֹךְ בְּנֵ֥י יִשְׂרָאֵ֖ל לְכַהֲנוֹ־לִ֑י אַהֲרֹ֕ן נָדָ֧ב וַאֲבִיה֛וּא אֶלְעָזָ֥ר וְאִיתָמָ֖ר בְּנֵ֥י אַהֲרֹֽן׃ וְאַתָּה תְּקָרֵב אֵלֶיךָ מִתּוֹךְ בְּנֵי יִשְׂרָאֵל אֶת אַהֲרֹן אָחִיךָ וְאֶת בָּנָיו אִתּוֹ שֶׁיִּהְיוּ הַכֹּהֲנִים שֶׁלִּי – אַהֲרֹן, וְנָדָב אֲבִיהוּא אֶלְעָזָר וְאִיתָמָר בְּנֵי אַהֲרֹן.
ב וְעָשִׂ֥יתָ בִגְדֵי־קֹ֖דֶשׁ לְאַהֲרֹ֣ן אָחִ֑יךָ לְכָב֖וֹד וּלְתִפְאָֽרֶת׃ וְתַעֲשֶׂה בִּגְדֵי קוֹדֶשׁ לְאַהֲרֹן אָחִיךָ לְכָבוֹד וּלְתִפְאֶרֶת.
ג וְאַתָּ֗ה תְּדַבֵּר֙ אֶל־כׇּל־חַכְמֵי־לֵ֔ב אֲשֶׁ֥ר מִלֵּאתִ֖יו ר֣וּחַ חׇכְמָ֑ה וְעָשׂ֞וּ אֶת־בִּגְדֵ֧י אַהֲרֹ֛ן לְקַדְּשׁ֖וֹ לְכַהֲנוֹ־לִֽי׃ וְאַתָּה תְּדַבֵּר אֶל כָּל הַחֲכָמִים שֶׁמִּלֵּאתִי אוֹתָם בְּרוּחַ חָכְמָה שֶׁיַּעֲשׂוּ אֶת בִּגְדֵי אַהֲרֹן לְקַדֵּשׁ אוֹתוֹ שֶׁיִּהְיֶה כֹּהֵן לַעֲבוֹדָתִי.
ד וְאֵ֨לֶּה הַבְּגָדִ֜ים אֲשֶׁ֣ר יַעֲשׂ֗וּ חֹ֤שֶׁן וְאֵפוֹד֙ וּמְעִ֔יל וּכְתֹ֥נֶת תַּשְׁבֵּ֖ץ מִצְנֶ֣פֶת וְאַבְנֵ֑ט וְעָשׂ֨וּ בִגְדֵי־קֹ֜דֶשׁ לְאַהֲרֹ֥ן אָחִ֛יךָ וּלְבָנָ֖יו לְכַהֲנוֹ־לִֽי׃ וְאֵלֶּה הַבְּגָדִים שֶׁיַּעֲשׂוּ: חוֹשֶׁן וְאֵפוֹד וּמְעִיל וּכְתוֹנֶת מְשֻׁבֶּצֶת, מִצְנֶפֶת וַחֲגוֹרָה, וְיַעֲשׂוּ אֶת בִּגְדֵי הַקּוֹדֶשׁ הָאֵלֶּה לְאַהֲרֹן אָחִיךָ וּלְבָנָיו שֶׁיִּהְיוּ כֹּהֲנִים לַעֲבוֹדָתִי.
ה וְהֵם֙ יִקְח֣וּ אֶת־הַזָּהָ֔ב וְאֶת־הַתְּכֵ֖לֶת וְאֶת־הָֽאַרְגָּמָ֑ן וְאֶת־תּוֹלַ֥עַת הַשָּׁנִ֖י וְאֶת־הַשֵּֽׁשׁ׃ וְהֵם יִקְחוּ אֶת הַזָּהָב וְאֶת הַתְּכֵלֶת וְאֶת הָאַרְגָּמָן וְאֶת הַצֶּבַע הָאָדוֹם מֵהַתּוֹלַעַת וְאֶת הַפִּשְׁתָּן.
פ פָּרָשַׁת הָאֵפוֹד
ו וְעָשׂ֖וּ אֶת־הָאֵפֹ֑ד זָ֠הָ֠ב תְּכֵ֨לֶת וְאַרְגָּמָ֜ן תּוֹלַ֧עַת שָׁנִ֛י וְשֵׁ֥שׁ מׇשְׁזָ֖ר מַעֲשֵׂ֥ה חֹשֵֽׁב׃ יַעֲשׂוּ אֶת הָאֵפוֹד מֵחוּטֵי זָהָב תְּכֵלֶת וְאַרְגָּמָן, צֶבַע אָדוֹם מִתּוֹלַעַת וּפִשְׁתָּן שָׁזוּר, בִּמְלֶאכֶת אֳמָנוּת.
ז שְׁתֵּ֧י כְתֵפֹ֣ת חֹֽבְרֹ֗ת יִֽהְיֶה־לּ֛וֹ אֶל־שְׁנֵ֥י קְצוֹתָ֖יו וְחֻבָּֽר׃ יִהְיוּ לוֹ שְׁתֵּי כְּתֵפִיּוֹת מְחֻבָּרוֹת אֵלָיו בִּשְׁנֵי קְצוֹתָיו, וִיחַבְּרוּ אוֹתָן בִּתְפִירָה.
ח וְחֵ֤שֶׁב אֲפֻדָּתוֹ֙ אֲשֶׁ֣ר עָלָ֔יו כְּמַעֲשֵׂ֖הוּ מִמֶּ֣נּוּ יִהְיֶ֑ה זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ מׇשְׁזָֽר׃ וַחֲגוֹרַת אֲפֻדָּתוֹ שֶׁעָלָיו תֵּעָשֶׂה כָּמוֹהוּ וְתִהְיֶה חֵלֶק מִמֶּנּוּ, מִזָּהָב תְּכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת וּפִשְׁתָּן שָׁזוּר.
ט וְלָ֣קַחְתָּ֔ אֶת־שְׁתֵּ֖י אַבְנֵי־שֹׁ֑הַם וּפִתַּחְתָּ֣ עֲלֵיהֶ֔ם שְׁמ֖וֹת בְּנֵ֥י יִשְׂרָאֵֽל׃ וְתִקַּח אֶת שְׁתֵּי אַבְנֵי הַשּׁוֹהַם וְתַחֲקוֹק עֲלֵיהֶן אֶת שְׁמוֹת בְּנֵי יִשְׂרָאֵל.
י שִׁשָּׁה֙ מִשְּׁמֹתָ֔ם עַ֖ל הָאֶ֣בֶן הָאֶחָ֑ת וְאֶת־שְׁמ֞וֹת הַשִּׁשָּׁ֧ה הַנּוֹתָרִ֛ים עַל־הָאֶ֥בֶן הַשֵּׁנִ֖ית כְּתוֹלְדֹתָֽם׃ שִׁשָּׁה מִשְּׁמוֹתָם עַל אַחַת הָאֲבָנִים וְאֶת שְׁמוֹת הַשִּׁשָּׁה הַנּוֹתָרִים עַל הָאֶבֶן הַשְּׁנִיָּה, כְּסֵדֶר לֵידָתָם.
יא מַעֲשֵׂ֣ה חָרַשׁ֮ אֶ֒בֶן֒ פִּתּוּחֵ֣י חֹתָ֗ם תְּפַתַּח֙ אֶת־שְׁתֵּ֣י הָאֲבָנִ֔ים עַל־שְׁמֹ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל מֻסַבֹּ֛ת מִשְׁבְּצ֥וֹת זָהָ֖ב תַּעֲשֶׂ֥ה אֹתָֽם׃ תַּחֲקוֹק בִּשְׁתֵּי הָאֲבָנִים אֶת שְׁמוֹת בְּנֵי יִשְׂרָאֵל חֲקוּקִים כְּחוֹתֶמֶת בִּמְלֶאכֶת אֳמָנוּת שֶׁל אַבְנֵי חֵן, תַּעֲשֶׂה אוֹתָן מֻקָּפוֹת בְּמִשְׁבְּצוֹת זָהָב.
יב וְשַׂמְתָּ֞ אֶת־שְׁתֵּ֣י הָאֲבָנִ֗ים עַ֚ל כִּתְפֹ֣ת הָֽאֵפֹ֔ד אַבְנֵ֥י זִכָּרֹ֖ן לִבְנֵ֣י יִשְׂרָאֵ֑ל וְנָשָׂא֩ אַהֲרֹ֨ן אֶת־שְׁמוֹתָ֜ם לִפְנֵ֧י יְהֹוָ֛ה עַל־שְׁתֵּ֥י כְתֵפָ֖יו לְזִכָּרֹֽן׃ תָּשִׂים אֶת שְׁתֵּי הָאֲבָנִים עַל כְּתֵפִיּוֹת הָאֵפוֹד – אַבְנֵי זִכָּרוֹן לִבְנֵי יִשְׂרָאֵל, וְאַהֲרֹן יִשָּׂא אֶת שְׁמוֹתָם עַל שְׁתֵּי כְּתֵפָיו לְזִכָּרוֹן לִפְנֵי ה'.

שני[עריכה]

ס
יג וְעָשִׂ֥יתָ מִשְׁבְּצֹ֖ת זָהָֽב׃ תַּעֲשֶׂה מִשְׁבְּצוֹת זָהָב.
יד וּשְׁתֵּ֤י שַׁרְשְׁרֹת֙ זָהָ֣ב טָה֔וֹר מִגְבָּלֹ֛ת תַּעֲשֶׂ֥ה אֹתָ֖ם מַעֲשֵׂ֣ה עֲבֹ֑ת וְנָתַתָּ֛ה אֶת־שַׁרְשְׁרֹ֥ת הָעֲבֹתֹ֖ת עַל־הַֽמִּשְׁבְּצֹֽת׃ וּשְׁתֵּי שַׁרְשְׁרוֹת מִזָּהָב טָהוֹר – תַּעֲשֶׂה אוֹתָן בַּקָּצֶה, שְׁזוּרוֹת כְּחֶבֶל, וְתָשִׂים אֶת שַׁרְשְׁרוֹת הַשְּׂרוֹכִים עַל הַמִּשְׁבְּצוֹת.
ס פָּרָשַׁת הַחֹשֶׁן
טו וְעָשִׂ֜יתָ חֹ֤שֶׁן מִשְׁפָּט֙ מַעֲשֵׂ֣ה חֹשֵׁ֔ב כְּמַעֲשֵׂ֥ה אֵפֹ֖ד תַּעֲשֶׂ֑נּוּ זָ֠הָ֠ב תְּכֵ֨לֶת וְאַרְגָּמָ֜ן וְתוֹלַ֧עַת שָׁנִ֛י וְשֵׁ֥שׁ מׇשְׁזָ֖ר תַּעֲשֶׂ֥ה אֹתֽוֹ׃ תַּעֲשֶׂה בִּמְלֶאכֶת אֳמָנוּת חוֹשֶׁן שֶׁמּוֹדִיעַ מַה לַּעֲשׂוֹת, תַּעֲשֶׂה אוֹתוֹ כְּמוֹ מְלֶאכֶת הָאֵפוֹד – תַּעֲשֶׂה אוֹתוֹ מִזָּהָב תְּכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת וּפִשְׁתָּן שָׁזוּר.
טז רָב֥וּעַ יִֽהְיֶ֖ה כָּפ֑וּל זֶ֥רֶת אׇרְכּ֖וֹ וְזֶ֥רֶת רׇחְבּֽוֹ׃ הוּא יִהְיֶה רִבּוּעַ מְקֻפָּל לִשְׁנַיִם, חֲצִי אַמָּה אָרְכּוֹ וַחֲצִי אַמָּה רָחְבּוֹ (כְּ24 ס"מ).
יז וּמִלֵּאתָ֥ בוֹ֙ מִלֻּ֣אַת אֶ֔בֶן אַרְבָּעָ֖ה טוּרִ֣ים אָ֑בֶן ט֗וּר אֹ֤דֶם פִּטְדָה֙ וּבָרֶ֔קֶת הַטּ֖וּר הָאֶחָֽד׃ תִּקְבַּע בּוֹ אֲבָנִים – אַרְבָּעָה טוּרִים שֶׁל אֲבָנִים, הַטּוּר הָרִאשׁוֹן: אוֹדֶם פִּטְדָה וּבָרֶקֶת.
יח וְהַטּ֖וּר הַשֵּׁנִ֑י נֹ֥פֶךְ סַפִּ֖יר וְיָהֲלֹֽם׃ וְהַטּוּר הַשֵּׁנִי נוֹפֶךְ סַפִּיר וְיָהֲלוֹם.
יט וְהַטּ֖וּר הַשְּׁלִישִׁ֑י לֶ֥שֶׁם שְׁב֖וֹ וְאַחְלָֽמָה׃ וְהַטּוּר הַשְּׁלִישִׁי לֶשֶׁם שְׁבוֹ וְאַחְלָמָה.
כ וְהַטּוּר֙ הָרְבִיעִ֔י תַּרְשִׁ֥ישׁ וְשֹׁ֖הַם וְיָשְׁפֵ֑ה מְשֻׁבָּצִ֥ים זָהָ֛ב יִהְי֖וּ בְּמִלּוּאֹתָֽם׃ וְהַטּוּר הָרְבִיעִי תַּרְשִׁישׁ וְשׁוֹהַם וְיָשְׁפֵה – הֵם יִהְיוּ מֻקָּפִים בְּמִשְׁבְּצוֹת זָהָב הַמְּמֻלָּאוֹת בְּאוֹתָן הָאֲבָנִים.
כא וְ֠הָאֲבָנִ֠ים תִּֽהְיֶ֜יןָ עַל־שְׁמֹ֧ת בְּנֵֽי־יִשְׂרָאֵ֛ל שְׁתֵּ֥ים עֶשְׂרֵ֖ה עַל־שְׁמֹתָ֑ם פִּתּוּחֵ֤י חוֹתָם֙ אִ֣ישׁ עַל־שְׁמ֔וֹ תִּֽהְיֶ֕יןָ לִשְׁנֵ֥י עָשָׂ֖ר שָֽׁבֶט׃ וְהָאֲבָנִים תִּהְיֶינָה עַל שֵׁם שְׁתֵּים עֶשְׂרֵה בְּנֵי יִשְׂרָאֵל, וּשְׁמוֹתֵיהֶם חֲקוּקִים עֲלֵיהֶן כְּחוֹתֶמֶת, לְכָל אֶחָד מִשְּׁנֵים עָשָׂר הַשְּׁבָטִים – אֶת שְׁמוֹ.
כב וְעָשִׂ֧יתָ עַל־הַחֹ֛שֶׁן שַֽׁרְשֹׁ֥ת גַּבְלֻ֖ת מַעֲשֵׂ֣ה עֲבֹ֑ת זָהָ֖ב טָהֽוֹר׃ וְתַעֲשֶׂה עַל הַחוֹשֶׁן בְּקָצֵהוּ שַׁרְשְׁרוֹת שְׁזוּרוֹת כְּחֶבֶל, מִזָּהָב טָהוֹר.
כג וְעָשִׂ֙יתָ֙ עַל־הַחֹ֔שֶׁן שְׁתֵּ֖י טַבְּע֣וֹת זָהָ֑ב וְנָתַתָּ֗ אֶת־שְׁתֵּי֙ הַטַּבָּע֔וֹת עַל־שְׁנֵ֖י קְצ֥וֹת הַחֹֽשֶׁן׃ וְתַעֲשֶׂה עַל הַחוֹשֶׁן שְׁתֵּי טַבְּעוֹת זָהָב וְתָשִׂים אֶת שְׁתֵּי הַטַּבָּעוֹת עַל שְׁנֵי קְצָווֹת הַחוֹשֶׁן הָעֶלְיוֹנוֹת.
כד וְנָתַתָּ֗ה אֶת־שְׁתֵּי֙ עֲבֹתֹ֣ת הַזָּהָ֔ב עַל־שְׁתֵּ֖י הַטַּבָּעֹ֑ת אֶל־קְצ֖וֹת הַחֹֽשֶׁן׃ וְתָשִׂים אֶת שְׁתֵּי חֶבְלֵי הַזָּהָב עַל שְׁתֵּי הַטַּבָּעוֹת עַל קְצָווֹת הַחוֹשֶׁן.
כה וְאֵ֨ת שְׁתֵּ֤י קְצוֹת֙ שְׁתֵּ֣י הָעֲבֹתֹ֔ת תִּתֵּ֖ן עַל־שְׁתֵּ֣י הַֽמִּשְׁבְּצ֑וֹת וְנָתַתָּ֛ה עַל־כִּתְפ֥וֹת הָאֵפֹ֖ד אֶל־מ֥וּל פָּנָֽיו׃ וְאֶת שְׁנֵי קְצָווֹת שְׁנֵי הַשְּׂרוֹכִים תָּשִׂים עַל שְׁתֵּי הַמִּשְׁבְּצוֹת וְתָשִׂים עַל כְּתֵפִיּוֹת הָאֵפוֹד כְּלַפֵּי חוּץ.
כו וְעָשִׂ֗יתָ שְׁתֵּי֙ טַבְּע֣וֹת זָהָ֔ב וְשַׂמְתָּ֣ אֹתָ֔ם עַל־שְׁנֵ֖י קְצ֣וֹת הַחֹ֑שֶׁן עַל־שְׂפָת֕וֹ אֲשֶׁ֛ר אֶל־עֵ֥בֶר הָאֵפֹ֖ד‏[1] בָּֽיְתָה׃ וְתַעֲשֶׂה שְׁתֵּי טַבְּעוֹת זָהָב וְתָשִׂים אוֹתָן עַל שְׁנֵי קְצוֹת הַחוֹשֶׁן עַל שְׂפָתוֹ הַתַּחְתּוֹנָה, בַּצַּד הַפְּנִימִי שֶׁל הָאֵפוֹד.
כז וְעָשִׂ֘יתָ֮ שְׁתֵּ֣י טַבְּע֣וֹת זָהָב֒ וְנָתַתָּ֣ה אֹתָ֡ם עַל־שְׁתֵּי֩ כִתְפ֨וֹת הָאֵפ֤וֹד מִלְּמַ֙טָּה֙ מִמּ֣וּל פָּנָ֔יו לְעֻמַּ֖ת מַחְבַּרְתּ֑וֹ מִמַּ֕עַל לְחֵ֖שֶׁב הָאֵפֽוֹד׃ תַּעֲשֶׂה שְׁתֵּי טַבְּעוֹת זָהָב וְתָשִׂים אוֹתָן עַל שְׁתֵּי כְּתֵפִיּוֹת הָאֵפוֹד מִלְּמַטָּה כְּלַפֵּי חוּץ, מוּל מְקוֹם הִתְחַבְּרוּתָן לְגוּף הָאֵפוֹד, מֵעַל לַחֲגוֹרַת הָאֵפוֹד.
כח וְיִרְכְּס֣וּ אֶת־הַ֠חֹ֠שֶׁן מִֽטַּבְּעֹתָ֞ו אֶל־טַבְּעֹ֤ת הָאֵפוֹד֙ בִּפְתִ֣יל תְּכֵ֔לֶת לִֽהְי֖וֹת עַל־חֵ֣שֶׁב הָאֵפ֑וֹד וְלֹֽא־יִזַּ֣ח הַחֹ֔שֶׁן מֵעַ֖ל הָאֵפֽוֹד׃ וִיחַבְּרוּ אֶת הַחוֹשֶׁן מִטַּבְּעוֹתָיו אֶל טַבְּעוֹת הָאֵפוֹד בִּפְתִיל תְּכֵלֶת כְּדֵי שֶׁיִּהְיֶה עַל חֲגוֹרַת הָאֵפוֹד וְכָךְ הַחוֹשֶׁן לֹא יִתְנַתֵּק מֵהָאֵפוֹד.
כט וְנָשָׂ֣א אַ֠הֲרֹ֠ן אֶת־שְׁמ֨וֹת בְּנֵֽי־יִשְׂרָאֵ֜ל בְּחֹ֧שֶׁן הַמִּשְׁפָּ֛ט עַל־לִבּ֖וֹ בְּבֹא֣וֹ אֶל־הַקֹּ֑דֶשׁ לְזִכָּרֹ֥ן לִפְנֵֽי־יְהֹוָ֖ה תָּמִֽיד׃ אַהֲרֹן יִשָּׂא עַל לִבּוֹ בְּבוֹאוֹ לַקּוֹדֶשׁ אֶת שְׁמוֹת בְּנֵי יִשְׂרָאֵל בַּחוֹשֶׁן שֶׁמּוֹדִיעַ מַה לַּעֲשׂוֹת, לְזִכָּרוֹן לִפְנֵי ה' תָּמִיד.
ל וְנָתַתָּ֞ אֶל־חֹ֣שֶׁן הַמִּשְׁפָּ֗ט אֶת־הָאוּרִים֙ וְאֶת־הַתֻּמִּ֔ים וְהָיוּ֙ עַל־לֵ֣ב אַהֲרֹ֔ן בְּבֹא֖וֹ לִפְנֵ֣י יְהֹוָ֑ה וְנָשָׂ֣א אַ֠הֲרֹ֠ן אֶת־מִשְׁפַּ֨ט בְּנֵי־יִשְׂרָאֵ֧ל עַל־לִבּ֛וֹ לִפְנֵ֥י יְהֹוָ֖ה תָּמִֽיד׃ וְתָשִׂים בַּחוֹשֶׁן שֶׁמּוֹדִיעַ מַה לַּעֲשׂוֹת אֶת הָאֲבָנִים הַמְּאִירוֹת שֶׁעוֹנוֹת לִשְׁאֵלוֹת, וְהֵן תִּהְיֶינָה עַל לֵב אַהֲרֹן בְּבוֹאוֹ לִפְנֵי ה', וְאַהֲרֹן יִשָּׂא עַל לִבּוֹ לִפְנֵי ה' אֶת הַהוֹרָאוֹת לִבְנֵי יִשְׂרָאֵל, תָּמִיד.

שלישי[עריכה]

ס פָּרָשַׁת הַמְּעִיל
לא וְעָשִׂ֛יתָ אֶת־מְעִ֥יל הָאֵפ֖וֹד כְּלִ֥יל תְּכֵֽלֶת׃ תַּעֲשֶׂה אֶת הַמְּעִיל שֶׁעָלָיו הָאֵפוֹד, כֻּלּוֹ מִתְּכֵלֶת.
לב וְהָיָ֥ה פִֽי־רֹאשׁ֖וֹ בְּתוֹכ֑וֹ שָׂפָ֡ה יִֽהְיֶה֩ לְפִ֨יו סָבִ֜יב מַעֲשֵׂ֣ה אֹרֵ֗ג כְּפִ֥י תַחְרָ֛א יִֽהְיֶה־לּ֖וֹ לֹ֥א יִקָּרֵֽעַ׃ קָצֵהוּ הָעֶלְיוֹן יִהְיֶה מְקֻפָּל לְתוֹכוֹ, וְכָךְ תִּוָּצֵר שָׂפָה מִסָּבִיב לְפִתְחוֹ מְחֻבֶּרֶת בַּאֲרִיגָה וְיִהְיֶה לוֹ כְּמוֹ צַוָּארוֹן הַשִּׁרְיוֹן כְּדֵי שֶׁלֹא יִקָּרַע.
לג וְעָשִׂ֣יתָ עַל־שׁוּלָ֗יו רִמֹּנֵי֙ תְּכֵ֤לֶת וְאַרְגָּמָן֙ וְתוֹלַ֣עַת שָׁנִ֔י עַל־שׁוּלָ֖יו סָבִ֑יב וּפַעֲמֹנֵ֥י זָהָ֛ב בְּתוֹכָ֖ם סָבִֽיב׃ וְתַעֲשֶׂה עַל שׁוּלָיו רִמּוֹנִים מִתְּכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת עַל שׁוּלָיו סָבִיב, וּפַעֲמוֹנֵי זָהָב בֵּינֵיהֶם מִסָּבִיב.
לד פַּעֲמֹ֤ן זָהָב֙ וְרִמּ֔וֹן פַּֽעֲמֹ֥ן זָהָ֖ב וְרִמּ֑וֹן עַל־שׁוּלֵ֥י הַמְּעִ֖יל סָבִֽיב׃ פַּעֲמוֹן זָהָב וְרִמּוֹן פַּעֲמוֹן זָהָב וְרִמּוֹן עַל שׁוּלֵי הַמְּעִיל מִסָּבִיב.
לה וְהָיָ֥ה עַֽל־אַהֲרֹ֖ן לְשָׁרֵ֑ת וְנִשְׁמַ֣ע ק֠וֹל֠וֹ בְּבֹא֨וֹ אֶל־הַקֹּ֜דֶשׁ לִפְנֵ֧י יְהֹוָ֛ה וּבְצֵאת֖וֹ וְלֹ֥א יָמֽוּת׃ וְיִהְיֶה עַל אַהֲרֹן כְּדֵי שֶׁיְּשָׁרֵת אִתּוֹ, וְיִשָּׁמַע קוֹלוֹ בְּבוֹאוֹ לַקּוֹדֶשׁ לִפְנֵי ה' וּבְצֵאתוֹ, וְלֹא יָמוּת.
ס פָּרָשַׁת הַצִּיץ
לו וְעָשִׂ֥יתָ צִּ֖יץ זָהָ֣ב טָה֑וֹר וּפִתַּחְתָּ֤ עָלָיו֙ פִּתּוּחֵ֣י חֹתָ֔ם קֹ֖דֶשׁ לַֽיהֹוָֽה׃ תַּעֲשֶׂה צִיץ מִזָּהָב טָהוֹר וְתַחֲקוֹק בּוֹ בַּחֲקִיקָה כְּמוֹ בְּחוֹתֶמֶת: קוֹדֶשׁ לַה'.
לז וְשַׂמְתָּ֤ אֹתוֹ֙ עַל־פְּתִ֣יל תְּכֵ֔לֶת וְהָיָ֖ה עַל־הַמִּצְנָ֑פֶת אֶל־מ֥וּל פְּנֵֽי־הַמִּצְנֶ֖פֶת יִהְיֶֽה׃ תִּקְשׁוֹר אוֹתוֹ עַל יְדֵי חוּט תְּכֵלֶת, וְיִהְיֶה עַל הַמִּצְנֶפֶת, מוּל קִדְמַת הַמִּצְנֶפֶת יִהְיֶה.
לח וְהָיָה֮ עַל־מֵ֣צַח אַהֲרֹן֒ וְנָשָׂ֨א אַהֲרֹ֜ן אֶת־עֲוֺ֣ן הַקֳּדָשִׁ֗ים אֲשֶׁ֤ר יַקְדִּ֙ישׁוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לְכׇֽל־מַתְּנֹ֖ת קׇדְשֵׁיהֶ֑ם וְהָיָ֤ה עַל־מִצְחוֹ֙ תָּמִ֔יד לְרָצ֥וֹן לָהֶ֖ם לִפְנֵ֥י יְהֹוָֽה׃ וְיִהְיֶה עַל מֵצַח אַהֲרֹן, וִיכַפֵּר בּוֹ אַהֲרֹן עַל עֲווֹן טֻמְאַת הַקֳּדָשִׁים שֶׁיַּקְדִּישׁוּ בְּנֵי יִשְׂרָאֵל – כָּל מַתְּנוֹת קָדְשֵׁיהֶם, וְיִהְיֶה עַל מִצְחוֹ תָּמִיד כְּדֵי שֶׁיִּהְיוּ רְצוּיִים לִפְנֵי ה'.
פָּרָשַׁת בִּגְדֵי הַכֹּהֲנִים
לט וְשִׁבַּצְתָּ֙ הַכְּתֹ֣נֶת שֵׁ֔שׁ וְעָשִׂ֖יתָ מִצְנֶ֣פֶת שֵׁ֑שׁ וְאַבְנֵ֥ט תַּעֲשֶׂ֖ה מַעֲשֵׂ֥ה רֹקֵֽם׃ תַּעֲשֶׂה מִשְׁבְּצוֹת לִכְתוֹנֶת מִפִּשְׁתָּן וְתַעֲשֶׂה מִצְנֶפֶת מִפִּשְׁתָּן, וְתַעֲשֶׂה אַבְנֵט בִּמְלֶאכֶת רִקְמָה.
מ וְלִבְנֵ֤י אַהֲרֹן֙ תַּעֲשֶׂ֣ה כֻתֳּנֹ֔ת וְעָשִׂ֥יתָ לָהֶ֖ם אַבְנֵטִ֑ים וּמִגְבָּעוֹת֙ תַּעֲשֶׂ֣ה לָהֶ֔ם לְכָב֖וֹד וּלְתִפְאָֽרֶת׃ תַּעֲשֶׂה לִבְנֵי אַהֲרֹן כֻּתּוֹנוֹת וְתַעֲשֶׂה לָהֶם אַבְנֵטִים וְתַעֲשֶׂה לָהֶם מִגְבָּעוֹת לְכָבוֹד וּלְתִפְאֶרֶת.
מא וְהִלְבַּשְׁתָּ֤ אֹתָם֙ אֶת־אַהֲרֹ֣ן אָחִ֔יךָ וְאֶת־בָּנָ֖יו אִתּ֑וֹ וּמָשַׁחְתָּ֨ אֹתָ֜ם וּמִלֵּאתָ֧ אֶת־יָדָ֛ם וְקִדַּשְׁתָּ֥ אֹתָ֖ם וְכִהֲנ֥וּ לִֽי׃ תַּלְבִּישׁ אוֹתָם עַל אַהֲרֹן אָחִיךָ וְעַל בָּנָיו אִתּוֹ, וְתִמְרַח עֲלֵיהֶם שֶׁמֶן וּתְמַנֶּה אוֹתָם לַתַּפְקִיד, וּתְקַדֵּשׁ אוֹתָם, וְיִהְיוּ לִי לְכֹהֲנִים.
מב וַעֲשֵׂ֤ה לָהֶם֙ מִכְנְסֵי־בָ֔ד לְכַסּ֖וֹת בְּשַׂ֣ר עֶרְוָ֑ה מִמׇּתְנַ֥יִם וְעַד־יְרֵכַ֖יִם יִהְיֽוּ׃ וַעֲשֵׂה לָהֶם מִכְנְסֵי בָּד לְכַסּוֹת מָקוֹם צָנוּעַ, הֵם יִהְיוּ מִמָּתְנַיִם עַד יְרֵכַיִם.
מג וְהָיוּ֩ עַל־אַהֲרֹ֨ן וְעַל־בָּנָ֜יו בְּבֹאָ֣ם ׀ אֶל־אֹ֣הֶל מוֹעֵ֗ד א֣וֹ בְגִשְׁתָּ֤ם אֶל־הַמִּזְבֵּ֙חַ֙ לְשָׁרֵ֣ת בַּקֹּ֔דֶשׁ וְלֹא־יִשְׂא֥וּ עָוֺ֖ן וָמֵ֑תוּ חֻקַּ֥ת עוֹלָ֛ם ל֖וֹ וּלְזַרְע֥וֹ אַחֲרָֽיו׃ וְיִהְיוּ עַל אַהֲרֹן וְעַל בָּנָיו בְּבוֹאָם לְאוֹהֶל הַהִתְכַּנְּסוּת אוֹ בְּגִשְׁתָּם לַמִּזְבֵּחַ לְשָׁרֵת בַּקּוֹדֶשׁ, וְלֹא יִהְיֶה לָהֶם עָווֹן וְיָמוּתוּ, חוֹק לְעוֹלָם לוֹ וּלְצֶאֱצָאָיו אַחֲרָיו.

רביעי[עריכה]

ס פָּרָשַׁת הַמִּלּוּאִים
כט א וְזֶ֨ה הַדָּבָ֜ר אֲשֶֽׁר־תַּעֲשֶׂ֥ה לָהֶ֛ם לְקַדֵּ֥שׁ אֹתָ֖ם לְכַהֵ֣ן לִ֑י לְ֠קַ֠ח פַּ֣ר אֶחָ֧ד בֶּן־בָּקָ֛ר וְאֵילִ֥ם שְׁנַ֖יִם תְּמִימִֽם׃ וְזֶה הַדָּבָר שֶׁתַּעֲשֶׂה לָהֶם לְקַדֵּשׁ אוֹתָם לְכַהֵן לִי – קַח פַּר אֶחָד וּשְׁנֵי אֵילִים לְלֹא מוּם.
ב וְלֶ֣חֶם מַצּ֗וֹת וְחַלֹּ֤ת מַצֹּת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֣ים בַּשָּׁ֑מֶן סֹ֥לֶת חִטִּ֖ים תַּעֲשֶׂ֥ה אֹתָֽם׃ וְלֶחֶם שֶׁלֹּא הֶחְמִיץ, וְחַלּוֹת שֶׁלֹּא הֶחְמִיצוּ מְעֻרְבָּבוֹת בְּשֶׁמֶן, וּפִתּוֹת דַּקּוֹת שֶׁלֹּא הֶחְמִיצוּ מְרוּחִים בְּשֶׁמֶן – וְתַעֲשֶׂה אוֹתָם מִסּוֹלֶת שֶׁל חִטִּים.
ג וְנָתַתָּ֤ אוֹתָם֙ עַל־סַ֣ל אֶחָ֔ד וְהִקְרַבְתָּ֥ אֹתָ֖ם בַּסָּ֑ל וְאֶ֨ת־הַפָּ֔ר וְאֵ֖ת שְׁנֵ֥י הָאֵילִֽם׃ תָּשִׂים אוֹתָם בְּסַל אֶחָד וְתַגִּישׁ אוֹתָם בַּסַּל, וְאֶת הַפָּר וְאֶת שְׁנֵי הָאֵילִים.
ד וְאֶת־אַהֲרֹ֤ן וְאֶת־בָּנָיו֙ תַּקְרִ֔יב אֶל־פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְרָחַצְתָּ֥ אֹתָ֖ם בַּמָּֽיִם׃ וְאֶת אַהֲרֹן וְאֶת בָּנָיו תְּקָרֵב לְפֶתַח אוֹהֶל הַהִתְכַּנְּסוּת וְתִרְחַץ אוֹתָם בְּמַיִם.
ה וְלָקַחְתָּ֣ אֶת־הַבְּגָדִ֗ים וְהִלְבַּשְׁתָּ֤ אֶֽת־אַהֲרֹן֙ אֶת־הַכֻּתֹּ֔נֶת וְאֵת֙ מְעִ֣יל הָאֵפֹ֔ד וְאֶת־הָאֵפֹ֖ד וְאֶת־הַחֹ֑שֶׁן וְאָפַדְתָּ֣ ל֔וֹ בְּחֵ֖שֶׁב הָאֵפֹֽד׃ תִּקַּח אֶת הַבְּגָדִים וְתַלְבִּישׁ אֶת אַהֲרֹן אֶת הַכֻּתּוֹנֶת וְאֶת הַמְּעִיל שֶׁעָלָיו הָאֵפוֹד וְאֶת הָאֵפוֹד וְאֶת הַחוֹשֶׁן, וְתַחְגּוֹר אוֹתוֹ בַּחֲגוֹרַת הָאֵפוֹד.
ו וְשַׂמְתָּ֥ הַמִּצְנֶ֖פֶת עַל־רֹאשׁ֑וֹ וְנָתַתָּ֛ אֶת־נֵ֥זֶר הַקֹּ֖דֶשׁ עַל־הַמִּצְנָֽפֶת׃ וְתָשִׂים אֶת הַמִּצְנֶפֶת עַל רֹאשׁוֹ, וְתָשִׂים אֶת הַצִּיץ – כֶּתֶר הַקּוֹדֶשׁ, עַל הַמִּצְנֶפֶת.
ז וְלָֽקַחְתָּ֙ אֶת־שֶׁ֣מֶן הַמִּשְׁחָ֔ה וְיָצַקְתָּ֖ עַל־רֹאשׁ֑וֹ וּמָשַׁחְתָּ֖ אֹתֽוֹ׃ וְתִקַּח אֶת הַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ וְתִיצוֹק עַל רֹאשׁוֹ וְתִמְרַח עָלָיו.
ח וְאֶת־בָּנָ֖יו תַּקְרִ֑יב וְהִלְבַּשְׁתָּ֖ם כֻּתֳּנֹֽת׃ וְאֶת בָּנָיו תְּקָרֵב וְתַלְבִּישׁ אוֹתָם כֻּתּוֹנוֹת.
ט וְחָגַרְתָּ֩ אֹתָ֨ם אַבְנֵ֜ט אַהֲרֹ֣ן וּבָנָ֗יו וְחָבַשְׁתָּ֤ לָהֶם֙ מִגְבָּעֹ֔ת וְהָיְתָ֥ה לָהֶ֛ם כְּהֻנָּ֖ה לְחֻקַּ֣ת עוֹלָ֑ם וּמִלֵּאתָ֥ יַֽד־אַהֲרֹ֖ן וְיַד־בָּנָֽיו׃ וְתַחְגּוֹר אוֹתָם בְּאַבְנֵטִים – אֶת אַהֲרֹן וּבָנָיו, וְתַחֲבוֹשׁ לָהֶם מִגְבָּעוֹת, וְהַכְּהֻנָּה תִּהְיֶה לָהֶם חוֹק לְעוֹלָם, וּתְמַנֶּה אֶת אַהֲרֹן וְאֶת בָּנָיו לְתַפְקִידָם.
י וְהִקְרַבְתָּ֙ אֶת־הַפָּ֔ר לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וְסָמַ֨ךְ אַהֲרֹ֧ן וּבָנָ֛יו אֶת־יְדֵיהֶ֖ם עַל־רֹ֥אשׁ הַפָּֽר׃ תַּגִּישׁ אֶת הַפָּר אֶל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת וְאַהֲרֹן וּבָנָיו יַשְׁעִינוּ אֶת יְדֵיהֶם עַל רֹאשׁ הַפָּר.
יא וְשָׁחַטְתָּ֥ אֶת־הַפָּ֖ר לִפְנֵ֣י יְהֹוָ֑ה פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ תִּשְׁחַט אֶת הַפָּר לִפְנֵי ה' מוּל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.
יב וְלָֽקַחְתָּ֙ מִדַּ֣ם הַפָּ֔ר וְנָתַתָּ֛ה עַל־קַרְנֹ֥ת הַמִּזְבֵּ֖חַ בְּאֶצְבָּעֶ֑ךָ וְאֶת־כׇּל־הַדָּ֣ם תִּשְׁפֹּ֔ךְ אֶל־יְס֖וֹד הַמִּזְבֵּֽחַ׃ תִּקַּח מִדַּם הַפָּר וְתָשִׂים עַל קַרְנוֹת הַמִּזְבֵּחַ בָּאֶצְבַּע שֶׁלְּךָ, וְאֶת כָּל הַדָּם תִּשְׁפּוֹךְ עַל יְסוֹד הַמִּזְבֵּחַ.
יג וְלָֽקַחְתָּ֗ אֶֽת־כׇּל־הַחֵ֘לֶב֮ הַֽמְכַסֶּ֣ה אֶת־הַקֶּ֒רֶב֒ וְאֵ֗ת הַיֹּתֶ֙רֶת֙ עַל־הַכָּבֵ֔ד וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־הַחֵ֖לֶב אֲשֶׁ֣ר עֲלֵיהֶ֑ן וְהִקְטַרְתָּ֖ הַמִּזְבֵּֽחָה׃ תִּקַּח אֶת כָּל הַחֵלֶב שֶׁמְּכַסֶּה אֶת הַמֵּעַיִם וְאֶת הַתּוֹסֶפֶת שֶׁעַל הַכָּבֵד וְאֶת שְׁתֵּי הַכְּלָיוֹת וְאֶת הַחֵלֶב שֶׁעֲלֵיהֶן, וְתַקְטִיר עַל הַמִּזְבֵּחַ.
יד וְאֶת־בְּשַׂ֤ר הַפָּר֙ וְאֶת־עֹר֣וֹ וְאֶת־פִּרְשׁ֔וֹ תִּשְׂרֹ֣ף בָּאֵ֔שׁ מִח֖וּץ לַֽמַּחֲנֶ֑ה חַטָּ֖את הֽוּא׃ וְאֶת בְּשַׂר הַפָּר וְאֶת עוֹרוֹ וְאֶת צוֹאָתוֹ תִּשְׂרוֹף בָּאֵשׁ מִחוּץ לַמַּחֲנֶה – הוּא קָרְבַּן חַטָּאת.
טו וְאֶת־הָאַ֥יִל הָאֶחָ֖ד תִּקָּ֑ח וְסָ֨מְכ֜וּ אַהֲרֹ֧ן וּבָנָ֛יו אֶת־יְדֵיהֶ֖ם עַל־רֹ֥אשׁ הָאָֽיִל׃ תִּקַּח אֶת הָאַיִל הָרִאשׁוֹן, וְאַהֲרֹן וּבָנָיו יַשְׁעִינוּ אֶת יְדֵיהֶם עַל רֹאשׁ הָאַיִל.
טז וְשָׁחַטְתָּ֖ אֶת־הָאָ֑יִל וְלָֽקַחְתָּ֙ אֶת־דָּמ֔וֹ וְזָרַקְתָּ֥ עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ תִּשְׁחַט אֶת הָאַיִל וְתִקַּח אֶת דָּמוֹ וְתִזְרוֹק עַל הַמִּזְבֵּחַ מִסָּבִיב.
יז וְאֶ֨ת־הָאַ֔יִל תְּנַתֵּ֖חַ לִנְתָחָ֑יו וְרָחַצְתָּ֤ קִרְבּוֹ֙ וּכְרָעָ֔יו וְנָתַתָּ֥ עַל־נְתָחָ֖יו וְעַל־רֹאשֽׁוֹ׃ וּתְחַלֵּק אֶת הָאַיִל לַחֲלָקָיו וְתִרְחַץ אֶת מֵעָיו וְרַגְלָיו וְתָשִׂים אוֹתָם עַל שְׁאָר אֵבָרָיו וְעַל רֹאשׁוֹ.
יח וְהִקְטַרְתָּ֤ אֶת־כׇּל־הָאַ֙יִל֙ הַמִּזְבֵּ֔חָה עֹלָ֥ה ה֖וּא לַֽיהֹוָ֑ה רֵ֣יחַ נִיח֔וֹחַ אִשֶּׁ֥ה לַיהֹוָ֖ה הֽוּא׃ וְתַקְטִיר אֶת כָּל הָאַיִל עַל הַמִּזְבֵּחַ, הוּא קָרְבַּן עוֹלָה לַה' – הוּא עוֹלֶה בָּאֵשׁ לְנַחַת רוּחַ לַה'.

חמישי[עריכה]

יט וְלָ֣קַחְתָּ֔ אֵ֖ת הָאַ֣יִל הַשֵּׁנִ֑י וְסָמַ֨ךְ אַהֲרֹ֧ן וּבָנָ֛יו אֶת־יְדֵיהֶ֖ם עַל־רֹ֥אשׁ הָאָֽיִל׃ תִּקַּח אֶת הָאַיִל הַשֵּׁנִי וְאַהֲרֹן וּבָנָיו יַשְׁעִינוּ אֶת יְדֵיהֶם עַל רֹאשׁ הָאַיִל.
כ וְשָׁחַטְתָּ֣ אֶת־הָאַ֗יִל וְלָקַחְתָּ֤ מִדָּמוֹ֙ וְנָֽתַתָּ֡ה עַל־תְּנוּךְ֩ אֹ֨זֶן אַהֲרֹ֜ן וְעַל־תְּנ֨וּךְ אֹ֤זֶן בָּנָיו֙ הַיְמָנִ֔ית וְעַל־בֹּ֤הֶן יָדָם֙ הַיְמָנִ֔ית וְעַל־בֹּ֥הֶן רַגְלָ֖ם הַיְמָנִ֑ית וְזָרַקְתָּ֧ אֶת־הַדָּ֛ם עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ תִּשְׁחַט אֶת הָאַיִל וְתִקַּח מִדָּמוֹ וְתָשִׂים עַל תְּנוּךְ אוֹזֶן אַהֲרֹן הַיְּמָנִית וְעַל תְּנוּךְ אוֹזֶן בָּנָיו הַיְּמָנִית וְעַל בֹּהֶן יָדָם הַיְּמָנִית וְעַל בּוֹהֶן רַגְלָם הַיְּמָנִית, וְתִזְרוֹק אֶת הַדָּם עַל הַמִּזְבֵּחַ מִסָּבִיב.
כא וְלָקַחְתָּ֞ מִן־הַדָּ֨ם אֲשֶׁ֥ר עַֽל־הַמִּזְבֵּ֘חַ֮ וּמִשֶּׁ֣מֶן הַמִּשְׁחָה֒ וְהִזֵּיתָ֤ עַֽל־אַהֲרֹן֙ וְעַל־בְּגָדָ֔יו וְעַל־בָּנָ֛יו וְעַל־בִּגְדֵ֥י בָנָ֖יו אִתּ֑וֹ וְקָדַ֥שׁ הוּא֙ וּבְגָדָ֔יו וּבָנָ֛יו וּבִגְדֵ֥י בָנָ֖יו אִתּֽוֹ׃ תִּקַּח מֵהַדָּם שֶׁעַל הַמִּזְבֵּחַ וּמֵהַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ, וְתַתִּיז עַל אַהֲרֹן וְעַל בְּגָדָיו וְעַל בָּנָיו וְעַל בִּגְדֵי בָּנָיו אִתּוֹ, וְיִתְקַדְּשׁוּ הוּא וּבְגָדָיו וְאִתּוֹ בָּנָיו וּבִגְדֵי בָּנָיו.
כב וְלָקַחְתָּ֣ מִן־הָ֠אַ֠יִל הַחֵ֨לֶב וְהָֽאַלְיָ֜ה וְאֶת־הַחֵ֣לֶב ׀ הַֽמְכַסֶּ֣ה אֶת־הַקֶּ֗רֶב וְאֵ֨ת יֹתֶ֤רֶת הַכָּבֵד֙ וְאֵ֣ת ׀ שְׁתֵּ֣י הַכְּלָיֹ֗ת וְאֶת־הַחֵ֙לֶב֙ אֲשֶׁ֣ר עֲלֵיהֶ֔ן וְאֵ֖ת שׁ֣וֹק הַיָּמִ֑ין כִּ֛י אֵ֥יל מִלֻּאִ֖ים הֽוּא׃ תִּקַּח מֵהָאַיִל אֶת הַחֵלֶב וְאֶת שׁוּמַן הַזָּנָב וְאֶת הַחֵלֶב שֶׁמְּכַסֶּה אֶת הַמֵּעַיִם וְאֶת הַתּוֹסֶפֶת שֶׁעַל הַכָּבֵד וְאֶת שְׁתֵּי הַכְּלָיוֹת וְאֶת הַחֵלֶב שֶׁעֲלֵיהֶן וְאֶת שׁוֹק הַיָּמִין, עֲשֵׂה כָּךְ כִּי אַיִל זֶה הוּא קָרְבָּן מִנּוּי לְתַפְקִיד.
כג וְכִכַּ֨ר לֶ֜חֶם אַחַ֗ת וְֽחַלַּ֨ת לֶ֥חֶם שֶׁ֛מֶן אַחַ֖ת וְרָקִ֣יק אֶחָ֑ד מִסַּל֙ הַמַּצּ֔וֹת אֲשֶׁ֖ר לִפְנֵ֥י יְהֹוָֽה׃ וְתִקַּח כִּכַּר לֶחֶם אַחַת, וְחַלַּת לֶחֶם אַחַת מְטֻגֶּנֶת בְּשֶׁמֶן, וּפִתָּה דַּקָּה אַחַת מִסַּל הַלֶּחֶם שֶׁלֹּא הֶחְמִיץ שֶׁלִּפְנֵי ה'.
כד וְשַׂמְתָּ֣ הַכֹּ֔ל עַ֚ל כַּפֵּ֣י אַהֲרֹ֔ן וְעַ֖ל כַּפֵּ֣י בָנָ֑יו וְהֵנַפְתָּ֥ אֹתָ֛ם תְּנוּפָ֖ה לִפְנֵ֥י יְהֹוָֽה׃ תָּשִׂים אֶת הַכֹּל עַל כַּפּוֹת יְדֵי אַהֲרֹן וְעַל כַּפּוֹת יְדֵי בָּנָיו וְתָנִיף אוֹתָם לִפְנֵי ה'.
כה וְלָקַחְתָּ֤ אֹתָם֙ מִיָּדָ֔ם וְהִקְטַרְתָּ֥ הַמִּזְבֵּ֖חָה עַל־הָעֹלָ֑ה לְרֵ֤יחַ נִיח֙וֹחַ֙ לִפְנֵ֣י יְהֹוָ֔ה אִשֶּׁ֥ה ה֖וּא לַיהֹוָֽה׃ תִּקַּח אוֹתָם מִיָּדָם וְתַקְטִיר עַל הַמִּזְבֵּחַ עַל הָעוֹלָה, לְנַחַת רוּחַ לִפְנֵי ה' – זֶה קָרְבָּן עוֹלֶה בָּאֵשׁ לַה'.
כו וְלָקַחְתָּ֣ אֶת־הֶֽחָזֶ֗ה מֵאֵ֤יל הַמִּלֻּאִים֙ אֲשֶׁ֣ר לְאַהֲרֹ֔ן וְהֵנַפְתָּ֥ אֹת֛וֹ תְּנוּפָ֖ה לִפְנֵ֣י יְהֹוָ֑ה וְהָיָ֥ה לְךָ֖ לְמָנָֽה׃ תִּקַּח אֶת הֶחָזֶה מֵהָאַיִל שֶׁל קָרְבַּן הַמִּנּוּי שֶׁל אַהֲרֹן, וְתָנִיף אוֹתוֹ לִפְנֵי ה', וְהוּא יִהְיֶה הַמָּנָה שֶׁלְּךָ.
כז וְקִדַּשְׁתָּ֞ אֵ֣ת ׀ חֲזֵ֣ה הַתְּנוּפָ֗ה וְאֵת֙ שׁ֣וֹק הַתְּרוּמָ֔ה אֲשֶׁ֥ר הוּנַ֖ף וַאֲשֶׁ֣ר הוּרָ֑ם מֵאֵיל֙ הַמִּלֻּאִ֔ים מֵאֲשֶׁ֥ר לְאַהֲרֹ֖ן וּמֵאֲשֶׁ֥ר לְבָנָֽיו׃ תְּקַדֵּשׁ אֶת הַחֲזֵה הַמּוּנָף וְאֶת הַשּׁוֹק שֶׁהוּרַם, שֶׁהוּנְפוּ וְשֶׁהוּרְמוּ מֵהָאַיִל שֶׁל קָרְבַּן הַמִּנּוּי, מֵהַשַּׁיָּךְ לְאַהֲרֹן וּמֵהַשַּׁיָּךְ לְבָנָיו.
כח וְהָיָה֩ לְאַהֲרֹ֨ן וּלְבָנָ֜יו לְחׇק־עוֹלָ֗ם מֵאֵת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל כִּ֥י תְרוּמָ֖ה ה֑וּא וּתְרוּמָ֞ה יִהְיֶ֨ה מֵאֵ֤ת בְּנֵֽי־יִשְׂרָאֵל֙ מִזִּבְחֵ֣י שַׁלְמֵיהֶ֔ם תְּרוּמָתָ֖ם לַיהֹוָֽה׃ וּמֵעַכְשָׁיו הֶחָזֶה וְהַשּׁוֹק יִהְיוּ לְאַהֲרֹן וּלְבָנָיו קִצְבָּה לְעוֹלָם מִבְּנֵי יִשְׂרָאֵל, כִּי הֵם מֻפְרָשִׁים מֵהַקָּרְבָּן, וְהַמֻּפְרָשׁ מִזִּבְחֵי שַׁלְמֵיהֶם שֶׁל בְּנֵי יִשְׂרָאֵל הוּא תְּרוּמָתָם לַה'.
כט וּבִגְדֵ֤י הַקֹּ֙דֶשׁ֙ אֲשֶׁ֣ר לְאַהֲרֹ֔ן יִהְי֥וּ לְבָנָ֖יו אַחֲרָ֑יו לְמׇשְׁחָ֣ה בָהֶ֔ם וּלְמַלֵּא־בָ֖ם אֶת־יָדָֽם׃ וּבִגְדֵי הַקּוֹדֶשׁ שֶׁל אַהֲרֹן יִהְיוּ גַּם לְבָנָיו אַחֲרָיו, יִמְרְחוּ עֲלֵיהֶם אֶת שֶׁמֶן הַקּוֹדֶשׁ כְּשֶׁיִּלְבְּשׁוּ אוֹתָם וְכָךְ יְמַנּוּ אוֹתָם לְכֹהֵן גָּדוֹל.
ל שִׁבְעַ֣ת יָמִ֗ים יִלְבָּשָׁ֧ם הַכֹּהֵ֛ן תַּחְתָּ֖יו מִבָּנָ֑יו אֲשֶׁ֥ר יָבֹ֛א אֶל־אֹ֥הֶל מוֹעֵ֖ד לְשָׁרֵ֥ת בַּקֹּֽדֶשׁ׃ הַכֹּהֵן מִבָּנָיו שֶׁיִּתְמַנֶּה בִּמְקוֹמוֹ יִלְבַּשׁ אוֹתָם שִׁבְעָה יָמִים כְּשֶׁיָּבוֹא לְאוֹהֶל הַהִתְכַּנְּסוּת לְשָׁרֵת בַּקּוֹדֶשׁ.
לא וְאֵ֛ת אֵ֥יל הַמִּלֻּאִ֖ים תִּקָּ֑ח וּבִשַּׁלְתָּ֥ אֶת־בְּשָׂר֖וֹ בְּמָקֹ֥ם קָדֹֽשׁ׃ וְתִקַּח אֶת אֵיל הַמִּנּוּי לְתַפְקִיד, וּתְבַשֵּׁל אֶת בְּשָׂרוֹ בְּמָקוֹם קָדוֹשׁ.
לב וְאָכַ֨ל אַהֲרֹ֤ן וּבָנָיו֙ אֶת־בְּשַׂ֣ר הָאַ֔יִל וְאֶת־הַלֶּ֖חֶם אֲשֶׁ֣ר בַּסָּ֑ל פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וְאַהֲרֹן וּבָנָיו יֹאכְלוּ אֶת בְּשַׂר הָאַיִל וְאֶת הַלֶּחֶם שֶׁבַּסַּל בְּפֶתַח אוֹהֶל הַהִתְכַּנְּסוּת –
לג וְאָכְל֤וּ אֹתָם֙ אֲשֶׁ֣ר כֻּפַּ֣ר בָּהֶ֔ם לְמַלֵּ֥א אֶת־יָדָ֖ם לְקַדֵּ֣שׁ אֹתָ֑ם וְזָ֥ר לֹא־יֹאכַ֖ל כִּי־קֹ֥דֶשׁ הֵֽם׃ יֹאכְלוּ אוֹתָם אֵלֶּה שֶׁהִתְכַּפְּרוּ בָּהֶם כְּדֵי לְמַנּוֹתָם לְתַפְקִידָם וּלְקַדֵּשׁ אוֹתָם, וְאָדָם זָר לֹא יֹאכַל אוֹתָם כִּי הֵם קוֹדֶשׁ.
לד וְֽאִם־יִוָּתֵ֞ר מִבְּשַׂ֧ר הַמִּלֻּאִ֛ים וּמִן־הַלֶּ֖חֶם עַד־הַבֹּ֑קֶר וְשָׂרַפְתָּ֤ אֶת־הַנּוֹתָר֙ בָּאֵ֔שׁ לֹ֥א יֵאָכֵ֖ל כִּי־קֹ֥דֶשׁ הֽוּא׃ וְאִם יִשָּׁאֵר מִבְּשַׂר קָרְבַּן הַמִּנּוּי וּמֵהַלֶּחֶם עַד הַבּוֹקֶר, תִּשְׂרוֹף אֶת הַנּוֹתָר בָּאֵשׁ, הוּא לֹא יֵאָכֵל כִּי הוּא קוֹדֶשׁ.
לה וְעָשִׂ֜יתָ לְאַהֲרֹ֤ן וּלְבָנָיו֙ כָּ֔כָה כְּכֹ֥ל אֲשֶׁר־צִוִּ֖יתִי אֹתָ֑כָה שִׁבְעַ֥ת יָמִ֖ים תְּמַלֵּ֥א יָדָֽם׃ תַּעֲשֶׂה לְאַהֲרֹן וּלְבָנָיו כָּכָה כְּכָל מַה שֶּׁצִּוִּיתִי אוֹתְךָ, שִׁבְעָה יָמִים תְּמַנֶּה אוֹתָם לַתַּפְקִיד.
לו וּפַ֨ר חַטָּ֜את תַּעֲשֶׂ֤ה לַיּוֹם֙ עַל־הַכִּפֻּרִ֔ים וְחִטֵּאתָ֙ עַל־הַמִּזְבֵּ֔חַ בְּכַפֶּרְךָ֖ עָלָ֑יו וּמָֽשַׁחְתָּ֥ אֹת֖וֹ לְקַדְּשֽׁוֹ׃ וְתַעֲשֶׂה כָּל יוֹם פַּר חַטָּאת לְכַפָּרָה וּתְטַהֵר אֶת הַמִּזְבֵּחַ בְּכָךְ שֶׁתַּזֶּה עָלָיו לְכַפָּרָה, וְתִמְרַח עָלָיו מֵהַשֶּׁמֶן כְּדֵי לְקַדְּשׁוֹ.
לז שִׁבְעַ֣ת יָמִ֗ים תְּכַפֵּר֙ עַל־הַמִּזְבֵּ֔חַ וְקִדַּשְׁתָּ֖ אֹת֑וֹ וְהָיָ֤ה הַמִּזְבֵּ֙חַ֙ קֹ֣דֶשׁ קׇֽדָשִׁ֔ים כׇּל־הַנֹּגֵ֥עַ בַּמִּזְבֵּ֖חַ יִקְדָּֽשׁ׃ שִׁבְעָה יָמִים תַּזֶּה עַל הַמִּזְבֵּחַ לְכַפָּרָה וּתְקַדֵּשׁ אוֹתוֹ, וְהַמִּזְבֵּחַ יִהְיֶה קוֹדֶשׁ קָדָשִׁים – כָּל מִי שֶׁנּוֹגֵעַ בַּמִּזְבֵּחַ צָרִיךְ לְהִתְקַדֵּשׁ.

ששי[עריכה]

ס פָּרָשַׁת הַתָּמִיד
לח וְזֶ֕ה אֲשֶׁ֥ר תַּעֲשֶׂ֖ה עַל־הַמִּזְבֵּ֑חַ כְּבָשִׂ֧ים בְּנֵֽי־שָׁנָ֛ה שְׁנַ֥יִם לַיּ֖וֹם תָּמִֽיד׃ וְזֶה מַה שֶּׁתַּקְרִיב עַל הַמִּזְבֵּחַ: כְּבָשִׂים שֶׁגִּילָם עַד שָׁנָה, שְׁנַיִם לְכָל יוֹם, תָּמִיד.
לט אֶת־הַכֶּ֥בֶשׂ הָאֶחָ֖ד תַּעֲשֶׂ֣ה בַבֹּ֑קֶר וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּעֲשֶׂ֖ה בֵּ֥ין הָעַרְבָּֽיִם׃ אֶת הַכֶּבֶשׂ הָרִאשׁוֹן תַּעֲשֶׂה בַּבּוֹקֶר וְאֶת הַכֶּבֶשׂ הַשֵּׁנִי תַּעֲשֶׂה אַחַר הַצָּהֳרַיִם.
מ וְעִשָּׂרֹ֨ן סֹ֜לֶת בָּל֨וּל בְּשֶׁ֤מֶן כָּתִית֙ רֶ֣בַע הַהִ֔ין וְנֵ֕סֶךְ רְבִיעִ֥ת הַהִ֖ין יָ֑יִן לַכֶּ֖בֶשׂ הָאֶחָֽד׃ וַעֲשִׂירִית הָאֵיפָה סוֹלֶת (כְּ2.16 לִיטֶר) מְעוֹרָב בְּרֶבַע הַהִין שֶׁמֶן זַיִת כָּתוּשׁ (כִּ900 מ"ל), וּרְבִיעִית הַהִין יַיִן (כִּ900 מ"ל), לְהַגָּשָׁה לַמִּזְבֵּחַ עֲבוּר כָּל כֶּבֶשׂ.
מא וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּעֲשֶׂ֖ה בֵּ֣ין הָעַרְבָּ֑יִם כְּמִנְחַ֨ת הַבֹּ֤קֶר וּכְנִסְכָּהּ֙ תַּֽעֲשֶׂה־לָּ֔הּ לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַיהֹוָֽה׃ וְאֶת הַכֶּבֶשׂ הַשֵּׁנִי תַּקְרִיב אַחַר הַצָּהֳרַיִם, תַּעֲשֶׂה לוֹ כְּמוֹ הַמִּנְחָה שֶׁל הַבּוֹקֶר וּשְׁפִיכַת הַיַּיִן שֶׁלָּהּ – לְקָרְבָּן עוֹלֶה בָּאֵשׁ לְנַחַת רוּחַ לַה'.
מב עֹלַ֤ת תָּמִיד֙ לְדֹרֹ֣תֵיכֶ֔ם פֶּ֥תַח אֹֽהֶל־מוֹעֵ֖ד לִפְנֵ֣י יְהֹוָ֑ה אֲשֶׁ֨ר אִוָּעֵ֤ד לָכֶם֙ שָׁ֔מָּה לְדַבֵּ֥ר אֵלֶ֖יךָ שָֽׁם׃ עוֹלָה הַמֻּקְרֶבֶת תָּמִיד חוֹק לְדוֹרוֹת, בְּפֶתַח אוֹהֶל הַהִתְכַּנְּסוּת לִפְנֵי ה', שֶׁאֶפָּגֵשׁ אִתְּכֶם שָׁם לְדַבֵּר אֵלֶיךָ שָׁם.
מג וְנֹעַדְתִּ֥י שָׁ֖מָּה לִבְנֵ֣י יִשְׂרָאֵ֑ל וְנִקְדַּ֖שׁ בִּכְבֹדִֽי׃ אֶפָּגֵשׁ שָׁם עִם בְּנֵי יִשְׂרָאֵל וְהַמָּקוֹם יִתְקַדֵּשׁ בִּכְבוֹדִי.
מד וְקִדַּשְׁתִּ֛י אֶת־אֹ֥הֶל מוֹעֵ֖ד וְאֶת־הַמִּזְבֵּ֑חַ וְאֶת־אַהֲרֹ֧ן וְאֶת־בָּנָ֛יו אֲקַדֵּ֖שׁ לְכַהֵ֥ן לִֽי׃ אֲקַדֵּשׁ אֶת אוֹהֶל הַהִתְכַּנְּסוּת וְאֶת הַמִּזְבֵּחַ, וְאֶת אַהֲרֹן וְאֶת בָּנָיו אֲקַדֵּשׁ לְכַהֵן לִי.
מה וְשָׁ֣כַנְתִּ֔י בְּת֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל וְהָיִ֥יתִי לָהֶ֖ם לֵאלֹהִֽים׃ וְאֶשְׁכּוֹן בְּתוֹךְ בְּנֵי יִשְׂרָאֵל, וְאֶהְיֶה לָהֶם לֵאלֹהִים.
מו וְיָדְע֗וּ כִּ֣י אֲנִ֤י יְהֹוָה֙ אֱלֹ֣הֵיהֶ֔ם אֲשֶׁ֨ר הוֹצֵ֧אתִי אֹתָ֛ם מֵאֶ֥רֶץ מִצְרַ֖יִם לְשׇׁכְנִ֣י בְתוֹכָ֑ם אֲנִ֖י יְהֹוָ֥ה אֱלֹהֵיהֶֽם׃ וְיֵדְעוּ שֶׁאֲנִי ה' אֱלֹהֵיהֶם שֶׁהוֹצֵאתִי אוֹתָם מֵאֶרֶץ מִצְרַיִם כְּדֵי לְהַשְׁרוֹת אֶת שְׁכִינָתִי בְּתוֹכָם, אֲנִי ה' אֱלֹהֵיהֶם.

שביעי[עריכה]

פ פָּרָשַׁת מִזְבַּח הַזָּהָב
ל א וְעָשִׂ֥יתָ מִזְבֵּ֖חַ מִקְטַ֣ר קְטֹ֑רֶת עֲצֵ֥י שִׁטִּ֖ים תַּעֲשֶׂ֥ה אֹתֽוֹ׃ תַּעֲשֶׂה מִזְבֵּחַ לְהַקְטִיר קְטוֹרֶת, תַּעֲשֶׂה אוֹתוֹ מֵעֲצֵי שִׁטָּה.
ב אַמָּ֨ה אׇרְכּ֜וֹ וְאַמָּ֤ה רׇחְבּוֹ֙ רָב֣וּעַ יִהְיֶ֔ה וְאַמָּתַ֖יִם קֹמָת֑וֹ מִמֶּ֖נּוּ קַרְנֹתָֽיו׃ הוּא יִהְיֶה בְּצוּרַת רִבּוּעַ – אַמָּה אָרְכּוֹ וְאַמָּה רָחְבּוֹ (כְּ47 ס"מ), וּשְׁתֵּי אַמּוֹת גָּבְהוֹ, קַרְנוֹתָיו יֵעָשׂוּ כְּחֵלֶק מִמֶּנּוּ.
ג וְצִפִּיתָ֨ אֹת֜וֹ זָהָ֣ב טָה֗וֹר אֶת־גַּגּ֧וֹ וְאֶת־קִירֹתָ֛יו סָבִ֖יב וְאֶת־קַרְנֹתָ֑יו וְעָשִׂ֥יתָ לּ֛וֹ זֵ֥ר זָהָ֖ב סָבִֽיב׃ תְּצַפֶּה אוֹתוֹ בְּזָהָב טָהוֹר – אֶת גַּגּוֹ וְאֶת קִירוֹתָיו מִסָּבִיב וְאֶת קַרְנוֹתָיו, וְתַעֲשֶׂה לוֹ זֵר מִזָּהָב מִסָּבִיב.
ד וּשְׁתֵּי֩ טַבְּעֹ֨ת זָהָ֜ב תַּֽעֲשֶׂה־לּ֣וֹ ׀ מִתַּ֣חַת לְזֵר֗וֹ עַ֚ל שְׁתֵּ֣י צַלְעֹתָ֔יו תַּעֲשֶׂ֖ה עַל־שְׁנֵ֣י צִדָּ֑יו וְהָיָה֙ לְבָתִּ֣ים לְבַדִּ֔ים לָשֵׂ֥את אֹת֖וֹ בָּהֵֽמָּה׃ וְתַעֲשֶׂה לוֹ שְׁתֵּי טַבָּעוֹת מִזָּהָב מִתַּחַת לְזֵרוֹ עַל שְׁתֵּי דְּפָנוֹתָיו – תַּעֲשֶׂה אוֹתָם עַל שְׁנֵי צִדָּיו, וְהֵם יִהְיוּ מְקוֹם הַמּוֹטוֹת לְהָרִים אוֹתוֹ בָּהֵם.
ה וְעָשִׂ֥יתָ אֶת־הַבַּדִּ֖ים עֲצֵ֣י שִׁטִּ֑ים וְצִפִּיתָ֥ אֹתָ֖ם זָהָֽב׃ תַּעֲשֶׂה אֶת הַמּוֹטוֹת מֵעֲצֵי שִׁטָּה, וּתְצַפֶּה אוֹתָם זָהָב.
ו וְנָתַתָּ֤ה אֹתוֹ֙ לִפְנֵ֣י הַפָּרֹ֔כֶת אֲשֶׁ֖ר עַל־אֲרֹ֣ן הָעֵדֻ֑ת לִפְנֵ֣י הַכַּפֹּ֗רֶת אֲשֶׁר֙ עַל־הָ֣עֵדֻ֔ת אֲשֶׁ֛ר אִוָּעֵ֥ד לְךָ֖ שָֽׁמָּה׃ תָּשִׂים אוֹתוֹ לִפְנֵי הַפָּרוֹכֶת שֶׁעַל יַד אֲרוֹן לוּחוֹת הַבְּרִית – לִפְנֵי הַכִּסּוּי שֶׁעַל אֲרוֹן לוּחוֹת הַבְּרִית, שֶׁאֶתְגַּלֶּה אֵלֶיךָ בְּדִבּוּרִי שָׁם.
ז וְהִקְטִ֥יר עָלָ֛יו אַהֲרֹ֖ן קְטֹ֣רֶת סַמִּ֑ים בַּבֹּ֣קֶר בַּבֹּ֗קֶר בְּהֵיטִיב֛וֹ אֶת־הַנֵּרֹ֖ת יַקְטִירֶֽנָּה׃ אַהֲרֹן יַקְטִיר עָלָיו קְטוֹרֶת בְּשָׂמִים מוּקְדָּם בַּבּוֹקֶר, בִּזְמַן הֲכָנָתוֹ אֶת הַנֵּרוֹת יַקְטִיר אוֹתָהּ.

מפטיר[עריכה]

ח וּבְהַעֲלֹ֨ת אַהֲרֹ֧ן אֶת־הַנֵּרֹ֛ת בֵּ֥ין הָעַרְבַּ֖יִם יַקְטִירֶ֑נָּה קְטֹ֧רֶת תָּמִ֛יד לִפְנֵ֥י יְהֹוָ֖ה לְדֹרֹתֵיכֶֽם׃ וּכְשֶׁאַהֲרֹן יַדְלִיק אֶת הַנֵּרוֹת אַחַר הַצָּהֳרַיִם, הוּא יַקְטִיר אוֹתָהּ שׁוּב – קְטוֹרֶת תָּמִיד בְּכָל יוֹם לִפְנֵי ה' לְדוֹרוֹת.
ט לֹא־תַעֲל֥וּ עָלָ֛יו קְטֹ֥רֶת זָרָ֖ה וְעֹלָ֣ה וּמִנְחָ֑ה וְנֵ֕סֶךְ לֹ֥א תִסְּכ֖וּ עָלָֽיו׃ אַל תַּעֲלוּ עָלָיו קְטוֹרֶת אֲחֶרֶת וְעוֹלָה וּמִנְחָה, וְלֹא תִּשְׁפְּכוּ עָלָיו יַיִן.
י וְכִפֶּ֤ר אַהֲרֹן֙ עַל־קַרְנֹתָ֔יו אַחַ֖ת בַּשָּׁנָ֑ה מִדַּ֞ם חַטַּ֣את הַכִּפֻּרִ֗ים אַחַ֤ת בַּשָּׁנָה֙ יְכַפֵּ֤ר עָלָיו֙ לְדֹרֹ֣תֵיכֶ֔ם קֹֽדֶשׁ־קׇדָשִׁ֥ים ה֖וּא לַיהֹוָֽה׃ אַהֲרֹן יַזֶּה דָּם לְכַפָּרָה עַל קַרְנוֹתָיו פַּעַם אַחַת בַּשָּׁנָה, הוּא יַזֶּה עָלָיו לְכַפָּרָה מִדַּם חַטַּאת הַכִּפּוּרִים פַּעַם אַחַת בַּשָּׁנָה בְּכָל הַדּוֹרוֹת, הוּא קוֹדֶשׁ קָדָשִׁים לַה'.
