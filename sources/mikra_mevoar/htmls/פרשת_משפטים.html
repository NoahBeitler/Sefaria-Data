ראשון[עריכה]

פ פָּרָשַׁת הָעֲבָדִים
כא א וְאֵ֙לֶּה֙ הַמִּשְׁפָּטִ֔ים אֲשֶׁ֥ר תָּשִׂ֖ים לִפְנֵיהֶֽם׃ אֵלֶּה הַדִּינִים שֶׁתְּסַדֵּר לִפְנֵיהֶם.
ב כִּ֤י תִקְנֶה֙ עֶ֣בֶד עִבְרִ֔י שֵׁ֥שׁ שָׁנִ֖ים יַעֲבֹ֑ד וּבַ֨שְּׁבִעִ֔ת יֵצֵ֥א לַֽחׇפְשִׁ֖י חִנָּֽם׃ אִם תִּקְנֶה עֶבֶד עִבְרִי הוּא יַעֲבוֹד שֵׁשׁ שָׁנִים, וּבַשָּׁנָה הַשְּׁבִיעִית יֵצֵא לַחָפְשִׁי בְּחִנָּם.
ג אִם־בְּגַפּ֥וֹ יָבֹ֖א בְּגַפּ֣וֹ יֵצֵ֑א אִם־בַּ֤עַל אִשָּׁה֙ ה֔וּא וְיָצְאָ֥ה אִשְׁתּ֖וֹ עִמּֽוֹ׃ אִם הוּא יַגִּיעַ לְבַדּוֹ – יֵצֵא לְבַדּוֹ, וְאִם הוּא בַּעַל אִשָּׁה – אִשְׁתּוֹ תֵּצֵא אִתּוֹ.
ד אִם־אֲדֹנָיו֙ יִתֶּן־ל֣וֹ אִשָּׁ֔ה וְיָלְדָה־ל֥וֹ בָנִ֖ים א֣וֹ בָנ֑וֹת הָאִשָּׁ֣ה וִילָדֶ֗יהָ תִּהְיֶה֙ לַֽאדֹנֶ֔יהָ וְה֖וּא יֵצֵ֥א בְגַפּֽוֹ׃ אִם אֲדוֹנוֹ יִתֵּן לוֹ אִשָּׁה וְהִיא תֵּלֵד לוֹ בָּנִים אוֹ בָּנוֹת – הָאִשָּׁה עִם יְלָדֶיהָ יִהְיוּ שֶׁל הָאָדוֹן שֶׁלָּהּ, וְהָעֶבֶד יֵצֵא לְבַדּוֹ.
ה וְאִם־אָמֹ֤ר יֹאמַר֙ הָעֶ֔בֶד אָהַ֙בְתִּי֙ אֶת־אֲדֹנִ֔י אֶת־אִשְׁתִּ֖י וְאֶת־בָּנָ֑י לֹ֥א אֵצֵ֖א חׇפְשִֽׁי׃ וְאִם יֹאמַר הָעֶבֶד: אֲנִי אוֹהֵב אֶת אֲדוֹנִי אֶת אִשְׁתִּי וְאֶת בָּנַי, לֹא אֵצֵא לַחָפְשִׁי!
ו וְהִגִּישׁ֤וֹ אֲדֹנָיו֙ אֶל־הָ֣אֱלֹהִ֔ים וְהִגִּישׁוֹ֙ אֶל־הַדֶּ֔לֶת א֖וֹ אֶל־הַמְּזוּזָ֑ה וְרָצַ֨ע אֲדֹנָ֤יו אֶת־אׇזְנוֹ֙ בַּמַּרְצֵ֔עַ וַעֲבָד֖וֹ לְעֹלָֽם׃ אֲדוֹנוֹ יָבִיא אוֹתוֹ לְבֵית הַדִּין וְיָבִיא אוֹתוֹ לַדֶּלֶת אוֹ לִמְזוּזַת הַדֶּלֶת וְיִנְקוֹב אֶת אָזְנוֹ בְּמַרְצֵעַ, וְהוּא יַעֲבוֹד אוֹתוֹ לְעוֹלָם.
ס
ז וְכִֽי־יִמְכֹּ֥ר אִ֛ישׁ אֶת־בִּתּ֖וֹ לְאָמָ֑ה לֹ֥א תֵצֵ֖א כְּצֵ֥את הָעֲבָדִֽים׃ וְאִם אִישׁ יִמְכּוֹר אֶת בִּתּוֹ לְשִׁפְחָה, הִיא לֹא תֵּצֵא לַחָפְשִׁי כִּיצִיאַת שְׁאָר הָעֲבָדִים.
ח אִם־רָעָ֞ה בְּעֵינֵ֧י אֲדֹנֶ֛יהָ אֲשֶׁר־ל֥וֹ לא יְעָדָ֖הּ וְהֶפְדָּ֑הּ לְעַ֥ם נׇכְרִ֛י לֹא־יִמְשֹׁ֥ל לְמׇכְרָ֖הּ בְּבִגְדוֹ־בָֽהּ׃ אִם אֵינָהּ מוֹצֵאת חֵן בְּעֵינֵי אֲדוֹנָהּ שֶׁאֵינוֹ רוֹצֶה לָשֵׂאת אוֹתָהּ לְאִשָּׁה, עָלָיו לְאַפְשֵׁר שֶׁיִּפְדּוּ אוֹתָהּ, וְלֹא יִבְגּוֹד בָּהּ לִשְׁלוֹט בָּהּ וְלִמְכּוֹר אוֹתָהּ לְאִישׁ אַחֵר.
ט וְאִם־לִבְנ֖וֹ יִֽיעָדֶ֑נָּה כְּמִשְׁפַּ֥ט הַבָּנ֖וֹת יַעֲשֶׂה־לָּֽהּ׃ וְאִם הוּא יְיַחֵד אוֹתָהּ שֶׁתִּהְיֶה אִשָּׁה לִבְנוֹ – יַעֲשֶׂה לָהּ כְּדִין כָּל בְּנוֹת יִשְׂרָאֵל.
י אִם־אַחֶ֖רֶת יִֽקַּֽח־ל֑וֹ שְׁאֵרָ֛הּ כְּסוּתָ֥הּ וְעֹנָתָ֖הּ לֹ֥א יִגְרָֽע׃ אִם יִקַּח לוֹ אִשָּׁה אַחֶרֶת – הוּא לֹא יַחְסִיר מִמְּזוֹנָהּ, בְּגָדֶיהָ, וּזְמַנֵּי הַשְּׁכִיבָה אִתָּהּ.
יא וְאִ֨ם־שְׁלׇשׁ־אֵ֔לֶּה לֹ֥א יַעֲשֶׂ֖ה לָ֑הּ וְיָצְאָ֥ה חִנָּ֖ם אֵ֥ין כָּֽסֶף׃ וְאִם לֹא יַעֲשֶׂה לָהּ אַחַת מִשְּׁלוֹשׁ הָאֶפְשָׁרוּיוֹת הָאֵלֶּה, הִיא תֵּצֵא בְּחִנָּם בְּלִי לְשַׁלֵּם כֶּסֶף.
ס פָּרָשַׁת הַמַּכִּים
יב מַכֵּ֥ה אִ֛ישׁ וָמֵ֖ת מ֥וֹת יוּמָֽת׃ מִי שֶׁמַּכֶּה אִישׁ וְהוּא מֵת – יוּמַת.
יג וַאֲשֶׁר֙ לֹ֣א צָדָ֔ה וְהָאֱלֹהִ֖ים אִנָּ֣ה לְיָד֑וֹ וְשַׂמְתִּ֤י לְךָ֙ מָק֔וֹם אֲשֶׁ֥ר יָנ֖וּס שָֽׁמָּה׃ וּמִי שֶׁלֹּא הִתְכַּוֵּן לַהֲרוֹג וְהָאֱלֹהִים זִמֵּן לְיָדוֹ אָסוֹן בְּטָעוּת, אֲיַעֵד לְךָ מָקוֹם שֶׁיִּבְרַח לְשָׁם.
ס
יד וְכִֽי־יָזִ֥ד אִ֛ישׁ עַל־רֵעֵ֖הוּ לְהׇרְג֣וֹ בְעׇרְמָ֑ה מֵעִ֣ם מִזְבְּחִ֔י תִּקָּחֶ֖נּוּ לָמֽוּת׃ וְאִם אִישׁ יְתַכְנֵן בְּרִשְׁעוּת עַל חֲבֵירוֹ לְהָרְגוֹ בְּעַרְמוּמִיּוּת, אֲפִלּוּ אִם יִבְרַח לְמִזְבְּחִי תִּקַּח אוֹתוֹ לָמוּת.
ס
טו וּמַכֵּ֥ה אָבִ֛יו וְאִמּ֖וֹ מ֥וֹת יוּמָֽת׃ וּמִי שֶׁמַּכֵּה אֶת אָבִיו אוֹ אֶת אִמּוֹ – יוּמַת.
ס
טז וְגֹנֵ֨ב אִ֧ישׁ וּמְכָר֛וֹ וְנִמְצָ֥א בְיָד֖וֹ מ֥וֹת יוּמָֽת׃ וְגוֹנֵב אִישׁ וּמְכָרוֹ לְאַחַר שֶׁהָיָה בְּיָדָיו – יוּמַת.
ס
יז וּמְקַלֵּ֥ל אָבִ֛יו וְאִמּ֖וֹ מ֥וֹת יוּמָֽת׃ וּמִי שֶׁמְּקַלֵּל אֶת אָבִיו אוֹ אֶת אִמּוֹ – יוּמַת.
ס
יח וְכִֽי־יְרִיבֻ֣ן אֲנָשִׁ֔ים וְהִכָּה־אִישׁ֙ אֶת־רֵעֵ֔הוּ בְּאֶ֖בֶן א֣וֹ בְאֶגְרֹ֑ף וְלֹ֥א יָמ֖וּת וְנָפַ֥ל לְמִשְׁכָּֽב׃ וְאִם אֲנָשִׁים יָרִיבוּ וְאֶחָד יַכֶּה אֶת הַשֵּׁנִי בְּאֶבֶן אוֹ בְּאֶגְרוֹף וְהוּא לֹא יָמוּת אֶלָּא יֶחֱלֶה.
יט אִם־יָק֞וּם וְהִתְהַלֵּ֥ךְ בַּח֛וּץ עַל־מִשְׁעַנְתּ֖וֹ וְנִקָּ֣ה הַמַּכֶּ֑ה רַ֥ק שִׁבְתּ֛וֹ יִתֵּ֖ן וְרַפֹּ֥א יְרַפֵּֽא׃ אִם הוּא יָקוּם וְיִתְהַלֵּךְ בַּחוּץ אֲפִלּוּ עַל מַקֵּל, הַמַּכֶּה יִפָּטֵר מִמָּוֶת, הוּא יְשַׁלֵּם רַק פִּצּוּיִים עַל שֶׁלֹּא הָיָה יָכוֹל לַעֲבוֹד, וְאֶת הוֹצְאוֹת הָרִפּוּי.

ס
כ וְכִֽי־יַכֶּה֩ אִ֨ישׁ אֶת־עַבְדּ֜וֹ א֤וֹ אֶת־אֲמָתוֹ֙ בַּשֵּׁ֔בֶט וּמֵ֖ת תַּ֣חַת יָד֑וֹ נָקֹ֖ם יִנָּקֵֽם׃ וְאִם אִישׁ יַכֶּה אֶת עַבְדּוֹ אוֹ אֶת שִׁפְחָתוֹ בְּמַקֵּל וְהוּא יָמוּת מִמַּכָּתוֹ – יִנְקְמוּ בּוֹ וְיַהַרְגוּהוּ.
כא אַ֥ךְ אִם־י֛וֹם א֥וֹ יוֹמַ֖יִם יַעֲמֹ֑ד לֹ֣א יֻקַּ֔ם כִּ֥י כַסְפּ֖וֹ הֽוּא׃ אֲבָל אִם הוּא יַעֲמוֹד יוֹם אוֹ יוֹמַיִם בַּחַיִּים – לֹא יִנְקְמוּ בּוֹ, כִּי הָעֶבֶד הוּא רְכוּשׁוֹ.
ס
כב וְכִֽי־יִנָּצ֣וּ אֲנָשִׁ֗ים וְנָ֨גְפ֜וּ אִשָּׁ֤ה הָרָה֙ וְיָצְא֣וּ יְלָדֶ֔יהָ וְלֹ֥א יִהְיֶ֖ה אָס֑וֹן עָנ֣וֹשׁ יֵעָנֵ֗שׁ כַּֽאֲשֶׁ֨ר יָשִׁ֤ית עָלָיו֙ בַּ֣עַל הָֽאִשָּׁ֔ה וְנָתַ֖ן בִּפְלִלִֽים׃ וְאִם אֲנָשִׁים יָרִיבוּ וְיִדְחֲפוּ אִשָּׁה בְּהֵרָיוֹן וְעֻבָּרֶיהָ יֵצְאוּ אֲבָל לֹא יָמוּת אָדָם חַי – הַדּוֹחֵף יְחֻיָּב כְּשֶׁיִּתְבַּע אוֹתוֹ בַּעַל הָאִשָּׁה, וְהוּא יְשַׁלֵּם כִּפְסַק בֵּית הַדִּין.
כג וְאִם־אָס֖וֹן יִהְיֶ֑ה וְנָתַתָּ֥ה נֶ֖פֶשׁ תַּ֥חַת נָֽפֶשׁ׃ וְאִם יָמוּת אָז אָדָם חַי, הוּא יוּמַת בִּגְלַל הַנֶּפֶשׁ שֶׁרָצַח.
כד עַ֚יִן תַּ֣חַת עַ֔יִן שֵׁ֖ן תַּ֣חַת שֵׁ֑ן יָ֚ד תַּ֣חַת יָ֔ד רֶ֖גֶל תַּ֥חַת רָֽגֶל׃ וִישַׁלֵּם עֵרֶךְ שֶׁל עַיִן עַל הַשְׁחָתַת עַיִן, עֵרֶךְ שֶׁל שֵׁן עַל הַשְׁחָתַת שֵׁן, שֶׁל יָד עֲבוּר הַשְׁחָתַת יָד, וְשֶׁל רֶגֶל עֲבוּר הַשְׁחָתַת רֶגֶל,

ט כִּֽי־יִתֵּן֩ אִ֨ישׁ אֶל־רֵעֵ֜הוּ חֲמ֨וֹר אוֹ־שׁ֥וֹר אוֹ־שֶׂ֛ה וְכׇל־בְּהֵמָ֖ה לִשְׁמֹ֑ר וּמֵ֛ת אוֹ־נִשְׁבַּ֥ר אוֹ־נִשְׁבָּ֖ה אֵ֥ין רֹאֶֽה׃ אִם אִישׁ יִתֵּן לַחֲבֵרוֹ חֲמוֹר אוֹ שׁוֹר אוֹ שֶׂה אוֹ כָּל בְּהֵמָה אַחֶרֶת לִשְׁמוֹר, וּמֵת אוֹ נִפְצַע אוֹ נֶחְטַף, וְאֵין רוֹאֶה.
י שְׁבֻעַ֣ת יְהֹוָ֗ה תִּהְיֶה֙ בֵּ֣ין שְׁנֵיהֶ֔ם אִם־לֹ֥א שָׁלַ֛ח יָד֖וֹ בִּמְלֶ֣אכֶת רֵעֵ֑הוּ וְלָקַ֥ח בְּעָלָ֖יו וְלֹ֥א יְשַׁלֵּֽם׃ שְׁבוּעַת ה' תִּהְיֶה בֵּין שְׁנֵיהֶם שֶׁלֹּא הוֹשִׁיט אֶת יָדוֹ לִגְנוֹב אוֹ לְהִשְׁתַּמֵּשׁ בִּרְכוּשׁ חֲבֵרוֹ; בְּעָלָיו יוּכַל לִדְרוֹשׁ שֶׁיִּשָּׁבַע, וְאָז לֹא יְשַׁלֵּם כֶּסֶף.
יא וְאִם־גָּנֹ֥ב יִגָּנֵ֖ב מֵעִמּ֑וֹ יְשַׁלֵּ֖ם לִבְעָלָֽיו׃ וְאִם יִגָּנֵב מִמֶּנּוּ יְשַׁלֵּם לִבְעָלָיו.
יב אִם־טָרֹ֥ף יִטָּרֵ֖ף יְבִאֵ֣הוּ עֵ֑ד הַטְּרֵפָ֖ה לֹ֥א יְשַׁלֵּֽם׃ אִם יִטָּרֵף – יָבִיא אֶת הַנְּבֵלָה לִרְאָיָה שֶׁנִּטְרְפָה, וְלֹא יְשַׁלֵּם עַל הַטְּרֵפָה.
פ
יג וְכִֽי־יִשְׁאַ֥ל אִ֛ישׁ מֵעִ֥ם רֵעֵ֖הוּ וְנִשְׁבַּ֣ר אוֹ־מֵ֑ת בְּעָלָ֥יו אֵין־עִמּ֖וֹ שַׁלֵּ֥ם יְשַׁלֵּֽם׃ וְאִם אִישׁ יִקַּח מֵחֲבֵרוֹ בְּהֵמָה בְּהַשְׁאָלָה וְהִיא תִּפָּצַע אוֹ תָּמוּת, אִם בְּעָלָיו לֹא אִתּוֹ – יְשַׁלֵּם.
יד אִם־בְּעָלָ֥יו עִמּ֖וֹ לֹ֣א יְשַׁלֵּ֑ם אִם־שָׂכִ֣יר ה֔וּא בָּ֖א בִּשְׂכָרֽוֹ׃ אִם בַּעַל הַבְּהֵמָה אִתּוֹ – לֹא יְשַׁלֵּם, וְכֵן אִם הַבְּעָלִים שָׂכִיר שֶׁל הַשּׁוֹאֵל, כִּי הוּא כְּבָר קִבֵּל אֶת הַפִּצּוּי עַל הַנֵּזֶק בִּשְׂכָרוֹ.
ס פָּרָשַׁת דִּינֵי חוֹטְאִים
טו וְכִֽי־יְפַתֶּ֣ה אִ֗ישׁ בְּתוּלָ֛ה אֲשֶׁ֥ר לֹא־אֹרָ֖שָׂה וְשָׁכַ֣ב עִמָּ֑הּ מָהֹ֛ר יִמְהָרֶ֥נָּה לּ֖וֹ לְאִשָּֽׁה׃ וְאִם אִישׁ יְפַתֶּה בְּתוּלָה שֶׁאֵינֶנָּה מְאוֹרֶסֶת וְיִשְׁכַּב אִתָּהּ – יְשַׁלֵּם לְאָבִיהָ כַּנָּהוּג כְּדֵי שֶׁתִּהְיֶה לוֹ לְאִשָּׁה.
טז אִם־מָאֵ֧ן יְמָאֵ֛ן אָבִ֖יהָ לְתִתָּ֣הּ ל֑וֹ כֶּ֣סֶף יִשְׁקֹ֔ל כְּמֹ֖הַר הַבְּתוּלֹֽת׃ אִם אָבִיהָ יְסָרֵב לָתֵת לוֹ אוֹתָהּ, יִתֵּן לוֹ מִשְׁקַל שֶׁל כֶּסֶף כַּנָּהוּג בִּכְתוּבַּת הַבְּתוּלוֹת.
ס
יז מְכַשֵּׁפָ֖ה לֹ֥א תְחַיֶּֽה׃ אַל תַּשְׁאִיר מְכַשֵּׁפָה בַּחַיִּים.
יח כׇּל־שֹׁכֵ֥ב עִם־בְּהֵמָ֖ה מ֥וֹת יוּמָֽת׃ כָּל שׁוֹכֵב עִם בְּהֵמָה – יוּמַת.
ס
יט זֹבֵ֥חַ לָאֱלֹהִ֖ים יׇֽחֳרָ֑ם בִּלְתִּ֥י לַיהֹוָ֖ה לְבַדּֽוֹ׃ הַמַּקְרִיב לְכָל אֱלֹהוּת חוּץ מִלַּה' בִּלְבָד – יוּמַת.
כ וְגֵ֥ר לֹא־תוֹנֶ֖ה וְלֹ֣א תִלְחָצֶ֑נּוּ כִּֽי־גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֥רֶץ מִצְרָֽיִם׃ וְאַל תְּצַעֵר גֵּר וְאַל תָּצִיק לוֹ, הֲרֵי הֱיִיתֶם גֵּרִים בְּאֶרֶץ מִצְרַיִם.
כא כׇּל־אַלְמָנָ֥ה וְיָת֖וֹם לֹ֥א תְעַנּֽוּן׃ אַל תַּשְׁפִּילוּ שׁוּם אַלְמָנָה וְיָתוֹם.
כב אִם־עַנֵּ֥ה תְעַנֶּ֖ה אֹת֑וֹ כִּ֣י אִם־צָעֹ֤ק יִצְעַק֙ אֵלַ֔י שָׁמֹ֥עַ אֶשְׁמַ֖ע צַעֲקָתֽוֹ׃ אִם תַּשְׁפִּיל אוֹתוֹ, כְּשֶׁיִּצְעַק אֵלַי בִּתְפִלָּה אֶשְׁמַע אֶת צַעֲקָתוֹ.
כג וְחָרָ֣ה אַפִּ֔י וְהָרַגְתִּ֥י אֶתְכֶ֖ם בֶּחָ֑רֶב וְהָי֤וּ נְשֵׁיכֶם֙ אַלְמָנ֔וֹת וּבְנֵיכֶ֖ם יְתֹמִֽים׃ וְאֶכְעַס וְאֶהֱרוֹג אֶתְכֶם בַּחֶרֶב, וּנְשֵׁיכֶם תִּהְיֶינָה אַלְמָנוֹת וּבְנֵיכֶם יְתוֹמִים.
פ
כד אִם־כֶּ֣סֶף ׀ תַּלְוֶ֣ה אֶת־עַמִּ֗י אֶת־הֶֽעָנִי֙ עִמָּ֔ךְ לֹא־תִהְיֶ֥ה ל֖וֹ כְּנֹשֶׁ֑ה לֹֽא־תְשִׂימ֥וּן עָלָ֖יו נֶֽשֶׁךְ׃ אִם תַּלְוֶה כֶּסֶף לְאָדָם מֵעַמִּי – לֶעָנִי שֶׁאִתְּךָ, אַל תִּגְבֶּה מִמֶּנּוּ בְּחָזְקָה, וְאַל תִּגְבֶּה מִמֶּנּוּ רִבִּית.
כה אִם־חָבֹ֥ל תַּחְבֹּ֖ל שַׂלְמַ֣ת רֵעֶ֑ךָ עַד־בֹּ֥א הַשֶּׁ֖מֶשׁ תְּשִׁיבֶ֥נּוּ לֽוֹ׃ אִם תִּקַּח אֶת בֶּגֶד חֲבֵרְךָ לְמַשְׁכּוֹן תַּחֲזִיר לוֹ אוֹתוֹ עַד שְׁקִיעַת הַשֶּׁמֶשׁ.
כו כִּ֣י הִ֤וא כְסוּתֹה֙ לְבַדָּ֔הּ הִ֥וא שִׂמְלָת֖וֹ לְעֹר֑וֹ בַּמֶּ֣ה יִשְׁכָּ֔ב וְהָיָה֙ כִּֽי־יִצְעַ֣ק אֵלַ֔י וְשָׁמַעְתִּ֖י כִּֽי־חַנּ֥וּן אָֽנִי׃ כִּי בֶּגֶד זֶה הוּא לְבוּשׁוֹ הַיָּחִיד, הוּא בִּגְדוֹ לְעוֹרוֹ, עִם מַה יִּשְׁכָּב?! וְאָז, אִם יִצְעַק אֵלַי בִּתְפִלָּה אֶשְׁמַע, כִּי אֲנִי רַחְמָן.

רביעי[עריכה]

ס
כז אֱלֹהִ֖ים לֹ֣א תְקַלֵּ֑ל וְנָשִׂ֥יא בְעַמְּךָ֖ לֹ֥א תָאֹֽר׃ אַל תְּבַזֶּה דַּיָּן, וְאַל תְּקַלֵּל מַנְהִיג בְּעַמְּךָ.
כח מְלֵאָתְךָ֥ וְדִמְעֲךָ֖ לֹ֣א תְאַחֵ֑ר בְּכ֥וֹר בָּנֶ֖יךָ תִּתֶּן־לִֽי׃ אַל תְּאַחֵר לְהַפְרִישׁ אֶת הַבִּכּוּרִים וְהַתְּרוּמָה שֶׁלְּךָ, אֶת בְּכוֹר בָּנֶיךָ תִּתֵּן לִי.
כט כֵּֽן־תַּעֲשֶׂ֥ה לְשֹׁרְךָ֖ לְצֹאנֶ֑ךָ שִׁבְעַ֤ת יָמִים֙ יִהְיֶ֣ה עִם־אִמּ֔וֹ בַּיּ֥וֹם הַשְּׁמִינִ֖י תִּתְּנוֹ־לִֽי׃ וְכָךְ תַּעֲשֶׂה גַּם לְשׁוֹרְךָ וּלְצֹאנְךָ – שִׁבְעָה יָמִים יִהְיֶה עִם אִמּוֹ וּבַיּוֹם הַשְּׁמִינִי תִּתֵּן לִי אוֹתוֹ.
ל וְאַנְשֵׁי־קֹ֖דֶשׁ תִּהְי֣וּן לִ֑י וּבָשָׂ֨ר בַּשָּׂדֶ֤ה טְרֵפָה֙ לֹ֣א תֹאכֵ֔לוּ לַכֶּ֖לֶב תַּשְׁלִכ֥וּן אֹתֽוֹ׃ וְתִהְיוּ לִי אֲנָשִׁים קְדוֹשִׁים, אַל תֹּאכְלוּ בְּשַׂר בְּהֵמָה שֶׁנִּטְרְפָה בַּשָּׂדֶה, אֶלָּא הַשְׁלִיכוּ אוֹתוֹ לַכֶּלֶב.
ס
כג א לֹ֥א תִשָּׂ֖א שֵׁ֣מַע שָׁ֑וְא אַל־תָּ֤שֶׁת יָֽדְךָ֙ עִם־רָשָׁ֔ע לִהְיֹ֖ת עֵ֥ד חָמָֽס׃ אַל תַּאֲמִין לִשְׁמוּעַת שֶׁקֶר, אַל תּוֹשִׁיט אֶת יָדְךָ לְסַיֵּעַ לְרָשָׁע – לִהְיוֹת עֵד שֶׁקֶר.
ב לֹֽא־תִהְיֶ֥ה אַחֲרֵֽי־רַבִּ֖ים לְרָעֹ֑ת וְלֹא־תַעֲנֶ֣ה עַל־רִ֗ב לִנְטֹ֛ת אַחֲרֵ֥י רַבִּ֖ים לְהַטֹּֽת׃ אַל תִּמָּשֵׁךְ אַחֲרֵי רַבִּים לַעֲשׂוֹת רָע, וְאַל תַּסְכִּים בְּוִכּוּחַ לִנְטוֹת מִדַּעַתְךָ וּלְהִמָּשֵׁךְ אַחֲרֵי רַבִּים, לְשַׁנּוֹת דִּין.
ג וְדָ֕ל לֹ֥א תֶהְדַּ֖ר בְּרִיבֽוֹ׃ וְאַל תִּתֵּן לְעָנִי עֲדִיפוּת בְּמִשְׁפָּטוֹ.
ס
ד כִּ֣י תִפְגַּ֞ע שׁ֧וֹר אֹֽיִבְךָ֛ א֥וֹ חֲמֹר֖וֹ תֹּעֶ֑ה הָשֵׁ֥ב תְּשִׁיבֶ֖נּוּ לֽוֹ׃ אִם תִּפְגּוֹשׁ שׁוֹר שֶׁל אוֹיֵב שֶׁלְּךָ אוֹ חֲמוֹר שֶׁלּוֹ תּוֹעֶה בַּדֶּרֶךְ – הַחֲזֵר לוֹ אוֹתוֹ.
ס
ה כִּֽי־תִרְאֶ֞ה חֲמ֣וֹר שֹׂנַאֲךָ֗ רֹבֵץ֙ תַּ֣חַת מַשָּׂא֔וֹ וְחָדַלְתָּ֖ מֵעֲזֹ֣ב ל֑וֹ עָזֹ֥ב תַּעֲזֹ֖ב עִמּֽוֹ׃ אִם תִּרְאֶה חֲמוֹר שֶׁל שׂוֹנֵא שֶׁלְּךָ רוֹבֵץ תַּחַת מַשָּׂאוֹ וּבְשִׂנְאָתְךָ תִּמָּנַע מִלַּעֲזוֹר לוֹ – עֲזוֹב שִׂנְאָתְךָ וַעֲזוֹר לוֹ.

ז מִדְּבַר־שֶׁ֖קֶר תִּרְחָ֑ק וְנָקִ֤י וְצַדִּיק֙ אַֽל־תַּהֲרֹ֔ג כִּ֥י לֹא־אַצְדִּ֖יק רָשָֽׁע׃ הִתְרַחֵק מִדְּבַר שֶׁקֶר, אַל תַּהֲרוֹג חַף מִפֶּשַׁע וְאֶת מִי שֶׁיָּצָא זַכַּאי בַּדִּין, כִּי אֲנִי לֹא אַצְדִּיק בַּדִּין אֶת מִי שֶׁבֶּאֱמֶת רָשָׁע.
ח וְשֹׁ֖חַד לֹ֣א תִקָּ֑ח כִּ֤י הַשֹּׁ֙חַד֙ יְעַוֵּ֣ר פִּקְחִ֔ים וִֽיסַלֵּ֖ף דִּבְרֵ֥י צַדִּיקִֽים׃ וְאַל תִּקַּח שׁוֹחַד כִּי הַשּׁוֹחַד יְעַוֵּר פִּקְחִים מִלִּרְאוֹת אֶת הָאֱמֶת, וִיעַוֵּת דִּבְרֵי צַדִּיקִים
ט וְגֵ֖ר לֹ֣א תִלְחָ֑ץ וְאַתֶּ֗ם יְדַעְתֶּם֙ אֶת־נֶ֣פֶשׁ הַגֵּ֔ר כִּֽי־גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֥רֶץ מִצְרָֽיִם׃ וְאַל תָּצִיק לְגֵר, וְאַתֶּם מַכִּירִים אֶת נֶפֶשׁ הַגֵּר שֶׁהֲרֵי הֱיִיתֶם גֵּרִים בְּאֶרֶץ מִצְרַיִם.
פָּרָשַׁת מוֹעֲדִים
י וְשֵׁ֥שׁ שָׁנִ֖ים תִּזְרַ֣ע אֶת־אַרְצֶ֑ךָ וְאָסַפְתָּ֖ אֶת־תְּבוּאָתָֽהּ׃ וְשֵׁשׁ שָׁנִים תִּזְרַע אֶת אַרְצְךָ וְתֶאֱסוֹף אֶת הַתְּבוּאָה שֶׁלָּהּ.
יא וְהַשְּׁבִיעִ֞ת תִּשְׁמְטֶ֣נָּה וּנְטַשְׁתָּ֗הּ וְאָֽכְלוּ֙ אֶבְיֹנֵ֣י עַמֶּ֔ךָ וְיִתְרָ֕ם תֹּאכַ֖ל חַיַּ֣ת הַשָּׂדֶ֑ה כֵּֽן־תַּעֲשֶׂ֥ה לְכַרְמְךָ֖ לְזֵיתֶֽךָ׃ וּבַשָּׁנָה הַשְּׁבִיעִית תַּפְקִיר אֶת הַשָּׂדֶה וְתַעֲזוֹב אֶת פֵּרוֹתֶיהָ וְיֹאכְלוּ אוֹתָם עֲנִיֵּי עַמְּךָ, וְאֶת מַה שֶּׁנִּשְׁאָר יֹאכְלוּ חַיּוֹת הַשָּׂדֶה, כָּךְ תַּעֲשֶׂה גַּם לְכַרְמְךָ וְלַעֲצֵי הַזַּיִת שֶׁלְּךָ.
יב שֵׁ֤שֶׁת יָמִים֙ תַּעֲשֶׂ֣ה מַעֲשֶׂ֔יךָ וּבַיּ֥וֹם הַשְּׁבִיעִ֖י תִּשְׁבֹּ֑ת לְמַ֣עַן יָנ֗וּחַ שֽׁוֹרְךָ֙ וַחֲמֹרֶ֔ךָ וְיִנָּפֵ֥שׁ בֶּן־אֲמָתְךָ֖ וְהַגֵּֽר׃ שִׁשָּׁה יָמִים תַּעֲשֶׂה אֶת מַעֲשֶׂיךָ וּבַיּוֹם הַשְּׁבִיעִי תִּשְׁבּוֹת, כְּדֵי שֶׁשּׁוֹרְךָ וַחֲמוֹרְךָ יָנוּחוּ וּלְבֶן שִׁפְחָתְךָ וְלַגֵּר תִּהְיֶה מְנוּחָה.
יג וּבְכֹ֛ל אֲשֶׁר־אָמַ֥רְתִּי אֲלֵיכֶ֖ם תִּשָּׁמֵ֑רוּ וְשֵׁ֨ם אֱלֹהִ֤ים אֲחֵרִים֙ לֹ֣א תַזְכִּ֔ירוּ לֹ֥א יִשָּׁמַ֖ע עַל־פִּֽיךָ׃ וְתִזָּהֲרוּ בְּכָל הַמִּצְווֹת שֶׁצִּוִּיתִי אֶתְכֶם, וְאַל תַּזְכִּירוּ שֵׁם אֱלֹהִים אֲחֵרִים – לֹא יִשָּׁמַע מִפִּיךָ.
יד שָׁלֹ֣שׁ רְגָלִ֔ים תָּחֹ֥ג לִ֖י בַּשָּׁנָֽה׃ תַּחְגּוֹג לִי שָׁלוֹשׁ פְּעָמִים בַּשָּׁנָה:
טו אֶת־חַ֣ג הַמַּצּוֹת֮ תִּשְׁמֹר֒ שִׁבְעַ֣ת יָמִים֩ תֹּאכַ֨ל מַצּ֜וֹת כַּֽאֲשֶׁ֣ר צִוִּיתִ֗ךָ לְמוֹעֵד֙ חֹ֣דֶשׁ הָֽאָבִ֔יב כִּי־ב֖וֹ יָצָ֣אתָ מִמִּצְרָ֑יִם וְלֹא־יֵרָא֥וּ פָנַ֖י רֵיקָֽם׃ אֶת חַג הַמַּצּוֹת תִּשְׁמוֹר – שִׁבְעָה יָמִים תֹּאכַל מַצּוֹת כְּמוֹ שֶׁצִּוִּיתִי אוֹתְךָ – בִּזְמַן חוֹדֶשׁ הָאָבִיב, כִּי בּוֹ יָצָאתָ מִמִּצְרַיִם; וְאַל תָּבוֹא לְפָנַי לְבֵית הַמִּקְדָּשׁ בְּלִי קָרְבָּן.
טז וְחַ֤ג הַקָּצִיר֙ בִּכּוּרֵ֣י מַעֲשֶׂ֔יךָ אֲשֶׁ֥ר תִּזְרַ֖ע בַּשָּׂדֶ֑ה וְחַ֤ג הָֽאָסִף֙ בְּצֵ֣את הַשָּׁנָ֔ה בְּאׇסְפְּךָ֥ אֶֽת־מַעֲשֶׂ֖יךָ מִן־הַשָּׂדֶֽה׃ וְאֶת חַג הַקָּצִיר בִּזְמַן תְּחִלַּת קְצִירַת תְּבוּאָתְךָ שֶׁתִּזְרַע בַּשָּׂדֶה, וְאֶת חַג הָאָסִיף בְּצֵאת שְׁנַת הַיְּבוּל, כְּשֶׁתַּאֲסוֹף אֶת תְּבוּאָתְךָ מֵהַשָּׂדֶה.
יז שָׁלֹ֥שׁ פְּעָמִ֖ים בַּשָּׁנָ֑ה יֵרָאֶה֙ כׇּל־זְכ֣וּרְךָ֔ אֶל־פְּנֵ֖י הָאָדֹ֥ן ׀ יְהֹוָֽה׃ שָׁלוֹשׁ פְּעָמִים בַּשָּׁנָה יֵרָאוּ כָּל הַזְּכָרִים שֶׁלָּכֶם לִפְנֵי הָאָדוֹן ה'.
יח לֹֽא־תִזְבַּ֥ח עַל־חָמֵ֖ץ דַּם־זִבְחִ֑י וְלֹֽא־יָלִ֥ין חֵֽלֶב־חַגִּ֖י עַד־בֹּֽקֶר׃ אַל תִּשְׁחַט אֶת קָרְבַּן הַפֶּסַח שֶׁדָּמוֹ נִזְרַק לַמִּזְבֵּחַ כְּשֶׁיֵּשׁ אֶצְלְךָ חָמֵץ, וְאַל תַּשְׁאִיר עַד הַבּוֹקֶר אֶת חֵלֶב קָרְבַּן הֶחָג שֶׁקָּרֵב לַמִּזְבֵּחַ.
יט רֵאשִׁ֗ית בִּכּוּרֵי֙ אַדְמָ֣תְךָ֔ תָּבִ֕יא בֵּ֖ית יְהֹוָ֣ה אֱלֹהֶ֑יךָ לֹֽא־תְבַשֵּׁ֥ל גְּדִ֖י בַּחֲלֵ֥ב אִמּֽוֹ׃ אֶת הַתְחָלַת בִּכּוּרֵי אַדְמָתְךָ תָּבִיא לְבֵית ה' אֱלֹהֶיךָ, אַל תְּבַשֵּׁל וְאַל תֹּאכַל בְּשַׂר בְּהֵמָה כְּשֵׁרָה בְּחָלָב בְּהֵמָה כְּשֵׁרָה.

ששי[עריכה]

פ פָּרָשַׁת הַהֲכָנָה לַכְּנִיסָה לָאָרֶץ
כ הִנֵּ֨ה אָנֹכִ֜י שֹׁלֵ֤חַ מַלְאָךְ֙ לְפָנֶ֔יךָ לִשְׁמׇרְךָ֖ בַּדָּ֑רֶךְ וְלַהֲבִ֣יאֲךָ֔ אֶל־הַמָּק֖וֹם אֲשֶׁ֥ר הֲכִנֹֽתִי׃ הִנֵּה אֲנִי שׁוֹלֵחַ מַלְאָךְ לְפָנֶיךָ לִשְׁמוֹר אוֹתְךָ בַּדֶּרֶךְ וּלְהָבִיא אוֹתְךָ לַמָּקוֹם שֶׁהֵכַנְתִּי.
כא הִשָּׁ֧מֶר מִפָּנָ֛יו וּשְׁמַ֥ע בְּקֹל֖וֹ אַל־תַּמֵּ֣ר בּ֑וֹ כִּ֣י לֹ֤א יִשָּׂא֙ לְפִשְׁעֲכֶ֔ם כִּ֥י שְׁמִ֖י בְּקִרְבּֽוֹ׃ הִזָּהֵר מִמֶּנּוּ וּשְׁמַע בְּקוֹלוֹ וְאַל תְּסָרֵב לְצִוּוּיוֹ, כִּי הוּא לֹא יִסְלַח לְפִשְׁעֲכֶם כִּי אֲנִי מִתְגַּלֶּה דַּרְכּוֹ.
כב כִּ֣י אִם־שָׁמ֤וֹעַ תִּשְׁמַע֙ בְּקֹל֔וֹ וְעָשִׂ֕יתָ כֹּ֖ל אֲשֶׁ֣ר אֲדַבֵּ֑ר וְאָֽיַבְתִּי֙ אֶת־אֹ֣יְבֶ֔יךָ וְצַרְתִּ֖י אֶת־צֹרְרֶֽיךָ׃ אֶלָּא שְׁמַע בְּקוֹלוֹ וְתַעֲשֶׂה אֶת כָּל מַה שֶּׁאֲדַבֵּר, וְאֶהְיֶה אוֹיֵב לְאוֹיְבֶיךָ וְאֶעֱשֶׂה צָרוֹת לְשׂוֹנְאֶיךָ.
כג כִּֽי־יֵלֵ֣ךְ מַלְאָכִי֮ לְפָנֶ֒יךָ֒ וֶהֱבִֽיאֲךָ֗ אֶל־הָֽאֱמֹרִי֙ וְהַ֣חִתִּ֔י וְהַפְּרִזִּי֙ וְהַֽכְּנַעֲנִ֔י הַחִוִּ֖י וְהַיְבוּסִ֑י וְהִכְחַדְתִּֽיו׃ כְּשֶׁיֵּלֵךְ מַלְאָכִי לְפָנֶיךָ הוּא יָבִיא אוֹתְךָ אֶל הָאֱמוֹרִי וְהַחִתִּי וְהַפְּרִזִּי וְהַכְּנַעֲנִי הַחִוִּי וְהַיְּבוּסִי, וַאֲנִי אַשְׁמִיד אוֹתָם.
כד לֹֽא־תִשְׁתַּחֲוֶ֤ה לֵאלֹֽהֵיהֶם֙ וְלֹ֣א תׇֽעׇבְדֵ֔ם וְלֹ֥א תַעֲשֶׂ֖ה כְּמַֽעֲשֵׂיהֶ֑ם כִּ֤י הָרֵס֙ תְּהָ֣רְסֵ֔ם וְשַׁבֵּ֥ר תְּשַׁבֵּ֖ר מַצֵּבֹתֵיהֶֽם׃ אַל תִּשְׁתַּחֲוֶה לֵאלֹהֵיהֶם וְאַל תַּעֲבוֹד אוֹתָם, וְאַל תַּעֲשֶׂה כְּמַעֲשֵׂיהֶם, אֶלָּא תַּהֲרוֹס אוֹתָם וְתִשְׁבּוֹר אֶת מַצֵּבוֹתֵיהֶם.
כה וַעֲבַדְתֶּ֗ם אֵ֚ת יְהֹוָ֣ה אֱלֹֽהֵיכֶ֔ם וּבֵרַ֥ךְ אֶֽת־לַחְמְךָ֖ וְאֶת־מֵימֶ֑יךָ וַהֲסִרֹתִ֥י מַחֲלָ֖ה מִקִּרְבֶּֽךָ׃ עִבְדוּ אֶת ה' אֱלֹהֵיכֶם וְה' יְבָרֵךְ אֶת מְזוֹנְךָ וְאֶת מֵימֶיךָ, וַאֲסַלֵּק מִמְּךָ מַחֲלוֹת.

שביעי[עריכה]

ס
כו לֹ֥א תִהְיֶ֛ה מְשַׁכֵּלָ֥ה וַעֲקָרָ֖ה בְּאַרְצֶ֑ךָ אֶת־מִסְפַּ֥ר יָמֶ֖יךָ אֲמַלֵּֽא׃ לֹא תִּהְיֶה בְּאַרְצְךָ אִשָּׁה שֶׁבְּנָהּ יָמוּת אוֹ שֶׁאֵינָהּ יוֹלֶדֶת, אַשְׁלִים אֶת מִסְפַּר הַשָּׁנִים הַמַּגִּיעוֹת לְךָ.
כז אֶת־אֵֽימָתִי֙ אֲשַׁלַּ֣ח לְפָנֶ֔יךָ וְהַמֹּתִי֙ אֶת־כׇּל־הָעָ֔ם אֲשֶׁ֥ר תָּבֹ֖א בָּהֶ֑ם וְנָתַתִּ֧י אֶת־כׇּל־אֹיְבֶ֛יךָ אֵלֶ֖יךָ עֹֽרֶף׃ אֶת פַּחְדִּי אֶשְׁלַח לְפָנֶיךָ – אֶעֱשֶׂה אֶת כָּל הָעָם שֶׁתָּבוֹא בְּאַרְצָם הֲמוּמִים, וְאֶעֱשֶׂה שֶׁכָּל אוֹיְבֶיךָ יִבְרְחוּ וְיַפְנוּ אֵלֶיךָ עוֹרֶף.
כח וְשָׁלַחְתִּ֥י אֶת־הַצִּרְעָ֖ה לְפָנֶ֑יךָ וְגֵרְשָׁ֗ה אֶת־הַחִוִּ֧י אֶת־הַֽכְּנַעֲנִ֛י וְאֶת־הַחִתִּ֖י מִלְּפָנֶֽיךָ׃ אֶשְׁלַח אֶת הַצִּרְעָה לְפָנֶיךָ וְהִיא תְּגָרֵשׁ אֶת הַחִוִּי אֶת הַכְּנַעֲנִי וְאֶת הַחִתִּי מִלְּפָנֶיךָ.
כט לֹ֧א אֲגָרְשֶׁ֛נּוּ מִפָּנֶ֖יךָ בְּשָׁנָ֣ה אֶחָ֑ת פֶּן־תִּהְיֶ֤ה הָאָ֙רֶץ֙ שְׁמָמָ֔ה וְרַבָּ֥ה עָלֶ֖יךָ חַיַּ֥ת הַשָּׂדֶֽה׃ לֹא אֲגָרֵשׁ אוֹתוֹ מִפָּנֶיךָ תּוֹךְ שָׁנָה אַחַת, כְּדֵי שֶׁהָאָרֶץ לֹא תִּהְיֶה שְׁמָמָה וְלֹא יִתְרַבּוּ עָלֶיךָ חַיּוֹת הַשָּׂדֶה.
ל מְעַ֥ט מְעַ֛ט אֲגָרְשֶׁ֖נּוּ מִפָּנֶ֑יךָ עַ֚ד אֲשֶׁ֣ר תִּפְרֶ֔ה וְנָחַלְתָּ֖ אֶת־הָאָֽרֶץ׃ אֲגָרֵשׁ אוֹתוֹ מִפָּנֶיךָ לְאַט לְאַט, עַד שֶׁתּוֹלִיד בָּנִים וְתִתְרַבֶּה וְתִנְחַל אֶת הָאָרֶץ.
לא וְשַׁתִּ֣י אֶת־גְּבֻלְךָ֗ מִיַּם־סוּף֙ וְעַד־יָ֣ם פְּלִשְׁתִּ֔ים וּמִמִּדְבָּ֖ר עַד־הַנָּהָ֑ר כִּ֣י ׀ אֶתֵּ֣ן בְּיֶדְכֶ֗ם אֵ֚ת יֹשְׁבֵ֣י הָאָ֔רֶץ וְגֵרַשְׁתָּ֖מוֹ מִפָּנֶֽיךָ׃ וְאָשִׂים אֶת גְּבוּלְךָ מִיַּם סוּף וְעַד הַיָּם הַתִּיכוֹן וּמִמִּדְבַּר עֲרָב עַד נְהַר פְּרָת, כִּי אֶתֵּן בְּיֶדְכֶם אֶת יוֹשְׁבֵי הָאָרֶץ וּתְגָרֵשׁ אוֹתָם מִפָּנֶיךָ.
לב לֹֽא־תִכְרֹ֥ת לָהֶ֛ם וְלֵאלֹֽהֵיהֶ֖ם בְּרִֽית׃ אַל תַּעֲשֶׂה אִתָּם וְעִם אֱלֹהֵיהֶם הֶסְכֵּם.
לג לֹ֤א יֵשְׁבוּ֙ בְּאַרְצְךָ֔ פֶּן־יַחֲטִ֥יאוּ אֹתְךָ֖ לִ֑י כִּ֤י תַעֲבֹד֙ אֶת־אֱלֹ֣הֵיהֶ֔ם כִּֽי־יִהְיֶ֥ה לְךָ֖ לְמוֹקֵֽשׁ׃ הֵם לֹא יֵשְׁבוּ בְּאַרְצְךָ שֶׁלֹּא יַחֲטִיאוּ אוֹתְךָ נֶגְדִּי בְּכָךְ שֶׁתַּעֲבוֹד אֶת אֱלֹהֵיהֶם וְזֶה יִהְיֶה לְךָ לְמִכְשׁוֹל.
פ פָּרָשַׁת עֲבוֹדַת הָאֱלֹהִים
כד א וְאֶל־מֹשֶׁ֨ה אָמַ֜ר עֲלֵ֣ה אֶל־יְהֹוָ֗ה אַתָּה֙ וְאַהֲרֹן֙ נָדָ֣ב וַאֲבִיה֔וּא וְשִׁבְעִ֖ים מִזִּקְנֵ֣י יִשְׂרָאֵ֑ל וְהִשְׁתַּחֲוִיתֶ֖ם מֵרָחֹֽק׃ וּלְמֹשֶׁה ה' אָמַר: עֲלֵה אֶל ה' אַתָּה וְאַהֲרֹן נָדָב וַאֲבִיהוּא וְשִׁבְעִים מִזִּקְנֵי יִשְׂרָאֵל וְתִשְׁתַּחֲווּ מֵרָחוֹק.
ב וְנִגַּ֨שׁ מֹשֶׁ֤ה לְבַדּוֹ֙ אֶל־יְהֹוָ֔ה וְהֵ֖ם לֹ֣א יִגָּ֑שׁוּ וְהָעָ֕ם לֹ֥א יַעֲל֖וּ עִמּֽוֹ׃ וּמֹשֶׁה לְבַדּוֹ יִגַּשׁ אֶל ה' וְהֵם לֹא יִגְּשׁוּ, וְהָעָם לֹא יַעֲלוּ אִתּוֹ.
ג וַיָּבֹ֣א מֹשֶׁ֗ה וַיְסַפֵּ֤ר לָעָם֙ אֵ֚ת כׇּל־דִּבְרֵ֣י יְהֹוָ֔ה וְאֵ֖ת כׇּל־הַמִּשְׁפָּטִ֑ים וַיַּ֨עַן כׇּל־הָעָ֜ם ק֤וֹל אֶחָד֙ וַיֹּ֣אמְר֔וּ כׇּל־הַדְּבָרִ֛ים אֲשֶׁר־דִּבֶּ֥ר יְהֹוָ֖ה נַעֲשֶֽׂה׃ מֹשֶׁה בָּא וְסִפֵּר לָעָם אֶת כָּל דִּבְרֵי ה' וְאֶת כָּל הַמִּשְׁפָּטִים, כָּל הָעָם עָנוּ יַחַד וְאָמְרוּ: כָּל הַדְּבָרִים שֶׁה' דִּבֵּר – נַעֲשֶׂה.
ד וַיִּכְתֹּ֣ב מֹשֶׁ֗ה אֵ֚ת כׇּל־דִּבְרֵ֣י יְהֹוָ֔ה וַיַּשְׁכֵּ֣ם בַּבֹּ֔קֶר וַיִּ֥בֶן מִזְבֵּ֖חַ תַּ֣חַת הָהָ֑ר וּשְׁתֵּ֤ים עֶשְׂרֵה֙ מַצֵּבָ֔ה לִשְׁנֵ֥ים עָשָׂ֖ר שִׁבְטֵ֥י יִשְׂרָאֵֽל׃ מֹשֶׁה כָּתַב אֶת כָּל דִּבְרֵי ה', וְהִשְׁכִּים בַּבּוֹקֶר וּבָנָה לְמַרְגְּלוֹת הָהָר מִזְבֵּחַ וּשְׁתֵּים עֶשְׂרֵה מַצֵּבוֹת לִשְׁנֵים עָשָׂר שִׁבְטֵי יִשְׂרָאֵל.
ה וַיִּשְׁלַ֗ח אֶֽת־נַעֲרֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַיַּֽעֲל֖וּ עֹלֹ֑ת וַֽיִּזְבְּח֞וּ זְבָחִ֧ים שְׁלָמִ֛ים לַיהֹוָ֖ה פָּרִֽים׃ הוּא שָׁלַח אֶת נַעֲרֵי בְּנֵי יִשְׂרָאֵל הַבְּכוֹרִים וְהֵם הֶעֱלוּ עוֹלוֹת, וְשָׁחֲטוּ פָּרִים זִבְחֵי שְׁלָמִים לַה'.
ו וַיִּקַּ֤ח מֹשֶׁה֙ חֲצִ֣י הַדָּ֔ם וַיָּ֖שֶׂם בָּאַגָּנֹ֑ת וַחֲצִ֣י הַדָּ֔ם זָרַ֖ק עַל־הַמִּזְבֵּֽחַ׃ מֹשֶׁה לָקַח חֵצִי מֵהַדָּם וְשָׂם בַּכּוֹסוֹת, וְחֵצִי מֵהַדָּם זָרַק עַל הַמִּזְבֵּחַ.
ז וַיִּקַּח֙ סֵ֣פֶר הַבְּרִ֔ית וַיִּקְרָ֖א בְּאׇזְנֵ֣י הָעָ֑ם וַיֹּ֣אמְר֔וּ כֹּ֛ל אֲשֶׁר־דִּבֶּ֥ר יְהֹוָ֖ה נַעֲשֶׂ֥ה וְנִשְׁמָֽע׃ הוּא לָקַח אֶת כְּתַב הַהֶסְכֵּם שֶׁכָּתַב בּוֹ אֶת דִּבְרֵי ה', וְקָרָא בְּאָזְנֵי הָעָם, וְהֵם אָמְרוּ: כָּל מַה שֶּׁה' דִּבֵּר נַעֲשֶׂה וְנִשְׁמַע!
ח וַיִּקַּ֤ח מֹשֶׁה֙ אֶת־הַדָּ֔ם וַיִּזְרֹ֖ק עַל־הָעָ֑ם וַיֹּ֗אמֶר הִנֵּ֤ה דַֽם־הַבְּרִית֙ אֲשֶׁ֨ר כָּרַ֤ת יְהֹוָה֙ עִמָּכֶ֔ם עַ֥ל כׇּל־הַדְּבָרִ֖ים הָאֵֽלֶּה׃ מֹשֶׁה לָקַח אֶת הַדָּם וְזָרַק עַל הָעָם וְאָמַר: הִנֵּה דַּם כְּרִיתַת הַבְּרִית וְהַהֶסְכֵּם שֶׁה' עוֹשֶׂה אִתְּכֶם עַל כָּל הַדְּבָרִים הָאֵלֶּה.
ט וַיַּ֥עַל מֹשֶׁ֖ה וְאַהֲרֹ֑ן נָדָב֙ וַאֲבִיה֔וּא וְשִׁבְעִ֖ים מִזִּקְנֵ֥י יִשְׂרָאֵֽל׃ מֹשֶׁה וְאַהֲרֹן נָדָב וַאֲבִיהוּא וְשִׁבְעִים מִזִּקְנֵי יִשְׂרָאֵל עָלוּ.
י וַיִּרְא֕וּ אֵ֖ת אֱלֹהֵ֣י יִשְׂרָאֵ֑ל וְתַ֣חַת רַגְלָ֗יו כְּמַעֲשֵׂה֙ לִבְנַ֣ת הַסַּפִּ֔יר וּכְעֶ֥צֶם הַשָּׁמַ֖יִם לָטֹֽהַר׃ וְהֵם רָאוּ אֶת כְּבוֹד אֱלֹהֵי יִשְׂרָאֵל, וּמִתַּחַת לְרַגְלָיו כְּמוֹ לְבֵנָה עֲשׂוּיָה מֵאֶבֶן סַפִּיר, וְהוּא צָלוּל כְּמַרְאֶה הַשָּׁמַיִם.
יא וְאֶל־אֲצִילֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לֹ֥א שָׁלַ֖ח יָד֑וֹ וַֽיֶּחֱזוּ֙ אֶת־הָ֣אֱלֹהִ֔ים וַיֹּאכְל֖וּ וַיִּשְׁתּֽוּ׃ וְאֶל אוֹתָם הַחֲשׁוּבִים מִבְּנֵי יִשְׂרָאֵל הוּא לֹא הוֹשִׂיט אֶת יָדוֹ לִפְגּוֹעַ בָּהֶם לַמְרוֹת שֶׁרָאוּ אֶת כְּבוֹד הָאֱלֹהִים וְאָכְלוּ וְשָׁתוּ.
ס פָּרָשַׁת הָעֲלִיָּה לָהָר
יב וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה עֲלֵ֥ה אֵלַ֛י הָהָ֖רָה וֶהְיֵה־שָׁ֑ם וְאֶתְּנָ֨ה לְךָ֜ אֶת־לֻחֹ֣ת הָאֶ֗בֶן וְהַתּוֹרָה֙ וְהַמִּצְוָ֔ה אֲשֶׁ֥ר כָּתַ֖בְתִּי לְהוֹרֹתָֽם׃ ה' אָמַר לְמֹשֶׁה: עֲלֵה אֵלַי לָהָר וְתִהְיֶה שָׁם, וְאֶתֵּן לְךָ אֶת לוּחוֹת הָאֶבֶן וְהַתּוֹרָה וְהַמִּצְוָה שֶׁכָּתַבְתִּי לְצַוּוֹת אוֹתָם.
יג וַיָּ֣קׇם מֹשֶׁ֔ה וִיהוֹשֻׁ֖עַ מְשָׁרְת֑וֹ וַיַּ֥עַל מֹשֶׁ֖ה אֶל־הַ֥ר הָאֱלֹהִֽים׃ מֹשֶׁה וִיהוֹשֻׁעַ מְשָׁרְתוֹ קָמוּ, וּמֹשֶׁה עָלָה לְהַר הָאֱלֹהִים.
יד וְאֶל־הַזְּקֵנִ֤ים אָמַר֙ שְׁבוּ־לָ֣נוּ בָזֶ֔ה עַ֥ד אֲשֶׁר־נָשׁ֖וּב אֲלֵיכֶ֑ם וְהִנֵּ֨ה אַהֲרֹ֤ן וְחוּר֙ עִמָּכֶ֔ם מִי־בַ֥עַל דְּבָרִ֖ים יִגַּ֥שׁ אֲלֵהֶֽם׃ וְלַזְּקֵנִים אָמַר: שְׁבוּ וְחַכּוּ לָנוּ כָּאן עַד שֶׁנַּחֲזוֹר אֲלֵיכֶם, וְהִנֵּה אַהֲרֹן וְחוּר אִתְּכֶם, מִי שֶׁיֵּשׁ לוֹ תְּבִיעָה יִגַּשׁ אֲלֵיהֶם.
טו וַיַּ֥עַל מֹשֶׁ֖ה אֶל־הָהָ֑ר וַיְכַ֥ס הֶעָנָ֖ן אֶת־הָהָֽר׃ מֹשֶׁה עָלָה לָהָר וְהֶעָנָן כִּסָּה אֶת הָהָר.

מפטיר[עריכה]

טז וַיִּשְׁכֹּ֤ן כְּבוֹד־יְהֹוָה֙ עַל־הַ֣ר סִינַ֔י וַיְכַסֵּ֥הוּ הֶעָנָ֖ן שֵׁ֣שֶׁת יָמִ֑ים וַיִּקְרָ֧א אֶל־מֹשֶׁ֛ה בַּיּ֥וֹם הַשְּׁבִיעִ֖י מִתּ֥וֹךְ הֶעָנָֽן׃ כְּבוֹד ה' שָׁכַן עַל הַר סִינַי – הֶעָנָן כִּסָּה אוֹתוֹ שִׁשָּׁה יָמִים, וַה' קָרָא לְמֹשֶׁה בַּיּוֹם הַשְּׁבִיעִי מִתּוֹךְ הֶעָנָן.
יז וּמַרְאֵה֙ כְּב֣וֹד יְהֹוָ֔ה כְּאֵ֥שׁ אֹכֶ֖לֶת בְּרֹ֣אשׁ הָהָ֑ר לְעֵינֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל׃ וּמַרְאֵה כְּבוֹד ה' הָיָה כְּמוֹ אֵשׁ שׂוֹרֶפֶת בְּרֹאשׁ הָהָר לְעֵינֵי בְּנֵי יִשְׂרָאֵל.
יח וַיָּבֹ֥א מֹשֶׁ֛ה בְּת֥וֹךְ הֶעָנָ֖ן וַיַּ֣עַל אֶל־הָהָ֑ר וַיְהִ֤י מֹשֶׁה֙ בָּהָ֔ר אַרְבָּעִ֣ים י֔וֹם וְאַרְבָּעִ֖ים לָֽיְלָה׃ מֹשֶׁה נִכְנַס בְּתוֹךְ הֶעָנָן וְעָלָה לָהָר, מֹשֶׁה הָיָה בָּהָר אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה.
