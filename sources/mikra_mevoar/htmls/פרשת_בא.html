ראשון[עריכה]

פ פָּרָשַׁת מַכַּת הָאַרְבֶּה
י א וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה בֹּ֖א אֶל־פַּרְעֹ֑ה כִּֽי־אֲנִ֞י הִכְבַּ֤דְתִּי אֶת־לִבּוֹ֙ וְאֶת־לֵ֣ב עֲבָדָ֔יו לְמַ֗עַן שִׁתִ֛י אֹתֹתַ֥י אֵ֖לֶּה בְּקִרְבּֽוֹ׃ ה' אָמַר לְמֹשֶׁה: בֹּא אֶל פַּרְעֹה כִּי אֲנִי הִקְשַׁחְתִּי אֶת לִבּוֹ וְאֶת לֵב עֲבָדָיו כְּדֵי שֶׁאֶעֱשֶׂה בָּהֶם אֶת הַסִּימָנִים הָאֵלֶּה לְכוֹחִי.
ב וּלְמַ֡עַן תְּסַפֵּר֩ בְּאׇזְנֵ֨י בִנְךָ֜ וּבֶן־בִּנְךָ֗ אֵ֣ת אֲשֶׁ֤ר הִתְעַלַּ֙לְתִּי֙ בְּמִצְרַ֔יִם וְאֶת־אֹתֹתַ֖י אֲשֶׁר־שַׂ֣מְתִּי בָ֑ם וִֽידַעְתֶּ֖ם כִּי־אֲנִ֥י יְהֹוָֽה׃ וּכְדֵי שֶׁתְּסַפֵּר לְבִנְךָ וְנֶכְדְךָ אֵיךְ הִתְעַלַּלְתִּי בְּמִצְרַיִם וְעַל הַסִּימָנִים לְכוֹחִי שֶׁהֶרְאֵתִי בְּכָךְ שֶׁהִכֵּיתִי אוֹתָם, וְתֵדְעוּ שֶׁאֲנִי ה'.
ג וַיָּבֹ֨א מֹשֶׁ֣ה וְאַהֲרֹן֮ אֶל־פַּרְעֹה֒ וַיֹּאמְר֣וּ אֵלָ֗יו כֹּֽה־אָמַ֤ר יְהֹוָה֙ אֱלֹהֵ֣י הָֽעִבְרִ֔ים עַד־מָתַ֣י מֵאַ֔נְתָּ לֵעָנֹ֖ת מִפָּנָ֑י שַׁלַּ֥ח עַמִּ֖י וְיַֽעַבְדֻֽנִי׃ מֹשֶׁה וְאַהֲרֹן בָּאוּ לְפַרְעֹה וְאָמְרוּ לוֹ: כָּךְ אָמַר ה' אֱלֹהֵי הָעִבְרִים: עַד מָתַי אַתָּה מְסָרֵב לְהִכָּנַע בְּפָנַי? שַׁחְרֵר אֶת עַמִּי וְיַעַבְדוּ לְפָנַי.
ד כִּ֛י אִם־מָאֵ֥ן אַתָּ֖ה לְשַׁלֵּ֣חַ אֶת־עַמִּ֑י הִנְנִ֨י מֵבִ֥יא מָחָ֛ר אַרְבֶּ֖ה בִּגְבֻלֶֽךָ׃ כִּי אִם אַתָּה מְסָרֵב לְשַׁחְרֵר אֶת עַמִּי הִנֵּה אֲנִי מֵבִיא מָחָר חֲגָבִים בְּאַרְצְךָ.
ה וְכִסָּה֙ אֶת־עֵ֣ין הָאָ֔רֶץ וְלֹ֥א יוּכַ֖ל לִרְאֹ֣ת אֶת־הָאָ֑רֶץ וְאָכַ֣ל ׀ אֶת־יֶ֣תֶר הַפְּלֵטָ֗ה הַנִּשְׁאֶ֤רֶת לָכֶם֙ מִן־הַבָּרָ֔ד וְאָכַל֙ אֶת־כׇּל־הָעֵ֔ץ הַצֹּמֵ֥חַ לָכֶ֖ם מִן־הַשָּׂדֶֽה׃ הֵם יְכַסּוּ אֶת מַרְאֶה הָאָרֶץ עַד שֶׁלֹּא יוּכַל אָדָם לִרְאוֹת אֶת הָאָרֶץ, וְהֵם יֹאכְלוּ אֶת מַה שֶּׁנִּשְׁאַר לָכֶם מֵהַבָּרָד, וְיֹאכְלוּ אֶת כָּל הָעֵצִים שֶׁצּוֹמְחִים לָכֶם בַּשָּׂדֶה.
ו וּמָלְא֨וּ בָתֶּ֜יךָ וּבָתֵּ֣י כׇל־עֲבָדֶ֘יךָ֮ וּבָתֵּ֣י כׇל־מִצְרַ֒יִם֒ אֲשֶׁ֨ר לֹֽא־רָא֤וּ אֲבֹתֶ֙יךָ֙ וַאֲב֣וֹת אֲבֹתֶ֔יךָ מִיּ֗וֹם הֱיוֹתָם֙ עַל־הָ֣אֲדָמָ֔ה עַ֖ד הַיּ֣וֹם הַזֶּ֑ה וַיִּ֥פֶן וַיֵּצֵ֖א מֵעִ֥ם פַּרְעֹֽה׃ וְיִתְמַלְּאוּ בָּתֶּיךָ וּבָתֵּי כָּל עֲבָדֶיךָ וּבָתֵּי כָּל מִצְרַיִם בְּאוֹפֶן שֶׁלֹּא רָאוּ אֲבוֹתֶיךָ וַאֲבוֹת אֲבוֹתֶיךָ מִיּוֹם שֶׁהָיוּ עַל הָאֲדָמָה עַד הַיּוֹם הַזֶּה, מֹשֶׁה פָּנָה וְיָצָא מֵעִם פַּרְעֹה.
ז וַיֹּאמְרוּ֩ עַבְדֵ֨י פַרְעֹ֜ה אֵלָ֗יו עַד־מָתַי֙ יִהְיֶ֨ה זֶ֥ה לָ֙נוּ֙ לְמוֹקֵ֔שׁ שַׁלַּח֙ אֶת־הָ֣אֲנָשִׁ֔ים וְיַֽעַבְד֖וּ אֶת־יְהֹוָ֣ה אֱלֹהֵיהֶ֑ם הֲטֶ֣רֶם תֵּדַ֔ע כִּ֥י אָבְדָ֖ה מִצְרָֽיִם׃ עַבְדֵי פַּרְעֹה אָמְרוּ אֵלָיו: עַד מָתַי הָאִישׁ הַזֶּה יִהְיֶה לָנוּ לְמִכְשׁוֹל?! שַׁחְרֵר אֶת הָאֲנָשִׁים וְיַעַבְדוּ לִפְנֵי ה' אֱלֹהֵיהֶם! וְכִי עֲדַיִן אֵינְךָ יוֹדֵעַ שֶׁמִּצְרַיִם אָבְדָה?!
ח וַיּוּשַׁ֞ב אֶת־מֹשֶׁ֤ה וְאֶֽת־אַהֲרֹן֙ אֶל־פַּרְעֹ֔ה וַיֹּ֣אמֶר אֲלֵהֶ֔ם לְכ֥וּ עִבְד֖וּ אֶת־יְהֹוָ֣ה אֱלֹהֵיכֶ֑ם מִ֥י וָמִ֖י הַהֹלְכִֽים׃ הֶחֱזִירוּ אֶת מֹשֶׁה וְאֶת אַהֲרֹן אֶל פַּרְעֹה, וְהוּא אָמַר לָהֶם: לְכוּ עִבְדוּ אֶת ה' אֱלֹהֵיכֶם, מִי וָמִי הַהוֹלְכִים?
ט וַיֹּ֣אמֶר מֹשֶׁ֔ה בִּנְעָרֵ֥ינוּ וּבִזְקֵנֵ֖ינוּ נֵלֵ֑ךְ בְּבָנֵ֨ינוּ וּבִבְנוֹתֵ֜נוּ בְּצֹאנֵ֤נוּ וּבִבְקָרֵ֙נוּ֙ נֵלֵ֔ךְ כִּ֥י חַג־יְהֹוָ֖ה לָֽנוּ׃ אָמַר מֹשֶׁה: נֵלֵךְ עִם נְעָרֵינוּ וּזְקֵנֵינוּ, נֵלֵךְ עִם בָּנֵינוּ וּבְנוֹתֵנוּ, עִם צֹאנֵנוּ וּבְקָרֵנוּ, כִּי יֵשׁ לָנוּ חַג לַה'.
י וַיֹּ֣אמֶר אֲלֵהֶ֗ם יְהִ֨י כֵ֤ן יְהֹוָה֙ עִמָּכֶ֔ם כַּאֲשֶׁ֛ר אֲשַׁלַּ֥ח אֶתְכֶ֖ם וְאֶֽת־טַפְּכֶ֑ם רְא֕וּ כִּ֥י רָעָ֖ה נֶ֥גֶד פְּנֵיכֶֽם׃ אָמַר לָהֶם: לוּ יִהְיֶה כָּךְ וַה' יַעֲזוֹר לָכֶם כְּשֶׁאֲשַׁחְרֵר אֶתְכֶם וְאֶת יַלְדֵיכֶם, בָּרוּר שֶׁאַתֶּם זוֹמְמִים רָעָה – לִבְרוֹחַ!
יא לֹ֣א כֵ֗ן לְכֽוּ־נָ֤א הַגְּבָרִים֙ וְעִבְד֣וּ אֶת־יְהֹוָ֔ה כִּ֥י אֹתָ֖הּ אַתֶּ֣ם מְבַקְשִׁ֑ים וַיְגָ֣רֶשׁ אֹתָ֔ם מֵאֵ֖ת פְּנֵ֥י פַרְעֹֽה׃ לֹא כָּךְ, לְכוּ נָא רַק הַגְּבָרִים וְעִבְדוּ אֶת ה', הֲרֵי רַק אֶת הָעֲבוֹדָה אַתֶּם רוֹצִים לַעֲשׂוֹת! פַּרְעֹה גֵּרֵשׁ אוֹתָם מִלְּפָנָיו.

שני[עריכה]

ס
יב וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה נְטֵ֨ה יָדְךָ֜ עַל־אֶ֤רֶץ מִצְרַ֙יִם֙ בָּֽאַרְבֶּ֔ה וְיַ֖עַל עַל־אֶ֣רֶץ מִצְרָ֑יִם וְיֹאכַל֙ אֶת־כׇּל־עֵ֣שֶׂב הָאָ֔רֶץ אֵ֛ת כׇּל־אֲשֶׁ֥ר הִשְׁאִ֖יר הַבָּרָֽד׃ ה' אָמַר לְמֹשֶׁה: הָנֵף אֶת יָדְךָ עַל אֶרֶץ מִצְרַיִם לְהָבִיא חֲגָבִים וְיַעֲלוּ עַל אֶרֶץ מִצְרַיִם וְיֹאכְלוּ אֶת כָּל הָעֵשֶׂב בָּאָרֶץ – אֶת כָּל מַה שֶּׁהַבָּרָד הִשְׁאִיר.
יג וַיֵּ֨ט מֹשֶׁ֣ה אֶת־מַטֵּ֘הוּ֮ עַל־אֶ֣רֶץ מִצְרַ֒יִם֒ וַֽיהֹוָ֗ה נִהַ֤ג רֽוּחַ־קָדִים֙ בָּאָ֔רֶץ כׇּל־הַיּ֥וֹם הַה֖וּא וְכׇל־הַלָּ֑יְלָה הַבֹּ֣קֶר הָיָ֔ה וְר֙וּחַ֙ הַקָּדִ֔ים נָשָׂ֖א אֶת־הָאַרְבֶּֽה׃ מֹשֶׁה הֵנִיף אֶת מַקְלוֹ עַל אֶרֶץ מִצְרַיִם וַה' הִנְהִיג רוּחַ מִזְרָחִית בָּאָרֶץ כָּל הַיּוֹם הַהוּא וְכָל הַלַּיְלָה, הַבּוֹקֶר הִגִּיעַ וְהָרוּחַ הַמִּזְרָחִית הֵרִימָה אֶת הַחֲגָבִים.
יד וַיַּ֣עַל הָֽאַרְבֶּ֗ה עַ֚ל כׇּל־אֶ֣רֶץ מִצְרַ֔יִם וַיָּ֕נַח בְּכֹ֖ל גְּב֣וּל מִצְרָ֑יִם כָּבֵ֣ד מְאֹ֔ד לְ֠פָנָ֠יו לֹא־הָ֨יָה כֵ֤ן אַרְבֶּה֙ כָּמֹ֔הוּ וְאַחֲרָ֖יו לֹ֥א יִֽהְיֶה־כֵּֽן׃ הָחֲגָבִים עָלוּ עַל כָּל אֶרֶץ מִצְרַיִם וְנָחוּ בְּכָל שֶׁטַח מִצְרַיִם, לַהֲקָה גְּדוֹלָה מְאוֹד, לְפָנֶיהָ לֹא הָיָה כָּךְ – כַּמּוּת חֲגָבִים כָּזוֹ, וְאַחֲרֶיהָ לֹא יִהְיֶה כָּךְ.
טו וַיְכַ֞ס אֶת־עֵ֣ין כׇּל־הָאָ֘רֶץ֮ וַתֶּחְשַׁ֣ךְ הָאָ֒רֶץ֒ וַיֹּ֜אכַל אֶת־כׇּל־עֵ֣שֶׂב הָאָ֗רֶץ וְאֵת֙ כׇּל־פְּרִ֣י הָעֵ֔ץ אֲשֶׁ֥ר הוֹתִ֖יר הַבָּרָ֑ד וְלֹא־נוֹתַ֨ר כׇּל־יֶ֧רֶק בָּעֵ֛ץ וּבְעֵ֥שֶׂב הַשָּׂדֶ֖ה בְּכׇל־אֶ֥רֶץ מִצְרָֽיִם׃ הֵם כִּסּוּ אֶת מַרְאֶה כָּל הָאָרֶץ וְנַעֲשָׂה חוֹשֶׁךְ בָּאָרֶץ, הֵם אָכְלוּ אֶת כָּל הָעֵשֶׂב וְאֶת כָּל הַפֵּרוֹת שֶׁהִשְׁאִיר הַבָּרָד, לֹא נִשְׁאַר שׁוּם דָּבָר יָרוֹק בָּעֵצִים וּבְעִשְׂבֵי הַשָּׂדֶה בְּכָל אֶרֶץ מִצְרַיִם.
טז וַיְמַהֵ֣ר פַּרְעֹ֔ה לִקְרֹ֖א לְמֹשֶׁ֣ה וּֽלְאַהֲרֹ֑ן וַיֹּ֗אמֶר חָטָ֛אתִי לַיהֹוָ֥ה אֱלֹֽהֵיכֶ֖ם וְלָכֶֽם׃ פַּרְעֹה מִהֵר לִקְרֹא לְמֹשֶׁה וּלְאַהֲרֹן, וְאָמַר: חָטָאתִי לַה' אֱלֹהֵיכֶם וְלָכֶם.
יז וְעַתָּ֗ה שָׂ֣א נָ֤א חַטָּאתִי֙ אַ֣ךְ הַפַּ֔עַם וְהַעְתִּ֖ירוּ לַיהֹוָ֣ה אֱלֹהֵיכֶ֑ם וְיָסֵר֙ מֵֽעָלַ֔י רַ֖ק אֶת־הַמָּ֥וֶת הַזֶּֽה׃ וְעַכְשָׁיו סְלַח בְּבַקָּשָׁה עַל חַטָּאתִי, רַק הַפַּעַם, וְהִתְפַּלְּלוּ לַה' אֱלֹהֵיכֶם, שֶׁיָּסִיר מֵעָלַי רַק אֶת הַמָּוֶת הַזֶּה.
יח וַיֵּצֵ֖א מֵעִ֣ם פַּרְעֹ֑ה וַיֶּעְתַּ֖ר אֶל־יְהֹוָֽה׃ מֹשֶׁה יָצָא מִלִּפְנֵי פַּרְעֹה וְהִתְפַּלֵּל אֶל ה'.
יט וַיַּהֲפֹ֨ךְ יְהֹוָ֤ה רֽוּחַ־יָם֙ חָזָ֣ק מְאֹ֔ד וַיִּשָּׂא֙ אֶת־הָ֣אַרְבֶּ֔ה וַיִּתְקָעֵ֖הוּ יָ֣מָּה סּ֑וּף לֹ֤א נִשְׁאַר֙ אַרְבֶּ֣ה אֶחָ֔ד בְּכֹ֖ל גְּב֥וּל מִצְרָֽיִם׃ ה' הָפַךְ אֶת כִּוּוּן הָרוּחַ לְרוּחַ מַעֲרָבִית חֲזָקָה מְאוֹד, הִיא לָקְחָה אֶת הָחֲגָבִים וְתָקְעָה אוֹתָם בְּיַם סוּף, וְלֹא נִשְׁאַר חָגָב אֶחָד בְּכָל שֶׁטַח מִצְרַיִם.
כ וַיְחַזֵּ֥ק יְהֹוָ֖ה אֶת־לֵ֣ב פַּרְעֹ֑ה וְלֹ֥א שִׁלַּ֖ח אֶת־בְּנֵ֥י יִשְׂרָאֵֽל׃ ה' הִקְשִׁיחַ אֶת לֵב פַּרְעֹה וְהוּא לֹא שִׁחְרֵר אֶת בְּנֵי יִשְׂרָאֵל.
פ פָּרָשַׁת מַכַּת הַחוֹשֶׁךְ
כא וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה נְטֵ֤ה יָֽדְךָ֙ עַל־הַשָּׁמַ֔יִם וִ֥יהִי חֹ֖שֶׁךְ עַל־אֶ֣רֶץ מִצְרָ֑יִם וְיָמֵ֖שׁ חֹֽשֶׁךְ׃ ה' אָמַר לְמֹשֶׁה: הָנֵף אֶת יָדְךָ אֶל הַשָּׁמַיִם וְיִהְיֶה חוֹשֶׁךְ עַל אֶרֶץ מִצְרַיִם, וְהַחוֹשֶׁךְ שֶׁל הַלַּיְלָה יִמָּשֵׁךְ אֶל הַיּוֹם.
כב וַיֵּ֥ט מֹשֶׁ֛ה אֶת־יָד֖וֹ עַל־הַשָּׁמָ֑יִם וַיְהִ֧י חֹֽשֶׁךְ־אֲפֵלָ֛ה בְּכׇל־אֶ֥רֶץ מִצְרַ֖יִם שְׁלֹ֥שֶׁת יָמִֽים׃ מֹשֶׁה הֵנִיף אֶת יָדוֹ אֶל הַשָּׁמַיִם, וְהָיָה חוֹשֶׁךְ אֲפֵלָה בְּכָל אֶרֶץ מִצְרַיִם שְׁלוֹשָׁה יָמִים.
כג לֹֽא־רָא֞וּ אִ֣ישׁ אֶת־אָחִ֗יו וְלֹא־קָ֛מוּ אִ֥ישׁ מִתַּחְתָּ֖יו שְׁלֹ֣שֶׁת יָמִ֑ים וּֽלְכׇל־בְּנֵ֧י יִשְׂרָאֵ֛ל הָ֥יָה א֖וֹר בְּמוֹשְׁבֹתָֽם׃ לֹא רָאוּ אֶחָד אֶת הַשֵּׁנִי וְאַף אֶחָד לֹא זָז מִמְּקוֹמוֹ שְׁלוֹשָׁה יָמִים, וּלְכָל בְּנֵי יִשְׂרָאֵל הָיָה אוֹר בִּמְקוֹם מְגוּרֵיהֶם.

שלישי[עריכה]

כד וַיִּקְרָ֨א פַרְעֹ֜ה אֶל־מֹשֶׁ֗ה וַיֹּ֙אמֶר֙ לְכוּ֙ עִבְד֣וּ אֶת־יְהֹוָ֔ה רַ֛ק צֹאנְכֶ֥ם וּבְקַרְכֶ֖ם יֻצָּ֑ג גַּֽם־טַפְּכֶ֖ם יֵלֵ֥ךְ עִמָּכֶֽם׃ פַּרְעֹה קָרָא לְמֹשֶׁה וְאָמַר: לְכוּ עִבְדוּ אֶת ה', רַק צֹאנְכֶם וּבְקַרְכֶם יִשָּׁאֵר כָּאן, אֲבָל גַּם יַלְדֵיכֶם יֵלְכוּ אִתְּכֶם.
כה וַיֹּ֣אמֶר מֹשֶׁ֔ה גַּם־אַתָּ֛ה תִּתֵּ֥ן בְּיָדֵ֖נוּ זְבָחִ֣ים וְעֹלֹ֑ת וְעָשִׂ֖ינוּ לַיהֹוָ֥ה אֱלֹהֵֽינוּ׃ אָמַר מֹשֶׁה: לְהֵפֶךְ! גַּם אַתָּה תִּתֵּן לָנוּ זְבָחִים וְעוֹלוֹת וְנַקְרִיב לַה' אֱלֹהֵינוּ.
כו וְגַם־מִקְנֵ֜נוּ יֵלֵ֣ךְ עִמָּ֗נוּ לֹ֤א תִשָּׁאֵר֙ פַּרְסָ֔ה כִּ֚י מִמֶּ֣נּוּ נִקַּ֔ח לַעֲבֹ֖ד אֶת־יְהֹוָ֣ה אֱלֹהֵ֑ינוּ וַאֲנַ֣חְנוּ לֹֽא־נֵדַ֗ע מַֽה־נַּעֲבֹד֙ אֶת־יְהֹוָ֔ה עַד־בֹּאֵ֖נוּ שָֽׁמָּה׃ וְגַם בֶּהֱמוֹתֵינוּ יֵלְכוּ אִתָּנוּ, לֹא תִּשָּׁאֵר אֲפִלּוּ פַּרְסָה שֶׁל בְּהֵמָה, כִּי מֵהֶם נִקַּח לַעֲבוֹד אֶת ה' אֱלֹהֵינוּ, כִּי אֲנַחְנוּ לֹא נֵדַע בְּמַה נַּעֲבוֹד אֶת ה' עַד שֶׁנַּגִּיעַ לְשָׁם.
כז וַיְחַזֵּ֥ק יְהֹוָ֖ה אֶת־לֵ֣ב פַּרְעֹ֑ה וְלֹ֥א אָבָ֖ה לְשַׁלְּחָֽם׃ ה' הִקְשִׁיחַ אֶת לֵב פַּרְעֹה וְהוּא לֹא הִסְכִּים לְשַׁחְרֵר אוֹתָם.
כח וַיֹּֽאמֶר־ל֥וֹ פַרְעֹ֖ה לֵ֣ךְ מֵעָלָ֑י הִשָּׁ֣מֶר לְךָ֗ אַל־תֹּ֙סֶף֙ רְא֣וֹת פָּנַ֔י כִּ֗י בְּי֛וֹם רְאֹתְךָ֥ פָנַ֖י תָּמֽוּת׃ פַּרְעֹה אָמַר לוֹ: לֵךְ מִמֶּנִּי! הִזָּהֵר לְךָ וְאַל תָּבוֹא עוֹד לִפְגּוֹשׁ אוֹתִי, כִּי בַּיּוֹם שֶׁתָּבוֹא לִפְגּוֹשׁ אוֹתִי תָּמוּת.
כט וַיֹּ֥אמֶר מֹשֶׁ֖ה כֵּ֣ן דִּבַּ֑רְתָּ לֹא־אֹסִ֥ף ע֖וֹד רְא֥וֹת פָּנֶֽיךָ׃ אָמַר מֹשֶׁה: אֱמֶת דִּבַּרְתָּ, לֹא אוֹסִיף עוֹד לָבוֹא לִפְגּוֹשׁ אוֹתְךָ.
פ פָּרָשַׁת אַזְהָרַת מַכַּת בְּכוֹרוֹת
יא א וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה ע֣וֹד נֶ֤גַע אֶחָד֙ אָבִ֤יא עַל־פַּרְעֹה֙ וְעַל־מִצְרַ֔יִם אַֽחֲרֵי־כֵ֕ן יְשַׁלַּ֥ח אֶתְכֶ֖ם מִזֶּ֑ה כְּשַׁ֨לְּח֔וֹ כָּלָ֕ה גָּרֵ֛שׁ יְגָרֵ֥שׁ אֶתְכֶ֖ם מִזֶּֽה׃ ה' אָמַר לְמֹשֶׁה: עוֹד מַכָּה אַחַת אָבִיא עַל פַּרְעֹה וְעַל מִצְרַיִם אַחֲרֵי כֵן הוּא יְשַׁחְרֵר אֶתְכֶם מִכָּאן, וּכְשֶׁיְּשַׁחְרֵר אֶתְכֶם הוּא יְגָרֵשׁ אֶתְכֶם מִכָּאן סוֹפִית.
ב דַּבֶּר־נָ֖א בְּאׇזְנֵ֣י הָעָ֑ם וְיִשְׁאֲל֞וּ אִ֣ישׁ ׀ מֵאֵ֣ת רֵעֵ֗הוּ וְאִשָּׁה֙ מֵאֵ֣ת רְעוּתָ֔הּ כְּלֵי־כֶ֖סֶף וּכְלֵ֥י זָהָֽב׃ דַּבֵּר בְּבַקָּשָׁה בְּאָזְנֵי הָעָם וִיבַקְּשׁוּ כָּל אֶחָד מֵחֲבֵרוֹ וְכָל אִשָּׁה מֵחֲבֶרְתָּהּ כְּלֵי כֶּסֶף וּכְלֵי זָהָב.
ג וַיִּתֵּ֧ן יְהֹוָ֛ה אֶת־חֵ֥ן הָעָ֖ם בְּעֵינֵ֣י מִצְרָ֑יִם גַּ֣ם ׀ הָאִ֣ישׁ מֹשֶׁ֗ה גָּד֤וֹל מְאֹד֙ בְּאֶ֣רֶץ מִצְרַ֔יִם בְּעֵינֵ֥י עַבְדֵֽי־פַרְעֹ֖ה וּבְעֵינֵ֥י הָעָֽם׃ ה' עָשָׂה שֶׁהָעָם מָצָא חֵן בְּעֵינֵי מִצְרַיִם, גַּם הָאִישׁ מֹשֶׁה חָשׁוּב מְאוֹד בְּאֶרֶץ מִצְרַיִם בְּעֵינֵי עַבְדֵי פַּרְעֹה וּבְעֵינֵי הָעָם.

רביעי[עריכה]

ס
ד וַיֹּ֣אמֶר מֹשֶׁ֔ה כֹּ֖ה אָמַ֣ר יְהֹוָ֑ה כַּחֲצֹ֣ת הַלַּ֔יְלָה אֲנִ֥י יוֹצֵ֖א בְּת֥וֹךְ מִצְרָֽיִם׃ מֹשֶׁה אָמַר: כָּךְ אָמַר ה' כְּשֶׁיַּגִּיעַ חֲצוֹת הַלַּיְלָה אֵצֵא לְתוֹךְ מִצְרַיִם.
ה וּמֵ֣ת כׇּל־בְּכוֹר֮ בְּאֶ֣רֶץ מִצְרַ֒יִם֒ מִבְּכ֤וֹר פַּרְעֹה֙ הַיֹּשֵׁ֣ב עַל־כִּסְא֔וֹ עַ֚ד בְּכ֣וֹר הַשִּׁפְחָ֔ה אֲשֶׁ֖ר אַחַ֣ר הָרֵחָ֑יִם וְכֹ֖ל בְּכ֥וֹר בְּהֵמָֽה׃ וְכָל בְּכוֹר בְּאֶרֶץ מִצְרַיִם יָמוּת, מִבְּכוֹר פַּרְעֹה שֶׁאָמוּר לָרֶשֶׁת אוֹתוֹ וְלָשֶׁבֶת עַל כִּסְאוֹ עַד בְּכוֹר הַשִּׁפְחָה שֶׁעוֹמֵד מֵאֲחוֹרֵי הָרֵחַיִם לִטְחוֹן, וְכָל בְּכוֹר בְּהֵמָה.
ו וְהָ֥יְתָ֛ה צְעָקָ֥ה גְדֹלָ֖ה בְּכׇל־אֶ֣רֶץ מִצְרָ֑יִם אֲשֶׁ֤ר כָּמֹ֙הוּ֙ לֹ֣א נִהְיָ֔תָה וְכָמֹ֖הוּ לֹ֥א תֹסִֽף׃ תִּהְיֶה צְעָקָה גְּדוֹלָה בְּכָל אֶרֶץ מִצְרַיִם שֶׁכְּמוֹתָהּ לֹא הָיְתָה וּכְמוֹתָהּ לֹא תִּהְיֶה עוֹד.
ז וּלְכֹ֣ל ׀ בְּנֵ֣י יִשְׂרָאֵ֗ל לֹ֤א יֶֽחֱרַץ־כֶּ֙לֶב֙ לְשֹׁנ֔וֹ לְמֵאִ֖ישׁ וְעַד־בְּהֵמָ֑ה לְמַ֙עַן֙ תֵּֽדְע֔וּן אֲשֶׁר֙ יַפְלֶ֣ה יְהֹוָ֔ה בֵּ֥ין מִצְרַ֖יִם וּבֵ֥ין יִשְׂרָאֵֽל׃ וְאֵצֶל כָּל בְּנֵי יִשְׂרָאֵל אֲפִלּוּ כֶּלֶב לֹא יִנְבַּח – לֹא אִישׁ יִצְרַח וְלֹא בְּהֵמָה, כְּדֵי שֶׁתֵּדְעוּ שֶׁה' מַבְדִּיל בֵּין מִצְרַיִם וּבֵין יִשְׂרָאֵל.
ח וְיָרְד֣וּ כׇל־עֲבָדֶ֩יךָ֩ אֵ֨לֶּה אֵלַ֜י וְהִשְׁתַּֽחֲווּ־לִ֣י לֵאמֹ֗ר צֵ֤א אַתָּה֙ וְכׇל־הָעָ֣ם אֲשֶׁר־בְּרַגְלֶ֔יךָ וְאַחֲרֵי־כֵ֖ן אֵצֵ֑א וַיֵּצֵ֥א מֵֽעִם־פַּרְעֹ֖ה בׇּחֳרִי־אָֽף׃ וְיֵרְדוּ כָּל עֲבָדֶיךָ אֵלֶּה אֵלַי וְיִשְׁתַּחֲווּ לִי לוֹמַר לִי: צֵא אַתָּה וְכָל הָעָם הַמִּתְלַוִּים אֵלֶיךָ, וְאַחֲרֵי כֵן אֵצֵא! וְהוּא יָצָא מֵאֵצֶל פַּרְעֹה בְּכַעַס.
ס
ט וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה לֹא־יִשְׁמַ֥ע אֲלֵיכֶ֖ם פַּרְעֹ֑ה לְמַ֛עַן רְב֥וֹת מוֹפְתַ֖י בְּאֶ֥רֶץ מִצְרָֽיִם׃ ה' אָמַר לְמֹשֶׁה: פַּרְעֹה לֹא יִשְׁמַע אֲלֵיכֶם כְּדֵי שֶׁאוּכַל לְהַרְבּוֹת אֶת הוֹכָחוֹתַי לְכוֹחִי בְּאֶרֶץ מִצְרַיִם.
י וּמֹשֶׁ֣ה וְאַהֲרֹ֗ן עָשׂ֛וּ אֶת־כׇּל־הַמֹּפְתִ֥ים הָאֵ֖לֶּה לִפְנֵ֣י פַרְעֹ֑ה וַיְחַזֵּ֤ק יְהֹוָה֙ אֶת־לֵ֣ב פַּרְעֹ֔ה וְלֹֽא־שִׁלַּ֥ח אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל מֵאַרְצֽוֹ׃ וּמֹשֶׁה וְאַהֲרֹן עָשׂוּ אֶת כָּל הַנִּסִּים הָאֵלֶּה לִפְנֵי פַּרְעֹה, וַה' הִקְשִׁיחַ אֶת לֵב פַּרְעֹה וְהוּא לֹא שִׁחְרֵר אֶת בְּנֵי יִשְׂרָאֵל מֵאַרְצוֹ.
ס פָּרָשַׁת הַחוֹדֶשׁ
יב א וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֣ה וְאֶֽל־אַהֲרֹ֔ן בְּאֶ֥רֶץ מִצְרַ֖יִם לֵאמֹֽר׃ ה' אָמַר לְמֹשֶׁה וּלְאַהֲרֹן בְּאֶרֶץ מִצְרַיִם:
ב הַחֹ֧דֶשׁ הַזֶּ֛ה לָכֶ֖ם רֹ֣אשׁ חֳדָשִׁ֑ים רִאשׁ֥וֹן הוּא֙ לָכֶ֔ם לְחׇדְשֵׁ֖י הַשָּׁנָֽה׃ הַחוֹדֶשׁ הַזֶּה יִהְיֶה לָכֶם רֹאשׁ הַחֳדָשִׁים – הוּא הַחוֹדֶשׁ הָרִאשׁוֹן שֶׁלָּכֶם לְחָדְשֵׁי הַשָּׁנָה.
ג דַּבְּר֗וּ אֶֽל־כׇּל־עֲדַ֤ת יִשְׂרָאֵל֙ לֵאמֹ֔ר בֶּעָשֹׂ֖ר לַחֹ֣דֶשׁ הַזֶּ֑ה וְיִקְח֣וּ לָהֶ֗ם אִ֛ישׁ שֶׂ֥ה לְבֵית־אָבֹ֖ת שֶׂ֥ה לַבָּֽיִת׃ דַּבְּרוּ אֶל כָּל הַצִּבּוּר שֶׁל יִשְׂרָאֵל וְאִמְרוּ: בָּעֲשִׂירִי לַחוֹדֶשׁ הַזֶּה יִקְחוּ לָהֶם, כָּל אִישׁ, גְּדִי אוֹ טָלֶה לְכָל בֵּית אָב – שֶׂה לְכָל מִשְׁפָּחָה.
ד וְאִם־יִמְעַ֣ט הַבַּ֘יִת֮ מִהְי֣וֹת מִשֶּׂה֒ וְלָקַ֣ח ה֗וּא וּשְׁכֵנ֛וֹ הַקָּרֹ֥ב אֶל־בֵּית֖וֹ בְּמִכְסַ֣ת נְפָשֹׁ֑ת אִ֚ישׁ לְפִ֣י אׇכְל֔וֹ תָּכֹ֖סּוּ עַל־הַשֶּֽׂה׃ וְאִם הַמִּשְׁפָּחָה תִּהְיֶה קְטַנָּה וְלֹא תּוּכַל לֶאֱכוֹל לְבַדָּהּ שֶׂה שָׁלֵם, יִקְחוּ הֵם וּשְׁכֵנֵיהֶם הַקְּרוֹבִים לְבֵיתָם לְפִי מִסְפַּר הָאֲנָשִׁים וּלְפִי יְכוֹלֶת הָאֲכִילָה שֶׁל כָּל אֶחָד, וְיִהְיוּ מְנוּיִים עַל הַשֶּׂה.
ה שֶׂ֥ה תָמִ֛ים זָכָ֥ר בֶּן־שָׁנָ֖ה יִהְיֶ֣ה לָכֶ֑ם מִן־הַכְּבָשִׂ֥ים וּמִן־הָעִזִּ֖ים תִּקָּֽחוּ׃ יִהְיֶה לָכֶם שֶׂה שָׁלֵם בְּלִי מוּם, זָכָר שֶׁגִּילוֹ עַד שָׁנָה, תִּקְּחוּ אוֹ מֵהַכְּבָשִׂים אוֹ מֵהָעִזִּים.
ו וְהָיָ֤ה לָכֶם֙ לְמִשְׁמֶ֔רֶת עַ֣ד אַרְבָּעָ֥ה עָשָׂ֛ר י֖וֹם לַחֹ֣דֶשׁ הַזֶּ֑ה וְשָׁחֲט֣וּ אֹת֗וֹ כֹּ֛ל קְהַ֥ל עֲדַֽת־יִשְׂרָאֵ֖ל בֵּ֥ין הָעַרְבָּֽיִם׃ וְתִשְׁמְרוּ אוֹתוֹ עַד אַרְבָּעָה עָשָׂר יוֹם לַחוֹדֶשׁ הַזֶּה, וְיִשְׁחֲטוּ אוֹתוֹ כָּל קְהַל הַצִּבּוּר שֶׁל יִשְׂרָאֵל אַחַר הַצָּהֳרַיִם.
ז וְלָֽקְחוּ֙ מִן־הַדָּ֔ם וְנָ֥תְנ֛וּ עַל־שְׁתֵּ֥י הַמְּזוּזֹ֖ת וְעַל־הַמַּשְׁק֑וֹף עַ֚ל הַבָּ֣תִּ֔ים אֲשֶׁר־יֹאכְל֥וּ אֹת֖וֹ בָּהֶֽם׃ וְיִקְחוּ מֵהַדָּם וְיָשִׂימוּ עַל שְׁנֵי צִדֵּי פֶּתַח הַדֶּלֶת וְעַל הַמַּשְׁקוֹף שֶׁבַּבָּתִּים שֶׁיֹּאכְלוּ אוֹתוֹ בָּהֶם.
ח וְאָכְל֥וּ אֶת־הַבָּשָׂ֖ר בַּלַּ֣יְלָה הַזֶּ֑ה צְלִי־אֵ֣שׁ וּמַצּ֔וֹת עַל־מְרֹרִ֖ים יֹאכְלֻֽהוּ׃ וְיֹאכְלוּ אֶת הַבָּשָׂר בַּלַּיְלָה הַזֶּה, יֹאכְלוּ אוֹתוֹ צָלוּי בָּאֵשׁ עִם מַצּוֹת וִירָקוֹת מָרִים.
ט אַל־תֹּאכְל֤וּ מִמֶּ֙נּוּ֙ נָ֔א וּבָשֵׁ֥ל מְבֻשָּׁ֖ל בַּמָּ֑יִם כִּ֣י אִם־צְלִי־אֵ֔שׁ רֹאשׁ֥וֹ עַל־כְּרָעָ֖יו וְעַל־קִרְבּֽוֹ׃ אַל תֹּאכְלוּ מִמֶּנּוּ כְּשֶׁאֵינוֹ צָלוּי מַסְפִּיק אוֹ כְּשֶׁהוּא מְבֻשָּׁל בְּמַיִם, אֶלָּא צָלוּי בָּאֵשׁ הֵיטֵב רֹאשׁוֹ עִם רַגְלָיו וְעִם אֵבָרָיו הַפְּנִימִיִּים.
י וְלֹא־תוֹתִ֥ירוּ מִמֶּ֖נּוּ עַד־בֹּ֑קֶר וְהַנֹּתָ֥ר מִמֶּ֛נּוּ עַד־בֹּ֖קֶר בָּאֵ֥שׁ תִּשְׂרֹֽפוּ׃ וְאַל תַּשְׁאִירוּ מִמֶּנּוּ עַד הַבּוֹקֶר, וּמַה שֶּׁיִּשָּׁאֵר מִמֶּנּוּ עַד הַבּוֹקֶר תִּשְׂרְפוּ בָּאֵשׁ.
יא וְכָ֘כָה֮ תֹּאכְל֣וּ אֹתוֹ֒ מׇתְנֵיכֶ֣ם חֲגֻרִ֔ים נַֽעֲלֵיכֶם֙ בְּרַגְלֵיכֶ֔ם וּמַקֶּלְכֶ֖ם בְּיֶדְכֶ֑ם וַאֲכַלְתֶּ֤ם אֹתוֹ֙ בְּחִפָּז֔וֹן פֶּ֥סַח ה֖וּא לַיהֹוָֽה׃ וְכָכָה תֹּאכְלוּ אוֹתוֹ: מָתְנֵיכֶם חֲגוּרִים, נַעֲלֵיכֶם בְּרַגְלֵיכֶם, וּמַקֶּלְכֶם בְּיֶדְכֶם, וְתֹאכְלוּ אוֹתוֹ בְּחָפְזָה, זֶהוּ קָרְבָּן פֶּסַח לַה'.
יב וְעָבַרְתִּ֣י בְאֶֽרֶץ־מִצְרַ֘יִם֮ בַּלַּ֣יְלָה הַזֶּה֒ וְהִכֵּיתִ֤י כׇל־בְּכוֹר֙ בְּאֶ֣רֶץ מִצְרַ֔יִם מֵאָדָ֖ם וְעַד־בְּהֵמָ֑ה וּבְכׇל־אֱלֹהֵ֥י מִצְרַ֛יִם אֶֽעֱשֶׂ֥ה שְׁפָטִ֖ים אֲנִ֥י יְהֹוָֽה׃ וְאֶעֱבוֹר בְּאֶרֶץ מִצְרַיִם בַּלַּיְלָה הַזֶּה וְאֶהֱרוֹג כָּל בְּכוֹר בְּאֶרֶץ מִצְרַיִם מֵאָדָם וְעַד בְּהֵמָה, וּבְכָל אֱלֹהֵי מִצְרַיִם אֶתֵּן עֳנָשִׁים, אֲנִי ה'.
יג וְהָיָה֩ הַדָּ֨ם לָכֶ֜ם לְאֹ֗ת עַ֤ל הַבָּתִּים֙ אֲשֶׁ֣ר אַתֶּ֣ם שָׁ֔ם וְרָאִ֙יתִי֙ אֶת־הַדָּ֔ם וּפָסַחְתִּ֖י עֲלֵכֶ֑ם וְלֹֽא־יִֽהְיֶ֨ה בָכֶ֥ם נֶ֙גֶף֙ לְמַשְׁחִ֔ית בְּהַכֹּתִ֖י בְּאֶ֥רֶץ מִצְרָֽיִם׃ הַדָּם יִהְיֶה לָכֶם סִימָן עַל הַבָּתִּים שֶׁאַתֶּם שָׁם, אֶרְאֶה אֶת הַדָּם וַאֲדַלֵּג עֲלֵיכֶם וְלֹא תִּהְיֶה בָּכֶם מַגֵּפָה מַשְׁחִיתָה כְּשֶׁאַכֶּה בְּאֶרֶץ מִצְרַיִם.
יד וְהָיָה֩ הַיּ֨וֹם הַזֶּ֤ה לָכֶם֙ לְזִכָּר֔וֹן וְחַגֹּתֶ֥ם אֹת֖וֹ חַ֣ג לַֽיהֹוָ֑ה לְדֹרֹ֣תֵיכֶ֔ם חֻקַּ֥ת עוֹלָ֖ם תְּחׇגֻּֽהוּ׃ הַיּוֹם הַזֶּה יִהְיֶה לָכֶם לְזִכָּרוֹן, תַּחְגְּגוּ אוֹתוֹ חַג לַה' בְּכָל הַדּוֹרוֹת – חוֹק לְעוֹלָם שֶׁתַּחְגְּגוּ אוֹתוֹ.
טו שִׁבְעַ֤ת יָמִים֙ מַצּ֣וֹת תֹּאכֵ֔לוּ אַ֚ךְ בַּיּ֣וֹם הָרִאשׁ֔וֹן תַּשְׁבִּ֥יתוּ שְּׂאֹ֖ר מִבָּתֵּיכֶ֑ם כִּ֣י ׀ כׇּל־אֹכֵ֣ל חָמֵ֗ץ וְנִכְרְתָ֞ה הַנֶּ֤פֶשׁ הַהִוא֙ מִיִּשְׂרָאֵ֔ל מִיּ֥וֹם הָרִאשֹׁ֖ן עַד־י֥וֹם הַשְּׁבִעִֽי׃ שִׁבְעָה יָמִים תֹּאכְלוּ מַצּוֹת, וְאִלּוּ בָּצֵק שֶׁהֶחְמִיץ שֶׁמַּתְפִּיחַ בָּצֵק אַחֵר תַּעֲלִימוּ מִבָּתֵּיכֶם בַּיּוֹם הָרִאשׁוֹן, כִּי כָּל מִי שֶׁיֹּאכַל חָמֵץ מִיּוֹם הָרִאשׁוֹן עַד יוֹם הַשְּׁבִיעִי – הַנֶּפֶשׁ הַהִיא תֹּאבַד מִיִּשְׂרָאֵל.
טז וּבַיּ֤וֹם הָרִאשׁוֹן֙ מִקְרָא־קֹ֔דֶשׁ וּבַיּוֹם֙ הַשְּׁבִיעִ֔י מִקְרָא־קֹ֖דֶשׁ יִהְיֶ֣ה לָכֶ֑ם כׇּל־מְלָאכָה֙ לֹא־יֵעָשֶׂ֣ה בָהֶ֔ם אַ֚ךְ אֲשֶׁ֣ר יֵאָכֵ֣ל לְכׇל־נֶ֔פֶשׁ ה֥וּא לְבַדּ֖וֹ יֵעָשֶׂ֥ה לָכֶֽם׃ וּבַיּוֹם הָרִאשׁוֹן מְאוֹרָע קָדוֹשׁ, וּבַיּוֹם הַשְּׁבִיעִי יִהְיֶה לָכֶם מְאוֹרָע קָדוֹשׁ – אָסוּר לַעֲשׂוֹת שׁוּם מְלָאכָה בָּהֶם, רַק מַה שֶּׁרָאוּי לַאֲכִילָה לְכָל אָדָם – אוֹתוֹ בִּלְבַד מֻתָּר לָכֶם לַעֲשׂוֹת.
יז וּשְׁמַרְתֶּם֮ אֶת־הַמַּצּוֹת֒ כִּ֗י בְּעֶ֙צֶם֙ הַיּ֣וֹם הַזֶּ֔ה הוֹצֵ֥אתִי אֶת־צִבְאוֹתֵיכֶ֖ם מֵאֶ֣רֶץ מִצְרָ֑יִם וּשְׁמַרְתֶּ֞ם אֶת־הַיּ֥וֹם הַזֶּ֛ה לְדֹרֹתֵיכֶ֖ם חֻקַּ֥ת עוֹלָֽם׃ זִכְרוּ לְקַיֵּם אֶת מִצְוַת הַמַּצּוֹת כִּי בַּיּוֹם הַזֶּה עַצְמוֹ הוֹצֵאתִי אֶת קְהִלּוֹתֵיכֶם מֵאֶרֶץ מִצְרַיִם בִּמְהִירוּת וּבְצֵקְכֶם לֹא הֶחְמִיץ, שִׁמְרוּ אֶת הַיּוֹם הַזֶּה לְדוֹרוֹת, חוֹק לְעוֹלָם.
יח בָּרִאשֹׁ֡ן בְּאַרְבָּעָה֩ עָשָׂ֨ר י֤וֹם לַחֹ֙דֶשׁ֙ בָּעֶ֔רֶב תֹּאכְל֖וּ מַצֹּ֑ת עַ֠ד י֣וֹם הָאֶחָ֧ד וְעֶשְׂרִ֛ים לַחֹ֖דֶשׁ בָּעָֽרֶב׃ בַּחוֹדֶשׁ הָרִאשׁוֹן בְּאַרְבָּעָה עָשָׂר יוֹם לַחוֹדֶשׁ בָּעֶרֶב תֹּאכְלוּ מַצּוֹת, עַד יוֹם הָעֶשְׂרִים וְאֶחָד לַחוֹדֶשׁ בָּעֶרֶב.
יט שִׁבְעַ֣ת יָמִ֔ים שְׂאֹ֕ר לֹ֥א יִמָּצֵ֖א בְּבָתֵּיכֶ֑ם כִּ֣י ׀ כׇּל־אֹכֵ֣ל מַחְמֶ֗צֶת וְנִכְרְתָ֞ה הַנֶּ֤פֶשׁ הַהִוא֙ מֵעֲדַ֣ת יִשְׂרָאֵ֔ל בַּגֵּ֖ר וּבְאֶזְרַ֥ח הָאָֽרֶץ׃ שִׁבְעָה יָמִים לֹא יִמָּצֵא בְּבָתֵּיכֶם בָּצֵק שֶׁהֶחְמִיץ שֶׁמַּתְפִּיחַ בָּצֵק אַחֵר, כִּי מִי שֶׁיֹּאכַל חָמֵץ – בֵּין גֵּר וּבֵין אֶזְרַח הָאָרֶץ – הַנֶּפֶשׁ הַהִיא תֹּאבַד מֵהַצִּבּוּר שֶׁל יִשְׂרָאֵל.
כ כׇּל־מַחְמֶ֖צֶת לֹ֣א תֹאכֵ֑לוּ בְּכֹל֙ מוֹשְׁבֹ֣תֵיכֶ֔ם תֹּאכְל֖וּ מַצּֽוֹת׃ אַל תֹּאכְלוּ שׁוּם מַאֲכָל מִבָּצֵק שֶׁהֶחְמִיץ, בְּכָל מְקוֹמוֹת מְגוּרֵיכֶם תֹּאכְלוּ מַצּוֹת.

חמישי[עריכה]

פ פָּרָשַׁת מְשִׁיכַת הַפֶּסַח
כא וַיִּקְרָ֥א מֹשֶׁ֛ה לְכׇל־זִקְנֵ֥י יִשְׂרָאֵ֖ל וַיֹּ֣אמֶר אֲלֵהֶ֑ם מִֽשְׁכ֗וּ וּקְח֨וּ לָכֶ֥ם צֹ֛אן לְמִשְׁפְּחֹתֵיכֶ֖ם וְשַׁחֲט֥וּ הַפָּֽסַח׃ מֹשֶׁה קָרָא לְכָל זִקְנֵי יִשְׂרָאֵל וְאָמַר לָהֶם: מִשְׁכוּ וּקְחוּ לָכֶם צֹאן לְמִשְׁפְּחוֹתֵיכֶם וְתִשְׁחֲטוּ אֶת הַפֶּסַח.
כב וּלְקַחְתֶּ֞ם אֲגֻדַּ֣ת אֵז֗וֹב וּטְבַלְתֶּם֮ בַּדָּ֣ם אֲשֶׁר־בַּסַּף֒ וְהִגַּעְתֶּ֤ם אֶל־הַמַּשְׁקוֹף֙ וְאֶל־שְׁתֵּ֣י הַמְּזוּזֹ֔ת מִן־הַדָּ֖ם אֲשֶׁ֣ר בַּסָּ֑ף וְאַתֶּ֗ם לֹ֥א תֵצְא֛וּ אִ֥ישׁ מִפֶּֽתַח־בֵּית֖וֹ עַד־בֹּֽקֶר׃ וְתִקְחוּ חֲבִילַת עַנְפֵי זַעְתָּר וְתִטְבְּלוּ בַּדָּם שֶׁבַּכְּלִי, וְתַעֲשׂוּ שֶׁיִּגַּע בַּמַּשְׁקוֹף וּבִשְׁנֵי צִדֵּי הַדֶּלֶת מֵהַדָּם שֶׁבַּכְּלִי, וְאַתֶּם אַל תֵּצְאוּ אַף אֶחָד מִפֶּתַח בֵּיתוֹ עַד הַבּוֹקֶר.
כג וְעָבַ֣ר יְהֹוָה֮ לִנְגֹּ֣ף אֶת־מִצְרַ֒יִם֒ וְרָאָ֤ה אֶת־הַדָּם֙ עַל־הַמַּשְׁק֔וֹף וְעַ֖ל שְׁתֵּ֣י הַמְּזוּזֹ֑ת וּפָסַ֤ח יְהֹוָה֙ עַל־הַפֶּ֔תַח וְלֹ֤א יִתֵּן֙ הַמַּשְׁחִ֔ית לָבֹ֥א אֶל־בָּתֵּיכֶ֖ם לִנְגֹּֽף׃ ה' יַעֲבוֹר לְהָבִיא מַגֵּפָה בְּמִצְרַיִם, וְיִרְאֶה אֶת הַדָּם עַל הַמַּשְׁקוֹף וְעַל שְׁתֵּי הַמְּזוּזוֹת, וַה' יְדַלֵּג עַל הַפֶּתַח וְלֹא יִתֵּן לַמַּשְׁחִית לָבוֹא לְבָתֵּיכֶם לְהָבִיא מַגֵּפָה.
כד וּשְׁמַרְתֶּ֖ם אֶת־הַדָּבָ֣ר הַזֶּ֑ה לְחׇק־לְךָ֥ וּלְבָנֶ֖יךָ עַד־עוֹלָֽם׃ הַקְפִּידוּ עַל הַדָּבָר הַזֶּה וְתִקְבְּעוּ אוֹתוֹ לְחוֹק לָכֶם וְלִבְנֵיכֶם לְעוֹלָם.
כה וְהָיָ֞ה כִּֽי־תָבֹ֣אוּ אֶל־הָאָ֗רֶץ אֲשֶׁ֨ר יִתֵּ֧ן יְהֹוָ֛ה לָכֶ֖ם כַּאֲשֶׁ֣ר דִּבֵּ֑ר וּשְׁמַרְתֶּ֖ם אֶת־הָעֲבֹדָ֥ה הַזֹּֽאת׃ כְּשֶׁתָּבוֹאוּ לָאָרֶץ שֶׁה' יִתֵּן לָכֶם כְּמוֹ שֶׁדִּבֵּר, תַּקְפִּידוּ עַל עֲבוֹדָה זוֹ שֶׁל הַקְרָבַת הַקָּרְבָּן.
כו וְהָיָ֕ה כִּֽי־יֹאמְר֥וּ אֲלֵיכֶ֖ם בְּנֵיכֶ֑ם מָ֛ה הָעֲבֹדָ֥ה הַזֹּ֖את לָכֶֽם׃ וּכְשֶׁיֹּאמְרוּ לָכֶם בְּנֵיכֶם: מָה הָעֲבוֹדָה הַזֹּאת לָכֶם?
כז וַאֲמַרְתֶּ֡ם זֶֽבַח־פֶּ֨סַח ה֜וּא לַֽיהֹוָ֗ה אֲשֶׁ֣ר פָּ֠סַ֠ח עַל־בָּתֵּ֤י בְנֵֽי־יִשְׂרָאֵל֙ בְּמִצְרַ֔יִם בְּנׇגְפּ֥וֹ אֶת־מִצְרַ֖יִם וְאֶת־בָּתֵּ֣ינוּ הִצִּ֑יל וַיִּקֹּ֥ד הָעָ֖ם וַיִּֽשְׁתַּחֲוֽוּ׃ תֹּאמְרוּ: זֶה קָרְבַּן פֶּסַח לַה' שֶׁדִּלֵּג עַל בָּתֵּי בְּנֵי יִשְׂרָאֵל בְּמִצְרַיִם כְּשֶׁהֵבִיא מַגֵּפָה בְּמִצְרַיִם וְהִצִּיל אֶת בָּתֵּינוּ. הָעָם קָדוּ וְהִשְׁתַּחֲווּ.
כח וַיֵּלְכ֥וּ וַֽיַּעֲשׂ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל כַּאֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֶת־מֹשֶׁ֥ה וְאַהֲרֹ֖ן כֵּ֥ן עָשֽׂוּ׃ בְּנֵי יִשְׂרָאֵל הָלְכוּ וּבִצְּעוּ – כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה וְאַהֲרֹן כָּךְ הֵם עָשׂוּ.

ששי[עריכה]

ס פָּרָשַׁת מַכַּת בְּכוֹרוֹת
כט וַיְהִ֣י ׀ בַּחֲצִ֣י הַלַּ֗יְלָה וַֽיהֹוָה֮ הִכָּ֣ה כׇל־בְּכוֹר֮ בְּאֶ֣רֶץ מִצְרַ֒יִם֒ מִבְּכֹ֤ר פַּרְעֹה֙ הַיֹּשֵׁ֣ב עַל־כִּסְא֔וֹ עַ֚ד בְּכ֣וֹר הַשְּׁבִ֔י אֲשֶׁ֖ר בְּבֵ֣ית הַבּ֑וֹר וְכֹ֖ל בְּכ֥וֹר בְּהֵמָֽה׃ בַּחֲצוֹת הַלַּיְלָה ה' הִכָּה כָּל בְּכוֹר בְּאֶרֶץ מִצְרַיִם, מֵהַבְּכוֹר שֶׁל פַּרְעֹה שֶׁאָמוּר לָרֶשֶׁת אוֹתוֹ וְלָשֶׁבֶת עַל כִּסְאוֹ עַד בְּכוֹר שֶׁל שָׁבוּי שֶׁבְּבֵית הַסּוֹהַר, וְכָל בְּכוֹר בְּהֵמָה.
ל וַיָּ֨קׇם פַּרְעֹ֜ה לַ֗יְלָה ה֤וּא וְכׇל־עֲבָדָיו֙ וְכׇל־מִצְרַ֔יִם וַתְּהִ֛י צְעָקָ֥ה גְדֹלָ֖ה בְּמִצְרָ֑יִם כִּֽי־אֵ֣ין בַּ֔יִת אֲשֶׁ֥ר אֵֽין־שָׁ֖ם מֵֽת׃ פַּרְעֹה קָם בַּלַּיְלָה, הוּא וְכָל עֲבָדָיו וְכָל מִצְרַיִם, וְהָיְתָה צְעָקָה גְּדוֹלָה בְּמִצְרַיִם, כִּי אֵין בַּיִת שֶׁאֵין שָׁם מֵת.
לא וַיִּקְרָא֩ לְמֹשֶׁ֨ה וּֽלְאַהֲרֹ֜ן לַ֗יְלָה וַיֹּ֙אמֶר֙ ק֤וּמוּ צְּאוּ֙ מִתּ֣וֹךְ עַמִּ֔י גַּם־אַתֶּ֖ם גַּם־בְּנֵ֣י יִשְׂרָאֵ֑ל וּלְכ֛וּ עִבְד֥וּ אֶת־יְהֹוָ֖ה כְּדַבֶּרְכֶֽם׃ פַּרְעֹה קָרָא לְמֹשֶׁה וּלְאַהֲרֹן בַּלַּיְלָה וְאָמַר: קוּמוּ צְאוּ מִתּוֹךְ עַמִּי, גַּם אַתֶּם גַּם בְּנֵי יִשְׂרָאֵל, וּלְכוּ עִבְדוּ אֶת ה' כְּמוֹ שֶׁדִּבַּרְתֶּם!
לב גַּם־צֹאנְכֶ֨ם גַּם־בְּקַרְכֶ֥ם קְח֛וּ כַּאֲשֶׁ֥ר דִּבַּרְתֶּ֖ם וָלֵ֑כוּ וּבֵֽרַכְתֶּ֖ם גַּם־אֹתִֽי׃ קְחוּ גַּם אֶת צֹאנְכֶם וְגַם אֶת בְּקַרְכֶם כְּמוֹ שֶׁדִּבַּרְתֶּם וּלְכוּ, וְתִתְפַּלְּלוּ גַּם עָלַי.
לג וַתֶּחֱזַ֤ק מִצְרַ֙יִם֙ עַל־הָעָ֔ם לְמַהֵ֖ר לְשַׁלְּחָ֣ם מִן־הָאָ֑רֶץ כִּ֥י אָמְר֖וּ כֻּלָּ֥נוּ מֵתִֽים׃ מִצְרַיִם לָחֲצוּ עַל הָעָם לְמַהֵר לְגָרֵשׁ אוֹתָם מֵהָאָרֶץ, כִּי אָמְרוּ: כֻּלָּנוּ עוֹמְדִים לָמוּת!
לד וַיִּשָּׂ֥א הָעָ֛ם אֶת־בְּצֵק֖וֹ טֶ֣רֶם יֶחְמָ֑ץ מִשְׁאֲרֹתָ֛ם צְרֻרֹ֥ת בְּשִׂמְלֹתָ֖ם עַל־שִׁכְמָֽם׃ הָעָם לָקַח אֶת בְּצֵקוֹ בִּמְהִירוּת לִפְנֵי שֶׁהֶחְמִיץ, בְּצֵקָם אָרוּז בְּתוֹךְ בִּגְדֵיהֶם עַל כְּתֵפָם.
לה וּבְנֵי־יִשְׂרָאֵ֥ל עָשׂ֖וּ כִּדְבַ֣ר מֹשֶׁ֑ה וַֽיִּשְׁאֲלוּ֙ מִמִּצְרַ֔יִם כְּלֵי־כֶ֛סֶף וּכְלֵ֥י זָהָ֖ב וּשְׂמָלֹֽת׃ וּבְנֵי יִשְׂרָאֵל עָשׂוּ כְּמוֹ שֶׁמֹּשֶׁה אָמַר, וּבִקְשׁוּ מִמִּצְרַיִם כְּלֵי כֶּסֶף וּכְלֵי זָהָב וּבְגָדִים.
לו וַֽיהֹוָ֞ה נָתַ֨ן אֶת־חֵ֥ן הָעָ֛ם בְּעֵינֵ֥י מִצְרַ֖יִם וַיַּשְׁאִל֑וּם וַֽיְנַצְּל֖וּ אֶת־מִצְרָֽיִם׃ וַה' עָשָׂה שֶׁהָעָם מָצָא חֵן בְּעֵינֵי מִצְרַיִם וְהֵם נַעֲנוּ לְבַקָּשָׁתָם וְהֵם רוֹקְנוּ אֶת מִצְרַיִם מֵרְכוּשׁ.
פ פָּרָשַׁת יְצִיאַת מִצְרַיִם
לז וַיִּסְע֧וּ בְנֵֽי־יִשְׂרָאֵ֛ל מֵרַעְמְסֵ֖ס סֻכֹּ֑תָה כְּשֵׁשׁ־מֵא֨וֹת אֶ֧לֶף רַגְלִ֛י הַגְּבָרִ֖ים לְבַ֥ד מִטָּֽף׃ בְּנֵי יִשְׂרָאֵל נָסְעוּ מֵרַעְמְסֵס לְסֻכּוֹת, כְּשֵׁשׁ מֵאוֹת אֶלֶף הוֹלְכֵי רֶגֶל – הַגְּבָרִים חוּץ מִיְּלָדִים.
לח וְגַם־עֵ֥רֶב רַ֖ב עָלָ֣ה אִתָּ֑ם וְצֹ֣אן וּבָקָ֔ר מִקְנֶ֖ה כָּבֵ֥ד מְאֹֽד׃ וְגַם נָכְרִים רַבִּים עָלוּ אִתָּם, וְצֹאן וּבָקָר –בְּהֵמוֹת רַבּוֹת מְאוֹד.
לט וַיֹּאפ֨וּ אֶת־הַבָּצֵ֜ק אֲשֶׁ֨ר הוֹצִ֧יאוּ מִמִּצְרַ֛יִם עֻגֹ֥ת מַצּ֖וֹת כִּ֣י לֹ֣א חָמֵ֑ץ כִּֽי־גֹרְשׁ֣וּ מִמִּצְרַ֗יִם וְלֹ֤א יָֽכְלוּ֙ לְהִתְמַהְמֵ֔הַּ וְגַם־צֵדָ֖ה לֹא־עָשׂ֥וּ לָהֶֽם׃ הֵם אָפוּ מֵהַבָּצֵק שֶׁהוֹצִיאוּ מִמִּצְרַיִם פִּתּוֹת שֶׁלֹּא הֶחְמִיצוּ, כִּי הוּא לֹא הִסְפִּיק לִתְפּוֹחַ, כִּי הֵם גּוֹרְשׁוּ מִמִּצְרַיִם וְלֹא יָכְלוּ לְהִתְעַכֵּב, וְגַם לֹא עָשׂוּ לָהֶם צֵדָה לַדֶּרֶךְ.
מ וּמוֹשַׁב֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁ֥ר יָשְׁב֖וּ בְּמִצְרָ֑יִם שְׁלֹשִׁ֣ים שָׁנָ֔ה וְאַרְבַּ֥ע מֵא֖וֹת שָׁנָֽה׃ זְמַן הַמְּגוּרִים שֶׁבְּנֵי יִשְׂרָאֵל גָּרוּ בְּמִצְרַיִם הָיָה אַרְבַּע מֵאוֹת וּשְׁלוֹשִׁים שָׁנָה.
מא וַיְהִ֗י מִקֵּץ֙ שְׁלֹשִׁ֣ים שָׁנָ֔ה וְאַרְבַּ֥ע מֵא֖וֹת שָׁנָ֑ה וַיְהִ֗י בְּעֶ֙צֶם֙ הַיּ֣וֹם הַזֶּ֔ה יָ֥צְא֛וּ כׇּל־צִבְא֥וֹת יְהֹוָ֖ה מֵאֶ֥רֶץ מִצְרָֽיִם׃ בְּסוֹף אַרְבַּע מֵאוֹת וּשְׁלוֹשִׁים שָׁנָה, בְּאוֹתוֹ יוֹם עַצְמוֹ יָצְאוּ כָּל הַקְּהִלּוֹת שֶׁל ה' מֵאֶרֶץ מִצְרַיִם.
מב לֵ֣יל שִׁמֻּרִ֥ים הוּא֙ לַֽיהֹוָ֔ה לְהוֹצִיאָ֖ם מֵאֶ֣רֶץ מִצְרָ֑יִם הֽוּא־הַלַּ֤יְלָה הַזֶּה֙ לַֽיהֹוָ֔ה שִׁמֻּרִ֛ים לְכׇל־בְּנֵ֥י יִשְׂרָאֵ֖ל לְדֹרֹתָֽם׃ לַיְלָה זֶה ה' יִיעֵד לְהוֹצִיא אוֹתָם מֵאֶרֶץ מִצְרַיִם, לַיְלָה זֶה שֶׁה' יִיעֵד, כָּל בְּנֵי יִשְׂרָאֵל לְדוֹרוֹתֵיהֶם מְיַעֲדִים אוֹתוֹ לַעֲבוֹדַת ה'.
פ פָּרָשַׁת מִצְוַת הַפֶּסַח
מג וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֣ה וְאַהֲרֹ֔ן זֹ֖את חֻקַּ֣ת הַפָּ֑סַח כׇּל־בֶּן־נֵכָ֖ר לֹא־יֹ֥אכַל בּֽוֹ׃ ה' אָמַר לְמֹשֶׁה וְאַהֲרֹן: אֵלֶּה חֻקֵּי הַפֶּסַח: כָּל נָכְרִי לֹא יֹאכַל מִמֶּנּוּ.
מד וְכׇל־עֶ֥בֶד אִ֖ישׁ מִקְנַת־כָּ֑סֶף וּמַלְתָּ֣ה אֹת֔וֹ אָ֖ז יֹ֥אכַל בּֽוֹ׃ וְכָל עֶבֶד שֶׁל אִישׁ שֶׁנִּקְנָה בְּכֶסֶף – תָּמוּל אוֹתוֹ וְרַק אָז הוּא יֹאכַל מִמֶּנּוּ.
מה תּוֹשָׁ֥ב וְשָׂכִ֖יר לֹא־יֹ֥אכַל בּֽוֹ׃ בֶּן מִיעוּטִים וְשָׂכִיר לֹא יֹאכְלוּ מִמֶּנּוּ.
מו בְּבַ֤יִת אֶחָד֙ יֵאָכֵ֔ל לֹא־תוֹצִ֧יא מִן־הַבַּ֛יִת מִן־הַבָּשָׂ֖ר ח֑וּצָה וְעֶ֖צֶם לֹ֥א תִשְׁבְּרוּ־בֽוֹ׃ הוּא יֵאָכֵל בְּחֶדֶר אֶחָד, אַל תּוֹצִיאוּ מֵהַבָּשָׂר מִחוּץ לַחֶדֶר, וְאַל תִּשְׁבְּרוּ בּוֹ עֶצֶם.
מז כׇּל־עֲדַ֥ת יִשְׂרָאֵ֖ל יַעֲשׂ֥וּ אֹתֽוֹ׃ כָּל הַצִּבּוּר שֶׁל יִשְׂרָאֵל יַעֲשׂוּ אוֹתוֹ.
מח וְכִֽי־יָג֨וּר אִתְּךָ֜ גֵּ֗ר וְעָ֣שָׂה פֶ֘סַח֮ לַיהֹוָה֒ הִמּ֧וֹל ל֣וֹ כׇל־זָכָ֗ר וְאָז֙ יִקְרַ֣ב לַעֲשֹׂת֔וֹ וְהָיָ֖ה כְּאֶזְרַ֣ח הָאָ֑רֶץ וְכׇל־עָרֵ֖ל לֹֽא־יֹ֥אכַל בּֽוֹ׃ וְאִם יָגוּר אִתְּךָ גֵּר הוּא יַעֲשֶׂה פֶּסַח לַה' – כָּל גֵּר זָכָר יָמוּל וְרַק אָז הוּא יִגַּשׁ לַעֲשׂוֹת אוֹתוֹ וְיִהְיֶה כְּאֶזְרַח הָאָרֶץ, וְכָל מִי שֶׁאֵינוֹ מָהוּל לֹא יֹאכַל מִמֶּנּוּ.
מט תּוֹרָ֣ה אַחַ֔ת יִהְיֶ֖ה לָֽאֶזְרָ֑ח וְלַגֵּ֖ר הַגָּ֥ר בְּתוֹכְכֶֽם׃ דִּין אֶחָד יִהְיֶה לָאֶזְרָח וְלַגֵּר שֶׁגָּר בֵּינֵיכֶם.
נ וַיַּֽעֲשׂ֖וּ כׇּל־בְּנֵ֣י יִשְׂרָאֵ֑ל כַּאֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֶת־מֹשֶׁ֥ה וְאֶֽת־אַהֲרֹ֖ן כֵּ֥ן עָשֽׂוּ׃ בְּנֵי יִשְׂרָאֵל בִּצְּעוּ – כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה וְאֶת אַהֲרֹן כָּךְ הֵם עָשׂוּ.
ס
נא וַיְהִ֕י בְּעֶ֖צֶם הַיּ֣וֹם הַזֶּ֑ה הוֹצִ֨יא יְהֹוָ֜ה אֶת־בְּנֵ֧י יִשְׂרָאֵ֛ל מֵאֶ֥רֶץ מִצְרַ֖יִם עַל־צִבְאֹתָֽם׃ בְּאוֹתוֹ יוֹם עַצְמוֹ ה' הוֹצִיא אֶת בְּנֵי יִשְׂרָאֵל וְכָל קְהִלּוֹתֵיהֶם מֵאֶרֶץ מִצְרַיִם.

שביעי[עריכה]

פ פָּרָשַׁת קַדֵּשׁ
יג א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב קַדֶּשׁ־לִ֨י כׇל־בְּכ֜וֹר פֶּ֤טֶר כׇּל־רֶ֙חֶם֙ בִּבְנֵ֣י יִשְׂרָאֵ֔ל בָּאָדָ֖ם וּבַבְּהֵמָ֑ה לִ֖י הֽוּא׃ קַדֵּשׁ לִי אֶת כָּל הַבְּכוֹרוֹת שֶׁל בְּנֵי יִשְׂרָאֵל – כָּל מִי שֶׁיָּצָא רִאשׁוֹן מֵהָרֶחֶם בָּאָדָם וּבַבְּהֵמָה, הוּא שֶׁלִּי.
ג וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶל־הָעָ֗ם זָכ֞וֹר אֶת־הַיּ֤וֹם הַזֶּה֙ אֲשֶׁ֨ר יְצָאתֶ֤ם מִמִּצְרַ֙יִם֙ מִבֵּ֣ית עֲבָדִ֔ים כִּ֚י בְּחֹ֣זֶק יָ֔ד הוֹצִ֧יא יְהֹוָ֛ה אֶתְכֶ֖ם מִזֶּ֑ה וְלֹ֥א יֵאָכֵ֖ל חָמֵֽץ׃ מֹשֶׁה אָמַר לָעָם: זִכְרוּ אֶת הַיּוֹם הַזֶּה שֶׁיְּצָאתֶם מִמִּצְרַיִם מִבֵּית עֲבָדִים – שֶׁה' הוֹצִיא אֶתְכֶם מִכָּאן בְּיָד חֲזָקָה, וְלֹא יֵאָכֵל בּוֹ חָמֵץ.
ד הַיּ֖וֹם אַתֶּ֣ם יֹצְאִ֑ים בְּחֹ֖דֶשׁ הָאָבִֽיב׃ הַיּוֹם אַתֶּם יוֹצְאִים בְּחוֹדֶשׁ הָאָבִיב.
ה וְהָיָ֣ה כִֽי־יְבִיאֲךָ֣ יְהֹוָ֡ה אֶל־אֶ֣רֶץ הַֽ֠כְּנַעֲנִ֠י וְהַחִתִּ֨י וְהָאֱמֹרִ֜י וְהַחִוִּ֣י וְהַיְבוּסִ֗י אֲשֶׁ֨ר נִשְׁבַּ֤ע לַאֲבֹתֶ֙יךָ֙ לָ֣תֶת לָ֔ךְ אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָ֑שׁ וְעָבַדְתָּ֛ אֶת־הָעֲבֹדָ֥ה הַזֹּ֖את בַּחֹ֥דֶשׁ הַזֶּֽה׃ כְּשֶׁה' יָבִיא אוֹתְךָ לְאֶרֶץ הַכְּנַעֲנִי וְהַחִתִּי וְהָאֱמוֹרִי וְהַחִוִּי וְהַיְּבוּסִי שֶׁנִּשְׁבַּע לַאֲבוֹתֶיךָ לָתֵת לְךָ – אֶרֶץ שׁוֹפַעַת חָלָב וּדְבָשׁ, תַּעֲבוֹד אֶת הָעֲבוֹדָה הַזֹּאת בַּחוֹדֶשׁ הַזֶּה.
ו שִׁבְעַ֥ת יָמִ֖ים תֹּאכַ֣ל מַצֹּ֑ת וּבַיּוֹם֙ הַשְּׁבִיעִ֔י חַ֖ג לַיהֹוָֽה׃ שִׁבְעָה יָמִים תֹּאכַל מַצּוֹת וּבַיּוֹם הַשְּׁבִיעִי חַג לַה'.
ז מַצּוֹת֙ יֵֽאָכֵ֔ל אֵ֖ת שִׁבְעַ֣ת הַיָּמִ֑ים וְלֹֽא־יֵרָאֶ֨ה לְךָ֜ חָמֵ֗ץ וְלֹֽא־יֵרָאֶ֥ה לְךָ֛ שְׂאֹ֖ר בְּכׇל־גְּבֻלֶֽךָ׃ בְּשִׁבְעַת הַיָּמִים יֹאכְלוּ מַצּוֹת, וְלֹא יִהְיֶה אֶצְלְךָ חָמֵץ, וְלֹא יִהְיֶה אֶצְלְךָ בָּצֵק שֶׁהֶחְמִיץ שֶׁמַּתְפִּיחַ בָּצֵק, בְּכָל מְקוֹמְךָ.
ח וְהִגַּדְתָּ֣ לְבִנְךָ֔ בַּיּ֥וֹם הַה֖וּא לֵאמֹ֑ר בַּעֲב֣וּר זֶ֗ה עָשָׂ֤ה יְהֹוָה֙ לִ֔י בְּצֵאתִ֖י מִמִּצְרָֽיִם׃ סַפֵּר לְבִנְךָ בַּיּוֹם הַהוּא וְתֹאמַר: בִּשְׁבִיל שֶׁאֶעֱשֶׂה עֲבוֹדָה זוֹ ה' גָּאַל אוֹתִי כְּשֶׁיָּצָאתִי מִמִּצְרַיִם.
ט וְהָיָה֩ לְךָ֨ לְא֜וֹת עַל־יָדְךָ֗ וּלְזִכָּרוֹן֙ בֵּ֣ין עֵינֶ֔יךָ לְמַ֗עַן תִּהְיֶ֛ה תּוֹרַ֥ת יְהֹוָ֖ה בְּפִ֑יךָ כִּ֚י בְּיָ֣ד חֲזָקָ֔ה הוֹצִֽאֲךָ֥ יְהֹוָ֖ה מִמִּצְרָֽיִם׃ כְּתוֹב לְךָ אֶת הַדְּבָרִים לְסִימָן עַל יָדְךָ וּלְזִכָּרוֹן בֵּין עֵינֶיךָ כְּדֵי שֶׁתּוֹרַת ה' תִּהְיֶה בְּפִיךָ – לְתֵעוּד שֶׁה' הוֹצִיא אוֹתְךָ מִמִּצְרַיִם בְּיָד חֲזָקָה.
י וְשָׁמַרְתָּ֛ אֶת־הַחֻקָּ֥ה הַזֹּ֖את לְמוֹעֲדָ֑הּ מִיָּמִ֖ים יָמִֽימָה׃ שְׁמוֹר אֶת הַחוֹק הַזֶּה בִּזְמַנּוֹ בְּכָל שָׁנָה וְשָׁנָה.
פ פָּרָשַׁת וְהָיָה כִּי יְבִאֲךָ
יא וְהָיָ֞ה כִּֽי־יְבִאֲךָ֤ יְהֹוָה֙ אֶל־אֶ֣רֶץ הַֽכְּנַעֲנִ֔י כַּאֲשֶׁ֛ר נִשְׁבַּ֥ע לְךָ֖ וְלַֽאֲבֹתֶ֑יךָ וּנְתָנָ֖הּ לָֽךְ׃ כְּשֶׁה' יָבִיא אוֹתְךָ לְאֶרֶץ הַכְּנַעֲנִי כְּמוֹ שֶׁנִּשְׁבַּע לְךָ וְלַאֲבוֹתֶיךָ, וְיִתֵּן לְךָ אוֹתָהּ.
יב וְהַעֲבַרְתָּ֥ כׇל־פֶּֽטֶר־רֶ֖חֶם לַֽיהֹוָ֑ה וְכׇל־פֶּ֣טֶר ׀ שֶׁ֣גֶר בְּהֵמָ֗ה אֲשֶׁ֨ר יִהְיֶ֥ה לְךָ֛ הַזְּכָרִ֖ים לַיהֹוָֽה׃ תִּתֵּן אֶת כָּל מִי שֶׁיָּצָא רִאשוֹן מֵהָרֶחֶם לַה', וְכָל בְּכוֹר וְלַד בְּהֵמָה זָכָר שֶׁיִּהְיֶה לְךָ – תַּקְדִּישׁ לַה'.
יג וְכׇל־פֶּ֤טֶר חֲמֹר֙ תִּפְדֶּ֣ה בְשֶׂ֔ה וְאִם־לֹ֥א תִפְדֶּ֖ה וַעֲרַפְתּ֑וֹ וְכֹ֨ל בְּכ֥וֹר אָדָ֛ם בְּבָנֶ֖יךָ תִּפְדֶּֽה׃ וְכָל בְּכוֹר חֲמוֹר תִּפְדֶּה בְּשֶׂה וְאִם לֹא תִּפְדֶּה תַּעֲרוֹף אוֹתוֹ, וְכָל בְּכוֹר אָדָם מִבָּנֶיךָ תִּפְדֶּה.

מפטיר[עריכה]

יד וְהָיָ֞ה כִּֽי־יִשְׁאָלְךָ֥ בִנְךָ֛ מָחָ֖ר לֵאמֹ֣ר מַה־זֹּ֑את וְאָמַרְתָּ֣ אֵלָ֔יו בְּחֹ֣זֶק יָ֗ד הוֹצִיאָ֧נוּ יְהֹוָ֛ה מִמִּצְרַ֖יִם מִבֵּ֥ית עֲבָדִֽים׃ וּכְשֶׁבִּנְךָ יִשְׁאַל אוֹתְךָ מָחָר וְיֹאמַר: מַה זֶּה? תֹּאמַר לוֹ: ה' הוֹצִיא אוֹתָנוּ מִמִּצְרַיִם מִבֵּית עֲבָדִים בְּיָד חֲזָקָה.
טו וַיְהִ֗י כִּֽי־הִקְשָׁ֣ה פַרְעֹה֮ לְשַׁלְּחֵ֒נוּ֒ וַיַּהֲרֹ֨ג יְהֹוָ֤ה כׇּל־בְּכוֹר֙ בְּאֶ֣רֶץ מִצְרַ֔יִם מִבְּכֹ֥ר אָדָ֖ם וְעַד־בְּכ֣וֹר בְּהֵמָ֑ה עַל־כֵּן֩ אֲנִ֨י זֹבֵ֜חַ לַֽיהֹוָ֗ה כׇּל־פֶּ֤טֶר רֶ֙חֶם֙ הַזְּכָרִ֔ים וְכׇל־בְּכ֥וֹר בָּנַ֖י אֶפְדֶּֽה׃ וּכְשֶׁפַּרְעֹה הִתְעַקֵּשׁ לֹא לְשַׁחְרֵר אוֹתָנוּ, ה' הָרַג כָּל בְּכוֹר בְּאֶרֶץ מִצְרַיִם מִבְּכוֹר אָדָם וְעַד בְּכוֹר בְּהֵמָה, לָכֵן אֲנִי שׁוֹחֵט לַה' כָּל זָכָר שֶׁיּוֹצֵא רִאשׁוֹן מֵהָרֶחֶם, וַאֲנִי פּוֹדֶה כָּל בְּכוֹר בְּבָנַי.
טז וְהָיָ֤ה לְאוֹת֙ עַל־יָ֣דְכָ֔ה וּלְטוֹטָפֹ֖ת בֵּ֣ין עֵינֶ֑יךָ כִּ֚י בְּחֹ֣זֶק יָ֔ד הוֹצִיאָ֥נוּ יְהֹוָ֖ה מִמִּצְרָֽיִם׃ כְּתוֹב לְךָ אֶת הַדְּבָרִים לְסִימָן עַל יָדְךָ וּלְתַכְשִׁיט בֵּין עֵינֶיךָ שֶׁה' הוֹצִיא אוֹתָנוּ מִמִּצְרַיִם בְּיָד חֲזָקָה.
