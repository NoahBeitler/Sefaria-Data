ראשון[עריכה]

פָּרָשַׁת הַחֲטָאִים
א א אֵ֣לֶּה הַדְּבָרִ֗ים אֲשֶׁ֨ר דִּבֶּ֤ר מֹשֶׁה֙ אֶל־כׇּל־יִשְׂרָאֵ֔ל בְּעֵ֖בֶר הַיַּרְדֵּ֑ן בַּמִּדְבָּ֡ר בָּֽעֲרָבָה֩ מ֨וֹל ס֜וּף בֵּֽין־פָּארָ֧ן וּבֵֽין־תֹּ֛פֶל וְלָבָ֥ן וַחֲצֵרֹ֖ת וְדִ֥י זָהָֽב׃ אֵלֶּה הַדְּבָרִים שֶׁמֹּשֶׁה דִּבֵּר אֶל כָּל יִשְׂרָאֵל בְּעֵבֶר הַיַּרְדֵּן בַּמִּדְבָּר שֶׁבָּעֲרָבָה, מוּל יַם סוּף בֵּין פָּארָן וּבֵין תּוֹפֶל וְלָבָן וַחֲצֵרוֹת וְדִי זָהָב.
ב אַחַ֨ד עָשָׂ֥ר יוֹם֙ מֵֽחֹרֵ֔ב דֶּ֖רֶךְ הַר־שֵׂעִ֑יר עַ֖ד קָדֵ֥שׁ בַּרְנֵֽעַ׃ הַמֶּרְחַק מֵחוֹרֵב לְקָדֵשׁ בַּרְנֵעַ דֶּרֶךְ הַר שֵׂעִיר הוּא אַחַד עָשָׂר יְמֵי הֲלִיכָה.
ג וַֽיְהִי֙ בְּאַרְבָּעִ֣ים שָׁנָ֔ה בְּעַשְׁתֵּֽי־עָשָׂ֥ר חֹ֖דֶשׁ בְּאֶחָ֣ד לַחֹ֑דֶשׁ דִּבֶּ֤ר מֹשֶׁה֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל כְּ֠כֹ֠ל אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֹת֖וֹ אֲלֵהֶֽם׃ אַךְ רַק בַּשָּׁנָה הָאַרְבָּעִים בַּחוֹדֶשׁ הָאַחַד עָשָׂר בַּיּוֹם הָרִאשׁוֹן בַּחוֹדֶשׁ מֹשֶׁה דִּבֵּר אֶל בְּנֵי יִשְׂרָאֵל אֶת כָּל מַה שֶּׁה' צִוָּה אוֹתוֹ לוֹמַר לָהֶם.
ד אַחֲרֵ֣י הַכֹּת֗וֹ אֵ֚ת סִיחֹן֙ מֶ֣לֶךְ הָֽאֱמֹרִ֔י אֲשֶׁ֥ר יוֹשֵׁ֖ב בְּחֶשְׁבּ֑וֹן וְאֵ֗ת ע֚וֹג מֶ֣לֶךְ הַבָּשָׁ֔ן אֲשֶׁר־יוֹשֵׁ֥ב בְּעַשְׁתָּרֹ֖ת בְּאֶדְרֶֽעִי׃ אַחֲרֵי שֶׁנִּצֵּחַ אֶת סִיחוֹן מֶלֶךְ הָאֱמוֹרִי שֶׁיּוֹשֵׁב בְּחֶשְׁבּוֹן וְאֶת עוֹג מֶלֶךְ הַבָּשָׁן שֶׁיּוֹשֵׁב בְּעַשְׁתָּרוֹת שֶׁבְּמַלְכוּת אֶדְרֶעִי.
ה בְּעֵ֥בֶר הַיַּרְדֵּ֖ן בְּאֶ֣רֶץ מוֹאָ֑ב הוֹאִ֣יל מֹשֶׁ֔ה בֵּאֵ֛ר אֶת־הַתּוֹרָ֥ה הַזֹּ֖את לֵאמֹֽר׃ מֹשֶׁה הִתְחִיל לְבָאֵר אֶת הַתּוֹרָה הַזֹּאת בְּעֵבֶר הַיַּרְדֵּן בְּאֶרֶץ מוֹאָב, וְאָמַר:
ו יְהֹוָ֧ה אֱלֹהֵ֛ינוּ דִּבֶּ֥ר אֵלֵ֖ינוּ בְּחֹרֵ֣ב לֵאמֹ֑ר רַב־לָכֶ֥ם שֶׁ֖בֶת בָּהָ֥ר הַזֶּֽה׃ ה' אֱלֹהֵינוּ דִּבֵּר אֵלֵינוּ בְּהַר חוֹרֵב וְאָמַר: מַסְפִּיק לָכֶם לָשֶׁבֶת בָּהָר הַזֶּה.
ז פְּנ֣וּ ׀ וּסְע֣וּ לָכֶ֗ם וּבֹ֨אוּ הַ֥ר הָֽאֱמֹרִי֮ וְאֶל־כׇּל־שְׁכֵנָיו֒ בָּעֲרָבָ֥ה בָהָ֛ר וּבַשְּׁפֵלָ֥ה וּבַנֶּ֖גֶב וּבְח֣וֹף הַיָּ֑ם אֶ֤רֶץ הַֽכְּנַעֲנִי֙ וְהַלְּבָנ֔וֹן עַד־הַנָּהָ֥ר הַגָּדֹ֖ל נְהַר־פְּרָֽת׃ פְּנוּ מִכָּאן וּסְעוּ לָכֶם וּבוֹאוּ לְאֶרֶץ יִשְׂרָאֵל, לְהַר הָאֱמוֹרִי וְאֶל כָּל שְׁכֵנָיו שֶׁבָּעֲרָבָה בָּהָר וּבַשְּׁפֵלָה וּבַנֶּגֶב וּבְחוֹף הַיָּם, אֶרֶץ הַכְּנַעֲנִי וְהַלְּבָנוֹן עַד הַנָּהָר הַגָּדוֹל – נְהַר פְּרָת.
ח רְאֵ֛ה נָתַ֥תִּי לִפְנֵיכֶ֖ם אֶת־הָאָ֑רֶץ בֹּ֚אוּ וּרְשׁ֣וּ אֶת־הָאָ֔רֶץ אֲשֶׁ֣ר נִשְׁבַּ֣ע יְ֠הֹוָ֠ה לַאֲבֹ֨תֵיכֶ֜ם לְאַבְרָהָ֨ם לְיִצְחָ֤ק וּֽלְיַעֲקֹב֙ לָתֵ֣ת לָהֶ֔ם וּלְזַרְעָ֖ם אַחֲרֵיהֶֽם׃ עַם יִשְׂרָאֵל, רְאֵה! נָתַתִּי לִפְנֵיכֶם אֶת הָאָרֶץ, בּוֹאוּ וְכִבְשׁוּ אֶת הָאָרֶץ שֶׁה' נִשְׁבַּע לַאֲבוֹתֵיכֶם – לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לָתֵת לָהֶם, וּלְצֶאֱצָאֵיהֶם אַחֲרֵיהֶם.
ט וָאֹמַ֣ר אֲלֵכֶ֔ם בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר לֹא־אוּכַ֥ל לְבַדִּ֖י שְׂאֵ֥ת אֶתְכֶֽם׃ אָמַרְתִּי לָכֶם בְּאוֹתוֹ זְמָן: לֹא אוּכַל לָשֵׂאת אֶתְכֶם לְבַדִּי.
י יְהֹוָ֥ה אֱלֹהֵיכֶ֖ם הִרְבָּ֣ה אֶתְכֶ֑ם וְהִנְּכֶ֣ם הַיּ֔וֹם כְּכוֹכְבֵ֥י הַשָּׁמַ֖יִם לָרֹֽב׃ ה' אֱלֹהֵיכֶם הִרְבָּה אֶתְכֶם וְהִנֵּה אַתֶּם הַיּוֹם רַבִּים כְּכוֹכְבֵי הַשָּׁמַיִם.

שני[עריכה]

יא יְהֹוָ֞ה אֱלֹהֵ֣י אֲבֽוֹתֵכֶ֗ם יֹסֵ֧ף עֲלֵיכֶ֛ם כָּכֶ֖ם אֶ֣לֶף פְּעָמִ֑ים וִיבָרֵ֣ךְ אֶתְכֶ֔ם כַּאֲשֶׁ֖ר דִּבֶּ֥ר לָכֶֽם׃ ה' אֱלֹהֵי אֲבוֹתֵיכֶם יוֹסִיף עֲלֵיכֶם פִּי אֶלֶף מִמַּה שֶּׁהִנְּכֶם, וְהוּא יְבָרֵךְ אֶתְכֶם כְּמוֹ שֶׁאָמַר לָכֶם.
יב אֵיכָ֥ה אֶשָּׂ֖א לְבַדִּ֑י טׇרְחֲכֶ֥ם וּמַֽשַּׂאֲכֶ֖ם וְרִֽיבְכֶֽם׃ כֵּיצַד אֶשָּׂא וְאֶסְבּוֹל לְבַדִּי אֶת הַטִּרְחָה וְאֶת הַמַּשָּׂא הַכָּבֵד לְהַנְהִיג אֶתְכֶם וּלְהִתְעַסֵּק בִּמְרִיבוֹתֵיכֶם?!
יג הָב֣וּ לָ֠כֶ֠ם אֲנָשִׁ֨ים חֲכָמִ֧ים וּנְבֹנִ֛ים וִידֻעִ֖ים לְשִׁבְטֵיכֶ֑ם וַאֲשִׂימֵ֖ם בְּרָאשֵׁיכֶֽם׃ הֲבִיאוּ לָכֶם אֲנָשִׁים חֲכָמִים וּנְבוֹנִים וִידוּעִים מִכָּל שִׁבְטֵיכֶם וַאֲמַנֶּה אוֹתָם שֶׁיִּהְיוּ מַנְהִיגֵיכֶם.
יד וַֽתַּעֲנ֖וּ אֹתִ֑י וַתֹּ֣אמְר֔וּ טֽוֹב־הַדָּבָ֥ר אֲשֶׁר־דִּבַּ֖רְתָּ לַעֲשֽׂוֹת׃ עֲנִיתֶם לִי וַאֲמַרְתֶּם: טוֹב לַעֲשׂוֹת אֶת הַדָּבָר שֶׁאָמַרְתָּ.
טו וָאֶקַּ֞ח אֶת־רָאשֵׁ֣י שִׁבְטֵיכֶ֗ם אֲנָשִׁ֤ים חֲכָמִים֙ וִֽידֻעִ֔ים וָאֶתֵּ֥ן אוֹתָ֛ם רָאשִׁ֖ים עֲלֵיכֶ֑ם שָׂרֵ֨י אֲלָפִ֜ים וְשָׂרֵ֣י מֵא֗וֹת וְשָׂרֵ֤י חֲמִשִּׁים֙ וְשָׂרֵ֣י עֲשָׂרֹ֔ת וְשֹׁטְרִ֖ים לְשִׁבְטֵיכֶֽם׃ לָקַחְתִּי אֶת רָאשֵׁי שִׁבְטֵיכֶם – אֲנָשִׁים חֲכָמִים וִידוּעִים, וּמִנִּיתִי אוֹתָם לְמַנְהִיגִים עֲלֵיכֶם, מִכָּל שֵׁבֶט – שָׂרִים הַמְּמֻנִּים עַל אֲלָפִים, שָׂרִים הַמְּמֻנִּים עַל מֵאוֹת, שָׂרִים הַמְּמֻנִּים עַל חֲמִשִּׁים וְשָׂרִים הַמְּמֻנִּים עַל עֲשָׂרוֹת, וְשׁוֹטְרִים.
טז וָאֲצַוֶּה֙ אֶת־שֹׁ֣פְטֵיכֶ֔ם בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר שָׁמֹ֤עַ בֵּין־אֲחֵיכֶם֙ וּשְׁפַטְתֶּ֣ם צֶ֔דֶק בֵּֽין־אִ֥ישׁ וּבֵין־אָחִ֖יו וּבֵ֥ין גֵּרֽוֹ׃ צִוִּיתִי אֶת שׁוֹפְטֵיכֶם בְּאוֹתוֹ זְמָן וְאָמַרְתִּי: שִׁמְעוּ אֶת הַתְּלֻנּוֹת שֶׁבֵּין אֲחֵיכֶם וְשִׁפְטוּ בְּצֶדֶק בֵּין אִישׁ לְבֵין אָחִיו וּבֵין בֶּן הַמִּיעוּטִים.
יז לֹֽא־תַכִּ֨ירוּ פָנִ֜ים בַּמִּשְׁפָּ֗ט כַּקָּטֹ֤ן כַּגָּדֹל֙ תִּשְׁמָע֔וּן לֹ֤א תָג֙וּרוּ֙ מִפְּנֵי־אִ֔ישׁ כִּ֥י הַמִּשְׁפָּ֖ט לֵאלֹהִ֣ים ה֑וּא וְהַדָּבָר֙ אֲשֶׁ֣ר יִקְשֶׁ֣ה מִכֶּ֔ם תַּקְרִב֥וּן אֵלַ֖י וּשְׁמַעְתִּֽיו׃ אַל תִּתְּנוּ עֲדִיפוּת לְאַחַד הַצְּדָדִים בַּמִּשְׁפָּט, תִּשְׁמְעוּ לַקָּטָן כְּמוֹ לַגָּדוֹל, אַל תִּפְחֲדוּ מִשּׁוּם אִישׁ, כִּי הַמִּשְׁפָּט הוּא לֵאלֹהִים, וְדָבָר שֶׁיִּהְיֶה קָשֶׁה לָכֶם לְהַכְרִיע תַּגִּישׁוּ אֵלַי וְאֶשְׁמַע אוֹתוֹ.
יח וָאֲצַוֶּ֥ה אֶתְכֶ֖ם בָּעֵ֣ת הַהִ֑וא אֵ֥ת כׇּל־הַדְּבָרִ֖ים אֲשֶׁ֥ר תַּעֲשֽׂוּן׃ וְאָז צִוִּיתִי אֶתְכֶם אֶת כָּל הַדְּבָרִים וְסִדְרֵי הַדִּינִים שֶׁתַּעֲשׂוּ בַּמִּשְׁפָּט.
יט וַנִּסַּ֣ע מֵחֹרֵ֗ב וַנֵּ֡לֶךְ אֵ֣ת כׇּל־הַמִּדְבָּ֣ר הַגָּדוֹל֩ וְהַנּוֹרָ֨א הַה֜וּא אֲשֶׁ֣ר רְאִיתֶ֗ם דֶּ֚רֶךְ הַ֣ר הָֽאֱמֹרִ֔י כַּאֲשֶׁ֥ר צִוָּ֛ה יְהֹוָ֥ה אֱלֹהֵ֖ינוּ אֹתָ֑נוּ וַנָּבֹ֕א עַ֖ד קָדֵ֥שׁ בַּרְנֵֽעַ׃ וְנָסַעְנוּ מֵהַר חוֹרֵב וְהָלַכְנוּ אֶת כָּל הַמִּדְבָּר הַגָּדוֹל וְהַנּוֹרָא הַהוּא שֶׁרְאִיתֶם, בַּדֶּרֶךְ לְהַר הָאֱמוֹרִי, כְּמוֹ שֶׁה' אֱלֹהֵינוּ צִוָּה אוֹתָנוּ, וּבָאנוּ עַד לְקָדֵשׁ בַּרְנֵעַ.
כ וָאֹמַ֖ר אֲלֵכֶ֑ם בָּאתֶם֙ עַד־הַ֣ר הָאֱמֹרִ֔י אֲשֶׁר־יְהֹוָ֥ה אֱלֹהֵ֖ינוּ נֹתֵ֥ן לָֽנוּ׃ אָמַרְתִּי לָכֶם: הִנֵּה בָּאתֶם עַד לְהַר הָאֱמוֹרִי שֶׁה' אֱלֹהֵינוּ נוֹתֵן לָנוּ.
כא רְ֠אֵ֠ה נָתַ֨ן יְהֹוָ֧ה אֱלֹהֶ֛יךָ לְפָנֶ֖יךָ אֶת־הָאָ֑רֶץ עֲלֵ֣ה רֵ֗שׁ כַּאֲשֶׁר֩ דִּבֶּ֨ר יְהֹוָ֜ה אֱלֹהֵ֤י אֲבֹתֶ֙יךָ֙ לָ֔ךְ אַל־תִּירָ֖א וְאַל־תֵּחָֽת׃ רְאֵה, ה' אֱלֹהֶיךָ נָתַן לְפָנֶיךָ אֶת הָאָרֶץ, עֲלֵה וּכְבוֹשׁ כְּמוֹ שֶׁה' אֱלֹהֵי אֲבוֹתֶיךָ דִּבֵּר אֵלֶיךָ, אַל תְּפַחֵד וְאַל תִּשָּׁבֵר.

שלישי[עריכה]

כב וַתִּקְרְב֣וּן אֵלַי֮ כֻּלְּכֶם֒ וַתֹּאמְר֗וּ נִשְׁלְחָ֤ה אֲנָשִׁים֙ לְפָנֵ֔ינוּ וְיַחְפְּרוּ־לָ֖נוּ אֶת־הָאָ֑רֶץ וְיָשִׁ֤בוּ אֹתָ֙נוּ֙ דָּבָ֔ר אֶת־הַדֶּ֙רֶךְ֙ אֲשֶׁ֣ר נַעֲלֶה־בָּ֔הּ וְאֵת֙ הֶֽעָרִ֔ים אֲשֶׁ֥ר נָבֹ֖א אֲלֵיהֶֽן׃ נִגַּשְׁתֶּם אֵלַי כֻּלְּכֶם וַאֲמַרְתֶּם: נִשְׁלַח אֲנָשִׁים לְפָנֵינוּ שֶׁיְּחַפְּשׂוּ וִירַגְּלוּ עֲבוּרֵנוּ בָּאָרֶץ, וְיַחֲזִירוּ לָנוּ תְּשׁוּבוֹת עַל הַדֶּרֶךְ שֶׁנַּעֲלֶה בָּהּ וְעַל הֶעָרִים שֶׁנָּבוֹא אֲלֵיהֶן.
כג וַיִּיטַ֥ב בְּעֵינַ֖י הַדָּבָ֑ר וָאֶקַּ֤ח מִכֶּם֙ שְׁנֵ֣ים עָשָׂ֣ר אֲנָשִׁ֔ים אִ֥ישׁ אֶחָ֖ד לַשָּֽׁבֶט׃ הַדָּבָר נִרְאָה טוֹב בְּעֵינַי, לָקַחְתִּי מִכֶּם שְׁנֵים עָשָׂר אֲנָשִׁים – אִישׁ אֶחָד מִכָּל שֵׁבֶט.
כד וַיִּפְנוּ֙ וַיַּעֲל֣וּ הָהָ֔רָה וַיָּבֹ֖אוּ עַד־נַ֣חַל אֶשְׁכֹּ֑ל וַֽיְרַגְּל֖וּ אֹתָֽהּ׃ הֵם פָּנוּ לָלֶכֶת וְעָלוּ לָהָר, וּבָאוּ עַד נַחַל אֶשְׁכֹּל, וְרִגְּלוּ בָּאָרֶץ.
כה וַיִּקְח֤וּ בְיָדָם֙ מִפְּרִ֣י הָאָ֔רֶץ וַיּוֹרִ֖דוּ אֵלֵ֑ינוּ וַיָּשִׁ֨בוּ אֹתָ֤נוּ דָבָר֙ וַיֹּ֣אמְר֔וּ טוֹבָ֣ה הָאָ֔רֶץ אֲשֶׁר־יְהֹוָ֥ה אֱלֹהֵ֖ינוּ נֹתֵ֥ן לָֽנוּ׃ הֵם לָקְחוּ בְּיָדָם מִפֵּרוֹת הָאָרֶץ וְהוֹרִידוּ אֵלֵינוּ, וְעָנוּ לָנוּ וְאָמְרוּ: הָאָרֶץ שֶׁה' אֱלֹהֵינוּ נוֹתֵן לָנוּ – טוֹבָה.
כו וְלֹ֥א אֲבִיתֶ֖ם לַעֲלֹ֑ת וַתַּמְר֕וּ אֶת־פִּ֥י יְהֹוָ֖ה אֱלֹהֵיכֶֽם׃ וְלֹא רְצִיתֶם לַעֲלוֹת, וַעֲבַרְתֶּם עַל צִוּוּי ה' אֱלֹהֵיכֶם.
כז וַתֵּרָגְנ֤וּ בְאׇהֳלֵיכֶם֙ וַתֹּ֣אמְר֔וּ בְּשִׂנְאַ֤ת יְהֹוָה֙ אֹתָ֔נוּ הוֹצִיאָ֖נוּ מֵאֶ֣רֶץ מִצְרָ֑יִם לָתֵ֥ת אֹתָ֛נוּ בְּיַ֥ד הָאֱמֹרִ֖י לְהַשְׁמִידֵֽנוּ׃ הִתְלוֹנַנְתֶּם בְּאָהֳלֵיכֶם וַאֲמַרְתֶּם: ה' הוֹצִיא אוֹתָנוּ מֵאֶרֶץ מִצְרַיִם מִפְּנֵי שֶׁהוּא שׂוֹנֵא אוֹתָנוּ, כְּדֵי לִמְסוֹר אוֹתָנוּ לִידֵי הָאֱמוֹרִי לְהַשְׁמִיד אוֹתָנוּ.
כח אָנָ֣ה ׀ אֲנַ֣חְנוּ עֹלִ֗ים אַחֵ֩ינוּ֩ הֵמַ֨סּוּ אֶת־לְבָבֵ֜נוּ לֵאמֹ֗ר עַ֣ם גָּד֤וֹל וָרָם֙ מִמֶּ֔נּוּ עָרִ֛ים גְּדֹלֹ֥ת וּבְצוּרֹ֖ת בַּשָּׁמָ֑יִם וְגַם־בְּנֵ֥י עֲנָקִ֖ים רָאִ֥ינוּ שָֽׁם׃ לְאָן אֲנַחְנוּ עוֹלִים?! אַחֵינוּ הֵמִיסוּ אֶת לִבֵּנוּ וְהִפְחִידוּ אוֹתָנוּ בְּאָמְרָם שֶׁרָאוּ עַם גָּדוֹל וְגָבוֹהַּ מֵאִתָּנוּ, עָרִים גְּדוֹלוֹת וּמְבֻצָּרוֹת, גְּבוֹהוֹת "עַד לַשָּׁמַיִם", וְאָמְרוּ: רָאִינוּ שָׁם גַּם בְּנֵי עֲנָקִים!
כט וָאֹמַ֖ר אֲלֵכֶ֑ם לֹא־תַעַרְצ֥וּן וְלֹא־תִֽירְא֖וּן מֵהֶֽם׃ אָמַרְתִּי לָכֶם: אַל תִּשָּׁבְרוּ וְאַל תִּפְחֲדוּ מֵהֶם!
ל יְהֹוָ֤ה אֱלֹֽהֵיכֶם֙ הַהֹלֵ֣ךְ לִפְנֵיכֶ֔ם ה֖וּא יִלָּחֵ֣ם לָכֶ֑ם כְּ֠כֹ֠ל אֲשֶׁ֨ר עָשָׂ֧ה אִתְּכֶ֛ם בְּמִצְרַ֖יִם לְעֵינֵיכֶֽם׃ ה' אֱלֹהֵיכֶם שֶׁהוֹלֵךְ לִפְנֵיכֶם הוּא יִלָּחֵם עֲבוּרְכֶם כְּמוֹ כָּל מַה שֶּׁעָשָׂה אִתְּכֶם בְּמִצְרַיִם לְעֵינֵיכֶם.
לא וּבַמִּדְבָּר֙ אֲשֶׁ֣ר רָאִ֔יתָ אֲשֶׁ֤ר נְשָׂאֲךָ֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ כַּאֲשֶׁ֥ר יִשָּׂא־אִ֖ישׁ אֶת־בְּנ֑וֹ בְּכׇל־הַדֶּ֙רֶךְ֙ אֲשֶׁ֣ר הֲלַכְתֶּ֔ם עַד־בֹּאֲכֶ֖ם עַד־הַמָּק֥וֹם הַזֶּֽה׃ וּכְמוֹ כָּל מַה שֶּׁעָשָׂה בַּמִּדְבָּר כְּמוֹ שֶׁרָאִיתָ, שֶׁה' אֱלֹהֶיךָ נָשָׂא אוֹתְךָ כְּמוֹ שֶׁאִישׁ נוֹשֵׂא אֶת בְּנוֹ, בְּכָל הַדֶּרֶךְ שֶׁהֲלַכְתֶּם עַד שֶׁבָּאתֶם לַמָּקוֹם הַזֶּה.
לב וּבַדָּבָ֖ר הַזֶּ֑ה אֵֽינְכֶם֙ מַאֲמִינִ֔ם בַּיהֹוָ֖ה אֱלֹהֵיכֶֽם׃ וּבַדָּבָר הַזֶּה – שֶׁהוּא יָבִיא אֶתְכֶם לְאֶרֶץ יִשְׂרָאֵל, אֵינְכֶם מַאֲמִינִים בַּה' אֱלֹהֵיכֶם.
לג הַהֹלֵ֨ךְ לִפְנֵיכֶ֜ם בַּדֶּ֗רֶךְ לָת֥וּר לָכֶ֛ם מָק֖וֹם לַחֲנֹֽתְכֶ֑ם בָּאֵ֣שׁ ׀ לַ֗יְלָה לַרְאֹֽתְכֶם֙ בַּדֶּ֙רֶךְ֙ אֲשֶׁ֣ר תֵּֽלְכוּ־בָ֔הּ וּבֶעָנָ֖ן יוֹמָֽם׃ שֶׁהוֹלֵךְ לִפְנֵיכֶם בַּדֶּרֶךְ לְחַפֵּשׂ עֲבוּרְכֶם מָקוֹם לַחֲנִיַּתְכֶם – בַּלַּיְלָה הוֹלֵךְ בְּעַמּוּד אֵשׁ כְּדֵי לְהַרְאוֹת לָכֶם אֶת הַדֶּרֶךְ שֶׁתֵּלְכוּ בָּהּ, וּבַיּוֹם בְּעַמּוּד עָנָן.
לד וַיִּשְׁמַ֥ע יְהֹוָ֖ה אֶת־ק֣וֹל דִּבְרֵיכֶ֑ם וַיִּקְצֹ֖ף וַיִּשָּׁבַ֥ע לֵאמֹֽר׃ ה' שָׁמַע אֶת קוֹל דִּבְרֵיכֶם, וְכָעַס וְנִשְׁבַּע:
לה אִם־יִרְאֶ֥ה אִישׁ֙ בָּאֲנָשִׁ֣ים הָאֵ֔לֶּה הַדּ֥וֹר הָרָ֖ע הַזֶּ֑ה אֵ֚ת הָאָ֣רֶץ הַטּוֹבָ֔ה אֲשֶׁ֣ר נִשְׁבַּ֔עְתִּי לָתֵ֖ת לַאֲבֹתֵיכֶֽם׃ אַף אֶחָד מֵהָאֲנָשִׁים הָאֵלֶּה – הַדּוֹר הָרָע הַזֶּה, לֹא יִרְאֶה אֶת הָאָרֶץ הַטּוֹבָה שֶׁנִּשְׁבַּעְתִּי לָתֵת לַאֲבוֹתֵיכֶם.
לו זֽוּלָתִ֞י כָּלֵ֤ב בֶּן־יְפֻנֶּה֙ ה֣וּא יִרְאֶ֔נָּה וְלֽוֹ־אֶתֵּ֧ן אֶת־הָאָ֛רֶץ אֲשֶׁ֥ר דָּֽרַךְ־בָּ֖הּ וּלְבָנָ֑יו יַ֕עַן אֲשֶׁ֥ר מִלֵּ֖א אַחֲרֵ֥י יְהֹוָֽה׃ חוּץ מִכָּלֵב בֶּן יְפֻנֶּה, הוּא יִרְאֶה אוֹתָהּ וְלוֹ וּלְבָנָיו אֶתֵּן אֶת הָאָרֶץ שֶׁדָּרַךְ בָּהּ – שָׂכָר עַל שֶׁקִּיֵּם אֶת מִצְוַת ה'.
לז גַּם־בִּי֙ הִתְאַנַּ֣ף יְהֹוָ֔ה בִּגְלַלְכֶ֖ם לֵאמֹ֑ר גַּם־אַתָּ֖ה לֹא־תָבֹ֥א שָֽׁם׃ גַּם עָלַי ה' כָּעַס בִּגְלַלְכֶם, וְאָמַר לִי: גַּם אַתָּה לֹא תָּבוֹא לְשָׁם.
לח יְהוֹשֻׁ֤עַ בִּן־נוּן֙ הָעֹמֵ֣ד לְפָנֶ֔יךָ ה֖וּא יָ֣בֹא שָׁ֑מָּה אֹת֣וֹ חַזֵּ֔ק כִּי־ה֖וּא יַנְחִלֶ֥נָּה אֶת־יִשְׂרָאֵֽל׃ יְהוֹשֻׁעַ בִּן נוּן הַתַּלְמִיד וְהַמְשָׁרֵת שֶׁלְּךָ הוּא יָבוֹא לְשָׁם, אוֹתוֹ תְּחַזֵּק וּתְעוֹדֵד כִּי הוּא יַנְחִיל אוֹתָהּ לְיִשְׂרָאֵל.

רביעי[עריכה]

לט וְטַפְּכֶם֩ אֲשֶׁ֨ר אֲמַרְתֶּ֜ם לָבַ֣ז יִהְיֶ֗ה וּ֠בְנֵיכֶ֠ם אֲשֶׁ֨ר לֹא־יָדְע֤וּ הַיּוֹם֙ ט֣וֹב וָרָ֔ע הֵ֖מָּה יָבֹ֣אוּ שָׁ֑מָּה וְלָהֶ֣ם אֶתְּנֶ֔נָּה וְהֵ֖ם יִירָשֽׁוּהָ׃ וְיַלְדֵיכֶם שֶׁאֲמַרְתֶּם שֶׁיִּהְיוּ שְׁבוּיִים, וּבְנֵיכֶם הַקְּטַנִּים שֶׁאֵינָם יוֹדְעִים הַיּוֹם לְהַבְחִין בֵּין טוֹב לְרָע, הֵם שֶׁיָּבוֹאוּ לְשָׁם וְלָהֶם אֶתֵּן אוֹתָהּ וְהֵם יִכְבְּשׁוּ אוֹתָהּ.
מ וְאַתֶּ֖ם פְּנ֣וּ לָכֶ֑ם וּסְע֥וּ הַמִּדְבָּ֖רָה דֶּ֥רֶךְ יַם־סֽוּף׃ וְאַתֶּם פְּנוּ לָכֶם וּסְעוּ לַמִּדְבָּר דֶּרֶךְ יַם סוּף.
מא וַֽתַּעֲנ֣וּ ׀ וַתֹּאמְר֣וּ אֵלַ֗י חָטָ֘אנוּ֮ לַיהֹוָה֒ אֲנַ֤חְנוּ נַעֲלֶה֙ וְנִלְחַ֔מְנוּ כְּכֹ֥ל אֲשֶׁר־צִוָּ֖נוּ יְהֹוָ֣ה אֱלֹהֵ֑ינוּ וַֽתַּחְגְּר֗וּ אִ֚ישׁ אֶת־כְּלֵ֣י מִלְחַמְתּ֔וֹ וַתָּהִ֖ינוּ לַעֲלֹ֥ת הָהָֽרָה׃ עֲנִיתֶם וַאֲמַרְתֶּם לִי: חָטָאנוּ לַה', אֲנַחְנוּ נַעֲלֶה וְנִלָּחֵם כְּכָל מַה שֶּׁה' אֱלֹהֵינוּ צִוָּה אוֹתָנוּ! חֲגַרְתֶּם כָּל אֶחָד אֶת כְּלֵי מִלְחַמְתּוֹ וְהִתְחַלְתֶּם מִיָּד לַעֲלוֹת לָהָר.
מב וַיֹּ֨אמֶר יְהֹוָ֜ה אֵלַ֗י אֱמֹ֤ר לָהֶם֙ לֹ֤א תַֽעֲלוּ֙ וְלֹא־תִלָּ֣חֲמ֔וּ כִּ֥י אֵינֶ֖נִּי בְּקִרְבְּכֶ֑ם וְלֹא֙ תִּנָּ֣גְפ֔וּ לִפְנֵ֖י אֹיְבֵיכֶֽם׃ ה' אָמַר אֵלַי, אֱמוֹר לָהֶם: אַל תַּעֲלוּ וְאַל תִּלָּחֲמוּ כִּי אֵינֶנִּי אִתְּכֶם, וְלֹא תַּפְסִידוּ בַקְּרָב לִפְנֵי אוֹיְבֵיכֶם.
מג וָאֲדַבֵּ֥ר אֲלֵיכֶ֖ם וְלֹ֣א שְׁמַעְתֶּ֑ם וַתַּמְרוּ֙ אֶת־פִּ֣י יְהֹוָ֔ה וַתָּזִ֖דוּ וַתַּעֲל֥וּ הָהָֽרָה׃ דִּבַּרְתִּי אֲלֵיכֶם וְלֹא שְׁמַעְתֶּם, סֵרַבְתֶּם לְצִוּוּי ה', וְהִרְשַׁעְתֶּם וַעֲלִיתֶם לָהָר.
מד וַיֵּצֵ֨א הָאֱמֹרִ֜י הַיֹּשֵׁ֨ב בָּהָ֤ר הַהוּא֙ לִקְרַאתְכֶ֔ם וַיִּרְדְּפ֣וּ אֶתְכֶ֔ם כַּאֲשֶׁ֥ר תַּעֲשֶׂ֖ינָה הַדְּבֹרִ֑ים וַֽיַּכְּת֥וּ אֶתְכֶ֛ם בְּשֵׂעִ֖יר עַד־חׇרְמָֽה׃ הָאֱמוֹרִי שֶׁגָּר בָּהָר הַהוּא יָצָא לִקְרַאתְכֶם, הֵם רָדְפוּ אֶתְכֶם כְּמוֹ שֶׁעוֹשׂוֹת הַדְּבוֹרִים, וְכַתְּשׁוּ אֶתְכֶם בְּשֵׂעִיר וְרָדְפוּ אַחֲרֵיכֶם עַד הַשְׁמָדָה.
מה וַתָּשֻׁ֥בוּ וַתִּבְכּ֖וּ לִפְנֵ֣י יְהֹוָ֑ה וְלֹֽא־שָׁמַ֤ע יְהֹוָה֙ בְּקֹ֣לְכֶ֔ם וְלֹ֥א הֶאֱזִ֖ין אֲלֵיכֶֽם׃ חֲזַרְתֶּם וּבְכִיתֶם לִפְנֵי ה', וְה' לֹא שָׁמַע בְּקוֹלְכֶם, וְלֹא הֶאֱזִין אֲלֵיכֶם.
מו וַתֵּשְׁב֥וּ בְקָדֵ֖שׁ יָמִ֣ים רַבִּ֑ים כַּיָּמִ֖ים אֲשֶׁ֥ר יְשַׁבְתֶּֽם׃ יְשַׁבְתֶּם בְּקָדֵשׁ יָמִים רַבִּים – כְּמִסְפַּר הַיָּמִים שֶׁיָּדוּעַ לָכֶם שֶׁיְּשַׁבְתֶּם.
ב א וַנֵּ֜פֶן וַנִּסַּ֤ע הַמִּדְבָּ֙רָה֙ דֶּ֣רֶךְ יַם־ס֔וּף כַּאֲשֶׁ֛ר דִּבֶּ֥ר יְהֹוָ֖ה אֵלָ֑י וַנָּ֥סׇב אֶת־הַר־שֵׂעִ֖יר יָמִ֥ים רַבִּֽים׃ פָּנִינוּ וְנָסַעְנוּ לַמִּדְבָּר דֶּרֶךְ יַם סוּף כְּמוֹ שֶׁה' דִּבֵּר אֵלַי, וְהִקַּפְנוּ אֶת הַר שֵׂעִיר יָמִים רַבִּים.

ב וַיֹּ֥אמֶר יְהֹוָ֖ה אֵלַ֥י לֵאמֹֽר׃ ה' אָמַר לִי:
ג רַב־לָכֶ֕ם סֹ֖ב אֶת־הָהָ֣ר הַזֶּ֑ה פְּנ֥וּ לָכֶ֖ם צָפֹֽנָה׃ מַסְפִּיק הִקַּפְתֶּם אֶת הָהָר הַזֶּה, פְּנוּ לָכֶם צָפוֹנָה.
ד וְאֶת־הָעָם֮ צַ֣ו לֵאמֹר֒ אַתֶּ֣ם עֹֽבְרִ֗ים בִּגְבוּל֙ אֲחֵיכֶ֣ם בְּנֵי־עֵשָׂ֔ו הַיֹּשְׁבִ֖ים בְּשֵׂעִ֑יר וְיִֽירְא֣וּ מִכֶּ֔ם וְנִשְׁמַרְתֶּ֖ם מְאֹֽד׃ וְצַוֵּה אֶת הָעָם: אַתֶּם עוֹבְרִים בִּגְבוּל אֲחֵיכֶם בְּנֵי עֵשָׂו שֶׁגָּרִים בְּשֵׂעִיר, הֵם יִפְחֲדוּ מִכֶּם וְלֹא יִלָּחֲמוּ בָּכֶם, הִזָּהֲרוּ מְאוֹד –
ה אַל־תִּתְגָּר֣וּ בָ֔ם כִּ֠י לֹֽא־אֶתֵּ֤ן לָכֶם֙ מֵֽאַרְצָ֔ם עַ֖ד מִדְרַ֣ךְ כַּף־רָ֑גֶל כִּֽי־יְרֻשָּׁ֣ה לְעֵשָׂ֔ו נָתַ֖תִּי אֶת־הַ֥ר שֵׂעִֽיר׃ אַל תִּתְגָּרוּ בָּהֶם, כִּי לֹא אֶתֵּן לָכֶם מֵאַרְצָם אֲפִלּוּ מָקוֹם לִדְרִיכַת כַּף רֶגֶל, כִּי אֶת הַר שֵׂעִיר נָתַתִּי יְרֻשָּׁה לְעֵשָׂו.
ו אֹ֣כֶל תִּשְׁבְּר֧וּ מֵֽאִתָּ֛ם בַּכֶּ֖סֶף וַאֲכַלְתֶּ֑ם וְגַם־מַ֜יִם תִּכְר֧וּ מֵאִתָּ֛ם בַּכֶּ֖סֶף וּשְׁתִיתֶֽם׃ קְנוּ מֵאִתָּם אוֹכֶל בְּכֶסֶף וְתֹאכְלוּ, וְגַם מַיִם תִּקְנוּ מֵאִתָּם בְּכֶסֶף וְתִשְׁתּוּ, וְאַל תִּגְזְלוּ מֵהֶם.
ז כִּי֩ יְהֹוָ֨ה אֱלֹהֶ֜יךָ בֵּֽרַכְךָ֗ בְּכֹל֙ מַעֲשֵׂ֣ה יָדֶ֔ךָ יָדַ֣ע לֶכְתְּךָ֔ אֶת־הַמִּדְבָּ֥ר הַגָּדֹ֖ל הַזֶּ֑ה זֶ֣ה ׀ אַרְבָּעִ֣ים שָׁנָ֗ה יְהֹוָ֤ה אֱלֹהֶ֙יךָ֙ עִמָּ֔ךְ לֹ֥א חָסַ֖רְתָּ דָּבָֽר׃ אַל תַּחְשְׁשׁוּ לְשַׁלֵּם, כִּי ה' אֱלֹהֶיךָ בֵּרַךְ אוֹתְךָ בְּכָל מַעֲשֵׂה יָדֶךָ, הוּא יָדַע תָּמִיד אֶת צְרָכֶיךָ בִּזְמַן הֲלִיכָתְךָ בַּמִּדְבָּר הַגָּדוֹל הַזֶּה, זֶה אַרְבָּעִים שָׁנָה שֶׁה' אֱלֹהֶיךָ אִתְּךָ וְלֹא חָסַר לְךָ דָּבָר.
ח וַֽנַּעֲבֹ֞ר מֵאֵ֧ת אַחֵ֣ינוּ בְנֵי־עֵשָׂ֗ו הַיֹּֽשְׁבִים֙ בְּשֵׂעִ֔יר מִדֶּ֙רֶךְ֙ הָֽעֲרָבָ֔ה מֵאֵילַ֖ת וּמֵעֶצְיֹ֣ן גָּ֑בֶר

ט וַיֹּ֨אמֶר יְהֹוָ֜ה אֵלַ֗י אַל־תָּ֙צַר֙ אֶת־מוֹאָ֔ב וְאַל־תִּתְגָּ֥ר בָּ֖ם מִלְחָמָ֑ה כִּ֠י לֹֽא־אֶתֵּ֨ן לְךָ֤ מֵֽאַרְצוֹ֙ יְרֻשָּׁ֔ה כִּ֣י לִבְנֵי־ל֔וֹט נָתַ֥תִּי אֶת־עָ֖ר יְרֻשָּֽׁה׃ ה' אָמַר לִי: אַל תַּעֲשֶׂה מָצוֹר עַל מוֹאָב וְאַל תִּתְגָּרֶה בָּהֶם בְּמִלְחָמָה, כִּי לֹא אֶתֵּן לְךָ מֵאַרְצוֹ יְרֻשָּׁה, כִּי לִבְנֵי לוֹט נָתַתִּי אֶת עָר לִירֻשָּׁה.
י הָאֵמִ֥ים לְפָנִ֖ים יָ֣שְׁבוּ בָ֑הּ עַ֣ם גָּד֥וֹל וְרַ֛ב וָרָ֖ם כָּעֲנָקִֽים׃ עַם הָאֵמְתָּנִיִּים יָשְׁבוּ בָּהּ פַּעַם – עַם חָזָק וְגָדוֹל וְגָבוֹהַּ כָּעֲנָקִים.
יא רְפָאִ֛ים יֵחָשְׁב֥וּ אַף־הֵ֖ם כָּעֲנָקִ֑ים וְהַמֹּ֣אָבִ֔ים יִקְרְא֥וּ לָהֶ֖ם אֵמִֽים׃ הֵם – הָרְפָאִים, נֶחְשְׁבוּ גַּם הֵם כָּעֲנָקִים, וְהַמּוֹאָבִים קוֹרְאִים לָהֶם "הָאֵמְתָּנִיִּים".
יב וּבְשֵׂעִ֞יר יָשְׁב֣וּ הַחֹרִים֮ לְפָנִים֒ וּבְנֵ֧י עֵשָׂ֣ו יִֽירָשׁ֗וּם וַיַּשְׁמִידוּם֙ מִפְּנֵיהֶ֔ם וַיֵּשְׁב֖וּ תַּחְתָּ֑ם כַּאֲשֶׁ֧ר עָשָׂ֣ה יִשְׂרָאֵ֗ל לְאֶ֙רֶץ֙ יְרֻשָּׁת֔וֹ אֲשֶׁר־נָתַ֥ן יְהֹוָ֖ה לָהֶֽם׃ וּבְשֵׂעִיר יָשְׁבוּ פַּעַם הַחוֹרִים, וּבְנֵי עֵשָׂו כָּבְשׁוּ אוֹתָם וְהִשְׁמִידוּ אוֹתָם מִפְּנֵיהֶם וְיָשְׁבוּ בִּמְקוֹמָם, כְּמוֹ שֶׁיִּשְׂרָאֵל עָשָׂה לְאֶרֶץ יְרֻשָּׁתוֹ שֶׁה' נָתַן לָהֶם.
יג עַתָּ֗ה קֻ֛מוּ וְעִבְר֥וּ לָכֶ֖ם אֶת־נַ֣חַל זָ֑רֶד וַֽנַּעֲבֹ֖ר אֶת־נַ֥חַל זָֽרֶד׃ עַכְשָׁיו, קוּמוּ וְעִבְרוּ לָכֶם אֶת נַחַל זֶרֶד, וְעָבַרְנוּ אֶת נַחַל זֶרֶד.
יד וְהַיָּמִ֞ים אֲשֶׁר־הָלַ֣כְנוּ ׀ מִקָּדֵ֣שׁ בַּרְנֵ֗עַ עַ֤ד אֲשֶׁר־עָבַ֙רְנוּ֙ אֶת־נַ֣חַל זֶ֔רֶד שְׁלֹשִׁ֥ים וּשְׁמֹנֶ֖ה שָׁנָ֑ה עַד־תֹּ֨ם כׇּל־הַדּ֜וֹר אַנְשֵׁ֤י הַמִּלְחָמָה֙ מִקֶּ֣רֶב הַֽמַּחֲנֶ֔ה כַּאֲשֶׁ֛ר נִשְׁבַּ֥ע יְהֹוָ֖ה לָהֶֽם׃ וְהַזְּמָן שֶׁהָלַכְנוּ מִקָּדֵשׁ בַּרְנֵעַ עַד שֶׁעָבַרְנוּ אֶת נַחַל זֶרֶד הָיָה שְׁלוֹשִׁים וּשְׁמוֹנֶה שָׁנִים, עַד שֶׁנִּגְמְרוּ כָּל דּוֹר הָאֲנָשִׁים הַלּוֹחֲמִים מִתּוֹךְ הַמַּחֲנֶה, כְּמוֹ שֶׁה' נִשְׁבַּע לָהֶם.
טו וְגַ֤ם יַד־יְהֹוָה֙ הָ֣יְתָה בָּ֔ם לְהֻמָּ֖ם מִקֶּ֣רֶב הַֽמַּחֲנֶ֑ה עַ֖ד תֻּמָּֽם׃ וְגַם מַכַּת ה' הָיְתָה בָּהֶם כְּדֵי לְהַשְׁמִיד אוֹתָם מִתּוֹךְ הַמַּחֲנֶה, עַד שֶׁנִּגְמְרוּ.
טז וַיְהִ֨י כַאֲשֶׁר־תַּ֜מּוּ כׇּל־אַנְשֵׁ֧י הַמִּלְחָמָ֛ה לָמ֖וּת מִקֶּ֥רֶב הָעָֽם׃ כְּשֶׁנִּגְמְרוּ כָּל אַנְשֵׁי הַמִּלְחָמָה מִתּוֹךְ הָעָם וְהֵם מֵתוּ –

יז וַיְדַבֵּ֥ר יְהֹוָ֖ה אֵלַ֥י לֵאמֹֽר׃ ה' דִּבֵּר אֵלַי וְאָמַר:
יח אַתָּ֨ה עֹבֵ֥ר הַיּ֛וֹם אֶת־גְּב֥וּל מוֹאָ֖ב אֶת־עָֽר׃ אַתָּה עוֹבֵר הַיּוֹם אֶת גְּבוּל מוֹאָב אֶת עָר.
יט וְקָרַבְתָּ֗ מ֚וּל בְּנֵ֣י עַמּ֔וֹן אַל־תְּצֻרֵ֖ם וְאַל־תִּתְגָּ֣ר בָּ֑ם כִּ֣י לֹֽא־אֶ֠תֵּ֠ן מֵאֶ֨רֶץ בְּנֵי־עַמּ֤וֹן לְךָ֙ יְרֻשָּׁ֔ה כִּ֥י לִבְנֵי־ל֖וֹט נְתַתִּ֥יהָ יְרֻשָּֽׁה׃ תִּתְקָרֵב אֶל מוּל בְּנֵי עַמּוֹן, אַל תַּעֲשֶׂה עֲלֵיהֶם מָצוֹר וְאַל תִּתְגָּרֶה בָּהֶם, כִּי לֹא אֶתֵּן לְךָ יְרֻשָּׁה מֵאֶרֶץ בְּנֵי עַמּוֹן, כִּי לִבְנֵי לוֹט נָתַתִּי אוֹתָהּ לִירֻשָּׁה.
כ אֶֽרֶץ־רְפָאִ֥ים תֵּחָשֵׁ֖ב אַף־הִ֑וא רְפָאִ֤ים יָֽשְׁבוּ־בָהּ֙ לְפָנִ֔ים וְהָֽעַמֹּנִ֔ים יִקְרְא֥וּ לָהֶ֖ם זַמְזֻמִּֽים׃ גַּם הִיא נֶחְשֶׁבֶת אֶרֶץ רְפָאִים, הָרְפָאִים יָשְׁבוּ בָּהּ פַּעַם וְהָעַמּוֹנִים קָרְאוּ לָהֶם 'זַמְזֻמִּים'.
כא עַ֣ם גָּד֥וֹל וְרַ֛ב וָרָ֖ם כָּעֲנָקִ֑ים וַיַּשְׁמִידֵ֤ם יְהֹוָה֙ מִפְּנֵיהֶ֔ם וַיִּירָשֻׁ֖ם וַיֵּשְׁב֥וּ תַחְתָּֽם׃ הֵם הָיוּ עַם חָזָק וְגָדוֹל וְגָבוֹהַּ כָּעֲנָקִים, ה' הִשְׁמִיד אוֹתָם מִפְּנֵיהֶם, וְהָעַמּוֹנִים כָּבְשׁוּ אוֹתָם וְהִתְיַשְּׁבוּ בִּמְקוֹמָם,
כב כַּאֲשֶׁ֤ר עָשָׂה֙ לִבְנֵ֣י עֵשָׂ֔ו הַיֹּשְׁבִ֖ים בְּשֵׂעִ֑יר אֲשֶׁ֨ר הִשְׁמִ֤יד אֶת־הַחֹרִי֙ מִפְּנֵיהֶ֔ם וַיִּֽירָשֻׁם֙ וַיֵּשְׁב֣וּ תַחְתָּ֔ם עַ֖ד הַיּ֥וֹם הַזֶּֽה׃ כְּמוֹ שֶׁה' עָשָׂה לִבְנֵי עֵשָׂו שֶׁיּוֹשְׁבִים בְּשֵׂעִיר – שֶׁהִשְׁמִיד אֶת הַחוֹרִי מִפְּנֵיהֶם, הֵם כָּבְשׁוּ אוֹתָם, וְהִתְיַשְּׁבוּ בִּמְקוֹמָם עַד הַיּוֹם.
כג וְהָֽעַוִּ֛ים הַיֹּשְׁבִ֥ים בַּחֲצֵרִ֖ים עַד־עַזָּ֑ה כַּפְתֹּרִים֙ הַיֹּצְאִ֣ים מִכַּפְתֹּ֔ר הִשְׁמִידֻ֖ם וַיֵּשְׁב֥וּ תַחְתָּֽם׃ וְהָעַוִּים שֶׁיָּשְׁבוּ בְּרָפִיחַ עַד עַזָּה, כַּפְתּוֹרִים שֶׁיָּצְאוּ מִכְּרֵתִים הִשְׁמִידוּ אוֹתָם וְהִתְיַשְּׁבוּ בִּמְקוֹמָם.
פָּרָשַׁת סִיחוֹן
כד ק֣וּמוּ סְּע֗וּ וְעִבְרוּ֮ אֶת־נַ֣חַל אַרְנֹן֒ רְאֵ֣ה נָתַ֣תִּי בְ֠יָדְךָ֠ אֶת־סִיחֹ֨ן מֶֽלֶךְ־חֶשְׁבּ֧וֹן הָֽאֱמֹרִ֛י וְאֶת־אַרְצ֖וֹ הָחֵ֣ל רָ֑שׁ וְהִתְגָּ֥ר בּ֖וֹ מִלְחָמָֽה׃ קוּמוּ סְעוּ וְעִבְרוּ אֶת נַחַל אַרְנוֹן רְאֵה נָתַתִּי בְּיָדְךָ אֶת סִיחוֹן מֶלֶךְ חֶשְׁבּוֹן הָאֱמוֹרִי וְאֶת אַרְצוֹ, הַתְחֵל לִכְבּוֹשׁ, וְתִתְגָּרֶה בּוֹ כְּדֵי לַעֲשׂוֹת מִלְחָמָה.
כה הַיּ֣וֹם הַזֶּ֗ה אָחֵל֙ תֵּ֤ת פַּחְדְּךָ֙ וְיִרְאָ֣תְךָ֔ עַל־פְּנֵי֙ הָֽעַמִּ֔ים תַּ֖חַת כׇּל־הַשָּׁמָ֑יִם אֲשֶׁ֤ר יִשְׁמְעוּן֙ שִׁמְעֲךָ֔ וְרָגְז֥וּ וְחָל֖וּ מִפָּנֶֽיךָ׃ הַיּוֹם הַזֶּה אַתְחִיל לְהַפִּיל אֶת פַּחְדְּךָ וְאֶת יִרְאָתְךָ עַל כָּל הָעַמִּים שֶׁתַּחַת הַשָּׁמַיִם, שֶׁיִּשְׁמְעוּ עָלֶיךָ וְיִפְחֲדוּ וְיִרְעֲדוּ מִמְּךָ.
כו וָאֶשְׁלַ֤ח מַלְאָכִים֙ מִמִּדְבַּ֣ר קְדֵמ֔וֹת אֶל־סִיח֖וֹן מֶ֣לֶךְ חֶשְׁבּ֑וֹן דִּבְרֵ֥י שָׁל֖וֹם לֵאמֹֽר׃ שָׁלַחְתִּי שְׁלִיחִים מִמִּדְבַּר קְדֵמוֹת אֶל סִיחוֹן מֶלֶךְ חֶשְׁבּוֹן, וְאָמַרְתִּי דִּבְרֵי שָׁלוֹם:
כז אֶעְבְּרָ֣ה בְאַרְצֶ֔ךָ בַּדֶּ֥רֶךְ בַּדֶּ֖רֶךְ אֵלֵ֑ךְ לֹ֥א אָס֖וּר יָמִ֥ין וּשְׂמֹֽאול׃ אֶעֱבוֹר בְּאַרְצְךָ, אֵלֵךְ רַק בַּדֶּרֶךְ הָרָאשִׁית – לֹא אֶסְטֶה לְיָמִין אוֹ לִשְׂמֹאל.
כח אֹ֣כֶל בַּכֶּ֤סֶף תַּשְׁבִּרֵ֙נִי֙ וְאָכַ֔לְתִּי וּמַ֛יִם בַּכֶּ֥סֶף תִּתֶּן־לִ֖י וְשָׁתִ֑יתִי רַ֖ק אֶעְבְּרָ֥ה בְרַגְלָֽי׃ תִּמְכּוֹר לִי אוֹכֶל בְּכֶסֶף וְאוֹכַל, וְתִתֶּן לִי מַיִם תְּמוּרַת כֶּסֶף וְאֶשְׁתֶּה, רַק אֶעֱבוֹר בְּרַגְלַי,
כט כַּאֲשֶׁ֨ר עָֽשׂוּ־לִ֜י בְּנֵ֣י עֵשָׂ֗ו הַיֹּֽשְׁבִים֙ בְּשֵׂעִ֔יר וְהַמּ֣וֹאָבִ֔ים הַיֹּשְׁבִ֖ים בְּעָ֑ר עַ֤ד אֲשֶֽׁר־אֶֽעֱבֹר֙ אֶת־הַיַּרְדֵּ֔ן אֶל־הָאָ֕רֶץ אֲשֶׁר־יְהֹוָ֥ה אֱלֹהֵ֖ינוּ נֹתֵ֥ן לָֽנוּ׃ – כְּמוֹ שֶׁעָשׂוּ לִי בְּנֵי עֵשָׂו שֶׁגָּרִים בְּשֵׂעִיר וְהַמּוֹאָבִים שֶׁגָּרִים בְּעָר, עַד שֶׁאֶעֱבוֹר אֶת הַיַּרְדֵּן אֶל הָאָרֶץ שֶׁה' אֱלֹהֵינוּ נוֹתֵן לָנוּ.
ל וְלֹ֣א אָבָ֗ה סִיחֹן֙ מֶ֣לֶךְ חֶשְׁבּ֔וֹן הַעֲבִרֵ֖נוּ בּ֑וֹ כִּֽי־הִקְשָׁה֩ יְהֹוָ֨ה אֱלֹהֶ֜יךָ אֶת־רוּח֗וֹ וְאִמֵּץ֙ אֶת־לְבָב֔וֹ לְמַ֛עַן תִּתּ֥וֹ בְיָדְךָ֖ כַּיּ֥וֹם הַזֶּֽה׃ וְסִיחוֹן מֶלֶךְ חֶשְׁבּוֹן לֹא רָצָה לְהַעֲבִיר אוֹתָנוּ בְּאַרְצוֹ כִּי ה' אֱלֹהֶיךָ הִקְשִׁיחַ אוֹתוֹ וְנָתַן לוֹ אוֹמֶץ בְּלִבּוֹ כְּדֵי לָתֵת אוֹתוֹ בְּיָדְךָ כְּמוֹ שֶׁקָּרָה.

לא וַיֹּ֤אמֶר יְהֹוָה֙ אֵלַ֔י רְאֵ֗ה הַֽחִלֹּ֙תִי֙ תֵּ֣ת לְפָנֶ֔יךָ אֶת־סִיחֹ֖ן וְאֶת־אַרְצ֑וֹ הָחֵ֣ל רָ֔שׁ לָרֶ֖שֶׁת אֶת־אַרְצֽוֹ׃ ה' אָמַר לִי: רְאֵה הִתְחַלְתִּי לָתֵת לְפָנֶיךָ אֶת סִיחוֹן וְאֶת אַרְצוֹ, הַתְחֵל לִכְבּוֹשׁ, כְּדֵי לִכְבּוֹשׁ אֶת אַרְצוֹ.
לב וַיֵּצֵא֩ סִיחֹ֨ן לִקְרָאתֵ֜נוּ ה֧וּא וְכׇל־עַמּ֛וֹ לַמִּלְחָמָ֖ה יָֽהְצָה׃ סִיחוֹן יָצָא עִם כָּל עַמּוֹ לִקְרָאתֵנוּ לְיַהַץ לַמִּלְחָמָה.
לג וַֽיִּתְּנֵ֛הוּ יְהֹוָ֥ה אֱלֹהֵ֖ינוּ לְפָנֵ֑ינוּ וַנַּ֥ךְ אֹת֛וֹ וְאֶת־בָּנָ֖ו וְאֶת־כׇּל־עַמּֽוֹ׃ ה' אֱלֹהֵינוּ נָתַָן אוֹתוֹ בְּיָדֵנוּ, וְנִצַּחְנוּ אוֹתוֹ וְאֶת בָּנָיו וְאֶת כָּל עַמּוֹ.
לד וַנִּלְכֹּ֤ד אֶת־כׇּל־עָרָיו֙ בָּעֵ֣ת הַהִ֔וא וַֽנַּחֲרֵם֙ אֶת־כׇּל־עִ֣יר מְתִ֔ם וְהַנָּשִׁ֖ים וְהַטָּ֑ף לֹ֥א הִשְׁאַ֖רְנוּ שָׂרִֽיד׃ לָכַדְנוּ אָז אֶת כָּל עָרָיו וְהִשְׁמַדְנוּ אֶת כָּל תּוֹשְׁבֵי עָרָיו וְהַנָּשִׁים וְהַיְּלָדִים, לֹא הִשְׁאַרְנוּ שָׂרִיד.
לה רַ֥ק הַבְּהֵמָ֖ה בָּזַ֣זְנוּ לָ֑נוּ וּשְׁלַ֥ל הֶעָרִ֖ים אֲשֶׁ֥ר לָכָֽדְנוּ׃ רַק אֶת הַבְּהֵמָה לָקַחְנוּ לָנוּ לְשָׁלָל וְאֶת שְׁלַל הֶעָרִים שֶׁלָּכָדְנוּ.
לו מֵֽעֲרֹעֵ֡ר אֲשֶׁר֩ עַל־שְׂפַת־נַ֨חַל אַרְנֹ֜ן וְהָעִ֨יר אֲשֶׁ֤ר בַּנַּ֙חַל֙ וְעַד־הַגִּלְעָ֔ד לֹ֤א הָֽיְתָה֙ קִרְיָ֔ה אֲשֶׁ֥ר שָׂגְבָ֖ה מִמֶּ֑נּוּ אֶת־הַכֹּ֕ל נָתַ֛ן יְהֹוָ֥ה אֱלֹהֵ֖ינוּ לְפָנֵֽינוּ׃ מֵעֲרוֹעֵר שֶׁעַל שְׂפַת נַחַל אַרְנוֹן וְהָעִיר שֶׁבַּנַּחַל וְעַד הַגִּלְעָד, לֹא הָיְתָה עִיר שֶׁהָיְתָה חֲזָקָה מֵאִתָּנוּ, אֶת הַכֹּל ה' אֱלֹהֵינוּ נָתַן לְפָנֵינוּ.
לז רַ֛ק אֶל־אֶ֥רֶץ בְּנֵי־עַמּ֖וֹן לֹ֣א קָרָ֑בְתָּ כׇּל־יַ֞ד נַ֤חַל יַבֹּק֙ וְעָרֵ֣י הָהָ֔ר וְכֹ֥ל אֲשֶׁר־צִוָּ֖ה יְהֹוָ֥ה אֱלֹהֵֽינוּ׃ רַק אֶל אֶרֶץ בְּנֵי עַמּוֹן לֹא נִגַּשְׁתָּ – כָּל הָאֱזוֹר עַל יַד נַחַל יַבּוֹק וְעָרֵי הָהָר, וְכָל מַה שֶּׁה' אֱלֹהֵינוּ צִוָּה שֶׁלֹּא נִלָּחֵם בָּהֶם.
פָּרָשַׁת עוֹג
ג א וַנֵּ֣פֶן וַנַּ֔עַל דֶּ֖רֶךְ הַבָּשָׁ֑ן וַיֵּצֵ֣א עוֹג֩ מֶֽלֶךְ־הַבָּשָׁ֨ן לִקְרָאתֵ֜נוּ ה֧וּא וְכׇל־עַמּ֛וֹ לַמִּלְחָמָ֖ה אֶדְרֶֽעִי׃ פָּנִינוּ וְעָלִינוּ צָפוֹנָה דֶּרֶךְ הַבָּשָׁן, עוֹג מֶלֶךְ הַבָּשָׁן יָצָא לִקְרָאתֵנוּ לְאֶדְרֶעִי לַמִּלְחָמָה הוּא וְכָל עַמּוֹ.
ב וַיֹּ֨אמֶר יְהֹוָ֤ה אֵלַי֙ אַל־תִּירָ֣א אֹת֔וֹ כִּ֣י בְיָדְךָ֞ נָתַ֧תִּי אֹת֛וֹ וְאֶת־כׇּל־עַמּ֖וֹ וְאֶת־אַרְצ֑וֹ וְעָשִׂ֣יתָ לּ֔וֹ כַּאֲשֶׁ֣ר עָשִׂ֗יתָ לְסִיחֹן֙ מֶ֣לֶךְ הָֽאֱמֹרִ֔י אֲשֶׁ֥ר יוֹשֵׁ֖ב בְּחֶשְׁבּֽוֹן׃ ה' אָמַר לִי: אַל תִּפְחַד מִמֶּנּוּ, כִּי נָתַתִּי בְּיָדְךָ אוֹתוֹ וְאֶת כָּל עַמּוֹ וְאֶת אַרְצוֹ, עֲשֵׁה לוֹ כְּמוֹ שֶׁעָשִׂיתָ לְסִיחוֹן מֶלֶךְ הָאֱמוֹרִי שֶׁגָּר בְּחֶשְׁבּוֹן.
ג וַיִּתֵּן֩ יְהֹוָ֨ה אֱלֹהֵ֜ינוּ בְּיָדֵ֗נוּ גַּ֛ם אֶת־ע֥וֹג מֶֽלֶךְ־הַבָּשָׁ֖ן וְאֶת־כׇּל־עַמּ֑וֹ וַנַּכֵּ֕הוּ עַד־בִּלְתִּ֥י הִשְׁאִֽיר־ל֖וֹ שָׂרִֽיד׃ ה' אֱלֹהֵינוּ נָתַן בְּיָדֵנוּ גַּם אֶת עוֹג מֶלֶךְ הַבָּשָׁן וְאֶת כָּל עַמּוֹ, הִכִּינוּ אוֹתוֹ עַד שֶׁלֹּא נִשְׁאַר מִמֶּנּוּ שָׂרִיד.
ד וַנִּלְכֹּ֤ד אֶת־כׇּל־עָרָיו֙ בָּעֵ֣ת הַהִ֔וא לֹ֤א הָֽיְתָה֙ קִרְיָ֔ה אֲשֶׁ֥ר לֹא־לָקַ֖חְנוּ מֵֽאִתָּ֑ם שִׁשִּׁ֥ים עִיר֙ כׇּל־חֶ֣בֶל אַרְגֹּ֔ב מַמְלֶ֥כֶת ע֖וֹג בַּבָּשָֽׁן׃ לָכַדְנוּ אָז אֶת כָּל עָרָיו, לֹא הָיְתָה עִיר שֶׁלֹּא לָקַחְנוּ מֵאִתָּם, שִׁשִּׁים עִיר הָיוּ בְּכָל חֶבֶל אַרְגּוֹב, שֶׁהוּא מַמְלֶכֶת עוֹג בַּבָּשָׁן.
ה כׇּל־אֵ֜לֶּה עָרִ֧ים בְּצֻרֹ֛ת חוֹמָ֥ה גְבֹהָ֖ה דְּלָתַ֣יִם וּבְרִ֑יחַ לְבַ֛ד מֵעָרֵ֥י הַפְּרָזִ֖י הַרְבֵּ֥ה מְאֹֽד׃ כָּל אֵלֶּה הָיוּ עָרִים מְבֻצָּרוֹת עִם חוֹמָה גְּבוֹהָה דְּלָתוֹת וּבָרִיחַ, וּבְנוֹסָף הָיוּ גַּם הַרְבֵּה מְאוֹד עָרִים לְלֹא חוֹמָה.
ו וַנַּחֲרֵ֣ם אוֹתָ֔ם כַּאֲשֶׁ֣ר עָשִׂ֔ינוּ לְסִיחֹ֖ן מֶ֣לֶךְ חֶשְׁבּ֑וֹן הַחֲרֵם֙ כׇּל־עִ֣יר מְתִ֔ם הַנָּשִׁ֖ים וְהַטָּֽף׃ הִשְׁמַדְנוּ אוֹתָם כְּמוֹ שֶׁעָשִׂינוּ לְסִיחוֹן מֶלֶךְ חֶשְׁבּוֹן – הַשְׁמָדַת כָּל תּוֹשְׁבֵי הֶעָרִים וְהַנָּשִׁים וְהַיְּלָדִים.
ז וְכׇל־הַבְּהֵמָ֛ה וּשְׁלַ֥ל הֶעָרִ֖ים בַּזּ֥וֹנוּ לָֽנוּ׃ וְאֶת כָּל הַבְּהֵמוֹת וּשְׁלַל הֶעָרִים לָקַחְנוּ לָנוּ לְשָׁלָל.
ח וַנִּקַּ֞ח בָּעֵ֤ת הַהִוא֙ אֶת־הָאָ֔רֶץ מִיַּ֗ד שְׁנֵי֙ מַלְכֵ֣י הָאֱמֹרִ֔י אֲשֶׁ֖ר בְּעֵ֣בֶר הַיַּרְדֵּ֑ן מִנַּ֥חַל אַרְנֹ֖ן עַד־הַ֥ר חֶרְמֽוֹן׃ לָקַחְנוּ אָז אֶת הָאָרֶץ מִיַּד שְׁנֵי מַלְכֵי הָאֱמוֹרִי שֶׁבְּעֵבֶר הַיַּרְדֵּן מִנַּחַל אַרְנוֹן עַד הַר חֶרְמוֹן.
ט צִידֹנִ֛ים יִקְרְא֥וּ לְחֶרְמ֖וֹן שִׂרְיֹ֑ן וְהָ֣אֱמֹרִ֔י יִקְרְאוּ־ל֖וֹ שְׂנִֽיר׃ צִידוֹנִים קָרְאוּ לְחֶרְמוֹן 'שִׂרְיוֹן', וְהָאֱמוֹרִי קָרְאוּ לוֹ 'הַמֻּשְׁלָג'.
י כֹּ֣ל ׀ עָרֵ֣י הַמִּישֹׁ֗ר וְכׇל־הַגִּלְעָד֙ וְכׇל־הַבָּשָׁ֔ן עַד־סַלְכָ֖ה וְאֶדְרֶ֑עִי עָרֵ֛י מַמְלֶ֥כֶת ע֖וֹג בַּבָּשָֽׁן׃ לָקַחְנוּ אֶת כָּל עָרֵי הַמִּישׁוֹר וְכָל הַגִּלְעָד וְכָל הַבָּשָׁן עַד סַלְכָה וְאֶדְרֶעִי עָרֵי מַמְלֶכֶת עוֹג בַּבָּשָׁן.
יא כִּ֣י רַק־ע֞וֹג מֶ֣לֶךְ הַבָּשָׁ֗ן נִשְׁאַר֮ מִיֶּ֣תֶר הָרְפָאִים֒ הִנֵּ֤ה עַרְשׂוֹ֙ עֶ֣רֶשׂ בַּרְזֶ֔ל הֲלֹ֣ה הִ֔וא בְּרַבַּ֖ת בְּנֵ֣י עַמּ֑וֹן תֵּ֧שַׁע אַמּ֣וֹת אׇרְכָּ֗הּ וְאַרְבַּ֥ע אַמּ֛וֹת רׇחְבָּ֖הּ בְּאַמַּת־אִֽישׁ׃ הֲרֵי רַק עוֹג מֶלֶךְ הַבָּשָׁן נִשְׁאַר מִשְּׁאֵרִית הָרְפָאִים, הִנֵּה מִטָּתוֹ מִטַּת בַּרְזֶל, הֲרֵי הִיא בְּרַבַּת בְּנֵי עַמּוֹן, אָרְכָּהּ תֵּשַׁע אַמּוֹת וְרָחְבָּהּ אַרְבַּע אַמּוֹת שֶׁל אַמַּת אִישׁ מְמוּצָע.
יב וְאֶת־הָאָ֧רֶץ הַזֹּ֛את יָרַ֖שְׁנוּ בָּעֵ֣ת הַהִ֑וא מֵעֲרֹעֵ֞ר אֲשֶׁר־עַל־נַ֣חַל אַרְנֹ֗ן וַחֲצִ֤י הַֽר־הַגִּלְעָד֙ וְעָרָ֔יו נָתַ֕תִּי לָרֽאוּבֵנִ֖י וְלַגָּדִֽי׃ וְכָבַשְׁנוּ אָז אֶת הָאָרֶץ הַזֹּאת, נָתַתִּי לָרְאוּבֵנִי וְלַגָּדִי – מֵעֲרוֹעֵר שֶׁעַל יַד נַחַל אַרְנוֹן, וַחֲצִי הַר הַגִּלְעָד, וְעָרָיו.
יג וְיֶ֨תֶר הַגִּלְעָ֤ד וְכׇל־הַבָּשָׁן֙ מַמְלֶ֣כֶת ע֔וֹג נָתַ֕תִּי לַחֲצִ֖י שֵׁ֣בֶט הַֽמְנַשֶּׁ֑ה כֹּ֣ל חֶ֤בֶל הָֽאַרְגֹּב֙ לְכׇל־הַבָּשָׁ֔ן הַה֥וּא יִקָּרֵ֖א אֶ֥רֶץ רְפָאִֽים׃ וּשְׁאָר הַגִּלְעָד וְכָל הַבָּשָׁן מַמְלֶכֶת עוֹג – נָתַתִּי לַחֲצִי שֵׁבֶט הַמְנַשֶּׁה, כָּל חֶבֶל הָאַרְגּוֹב שֶׁבְּכָל הַבָּשָׁן, הָאֵזוֹר הַהוּא נִקְרָא אֶרֶץ רְפָאִים.
יד יָאִ֣יר בֶּן־מְנַשֶּׁ֗ה לָקַח֙ אֶת־כׇּל־חֶ֣בֶל אַרְגֹּ֔ב עַד־גְּב֥וּל הַגְּשׁוּרִ֖י וְהַמַּֽעֲכָתִ֑י וַיִּקְרָא֩ אֹתָ֨ם עַל־שְׁמ֤וֹ אֶת־הַבָּשָׁן֙ חַוֺּ֣ת יָאִ֔יר עַ֖ד הַיּ֥וֹם הַזֶּֽה׃ יָאִיר בֶּן מְנַשֶּׁה לָקַח אֶת כָּל חֶבֶל אַרְגּוֹב עַד גְּבוּל הַגְּשׁוּרִי וְהַמַּעֲכָתִי, וְקָרָא לָהֶן לְחַוּוֹת הַבָּשָׁן, עַל שְׁמוֹ – 'חַוּוֹת יָאִיר', וְכָךְ שְׁמָם עַד הַיּוֹם.

שביעי[עריכה]

טו וּלְמָכִ֖יר נָתַ֥תִּי אֶת־הַגִּלְעָֽד׃ וּלְמָכִיר נָתַתִּי אֶת הַגִּלְעָד.
טז וְלָראוּבֵנִ֨י וְלַגָּדִ֜י נָתַ֤תִּי מִן־הַגִּלְעָד֙ וְעַד־נַ֣חַל אַרְנֹ֔ן תּ֥וֹךְ הַנַּ֖חַל וּגְבֻ֑ל וְעַד֙ יַבֹּ֣ק הַנַּ֔חַל גְּב֖וּל בְּנֵ֥י עַמּֽוֹן׃ וְלָרְאוּבֵנִי וְלַגָּדִי נָתַתִּי מֵהַגִּלְעָד וְעַד נַחַל אַרְנוֹן כּוֹלֵל תְּחוּם שְׂפַת הַנַּחַל, וְעַד נַחַל יַבּוֹק שֶׁהוּא גְּבוּל בְּנֵי עַמּוֹן.
יז וְהָֽעֲרָבָ֖ה וְהַיַּרְדֵּ֣ן וּגְבֻ֑ל מִכִּנֶּ֗רֶת וְעַ֨ד יָ֤ם הָֽעֲרָבָה֙ יָ֣ם הַמֶּ֔לַח תַּ֛חַת אַשְׁדֹּ֥ת הַפִּסְגָּ֖ה מִזְרָֽחָה׃ וְאֶת הָעֲרָבָה, וְהַיַּרְדֵּן כּוֹלֵל תְּחוּם שְׂפָתוֹ, מִכִּנֶּרֶת וְעַד יָם הָעֲרָבָה – יָם הַמֶּלַח, לְמַרְגְּלוֹת הַמַּפָּלִים שֶׁמֵּהָהָר שֶׁמִּמִּזְרָח.
פָּרָשַׁת צִוּוּי הַנִּלְחָמִים
יח וָאֲצַ֣ו אֶתְכֶ֔ם בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר יְהֹוָ֣ה אֱלֹהֵיכֶ֗ם נָתַ֨ן לָכֶ֜ם אֶת־הָאָ֤רֶץ הַזֹּאת֙ לְרִשְׁתָּ֔הּ חֲלוּצִ֣ים תַּֽעַבְר֗וּ לִפְנֵ֛י אֲחֵיכֶ֥ם בְּנֵֽי־יִשְׂרָאֵ֖ל כׇּל־בְּנֵי־חָֽיִל׃ צִוִּיתִי אֶתְכֶם בַּזְּמָן הַהוּא – בְּנֵי רְאוּבֵן וְגָד, כָּךְ: ה' אֱלֹהֵיכֶם נָתַן לָכֶם אֶת הָאָרֶץ הַזֹּאת לִכְבּוֹשׁ אוֹתָהּ, אַתֶּם כָּל הַלּוֹחֲמִים תַּעַבְרוּ רִאשׁוֹנִים לִפְנֵי אֲחֵיכֶם בְּנֵי יִשְׂרָאֵל.
יט רַ֠ק נְשֵׁיכֶ֣ם וְטַפְּכֶם֮ וּמִקְנֵכֶם֒ יָדַ֕עְתִּי כִּֽי־מִקְנֶ֥ה רַ֖ב לָכֶ֑ם יֵֽשְׁבוּ֙ בְּעָ֣רֵיכֶ֔ם אֲשֶׁ֥ר נָתַ֖תִּי לָכֶֽם׃ רַק נְשֵׁיכֶם וִילָדֵיכֶם וַעֲדָרֵיכֶם, וַאֲנִי יוֹדֵעַ שֶׁיֵּשׁ לָכֶם עֲדָרִים רַבִּים, יֵשְׁבוּ בְּעָרֵיכֶם שֶׁנָּתַתִּי לָכֶם.

מפטיר[עריכה]

כ עַ֠ד אֲשֶׁר־יָנִ֨יחַ יְהֹוָ֥ה ׀ לַֽאֲחֵיכֶם֮ כָּכֶם֒ וְיָרְשׁ֣וּ גַם־הֵ֔ם אֶת־הָאָ֕רֶץ אֲשֶׁ֨ר יְהֹוָ֧ה אֱלֹהֵיכֶ֛ם נֹתֵ֥ן לָהֶ֖ם בְּעֵ֣בֶר הַיַּרְדֵּ֑ן וְשַׁבְתֶּ֗ם אִ֚ישׁ לִֽירֻשָּׁת֔וֹ אֲשֶׁ֥ר נָתַ֖תִּי לָכֶֽם׃ עַד שֶׁה' יִתֵּן לַאֲחֵיכֶם נַחֲלָה לִמְנוּחָה כְּמוֹ שֶׁנָּתַן מְנוּחָה לָכֶם, וְיִכְבְּשׁוּ גַּם הֵם אֶת הָאָרֶץ שֶׁה' אֱלֹהֵיכֶם נוֹתֵן לָהֶם בְּעֵבֶר הַיַּרְדֵּן, וְאָז תָּשׁוּבוּ אִישׁ לְנַחֲלָתוֹ שֶׁנָּתַתִּי לָכֶם.
כא וְאֶת־יְהוֹשׁ֣וּעַ צִוֵּ֔יתִי בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר עֵינֶ֣יךָ הָרֹאֹ֗ת אֵת֩ כׇּל־אֲשֶׁ֨ר עָשָׂ֜ה יְהֹוָ֤ה אֱלֹהֵיכֶם֙ לִשְׁנֵי֙ הַמְּלָכִ֣ים הָאֵ֔לֶּה כֵּֽן־יַעֲשֶׂ֤ה יְהֹוָה֙ לְכׇל־הַמַּמְלָכ֔וֹת אֲשֶׁ֥ר אַתָּ֖ה עֹבֵ֥ר שָֽׁמָּה׃ וְאֶת יְהוֹשֻׁעַ צִוֵּיתִי אָז: עֵינֶיךָ רָאוּ אֶת כָּל מַה שֶּׁה' אֱלֹהֵיכֶם עָשָׂה לִשְׁנֵי הַמְּלָכִים הָאֵלֶּה, כָּךְ ה' יַעֲשֶׂה לְכָל הַמַּמְלָכוֹת שֶׁתַּעֲבוֹר לְשָׁם.
כב לֹ֖א תִּֽירָא֑וּם כִּ֚י יְהֹוָ֣ה אֱלֹֽהֵיכֶ֔ם ה֖וּא הַנִּלְחָ֥ם לָכֶֽם׃ אַל תִּפְחֲדוּ מֵהֶם, כִּי ה' אֱלֹהֵיכֶם הוּא הַנִּלְחָם עֲבוּרְכֶם!
