קפיצה לניווט קפיצה לחיפוש

הערה לגבי חלוקת הפרשות: חלוקת הפרשות במגילת אסתר מסומנת כאן לפי מה שהיה בכתר ארם צובה (ע"פ עדותו של יהושע קמחי).


א א וַיְהִ֖י בִּימֵ֣י אֲחַשְׁוֵר֑וֹשׁ ה֣וּא אֲחַשְׁוֵר֗וֹשׁ הַמֹּלֵךְ֙ מֵהֹ֣דּוּ וְעַד־כּ֔וּשׁ שֶׁ֛בַע וְעֶשְׂרִ֥ים וּמֵאָ֖ה מְדִינָֽה׃ בִּימֵי אֲחַשְׁוֵרוֹשׁ – הוּא אֲחַשְׁוֵרוֹשׁ שֶׁמָּלַךְ מֵהוֹדּוּ וְעַד כּוּשׁ – עַל מֵאָה עֶשְׂרִים וְשֶׁבַע מְדִינוֹת.
ב בַּיָּמִ֖ים הָהֵ֑ם כְּשֶׁ֣בֶת ׀ הַמֶּ֣לֶךְ אֲחַשְׁוֵר֗וֹשׁ עַ֚ל כִּסֵּ֣א מַלְכוּת֔וֹ אֲשֶׁ֖ר בְּשׁוּשַׁ֥ן הַבִּירָֽה׃ בַּיָּמִים הָהֵם כְּשֶׁהַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ מָלַךְ וְיָשַׁב עַל כִּסֵּא מַלְכוּתוֹ שֶׁבְּשׁוּשַׁן הַבִּירָה,
ג בִּשְׁנַ֤ת שָׁלוֹשׁ֙ לְמׇלְכ֔וֹ עָשָׂ֣ה מִשְׁתֶּ֔ה לְכׇל־שָׂרָ֖יו וַעֲבָדָ֑יו חֵ֣יל ׀ פָּרַ֣ס וּמָדַ֗י הַֽפַּרְתְּמִ֛ים וְשָׂרֵ֥י הַמְּדִינ֖וֹת לְפָנָֽיו׃ בַּשָּׁנָה הַשְּׁלִישִׁית לְמַלְכוּתוֹ הוּא עָשָׂה מְסִבַּת שְׁתִיַּת יַיִן לְכָל שָׂרָיו וַעֲבָדָיו: לִצְבָא פָּרַס וּמָדַי וְלַשַּׁלִּיטִים וְשָׂרֵי הַמְּדִינוֹת שֶׁבְּשִׁלְטוֹנוֹ.
ד בְּהַרְאֹת֗וֹ אֶת־עֹ֙שֶׁר֙ כְּב֣וֹד מַלְכוּת֔וֹ וְאֶ֨ת־יְקָ֔ר תִּפְאֶ֖רֶת גְּדוּלָּת֑וֹ יָמִ֣ים רַבִּ֔ים שְׁמוֹנִ֥ים וּמְאַ֖ת יֽוֹם׃ הוּא הֶרְאָה בָּהּ אֶת עוֹשְׁרוֹ וְאֶת כְּבוֹד מַלְכוּתוֹ וְאֶת חֲשִׁיבוּתוֹ וּפְאֵר גְּדוּלָּתוֹ, בְּמֶשֶׁךְ יָמִים רַבִּים – מֵאָה וּשְׁמוֹנִים יָמִים.
ה וּבִמְל֣וֹאת ׀ הַיָּמִ֣ים הָאֵ֗לֶּה עָשָׂ֣ה הַמֶּ֡לֶךְ לְכׇל־הָעָ֣ם הַנִּמְצְאִים֩ בְּשׁוּשַׁ֨ן הַבִּירָ֜ה לְמִגָּד֧וֹל וְעַד־קָטָ֛ן מִשְׁתֶּ֖ה שִׁבְעַ֣ת יָמִ֑ים בַּחֲצַ֕ר גִּנַּ֥ת בִּיתַ֖ן הַמֶּֽלֶךְ׃ וּכְשֶׁעָבְרוּ הַיָּמִים הָאֵלֶּה, הַמֶּלֶךְ עָשָׂה מְסִבַּת שְׁתִיַּת יַיִן שִׁבְעָה יָמִים, לְכָל הָעָם שֶׁנִּמְצְאִים בְּשׁוּשַׁן הַבִּירָה מִגָּדוֹל וְעַד קָטָן בֶּחָצֵר שֶׁל גִּנַּת אַרְמוֹן הַמֶּלֶךְ.
ו ח֣וּר ׀ כַּרְפַּ֣ס וּתְכֵ֗לֶת אָחוּז֙ בְּחַבְלֵי־ב֣וּץ וְאַרְגָּמָ֔ן עַל־גְּלִ֥ילֵי כֶ֖סֶף וְעַמּ֣וּדֵי שֵׁ֑שׁ מִטּ֣וֹת ׀ זָהָ֣ב וָכֶ֗סֶף עַ֛ל רִֽצְפַ֥ת בַּהַט־וָשֵׁ֖שׁ וְדַ֥ר וְסֹחָֽרֶת׃ יְרִיעוֹת בְּלָבָן וִירַקְרַק וּתְכֵלֶת קְשׁוּרוֹת בְּחַבְלֵי פִשְׁתָּן וְאַרְגָּמָן עַל וָוֵי כֶּסֶף וְעַמּוּדֵי שַׁיִשׁ, וְיָשְׁבוּ עַל מִטּוֹת שֶׁל זָהָב וָכֶסֶף עַל רִצְפָּה מֵאַבְנֵי הַחֵן בַּהַט וְשַׁיִשׁ וְדַר וְסוֹחָרֶת.
ז וְהַשְׁקוֹת֙ בִּכְלֵ֣י זָהָ֔ב וְכֵלִ֖ים מִכֵּלִ֣ים שׁוֹנִ֑ים וְיֵ֥ין מַלְכ֛וּת רָ֖ב כְּיַ֥ד הַמֶּֽלֶךְ׃ וּשְׁתִיָּה בִּכְלֵי זָהָב וְכֵלִים מִסּוּגִים שׁוֹנִים, וְהַרְבֵּה יַיִן מַלְכוּתִי – כְּיַד הַמֶּלֶךְ.
ח וְהַשְּׁתִיָּ֥ה כַדָּ֖ת אֵ֣ין אֹנֵ֑ס כִּי־כֵ֣ן ׀ יִסַּ֣ד הַמֶּ֗לֶךְ עַ֚ל כׇּל־רַ֣ב בֵּית֔וֹ לַעֲשׂ֖וֹת כִּרְצ֥וֹן אִישׁ־וָאִֽישׁ׃ וְהַשְּׁתִיָּה כְּמוֹ שֶׁרְגִילִים – אֵין חוֹבָה לִשְׁתּוֹת, כִּי כָּךְ צִוָּה הַמֶּלֶךְ עַל כָּל הָאַחְרָאִים בְּבֵיתוֹ – לַעֲשׂוֹת כִּרְצוֹן כָּל אִישׁ.
ט גַּ֚ם וַשְׁתִּ֣י הַמַּלְכָּ֔ה עָשְׂתָ֖ה מִשְׁתֵּ֣ה נָשִׁ֑ים בֵּ֚ית הַמַּלְכ֔וּת אֲשֶׁ֖ר לַמֶּ֥לֶךְ אֲחַשְׁוֵרֽוֹשׁ׃ גַּם וַשְׁתִּי הַמַּלְכָּה עָשְׂתָה מְסִבַּת שְׁתִיַּת יַיִן לַנָּשִׁים בְּאַרְמוֹן הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ.
ס
י בַּיּוֹם֙ הַשְּׁבִיעִ֔י כְּט֥וֹב לֵב־הַמֶּ֖לֶךְ בַּיָּ֑יִן אָמַ֡ר לִ֠מְהוּמָ֠ן בִּזְּתָ֨א חַרְבוֹנָ֜א בִּגְתָ֤א וַאֲבַגְתָא֙ זֵתַ֣ר וְכַרְכַּ֔ס שִׁבְעַת֙ הַסָּ֣רִיסִ֔ים הַמְשָׁ֣רְתִ֔ים אֶת־פְּנֵ֖י הַמֶּ֥לֶךְ אֲחַשְׁוֵרֽוֹשׁ׃ בַּיּוֹם הַשְּׁבִיעִי, כְּשֶׁהַמֶּלֶךְ שָׂמַח בַּיַּיִן, הוּא אָמַר לִמְהוּמָן, בִּזְּתָא, חַרְבוֹנָא, בִּגְתָא וַאֲבַגְתָא, זֵתַר וְכַרְכַּס – שִׁבְעַת הַמְּשָׁרְתִים שֶׁלִּפְנֵי הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ,
יא לְ֠הָבִ֠יא אֶת־וַשְׁתִּ֧י הַמַּלְכָּ֛ה לִפְנֵ֥י הַמֶּ֖לֶךְ בְּכֶ֣תֶר מַלְכ֑וּת לְהַרְא֨וֹת הָֽעַמִּ֤ים וְהַשָּׂרִים֙ אֶת־יׇפְיָ֔הּ כִּֽי־טוֹבַ֥ת מַרְאֶ֖ה הִֽיא׃ לְהָבִיא אֶת וַשְׁתִּי הַמַּלְכָּה לִפְנֵי הַמֶּלֶךְ בְּכֶתֶר מַלְכוּת, כְּדֵי לְהַרְאוֹת לָעַמִּים וְלַשָּׂרִים אֶת יָפְיָהּ שֶׁהִיא נִרְאֵית טוֹב.
יב וַתְּמָאֵ֞ן הַמַּלְכָּ֣ה וַשְׁתִּ֗י לָבוֹא֙ בִּדְבַ֣ר הַמֶּ֔לֶךְ אֲשֶׁ֖ר בְּיַ֣ד הַסָּרִיסִ֑ים וַיִּקְצֹ֤ף הַמֶּ֙לֶךְ֙ מְאֹ֔ד וַחֲמָת֖וֹ בָּעֲרָ֥ה בֽוֹ׃ הַמַּלְכָּה וַשְׁתִּי סֵרְבָה לָבוֹא כִּפְקֻדַּת הַמֶּלֶךְ שֶׁנִּמְסְרָה עַל יְדֵי הַמְּשָׁרְתִים, הַמֶּלֶךְ הִתְרַגֵּז מְאוֹד וְכַעֲסוֹ בָּעַר בּוֹ.
יג וַיֹּ֣אמֶר הַמֶּ֔לֶךְ לַחֲכָמִ֖ים יֹדְעֵ֣י הָֽעִתִּ֑ים כִּי־כֵן֙ דְּבַ֣ר הַמֶּ֔לֶךְ לִפְנֵ֕י כׇּל־יֹדְעֵ֖י דָּ֥ת וָדִֽין׃ הַמֶּלֶךְ דִּבֵּר עִם הַחֲכָמִים שֶׁמְּבִינִים בְּאַסְטְרוֹלוֹגְיָה, כִּי כָּךְ מִנְהַג הַמֶּלֶךְ בְּמִשְׁפָּט – לְהִתְיַעֵץ עִם כָּל יוֹדְעֵי הַחֻקִּים וְהַדִּינִים.
יד וְהַקָּרֹ֣ב אֵלָ֗יו כַּרְשְׁנָ֤א שֵׁתָר֙ אַדְמָ֣תָא תַרְשִׁ֔ישׁ מֶ֥רֶס מַרְסְנָ֖א מְמוּכָ֑ן שִׁבְעַ֞ת שָׂרֵ֣י ׀ פָּרַ֣ס וּמָדַ֗י רֹאֵי֙ פְּנֵ֣י הַמֶּ֔לֶךְ הַיֹּשְׁבִ֥ים רִאשֹׁנָ֖ה בַּמַּלְכֽוּת׃ וְהַמְּקוֹרָבִים אֵלָיו הָיוּ: כַּרְשְׁנָא, שֵׁתָר, אַדְמָתָא, תַּרְשִׁישׁ, מֶרֶס, מַרְסְנָא וּמְמוּכָן – שִׁבְעַת שָׂרֵי פָּרַס וּמָדַי רוֹאֵי פְּנֵי הַמֶּלֶךְ שֶׁיּוֹשְׁבִים רִאשׁוֹנִים בִּישִׁיבוֹת הַמַּלְכוּת.
טו כְּדָת֙ מַֽה־לַּעֲשׂ֔וֹת בַּמַּלְכָּ֖ה וַשְׁתִּ֑י עַ֣ל ׀ אֲשֶׁ֣ר לֹֽא־עָשְׂתָ֗ה אֶֽת־מַאֲמַר֙ הַמֶּ֣לֶךְ אֲחַשְׁוֵר֔וֹשׁ בְּיַ֖ד הַסָּרִיסִֽים׃ שָׁאַל אוֹתָם הַמֶּלֶךְ: מַה לְפִי הַחוֹק צָּרִיךְ לַעֲשׂוֹת לַמַּלְכָּה וַשְׁתִּי עַל כָּךְ שֶׁלֹּא עָשְׂתָה אֶת פְּקֻדַּת הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ שֶׁמָּסַר לָהּ עַל יְדֵי הַמְּשָׁרְתִים?
פ
טז וַיֹּ֣אמֶר מומכן מְמוּכָ֗ן לִפְנֵ֤י הַמֶּ֙לֶךְ֙ וְהַשָּׂרִ֔ים לֹ֤א עַל־הַמֶּ֙לֶךְ֙ לְבַדּ֔וֹ עָוְתָ֖ה וַשְׁתִּ֣י הַמַּלְכָּ֑ה כִּ֤י עַל־כׇּל־הַשָּׂרִים֙ וְעַל־כׇּל־הָ֣עַמִּ֔ים אֲשֶׁ֕ר בְּכׇל־מְדִינ֖וֹת הַמֶּ֥לֶךְ אֲחַשְׁוֵרֽוֹשׁ׃ אָמַר מְמוּכָן לִפְנֵי הַמֶּלֶךְ וְהַשָּׂרִים: לֹא רַק כְּלַפֵּי הַמֶּלֶךְ פָּשְׁעָה וַשְׁתִּי הַמַּלְכָּה, אֶלָּא גַּם כְּלַפֵּי כָּל הַשָּׂרִים וְכָל הָעַמִּים שֶׁבְּכָל מְדִינוֹת הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ.
יז כִּֽי־יֵצֵ֤א דְבַר־הַמַּלְכָּה֙ עַל־כׇּל־הַנָּשִׁ֔ים לְהַבְז֥וֹת בַּעְלֵיהֶ֖ן בְּעֵינֵיהֶ֑ן בְּאׇמְרָ֗ם הַמֶּ֣לֶךְ אֲחַשְׁוֵר֡וֹשׁ אָמַ֞ר לְהָבִ֨יא אֶת־וַשְׁתִּ֧י הַמַּלְכָּ֛ה לְפָנָ֖יו וְלֹא־בָֽאָה׃ כִּי כְּשֶׁהַסִּפּוּר עִם הַמַּלְכָּה יִתְפַּרְסֵם אֵצֶל כָּל הַנָּשִׁים בַּעְלֵיהֶן יִתְבַּזּוּ בְּעֵינֵיהֶן, וְהֵן יֹאמְרוּ: אֲפִלּוּ הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ אָמַר לְהָבִיא לְפָנָיו אֶת וַשְׁתִּי הַמַּלְכָּה וְהִיא לֹא בָּאָה!
יח וְֽהַיּ֨וֹם הַזֶּ֜ה תֹּאמַ֣רְנָה ׀ שָׂר֣וֹת פָּֽרַס־וּמָדַ֗י אֲשֶׁ֤ר שָֽׁמְעוּ֙ אֶת־דְּבַ֣ר הַמַּלְכָּ֔ה לְכֹ֖ל שָׂרֵ֣י הַמֶּ֑לֶךְ וּכְדַ֖י בִּזָּי֥וֹן וָקָֽצֶף׃ וּמֵהַיּוֹם שָׂרוֹת פָּרַס וּמָדַי שֶׁשָּׁמְעוּ אֶת הַסִּפּוּר עִם הַמַּלְכָּה יֹאמְרוּ כָּךְ גַּם לְכָל שָׂרֵי הַמֶּלֶךְ, וְזֶה יִגְרוֹם לְהַרְבֵּה בִּזָּיוֹן וְכַעַס.
יט אִם־עַל־הַמֶּ֣לֶךְ ט֗וֹב יֵצֵ֤א דְבַר־מַלְכוּת֙ מִלְּפָנָ֔יו וְיִכָּתֵ֛ב בְּדָתֵ֥י פָֽרַס־וּמָדַ֖י וְלֹ֣א יַעֲב֑וֹר אֲשֶׁ֨ר לֹֽא־תָב֜וֹא וַשְׁתִּ֗י לִפְנֵי֙ הַמֶּ֣לֶךְ אֲחַשְׁוֵר֔וֹשׁ וּמַלְכוּתָהּ֙ יִתֵּ֣ן הַמֶּ֔לֶךְ לִרְעוּתָ֖הּ הַטּוֹבָ֥ה מִמֶּֽנָּה׃ לָכֵן, אִם זֶה טוֹב בְּעֵינֵי הַמֶּלֶךְ, יֵצֵא צַו מַלְכוּתִי מִלְּפָנָיו וְיִכָּתֵב בְּסֵפֶר חֻקֵּי פָּרַס וּמָדַי חוֹק שֶׁלֹּא יִשְׁתַּנֶּה, שֶׁוַּשְׁתִּי לֹא תָּבוֹא יוֹתֵר לִפְנֵי הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ, וְהַמֶּלֶךְ יִתֵּן אֶת מַלְכוּתָהּ לְאִשָּׁה אַחֶרֶת טוֹבָה מִמֶּנָּה.
כ וְנִשְׁמַע֩ פִּתְגָ֨ם הַמֶּ֤לֶךְ אֲשֶֽׁר־יַעֲשֶׂה֙ בְּכׇל־מַלְכוּת֔וֹ כִּ֥י רַבָּ֖ה הִ֑יא וְכׇל־הַנָּשִׁ֗ים יִתְּנ֤וּ יְקָר֙ לְבַעְלֵיהֶ֔ן לְמִגָּד֖וֹל וְעַד־קָטָֽן׃ וְהַדָּבָר הַזֶּה שֶׁהַמֶּלֶךְ יַעֲשֶׂה יִתְפַּרְסֵם בְּכָל מַמְלַכְתּוֹ – שֶׁהִיא גְּדוֹלָה, וְכָךְ כָּל הַנָּשִׁים יִתְּנוּ כָּבוֹד לְבַעְלֵיהֶן בֵּין אִם הוּא חָשׁוּב אוֹ פָּשׁוּט.
כא וַיִּיטַב֙ הַדָּבָ֔ר בְּעֵינֵ֥י הַמֶּ֖לֶךְ וְהַשָּׂרִ֑ים וַיַּ֥עַשׂ הַמֶּ֖לֶךְ כִּדְבַ֥ר מְמוּכָֽן׃ הָרַעְיוֹן נִרְאָה טוֹב בְּעֵינֵי הַמֶּלֶךְ וְהַשָּׂרִים, וְהַמֶּלֶךְ עָשָׂה כְּדִבְרֵי מְמוּכָן.
כב וַיִּשְׁלַ֤ח סְפָרִים֙ אֶל־כׇּל־מְדִינ֣וֹת הַמֶּ֔לֶךְ אֶל־מְדִינָ֤ה וּמְדִינָה֙ כִּכְתָבָ֔הּ וְאֶל־עַ֥ם וָעָ֖ם כִּלְשׁוֹנ֑וֹ לִהְי֤וֹת כׇּל־אִישׁ֙ שֹׂרֵ֣ר בְּבֵית֔וֹ וּמְדַבֵּ֖ר כִּלְשׁ֥וֹן עַמּֽוֹ׃ הוּא שָׁלַח מִכְתָּבִים אֶל כָּל מְדִינוֹת הַמֶּלֶךְ – אֶל כָּל מְדִינָה וּמְדִינָה בִּכְתָבָהּ וְאֶל כָּל עָם וְעָם בִּשְׂפָתוֹ, שֶׁכָּל אִישׁ יִשְׁלוֹט בְּבֵיתוֹ וִידַבֵּר בִּשְׁפַת עַמּוֹ וְלֹא בִּשְׁפַת אִשְׁתּוֹ.
פ
ב א אַחַר֙ הַדְּבָרִ֣ים הָאֵ֔לֶּה כְּשֹׁ֕ךְ חֲמַ֖ת הַמֶּ֣לֶךְ אֲחַשְׁוֵר֑וֹשׁ זָכַ֤ר אֶת־וַשְׁתִּי֙ וְאֵ֣ת אֲשֶׁר־עָשָׂ֔תָה וְאֵ֥ת אֲשֶׁר־נִגְזַ֖ר עָלֶֽיהָ׃ אַחֲרֵי הַדְּבָרִים הָאֵלֶּה כְּשֶׁהַמֶּלֶךְ אַחֲשְׁוֵרוֹשׁ נִרְגַּע מִכַּעֲסוֹ, הוּא נִזְכַּר בְּוַשְׁתִּי וּמַה שֶּׁעָשְׂתָה וּמַה שֶּׁנִּגְזַר עָלֶיהָ.
ב וַיֹּאמְר֥וּ נַעֲרֵֽי־הַמֶּ֖לֶךְ מְשָׁרְתָ֑יו יְבַקְשׁ֥וּ לַמֶּ֛לֶךְ נְעָר֥וֹת בְּתוּל֖וֹת טוֹב֥וֹת מַרְאֶֽה׃ הִצִּיעוּ לוֹ נַעֲרֵי הַמֶּלֶךְ מְשָׁרְתָיו: יְחַפְּשׂוּ לַמֶּלֶךְ נְעָרוֹת בְּתוּלוֹת יָפוֹת.
ג וְיַפְקֵ֨ד הַמֶּ֣לֶךְ פְּקִידִים֮ בְּכׇל־מְדִינ֣וֹת מַלְכוּתוֹ֒ וְיִקְבְּצ֣וּ אֶת־כׇּל־נַעֲרָֽה־בְ֠תוּלָ֠ה טוֹבַ֨ת מַרְאֶ֜ה אֶל־שׁוּשַׁ֤ן הַבִּירָה֙ אֶל־בֵּ֣ית הַנָּשִׁ֔ים אֶל־יַ֥ד הֵגֶ֛א סְרִ֥יס הַמֶּ֖לֶךְ שֹׁמֵ֣ר הַנָּשִׁ֑ים וְנָת֖וֹן תַּמְרֻקֵיהֶֽן׃ וְהַמֶּלֶךְ יְמַנֶּה פְּקִידִים בְּכָל מְדִינוֹת מַלְכוּתוֹ, שֶׁיַּאַסְפוּ כָּל נַעֲרָה בְּתוּלָה יָפָה לְשׁוּשַׁן הַבִּירָה לְבֵית הַנָּשִׁים שֶׁבְּנִיהוּל הֵגֶא שַׂר הַמֶּלֶךְ שׁוֹמֵר הַנָּשִׁים, וְיִתְנוּ לָהֶם תַּמְרוּקִים.
ד וְהַֽנַּעֲרָ֗ה אֲשֶׁ֤ר תִּיטַב֙ בְּעֵינֵ֣י הַמֶּ֔לֶךְ תִּמְלֹ֖ךְ תַּ֣חַת וַשְׁתִּ֑י וַיִּיטַ֧ב הַדָּבָ֛ר בְּעֵינֵ֥י הַמֶּ֖לֶךְ וַיַּ֥עַשׂ כֵּֽן׃ וְהַנַּעֲרָה שֶׁתִּמְצָא חֵן בְּעֵינֵי הַמֶּלֶךְ תִּמְלוֹךְ בִּמְקוֹם וַשְׁתִּי. הָרַעֲיוֹן מָצָא חֵן בְּעֵינֵי הַמֶּלֶךְ, וְכָךְ הוּא עָשָׂה.
פ
ה אִ֣ישׁ יְהוּדִ֔י הָיָ֖ה בְּשׁוּשַׁ֣ן הַבִּירָ֑ה וּשְׁמ֣וֹ מׇרְדֳּכַ֗י בֶּ֣ן יָאִ֧יר בֶּן־שִׁמְעִ֛י בֶּן־קִ֖ישׁ אִ֥ישׁ יְמִינִֽי׃ בְּשׁוּשַׁן הַבִּירָה הָיָה אִישׁ מֵאֶרֶץ יְהוּדָה וּשְׁמוֹ מָרְדֳּכַי בֶּן יָאִיר בֶּן שִׁמְעִי בֶּן קִישׁ, מִשֵׁבֶט בִּנְיָמִין.
ו אֲשֶׁ֤ר הׇגְלָה֙ מִיר֣וּשָׁלַ֔יִם עִם־הַגֹּלָה֙ אֲשֶׁ֣ר הׇגְלְתָ֔ה עִ֖ם יְכׇנְיָ֣ה מֶֽלֶךְ־יְהוּדָ֑ה אֲשֶׁ֣ר הֶגְלָ֔ה נְבוּכַדְנֶצַּ֖ר מֶ֥לֶךְ בָּבֶֽל׃ הוּא הֻגְלָה מִיְּרוּשָׁלַיִם עִם הַגּוֹלָה שֶׁהֻגְלְתָה עִם יְכָנְיָה מֶלֶךְ יְהוּדָה, שֶׁהֶגְלָה אוֹתָהּ נְבוּכַדְנֶאצַּר מֶלֶךְ בָּבֶל.
ז וַיְהִ֨י אֹמֵ֜ן אֶת־הֲדַסָּ֗ה הִ֤יא אֶסְתֵּר֙ בַּת־דֹּד֔וֹ כִּ֛י אֵ֥ין לָ֖הּ אָ֣ב וָאֵ֑ם וְהַנַּעֲרָ֤ה יְפַת־תֹּ֙אַר֙ וְטוֹבַ֣ת מַרְאֶ֔ה וּבְמ֤וֹת אָבִ֙יהָ֙ וְאִמָּ֔הּ לְקָחָ֧הּ מׇרְדֳּכַ֛י ל֖וֹ לְבַֽת׃ הוּא אִמֵּץ אֶת הֲדַסָּה – הִיא אֶסְתֵּר בַּת דּוֹדוֹ, כִּי לֹא הָיָה לָהּ אָב וָאֵם, וְהַנַּעֲרָה יָפָה בְּצוּרַת גּוּפָהּ וְנִרְאֵית טוֹב, וּכְשֶׁמֵּתוּ אָבִיהָ וְאִמָּהּ לָקַח אוֹתָהּ מָרְדֳּכַי אֵלָיו וְאִמֵּץ אוֹתָהּ כְּבַת.
ח וַיְהִ֗י בְּהִשָּׁמַ֤ע דְּבַר־הַמֶּ֙לֶךְ֙ וְדָת֔וֹ וּֽבְהִקָּבֵ֞ץ נְעָר֥וֹת רַבּ֛וֹת אֶל־שׁוּשַׁ֥ן הַבִּירָ֖ה אֶל־יַ֣ד הֵגָ֑י וַתִּלָּקַ֤ח אֶסְתֵּר֙ אֶל־בֵּ֣ית הַמֶּ֔לֶךְ אֶל־יַ֥ד הֵגַ֖י שֹׁמֵ֥ר הַנָּשִֽׁים׃ וּכְשֶׁהִתְפַּרְסֵם צַו הַמֶּלֶךְ וּגְזֵרָתוֹ וְנֶאֶסְפוּ נְעָרוֹת רַבּוֹת אֶל שׁוּשַׁן הַבִּירָה אֶל הֵגַי, אֶסְתֵּר נִלְקְחָה אֶל בֵּית הַמֶּלֶךְ אֶל הֵגַי שׁוֹמֵר הַנָּשִׁים.
ט וַתִּיטַ֨ב הַנַּעֲרָ֣ה בְעֵינָיו֮ וַתִּשָּׂ֣א חֶ֣סֶד לְפָנָיו֒ וַ֠יְבַהֵ֠ל אֶת־תַּמְרוּקֶ֤יהָ וְאֶת־מָנוֹתֶ֙הָ֙ לָתֵ֣ת לָ֔הּ וְאֵת֙ שֶׁ֣בַע הַנְּעָר֔וֹת הָרְאֻי֥וֹת לָֽתֶת־לָ֖הּ מִבֵּ֣ית הַמֶּ֑לֶךְ וַיְשַׁנֶּ֧הָ וְאֶת־נַעֲרוֹתֶ֛יהָ לְט֖וֹב בֵּ֥ית הַנָּשִֽׁים׃ הַנַּעֲרָה מָצְאָה חֵן בְּעֵינָיו וְהוּא חִבֵּב אוֹתָהּ, הוּא מִהֵר לָתֵת לָהּ אֶת תַּמְרוּקֶיהָ וְאֶת מְנוֹת הָאוֹכֶל שֶׁלָּהּ, וְאֶת שֶׁבַע הַמְּשָׁרְתוֹת שֶׁהִיא אֲמוּרָה לְקַבֵּל מִבֵּית הַמֶּלֶךְ, וְהוּא הִפְלָה לְטוֹבָה אוֹתָהּ וְאֶת נַעֲרוֹתֶיהָ בְּבֵית הַנָּשִׁים.
י לֹא־הִגִּ֣ידָה אֶסְתֵּ֔ר אֶת־עַמָּ֖הּ וְאֶת־מֽוֹלַדְתָּ֑הּ כִּ֧י מׇרְדֳּכַ֛י צִוָּ֥ה עָלֶ֖יהָ אֲשֶׁ֥ר לֹא־תַגִּֽיד׃ אֶסְתֵּר לֹא סִפְּרָה מַהוּ עַמָּהּ וְאֵיפֹה נוֹלְדָה כִּי מָרְדֳּכַי צִוָּה עָלֶיהָ שֶׁלֹּא תְּסַפֵּר.

יא וּבְכׇל־י֣וֹם וָי֔וֹם מׇרְדֳּכַי֙ מִתְהַלֵּ֔ךְ לִפְנֵ֖י חֲצַ֣ר בֵּית־הַנָּשִׁ֑ים לָדַ֙עַת֙ אֶת־שְׁל֣וֹם אֶסְתֵּ֔ר וּמַה־יֵּעָשֶׂ֖ה בָּֽהּ׃ וּבְכָל יוֹם מָרְדֳּכַי מִתְהַלֵּךְ לִפְנֵי חֲצַר בֵּית הַנָּשִׁים כְּדֵי לָדַעַת מַה שְּׁלוֹם אֶסְתֵּר וּמַה עוֹשִׂים אִתָּהּ.
יב וּבְהַגִּ֡יעַ תֹּר֩ נַעֲרָ֨ה וְנַעֲרָ֜ה לָב֣וֹא ׀ אֶל־הַמֶּ֣לֶךְ אֲחַשְׁוֵר֗וֹשׁ מִקֵּץ֩ הֱי֨וֹת לָ֜הּ כְּדָ֤ת הַנָּשִׁים֙ שְׁנֵ֣ים עָשָׂ֣ר חֹ֔דֶשׁ כִּ֛י כֵּ֥ן יִמְלְא֖וּ יְמֵ֣י מְרוּקֵיהֶ֑ן שִׁשָּׁ֤ה חֳדָשִׁים֙ בְּשֶׁ֣מֶן הַמֹּ֔ר וְשִׁשָּׁ֤ה חֳדָשִׁים֙ בַּבְּשָׂמִ֔ים וּבְתַמְרוּקֵ֖י הַנָּשִֽׁים׃ כְּשֶׁמַּגִּיעַ תּוֹר כָּל נַעֲרָה לָבוֹא אֶל הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ, כְּשֶׁמִּסְתַּיְּמִים שְׁנֵים עָשָׂר הֶחֳדָשִׁים שֶׁל הִתְקַשְּׁטוּת כְּמִנְהַג הַנָּשִׁים, כִּי כָּךְ זְמַן הַהִתְמָרְקוּת בַּבְּשָׂמִים – שִׁשָּׁה חֳדָשִׁים בְּשֶׁמֶן מוֹר וְשִׁשָּׁה חֳדָשִׁים בִּבְשָׂמִים וּבְתַמְרוּקֵי נָשִׁים.
יג וּבָזֶ֕ה הַֽנַּעֲרָ֖ה בָּאָ֣ה אֶל־הַמֶּ֑לֶךְ אֵת֩ כׇּל־אֲשֶׁ֨ר תֹּאמַ֜ר יִנָּ֤תֵֽן לָהּ֙ לָב֣וֹא עִמָּ֔הּ מִבֵּ֥ית הַנָּשִׁ֖ים עַד־בֵּ֥ית הַמֶּֽלֶךְ׃ עִם זֶה הַנַּעֲרָה בָּאָה אֶל הַמֶּלֶךְ: כָּל מַה שֶׁתְּבַקֵּשׁ יִנָּתֵן לָהּ לָבוֹא אִתָּהּ מִבֵּית הַנָּשִׁים לְבֵית הַמֶּלֶךְ.
יד בָּעֶ֣רֶב ׀ הִ֣יא בָאָ֗ה וּ֠בַבֹּ֠קֶר הִ֣יא שָׁבָ֞ה אֶל־בֵּ֤ית הַנָּשִׁים֙ שֵׁנִ֔י אֶל־יַ֧ד שַֽׁעַשְׁגַ֛ז סְרִ֥יס הַמֶּ֖לֶךְ שֹׁמֵ֣ר הַפִּֽילַגְשִׁ֑ים לֹא־תָב֥וֹא עוֹד֙ אֶל־הַמֶּ֔לֶךְ כִּ֣י אִם־חָפֵ֥ץ בָּ֛הּ הַמֶּ֖לֶךְ וְנִקְרְאָ֥ה בְשֵֽׁם׃ בָּעֶרֶב הִיא בָּאָה, וּבַבּוֹקֶר הִיא חוֹזֶרֶת אֶל בֵּית הַנָּשִׁים הַשֵּׁנִי אֶל שַׁעַשְׁגַז שַׂר הַמֶּלֶךְ שׁוֹמֵר הַפִּילַגְשִׁים, הִיא לֹא תָּבוֹא שׁוּב אֶל הַמֶּלֶךְ אֶלָּא אִם הַמֶּלֶךְ יִרְצֶה אוֹתָהּ וְיִקְרָא בִּשְׁמָהּ.
טו וּבְהַגִּ֣יעַ תֹּר־אֶסְתֵּ֣ר בַּת־אֲבִיחַ֣יִל דֹּ֣ד מׇרְדֳּכַ֡י אֲשֶׁר֩ לָקַֽח־ל֨וֹ לְבַ֜ת לָב֣וֹא אֶל־הַמֶּ֗לֶךְ לֹ֤א בִקְשָׁה֙ דָּבָ֔ר כִּ֠י אִ֣ם אֶת־אֲשֶׁ֥ר יֹאמַ֛ר הֵגַ֥י סְרִיס־הַמֶּ֖לֶךְ שֹׁמֵ֣ר הַנָּשִׁ֑ים וַתְּהִ֤י אֶסְתֵּר֙ נֹשֵׂ֣את חֵ֔ן בְּעֵינֵ֖י כׇּל־רֹאֶֽיהָ׃ וּכְשֶׁהִגִּיעַ הַתּוֹר שֶׁל אֶסְתֵּר בַּת אֲבִיחַיִל הַדּוֹד שֶׁל מָרְדֳּכַי שֶׁאִמֵּץ אוֹתָהּ כְּבַת, לָבוֹא אֶל הַמֶּלֶךְ, הִיא לֹא בִּקְשָׁה שׁוּם דָּבָר אֶלָּא רַק אֶת מַה שֶּׁיֹּאמַר הֵגַי מְשָׁרֵת הַמֶּלֶךְ שׁוֹמֵר הַנָּשִׁים. אֶסְתֵּר מָצְאָה חֵן בְּעֵינֵי כָּל רוֹאֶיהָ.
טז וַתִּלָּקַ֨ח אֶסְתֵּ֜ר אֶל־הַמֶּ֤לֶךְ אֲחַשְׁוֵרוֹשׁ֙ אֶל־בֵּ֣ית מַלְכוּת֔וֹ בַּחֹ֥דֶשׁ הָעֲשִׂירִ֖י הוּא־חֹ֣דֶשׁ טֵבֵ֑ת בִּשְׁנַת־שֶׁ֖בַע לְמַלְכוּתֽוֹ׃ אֶסְתֵּר נִלְקְחָה אֶל הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ אֶל בֵּית מַלְכוּתוֹ בַּחוֹדֶשׁ הָעֲשִׂירִי הוּא חוֹדֶשׁ טֵבֵת בַּשָּׁנָה הַשְּׁבִיעִית לְמַלְכוּתוֹ.
יז וַיֶּאֱהַ֨ב הַמֶּ֤לֶךְ אֶת־אֶסְתֵּר֙ מִכׇּל־הַנָּשִׁ֔ים וַתִּשָּׂא־חֵ֥ן וָחֶ֛סֶד לְפָנָ֖יו מִכׇּל־הַבְּתוּל֑וֹת וַיָּ֤שֶׂם כֶּֽתֶר־מַלְכוּת֙ בְּרֹאשָׁ֔הּ וַיַּמְלִיכֶ֖הָ תַּ֥חַת וַשְׁתִּֽי׃ הַמֶּלֶךְ אָהַב אֶת אֶסְתֵּר יוֹתֵר מִכָּל הַנָּשִׁים, וְהִיא מָצְאָה חֵן וְהָיְתָה חֲבִיבָה בְּעֵינָיו יוֹתֵר מִכָּל הַבְּתוּלוֹת, הוּא שָׂם כֶּתֶר מַלְכוּת עַל רֹאשָׁהּ וְהִמְלִיךְ אוֹתָהּ בִּמְקוֹם וַשְׁתִּי.
יח וַיַּ֨עַשׂ הַמֶּ֜לֶךְ מִשְׁתֶּ֣ה גָד֗וֹל לְכׇל־שָׂרָיו֙ וַעֲבָדָ֔יו אֵ֖ת מִשְׁתֵּ֣ה אֶסְתֵּ֑ר וַהֲנָחָ֤ה לַמְּדִינוֹת֙ עָשָׂ֔ה וַיִּתֵּ֥ן מַשְׂאֵ֖ת כְּיַ֥ד הַמֶּֽלֶךְ׃ הַמֶּלֶךְ עָשָׂה מְסִבַּת שְׁתִיַּת יַיִן גְּדוֹלָה לְכָל שָׂרָיו וַעֲבָדָיו לִכְבוֹד אֶסְתֵּר, וְהוּא עָשָׂה הֲנָחָה לַמְּדִינוֹת בְּתַשְׁלוּם הַמִּסִּים, וְנָתַן מַתָּנוֹת כְּיַד הַמֶּלֶךְ.
יט וּבְהִקָּבֵ֥ץ בְּתוּל֖וֹת שֵׁנִ֑ית וּמׇרְדֳּכַ֖י יֹשֵׁ֥ב בְּשַֽׁעַר־הַמֶּֽלֶךְ׃ גַּם כְּשֶׁאָסְפוּ בְּתוּלוֹת בְּפַעַם שְׁנִיָּה, מָרְדֳּכַי הָיָה יוֹשֵׁב בְּשַׁעַר הַמֶּלֶךְ,
כ אֵ֣ין אֶסְתֵּ֗ר מַגֶּ֤דֶת מֽוֹלַדְתָּהּ֙ וְאֶת־עַמָּ֔הּ כַּאֲשֶׁ֛ר צִוָּ֥ה עָלֶ֖יהָ מׇרְדֳּכָ֑י וְאֶת־מַאֲמַ֤ר מׇרְדֳּכַי֙ אֶסְתֵּ֣ר עֹשָׂ֔ה כַּאֲשֶׁ֛ר הָיְתָ֥ה בְאׇמְנָ֖ה אִתּֽוֹ׃ וְאֶסְתֵּר לֹא סִפְּרָה אֵיפֹה נוֹלְדָה וּמַהוּ עַמָּהּ כְּמוֹ שֶׁצִּוָּה עָלֶיהָ מָרְדֳּכַי, אֶסְתֵּר מִלְּאָה אֶת הַהוֹרָאוֹת שֶׁל מָרְדֳּכַי כְּמוֹ בַּזְּמָן שֶׁהָיְתָה מְאֻמֶּצֶת אֶצְלוֹ.

כא בַּיָּמִ֣ים הָהֵ֔ם וּמׇרְדֳּכַ֖י יוֹשֵׁ֣ב בְּשַֽׁעַר־הַמֶּ֑לֶךְ קָצַף֩ בִּגְתָ֨ן וָתֶ֜רֶשׁ שְׁנֵֽי־סָרִיסֵ֤י הַמֶּ֙לֶךְ֙ מִשֹּׁמְרֵ֣י הַסַּ֔ף וַיְבַקְשׁוּ֙ לִשְׁלֹ֣חַ יָ֔ד בַּמֶּ֖לֶךְ אֲחַשְׁוֵרֹֽשׁ׃ בַּיָּמִים הָהֵם שֶׁמָּרְדֳּכַי הָיָה יוֹשֵׁב בְּשַׁעַר הַמֶּלֶךְ כַּעֲסוּ בִּגְתָן וְתֶרֶשׁ שְׁנֵי מְשָׁרְתֵי הַמֶּלֶךְ מִשּׁוֹמְרֵי הַפֶּתַח, וְרָצוּ לְהִתְנַקֵּשׁ בַּמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ.
כב וַיִּוָּדַ֤ע הַדָּבָר֙ לְמׇרְדֳּכַ֔י וַיַּגֵּ֖ד לְאֶסְתֵּ֣ר הַמַּלְכָּ֑ה וַתֹּ֧אמֶר אֶסְתֵּ֛ר לַמֶּ֖לֶךְ בְּשֵׁ֥ם מׇרְדֳּכָֽי׃ הַדָּבָר נוֹדַע לְמָרְדֳּכַי וְהוּא סִפֵּר לְאֶסְתֵּר הַמַּלְכָּה, וְאֶסְתֵּר אָמְרָה לַמֶּלֶךְ בְּשֵׁם מָרְדֳּכָי.
כג וַיְבֻקַּ֤שׁ הַדָּבָר֙ וַיִּמָּצֵ֔א וַיִּתָּל֥וּ שְׁנֵיהֶ֖ם עַל־עֵ֑ץ וַיִּכָּתֵ֗ב בְּסֵ֛פֶר דִּבְרֵ֥י הַיָּמִ֖ים לִפְנֵ֥י הַמֶּֽלֶךְ׃ חָקְרוּ אֶת הַדָּבָר וְהוּא הִתְגַּלָּה, וּשְׁנֵיהֶם נִתְלוּ עַל עֵץ, וְזֶה נִכְתַּב בְּיוֹמַן הַזִּכְרוֹנוֹת שֶׁלִּפְנֵי הַמֶּלֶךְ.
פ
ג א אַחַ֣ר ׀ הַדְּבָרִ֣ים הָאֵ֗לֶּה גִּדַּל֩ הַמֶּ֨לֶךְ אֲחַשְׁוֵר֜וֹשׁ אֶת־הָמָ֧ן בֶּֽן־הַמְּדָ֛תָא הָאֲגָגִ֖י וַֽיְנַשְּׂאֵ֑הוּ וַיָּ֙שֶׂם֙ אֶת־כִּסְא֔וֹ מֵעַ֕ל כׇּל־הַשָּׂרִ֖ים אֲשֶׁ֥ר אִתּֽוֹ׃ אַחֲרֵי הַדְּבָרִים הָאֵלֶּה, הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ רוֹמֵם אֶת הָמָן בֶּן הַמְּדָתָא הַצֶּאֱצָא שֶׁל אֲגָג, וְכִבֵּד אוֹתוֹ, וְקָבַע אֶת הַדַּרְגָּה שֶׁלּוֹ מֵעַל כָּל הַשָּׂרִים שֶׁאֶצְלוֹ.
ב וְכׇל־עַבְדֵ֨י הַמֶּ֜לֶךְ אֲשֶׁר־בְּשַׁ֣עַר הַמֶּ֗לֶךְ כֹּרְעִ֤ים וּמִֽשְׁתַּחֲוִים֙ לְהָמָ֔ן כִּי־כֵ֖ן צִוָּה־ל֣וֹ הַמֶּ֑לֶךְ וּמׇ֨רְדֳּכַ֔י לֹ֥א יִכְרַ֖ע וְלֹ֥א יִֽשְׁתַּחֲוֶֽה׃ וְכָל עַבְדֵי הַמֶּלֶךְ שֶׁבְּשַׁעַר הַמֶּלֶךְ כּוֹרְעִים וּמִשְׁתַּחֲוִים לְהָמָן כִּי כָּךְ צִוָּה הַמֶּלֶךְ, וּמָרְדֳּכַי לֹא הָיָה כּוֹרֵע וְלֹא מִשְׁתַּחֲוֶה.
ג וַיֹּ֨אמְר֜וּ עַבְדֵ֥י הַמֶּ֛לֶךְ אֲשֶׁר־בְּשַׁ֥עַר הַמֶּ֖לֶךְ לְמׇרְדֳּכָ֑י מַדּ֙וּעַ֙ אַתָּ֣ה עוֹבֵ֔ר אֵ֖ת מִצְוַ֥ת הַמֶּֽלֶךְ׃ עַבְדֵי הַמֶּלֶךְ שֶׁבְּשַׁעַר הַמֶּלֶךְ אָמְרוּ לְמָרְדֳּכָי: מַדּוּעַ אַתָּה עוֹבֵר עַל מִצְוַת הַמֶּלֶךְ?
ד וַיְהִ֗י באמרם כְּאׇמְרָ֤ם אֵלָיו֙ י֣וֹם וָי֔וֹם וְלֹ֥א שָׁמַ֖ע אֲלֵיהֶ֑ם וַיַּגִּ֣ידוּ לְהָמָ֗ן לִרְאוֹת֙ הֲיַֽעַמְדוּ֙ דִּבְרֵ֣י מׇרְדֳּכַ֔י כִּֽי־הִגִּ֥יד לָהֶ֖ם אֲשֶׁר־ה֥וּא יְהוּדִֽי׃ הֵם הָיוּ אוֹמְרִים לוֹ כָּל יוֹם וְהוּא לֹא הִקְשִׁיב לָהֶם, הֵם סִפְּרוּ לְהָמָן כְּדֵי לִרְאוֹת אִם דִּבְרֵי מָרְדֳּכַי יַחֲזִיקוּ מַעֲמָד, כִּי הוּא הִסְבִּיר לָהֶם שֶׁאֵינוֹ כּוֹרֵעַ כִּי הוּא יְהוּדִי.
ה וַיַּ֣רְא הָמָ֔ן כִּי־אֵ֣ין מׇרְדֳּכַ֔י כֹּרֵ֥עַ וּמִֽשְׁתַּחֲוֶ֖ה ל֑וֹ וַיִּמָּלֵ֥א הָמָ֖ן חֵמָֽה׃ הָמָן רָאָה שֶׁמָרְדֳּכַי אֵינוֹ כּוֹרֵעַ וּמִשְׁתַּחֲוֶה לוֹ וְהוּא הִתְמַלֵּא בְּכַעַס.
ו וַיִּ֣בֶז בְּעֵינָ֗יו לִשְׁלֹ֤חַ יָד֙ בְּמׇרְדֳּכַ֣י לְבַדּ֔וֹ כִּֽי־הִגִּ֥ידוּ ל֖וֹ אֶת־עַ֣ם מׇרְדֳּכָ֑י וַיְבַקֵּ֣שׁ הָמָ֗ן לְהַשְׁמִ֧יד אֶת־כׇּל־הַיְּהוּדִ֛ים אֲשֶׁ֛ר בְּכׇל־מַלְכ֥וּת אֲחַשְׁוֵר֖וֹשׁ עַ֥ם מׇרְדֳּכָֽי׃ הָיָה בָּזוּי בְּעֵינָיו לִפְגּוֹעַ רַק בְּמָרְדֳּכַי כִּי סִפְּרוּ לוֹ מִי הָעָם שֶׁל מָרְדֳּכָי, הָמָן רָצָה לְהַשְׁמִיד אֶת כָּל הַיְּהוּדִים – הָעָם שֶׁל מָרְדֳּכָי, שֶׁבְּכָל מַלְכוּת אֲחַשְׁוֵרוֹשׁ.
ז בַּחֹ֤דֶשׁ הָרִאשׁוֹן֙ הוּא־חֹ֣דֶשׁ נִיסָ֔ן בִּשְׁנַת֙ שְׁתֵּ֣ים עֶשְׂרֵ֔ה לַמֶּ֖לֶךְ אֲחַשְׁוֵר֑וֹשׁ הִפִּ֣יל פּוּר֩ ה֨וּא הַגּוֹרָ֜ל לִפְנֵ֣י הָמָ֗ן מִיּ֧וֹם ׀ לְי֛וֹם וּמֵחֹ֛דֶשׁ לְחֹ֥דֶשׁ שְׁנֵים־עָשָׂ֖ר הוּא־חֹ֥דֶשׁ אֲדָֽר׃ בַּחוֹדֶשׁ הָרִאשׁוֹן הוּא חוֹדֶשׁ נִיסָן, בִּשְׁנַת שְׁתֵּים עֶשְׂרֵה לַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ הוּא עָשָׂה הַגְרָלָה, זֶה הַגּוֹרָל שֶׁעָשָׂה הָמָן, בְּאֵיזֶה יוֹם וּבְאֵיזֶה חוֹדֶשׁ, וְיָצָא הַחוֹדֶשׁ הַשְּׁנֵים עָשָׂר הוּא חוֹדֶשׁ אֲדָר.
ס
ח וַיֹּ֤אמֶר הָמָן֙ לַמֶּ֣לֶךְ אֲחַשְׁוֵר֔וֹשׁ יֶשְׁנ֣וֹ עַם־אֶחָ֗ד מְפֻזָּ֤ר וּמְפֹרָד֙ בֵּ֣ין הָֽעַמִּ֔ים בְּכֹ֖ל מְדִינ֣וֹת מַלְכוּתֶ֑ךָ וְדָתֵיהֶ֞ם שֹׁנ֣וֹת מִכׇּל־עָ֗ם וְאֶת־דָּתֵ֤י הַמֶּ֙לֶךְ֙ אֵינָ֣ם עֹשִׂ֔ים וְלַמֶּ֥לֶךְ אֵין־שֹׁוֶ֖ה לְהַנִּיחָֽם׃ אָמַר הָמָן לַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ: יֵשׁ עַם אֶחָד מְפֻזָּר וּמְפוֹרָד בֵּין הָעַמִּים בְּכָל מְדִינוֹת מַמְלַכְתְּךָ, וְחֻקֵּיהֶם שׁוֹנִים מִשֶּׁל כָּל עָם, וְהֵם אֵינָם עוֹשִׂים אֶת חֻקֵּי הַמֶּלֶךְ, וְלַמֶּלֶךְ לֹא מִשְׁתַּלֵּם לְהַשְׁאִיר אוֹתָם.
ט אִם־עַל־הַמֶּ֣לֶךְ ט֔וֹב יִכָּתֵ֖ב לְאַבְּדָ֑ם וַעֲשֶׂ֨רֶת אֲלָפִ֜ים כִּכַּר־כֶּ֗סֶף אֶשְׁקוֹל֙ עַל־יְדֵי֙ עֹשֵׂ֣י הַמְּלָאכָ֔ה לְהָבִ֖יא אֶל־גִּנְזֵ֥י הַמֶּֽלֶךְ׃ אִם זֶה רַעֲיוֹן טוֹב בְּעֵינֵי הַמֶּלֶךְ יִכָּתֵב צַו לְהַשְׁמִיד אוֹתָם, וְאֶשְׁקוֹל עֲשֶׂרֶת אֲלָפִים כִּכְּרוֹת כֶּסֶף וְאֶתֵּן לַמְּמוּנִים שֶׁיָּבִיאוּ אוֹתָם לְאוֹצַר הַמֶּלֶךְ.
י וַיָּ֧סַר הַמֶּ֛לֶךְ אֶת־טַבַּעְתּ֖וֹ מֵעַ֣ל יָד֑וֹ וַֽיִּתְּנָ֗הּ לְהָמָ֧ן בֶּֽן־הַמְּדָ֛תָא הָאֲגָגִ֖י צֹרֵ֥ר הַיְּהוּדִֽים׃ הַמֶּלֶךְ הֵסִיר אֶת טַבַּעַת הַחוֹתֶמֶת שֶׁלּוֹ מִיָּדוֹ וּמָסַר אוֹתָהּ לְהָמָן בֶּן הַמְּדָתָא הַצֶּאֱצָא שֶׁל אֲגָג, אוֹיֵב הַיְּהוּדִים.
יא וַיֹּ֤אמֶר הַמֶּ֙לֶךְ֙ לְהָמָ֔ן הַכֶּ֖סֶף נָת֣וּן לָ֑ךְ וְהָעָ֕ם לַעֲשׂ֥וֹת בּ֖וֹ כַּטּ֥וֹב בְּעֵינֶֽיךָ׃ אָמַר הַמֶּלֶךְ לְהָמָן: הַכֶּסֶף נָתוּן לְךָ, וְגַם הָעָם – לַעֲשׂוֹת בּוֹ כִּרְצוֹנְךָ.
יב וַיִּקָּרְאוּ֩ סֹפְרֵ֨י הַמֶּ֜לֶךְ בַּחֹ֣דֶשׁ הָרִאשׁ֗וֹן בִּשְׁלוֹשָׁ֨ה עָשָׂ֣ר יוֹם֮ בּוֹ֒ וַיִּכָּתֵ֣ב כְּֽכׇל־אֲשֶׁר־צִוָּ֣ה הָמָ֡ן אֶ֣ל אֲחַשְׁדַּרְפְּנֵֽי־הַ֠מֶּ֠לֶךְ וְֽאֶל־הַפַּח֞וֹת אֲשֶׁ֣ר ׀ עַל־מְדִינָ֣ה וּמְדִינָ֗ה וְאֶל־שָׂ֤רֵי עַם֙ וָעָ֔ם מְדִינָ֤ה וּמְדִינָה֙ כִּכְתָבָ֔הּ וְעַ֥ם וָעָ֖ם כִּלְשׁוֹנ֑וֹ בְּשֵׁ֨ם הַמֶּ֤לֶךְ אֲחַשְׁוֵרֹשׁ֙ נִכְתָּ֔ב וְנֶחְתָּ֖ם בְּטַבַּ֥עַת הַמֶּֽלֶךְ׃ סוֹפְרֵי הַמֶּלֶךְ הֻזְמְנוּ בְּי"ג לַחוֹדֶשׁ הָרִאשׁוֹן וְהֵם כָּתְבוּ אֶת כָּל מַה שֶּׁצִּוָּה הָמָן אֶל נְצִיבֵי הַמֶּלֶךְ וְאֶל הַמּוֹשְׁלִים שֶׁל כָּל מְדִינָה וּמְדִינָה וְאֶל שָׂרֵי כָּל עַם וְעַם, לְכָל מְדִינָה וּמְדִינָה בִּכְתָבָהּ וּלְכָל עָם וְעָם בִּשְׂפָתוֹ, נִכְתָּב בְּשֵׁם הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ וְנֶחְתָּם בְּחוֹתֶמֶת הַמֶּלֶךְ.
יג וְנִשְׁל֨וֹחַ סְפָרִ֜ים בְּיַ֣ד הָרָצִים֮ אֶל־כׇּל־מְדִינ֣וֹת הַמֶּ֒לֶךְ֒ לְהַשְׁמִ֡יד לַהֲרֹ֣ג וּלְאַבֵּ֣ד אֶת־כׇּל־הַ֠יְּהוּדִ֠ים מִנַּ֨עַר וְעַד־זָקֵ֜ן טַ֤ף וְנָשִׁים֙ בְּי֣וֹם אֶחָ֔ד בִּשְׁלוֹשָׁ֥ה עָשָׂ֛ר לְחֹ֥דֶשׁ שְׁנֵים־עָשָׂ֖ר הוּא־חֹ֣דֶשׁ אֲדָ֑ר וּשְׁלָלָ֖ם לָבֽוֹז׃ וְשָׁלְחוּ מִכְתָּבִים בִּידֵי הָרָצִים אֶל כָּל מְדִינוֹת הַמֶּלֶךְ, וְכָתוּב בָּהֶם לְהַשְׁמִיד לַהֲרוֹג וּלְאַבֵּד אֶת כָּל הַיְּהוּדִים מִנַּעַר וְעַד זָקֵן, יְלָדִים וְנָשִׁים, בְּיוֹם אֶחָד – בְּיוֹם י"ג לְחוֹדֶשׁ שְׁנֵים עָשָׂר הוּא חוֹדֶשׁ אֲדָר, וְלָקַחַת אֶת שְׁלָלָם.
יד פַּתְשֶׁ֣גֶן הַכְּתָ֗ב לְהִנָּ֤תֵֽן דָּת֙ בְּכׇל־מְדִינָ֣ה וּמְדִינָ֔ה גָּל֖וּי לְכׇל־הָֽעַמִּ֑ים לִהְי֥וֹת עֲתִדִ֖ים לַיּ֥וֹם הַזֶּֽה׃ תּוֹכֶן הַמִּכְתָב: לִקְבּוֹעַ בְּכָל מְדִינָה חוֹק וּלְפַרְסֵם אֵצֶל כָּל הָעַמִּים, שֶׁיִּהְיוּ מוּכָנִים לַיּוֹם הַזֶּה.
טו הָֽרָצִ֞ים יָצְא֤וּ דְחוּפִים֙ בִּדְבַ֣ר הַמֶּ֔לֶךְ וְהַדָּ֥ת נִתְּנָ֖ה בְּשׁוּשַׁ֣ן הַבִּירָ֑ה וְהַמֶּ֤לֶךְ וְהָמָן֙ יָשְׁב֣וּ לִשְׁתּ֔וֹת וְהָעִ֥יר שׁוּשָׁ֖ן נָבֽוֹכָה׃ הָרָצִים יָצְאוּ בִּדְחִיפוּת לַעֲשׂוֹת אֶת דְּבַר הַמֶּלֶךְ, וְנִקְבַּע הַחוֹק בְּשׁוּשַׁן הַבִּירָה, וְהַמֶּלֶךְ וְהָמָן יָשְׁבוּ לִשְׁתּוֹת יַיִן וּבָעִיר שׁוּשָׁן הָיוּ נְבוֹכִים.
ס
ד א וּמׇרְדֳּכַ֗י יָדַע֙ אֶת־כׇּל־אֲשֶׁ֣ר נַעֲשָׂ֔ה וַיִּקְרַ֤ע מׇרְדֳּכַי֙ אֶת־בְּגָדָ֔יו וַיִּלְבַּ֥שׁ שַׂ֖ק וָאֵ֑פֶר וַיֵּצֵא֙ בְּת֣וֹךְ הָעִ֔יר וַיִּזְעַ֛ק זְעָקָ֥ה גְדוֹלָ֖ה וּמָרָֽה׃ וּמָרְדֳּכַי יָדַע אֶת כָּל מַה שֶׁקָּרָה, מָרְדֳּכַי קָרַע אֶת בְּגָדָיו וְלָבַשׁ בֶּגֶד גַּס מִשֵּׂעָר עִזִּים שְׁחוֹרוֹת וְשָׂם אֵפֶר עַל רֹאשׁוֹ, וְיָצָא בְּתוֹךְ הָעִיר וְזָעַק זְעָקָה גְּדוֹלָה וּמָרָה.
ב וַיָּב֕וֹא עַ֖ד לִפְנֵ֣י שַֽׁעַר־הַמֶּ֑לֶךְ כִּ֣י אֵ֥ין לָב֛וֹא אֶל־שַׁ֥עַר הַמֶּ֖לֶךְ בִּלְב֥וּשׁ שָֽׂק׃ הוּא בָּא עַד לִפְנֵי שַׁעַר הַמֶּלֶךְ, כִּי אָסוּר לְהִכָּנֵס לְשַׁעַר הַמֶּלֶךְ בִּלְבוּשׁ גַּס.
ג וּבְכׇל־מְדִינָ֣ה וּמְדִינָ֗ה מְקוֹם֙ אֲשֶׁ֨ר דְּבַר־הַמֶּ֤לֶךְ וְדָתוֹ֙ מַגִּ֔יעַ אֵ֤בֶל גָּדוֹל֙ לַיְּהוּדִ֔ים וְצ֥וֹם וּבְכִ֖י וּמִסְפֵּ֑ד שַׂ֣ק וָאֵ֔פֶר יֻצַּ֖ע לָֽרַבִּֽים׃ וּבְכָל מְדִינָה וּמְדִינָה בְּמָקוֹם שֶׁפְּקֻדַּת הַמֶּלֶךְ וּגְזֵרָתוֹ מַגִּיעָה, אֵבֶל גָּדוֹל לַיְּהוּדִים וְצוֹם וּבְכִי וְקִינוֹת, וְרַבִּים שׁוֹכְבִים עַל בַּד גַּס מִשֵּׂעָר עִזִּים שְׁחוֹרוֹת וְעַל אֵפֶר.
ד ותבואינה וַ֠תָּב֠וֹאנָה נַעֲר֨וֹת אֶסְתֵּ֤ר וְסָרִיסֶ֙יהָ֙ וַיַּגִּ֣ידוּ לָ֔הּ וַתִּתְחַלְחַ֥ל הַמַּלְכָּ֖ה מְאֹ֑ד וַתִּשְׁלַ֨ח בְּגָדִ֜ים לְהַלְבִּ֣ישׁ אֶֽת־מׇרְדֳּכַ֗י וּלְהָסִ֥יר שַׂקּ֛וֹ מֵעָלָ֖יו וְלֹ֥א קִבֵּֽל׃ נַעֲרוֹת אֶסְתֵּר וְהַמְּשָׁרְתִים שֶׁלָּהּ בָּאוּ וְסִפְּרוּ לָהּ וְהַמַּלְכָּה הִזְדַּעַזְעָה מְאוֹד, הִיא שָׁלְחָה בְּגָדִים לְהַלְבִּישׁ אֶת מָרְדֳּכַי וְשֶׁיָּסִיר אֶת הַבֶּגֶד הַגַּס מֵעָלָיו, וְהוּא לֹא הִסְכִּים לְקַבֵּל.
ה וַתִּקְרָא֩ אֶסְתֵּ֨ר לַהֲתָ֜ךְ מִסָּרִיסֵ֤י הַמֶּ֙לֶךְ֙ אֲשֶׁ֣ר הֶעֱמִ֣יד לְפָנֶ֔יהָ וַתְּצַוֵּ֖הוּ עַֽל־מׇרְדֳּכָ֑י לָדַ֥עַת מַה־זֶּ֖ה וְעַל־מַה־זֶּֽה׃ אֶסְתֵּר קָרְאָה לַהֲתָךְ – אֶחָד מִשָּׂרֵי הַמֶּלֶךְ שֶׁהוּא הֶעֱמִיד לְשֵׁרוּתָהּ, הִיא צִוְּתָה עָלָיו לִשְׁאוֹל אֶת מָרְדֳּכָי כְּדֵי לָדַעַת מַה זֶּה וְעַל מַה זֶּה.
ו וַיֵּצֵ֥א הֲתָ֖ךְ אֶֽל־מׇרְדֳּכָ֑י אֶל־רְח֣וֹב הָעִ֔יר אֲשֶׁ֖ר לִפְנֵ֥י שַֽׁעַר־הַמֶּֽלֶךְ׃ הֲתָךְ יָצָא אֶל מָרְדֳּכָי אֶל רַחֲבַת הָעִיר שֶׁלִּפְנֵי שַׁעַר הַמֶּלֶךְ.
ז וַיַּגֶּד־ל֣וֹ מׇרְדֳּכַ֔י אֵ֖ת כׇּל־אֲשֶׁ֣ר קָרָ֑הוּ וְאֵ֣ת ׀ פָּרָשַׁ֣ת הַכֶּ֗סֶף אֲשֶׁ֨ר אָמַ֤ר הָמָן֙ לִ֠שְׁק֠וֹל עַל־גִּנְזֵ֥י הַמֶּ֛לֶךְ ביהודיים בַּיְּהוּדִ֖ים לְאַבְּדָֽם׃ מָרְדֳּכַי סִפֵּר לוֹ אֶת כָּל מַה שֶּׁקָּרָה לוֹ בִּגְלַל שֶׁלֹּא הִשְׁתַּחֲוָה לְהָמָן, וְאֶת סְכוּם הַכֶּסֶף שֶׁהָמָן אָמַר שֶׁיָּבִיא לְאוֹצְרוֹת הַמֶּלֶךְ בִּשְׁבִיל לְהַשְׁמִיד אֶת הַיְּהוּדִים.
ח וְאֶת־פַּתְשֶׁ֣גֶן כְּתָֽב־הַ֠דָּ֠ת אֲשֶׁר־נִתַּ֨ן בְּשׁוּשָׁ֤ן לְהַשְׁמִידָם֙ נָ֣תַן ל֔וֹ לְהַרְא֥וֹת אֶת־אֶסְתֵּ֖ר וּלְהַגִּ֣יד לָ֑הּ וּלְצַוּ֣וֹת עָלֶ֗יהָ לָב֨וֹא אֶל־הַמֶּ֧לֶךְ לְהִֽתְחַנֶּן־ל֛וֹ וּלְבַקֵּ֥שׁ מִלְּפָנָ֖יו עַל־עַמָּֽהּ׃ וְנָתַן לוֹ עוֹתֶק מִכְּתַב הַחוֹק שֶׁנִּקְבַּע בְּשׁוּשָׁן לְהַשְׁמִיד אוֹתָם, לְהַרְאוֹת לְאֶסְתֵּר וּלְסַפֵּר לָהּ, וּלְצַוּוֹת עָלֶיהָ לָבוֹא אֶל הַמֶּלֶךְ לְהִתְחַנֵּן אֵלָיו וּלְבַקֵּשׁ מִמֶּנּוּ עַל עַמָּהּ.
ט וַיָּב֖וֹא הֲתָ֑ךְ וַיַּגֵּ֣ד לְאֶסְתֵּ֔ר אֵ֖ת דִּבְרֵ֥י מׇרְדֳּכָֽי׃ הֲתָךְ בָּא וְסִפֵּר לְאֶסְתֵּר אֶת דִּבְרֵי מָרְדֳּכָי.
י וַתֹּ֤אמֶר אֶסְתֵּר֙ לַהֲתָ֔ךְ וַתְּצַוֵּ֖הוּ אֶֽל־מׇרְדֳּכָֽי׃ אָמְרָה אֶסְתֵּר לַהֲתָךְ וְצִוְּתָה אוֹתוֹ לוֹמַר לְמָרְדֳּכָי:
יא כׇּל־עַבְדֵ֣י הַמֶּ֡לֶךְ וְעַם־מְדִינ֨וֹת הַמֶּ֜לֶךְ יֹֽדְעִ֗ים אֲשֶׁ֣ר כׇּל־אִ֣ישׁ וְאִשָּׁ֡ה אֲשֶׁ֣ר יָבֽוֹא־אֶל־הַמֶּ֩לֶךְ֩ אֶל־הֶחָצֵ֨ר הַפְּנִימִ֜ית אֲשֶׁ֣ר לֹֽא־יִקָּרֵ֗א אַחַ֤ת דָּתוֹ֙ לְהָמִ֔ית לְ֠בַ֠ד מֵאֲשֶׁ֨ר יֽוֹשִׁיט־ל֥וֹ הַמֶּ֛לֶךְ אֶת־שַׁרְבִ֥יט הַזָּהָ֖ב וְחָיָ֑ה וַאֲנִ֗י לֹ֤א נִקְרֵ֙אתִי֙ לָב֣וֹא אֶל־הַמֶּ֔לֶךְ זֶ֖ה שְׁלוֹשִׁ֥ים יֽוֹם׃ כָּל עַבְדֵי הַמֶּלֶךְ וְעַמֵּי מְדִינוֹת הַמֶּלֶךְ יוֹדְעִים שֶׁכָּל אִישׁ אוֹ אִשָּׁה שֶׁיָּבוֹאוּ אֶל הַמֶּלֶךְ אֶל הֶחָצֵר הַפְּנִימִית בְּלִי שֶׁהֻזְמַן דִּינוֹ אֶחָד – לְהֵהָרֵג, אֶלָּא אִם כֵּן הַמֶּלֶךְ יוֹשִׁיט לוֹ אֶת שַׁרְבִיט הַזָּהָב וְהוּא יִחְיֶה, וַאֲנִי לֹא הֻזְמַנְתִּי לָבוֹא אֶל הַמֶּלֶךְ כְּבָר שְׁלוֹשִׁים יוֹם.
יב וַיַּגִּ֣ידוּ לְמׇרְדֳּכָ֔י אֵ֖ת דִּבְרֵ֥י אֶסְתֵּֽר׃ סִפְּרוּ לְמָרְדֳּכָי אֵת דִּבְרֵי אֶסְתֵּר.
פ
יג וַיֹּ֥אמֶר מׇרְדֳּכַ֖י לְהָשִׁ֣יב אֶל־אֶסְתֵּ֑ר אַל־תְּדַמִּ֣י בְנַפְשֵׁ֔ךְ לְהִמָּלֵ֥ט בֵּית־הַמֶּ֖לֶךְ מִכׇּל־הַיְּהוּדִֽים׃ מָרְדֳּכַי אָמַר שֶׁיַּעֲנוּ לְאֶסְתֵּר: אַל תַּחְשְׁבִי לְעַצְמֵךְ לְהִמָּלֵט בְּבֵית הַמֶּלֶךְ וּלְהִנָּצֵל מִמַּה שֶׁיִּקְרֶה לְכָל הַיְּהוּדִים.
יד כִּ֣י אִם־הַחֲרֵ֣שׁ תַּחֲרִ֘ישִׁי֮ בָּעֵ֣ת הַזֹּאת֒ רֶ֣וַח וְהַצָּלָ֞ה יַעֲמ֤וֹד לַיְּהוּדִים֙ מִמָּק֣וֹם אַחֵ֔ר וְאַ֥תְּ וּבֵית־אָבִ֖יךְ תֹּאבֵ֑דוּ וּמִ֣י יוֹדֵ֔עַ אִם־לְעֵ֣ת כָּזֹ֔את הִגַּ֖עַתְּ לַמַּלְכֽוּת׃ כִּי אִם תִּשְׁתְּקִי עַכְשָׁיו וְלֹא תְּבַקְּשִׁי עֲבוּר הַיְּהוּדִים, תַּגִּיעַ רְוָחָה וְהַצָּלָה לַיְּהוּדִים מִמָּקוֹם אַחֵר וְאַתְּ וּמִשְׁפַּחְתֵּךְ תֻּשְׁמְדוּ, וּמִי יוֹדֵעַ אִם לֹא בִּשְׁבִיל הִזְדַּמְּנוּת זוֹ הִגַּעַתְּ לַמַּלְכוּת.
טו וַתֹּ֥אמֶר אֶסְתֵּ֖ר לְהָשִׁ֥יב אֶֽל־מׇרְדֳּכָֽי׃ אָמְרָה אֶסְתֵּר לַעֲנוֹת לְמָרְדֳּכָי:
טז לֵךְ֩ כְּנ֨וֹס אֶת־כׇּל־הַיְּהוּדִ֜ים הַֽנִּמְצְאִ֣ים בְּשׁוּשָׁ֗ן וְצ֣וּמוּ עָ֠לַ֠י וְאַל־תֹּאכְל֨וּ וְאַל־תִּשְׁתּ֜וּ שְׁלֹ֤שֶׁת יָמִים֙ לַ֣יְלָה וָי֔וֹם גַּם־אֲנִ֥י וְנַעֲרֹתַ֖י אָצ֣וּם כֵּ֑ן וּבְכֵ֞ן אָב֤וֹא אֶל־הַמֶּ֙לֶךְ֙ אֲשֶׁ֣ר לֹֽא־כַדָּ֔ת וְכַאֲשֶׁ֥ר אָבַ֖דְתִּי אָבָֽדְתִּי׃ לֵךְ תְּכַנֵּס אֶת כָּל הַיְּהוּדִים שֶׁנִּמְצְאִים בְּשׁוּשָׁן וְתָצוּמוּ עָלַי, אַל תֹּאכְלוּ וְאַל תִּשְׁתּוּ שְׁלוֹשָׁה יָמִים בַּלַּיְלָה וּבַיּוֹם, גַּם אֲנִי וְהַמְּשָׁרְתוֹת שֶׁלִּי נָצוּם כָּךְ, וְכָךְ אָבוֹא לַמֶּלֶךְ שֶׁלֹּא כָּחוֹק, וְאִם אָמוּת - אָמוּת.
יז וַֽיַּעֲבֹ֖ר מׇרְדֳּכָ֑י וַיַּ֕עַשׂ כְּכֹ֛ל אֲשֶׁר־צִוְּתָ֥ה עָלָ֖יו אֶסְתֵּֽר׃ מָרְדֳּכָי עָבַר בְּשׁוּשָׁן וְעָשָׂה אֶת כָּל מַה שֶּׁצִּוְּתָה עָלָיו אֶסְתֵּר.
ס
ה א וַיְהִ֣י ׀ בַּיּ֣וֹם הַשְּׁלִישִׁ֗י וַתִּלְבַּ֤שׁ אֶסְתֵּר֙ מַלְכ֔וּת וַֽתַּעֲמֹ֞ד בַּחֲצַ֤ר בֵּית־הַמֶּ֙לֶךְ֙ הַפְּנִימִ֔ית נֹ֖כַח בֵּ֣ית הַמֶּ֑לֶךְ וְ֠הַמֶּ֠לֶךְ יוֹשֵׁ֞ב עַל־כִּסֵּ֤א מַלְכוּתוֹ֙ בְּבֵ֣ית הַמַּלְכ֔וּת נֹ֖כַח פֶּ֥תַח הַבָּֽיִת׃ בַּיּוֹם הַשְּׁלִישִׁי לָבְשָׁה אֶסְתֵּר בִּגְדֵי מַלְכוּת וְעָמְדָה בֶּחָצֵר הַפְּנִימִית שֶׁל בֵּית הַמֶּלֶךְ מוּל הֵיכַל הַמֶּלֶךְ, וְהַמֶּלֶךְ יָשַׁב עַל כִּסֵּא מַלְכוּתוֹ בְּהֵיכַל הַמַּלְכוּת מוּל פֶּתַח הַהֵיכָל.
ב וַיְהִי֩ כִרְא֨וֹת הַמֶּ֜לֶךְ אֶת־אֶסְתֵּ֣ר הַמַּלְכָּ֗ה עֹמֶ֙דֶת֙ בֶּֽחָצֵ֔ר נָשְׂאָ֥ה חֵ֖ן בְּעֵינָ֑יו וַיּ֨וֹשֶׁט הַמֶּ֜לֶךְ לְאֶסְתֵּ֗ר אֶת־שַׁרְבִ֤יט הַזָּהָב֙ אֲשֶׁ֣ר בְּיָד֔וֹ וַתִּקְרַ֣ב אֶסְתֵּ֔ר וַתִּגַּ֖ע בְּרֹ֥אשׁ הַשַּׁרְבִֽיט׃ כְּשֶׁהַמֶּלֶךְ רָאָה אֶת אֶסְתֵּר הַמַּלְכָּה עוֹמֶדֶת בֶּחָצֵר הִיא מָצְאָה חֵן בְּעֵינָיו, הַמֶּלֶךְ הוֹשִׁיט לְאֶסְתֵּר אֶת שַׁרְבִיט הַזָּהָב שֶׁבְּיָדוֹ, אֶסְתֵּר הִתְקָרְבָה וְנָגְעָה בִּקְצֵה הַשַּׁרְבִיט.
ס
ג וַיֹּ֤אמֶר לָהּ֙ הַמֶּ֔לֶךְ מַה־לָּ֖ךְ אֶסְתֵּ֣ר הַמַּלְכָּ֑ה וּמַה־בַּקָּשָׁתֵ֛ךְ עַד־חֲצִ֥י הַמַּלְכ֖וּת וְיִנָּ֥תֵֽן לָֽךְ׃ אָמַר לָהּ הַמֶּלֶךְ: מַה לָּךְ אֶסְתֵּר הַמַּלְכָּה, וּמַה בַּקָּשָׁתֵךְ? גַּם אִם תְּבַקְּשִׁי אֶת חֲצִי הַמַּמְלָכָה זֶה יִנָּתֵן לָךְ!
ד וַתֹּ֣אמֶר אֶסְתֵּ֔ר אִם־עַל־הַמֶּ֖לֶךְ ט֑וֹב יָב֨וֹא הַמֶּ֤לֶךְ וְהָמָן֙ הַיּ֔וֹם אֶל־הַמִּשְׁתֶּ֖ה אֲשֶׁר־עָשִׂ֥יתִי לֽוֹ׃ אָמְרָה אֶסְתֵּר: אִם הַמֶּלֶךְ מַסְכִּים, יָבוֹאוּ הַמֶּלֶךְ וְהָמָן הַיּוֹם לִמְסִבַּת שְׁתִיַּת הַיַּיִן שֶׁהֵכַנְתִּי לוֹ.
ה וַיֹּ֣אמֶר הַמֶּ֔לֶךְ מַהֲרוּ֙ אֶת־הָמָ֔ן לַעֲשׂ֖וֹת אֶת־דְּבַ֣ר אֶסְתֵּ֑ר וַיָּבֹ֤א הַמֶּ֙לֶךְ֙ וְהָמָ֔ן אֶל־הַמִּשְׁתֶּ֖ה אֲשֶׁר־עָשְׂתָ֥ה אֶסְתֵּֽר׃ אָמַר הַמֶּלֶךְ, מַהֲרוּ לִקְרוֹא לְהָמָן כְּדֵי לַעֲשׂוֹת אֶת דִּבְרֵי אֶסְתֵּר, הַמֶּלֶךְ וְהָמָן בָּאוּ לִמְסִבַּת שְׁתִיַּת הַיַּיִן שֶׁאֶסְתֵּר הֵכִינָה.
ו וַיֹּ֨אמֶר הַמֶּ֤לֶךְ לְאֶסְתֵּר֙ בְּמִשְׁתֵּ֣ה הַיַּ֔יִן מַה־שְּׁאֵלָתֵ֖ךְ וְיִנָּ֣תֵֽן לָ֑ךְ וּמַה־בַּקָּשָׁתֵ֛ךְ עַד־חֲצִ֥י הַמַּלְכ֖וּת וְתֵעָֽשׂ׃ אָמַר הַמֶּלֶךְ לְאֶסְתֵּר בִּמְסִבַּת שְׁתִיַּת הַיַּיִן: מַה רְצוֹנֵךְ? הוּא יִנָּתֵן לָךְ! מַה שֶּׁתְּבַקְּשִׁי גַּם עַד חֲצִי הַמַּמְלָכָה יִתְבַּצֵּעַ!
ז וַתַּ֥עַן אֶסְתֵּ֖ר וַתֹּאמַ֑ר שְׁאֵלָתִ֖י וּבַקָּשָׁתִֽי׃ עָנְתָה אֶסְתֵּר וְאָמְרָה, זֹאת בַּקָּשָׁתִי:
ח אִם־מָצָ֨אתִי חֵ֜ן בְּעֵינֵ֣י הַמֶּ֗לֶךְ וְאִם־עַל־הַמֶּ֙לֶךְ֙ ט֔וֹב לָתֵת֙ אֶת־שְׁאֵ֣לָתִ֔י וְלַעֲשׂ֖וֹת אֶת־בַּקָּשָׁתִ֑י יָב֧וֹא הַמֶּ֣לֶךְ וְהָמָ֗ן אֶל־הַמִּשְׁתֶּה֙ אֲשֶׁ֣ר אֶֽעֱשֶׂ֣ה לָהֶ֔ם וּמָחָ֥ר אֶֽעֱשֶׂ֖ה כִּדְבַ֥ר הַמֶּֽלֶךְ׃ אִם מָצָאתִי חֵן בְּעֵינֵי הַמֶּלֶךְ וְאִם הַמֶּלֶךְ מַסְכִּים לָתֵת לִי אֶת בַּקָּשָׁתִי, יָבוֹאוּ הַמֶּלֶךְ וְהָמָן לִמְסִבַּת שְׁתִיַּת הַיַּיִן שֶׁאֶעֱשֶׂה לָהֶם, וְאֶעֱשֶׂה אוֹתוֹ מָחָר אִם הַמֶּלֶךְ יַסְכִּים.
ט וַיֵּצֵ֤א הָמָן֙ בַּיּ֣וֹם הַה֔וּא שָׂמֵ֖חַ וְט֣וֹב לֵ֑ב וְכִרְאוֹת֩ הָמָ֨ן אֶֽת־מׇרְדֳּכַ֜י בְּשַׁ֣עַר הַמֶּ֗לֶךְ וְלֹא־קָם֙ וְלֹא־זָ֣ע מִמֶּ֔נּוּ וַיִּמָּלֵ֥א הָמָ֛ן עַֽל־מׇרְדֳּכַ֖י חֵמָֽה׃ הָמָן יָצָא בַּיּוֹם הַהוּא שָׂמֵחַ וְטוֹב לֵב, וּכְשֶׁהָמָן רָאָה אֶת מָרְדֳּכַי בְּשַׁעַר הַמֶּלֶךְ שֶׁאֵינוֹ קָם וְלֹא רוֹעֵד מִמֶּנּוּ, הָמָן הִתְמַלֵּא בְּכַעַס עַל מָרְדֳּכַי.
י וַיִּתְאַפַּ֣ק הָמָ֔ן וַיָּב֖וֹא אֶל־בֵּית֑וֹ וַיִּשְׁלַ֛ח וַיָּבֵ֥א אֶת־אֹהֲבָ֖יו וְאֶת־זֶ֥רֶשׁ אִשְׁתּֽוֹ׃ הָמָן הִתְאַפֵּק שֶׁלֹּא לְהִתְפָּרֵץ וּבָא לְבֵיתוֹ, וְשָׁלַח שֶׁיִּקְרְאוּ לַחֲבֵרָיו וּלְזֶרֶשׁ אִשְׁתּוֹ.
יא וַיְסַפֵּ֨ר לָהֶ֥ם הָמָ֛ן אֶת־כְּב֥וֹד עׇשְׁר֖וֹ וְרֹ֣ב בָּנָ֑יו וְאֵת֩ כׇּל־אֲשֶׁ֨ר גִּדְּל֤וֹ הַמֶּ֙לֶךְ֙ וְאֵ֣ת אֲשֶׁ֣ר נִשְּׂא֔וֹ עַל־הַשָּׂרִ֖ים וְעַבְדֵ֥י הַמֶּֽלֶךְ׃ הָמָן סִפֵּר לָהֶם עַל כְּבוֹדוֹ וְעָשְׁרוֹ וּבָנָיו הָרַבִּים וְכָל מַה שֶּׁהַמֶּלֶךְ רוֹמֵם אוֹתוֹ וְשֶׁקָּבַע אֶת דַּרְגָּתוֹ מֵעַל הַשָּׂרִים וְעַבְדֵי הַמֶּלֶךְ.
יב וַיֹּ֘אמֶר֮ הָמָן֒ אַ֣ף לֹא־הֵבִ֩יאָה֩ אֶסְתֵּ֨ר הַמַּלְכָּ֧ה עִם־הַמֶּ֛לֶךְ אֶל־הַמִּשְׁתֶּ֥ה אֲשֶׁר־עָשָׂ֖תָה כִּ֣י אִם־אוֹתִ֑י וְגַם־לְמָחָ֛ר אֲנִ֥י קָֽרוּא־לָ֖הּ עִם־הַמֶּֽלֶךְ׃ אָמַר הָמָן: אֲפִלּוּ אֶסְתֵּר הַמַּלְכָּה לֹא הִזְמִינָה עִם הַמֶּלֶךְ אֶל מְסִבַּת שְׁתִיַּת הַיַּיִן שֶׁהֵכִינָה אֶלָּא אוֹתִי, וַאֲנִי מֻזְמָן אֵלֶיהָ עִם הַמֶּלֶךְ גַּם מָחָר.
יג וְכׇל־זֶ֕ה אֵינֶ֥נּוּ שֹׁוֶ֖ה לִ֑י בְּכׇל־עֵ֗ת אֲשֶׁ֨ר אֲנִ֤י רֹאֶה֙ אֶת־מׇרְדֳּכַ֣י הַיְּהוּדִ֔י יוֹשֵׁ֖ב בְּשַׁ֥עַר הַמֶּֽלֶךְ׃ וְכָל זֶה לֹא שָׁוֶה לִי כְּלוּם כָּל זְמָן שֶׁאֲנִי רוֹאֶה אֶת מָרְדֳּכַי הַיְּהוּדִי יוֹשֵׁב בְּשַׁעַר הַמֶּלֶךְ.
יד וַתֹּ֣אמֶר לוֹ֩ זֶ֨רֶשׁ אִשְׁתּ֜וֹ וְכׇל־אֹֽהֲבָ֗יו יַֽעֲשׂוּ־עֵץ֮ גָּבֹ֣הַּ חֲמִשִּׁ֣ים אַמָּה֒ וּבַבֹּ֣קֶר ׀ אֱמֹ֣ר לַמֶּ֗לֶךְ וְיִתְל֤וּ אֶֽת־מׇרְדֳּכַי֙ עָלָ֔יו וּבֹֽא־עִם־הַמֶּ֥לֶךְ אֶל־הַמִּשְׁתֶּ֖ה שָׂמֵ֑חַ וַיִּיטַ֧ב הַדָּבָ֛ר לִפְנֵ֥י הָמָ֖ן וַיַּ֥עַשׂ הָעֵֽץ׃ אָמְרוּ לוֹ זֶרֶשׁ אִשְׁתּוֹ וְכָל חֲבֵרָיו: שֶׁיַּעֲשׂוּ עַמּוּד תְּלִיָּה בְּגוֹבַהּ חֲמִשִּׁים אַמָּה (כְּ25 מֶטֶר), וּבַבּוֹקֶר תֹּאמַר לַמֶּלֶךְ וְיִתְלוּ עָלָיו אֶת מָרְדֳּכַי, וְאָז תָּבוֹא עִם הַמֶּלֶךְ אֶל מְסִבַּת שְׁתִיַּת הַיַּיִן שָׂמֵחַ, הַדָּבָר מָצָא חֵן בְּעֵינֵי הָמָן וְהוּא עָשָׂה אֶת עַמּוּד הַתְּלִיָּה.
ס
ו א בַּלַּ֣יְלָה הַה֔וּא נָדְדָ֖ה שְׁנַ֣ת הַמֶּ֑לֶךְ וַיֹּ֗אמֶר לְהָבִ֞יא אֶת־סֵ֤פֶר הַזִּכְרֹנוֹת֙ דִּבְרֵ֣י הַיָּמִ֔ים וַיִּהְי֥וּ נִקְרָאִ֖ים לִפְנֵ֥י הַמֶּֽלֶךְ׃ בַּלַּיְלָה הַהוּא הַמֶּלֶךְ לֹא הִצְלִיחַ לְהֵרָדֵם, הוּא אָמַר שֶׁיָּבִיאוּ אֶת יוֹמַן הַזִּכְרוֹנוֹת וְקָרְאוּ מִמֶּנּוּ לִפְנֵי הַמֶּלֶךְ.
ב וַיִּמָּצֵ֣א כָת֗וּב אֲשֶׁר֩ הִגִּ֨יד מׇרְדֳּכַ֜י עַל־בִּגְתָ֣נָא וָתֶ֗רֶשׁ שְׁנֵי֙ סָרִיסֵ֣י הַמֶּ֔לֶךְ מִשֹּׁמְרֵ֖י הַסַּ֑ף אֲשֶׁ֤ר בִּקְשׁוּ֙ לִשְׁלֹ֣חַ יָ֔ד בַּמֶּ֖לֶךְ אֲחַשְׁוֵרֽוֹשׁ׃ וּמָצְאוּ כָּתוּב שֶׁמָרְדֳּכַי סִפֵּר עַל בִּגְתָנָא וָתֶרֶשׁ שְׁנֵי מְשָׁרְתֵי הַמֶּלֶךְ מִשּׁוֹמְרֵי הַפֶּתַח, שֶׁרָצוּ לְהִתְנַקֵּשׁ בַּמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ.
ג וַיֹּ֣אמֶר הַמֶּ֔לֶךְ מַֽה־נַּעֲשָׂ֞ה יְקָ֧ר וּגְדוּלָּ֛ה לְמׇרְדֳּכַ֖י עַל־זֶ֑ה וַיֹּ֨אמְר֜וּ נַעֲרֵ֤י הַמֶּ֙לֶךְ֙ מְשָׁ֣רְתָ֔יו לֹא־נַעֲשָׂ֥ה עִמּ֖וֹ דָּבָֽר׃ אָמַר הַמֶּלֶךְ: אֵיזֶה כָּבוֹד וּגְדוּלָּה עָשׂוּ לְמָרְדֳּכַי עַל זֶה? אָמְרוּ נַעֲרֵי הַמֶּלֶךְ מְשָׁרְתָיו: לֹא עָשׂוּ לוֹ שׁוּם דָּבָר.
ד וַיֹּ֥אמֶר הַמֶּ֖לֶךְ מִ֣י בֶחָצֵ֑ר וְהָמָ֣ן בָּ֗א לַחֲצַ֤ר בֵּית־הַמֶּ֙לֶךְ֙ הַחִ֣יצוֹנָ֔ה לֵאמֹ֣ר לַמֶּ֔לֶךְ לִתְלוֹת֙ אֶֽת־מׇרְדֳּכַ֔י עַל־הָעֵ֖ץ אֲשֶׁר־הֵכִ֥ין לֽוֹ׃ אָמַר הַמֶּלֶךְ: מִי בֶחָצֵר? וְהָמָן בָּא לֶחָצֵר הַחִיצוֹנָה שֶׁל בֵּית הַמֶּלֶךְ לוֹמַר לַמֶּלֶךְ לִתְלוֹת אֶת מָרְדֳּכַי עַל עַמּוּד הַתְּלִיָּה שֶׁהֵכִין לוֹ.
ה וַיֹּ֨אמְר֜וּ נַעֲרֵ֤י הַמֶּ֙לֶךְ֙ אֵלָ֔יו הִנֵּ֥ה הָמָ֖ן עֹמֵ֣ד בֶּחָצֵ֑ר וַיֹּ֥אמֶר הַמֶּ֖לֶךְ יָבֽוֹא׃ אָמְרוּ נַעֲרֵי הַמֶּלֶךְ אֵלָיו: הִנֵּה הָמָן עוֹמֵד בֶּחָצֵר, אָמַר הַמֶּלֶךְ: יָבוֹא!
ו וַיָּבוֹא֮ הָמָן֒ וַיֹּ֤אמֶר לוֹ֙ הַמֶּ֔לֶךְ מַה־לַּעֲשׂ֕וֹת בָּאִ֕ישׁ אֲשֶׁ֥ר הַמֶּ֖לֶךְ חָפֵ֣ץ בִּיקָר֑וֹ וַיֹּ֤אמֶר הָמָן֙ בְּלִבּ֔וֹ לְמִ֞י יַחְפֹּ֥ץ הַמֶּ֛לֶךְ לַעֲשׂ֥וֹת יְקָ֖ר יוֹתֵ֥ר מִמֶּֽנִּי׃ הָמָן בָּא, אָמַר לוֹ הַמֶּלֶךְ: מַה לַּעֲשׂוֹת לָאִישׁ שֶׁהַמֶּלֶךְ רוֹצֶה שֶׁיְּכַבְּדוּ אוֹתוֹ? הָמָן אָמַר לְעַצְמוֹ, לְמִי הַמֶּלֶךְ רוֹצֶה לָתֵת כָּבוֹד יוֹתֵר מִלִּי?!
ז וַיֹּ֥אמֶר הָמָ֖ן אֶל־הַמֶּ֑לֶךְ אִ֕ישׁ אֲשֶׁ֥ר הַמֶּ֖לֶךְ חָפֵ֥ץ בִּיקָרֽוֹ׃ אָמַר הָמָן לַמֶּלֶךְ: אִישׁ שֶׁהַמֶּלֶךְ רוֹצֶה שֶׁיְּכַבְּדוּ אוֹתוֹ –
ח יָבִ֙יאוּ֙ לְב֣וּשׁ מַלְכ֔וּת אֲשֶׁ֥ר לָֽבַשׁ־בּ֖וֹ הַמֶּ֑לֶךְ וְס֗וּס אֲשֶׁ֨ר רָכַ֤ב עָלָיו֙ הַמֶּ֔לֶךְ וַאֲשֶׁ֥ר נִתַּ֛ן כֶּ֥תֶר מַלְכ֖וּת בְּרֹאשֽׁוֹ׃ יָבִיאוּ בִּגְדֵי מַלְכוּת שֶׁהַמֶּלֶךְ לָבַשׁ, וְסוּס שֶׁהַמֶּלֶךְ רָכַב עָלָיו וְשֶׁשָּׂמוּ כֶּתֶר מַלְכוּת בְּרֹאשׁוֹ.
ט וְנָת֨וֹן הַלְּב֜וּשׁ וְהַסּ֗וּס עַל־יַד־אִ֞ישׁ מִשָּׂרֵ֤י הַמֶּ֙לֶךְ֙ הַֽפַּרְתְּמִ֔ים וְהִלְבִּ֙ישׁוּ֙ אֶת־הָאִ֔ישׁ אֲשֶׁ֥ר הַמֶּ֖לֶךְ חָפֵ֣ץ בִּֽיקָר֑וֹ וְהִרְכִּיבֻ֤הוּ עַל־הַסּוּס֙ בִּרְח֣וֹב הָעִ֔יר וְקָרְא֣וּ לְפָנָ֔יו כָּ֚כָה יֵעָשֶׂ֣ה לָאִ֔ישׁ אֲשֶׁ֥ר הַמֶּ֖לֶךְ חָפֵ֥ץ בִּיקָרֽוֹ׃ וְיִמְסְרוּ אֶת הַבְּגָדִים וְאֶת הַסּוּס, לִידֵי אֶחָד מִשָּׂרֵי הַמֶּלֶךְ הַשַּׁלִּיטִים, וְיַלְבִּישׁוּ אֶת הָאִישׁ שֶׁהַמֶּלֶךְ רוֹצֶה שֶׁיְּכַבְּדוּ אוֹתוֹ, וְיַרְכִּיבוּ אוֹתוֹ עַל הַסּוּס בְּכִכַּר הָעִיר, וְיַכְרִיזוּ לְפָנָיו: כָּכָה עוֹשִׂים לָאִישׁ שֶׁהַמֶּלֶךְ רוֹצֶה שֶׁיְּכַבְּדוּ אוֹתוֹ!
י וַיֹּ֨אמֶר הַמֶּ֜לֶךְ לְהָמָ֗ן מַ֠הֵ֠ר קַ֣ח אֶת־הַלְּב֤וּשׁ וְאֶת־הַסּוּס֙ כַּאֲשֶׁ֣ר דִּבַּ֔רְתָּ וַֽעֲשֵׂה־כֵן֙ לְמׇרְדֳּכַ֣י הַיְּהוּדִ֔י הַיּוֹשֵׁ֖ב בְּשַׁ֣עַר הַמֶּ֑לֶךְ אַל־תַּפֵּ֣ל דָּבָ֔ר מִכֹּ֖ל אֲשֶׁ֥ר דִּבַּֽרְתָּ׃ אָמַר הַמֶּלֶךְ לְהָמָן: תְּמַהֵר, קַח אֶת הַבְּגָדִים וְאֶת הַסּוּס כְּמוֹ שֶׁאָמַרְתָּ, וְכָךְ תַּעֲשֶׂה לְמָרְדֳּכַי הַיְּהוּדִי שֶׁיּוֹשֵׁב בְּשַׁעַר הַמֶּלֶךְ, אַל תַּחְסִיר שׁוּם דָּבָר מִכָּל מַה שֶּׁאָמַרְתָּ!
יא וַיִּקַּ֤ח הָמָן֙ אֶת־הַלְּב֣וּשׁ וְאֶת־הַסּ֔וּס וַיַּלְבֵּ֖שׁ אֶֽת־מׇרְדֳּכָ֑י וַיַּרְכִּיבֵ֙הוּ֙ בִּרְח֣וֹב הָעִ֔יר וַיִּקְרָ֣א לְפָנָ֔יו כָּ֚כָה יֵעָשֶׂ֣ה לָאִ֔ישׁ אֲשֶׁ֥ר הַמֶּ֖לֶךְ חָפֵ֥ץ בִּיקָרֽוֹ׃ הָמָן לָקַח אֶת הַבְּגָדִים וְאֶת הַסּוּס, וְהִלְבִּישׁ אֶת מָרְדֳּכָי, וְהִרְכִּיב אוֹתוֹ עַל הַסּוּס בְּכִכַּר הָעִיר, וְקָרָא לְפָנָיו: כָּכָה עוֹשִׂים לָאִישׁ שֶׁהַמֶּלֶךְ רוֹצֶה שֶׁיְּכַבְּדוּ אוֹתוֹ!
יב וַיָּ֥שׇׁב מׇרְדֳּכַ֖י אֶל־שַׁ֣עַר הַמֶּ֑לֶךְ וְהָמָן֙ נִדְחַ֣ף אֶל־בֵּית֔וֹ אָבֵ֖ל וַחֲפ֥וּי רֹֽאשׁ׃ מָרְדֳּכַי חָזַר לְשַׁעַר הַמֶּלֶךְ, וְהָמָן מִהֵר בִּדְחִיפוּת לְבֵיתוֹ, אָבֵל וּמְכַסֶּה אֶת רֹאשׁוֹ בְּבוּשָׁה.
יג וַיְסַפֵּ֨ר הָמָ֜ן לְזֶ֤רֶשׁ אִשְׁתּוֹ֙ וּלְכׇל־אֹ֣הֲבָ֔יו אֵ֖ת כׇּל־אֲשֶׁ֣ר קָרָ֑הוּ וַיֹּ֩אמְרוּ֩ ל֨וֹ חֲכָמָ֜יו וְזֶ֣רֶשׁ אִשְׁתּ֗וֹ אִ֣ם מִזֶּ֣רַע הַיְּהוּדִ֡ים מׇרְדֳּכַ֞י אֲשֶׁר֩ הַחִלּ֨וֹתָ לִנְפֹּ֤ל לְפָנָיו֙ לֹא־תוּכַ֣ל ל֔וֹ כִּֽי־נָפ֥וֹל תִּפּ֖וֹל לְפָנָֽיו׃ הָמָן סִפֵּר לְזֶרֶשׁ אִשְׁתּוֹ וּלְכָל חֲבֵרָיו אֶת כָּל מַה שֶּׁקָּרָה לוֹ, אָמְרוּ לוֹ חֲבֵרָיו הַחֲכָמִים וְזֶרֶשׁ אִשְׁתּוֹ: אִם מָרְדֳּכַי שֶׁהִתְחַלְתָּ לִפּוֹל בְּפָנָיו הוּא מִצֶּאֱצָאֵי הַיְּהוּדִים, לֹא תּוּכַל לְנַצֵּחַ אוֹתוֹ אֶלָּא תִּפּוֹל בְּפָנָיו.
יד עוֹדָם֙ מְדַבְּרִ֣ים עִמּ֔וֹ וְסָרִיסֵ֥י הַמֶּ֖לֶךְ הִגִּ֑יעוּ וַיַּבְהִ֙לוּ֙ לְהָבִ֣יא אֶת־הָמָ֔ן אֶל־הַמִּשְׁתֶּ֖ה אֲשֶׁר־עָשְׂתָ֥ה אֶסְתֵּֽר׃ בְּעוֹד הֵם מְדַבְּרִים אִתּוֹ הִגִּיעוּ מְשָׁרְתֵי הַמֶּלֶךְ וּמִהֲרוּ לְהָבִיא אֶת הָמָן לִמְסִבַּת שְׁתִיַּת הַיַּיִן שֶׁהֵכִינָה אֶסְתֵּר.
ז א וַיָּבֹ֤א הַמֶּ֙לֶךְ֙ וְהָמָ֔ן לִשְׁתּ֖וֹת עִם־אֶסְתֵּ֥ר הַמַּלְכָּֽה׃ הַמֶּלֶךְ וְהָמָן בָּאוּ לִשְׁתּוֹת עִם אֶסְתֵּר הַמַּלְכָּה.
ב וַיֹּ֩אמֶר֩ הַמֶּ֨לֶךְ לְאֶסְתֵּ֜ר גַּ֣ם בַּיּ֤וֹם הַשֵּׁנִי֙ בְּמִשְׁתֵּ֣ה הַיַּ֔יִן מַה־שְּׁאֵלָתֵ֛ךְ אֶסְתֵּ֥ר הַמַּלְכָּ֖ה וְתִנָּ֣תֵֽן לָ֑ךְ וּמַה־בַּקָּשָׁתֵ֛ךְ עַד־חֲצִ֥י הַמַּלְכ֖וּת וְתֵעָֽשׂ׃ הַמֶּלֶךְ אָמַר לְאֶסְתֵּר גַּם בַּיּוֹם הַשֵּׁנִי בִּמְסִבַּת שְׁתִיַּת הַיַּיִן: מָה רְצוֹנֵךְ אֶסְתֵּר הַמַּלְכָּה, זֶה יִנָּתֵן לָךְ! וּמַה בַּקָּשָׁתֵךְ, גַּם עַד חֲצִי הַמַּמְלָכָה זֶה יִתְבַּצֵּעַ!
ג וַתַּ֨עַן אֶסְתֵּ֤ר הַמַּלְכָּה֙ וַתֹּאמַ֔ר אִם־מָצָ֨אתִי חֵ֤ן בְּעֵינֶ֙יךָ֙ הַמֶּ֔לֶךְ וְאִם־עַל־הַמֶּ֖לֶךְ ט֑וֹב תִּנָּֽתֶן־לִ֤י נַפְשִׁי֙ בִּשְׁאֵ֣לָתִ֔י וְעַמִּ֖י בְּבַקָּשָׁתִֽי׃ עָנְתָה אֶסְתֵּר הַמַּלְכָּה וְאָמְרָה: אִם מָצָאתִי חֵן בְּעֵינֶיךָ הַמֶּלֶךְ וְאִם כָּךְ רְצוֹן הַמֶּלֶךְ, יִנָּתְנוּ לִי לְבַקָּשָׁתִי הַחַיִּים שֶׁלִּי וְשֶׁל עַמִּי!
ד כִּ֤י נִמְכַּ֙רְנוּ֙ אֲנִ֣י וְעַמִּ֔י לְהַשְׁמִ֖יד לַהֲר֣וֹג וּלְאַבֵּ֑ד וְ֠אִלּ֠וּ לַעֲבָדִ֨ים וְלִשְׁפָח֤וֹת נִמְכַּ֙רְנוּ֙ הֶחֱרַ֔שְׁתִּי כִּ֣י אֵ֥ין הַצָּ֛ר שֹׁוֶ֖ה בְּנֵ֥זֶק הַמֶּֽלֶךְ׃ כִּי נִמְכַּרְנוּ אֲנִי וְעַמִּי לְהַשְׁמִיד לַהֲרוֹג וּלְאַבֵּד אוֹתָנוּ, אִלּוּ הָיִינוּ נִמְכָּרִים לַעֲבָדִים וְלִשְׁפָחוֹת הָיִיתִי שׁוֹתֶקֶת כִּי עֲדִיפָה צָרָה כָּזוֹ כְּדֵי שֶׁלֹּא יִהְיֶה נֶזֶק לַמֶּלֶךְ.
ס
ה וַיֹּ֙אמֶר֙ הַמֶּ֣לֶךְ אֲחַשְׁוֵר֔וֹשׁ וַיֹּ֖אמֶר לְאֶסְתֵּ֣ר הַמַּלְכָּ֑ה מִ֣י ה֥וּא זֶה֙ וְאֵֽי־זֶ֣ה ה֔וּא אֲשֶׁר־מְלָא֥וֹ לִבּ֖וֹ לַעֲשׂ֥וֹת כֵּֽן׃ אָמַר הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ לְאֶסְתֵּר הַמַּלְכָּה: מִי הוּא זֶה וְאֵיפֹה הוּא הָאִישׁ שֶׁהֶעֱלָה עַל דַּעְתּוֹ לַעֲשׂוֹת כָּךְ?!
ו וַתֹּ֣אמֶר אֶסְתֵּ֔ר אִ֚ישׁ צַ֣ר וְאוֹיֵ֔ב הָמָ֥ן הָרָ֖ע הַזֶּ֑ה וְהָמָ֣ן נִבְעַ֔ת מִלִּפְנֵ֥י הַמֶּ֖לֶךְ וְהַמַּלְכָּֽה׃ אָמְרָה אֶסְתֵּר: אִישׁ עוֹשֶׂה צָרוֹת וְאוֹיֵב – הָמָן הָרָע הַזֶּה! וְהָמָן נִבְהַל וּפָחַד מֵהַמֶּלֶךְ וְהַמַּלְכָּה.
ז וְהַמֶּ֜לֶךְ קָ֤ם בַּחֲמָתוֹ֙ מִמִּשְׁתֵּ֣ה הַיַּ֔יִן אֶל־גִּנַּ֖ת הַבִּיתָ֑ן וְהָמָ֣ן עָמַ֗ד לְבַקֵּ֤שׁ עַל־נַפְשׁוֹ֙ מֵֽאֶסְתֵּ֣ר הַמַּלְכָּ֔ה כִּ֣י רָאָ֔ה כִּֽי־כָלְתָ֥ה אֵלָ֛יו הָרָעָ֖ה מֵאֵ֥ת הַמֶּֽלֶךְ׃ וְהַמֶּלֶךְ קָם בְּכַעֲסוֹ מִמְּסִבַּת שְׁתִיַּת הַיַּיִן וְהָלַךְ אֶל גִּנַּת הָאַרְמוֹן, וְהָמָן עָמַד לְבַקֵּשׁ עַל חַיָּיו מֵאֶסְתֵּר הַמַּלְכָּה כִּי הוּא רָאָה שֶׁהַמֶּלֶךְ כְּבָר הֶחְלִיט לַעֲשׂוֹת לוֹ רַע.
ח וְהַמֶּ֡לֶךְ שָׁב֩ מִגִּנַּ֨ת הַבִּיתָ֜ן אֶל־בֵּ֣ית ׀ מִשְׁתֵּ֣ה הַיַּ֗יִן וְהָמָן֙ נֹפֵ֗ל עַל־הַמִּטָּה֙ אֲשֶׁ֣ר אֶסְתֵּ֣ר עָלֶ֔יהָ וַיֹּ֣אמֶר הַמֶּ֔לֶךְ הֲ֠גַ֠ם לִכְבּ֧וֹשׁ אֶת־הַמַּלְכָּ֛ה עִמִּ֖י בַּבָּ֑יִת הַדָּבָ֗ר יָצָא֙ מִפִּ֣י הַמֶּ֔לֶךְ וּפְנֵ֥י הָמָ֖ן חָפֽוּ׃ וְהַמֶּלֶךְ חָזַר מִגִּנַּת הָאַרְמוֹן לְהֵיכַל שְׁתִיַּת הַיַּיִן, וְהָמָן נוֹפֵל עַל הַמִּטָּה שֶׁאֶסְתֵּר עָלֶיהָ, אָמַר הַמֶּלֶךְ: גַּם לֶאֱנוֹס אֶת הַמַּלְכָּה אֶצְלִי בַּבַּיִת?! הַמִּילִים יָצְאוּ מִפִּי הַמֶּלֶךְ וְהַמְּשָׁרְתִים כִּסּוּ אֶת פְּנֵי הָמָן.

ט וַיֹּ֣אמֶר חַ֠רְבוֹנָ֠ה אֶחָ֨ד מִן־הַסָּרִיסִ֜ים לִפְנֵ֣י הַמֶּ֗לֶךְ גַּ֣ם הִנֵּה־הָעֵ֣ץ אֲשֶׁר־עָשָׂ֪ה הָמָ֟ן לְֽמׇרְדֳּכַ֞י אֲשֶׁ֧ר דִּבֶּר־ט֣וֹב עַל־הַמֶּ֗לֶךְ עֹמֵד֙ בְּבֵ֣ית הָמָ֔ן גָּבֹ֖הַּ חֲמִשִּׁ֣ים אַמָּ֑ה וַיֹּ֥אמֶר הַמֶּ֖לֶךְ תְּלֻ֥הוּ עָלָֽיו׃ חַרְבוֹנָה, אֶחָד הַמְּשָׁרְתִים שֶׁלִּפְנֵי הַמֶּלֶךְ, אָמַר: גַּם הִנֵּה עַמּוּד הַתְּלִיָּה שֶׁהָמָן עָשָׂה לִתְלוֹת אֶת מָרְדֳּכָי שֶׁדִּבֵּר דְּבָרִים טוֹבִים בִּשְׁבִיל הַמֶּלֶךְ, עוֹמֵד בְּבֵית הָמָן, בְּגוֹבַהּ חֲמִשִּׁים אַמָּה! אָמַר הַמֶּלֶךְ: תְּלוּ אוֹתוֹ עָלָיו!

ג וַתּ֣וֹסֶף אֶסְתֵּ֗ר וַתְּדַבֵּר֙ לִפְנֵ֣י הַמֶּ֔לֶךְ וַתִּפֹּ֖ל לִפְנֵ֣י רַגְלָ֑יו וַתֵּ֣בְךְּ וַתִּתְחַנֶּן־ל֗וֹ לְהַֽעֲבִיר֙ אֶת־רָעַת֙ הָמָ֣ן הָֽאֲגָגִ֔י וְאֵת֙ מַֽחֲשַׁבְתּ֔וֹ אֲשֶׁ֥ר חָשַׁ֖ב עַל־הַיְּהוּדִֽים׃ אֶסְתֵּר הִמְשִׁיכָה לְדַבֵּר לִפְנֵי הַמֶּלֶךְ וְנָפְלָה לִפְנֵי רַגְלָיו, וּבָכְתָה וְהִתְחַנְּנָה לוֹ שֶׁיְּבַטֵּל אֶת רָעַת הָמָן הַצֶּאֱצָא שֶׁל אֲגָג, וְאֶת תָּכְנִיתוֹ שֶׁתִּכְנֵן לַעֲשׂוֹת לַיְּהוּדִים.
ד וַיּ֤וֹשֶׁט הַמֶּ֙לֶךְ֙ לְאֶסְתֵּ֔ר אֵ֖ת שַׁרְבִ֣ט הַזָּהָ֑ב וַתָּ֣קׇם אֶסְתֵּ֔ר וַֽתַּעֲמֹ֖ד לִפְנֵ֥י הַמֶּֽלֶךְ׃ הַמֶּלֶךְ הוֹשִׁיט לְאֶסְתֵּר אֶת שַׁרְבִיט הַזָּהָב, וְאֶסְתֵּר קָמָה וְנֶעֶמְדָה לִפְנֵי הַמֶּלֶךְ.
ה וַ֠תֹּ֠אמֶר אִם־עַל־הַמֶּ֨לֶךְ ט֜וֹב וְאִם־מָצָ֧אתִי חֵ֣ן לְפָנָ֗יו וְכָשֵׁ֤ר הַדָּבָר֙ לִפְנֵ֣י הַמֶּ֔לֶךְ וְטוֹבָ֥ה אֲנִ֖י בְּעֵינָ֑יו יִכָּתֵ֞ב לְהָשִׁ֣יב אֶת־הַסְּפָרִ֗ים מַחֲשֶׁ֜בֶת הָמָ֤ן בֶּֽן־הַמְּדָ֙תָא֙ הָאֲגָגִ֔י אֲשֶׁ֣ר כָּתַ֗ב לְאַבֵּד֙ אֶת־הַיְּהוּדִ֔ים אֲשֶׁ֖ר בְּכׇל־מְדִינ֥וֹת הַמֶּֽלֶךְ׃ וְאָמְרָה: אִם הַמֶּלֶךְ מַסְכִּים וְאִם אֲנִי מוֹצֵאת חֵן בְּעֵינָיו וְהַמֶּלֶךְ חוֹשֵׁב שֶׁזֶּה נָכוֹן וַאֲנִי טוֹבָה בְּעֵינָיו, יִכְתְּבוּ לְבַטֵּל אֶת הַצַּוִּים שֶׁל הַתָּכְנִית שֶׁל הָמָן בֶּן הַמְּדָתָא הַצֶּאֱצָא שֶׁל אֲגָג שֶׁכָּתַב לְהַשְׁמִיד אֶת הַיְּהוּדִים שֶׁבְּכָל מְדִינוֹת הַמֶּלֶךְ.
ו כִּ֠י אֵיכָכָ֤ה אוּכַל֙ וְֽרָאִ֔יתִי בָּרָעָ֖ה אֲשֶׁר־יִמְצָ֣א אֶת־עַמִּ֑י וְאֵֽיכָכָ֤ה אוּכַל֙ וְֽרָאִ֔יתִי בְּאׇבְדַ֖ן מוֹלַדְתִּֽי׃ כִּי אֵיךְ אוּכַל לִרְאוֹת אֶת הָרַע שֶׁיִּקְרֶה לְעַמִּי?! וְאֵיךְ אוּכַל לִרְאוֹת אֶת הַשְׁמָדַת מִשְׁפַּחְתִּי?!

ז וַיֹּ֨אמֶר הַמֶּ֤לֶךְ אֲחַשְׁוֵרֹשׁ֙ לְאֶסְתֵּ֣ר הַמַּלְכָּ֔ה וּֽלְמׇרְדֳּכַ֖י הַיְּהוּדִ֑י הִנֵּ֨ה בֵית־הָמָ֜ן נָתַ֣תִּי לְאֶסְתֵּ֗ר וְאֹתוֹ֙ תָּל֣וּ עַל־הָעֵ֔ץ עַ֛ל אֲשֶׁר־שָׁלַ֥ח יָד֖וֹ ביהודיים בַּיְּהוּדִֽים׃ אָמַר הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ לְאֶסְתֵּר הַמַּלְכָּה וּלְמָרְדֳּכַי הַיְּהוּדִי: הִנֵּה, אֶת בֵּיתוֹ שֶׁל הָמָן נָתַתִּי לְאֶסְתֵּר, וְאֶת הָמָן תָּלוּ עַל עַמּוּד הַתְּלִיָּה עַל שֶׁהִתְנַכֵּל לַיְּהוּדִים.
ח וְ֠אַתֶּ֠ם כִּתְב֨וּ עַל־הַיְּהוּדִ֜ים כַּטּ֤וֹב בְּעֵֽינֵיכֶם֙ בְּשֵׁ֣ם הַמֶּ֔לֶךְ וְחִתְמ֖וּ בְּטַבַּ֣עַת הַמֶּ֑לֶךְ כִּֽי־כְתָ֞ב אֲשֶׁר־נִכְתָּ֣ב בְּשֵׁם־הַמֶּ֗לֶךְ וְנַחְתּ֛וֹם בְּטַבַּ֥עַת הַמֶּ֖לֶךְ אֵ֥ין לְהָשִֽׁיב׃ וְאַתֶּם כִּתְבוּ עַל הַיְּהוּדִים מַה שֶּׁאַתֶּם רוֹצִים בְּצַו חָדָשׁ בְּשֵׁם הַמֶּלֶךְ וְתַחְתְּמוּ בְּחוֹתֶמֶת טַבַּעַת הַמֶּלֶךְ, כִּי צַו שֶׁנִּכְתָּב בְּשֵׁם הַמֶּלֶךְ וְנֶחְתַּם בְּחוֹתֶמֶת הַמֶּלֶךְ אִי אֶפְשָׁר לְבַטֵּל.
ט וַיִּקָּרְא֣וּ סֹפְרֵֽי־הַמֶּ֣לֶךְ בָּֽעֵת־הַ֠הִ֠יא בַּחֹ֨דֶשׁ הַשְּׁלִישִׁ֜י הוּא־חֹ֣דֶשׁ סִיוָ֗ן בִּשְׁלוֹשָׁ֣ה וְעֶשְׂרִים֮ בּוֹ֒ וַיִּכָּתֵ֣ב כְּֽכׇל־אֲשֶׁר־צִוָּ֣ה מׇרְדֳּכַ֣י אֶל־הַיְּהוּדִ֡ים וְאֶ֣ל הָאֲחַשְׁדַּרְפְּנִֽים־וְהַפַּחוֹת֩ וְשָׂרֵ֨י הַמְּדִינ֜וֹת אֲשֶׁ֣ר ׀ מֵהֹ֣דּוּ וְעַד־כּ֗וּשׁ שֶׁ֣בַע וְעֶשְׂרִ֤ים וּמֵאָה֙ מְדִינָ֔ה מְדִינָ֤ה וּמְדִינָה֙ כִּכְתָבָ֔הּ וְעַ֥ם וָעָ֖ם כִּלְשֹׁנ֑וֹ וְאֶ֨ל־הַיְּהוּדִ֔ים כִּכְתָבָ֖ם וְכִלְשׁוֹנָֽם׃ סוֹפְרֵי הַמֶּלֶךְ הֻזְמְנוּ אָז – בְּיוֹם כ"ג בַּחוֹדֶשׁ הַשְּׁלִישִׁי הוּא חוֹדֶשׁ סִיוָן, וְכָתְבוּ אֶת כָּל מַה שֶּׁצִּוָּה מָרְדֳּכַי אֶל הַיְּהוּדִים וְאֶל הַנְּצִיבִים וְהַמּוֹשְׁלִים וְשָׂרֵי הַמְּדִינוֹת שֶׁמֵּהוֹדּוּ וְעַד כּוּשׁ מֵאָה וְעֶשְׂרִים וְשֶׁבַע מְדִינוֹת, לְכָל מְדִינָה וּמְדִינָה בִּכְתָבָהּ וּלְכָל עַם וְעַם בִּשְׂפָתוֹ, וְאֶל הַיְּהוּדִים בִּכְתָבָם וּבִשְׂפָתָם.
י וַיִּכְתֹּ֗ב בְּשֵׁם֙ הַמֶּ֣לֶךְ אֲחַשְׁוֵרֹ֔שׁ וַיַּחְתֹּ֖ם בְּטַבַּ֣עַת הַמֶּ֑לֶךְ וַיִּשְׁלַ֣ח סְפָרִ֡ים בְּיַד֩ הָרָצִ֨ים בַּסּוּסִ֜ים רֹכְבֵ֤י הָרֶ֙כֶשׁ֙ הָֽאֲחַשְׁתְּרָנִ֔ים בְּנֵ֖י הָֽרַמָּכִֽים׃ הוּא כָּתַב בְּשֵׁם הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ וְחָתַם בְּחוֹתֶמֶת הַמֶּלֶךְ וְשָׁלַח מִכְתָּבִים בְּיַד הַדּוֹהֲרִים עַל הַפְּרָדִים בְּנֵי סוּסוֹת מֵרְכוּשׁ הַמֶּלֶךְ.
יא אֲשֶׁר֩ נָתַ֨ן הַמֶּ֜לֶךְ לַיְּהוּדִ֣ים ׀ אֲשֶׁ֣ר בְּכׇל־עִיר־וָעִ֗יר לְהִקָּהֵל֮ וְלַעֲמֹ֣ד עַל־נַפְשָׁם֒ לְהַשְׁמִיד֩ וְלַהֲרֹ֨ג‏[1] וּלְאַבֵּ֜ד אֶת־כׇּל־חֵ֨יל עַ֧ם וּמְדִינָ֛ה הַצָּרִ֥ים אֹתָ֖ם טַ֣ף וְנָשִׁ֑ים וּשְׁלָלָ֖ם לָבֽוֹז׃ וְנִכְתַּב שֶׁהַמֶּלֶךְ נָתַן רְשׁוּת לַיְּהוּדִים שֶׁבְּכָל עִיר לְהִתְכַּנֵּס וְלַעֲמוֹד בַּקְּרָב כְּדֵי לְהַצִּיל אֶת נַפְשָׁם, וּלְהַשְׁמִיד וְלַהֲרוֹג וּלְאַבֵּד כָּל צָבָא שֶׁל עַם וּמְדִינָה שֶׁנִּלְחָם בָּהֶם, וְאֶת הַיְּלָדִים וְהַנָּשִׁים, וְלָקַחַת אֶת שְׁלָלָם.
יב בְּי֣וֹם אֶחָ֔ד בְּכׇל־מְדִינ֖וֹת הַמֶּ֣לֶךְ אֲחַשְׁוֵר֑וֹשׁ בִּשְׁלוֹשָׁ֥ה עָשָׂ֛ר לְחֹ֥דֶשׁ שְׁנֵים־עָשָׂ֖ר הוּא־חֹ֥דֶשׁ אֲדָֽר׃ בְּיוֹם אֶחָד בְּכָל מְדִינוֹת הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ, בְּיוֹם י"ג לַחוֹדֶשׁ הַשְּׁנֵים עָשָׂר הוּא חוֹדֶשׁ אֲדָר.
יג פַּתְשֶׁ֣גֶן הַכְּתָ֗ב לְהִנָּ֤תֵֽן דָּת֙ בְּכׇל־מְדִינָ֣ה וּמְדִינָ֔ה גָּל֖וּי לְכׇל־הָעַמִּ֑ים וְלִהְי֨וֹת היהודיים הַיְּהוּדִ֤ים עתודים עֲתִידִים֙ לַיּ֣וֹם הַזֶּ֔ה לְהִנָּקֵ֖ם מֵאֹיְבֵיהֶֽם׃ תּוֹכֶן הַכְּתָב: יִקָּבַע חוֹק בְּכָל מְדִינָה, וְיִתְפַּרְסֵם אֵצֶל כָּל הָעַמִּים, שֶׁהַיְּהוּדִים יִתְכּוֹנְנוּ לַיּוֹם הַזֶּה לִנְקוֹם בְּאוֹיְבֵיהֶם.
יד הָרָצִ֞ים רֹכְבֵ֤י הָרֶ֙כֶשׁ֙ הָֽאֲחַשְׁתְּרָנִ֔ים יָ֥צְא֛וּ מְבֹהָלִ֥ים וּדְחוּפִ֖ים בִּדְבַ֣ר הַמֶּ֑לֶךְ וְהַדָּ֥ת נִתְּנָ֖ה בְּשׁוּשַׁ֥ן הַבִּירָֽה׃ הַדּוֹהֲרִים עַל הַפְּרָדִים מֵרְכוּשׁ הַמֶּלֶךְ יָצְאוּ בִּמְהִירוּת וּבִדְחִיפוּת כְּדִבְרֵי הַמֶּלֶךְ וְהַחוֹק נִקְבַּע בְּשׁוּשַׁן הַבִּירָה.

טו וּמׇרְדֳּכַ֞י יָצָ֣א ׀ מִלִּפְנֵ֣י הַמֶּ֗לֶךְ בִּלְב֤וּשׁ מַלְכוּת֙ תְּכֵ֣לֶת וָח֔וּר וַעֲטֶ֤רֶת זָהָב֙ גְּדוֹלָ֔ה וְתַכְרִ֥יךְ בּ֖וּץ וְאַרְגָּמָ֑ן וְהָעִ֣יר שׁוּשָׁ֔ן צָהֲלָ֖ה וְשָׂמֵֽחָה׃ וּמָרְדֳּכַי יָצָא מִלִּפְנֵי הַמֶּלֶךְ בִּלְבוּשׁ מַלְכוּת שֶׁל תְּכֵלֶת וְלָבָן וּבְכֶתֶר זָהָב גָּדוֹל וְצָעִיף פִּשְׁתָּן וְצָעִיף אַרְגָּמָן, וּבָעִיר שׁוּשָׁן צָהֲלוּ וְשָׂמְחוּ.
טז לַיְּהוּדִ֕ים הָֽיְתָ֥ה אוֹרָ֖ה וְשִׂמְחָ֑ה וְשָׂשֹׂ֖ן וִיקָֽר׃ לַיְּהוּדִים הָיָה אוֹר בָּעֵינַיִם וְשִׂמְחָה וְשָׂשׂוֹן וְכָבוֹד.
יז וּבְכׇל־מְדִינָ֨ה וּמְדִינָ֜ה וּבְכׇל־עִ֣יר וָעִ֗יר מְקוֹם֙ אֲשֶׁ֨ר דְּבַר־הַמֶּ֤לֶךְ וְדָתוֹ֙ מַגִּ֔יעַ שִׂמְחָ֤ה וְשָׂשׂוֹן֙ לַיְּהוּדִ֔ים מִשְׁתֶּ֖ה וְי֣וֹם ט֑וֹב וְרַבִּ֞ים מֵֽעַמֵּ֤י הָאָ֙רֶץ֙ מִֽתְיַהֲדִ֔ים כִּֽי־נָפַ֥ל פַּֽחַד־הַיְּהוּדִ֖ים עֲלֵיהֶֽם׃ וּבְכָל מְדִינָה וּבְכָל עִיר כָּל מָקוֹם שֶׁדִּבְרֵי הַמֶּלֶךְ וְהַחוֹק שֶׁלּוֹ מִתְפַּרְסְמִים, יֵשׁ שִׂמְחָה וְשָׂשׂוֹן לַיְּהוּדִים, סְעוּדָה עִם יַיִן וְיוֹם טוֹב, וְרַבִּים מִבְּנֵי הָעַמִּים שֶׁבָּאֲרָצוֹת מִתְגַּיְּרִים מֵרוֹב פַּחַד מֵהַיְּהוּדִים.
ט א וּבִשְׁנֵים֩ עָשָׂ֨ר חֹ֜דֶשׁ הוּא־חֹ֣דֶשׁ אֲדָ֗ר בִּשְׁלוֹשָׁ֨ה עָשָׂ֥ר יוֹם֙ בּ֔וֹ אֲשֶׁ֨ר הִגִּ֧יעַ דְּבַר־הַמֶּ֛לֶךְ וְדָת֖וֹ לְהֵעָשׂ֑וֹת בַּיּ֗וֹם אֲשֶׁ֨ר שִׂבְּר֜וּ אֹיְבֵ֤י הַיְּהוּדִים֙ לִשְׁל֣וֹט בָּהֶ֔ם וְנַהֲפ֣וֹךְ ה֔וּא אֲשֶׁ֨ר יִשְׁלְט֧וּ הַיְּהוּדִ֛ים הֵ֖מָּה בְּשֹׂנְאֵיהֶֽם׃ וּבַחוֹדֶשׁ הַשְּׁנֵים עָשָׂר הוּא חוֹדֶשׁ אֲדָר בְּיוֹם י"ג בּוֹ, כְּשֶׁהִגִּיעַ זְמַן בִּצּוּעַ הַחוֹק שֶׁל פְּקֻדַּת הַמֶּלֶךְ, בַּיּוֹם שֶׁאוֹיְבֵי הַיְּהוּדִים תִּכְנְנוּ לִשְׁלוֹט בָּהֶם, קָרָה הַהֵיפֶךְ, שֶׁהַיְּהוּדִים הֵם שֶׁשָּׁלְטוּ בְּשׂוֹנְאֵיהֶם.
ב נִקְהֲל֨וּ הַיְּהוּדִ֜ים בְּעָרֵיהֶ֗ם בְּכׇל־מְדִינוֹת֙ הַמֶּ֣לֶךְ אֲחַשְׁוֵר֔וֹשׁ לִשְׁלֹ֣חַ יָ֔ד בִּמְבַקְשֵׁ֖י רָֽעָתָ֑ם וְאִישׁ֙ לֹא־עָמַ֣ד לִפְנֵיהֶ֔ם‏[2] כִּֽי־נָפַ֥ל פַּחְדָּ֖ם עַל־כׇּל־הָעַמִּֽים׃ הַיְּהוּדִים הִתְכַּנְּסוּ בְּעָרֵיהֶם בְּכָל מְדִינוֹת הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ כְּדֵי לִפְגּוֹעַ בְּאֵלֶּה שֶׁחִפְּשׂוּ אֶת רָעָתָם, וְאַף אֶחָד לֹא נִסָּה לַעֲמוֹד בִּפְנֵיהֶם כִּי כָּל הָעַמִּים פָּחֲדוּ מֵהֶם.
ג וְכׇל־שָׂרֵ֨י הַמְּדִינ֜וֹת וְהָאֲחַשְׁדַּרְפְּנִ֣ים וְהַפַּח֗וֹת וְעֹשֵׂ֤י הַמְּלָאכָה֙ אֲשֶׁ֣ר לַמֶּ֔לֶךְ מְנַשְּׂאִ֖ים אֶת־הַיְּהוּדִ֑ים כִּֽי־נָפַ֥ל פַּֽחַד־מׇרְדֳּכַ֖י עֲלֵיהֶֽם׃ וְכָל שָׂרֵי הַמְּדִינוֹת וְהַנְּצִיבִים וְהַמּוֹשְׁלִים וְעוֹבְדֵי הַמֶּלֶךְ כִּבְּדוּ אֶת הַיְּהוּדִים כִּי הֵם פָּחֲדוּ מִמָּרְדֳּכַי.
ד כִּֽי־גָד֤וֹל מׇרְדֳּכַי֙ בְּבֵ֣ית הַמֶּ֔לֶךְ וְשׇׁמְע֖וֹ הוֹלֵ֣ךְ בְּכׇל־הַמְּדִינ֑וֹת כִּֽי־הָאִ֥ישׁ מׇרְדֳּכַ֖י הוֹלֵ֥ךְ וְגָדֽוֹל׃ כִּי מָרְדֳּכַי הָיָה חָשׁוּב בְּבֵית הַמֶּלֶךְ וְהַשֵּׁם שֶׁלּוֹ הִתְפַּרְסֵם בְּכָל הַמְּדִינוֹת כִּי הָאִישׁ מָרְדֳּכַי הוֹלֵךְ וּמִתְחַזֵּק.
ה וַיַּכּ֤וּ הַיְּהוּדִים֙ בְּכׇל־אֹ֣יְבֵיהֶ֔ם מַכַּת־חֶ֥רֶב וְהֶ֖רֶג וְאַבְדָ֑ן וַיַּֽעֲשׂ֥וּ בְשֹׂנְאֵיהֶ֖ם כִּרְצוֹנָֽם׃ הַיְּהוּדִים הִכּוּ אֶת כָּל אוֹיְבֵיהֶם בְּחֶרֶב וְהָרְגוּ וְהִשְׁמִידוּ אוֹתָם, וְעָשׂוּ בְּשׂוֹנְאֵיהֶם כִּרְצוֹנָם.
ו וּבְשׁוּשַׁ֣ן הַבִּירָ֗ה הָרְג֤וּ הַיְּהוּדִים֙ וְאַבֵּ֔ד חֲמֵ֥שׁ מֵא֖וֹת אִֽישׁ׃ וּבְשׁוּשַׁן הַבִּירָה הָרְגוּ הַיְּהוּדִים וְהִשְׁמִידוּ חֲמֵשׁ מֵאוֹת אִישׁ.

יא בַּיּ֣וֹם הַה֗וּא בָּ֣א מִסְפַּ֧ר הַֽהֲרוּגִ֛ים בְּשׁוּשַׁ֥ן הַבִּירָ֖ה לִפְנֵ֥י הַמֶּֽלֶךְ׃ בְּאוֹתוֹ יוֹם הוֹדִיעוּ לַמֶּלֶךְ אֶת מִסְפַּר הַהֲרוּגִים בְּשׁוּשַׁן הַבִּירָה.
יב וַיֹּ֨אמֶר הַמֶּ֜לֶךְ לְאֶסְתֵּ֣ר הַמַּלְכָּ֗ה בְּשׁוּשַׁ֣ן הַבִּירָ֡ה הָרְגוּ֩ הַיְּהוּדִ֨ים וְאַבֵּ֜ד חֲמֵ֧שׁ מֵא֣וֹת אִ֗ישׁ וְאֵת֙ עֲשֶׂ֣רֶת בְּנֵֽי־הָמָ֔ן בִּשְׁאָ֛ר מְדִינ֥וֹת הַמֶּ֖לֶךְ מֶ֣ה עָשׂ֑וּ וּמַה־שְּׁאֵֽלָתֵךְ֙ וְיִנָּ֣תֵֽן לָ֔ךְ וּמַה־בַּקָּשָׁתֵ֥ךְ ע֖וֹד וְתֵעָֽשׂ׃ אָמַר הַמֶּלֶךְ לְאֶסְתֵּר הַמַּלְכָּה: בְּשׁוּשַׁן הַבִּירָה הַיְּהוּדִים הָרְגוּ וְהִשְׁמִידוּ חֲמֵשׁ מֵאוֹת אִישׁ וְאֶת עֲשֶׂרֶת בְּנֵי הָמָן, בִּשְׁאָר מְדִינוֹת הַמֶּלֶךְ מִי יוֹדֵעַ מֶה עָשׂוּ! וּמָה רְצוֹנֵךְ וְיִנָּתֵן לָךְ וּמַה עוֹד אַתְּ מְבַקֶּשֶׁת וְזֶה יִתְבַּצֵּעַ!
יג וַתֹּ֤אמֶר אֶסְתֵּר֙ אִם־עַל־הַמֶּ֣לֶךְ ט֔וֹב יִנָּתֵ֣ן גַּם־מָחָ֗ר לַיְּהוּדִים֙ אֲשֶׁ֣ר בְּשׁוּשָׁ֔ן לַעֲשׂ֖וֹת כְּדָ֣ת הַיּ֑וֹם וְאֵ֛ת עֲשֶׂ֥רֶת בְּנֵֽי־הָמָ֖ן יִתְל֥וּ עַל־הָעֵֽץ׃ אָמְרָה אֶסְתֵּר: אִם הַמֶּלֶךְ מַסְכִּים, יִתְאַפְשֵׁר גַּם מָחָר לַיְּהוּדִים שֶׁבְּשׁוּשָׁן, לַעֲשׂוֹת כְּמוֹ הַחוֹק שֶׁל הַיּוֹם, וְאֶת גּוּפוֹתֵיהֶם שֶׁל עֲשֶׂרֶת בְּנֵי הָמָן יִתְלוּ עַל עַמּוּד הַתְּלִיָּה.
יד וַיֹּ֤אמֶר הַמֶּ֙לֶךְ֙ לְהֵֽעָשׂ֣וֹת כֵּ֔ן וַתִּנָּתֵ֥ן דָּ֖ת בְּשׁוּשָׁ֑ן וְאֵ֛ת עֲשֶׂ֥רֶת בְּנֵֽי־הָמָ֖ן תָּלֽוּ׃ הַמֶּלֶךְ אָמַר שֶׁכָּךְ יֵעָשֶׂה וְנִקְבַּע הַחוֹק בְּשׁוּשָׁן, וְאֶת עֲשֶׂרֶת בְּנֵי הָמָן תָּלוּ.
טו וַיִּֽקָּהֲל֞וּ היהודיים הַיְּהוּדִ֣ים אֲשֶׁר־בְּשׁוּשָׁ֗ן גַּ֠ם בְּי֣וֹם אַרְבָּעָ֤ה עָשָׂר֙ לְחֹ֣דֶשׁ אֲדָ֔ר וַיַּֽהַרְג֣וּ בְשׁוּשָׁ֔ן שְׁלֹ֥שׁ מֵא֖וֹת אִ֑ישׁ וּבַ֨בִּזָּ֔ה לֹ֥א שָׁלְח֖וּ אֶת־יָדָֽם׃ וְהַיְּהוּדִים שֶׁבְּשׁוּשָׁן הִתְכַּנְּסוּ גַּם בְּיוֹם י"ד לְחוֹדֶשׁ אֲדָר, וְהָרְגוּ בְּשׁוּשָׁן שְׁלוֹשׁ מֵאוֹת אִישׁ, וְלֹא הוֹשִׁיטוּ יָד לָקַחַת מֵהַשָּׁלָל.
טז וּשְׁאָ֣ר הַיְּהוּדִ֡ים אֲשֶׁר֩ בִּמְדִינ֨וֹת הַמֶּ֜לֶךְ נִקְהֲל֣וּ ׀ וְעָמֹ֣ד עַל־נַפְשָׁ֗ם וְנ֙וֹחַ֙ מֵאֹ֣יְבֵיהֶ֔ם וְהָרוֹג֙ בְּשֹׂ֣נְאֵיהֶ֔ם חֲמִשָּׁ֥ה וְשִׁבְעִ֖ים אָ֑לֶף וּבַ֨בִּזָּ֔ה לֹ֥א שָֽׁלְח֖וּ אֶת־יָדָֽם׃ וּשְׁאָר הַיְּהוּדִים שֶׁבִּמְדִינוֹת הַמֶּלֶךְ הִתְכַּנְּסוּ וְעָמְדוּ עַל נַפְשָׁם וְהָיָה לָהֶם שֶׁקֶט מֵאוֹיְבֵיהֶם, וְהֵם הָרְגוּ בְּשׂוֹנְאֵיהֶם שִׁבְעִים וַחֲמִשָּׁה אֶלֶף, וְלֹא הוֹשִׁיטוּ יָד לָקַחַת מֵהַשָּׁלָל.
יז בְּיוֹם־שְׁלוֹשָׁ֥ה עָשָׂ֖ר לְחֹ֣דֶשׁ אֲדָ֑ר וְנ֗וֹחַ בְּאַרְבָּעָ֤ה עָשָׂר֙ בּ֔וֹ וְעָשֹׂ֣ה אֹת֔וֹ י֖וֹם מִשְׁתֶּ֥ה וְשִׂמְחָֽה׃ בְּיוֹם י"ג לְחוֹדֶשׁ אֲדָר הֵם נִלְחֲמוּ, וְהָיָה לָהֶם שֶׁקֶט בְּיוֹם י"ד בּוֹ וְעָשׂוּ אוֹתוֹ יוֹם סְעוּדָה עִם יַיִן וְשִׂמְחָה.
יח והיהודיים וְהַיְּהוּדִ֣ים אֲשֶׁר־בְּשׁוּשָׁ֗ן נִקְהֲלוּ֙ בִּשְׁלוֹשָׁ֤ה עָשָׂר֙ בּ֔וֹ וּבְאַרְבָּעָ֥ה עָשָׂ֖ר בּ֑וֹ וְנ֗וֹחַ בַּחֲמִשָּׁ֤ה עָשָׂר֙ בּ֔וֹ וְעָשֹׂ֣ה אֹת֔וֹ י֖וֹם מִשְׁתֶּ֥ה וְשִׂמְחָֽה׃ וְהַיְּהוּדִים שֶׁבְּשׁוּשָׁן הִתְכַּנְּסוּ לְהִלָּחֵם בְּיוֹם י"ג בַּחוֹדֶשׁ וְגַם בְּיוֹם י"ד, וְהָיָה לָהֶם שֶׁקֶט בְּיוֹם ט"ו וְעָשׂוּ אוֹתוֹ יוֹם סְעוּדָה עִם יַיִן וְשִׂמְחָה.
יט עַל־כֵּ֞ן הַיְּהוּדִ֣ים הפרוזים הַפְּרָזִ֗ים הַיֹּשְׁבִים֮ בְּעָרֵ֣י הַפְּרָזוֹת֒ עֹשִׂ֗ים אֵ֠ת י֣וֹם אַרְבָּעָ֤ה עָשָׂר֙ לְחֹ֣דֶשׁ אֲדָ֔ר שִׂמְחָ֥ה וּמִשְׁתֶּ֖ה וְי֣וֹם ט֑וֹב וּמִשְׁלֹ֥חַ מָנ֖וֹת אִ֥ישׁ לְרֵעֵֽהוּ׃ לָכֵן הַיְּהוּדִים הַ"פְּרָזִים" – אֵלֶּה שֶׁיּוֹשְׁבִים בְּעָרִים שֶׁאֵינָם מֻקָּפוֹת חוֹמָה, עוֹשִׂים בְּיוֹם י"ד לְחוֹדֶשׁ אֲדָר יוֹם שִׂמְחָה וּסְעוּדָה עִם יַיִן וְיוֹם טוֹב וְשׁוֹלְחִים מָנוֹת אִישׁ לַחֲבֵרוֹ.

כ וַיִּכְתֹּ֣ב מׇרְדֳּכַ֔י אֶת־הַדְּבָרִ֖ים הָאֵ֑לֶּה וַיִּשְׁלַ֨ח סְפָרִ֜ים אֶל־כׇּל־הַיְּהוּדִ֗ים אֲשֶׁר֙ בְּכׇל־מְדִינוֹת֙ הַמֶּ֣לֶךְ אֲחַשְׁוֵר֔וֹשׁ הַקְּרוֹבִ֖ים וְהָרְחוֹקִֽים׃ מָרְדֳּכַי כָּתַב אֶת הַדְּבָרִים הָאֵלֶּה, וְשָׁלַח מִכְתָּבִים אֶל כָּל הַיְּהוּדִים שֶׁבְּכָל מְדִינוֹת הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ, הַקְּרוֹבִים וְהָרְחוֹקִים.
כא לְקַיֵּם֮ עֲלֵיהֶם֒ לִהְי֣וֹת עֹשִׂ֗ים אֵ֠ת י֣וֹם אַרְבָּעָ֤ה עָשָׂר֙ לְחֹ֣דֶשׁ אֲדָ֔ר וְאֵ֛ת יוֹם־חֲמִשָּׁ֥ה עָשָׂ֖ר בּ֑וֹ בְּכׇל־שָׁנָ֖ה וְשָׁנָֽה׃ שֶׁיְּקַבְּלוּ עַל עַצְמָם לַעֲשׂוֹת אֶת יוֹם י"ד לְחוֹדֶשׁ אֲדָר וְאֶת יוֹם ט"ו בּוֹ בְּכָל שָׁנָה וְשָׁנָה.
כב כַּיָּמִ֗ים אֲשֶׁר־נָ֨חוּ בָהֶ֤ם הַיְּהוּדִים֙ מֵאֹ֣יְבֵיהֶ֔ם וְהַחֹ֗דֶשׁ אֲשֶׁר֩ נֶהְפַּ֨ךְ לָהֶ֤ם מִיָּגוֹן֙ לְשִׂמְחָ֔ה וּמֵאֵ֖בֶל לְי֣וֹם ט֑וֹב לַעֲשׂ֣וֹת אוֹתָ֗ם יְמֵי֙ מִשְׁתֶּ֣ה וְשִׂמְחָ֔ה וּמִשְׁלֹ֤חַ מָנוֹת֙ אִ֣ישׁ לְרֵעֵ֔הוּ וּמַתָּנ֖וֹת לָֽאֶבְיֹנִֽים׃ כְּמוֹ הַיָּמִים שֶׁהָיָה לַיְּהוּדִים שֶׁקֶט מֵאוֹיְבֵיהֶם, וְהַחוֹדֶשׁ שֶׁנֶּהְפַּךְ לָהֶם מִצַּעַר לְשִׂמְחָה וּמֵאֵבֶל לְיוֹם טוֹב, לַעֲשׂוֹת אוֹתָם יָמִים שֶׁל סְעוּדָה עִם יַיִן וְשִׂמְחָה וּשְׁלִיחַת מָנוֹת אִישׁ לַחֲבֵרוֹ וּמַתָּנוֹת לַעֲנִיִּים.
כג וְקִבֵּל֙ הַיְּהוּדִ֔ים אֵ֥ת אֲשֶׁר־הֵחֵ֖לּוּ לַעֲשׂ֑וֹת וְאֵ֛ת אֲשֶׁר־כָּתַ֥ב מׇרְדֳּכַ֖י אֲלֵיהֶֽם׃ וְהַיְּהוּדִים קִבְּלוּ עַל עַצְמָם אֶת מַה שֶּׁהֵם כְּבָר הִתְחִילוּ לַעֲשׂוֹת וְאֶת מַה שֶׁמָּרְדֳּכַי כָּתַב לָהֶם.
כד כִּי֩ הָמָ֨ן בֶּֽן־הַמְּדָ֜תָא הָֽאֲגָגִ֗י צֹרֵר֙ כׇּל־הַיְּהוּדִ֔ים חָשַׁ֥ב עַל־הַיְּהוּדִ֖ים לְאַבְּדָ֑ם וְהִפִּ֥ל פּוּר֙ ה֣וּא הַגּוֹרָ֔ל לְהֻמָּ֖ם וּֽלְאַבְּדָֽם׃ כִּי הָמָן בֶּן הַמְּדָתָא הַצֶּאֱצָא שֶׁל אֲגָג אוֹיֵב כָּל הַיְּהוּדִים תִּכְנֵן לְהַשְׁמִיד אֶת הַיְּהוּדִים, וְהִגְרִיל גּוֹרָל מָתַי לַעֲשׂוֹת בָּהֶם מְהוּמוֹת וּלְהַשְׁמִיד אוֹתָם.
כה וּבְבֹאָהּ֮ לִפְנֵ֣י הַמֶּ֒לֶךְ֒ אָמַ֣ר עִם־הַסֵּ֔פֶר יָשׁ֞וּב מַחֲשַׁבְתּ֧וֹ הָרָעָ֛ה אֲשֶׁר־חָשַׁ֥ב עַל־הַיְּהוּדִ֖ים עַל־רֹאשׁ֑וֹ וְתָל֥וּ אֹת֛וֹ וְאֶת־בָּנָ֖יו עַל־הָעֵֽץ׃ וּכְשֶׁאֶסְתֵּר בָּאָה לִפְנֵי הַמֶּלֶךְ, הַמֶּלֶךְ אָמַר לִכְתּוֹב בַּמִּכְתָּבִים שֶׁהַתָּכְנִית הָרָעָה שֶׁהָמָן חָשַׁב עַל הַיְּהוּדִים תִּתְהַפֵּךְ עָלָיו, וְתָלוּ אוֹתוֹ וְאֶת בָּנָיו עַל הָעֵץ.
כו עַל־כֵּ֡ן קָֽרְאוּ֩ לַיָּמִ֨ים הָאֵ֤לֶּה פוּרִים֙ עַל־שֵׁ֣ם הַפּ֔וּר עַל־כֵּ֕ן עַל־כׇּל־דִּבְרֵ֖י הָאִגֶּ֣רֶת הַזֹּ֑את וּמָֽה־רָא֣וּ עַל־כָּ֔כָה וּמָ֥ה הִגִּ֖יעַ אֲלֵיהֶֽם׃ לָכֵן קָרְאוּ לַיָּמִים הָאֵלֶּה "פּוּרִים" שֶׁפֵּרוּשׁוֹ גּוֹרָלוֹת, עַל שֵׁם הַגּוֹרָל. לָכֵן נִכְתְּבוּ כָּל דִּבְרֵי הַמְּגִלָּה הַזֹּאת – מָה רָאוּ שֶׁמִּשּׁוּם כָּךְ קָבְעוּ לַעֲשׂוֹת כָּךְ וּמָה קָּרָה לָהֶם –
כז קִיְּמ֣וּ וקבל וְקִבְּל֣וּ הַיְּהוּדִים֩ ׀ עֲלֵיהֶ֨ם ׀ וְעַל־זַרְעָ֜ם וְעַ֨ל כׇּל־הַנִּלְוִ֤ים עֲלֵיהֶם֙ וְלֹ֣א יַעֲב֔וֹר לִהְי֣וֹת עֹשִׂ֗ים אֵ֣ת שְׁנֵ֤י הַיָּמִים֙ הָאֵ֔לֶּה כִּכְתָבָ֖ם וְכִזְמַנָּ֑ם בְּכׇל־שָׁנָ֖ה וְשָׁנָֽה׃ הַיְּהוּדִים קָבְעוּ וְקִבְּלוּ עַל עַצְמָם וְעַל צֶאֱצָאֵיהֶם וְעַל כָּל הַגֵּרִים שֶׁמִּתְלַוִּים אֲלֵיהֶם שֶׁלֹּא יִתְבַּטֵל לְעוֹלָם, שֶׁיַּחְגְּגוּ אֶת שְׁנֵי הַיָּמִים הָאֵלֶּה בִּזְמַנָּם כְּפִי שֶׁנִּכְתְּבוּ, בְּכָל שָׁנָה וְשָׁנָה.
כח וְהַיָּמִ֣ים הָ֠אֵ֠לֶּה נִזְכָּרִ֨ים וְנַעֲשִׂ֜ים בְּכׇל־דּ֣וֹר וָד֗וֹר מִשְׁפָּחָה֙ וּמִשְׁפָּחָ֔ה מְדִינָ֥ה וּמְדִינָ֖ה וְעִ֣יר וָעִ֑יר וִימֵ֞י הַפּוּרִ֣ים הָאֵ֗לֶּה לֹ֤א יַֽעַבְרוּ֙ מִתּ֣וֹךְ הַיְּהוּדִ֔ים וְזִכְרָ֖ם לֹא־יָס֥וּף מִזַּרְעָֽם׃ וְהַיָּמִים הָאֵלֶּה מְצֻיָּנִים וְנֶחְגָּגִים בְּכָל דּוֹר וָדוֹר, בְּכָל מִשְׁפָּחָה וּמִשְׁפָּחָה בְּכָל מְדִינָה וּמְדִינָה וּבְכָל עִיר וְעִיר, וִימֵי הַפּוּרִים הָאֵלֶּה לֹא יִתְבַּטְּלוּ מִמִּנְהֲגֵי הַיְּהוּדִים וְצֶאֱצָאֵיהֶם לֹא יַפְסִיקוּ לַחְגּוֹג בָּהֶם.
ס
כט וַ֠תִּכְתֹּ֠ב אֶסְתֵּ֨ר הַמַּלְכָּ֧ה בַת־אֲבִיחַ֛יִל וּמׇרְדֳּכַ֥י הַיְּהוּדִ֖י אֶת־כׇּל־תֹּ֑קֶף לְקַיֵּ֗ם אֵ֣ת אִגֶּ֧רֶת הַפֻּרִ֛ים הַזֹּ֖את הַשֵּׁנִֽית׃ אֶסְתֵּר הַמַּלְכָּה בַּת אֲבִיחַיִל וּמָרְדֳּכַי הַיְּהוּדִי כָּתְבוּ וְחִיְּבוּ בְּכָל תּוֹקֶף לְקַיֵּם אֶת צַו הָאִגֶּרֶת הַשְּׁנִיָּה – לְקַיֵּם אֶת יְמֵי הַפּוּרִים.
ל וַיִּשְׁלַ֨ח סְפָרִ֜ים אֶל־כׇּל־הַיְּהוּדִ֗ים אֶל־שֶׁ֨בַע וְעֶשְׂרִ֤ים וּמֵאָה֙ מְדִינָ֔ה מַלְכ֖וּת אֲחַשְׁוֵר֑וֹשׁ דִּבְרֵ֥י שָׁל֖וֹם וֶאֱמֶֽת׃ וּמָרְדֳּכַי שָׁלַח מִכְתָּבִים אֶל כָּל הַיְּהוּדִים, אֶל מֵאָה עֶשְׂרִים וְשֶׁבַע הַמְּדִינוֹת שֶׁל מַלְכוּת אֲחַשְׁוֵרוֹשׁ, וּבָהֶם דִּבְרֵי שָׁלוֹם וּקְבִיעָה,
לא לְקַיֵּ֡ם אֶת־יְמֵי֩ הַפֻּרִ֨ים הָאֵ֜לֶּה בִּזְמַנֵּיהֶ֗ם כַּאֲשֶׁר֩ קִיַּ֨ם עֲלֵיהֶ֜ם מׇרְדֳּכַ֤י הַיְּהוּדִי֙ וְאֶסְתֵּ֣ר הַמַּלְכָּ֔ה וְכַאֲשֶׁ֛ר קִיְּמ֥וּ עַל־נַפְשָׁ֖ם וְעַל־זַרְעָ֑ם דִּבְרֵ֥י הַצּוֹמ֖וֹת וְזַעֲקָתָֽם׃ שֶׁיִּקְבְּעוּ אֶת יְמֵי הַפּוּרִים הָאֵלֶּה בִּזְמַנֵּיהֶם כְּמוֹ שֶׁקָּבְעוּ עֲבוּרָם מָרְדֳּכַי הַיְּהוּדִי וְאֶסְתֵּר הַמַּלְכָּה, וּכְמוֹ שֶׁקִּבְּלוּ עַל עַצְמָם וְעַל צֶאֱצָאֵיהֶם, אֶת דִּבְרֵי הַצּוֹמוֹת וְהַזְּעָקָה בָּהֶם.
לב וּמַאֲמַ֣ר אֶסְתֵּ֔ר קִיַּ֕ם דִּבְרֵ֥י הַפֻּרִ֖ים הָאֵ֑לֶּה וְנִכְתָּ֖ב בַּסֵּֽפֶר׃ כְּפִי שֶׁאָמְרָה אֶסְתֵּר – הוּא קָבַע דִּבְרֵי הַפּוּרִים הָאֵלֶּה, כַּכָּתוּב בַּמִּכְתָּבִים.
ס
י א וַיָּ֩שֶׂם֩ הַמֶּ֨לֶךְ אחשרש אֲחַשְׁוֵר֧וֹשׁ ׀ מַ֛ס עַל־הָאָ֖רֶץ וְאִיֵּ֥י הַיָּֽם׃ הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ הֵטִיל מַס עַל תּוֹשְׁבֵי הָאָרֶץ וְאִיֵּי הַיָּם, לְאַחַר הַהֲנָחָה בַּמָּס שֶׁבְּהַמְלָכַת אֶסְתֵּר.
ב וְכׇל־מַעֲשֵׂ֤ה תׇקְפּוֹ֙ וּגְב֣וּרָת֔וֹ וּפָרָשַׁת֙ גְּדֻלַּ֣ת מׇרְדֳּכַ֔י אֲשֶׁ֥ר גִּדְּל֖וֹ הַמֶּ֑לֶךְ הֲלוֹא־הֵ֣ם כְּתוּבִ֗ים עַל־סֵ֙פֶר֙ דִּבְרֵ֣י הַיָּמִ֔ים לְמַלְכֵ֖י מָדַ֥י וּפָרָֽס׃ וְכָל סִפּוּרֵי הַחוֹזֶק וְהַגְּבוּרָה שֶׁל אֲחַשְׁוֵרוֹשׁ וּפִרְטֵי גְּדֻלָּתוֹ שֶׁל מָרְדֳּכַי שֶׁהַמֶּלֶךְ רוֹמֵם אוֹתוֹ – כְּתוּבִים בְּסֵפֶר הַהִסְטוֹרְיָה שֶׁל מַלְכֵי מָדַי וּפָרָס.
ג כִּ֣י ׀ מׇרְדֳּכַ֣י הַיְּהוּדִ֗י מִשְׁנֶה֙ לַמֶּ֣לֶךְ אֲחַשְׁוֵר֔וֹשׁ וְגָדוֹל֙ לַיְּהוּדִ֔ים וְרָצ֖וּי לְרֹ֣ב אֶחָ֑יו דֹּרֵ֥שׁ טוֹב֙ לְעַמּ֔וֹ וְדֹבֵ֥ר שָׁל֖וֹם לְכׇל־זַרְעֽוֹ׃ שֶׁמָּרְדֳּכַי הַיְּהוּדִי הָיָה מִשְׁנֶה לַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ וְחָשׁוּב בֵּין הַיְּהוּדִים וְרָצוּי לְאֶחָיו הָרַבִּים, מְחַפֵּשׂ לַעֲשׂוֹת טוֹב לְעַמּוֹ וּמְדַבֵּר לְמַעַן שְׁלוֹם כָּל בְּנֵי עַמּוֹ.
