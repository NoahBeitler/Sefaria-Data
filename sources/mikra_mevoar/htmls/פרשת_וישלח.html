ראשון[עריכה]

פ פָּרָשַׁת הַהֲכָנוֹת לַפְּגִישָׁה עִם עֵשָׂו
לב ד וַיִּשְׁלַ֨ח יַעֲקֹ֤ב מַלְאָכִים֙ לְפָנָ֔יו אֶל־עֵשָׂ֖ו אָחִ֑יו אַ֥רְצָה שֵׂעִ֖יר שְׂדֵ֥ה אֱדֽוֹם׃ יַעֲקֹב שָׁלַח שְׁלִיחִים שֶׁיֵּלְכוּ לְפָנָיו אֶל עֵשָׂו אָחִיו לְאֶרֶץ שֵׂעִיר – לִשְׂדֵה אֱדוֹם.
ה וַיְצַ֤ו אֹתָם֙ לֵאמֹ֔ר כֹּ֣ה תֹאמְר֔וּן לַֽאדֹנִ֖י לְעֵשָׂ֑ו כֹּ֤ה אָמַר֙ עַבְדְּךָ֣ יַעֲקֹ֔ב עִם־לָבָ֣ן גַּ֔רְתִּי וָאֵחַ֖ר עַד־עָֽתָּה׃ הוּא צִוָּה אוֹתָם וְאָמַר: כָּךְ תֹּאמְרוּ לַאדֹנִי לְעֵשָׂו: כָּךְ אָמַר עַבְדְּךָ יַעֲקֹב – גַּרְתִּי אֵצֶל לָבָן וְהִתְעַכַּבְתִּי עַד עַכְשָׁיו.
ו וַֽיְהִי־לִי֙ שׁ֣וֹר וַחֲמ֔וֹר צֹ֖אן וְעֶ֣בֶד וְשִׁפְחָ֑ה וָֽאֶשְׁלְחָה֙ לְהַגִּ֣יד לַֽאדֹנִ֔י לִמְצֹא־חֵ֖ן בְּעֵינֶֽיךָ׃ רָכַשְׁתִּי שְׁוָרִים, חֲמוֹרִים, צֹאן, עֲבָדִים וּשְׁפָחוֹת, וְשָׁלַחְתִּי שְׁלִיחִים לְסַפֵּר אֶת זֶה לַאדֹנִי לִמְצֹא חֵן בְּעֵינֶיךָ.
ז וַיָּשֻׁ֙בוּ֙ הַמַּלְאָכִ֔ים אֶֽל־יַעֲקֹ֖ב לֵאמֹ֑ר בָּ֤אנוּ אֶל־אָחִ֙יךָ֙ אֶל־עֵשָׂ֔ו וְגַם֙ הֹלֵ֣ךְ לִקְרָֽאתְךָ֔ וְאַרְבַּע־מֵא֥וֹת אִ֖ישׁ עִמּֽוֹ׃ הַשְּׁלִיחִים חָזְרוּ אֶל יַעֲקֹב וְאָמְרוּ: בָּאנוּ לְאָחִיךָ עֵשָׂו, וְגַם הוּא הוֹלֵךְ לִקְרָאתְךָ לִפְגּוֹשׁ אוֹתְךָ, וְיֵשׁ אִתּוֹ אַרְבַּע מֵאוֹת אִישׁ.
ח וַיִּירָ֧א יַעֲקֹ֛ב מְאֹ֖ד וַיֵּ֣צֶר ל֑וֹ וַיַּ֜חַץ אֶת־הָעָ֣ם אֲשֶׁר־אִתּ֗וֹ וְאֶת־הַצֹּ֧אן וְאֶת־הַבָּקָ֛ר וְהַגְּמַלִּ֖ים לִשְׁנֵ֥י מַחֲנֽוֹת׃ יַעֲקֹב פָּחַד מְאוֹד וְהִרְגִּישׁ שֶׁהוּא בְּצָרָה, הוּא חִלֵּק אֶת הָעָם שֶׁאִתּוֹ וְאֶת הַצֹּאן וְאֶת הַבָּקָר וְהַגְּמַלִּים לִשְׁנֵי מַחֲנוֹת.
ט וַיֹּ֕אמֶר אִם־יָב֥וֹא עֵשָׂ֛ו אֶל־הַמַּחֲנֶ֥ה הָאַחַ֖ת וְהִכָּ֑הוּ וְהָיָ֛ה הַמַּחֲנֶ֥ה הַנִּשְׁאָ֖ר לִפְלֵיטָֽה׃ וְאָמַר: אִם יָבוֹא עֵשָׂו לְמַחֲנֶה אֶחָד וְיַכֶּה אוֹתוֹ – הַמַּחֲנֶה שֶׁנִּשְׁאָר יִנָּצֵל.
י וַיֹּ֘אמֶר֮ יַעֲקֹב֒ אֱלֹהֵי֙ אָבִ֣י אַבְרָהָ֔ם וֵאלֹהֵ֖י אָבִ֣י יִצְחָ֑ק יְהֹוָ֞ה הָאֹמֵ֣ר אֵלַ֗י שׁ֧וּב לְאַרְצְךָ֛ וּלְמוֹלַדְתְּךָ֖ וְאֵיטִ֥יבָה עִמָּֽךְ׃ יַעֲקֹב אָמַר: אֱלֹהֵי אָבִי אַבְרָהָם וֵאלֹהֵי אָבִי יִצְחָק – ה' שֶׁאָמַר לִי חֲזוֹר לְאַרְצְךָ וּלְמוֹלַדְתְּךָ וְאֵיטִיב לְךָ.
יא קָטֹ֜נְתִּי מִכֹּ֤ל הַחֲסָדִים֙ וּמִכׇּל־הָ֣אֱמֶ֔ת אֲשֶׁ֥ר עָשִׂ֖יתָ אֶת־עַבְדֶּ֑ךָ כִּ֣י בְמַקְלִ֗י עָבַ֙רְתִּי֙ אֶת־הַיַּרְדֵּ֣ן הַזֶּ֔ה וְעַתָּ֥ה הָיִ֖יתִי לִשְׁנֵ֥י מַחֲנֽוֹת׃ קָטוֹנְתִּי מִלִּהְיוֹת רָאוּי לְקַבֵּל אֶת כָּל הַחֲסָדִים וְאֶת כָּל הַטּוֹבוֹת שֶׁעָשִׂיתָ לְעַבְדְּךָ, שֶׁהֲרֵי עָבַרְתִּי אֶת הַיַּרְדֵּן הַזֶּה רַק עִם מַקְלִי – וְעַכְשָׁיו נִהְיֵיתִי שְׁנֵי מַחֲנוֹת שְׁלֵמִים.
יב הַצִּילֵ֥נִי נָ֛א מִיַּ֥ד אָחִ֖י מִיַּ֣ד עֵשָׂ֑ו כִּֽי־יָרֵ֤א אָנֹכִי֙ אֹת֔וֹ פֶּן־יָב֣וֹא וְהִכַּ֔נִי אֵ֖ם עַל־בָּנִֽים׃ הַצִּילֵנִי בְּבַקָּשָׁה מִיַּד אָחִי – מִיַּד עֵשָׂו, כִּי אֲנִי פּוֹחֵד מִמֶּנּוּ, שֶׁלֹּא יָבוֹא וְיַכֶּה אוֹתָנוּ אֶת הָאִמָּהוֹת עִם הַבָּנִים.
יג וְאַתָּ֣ה אָמַ֔רְתָּ הֵיטֵ֥ב אֵיטִ֖יב עִמָּ֑ךְ וְשַׂמְתִּ֤י אֶֽת־זַרְעֲךָ֙ כְּח֣וֹל הַיָּ֔ם אֲשֶׁ֥ר לֹא־יִסָּפֵ֖ר מֵרֹֽב׃ וְאַתָּה אָמַרְתָּ: אֵיטִיב לְךְ, וְאַרְבֶּה אֶת צֶאֱצָאֶיךָ כְּמוֹ חוֹל הַיָּם – שֶׁלֹּא יִסָּפֵר מֵרוֹב.

שני[עריכה]

יד וַיָּ֥לֶן שָׁ֖ם בַּלַּ֣יְלָה הַה֑וּא וַיִּקַּ֞ח מִן־הַבָּ֧א בְיָד֛וֹ מִנְחָ֖ה לְעֵשָׂ֥ו אָחִֽיו׃ הוּא יָשַׁן שָׁם בַּלַּיְלָה הַהוּא, וְלָקַח מֵהָרְכוּשׁ שֶׁבְּיָדוֹ לָתֵת מַתָּנָה לְעֵשָׂו אָחִיו.
טו עִזִּ֣ים מָאתַ֔יִם וּתְיָשִׁ֖ים עֶשְׂרִ֑ים רְחֵלִ֥ים מָאתַ֖יִם וְאֵילִ֥ים עֶשְׂרִֽים׃ מָאתַיִם עִזִּים, עֶשְׂרִים תְּיָשִׁים, מָאתַיִם כְּבָשִׂים וְעֶשְׂרִים אֵילִים.
טז גְּמַלִּ֧ים מֵינִיק֛וֹת וּבְנֵיהֶ֖ם שְׁלֹשִׁ֑ים פָּר֤וֹת אַרְבָּעִים֙ וּפָרִ֣ים עֲשָׂרָ֔ה אֲתֹנֹ֣ת עֶשְׂרִ֔ים וַעְיָרִ֖ם עֲשָׂרָֽה׃ שְׁלוֹשִׁים גְּמַלִּים נְקֵבוֹת עִם בְּנֵיהֶם, אַרְבָּעִים פָּרוֹת וַעֲשָׂרָה פָּרִים, עֶשְׂרִים אֲתוֹנוֹת וַעֲשָׂרָה חֲמוֹרִים צְעִירִים.
יז וַיִּתֵּן֙ בְּיַד־עֲבָדָ֔יו עֵ֥דֶר עֵ֖דֶר לְבַדּ֑וֹ וַיֹּ֤אמֶר אֶל־עֲבָדָיו֙ עִבְר֣וּ לְפָנַ֔י וְרֶ֣וַח תָּשִׂ֔ימוּ בֵּ֥ין עֵ֖דֶר וּבֵ֥ין עֵֽדֶר׃ הוּא נָתַן בְּיַד עֲבָדָיו כָּל עֵדֶר בְּנִפְרָד, וְאָמַר לַעֲבָדָיו: עִבְרוּ לְפָנַי וְשִׂימוּ רֶוַח בֵּין כָּל עֵדֶר וְעֵדֶר!
יח וַיְצַ֥ו אֶת־הָרִאשׁ֖וֹן לֵאמֹ֑ר כִּ֣י יִֽפְגׇשְׁךָ֞ עֵשָׂ֣ו אָחִ֗י וּשְׁאֵֽלְךָ֙ לֵאמֹ֔ר לְמִי־אַ֙תָּה֙ וְאָ֣נָה תֵלֵ֔ךְ וּלְמִ֖י אֵ֥לֶּה לְפָנֶֽיךָ׃ הוּא צִוָּה אֶת הָרִאשׁוֹן וְאָמַר: כְּשֶׁעֵשָׂו אָחִי יִפְגוֹשׁ אוֹתְךָ וְיִשְׁאַל אוֹתְךָ: שֶׁל מִי אַתָּה וּלְאָן אַתָּה הוֹלֵךְ וְשֶׁל מִי הָעֲדָרִים הָאֵלֶּה שֶׁלְּפָנֶיךָ?
יט וְאָֽמַרְתָּ֙ לְעַבְדְּךָ֣ לְיַעֲקֹ֔ב מִנְחָ֥ה הִוא֙ שְׁלוּחָ֔ה לַֽאדֹנִ֖י לְעֵשָׂ֑ו וְהִנֵּ֥ה גַם־ה֖וּא אַחֲרֵֽינוּ׃ תֹּאמַר: אֲנִי שֶׁל עַבְדְּךָ יַעֲקֹב, וְזֹאת מַתָּנָה בְּמִשְׁלוֹחַ לַאדֹנִי לְעֵשָׂו, וְיַעֲקֹב גַּם הוּא אַחֲרֵינוּ.
כ וַיְצַ֞ו גַּ֣ם אֶת־הַשֵּׁנִ֗י גַּ֚ם אֶת־הַשְּׁלִישִׁ֔י גַּ֚ם אֶת־כׇּל־הַהֹ֣לְכִ֔ים אַחֲרֵ֥י הָעֲדָרִ֖ים לֵאמֹ֑ר כַּדָּבָ֤ר הַזֶּה֙ תְּדַבְּר֣וּן אֶל־עֵשָׂ֔ו בְּמֹצַאֲכֶ֖ם אֹתֽוֹ׃ הוּא צִוָּה גַּם אֶת הַשֵּׁנִי גַּם אֶת הַשְּׁלִישִׁי גַּם אֶת כָּל הַהוֹלְכִים אַחֲרֵי הָעֲדָרִים וְאָמַר: כַּדָּבָר הַזֶּה תְּדַבְּרוּ לְעֵשָׂו כְּשֶׁתִּפְגְּשׁוּ אוֹתוֹ.
כא וַאֲמַרְתֶּ֕ם גַּ֗ם הִנֵּ֛ה עַבְדְּךָ֥ יַעֲקֹ֖ב אַחֲרֵ֑ינוּ כִּֽי־אָמַ֞ר אֲכַפְּרָ֣ה פָנָ֗יו בַּמִּנְחָה֙ הַהֹלֶ֣כֶת לְפָנָ֔י וְאַחֲרֵי־כֵן֙ אֶרְאֶ֣ה פָנָ֔יו אוּלַ֖י יִשָּׂ֥א פָנָֽי׃ וְתֹאמְרוּ: גַּם עַבְדְּךָ יַעֲקֹב הִנֵּה הוּא אַחֲרֵינוּ, כִּי אָמַר אֲפַיֵּס אוֹתוֹ בַּמַּתָּנָה שֶׁמּוּבֵאת לְפָנַי וְאַחֲרֵי כֵן אֶרְאֶה אֶת פָּנָיו, אוּלַי יְקַבֵּל כָּךְ אֶת פָּנַי בְּרָצוֹן.
כב וַתַּעֲבֹ֥ר הַמִּנְחָ֖ה עַל־פָּנָ֑יו וְה֛וּא לָ֥ן בַּלַּֽיְלָה־הַה֖וּא בַּֽמַּחֲנֶֽה׃ הַמַּתָּנָה עָבְרָה לְפָנָיו וְהוּא יָשַׁן בַּלַּיְלָה הַהוּא בַּמַּחֲנֶה.
פָּרָשַׁת הַמַּאֲבָק
כג וַיָּ֣קׇם ׀ בַּלַּ֣יְלָה ה֗וּא וַיִּקַּ֞ח אֶת־שְׁתֵּ֤י נָשָׁיו֙ וְאֶת־שְׁתֵּ֣י שִׁפְחֹתָ֔יו וְאֶת־אַחַ֥ד עָשָׂ֖ר יְלָדָ֑יו וַֽיַּעֲבֹ֔ר אֵ֖ת מַעֲבַ֥ר יַבֹּֽק׃ הוּא קָם בַּלַּיְלָה הַהוּא וְלָקַח אֶת שְׁתֵּי נָשָׁיו וְאֶת שְׁתֵּי שִׁפְחוֹתָיו וְאֶת אַחַד עָשָׂר יְלָדָיו וְעָבַר אֶת הַמַּעֲבַר שֶׁבְּנַחַל יַבּוֹק.
כד וַיִּ֨קָּחֵ֔ם וַיַּֽעֲבִרֵ֖ם אֶת־הַנָּ֑חַל וַֽיַּעֲבֵ֖ר אֶת־אֲשֶׁר־לֽוֹ׃ הוּא לָקַח אוֹתָם וְהֶעֱבִיר אוֹתָם אֶת הַנָּחַל וְהֶעֱבִיר אֶת הָרְכוּשׁ שֶׁלּוֹ.
כה וַיִּוָּתֵ֥ר יַעֲקֹ֖ב לְבַדּ֑וֹ וַיֵּאָבֵ֥ק אִישׁ֙ עִמּ֔וֹ עַ֖ד עֲל֥וֹת הַשָּֽׁחַר׃ יַעֲקֹב נִשְׁאַר לְבַדּוֹ, וְאִישׁ אֶחָד נֶאֱבַק אִתּוֹ עַד עֲלוֹת הַשַּׁחַר.
כו וַיַּ֗רְא כִּ֣י לֹ֤א יָכֹל֙ ל֔וֹ וַיִּגַּ֖ע בְּכַף־יְרֵכ֑וֹ וַתֵּ֙קַע֙ כַּף־יֶ֣רֶךְ יַעֲקֹ֔ב בְּהֵאָֽבְק֖וֹ עִמּֽוֹ׃ הוּא רָאָה שֶׁאֵינוֹ יָכוֹל לְנַצֵּח אוֹתוֹ, וְהִכָּה בְּכַף יְרֵכוֹ, וְנַעֲשָׂה נֶקַע בְּכַף יֶרֶךְ יַעֲקֹב כְּשֶׁנֶּאֱבַק אִתּוֹ.
כז וַיֹּ֣אמֶר שַׁלְּחֵ֔נִי כִּ֥י עָלָ֖ה הַשָּׁ֑חַר וַיֹּ֙אמֶר֙ לֹ֣א אֲשַֽׁלֵּחֲךָ֔ כִּ֖י אִם־בֵּרַכְתָּֽנִי׃ אָמַר הָאִישׁ: תֵּן לִי לָלֶכֶת כִּי הַשַּׁחַר עָלָה, אָמַר לוֹ: לֹא אֶתֵּן לְךָ אֶלָּא אִם תְּבָרֵךְ אוֹתִי.
כח וַיֹּ֥אמֶר אֵלָ֖יו מַה־שְּׁמֶ֑ךָ וַיֹּ֖אמֶר יַעֲקֹֽב׃ אָמַר לוֹ: מַה שִּׁמְךָ? אָמַר לוֹ: יַעֲקֹב.
כט וַיֹּ֗אמֶר לֹ֤א יַעֲקֹב֙ יֵאָמֵ֥ר עוֹד֙ שִׁמְךָ֔ כִּ֖י אִם־יִשְׂרָאֵ֑ל כִּֽי־שָׂרִ֧יתָ עִם־אֱלֹהִ֛ים וְעִם־אֲנָשִׁ֖ים וַתּוּכָֽל׃ אָמַר לוֹ: שִׁמְךָ לֹא יִקָּרֵא יוֹתֵר יַעֲקֹב, אֶלָּא יִשְׂרָאֵל, כִּי שָׂרִיתָ – כְּלוֹמַר הִתְכַּבַּדְתָּ בְּמַאֲבָק עִם מַלְאָךְ אֱלֹהִים וְעִם אֲנָשִׁים – וְנִצַּחְתָּ!
ל וַיִּשְׁאַ֣ל יַעֲקֹ֗ב וַיֹּ֙אמֶר֙ הַגִּֽידָה־נָּ֣א שְׁמֶ֔ךָ וַיֹּ֕אמֶר לָ֥מָּה זֶּ֖ה תִּשְׁאַ֣ל לִשְׁמִ֑י וַיְבָ֥רֶךְ אֹת֖וֹ שָֽׁם׃ יַעֲקֹב שָׁאַל וְאָמַר: הַגֵּד בְּבַקָּשָׁה אֶת שִׁמְךָ! אָמַר לוֹ: לָמָּה תִּשְׁאַל לִשְׁמִי? וְהוּא בֵּרֵךְ אוֹתוֹ שָׁם.

שלישי[עריכה]

לא וַיִּקְרָ֧א יַעֲקֹ֛ב שֵׁ֥ם הַמָּק֖וֹם פְּנִיאֵ֑ל כִּֽי־רָאִ֤יתִי אֱלֹהִים֙ פָּנִ֣ים אֶל־פָּנִ֔ים וַתִּנָּצֵ֖ל נַפְשִֽׁי׃ יַעֲקֹב קָרָא אֶת שֵׁם הַמָּקוֹם פְּנִיאֵל, כִּי אָמַר: רָאִיתִי מַלְאַךְ אֱלֹהִים פָּנִים אֶל פָּנִים וְנִצַּלְתִּי.
לב וַיִּֽזְרַֽח־ל֣וֹ הַשֶּׁ֔מֶשׁ כַּאֲשֶׁ֥ר עָבַ֖ר אֶת־פְּנוּאֵ֑ל וְה֥וּא צֹלֵ֖עַ עַל־יְרֵכֽוֹ׃ הַשֶּׁמֶשׁ זָרְחָה כְּשֶׁעָבַר אֶת פְּנוּאֵל (פְּנִיאֵל), וְהוּא צוֹלֵעַ עַל יְרֵכוֹ.
לג עַל־כֵּ֡ן לֹֽא־יֹאכְל֨וּ בְנֵֽי־יִשְׂרָאֵ֜ל אֶת־גִּ֣יד הַנָּשֶׁ֗ה אֲשֶׁר֙ עַל־כַּ֣ף הַיָּרֵ֔ךְ עַ֖ד הַיּ֣וֹם הַזֶּ֑ה כִּ֤י נָגַע֙ בְּכַף־יֶ֣רֶךְ יַעֲקֹ֔ב בְּגִ֖יד הַנָּשֶֽׁה׃ לָכֵן עַד הַיּוֹם הַזֶּה בְּנֵי יִשְׂרָאֵל לֹא אוֹכְלִים אֶת 'גִּיד הַנָּשֶׁה' – כְּלוֹמַר הַגִּיד שֶׁהֻזַּז מִמְּקוֹמוֹ שֶׁעַל כַּף הַיָּרֵךְ, כִּי הָאִישׁ הִכָּה בְּכַף הַיָּרֵךְ שֶׁל יַעֲקֹב בְּגִיד הַנָּשֶׁה.
פָּרָשַׁת הַפְּגִישָׁה עִם עֵשָׂו
לג א וַיִּשָּׂ֨א יַעֲקֹ֜ב עֵינָ֗יו וַיַּרְא֙ וְהִנֵּ֣ה עֵשָׂ֣ו בָּ֔א וְעִמּ֕וֹ אַרְבַּ֥ע מֵא֖וֹת אִ֑ישׁ וַיַּ֣חַץ אֶת־הַיְלָדִ֗ים עַל־לֵאָה֙ וְעַל־רָחֵ֔ל וְעַ֖ל שְׁתֵּ֥י הַשְּׁפָחֽוֹת׃ יַעֲקֹב הֵרִים אֶת מַבָּטוֹ, וְרָאָה שֶׁעֵשָׂו בָּא וְאִתּוֹ אַרְבַּע מֵאוֹת אִישׁ, הוּא חִלֵּק לִקְבוּצוֹת אֶת הַיְּלָדִים עִם לֵאָה רָחֵל וּשְׁתֵּי הַשְּׁפָחוֹת.
ב וַיָּ֧שֶׂם אֶת־הַשְּׁפָח֛וֹת וְאֶת־יַלְדֵיהֶ֖ן רִֽאשֹׁנָ֑ה וְאֶת־לֵאָ֤ה וִֽילָדֶ֙יהָ֙ אַחֲרֹנִ֔ים וְאֶת־רָחֵ֥ל וְאֶת־יוֹסֵ֖ף אַחֲרֹנִֽים׃ הוּא שָׂם אֶת הַשְּׁפָחוֹת וְאֶת יַלְדֵיהֶן רִאשׁוֹנִים, וְאֶת לֵאָה וִילָדֶיהָ אַחֲרֵי כֵן, וְאֶת רָחֵל וְאֶת יוֹסֵף אַחֲרֵי כֵן.
ג וְה֖וּא עָבַ֣ר לִפְנֵיהֶ֑ם וַיִּשְׁתַּ֤חוּ אַ֙רְצָה֙ שֶׁ֣בַע פְּעָמִ֔ים עַד־גִּשְׁתּ֖וֹ עַד־אָחִֽיו׃ וְהוּא עָבַר לַעֲמוֹד לִפְנֵיהֶם וְהִשְׁתַּחֲוָה עַל הָאָרֶץ שֶׁבַע פְּעָמִים עַד שֶׁנִּגַּשׁ לְאָחִיו.
ד וַיָּ֨רׇץ עֵשָׂ֤ו לִקְרָאתוֹ֙ וַֽיְחַבְּקֵ֔הוּ וַיִּפֹּ֥ל עַל־צַוָּארָ֖ו וַׄיִּׄשָּׁׄקֵ֑ׄהׄוּׄ וַיִּבְכּֽוּ׃ עֵשָׂו רָץ לִקְרָאתוֹ, חִבֵּק אוֹתוֹ וְנָפַל עַל צַוָּארוֹ וְנִשֵּׁק אוֹתוֹ, וְהֵם בָּכוּ.
ה וַיִּשָּׂ֣א אֶת־עֵינָ֗יו וַיַּ֤רְא אֶת־הַנָּשִׁים֙ וְאֶת־הַיְלָדִ֔ים וַיֹּ֖אמֶר מִי־אֵ֣לֶּה לָּ֑ךְ וַיֹּאמַ֕ר הַיְלָדִ֕ים אֲשֶׁר־חָנַ֥ן אֱלֹהִ֖ים אֶת־עַבְדֶּֽךָ׃ הוּא הֵרִים אֶת מַבָּטוֹ וְרָאָה אֶת הַנָּשִׁים וְאֶת הַיְּלָדִים וְאָמַר: מִי אֵלֶּה עֲבוּרְךָ? אָמַר לוֹ: אֵלֶּה הַיְּלָדִים שֶׁאֱלֹהִים נָתַן בְּחַסְדּוֹ לְעַבְדְּךָ.

רביעי[עריכה]

ו וַתִּגַּ֧שְׁןָ הַשְּׁפָח֛וֹת הֵ֥נָּה וְיַלְדֵיהֶ֖ן וַתִּֽשְׁתַּחֲוֶֽיןָ׃ הַשְּׁפָחוֹת וְיַלְדֵיהֶן נִגְּשׁוּ וְהִשְׁתַּחֲווּ.
ז וַתִּגַּ֧שׁ גַּם־לֵאָ֛ה וִילָדֶ֖יהָ וַיִּֽשְׁתַּחֲו֑וּ וְאַחַ֗ר נִגַּ֥שׁ יוֹסֵ֛ף וְרָחֵ֖ל וַיִּֽשְׁתַּחֲוֽוּ׃ גַּם לֵאָה וִילָדֶיהָ נִגְּשׁוּ וְהִשְׁתַּחֲווּ, וְאַחַר כָּךְ נִגְּשׁוּ יוֹסֵף וְרָחֵל וְהִשְׁתַּחֲווּ.
ח וַיֹּ֕אמֶר מִ֥י לְךָ֛ כׇּל־הַמַּחֲנֶ֥ה הַזֶּ֖ה אֲשֶׁ֣ר פָּגָ֑שְׁתִּי וַיֹּ֕אמֶר לִמְצֹא־חֵ֖ן בְּעֵינֵ֥י אֲדֹנִֽי׃ אָמַר לוֹ: מַה כָּל הַמַּחֲנֶה הַזֶּה שֶׁלְּךָ שֶׁפָּגַשְׁתִּי בַּדֶּרֶךְ? אָמַר לוֹ: לִמְצֹא חֵן בְּעֵינֵי אֲדוֹנִי!
ט וַיֹּ֥אמֶר עֵשָׂ֖ו יֶשׁ־לִ֣י רָ֑ב אָחִ֕י יְהִ֥י לְךָ֖ אֲשֶׁר־לָֽךְ׃ אָמַר עֵשָׂו: יֵשׁ לִי הַרְבֵּה, אָחִי – שֶׁיִּהְיֶה שֶׁלְּךָ הָרְכוּשׁ שֶׁיֵּשׁ לְךָ!
י וַיֹּ֣אמֶר יַעֲקֹ֗ב אַל־נָא֙ אִם־נָ֨א מָצָ֤אתִי חֵן֙ בְּעֵינֶ֔יךָ וְלָקַחְתָּ֥ מִנְחָתִ֖י מִיָּדִ֑י כִּ֣י עַל־כֵּ֞ן רָאִ֣יתִי פָנֶ֗יךָ כִּרְאֹ֛ת פְּנֵ֥י אֱלֹהִ֖ים וַתִּרְצֵֽנִי׃ אָמַר יַעֲקֹב: בְּבַקָּשָׁה אַל תֹּאמַר כָּךְ! אִם מָצָאתִי חֵן בְּעֵינֶיךָ בְּבַקָּשָׁה קַח אֶת מַתְּנָתִי מִיָּדִי, מִפְּנֵי שֶׁהִקְבַּלְתִּי אֶת פָּנֶיךָ כְּמוֹ שֶׁמַּקְבִּילִים פְּנֵי אֱלֹהִים, וּתְקַבֵּל אוֹתִי בְּרָצוֹן.
יא קַח־נָ֤א אֶת־בִּרְכָתִי֙ אֲשֶׁ֣ר הֻבָ֣את לָ֔ךְ כִּֽי־חַנַּ֥נִי אֱלֹהִ֖ים וְכִ֣י יֶשׁ־לִי־כֹ֑ל וַיִּפְצַר־בּ֖וֹ וַיִּקָּֽח׃ קַח בְּבַקָּשָׁה אֶת מַתְּנָתִי שֶׁהוּבְאָה לְךָ כִּי אֱלֹהִים הֶעֱנִיק לִי בְּחַסְדּוֹ וְיֵשׁ לִי מַסְפִּיק לְכָל צְרָכַי! הוּא הִפְצִיר בּוֹ, וְהוּא לָקַח.
יב וַיֹּ֖אמֶר נִסְעָ֣ה וְנֵלֵ֑כָה וְאֵלְכָ֖ה לְנֶגְדֶּֽךָ׃ אָמַר עֵשָׂו: נִסַּע וְנֵלֵךְ, וְאֵלֶךְ לְיָדְךָ.
יג וַיֹּ֣אמֶר אֵלָ֗יו אֲדֹנִ֤י יֹדֵ֙עַ֙ כִּֽי־הַיְלָדִ֣ים רַכִּ֔ים וְהַצֹּ֥אן וְהַבָּקָ֖ר עָל֣וֹת עָלָ֑י וּדְפָקוּם֙ י֣וֹם אֶחָ֔ד וָמֵ֖תוּ כׇּל־הַצֹּֽאן׃ אָמַר לוֹ: אֲדוֹנִי יוֹדֵעַ שֶׁהַיְּלָדִים חֲלוּשִׁים וְהַצֹּאן וְהַבָּקָר יוֹנְקִים וְעָלַי לְהוֹבִילָם, וְאִם יְזָרְזוּ אוֹתָם בְּמֶשֶׁךְ יוֹם אֶחָד כָּל הַצֹּאן יָמוּת.
יד יַעֲבׇר־נָ֥א אֲדֹנִ֖י לִפְנֵ֣י עַבְדּ֑וֹ וַאֲנִ֞י אֶֽתְנָהֲלָ֣ה לְאִטִּ֗י לְרֶ֨גֶל הַמְּלָאכָ֤ה אֲשֶׁר־לְפָנַי֙ וּלְרֶ֣גֶל הַיְלָדִ֔ים עַ֛ד אֲשֶׁר־אָבֹ֥א אֶל־אֲדֹנִ֖י שֵׂעִֽירָה׃ בְּבַקָּשָׁה, אֲדוֹנִי יַעֲבוֹר לִפְנֵי עַבְדּוֹ וַאֲנִי אוֹבִיל אֶת עַצְמִי לְאַט כְּפִי צוֹרֶךְ הַמְּלָאכָה הַמֻּטֶּלֶת עָלַי וּכְפִי צוֹרֶךְ הַיְּלָדִים, עַד שֶׁאָבוֹא אֶל אֲדוֹנִי לְשֵׂעִיר.
טו וַיֹּ֣אמֶר עֵשָׂ֔ו אַצִּֽיגָה־נָּ֣א עִמְּךָ֔ מִן־הָעָ֖ם אֲשֶׁ֣ר אִתִּ֑י וַיֹּ֙אמֶר֙ לָ֣מָּה זֶּ֔ה אֶמְצָא־חֵ֖ן בְּעֵינֵ֥י אֲדֹנִֽי׃ אָמַר עֵשָׂו: אַשְׁאִיר אִתְּךָ מֵהָאֲנָשִׁים שֶׁאִתִּי, אָמַר לוֹ: לְשֵׁם מָה זֶה נָחוּץ לִי? דַּי לִי שֶׁאֶמְצָא חֵן בְּעֵינֵי אֲדוֹנִי!
טז וַיָּ֩שׇׁב֩ בַּיּ֨וֹם הַה֥וּא עֵשָׂ֛ו לְדַרְכּ֖וֹ שֵׂעִֽירָה׃ עֵשָׂו חָזַר בַּיּוֹם הַהוּא לְדַרְכּוֹ לְשֵׂעִיר.
יז וְיַעֲקֹב֙ נָסַ֣ע סֻכֹּ֔תָה וַיִּ֥בֶן ל֖וֹ בָּ֑יִת וּלְמִקְנֵ֙הוּ֙ עָשָׂ֣ה סֻכֹּ֔ת עַל־כֵּ֛ן קָרָ֥א שֵׁם־הַמָּק֖וֹם סֻכּֽוֹת׃ וְיַעֲקֹב נָסַע לְ'סֻכֹּת', הוּא בָּנָה לוֹ בַּיִת וּלְבֶהֱמוֹתָיו עָשָׂה סֻכּוֹת, לָכֵן הוּא קָרָא אֶת שֵׁם הַמָּקוֹם הַהוּא – סֻכּוֹת.
ס
יח וַיָּבֹא֩ יַעֲקֹ֨ב שָׁלֵ֜ם עִ֣יר שְׁכֶ֗ם אֲשֶׁר֙ בְּאֶ֣רֶץ כְּנַ֔עַן בְּבֹא֖וֹ מִפַּדַּ֣ן אֲרָ֑ם וַיִּ֖חַן אֶת־פְּנֵ֥י הָעִֽיר׃ יַעֲקֹב בָּא בְּשָׁלוֹם לָעִיר שֶׁל שְׁכֶם שֶׁבְּאֶרֶץ כְּנַעַן, בְּבוֹאוֹ מִפַּדַּן אֲרָם, וְהוּא חָנָה לִפְנֵי הָעִיר.
יט וַיִּ֜קֶן אֶת־חֶלְקַ֣ת הַשָּׂדֶ֗ה אֲשֶׁ֤ר נָֽטָה־שָׁם֙ אׇהֳל֔וֹ מִיַּ֥ד בְּנֵֽי־חֲמ֖וֹר אֲבִ֣י שְׁכֶ֑ם בְּמֵאָ֖ה קְשִׂיטָֽה׃ הוּא קָנָה מִבְּנֵי חֲמוֹר אֲבִי שְׁכֶם אֶת חֶלְקַת הַשָּׂדֶה שֶׁהוּא הֵקִים שָׁם אֶת אָהֳלוֹ תְּמוּרַת מֵאָה כְּבָשׂוֹת צְעִירוֹת.
כ וַיַּצֶּב־שָׁ֖ם מִזְבֵּ֑חַ וַיִּ֨קְרָא־ל֔וֹ אֵ֖ל אֱלֹהֵ֥י יִשְׂרָאֵֽל׃ הוּא הֵקִים שָׁם מִזְבֵּחַ, וְעָבַד עָלָיו לְאֵל אֱלֹהֵי יִשְׂרָאֵל.

חמישי[עריכה]

ס פָּרָשַׁת דִּינָה
לד א וַתֵּצֵ֤א דִינָה֙ בַּת־לֵאָ֔ה אֲשֶׁ֥ר יָלְדָ֖ה לְיַעֲקֹ֑ב לִרְא֖וֹת בִּבְנ֥וֹת הָאָֽרֶץ׃ דִּינָה, הַבַּת שֶׁלֵּאָה יָלְדָה לְיַעֲקֹב, יָצְאָה לִפְגּוֹשׁ אֶת בְּנוֹת הָאָרֶץ.
ב וַיַּ֨רְא אֹתָ֜הּ שְׁכֶ֧ם בֶּן־חֲמ֛וֹר הַֽחִוִּ֖י נְשִׂ֣יא הָאָ֑רֶץ וַיִּקַּ֥ח אֹתָ֛הּ וַיִּשְׁכַּ֥ב אֹתָ֖הּ וַיְעַנֶּֽהָ׃ שְׁכֶם בֶּן חֲמוֹר הַחִוִּי נְשִׂיא הָאָרֶץ רָאָה אוֹתָהּ, חָטַף אוֹתָהּ, שָׁכַב אִתָּהּ וְעִנָּה אוֹתָהּ.
ג וַתִּדְבַּ֣ק נַפְשׁ֔וֹ בְּדִינָ֖ה בַּֽת־יַעֲקֹ֑ב וַיֶּֽאֱהַב֙ אֶת־הַֽנַּעֲרָ֔ וַיְדַבֵּ֖ר עַל־לֵ֥ב הַֽנַּעֲרָֽ׃ נַפְשׁוֹ דָּבְקָה בְּדִינָה בַּת יַעֲקֹב, וְהוּא הִתְאָהֵב בַּנַּעֲרָה, וְדִבֵּר אִתָּהּ לְרַצּוֹת אֶת לִבָּהּ.
ד וַיֹּ֣אמֶר שְׁכֶ֔ם אֶל־חֲמ֥וֹר אָבִ֖יו לֵאמֹ֑ר קַֽח־לִ֛י אֶת־הַיַּלְדָּ֥ה הַזֹּ֖את לְאִשָּֽׁה׃ אָמַר שְׁכֶם לַחֲמוֹר אָבִיו: קַח לִי אֶת הַיַּלְדָּה הַזֹּאת לְאִשָּׁה!
ה וְיַעֲקֹ֣ב שָׁמַ֗ע כִּ֤י טִמֵּא֙ אֶת־דִּינָ֣ה בִתּ֔וֹ וּבָנָ֛יו הָי֥וּ אֶת־מִקְנֵ֖הוּ בַּשָּׂדֶ֑ה וְהֶחֱרִ֥שׁ יַעֲקֹ֖ב עַד־בֹּאָֽם׃ וְיַעֲקֹב שָׁמַע שֶׁשְּׁכֶם טִמֵּא אֶת דִּינָה בִּתּוֹ וּבָנָיו הָיוּ עִם בֶּהֱמוֹתָיו בַּשָּׂדֶה, יַעֲקֹב שָׁתַק עַד שֶׁבָּאוּ.
ו וַיֵּצֵ֛א חֲמ֥וֹר אֲבִֽי־שְׁכֶ֖ם אֶֽל־יַעֲקֹ֑ב לְדַבֵּ֖ר אִתּֽוֹ׃ יָצָא חֲמוֹר אֲבִי שְׁכֶם אֶל יַעֲקֹב לְדַבֵּר אִתּוֹ.
ז וּבְנֵ֨י יַעֲקֹ֜ב בָּ֤אוּ מִן־הַשָּׂדֶה֙ כְּשׇׁמְעָ֔ם וַיִּֽתְעַצְּבוּ֙ הָֽאֲנָשִׁ֔ים וַיִּ֥חַר לָהֶ֖ם מְאֹ֑ד כִּֽי־נְבָלָ֞ה עָשָׂ֣ה בְיִשְׂרָאֵ֗ל לִשְׁכַּב֙ אֶת־בַּֽת־יַעֲקֹ֔ב וְכֵ֖ן לֹ֥א יֵעָשֶֽׂה׃ כְּשֶׁבְּנֵי יַעֲקֹב שָׁמְעוּ זֹאת הֵם בָּאוּ מֵהַשָּׂדֶה, הֵם הִצְטַעֲרוּ וְכָאַב לָהֶם מְאוֹד, שֶׁשְּׁכֶם עָשָׂה מַעֲשֶׂה חֶרְפָּה בְּיִשְׂרָאֵל – לִשְׁכַּב עִם בַּת יַעֲקֹב, וְזֶהוּ מַעֲשֶׂה שֶׁלֹּא יֵעָשֶׂה.
ח וַיְדַבֵּ֥ר חֲמ֖וֹר אִתָּ֣ם לֵאמֹ֑ר שְׁכֶ֣ם בְּנִ֗י חָֽשְׁקָ֤ה נַפְשׁוֹ֙ בְּבִתְּכֶ֔ם תְּנ֨וּ נָ֥א אֹתָ֛הּ ל֖וֹ לְאִשָּֽׁה׃ חֲמוֹר דִּבֵּר אִתָּם וְאָמַר: שְׁכֶם בְּנִי חָשְׁקָה נַפְשׁוֹ בְּבִתְּכֶם תְּנוּ לוֹ אוֹתָהּ בְּבַקָּשָׁה לְאִשָּׁה.
ט וְהִֽתְחַתְּנ֖וּ אֹתָ֑נוּ בְּנֹֽתֵיכֶם֙ תִּתְּנוּ־לָ֔נוּ וְאֶת־בְּנֹתֵ֖ינוּ תִּקְח֥וּ לָכֶֽם׃ וְתִתְחַתְּנוּ אִתָּנוּ – אֶת בְּנוֹתֵיכֶם תִּתְּנוּ לָנוּ וְאֶת בְּנוֹתֵינוּ תִּקְחוּ לָכֶם.
י וְאִתָּ֖נוּ תֵּשֵׁ֑בוּ וְהָאָ֙רֶץ֙ תִּהְיֶ֣ה לִפְנֵיכֶ֔ם שְׁבוּ֙ וּסְחָר֔וּהָ וְהֵֽאָחֲז֖וּ בָּֽהּ׃ וְתִתְיַשְּׁבוּ אִתָּנוּ וְהָאָרֶץ תִּהְיֶה לִפְנֵיכֶם, תִּתְיַשְּׁבוּ וְתִסְחֲרוּ בָּהּ וְתֵאָחֲזוּ בָּהּ!
יא וַיֹּ֤אמֶר שְׁכֶם֙ אֶל־אָבִ֣יהָ וְאֶל־אַחֶ֔יהָ אֶמְצָא־חֵ֖ן בְּעֵינֵיכֶ֑ם וַאֲשֶׁ֥ר תֹּאמְר֛וּ אֵלַ֖י אֶתֵּֽן׃ שְׁכֶם אָמַר לְאָבִיהָ וּלְאַחֶיהָ: אֶמְצָא חֵן בְּעֵינֵיכֶם, וְאֶתֵּן מַה שֶּׁתֹּאמְרוּ לִי.
יב הַרְבּ֨וּ עָלַ֤י מְאֹד֙ מֹ֣הַר וּמַתָּ֔ן וְאֶ֨תְּנָ֔ה כַּאֲשֶׁ֥ר תֹּאמְר֖וּ אֵלָ֑י וּתְנוּ־לִ֥י אֶת־הַֽנַּעֲרָ֖ לְאִשָּֽׁה׃ בַּקְּשׁוּ מִמֶּנִּי תַּשְׁלוּם גָּבוֹהַּ מְאוֹד עַל הַכַּלָּה וּמַתָּנוֹת – וְאֶתֵּן כָּל מַה שֶּׁתֹּאמְרוּ לִי, וּתְנוּ לִי אֶת הַנַּעֲרָה לְאִשָּׁה.
יג וַיַּעֲנ֨וּ בְנֵֽי־יַעֲקֹ֜ב אֶת־שְׁכֶ֨ם וְאֶת־חֲמ֥וֹר אָבִ֛יו בְּמִרְמָ֖ה וַיְדַבֵּ֑רוּ אֲשֶׁ֣ר טִמֵּ֔א אֵ֖ת דִּינָ֥ה אֲחֹתָֽם׃ בְּנֵי יַעֲקֹב עָנוּ לִשְׁכֶם וּלְחֲמוֹר אָבִיו בְּעַרְמוּמִיּוּת וְדִבְּרוּ אִתָּם, כִּי הוּא טִמֵּא אֶת דִּינָה אֲחוֹתָם.
יד וַיֹּאמְר֣וּ אֲלֵיהֶ֗ם לֹ֤א נוּכַל֙ לַעֲשׂוֹת֙ הַדָּבָ֣ר הַזֶּ֔ה לָתֵת֙ אֶת־אֲחֹתֵ֔נוּ לְאִ֖ישׁ אֲשֶׁר־ל֣וֹ עׇרְלָ֑ה כִּֽי־חֶרְפָּ֥ה הִ֖וא לָֽנוּ׃ וְאָמְרוּ לָהֶם: לֹא נוּכַל לַעֲשׂוֹת אֶת הַדָּבָר הַזֶּה – לָתֵת אֶת אֲחוֹתֵנוּ לְאִישׁ שֶׁיֵּשׁ לוֹ עָרְלָה, כִּי זֶה בִּזָּיוֹן עֲבוּרֵנוּ.
טו אַךְ־בְּזֹ֖את נֵא֣וֹת לָכֶ֑ם אִ֚ם תִּהְי֣וּ כָמֹ֔נוּ לְהִמֹּ֥ל לָכֶ֖ם כׇּל־זָכָֽר׃ אֲבָל נַסְכִּים לָכֶם בִּתְנַאי זֶה – אִם תִּהְיוּ כָּמוֹנוּ שֶׁתָּמוּלוּ כָּל זָכָר מִכֶּם.
טז וְנָתַ֤נּוּ אֶת־בְּנֹתֵ֙ינוּ֙ לָכֶ֔ם וְאֶת־בְּנֹתֵיכֶ֖ם נִֽקַּֽח־לָ֑נוּ וְיָשַׁ֣בְנוּ אִתְּכֶ֔ם וְהָיִ֖ינוּ לְעַ֥ם אֶחָֽד׃ נִתֵּן לָכֶם אֶת בְּנוֹתֵינוּ וְאֶת בְּנוֹתֵיכֶם נִקַּח לָנוּ וְנֵשֵׁב אִתְּכֶם וְנִהְיֶה לְעַם אֶחָד.
יז וְאִם־לֹ֧א תִשְׁמְע֛וּ אֵלֵ֖ינוּ לְהִמּ֑וֹל וְלָקַ֥חְנוּ אֶת־בִּתֵּ֖נוּ וְהָלָֽכְנוּ׃ וְאִם לֹא תִּשְׁמְעוּ אֵלֵינוּ לָמוּל, נִקַּח אֶת בִּתֵּנוּ וְנֵלֵךְ.
יח וַיִּֽיטְב֥וּ דִבְרֵיהֶ֖ם בְּעֵינֵ֣י חֲמ֑וֹר וּבְעֵינֵ֖י שְׁכֶ֥ם בֶּן־חֲמֽוֹר׃ דִּבְרֵיהֶם מָצְאוּ חֵן בְּעֵינֵי חֲמוֹר וּבְעֵינֵי שְׁכֶם בֶּן חֲמוֹר.
יט וְלֹֽא־אֵחַ֤ר הַנַּ֙עַר֙ לַעֲשׂ֣וֹת הַדָּבָ֔ר כִּ֥י חָפֵ֖ץ בְּבַֽת־יַעֲקֹ֑ב וְה֣וּא נִכְבָּ֔ד מִכֹּ֖ל בֵּ֥ית אָבִֽיו׃ הַנַּעַר לֹא הִתְעַכֵּב מִלַּעֲשׂוֹת אֶת הַדָּבָר כִּי הוּא חָפַץ בַּבַּת שֶׁל יַעֲקֹב, וְהוּא הָיָה הַמְּכֻבָּד בְּיוֹתֵר מִכָּל מִשְׁפַּחְתּוֹ.
כ וַיָּבֹ֥א חֲמ֛וֹר וּשְׁכֶ֥ם בְּנ֖וֹ אֶל־שַׁ֣עַר עִירָ֑ם וַֽיְדַבְּר֛וּ אֶל־אַנְשֵׁ֥י עִירָ֖ם לֵאמֹֽר׃ חֲמוֹר וּשְׁכֶם בְּנוֹ בָּאוּ לַשַּׁעַר שֶׁל עִירָם וְדִבְּרוּ אֶל אַנְשֵׁי עִירָם וְאָמְרוּ:
כא הָאֲנָשִׁ֨ים הָאֵ֜לֶּה שְֽׁלֵמִ֧ים הֵ֣ם אִתָּ֗נוּ וְיֵשְׁב֤וּ בָאָ֙רֶץ֙ וְיִסְחֲר֣וּ אֹתָ֔הּ וְהָאָ֛רֶץ הִנֵּ֥ה רַֽחֲבַת־יָדַ֖יִם לִפְנֵיהֶ֑ם אֶת־בְּנֹתָם֙ נִקַּֽח־לָ֣נוּ לְנָשִׁ֔ים וְאֶת־בְּנֹתֵ֖ינוּ נִתֵּ֥ן לָהֶֽם׃ הָאֲנָשִׁים הָאֵלֶּה בְּשָׁלוֹם אִתָּנוּ וְהֵם יֵשְׁבוּ בָּאָרֶץ וְיִסְחֲרוּ בָּהּ, וְהָאָרֶץ הִנֵּה הִיא רַחֲבַת יָדַיִם לִפְנֵיהֶם, אֶת בְּנוֹתָם נִקַּח לָנוּ לְנָשִׁים וְאֶת בְּנוֹתֵינוּ נִתֵּן לָהֶם.
כב אַךְ־בְּ֠זֹ֠את יֵאֹ֨תוּ לָ֤נוּ הָאֲנָשִׁים֙ לָשֶׁ֣בֶת אִתָּ֔נוּ לִהְי֖וֹת לְעַ֣ם אֶחָ֑ד בְּהִמּ֥וֹל לָ֙נוּ֙ כׇּל־זָכָ֔ר כַּאֲשֶׁ֖ר הֵ֥ם נִמֹּלִֽים׃ אֲבָל הָאֲנָשִׁים יַסְכִּימוּ לָנוּ לָשֶׁבֶת אִתָּנוּ לִהְיוֹת לְעַם אֶחָד בִּתְנַאי שֶׁנָּמוּל לָנוּ כָּל זָכָר כְּמוֹ שֶׁהֵם נִמּוֹלִים.
כג מִקְנֵהֶ֤ם וְקִנְיָנָם֙ וְכׇל־בְּהֶמְתָּ֔ם הֲל֥וֹא לָ֖נוּ הֵ֑ם אַ֚ךְ נֵא֣וֹתָה לָהֶ֔ם וְיֵשְׁב֖וּ אִתָּֽנוּ׃ עֶדְרֵיהֶם וּרְכוּשָׁם וְכָל בֶּהֱמוֹתֵיהֶם הֲרֵי הֵם שֶׁלָּנוּ, רַק נִתְרַצֶּה לָהֶם וְיֵשְׁבוּ אִתָּנוּ.
כד וַיִּשְׁמְע֤וּ אֶל־חֲמוֹר֙ וְאֶל־שְׁכֶ֣ם בְּנ֔וֹ כׇּל־יֹצְאֵ֖י שַׁ֣עַר עִיר֑וֹ וַיִּמֹּ֙לוּ֙ כׇּל־זָכָ֔ר כׇּל־יֹצְאֵ֖י שַׁ֥עַר עִירֽוֹ׃ כָּל תּוֹשָׁבֵי הָעִיר שָׁמְעוּ בְּקוֹל חֲמוֹר וּבְקוֹל שְׁכֶם בְּנוֹ, וְהֵם מָלוּ כָּל זָכָר.
כה וַיְהִי֩ בַיּ֨וֹם הַשְּׁלִישִׁ֜י בִּֽהְיוֹתָ֣ם כֹּֽאֲבִ֗ים וַיִּקְח֣וּ שְׁנֵֽי־בְנֵי־יַ֠עֲקֹ֠ב שִׁמְע֨וֹן וְלֵוִ֜י אֲחֵ֤י דִינָה֙ אִ֣ישׁ חַרְבּ֔וֹ וַיָּבֹ֥אוּ עַל־הָעִ֖יר בֶּ֑טַח וַיַּֽהַרְג֖וּ כׇּל־זָכָֽר׃ בַּיּוֹם הַשְּׁלִישִׁי כְּשֶׁכָּאַב לָהֵם, שְׁנֵי בְּנֵי יַעֲקֹב שִׁמְעוֹן וְלֵוִי אֲחֵי דִּינָה לָקְחוּ כָּל אֶחָד אֶת חַרְבּוֹ, וּבָאוּ לָעִיר שֶׁיּוֹשֶׁבֶת בְּבִטְחָה, וְהָרְגוּ כָּל זָכָר.
כו וְאֶת־חֲמוֹר֙ וְאֶת־שְׁכֶ֣ם בְּנ֔וֹ הָרְג֖וּ לְפִי־חָ֑רֶב וַיִּקְח֧וּ אֶת־דִּינָ֛ה מִבֵּ֥ית שְׁכֶ֖ם וַיֵּצֵֽאוּ׃ וְאֶת חֲמוֹר וְאֶת שְׁכֶם בְּנוֹ הָרְגוּ בְּלַהַב הַחֶרֶב, וְלָקְחוּ אֶת דִּינָה מִבֵּית שְׁכֶם וְיָצְאוּ.
כז בְּנֵ֣י יַעֲקֹ֗ב בָּ֚אוּ עַל־הַ֣חֲלָלִ֔ים וַיָּבֹ֖זּוּ הָעִ֑יר אֲשֶׁ֥ר טִמְּא֖וּ אֲחוֹתָֽם׃ בְּנֵי יַעֲקֹב בָּאוּ לְהַפְשִׁיט תַּכְשִׁיטִים מֵהַהֲרוּגִים וּבָזְזוּ אֶת הָעִיר שֶׁתּוֹשָׁבֶיהָ טִמְּאוּ אֶת אֲחוֹתָם.
כח אֶת־צֹאנָ֥ם וְאֶת־בְּקָרָ֖ם וְאֶת־חֲמֹרֵיהֶ֑ם וְאֵ֧ת אֲשֶׁר־בָּעִ֛יר וְאֶת־אֲשֶׁ֥ר בַּשָּׂדֶ֖ה לָקָֽחוּ׃ הֵם לָקְחוּ אֶת צֹאנָם וְאֶת בְּקָרָם וְאֶת חֲמוֹרֵיהֶם – אֶת מַה שֶּׁבָּעִיר וְאֶת מַה שֶּׁבַּשָּׂדֶה.
כט וְאֶת־כׇּל־חֵילָ֤ם וְאֶת־כׇּל־טַפָּם֙ וְאֶת־נְשֵׁיהֶ֔ם שָׁב֖וּ וַיָּבֹ֑זּוּ וְאֵ֖ת כׇּל־אֲשֶׁ֥ר בַּבָּֽיִת׃ וְאֶת כָּל נִכְסֵיהֶם וְאֶת כָּל יַלְדֵיהֶם וְאֶת נְשֵׁיהֶם הֵם לָקְחוּ בַּשֶּׁבִי וּבָזְזוּ, וְאֶת כָּל מַה שֶּׁבַּבַּיִת.
ל וַיֹּ֨אמֶר יַעֲקֹ֜ב אֶל־שִׁמְע֣וֹן וְאֶל־לֵוִי֮ עֲכַרְתֶּ֣ם אֹתִי֒ לְהַבְאִישֵׁ֙נִי֙ בְּיֹשֵׁ֣ב הָאָ֔רֶץ בַּֽכְּנַעֲנִ֖י וּבַפְּרִזִּ֑י וַאֲנִי֙ מְתֵ֣י מִסְפָּ֔ר וְנֶאֶסְפ֤וּ עָלַי֙ וְהִכּ֔וּנִי וְנִשְׁמַדְתִּ֖י אֲנִ֥י וּבֵיתִֽי׃ יַעֲקֹב אָמַר לְשִׁמְעוֹן וּלְלֵוִי: הִכְתַּמְתֶּם אוֹתִי וְהוֹצֵאתֶם עָלַי שֵׁם רַע בְּעֵינֵי יוֹשְׁבֵי הָאָרֶץ – הַכְּנַעֲנִי וְהַפְּרִזִּי, וַאֲנִי רַק עִם אֲנָשִׁים סְפוּרִים, וְהֵם יֵאָסְפוּ עָלַי וְיַכּוּ אוֹתִי וְנֻשְׁמַד אֲנִי וּמִשְׁפַּחְתִּי!
לא וַיֹּאמְר֑וּ הַכְזוֹנָ֕ה יַעֲשֶׂ֖ה אֶת־אֲחוֹתֵֽנוּ׃ אָמְרוּ לוֹ: הַאִם הוּא יַעֲשֶׂה אֶת אֲחוֹתֵנוּ כְּמוֹ זוֹנָה?!
פ פָּרָשַׁת קִיּוּם הַנֶּדֶר
לה א וַיֹּ֤אמֶר אֱלֹהִים֙ אֶֽל־יַעֲקֹ֔ב ק֛וּם עֲלֵ֥ה בֵֽית־אֵ֖ל וְשֶׁב־שָׁ֑ם וַעֲשֵׂה־שָׁ֣ם מִזְבֵּ֔חַ לָאֵל֙ הַנִּרְאֶ֣ה אֵלֶ֔יךָ בְּבׇ֨רְחֲךָ֔ מִפְּנֵ֖י עֵשָׂ֥ו אָחִֽיךָ׃ אֱלֹהִים אָמַר לְיַעֲקֹב קוּם עֲלֵה לְבֵית אֵל וְתִתְיַשֵּׁב שָׁם, וַעֲשֵׂה שָׁם מִזְבֵּחַ לָאֵל שֶׁהִתְגַּלָּה אֵלֶיךָ כְּשֶׁבָּרַחְתָּ מֵעֵשָׂו אָחִיךָ, כְּמוֹ שֶׁנָּדַרְתָּ.
ב וַיֹּ֤אמֶר יַעֲקֹב֙ אֶל־בֵּית֔וֹ וְאֶ֖ל כׇּל־אֲשֶׁ֣ר עִמּ֑וֹ הָסִ֜רוּ אֶת־אֱלֹהֵ֤י הַנֵּכָר֙ אֲשֶׁ֣ר בְּתֹכְכֶ֔ם וְהִֽטַּהֲר֔וּ וְהַחֲלִ֖יפוּ שִׂמְלֹתֵיכֶֽם׃ יַעֲקֹב אָמַר לְאַנְשֵׁי בֵּיתוֹ וּלְכָל מִי שֶׁאִתּוֹ: הָסִירוּ מֵאִתְּכֶם אֶת הַפְּסָלִים שֶׁאִתְּכֶם, וְהִטַּהֲרוּ, וְהַחֲלִיפוּ אֶת בִּגְדֵיכֶם.
ג וְנָק֥וּמָה וְנַעֲלֶ֖ה בֵּֽית־אֵ֑ל וְאֶֽעֱשֶׂה־שָּׁ֣ם מִזְבֵּ֗חַ לָאֵ֞ל הָעֹנֶ֤ה אֹתִי֙ בְּי֣וֹם צָֽרָתִ֔י וַֽיְהִי֙ עִמָּדִ֔י בַּדֶּ֖רֶךְ אֲשֶׁ֥ר הָלָֽכְתִּי׃ וְנָקוּם וְנַעֲלֶה לְבֵית אֵל, וְאֶעֱשֶׂה שָׁם מִזְבֵּחַ לָאֵל שֶׁעָנָה לִתְפִלָּתִי בְּיוֹם צָרָתִי וְהָיָה בְּעֶזְרָתִי בַּדֶּרֶךְ שֶׁהָלַכְתִּי.
ד וַיִּתְּנ֣וּ אֶֽל־יַעֲקֹ֗ב אֵ֣ת כׇּל־אֱלֹהֵ֤י הַנֵּכָר֙ אֲשֶׁ֣ר בְּיָדָ֔ם וְאֶת־הַנְּזָמִ֖ים אֲשֶׁ֣ר בְּאׇזְנֵיהֶ֑ם וַיִּטְמֹ֤ן אֹתָם֙ יַעֲקֹ֔ב תַּ֥חַת הָאֵלָ֖ה אֲשֶׁ֥ר עִם־שְׁכֶֽם׃ הֵם נָתְנוּ לְיַעֲקֹב אֶת כָּל הַפְּסָלִים שֶׁבִּידֵיהֶם וְאֶת הַנְּזָמִים שֶׁבְּאָזְנֵיהֶם, וְיַעֲקֹב טָמַן אוֹתָם מִתַּחַת לְעֵץ הָאַלּוֹן שֶׁלְּיַד שְׁכֶם.
ה וַיִּסָּ֑עוּ וַיְהִ֣י ׀ חִתַּ֣ת אֱלֹהִ֗ים עַל־הֶֽעָרִים֙ אֲשֶׁר֙ סְבִיב֣וֹתֵיהֶ֔ם וְלֹ֣א רָֽדְפ֔וּ אַחֲרֵ֖י בְּנֵ֥י יַעֲקֹֽב׃ הֵם נָסְעוּ מִשָּׁם וְהָיָה פַּחַד אֱלֹהִים עַל הֶעָרִים שֶׁסְּבִיבוֹתֵיהֶם וְהֵם לֹא רָדְפוּ אַחֲרֵי בְּנֵי יַעֲקֹב.
ו וַיָּבֹ֨א יַעֲקֹ֜ב ל֗וּזָה אֲשֶׁר֙ בְּאֶ֣רֶץ כְּנַ֔עַן הִ֖וא בֵּֽית־אֵ֑ל ה֖וּא וְכׇל־הָעָ֥ם אֲשֶׁר־עִמּֽוֹ׃ יַעֲקֹב בָּא לְלוּז שֶׁבְּאֶרֶץ כְּנַעַן הִיא בֵּית אֵל, הוּא וְכָל הָעָם שֶׁאִתּוֹ.
ז וַיִּ֤בֶן שָׁם֙ מִזְבֵּ֔חַ וַיִּקְרָא֙ לַמָּק֔וֹם אֵ֖ל בֵּֽית־אֵ֑ל כִּ֣י שָׁ֗ם נִגְל֤וּ אֵלָיו֙ הָֽאֱלֹהִ֔ים בְּבׇרְח֖וֹ מִפְּנֵ֥י אָחִֽיו׃ הוּא בָּנָה שָׁם מִזְבֵּחַ וְקָרָא לַמָּקוֹם "הָאֵל שֶׁבְּבֵית-אֵל" כִּי שָׁם נִגְלוּ אֵלָיו מַלְאֲכֵי הָאֱלֹהִים כְּשֶׁבָּרַח מֵאָחִיו.
ח וַתָּ֤מׇת דְּבֹרָה֙ מֵינֶ֣קֶת רִבְקָ֔ה וַתִּקָּבֵ֛ר מִתַּ֥חַת לְבֵֽית־אֵ֖ל תַּ֣חַת הָֽאַלּ֑וֹן וַיִּקְרָ֥א שְׁמ֖וֹ אַלּ֥וֹן בָּכֽוּת׃ דְּבוֹרָה שֶׁהָיְתָה הַמֵּנִיקָה שֶׁל רִבְקָה מֵתָה וְהִיא נִקְבְּרָה לְמַרְגְּלוֹת הָהָר שֶׁל בֵּית אֵל מִתַּחַת לָאַלּוֹן, הוּא קָרָא אֶת שְׁמוֹ אַלּוֹן הַבְּכִיָּה.
פ פָּרָשַׁת שֵׁם יִשְׂרָאֵל
ט וַיֵּרָ֨א אֱלֹהִ֤ים אֶֽל־יַעֲקֹב֙ ע֔וֹד בְּבֹא֖וֹ מִפַּדַּ֣ן אֲרָ֑ם וַיְבָ֖רֶךְ אֹתֽוֹ׃ אֱלֹהִים הִתְגַּלָּה אֶל יַעֲקֹב שׁוּב כְּשֶׁבָּא מִפַּדַּן אֲרָם וּבֵרֵךְ אוֹתוֹ.
י וַיֹּֽאמֶר־ל֥וֹ אֱלֹהִ֖ים שִׁמְךָ֣ יַעֲקֹ֑ב לֹֽא־יִקָּרֵא֩ שִׁמְךָ֨ ע֜וֹד יַעֲקֹ֗ב כִּ֤י אִם־יִשְׂרָאֵל֙ יִהְיֶ֣ה שְׁמֶ֔ךָ וַיִּקְרָ֥א אֶת־שְׁמ֖וֹ יִשְׂרָאֵֽל׃ אָמַר לוֹ אֱלֹהִים: שִׁמְךָ יַעֲקֹב, מֵעַכְשָׁיו שִׁמְךָ לֹא יִקָּרֵא יוֹתֵר יַעֲקֹב אֶלָּא יִהְיֶה יִשְׂרָאֵל, וְהוּא קָרָא אֶת שְׁמוֹ יִשְׂרָאֵל.
יא וַיֹּ֩אמֶר֩ ל֨וֹ אֱלֹהִ֜ים אֲנִ֨י אֵ֤ל שַׁדַּי֙ פְּרֵ֣ה וּרְבֵ֔ה גּ֛וֹי וּקְהַ֥ל גּוֹיִ֖ם יִהְיֶ֣ה מִמֶּ֑ךָּ וּמְלָכִ֖ים מֵחֲלָצֶ֥יךָ יֵצֵֽאוּ׃ אָמַר לוֹ אֱלֹהִים: אֲנִי אֵל-שַׁדַּי, הוֹלֵד בָּנִים וְתִתְרַבֶּה, יֵצֵא מִמְּךָ עַם וַחֲבוּרַת עַמִּים, וְיִוָּלְדוּ מִמְּךָ מְלָכִים.

ששי[עריכה]

יב וְאֶת־הָאָ֗רֶץ אֲשֶׁ֥ר נָתַ֛תִּי לְאַבְרָהָ֥ם וּלְיִצְחָ֖ק לְךָ֣ אֶתְּנֶ֑נָּה וּֽלְזַרְעֲךָ֥ אַחֲרֶ֖יךָ אֶתֵּ֥ן אֶת־הָאָֽרֶץ׃ וְאֶת הָאָרֶץ שֶׁנָּתַתִּי לְאַבְרָהָם וּלְיִצְחָק אֶתֵּן אוֹתָהּ לְךָ, וּלְצֶאֱצָאֶיךָ אַחֲרֶיךָ אֶתֵּן אֶת הָאָרֶץ.
יג וַיַּ֥עַל מֵעָלָ֖יו אֱלֹהִ֑ים בַּמָּק֖וֹם אֲשֶׁר־דִּבֶּ֥ר אִתּֽוֹ׃ גִּלּוּי הָאֱלֹהִים הִסְתַּלֵּק מֵעָלָיו בְּאוֹתוֹ מָקוֹם שֶׁדִּבֵּר אִתּוֹ בַּפַּעַם הָרִאשׁוֹנָה.
יד וַיַּצֵּ֨ב יַעֲקֹ֜ב מַצֵּבָ֗ה בַּמָּק֛וֹם אֲשֶׁר־דִּבֶּ֥ר אִתּ֖וֹ מַצֶּ֣בֶת אָ֑בֶן וַיַּסֵּ֤ךְ עָלֶ֙יהָ֙ נֶ֔סֶךְ וַיִּצֹ֥ק עָלֶ֖יהָ שָֽׁמֶן׃ יַעֲקֹב הֶעֱמִיד מַצֵּבָה בַּמָּקוֹם שֶׁדִּבֵּר אִתּוֹ – מַצֵּבָה שֶׁל אֶבֶן, הוּא מָזַג עָלֶיהָ יַיִן וְיָצַק עָלֶיהָ שֶׁמֶן.
טו וַיִּקְרָ֨א יַעֲקֹ֜ב אֶת־שֵׁ֣ם הַמָּק֗וֹם אֲשֶׁר֩ דִּבֶּ֨ר אִתּ֥וֹ שָׁ֛ם אֱלֹהִ֖ים בֵּֽית־אֵֽל׃ יַעֲקֹב קָרָא אֶת שֵׁם הַמָּקוֹם שֶׁאֱלֹהִים דִּבֵּר אִתּוֹ שָׁם – בֵּית אֵל.
פָּרָשַׁת לֵידַת בִּנְיָמִין
טז וַיִּסְעוּ֙ מִבֵּ֣ית אֵ֔ל וַֽיְהִי־ע֥וֹד כִּבְרַת־הָאָ֖רֶץ לָב֣וֹא אֶפְרָ֑תָה וַתֵּ֥לֶד רָחֵ֖ל וַתְּקַ֥שׁ בְּלִדְתָּֽהּ׃ הֵם נָסְעוּ מִבֵּית אֵל, וּכְשֶׁהָיְתָה עוֹד דֶּרֶךְ אֲרֻכָּה לָבוֹא לְאֶפְרָת רָחֵל יָלְדָה וְהִתְקַשְּׁתָה לָלֶדֶת.
יז וַיְהִ֥י בְהַקְשֹׁתָ֖הּ בְּלִדְתָּ֑הּ וַתֹּ֨אמֶר לָ֤הּ הַמְיַלֶּ֙דֶת֙ אַל־תִּ֣ירְאִ֔י כִּֽי־גַם־זֶ֥ה לָ֖ךְ בֵּֽן׃ כְּשֶׁהִתְקַשְּׁתָה בַּלֵּדָה, אָמְרָה לָהּ הַמְּיַלֶּדֶת: אַל תִּפְחֲדִי כִּי גַּם זֶה לָךְ בֵּן זָכָר.
יח וַיְהִ֞י בְּצֵ֤את נַפְשָׁהּ֙ כִּ֣י מֵ֔תָה וַתִּקְרָ֥א שְׁמ֖וֹ בֶּן־אוֹנִ֑י וְאָבִ֖יו קָֽרָא־ל֥וֹ בִנְיָמִֽין׃ כְּשֶׁמֵּתָה וְיָצְאָה נַפְשָׁהּ, הִיא קָרְאָה אֶת שְׁמוֹ בֶּן אֶבְלִי, וְאָבִיו קָרָא לוֹ בִּנְיָמִין – בֶּן הַדָּרוֹם.
יט וַתָּ֖מׇת רָחֵ֑ל וַתִּקָּבֵר֙ בְּדֶ֣רֶךְ אֶפְרָ֔תָה הִ֖וא בֵּ֥ית לָֽחֶם׃ רָחֵל מֵתָה וְנִקְבְּרָה בַּדֶּרֶךְ לְאֶפְרָת הִיא בֵּית לֶחֶם.
כ וַיַּצֵּ֧ב יַעֲקֹ֛ב מַצֵּבָ֖ה עַל־קְבֻרָתָ֑הּ הִ֛וא מַצֶּ֥בֶת קְבֻֽרַת־רָחֵ֖ל עַד־הַיּֽוֹם׃ יַעֲקֹב הֶעֱמִיד מַצֵּבָה עַל קְבוּרָתָהּ, הִיא מַצֶּבֶת קְבוּרַת רָחֵל שֶׁעוֹמֶדֶת עַד הַיּוֹם.
כא וַיִּסַּ֖ע יִשְׂרָאֵ֑ל וַיֵּ֣ט אׇֽהֳלֹ֔ה מֵהָ֖לְאָה לְמִגְדַּל־עֵֽדֶר׃ יִשְׂרָאֵל נָסַע וְהֵקִים אֶת אָהֳלוֹ בְּמֶרְחָק אַחֲרֵי מִגְדַּל עֵדֶר.
כב וַיְהִ֗י בִּשְׁכֹּ֤ן יִשְׂרָאֵל֙ בָּאָ֣רֶץ הַהִ֔וא וַיֵּ֣לֶךְ רְאוּבֵ֗֔ן וַיִּשְׁכַּ֕ב֙ אֶת־בִּלְהָ֖ה֙ פִּילֶ֣גֶשׁ אָבִ֑֔יו וַיִּשְׁמַ֖ע יִשְׂרָאֵ֑͏ֽל


וַיִּֽהְי֥וּ בְנֵֽי־יַעֲקֹ֖ב שְׁנֵ֥ים עָשָֽׂר׃

כְּשֶׁיִּשְׂרָאֵל גָּר בָּאָרֶץ הַהִיא רְאוּבֵן הָלַךְ וְשָׁכַב עִ­ם בִּלְהָה שִׁפְחַת אָבִיו, וְיִשְׂרָאֵל שָׁמַע עַל כָּךְ.


בְּנֵי יַעֲקֹב הָיוּ שְׁנֵים עָשָׂר.

כג בְּנֵ֣י לֵאָ֔ה בְּכ֥וֹר יַעֲקֹ֖ב רְאוּבֵ֑ן וְשִׁמְעוֹן֙ וְלֵוִ֣י וִֽיהוּדָ֔ה וְיִשָּׂשכָ֖ר וּזְבֻלֽוּן׃ בְּנֵי לֵאָה: רְאוּבֵן – בְּכוֹר יַעֲקֹב, וְשִׁמְעוֹן וְלֵוִי וִיהוּדָה וְיִשָּׂשכָר וּזְבוּלוּן.
כד בְּנֵ֣י רָחֵ֔ל יוֹסֵ֖ף וּבִנְיָמִֽן׃ בְּנֵי רָחֵל: יוֹסֵף וּבִנְיָמִין.
כה וּבְנֵ֤י בִלְהָה֙ שִׁפְחַ֣ת רָחֵ֔ל דָּ֖ן וְנַפְתָּלִֽי׃ וּבְנֵי בִּלְהָה שִׁפְחַת רָחֵל: דָּן וְנַפְתָּלִי.
כו וּבְנֵ֥י זִלְפָּ֛ה שִׁפְחַ֥ת לֵאָ֖ה גָּ֣ד וְאָשֵׁ֑ר אֵ֚לֶּה בְּנֵ֣י יַעֲקֹ֔ב אֲשֶׁ֥ר יֻלַּד־ל֖וֹ בְּפַדַּ֥ן אֲרָֽם׃ וּבְנֵי זִלְפָּה שִׁפְחַת לֵאָה: גָּד וְאָשֵׁר, אֵלֶּה בְּנֵי יַעֲקֹב שֶׁנּוֹלְדוּ לוֹ בְּפַדַּן אֲרָם.
פָּרָשַׁת מוֹת יִצְחָק
כז וַיָּבֹ֤א יַעֲקֹב֙ אֶל־יִצְחָ֣ק אָבִ֔יו מַמְרֵ֖א קִרְיַ֣ת הָֽאַרְבַּ֑ע הִ֣וא חֶבְר֔וֹן אֲשֶׁר־גָּֽר־שָׁ֥ם אַבְרָהָ֖ם וְיִצְחָֽק׃ יַעֲקֹב בָּא אֶל יִצְחָק אָבִיו לְמַמְרֵא – לְקִרְיַת אַרְבַּע, הִיא חֶבְרוֹן שֶׁגָּרוּ שָׁם אַבְרָהָם וְיִצְחָק.
כח וַיִּֽהְי֖וּ יְמֵ֣י יִצְחָ֑ק מְאַ֥ת שָׁנָ֖ה וּשְׁמֹנִ֥ים שָׁנָֽה׃ יְמֵי יִצְחָק הָיוּ מֵאָה וּשְׁמוֹנִים שָׁנָה.
כט וַיִּגְוַ֨ע יִצְחָ֤ק וַיָּ֙מׇת֙ וַיֵּאָ֣סֶף אֶל־עַמָּ֔יו זָקֵ֖ן וּשְׂבַ֣ע יָמִ֑ים וַיִּקְבְּר֣וּ אֹת֔וֹ עֵשָׂ֥ו וְיַעֲקֹ֖ב בָּנָֽיו׃ נִשְׁמַת יִצְחָק יָצְאָה וְהוּא מֵת, נִשְׁמָתוֹ נֶאֶסְפָה לִמְקוֹם הַנְּשָׁמוֹת כְּשֶׁהוּא זָקֵן וְשָׂבֵעַ מֵרוֹב יָמִים, עֵשָׂו וְיַעֲקֹב בָּנָיו קָבְרוּ אוֹתוֹ.
פ פָּרָשַׁת תֹּלְדוֹת עֵשָׂו
לו א וְאֵ֛לֶּה תֹּלְד֥וֹת עֵשָׂ֖ו ה֥וּא אֱדֽוֹם׃ וְאֵלֶּה צֶאֱצָאֵי עֵשָׂו הוּא אֱדוֹם.
ב עֵשָׂ֛ו לָקַ֥ח אֶת־נָשָׁ֖יו מִבְּנ֣וֹת כְּנָ֑עַן אֶת־עָדָ֗ה בַּת־אֵילוֹן֙ הַֽחִתִּ֔י וְאֶת־אׇהֳלִֽיבָמָה֙ בַּת־עֲנָ֔ה בַּת־צִבְע֖וֹן הַֽחִוִּֽי׃ עֵשָׂו לָקַח אֶת נָשָׁיו מִבְּנוֹת כְּנַעַן, אֶת עָדָה בַּת אֵילוֹן הַחִתִּי, וְאֶת אָהֳלִיבָמָה בַּת עֲנָה בַּת צִבְעוֹן הַחִוִּי.
ג וְאֶת־בָּשְׂמַ֥ת בַּת־יִשְׁמָעֵ֖אל אֲח֥וֹת נְבָיֽוֹת׃ וְאֶת בָּשְׂמַת בַּת יִשְׁמָעֵאל אֲחוֹת נְבָיוֹת.
ד וַתֵּ֧לֶד עָדָ֛ה לְעֵשָׂ֖ו אֶת־אֱלִיפָ֑ז וּבָ֣שְׂמַ֔ת יָלְדָ֖ה אֶת־רְעוּאֵֽל׃ עָדָה יָלְדָה לְעֵשָׂו אֶת אֱלִיפָז, וּבָשְׂמַת יָלְדָה אֶת רְעוּאֵל.
ה וְאׇהֳלִֽיבָמָה֙ יָֽלְדָ֔ה אֶת־יְע֥וּשׁ יעיש וְאֶת־יַעְלָ֖ם וְאֶת־קֹ֑רַח אֵ֚לֶּה בְּנֵ֣י עֵשָׂ֔ו אֲשֶׁ֥ר יֻלְּדוּ־ל֖וֹ בְּאֶ֥רֶץ כְּנָֽעַן׃ וְאָהֳלִיבָמָה יָלְדָה אֶת יְעוּשׁ וְאֶת יַעְלָם וְאֶת קֹרַח, אֵלֶּה בְּנֵי עֵשָׂו שֶׁנּוֹלְדוּ לוֹ בְּאֶרֶץ כְּנַעַן.
ו וַיִּקַּ֣ח עֵשָׂ֡ו אֶת־נָ֠שָׁ֠יו וְאֶת־בָּנָ֣יו וְאֶת־בְּנֹתָיו֮ וְאֶת־כׇּל־נַפְשׁ֣וֹת בֵּיתוֹ֒ וְאֶת־מִקְנֵ֣הוּ וְאֶת־כׇּל־בְּהֶמְתּ֗וֹ וְאֵת֙ כׇּל־קִנְיָנ֔וֹ אֲשֶׁ֥ר רָכַ֖שׁ בְּאֶ֣רֶץ כְּנָ֑עַן וַיֵּ֣לֶךְ אֶל־אֶ֔רֶץ מִפְּנֵ֖י יַעֲקֹ֥ב אָחִֽיו׃ עֵשָׂו לָקַח אֶת נָשָׁיו וְאֶת בָּנָיו וְאֶת בְּנוֹתָיו וְאֶת כָּל אַנְשֵׁי בֵּיתוֹ וְאֶת עֲדָרָיו וְאֶת כָּל בֶּהֱמוֹתָיו וְאֶת כָּל רְכוּשׁוֹ שֶׁרָכַשׁ בְּאֶרֶץ כְּנַעַן, וְהָלַךְ לְאֶרֶץ אֲחֶרֶת, לְהִתְרַחֵק מִיַּעֲקֹב אָחִיו.
ז כִּֽי־הָיָ֧ה רְכוּשָׁ֛ם רָ֖ב מִשֶּׁ֣בֶת יַחְדָּ֑ו וְלֹ֨א יָֽכְלָ֜ה אֶ֤רֶץ מְגֽוּרֵיהֶם֙ לָשֵׂ֣את אֹתָ֔ם מִפְּנֵ֖י מִקְנֵיהֶֽם׃ כִּי רְכוּשָׁם הָיָה רָב מִלָּשֶׁבֶת יַחַד, וְאֶרֶץ מְגוּרֵיהֶם לֹא יָכְלָה לְהָכִיל אוֹתָם בִּגְלַל גּוֹדֶל עֶדְרֵיהֶם.
ח וַיֵּ֤שֶׁב עֵשָׂו֙ בְּהַ֣ר שֵׂעִ֔יר עֵשָׂ֖ו ה֥וּא אֱדֽוֹם׃ עֵשָׂו הִתְיַשֵּׁב בְּהַר שֵׂעִיר, עֵשָׂו הוּא אֱדוֹם.
ט וְאֵ֛לֶּה תֹּלְד֥וֹת עֵשָׂ֖ו אֲבִ֣י אֱד֑וֹם בְּהַ֖ר שֵׂעִֽיר׃ וְאֵלֶּה צֶאֱצָאֵי עֵשָׂו אֲבִי הָעָם אֱדוֹם, בְּהַר שֵׂעִיר.
י אֵ֖לֶּה שְׁמ֣וֹת בְּנֵֽי־עֵשָׂ֑ו אֱלִיפַ֗ז בֶּן־עָדָה֙ אֵ֣שֶׁת עֵשָׂ֔ו רְעוּאֵ֕ל בֶּן־בָּשְׂמַ֖ת אֵ֥שֶׁת עֵשָֽׂו׃ אֵלֶּה שְׁמוֹת בְּנֵי עֵשָׂו: אֱלִיפַז בֶּן עָדָה אֵשֶׁת עֵשָׂו רְעוּאֵל בֶּן בָּשְׂמַת אֵשֶׁת עֵשָׂו.
יא וַיִּהְי֖וּ בְּנֵ֣י אֱלִיפָ֑ז תֵּימָ֣ן אוֹמָ֔ר צְפ֥וֹ וְגַעְתָּ֖ם וּקְנַֽז׃ בְּנֵי אֱלִיפָז הָיוּ: תֵּימָן, אוֹמָר, צְפוֹ וְגַעְתָּם וּקְנַז.
יב וְתִמְנַ֣ע ׀ הָיְתָ֣ה פִילֶ֗גֶשׁ לֶֽאֱלִיפַז֙ בֶּן־עֵשָׂ֔ו וַתֵּ֥לֶד לֶאֱלִיפַ֖ז אֶת־עֲמָלֵ֑ק אֵ֕לֶּה בְּנֵ֥י עָדָ֖ה אֵ֥שֶׁת עֵשָֽׂו׃ וְתִמְנַע הָיְתָה שִׁפְחָה לֶאֱלִיפַז בֶּן עֵשָׂו, וְהִיא יָלְדָה לֶאֱלִיפַז אֶת עֲמָלֵק, אֵלֶּה בְּנֵי עָדָה אֵשֶׁת עֵשָׂו.
יג וְאֵ֙לֶּה֙ בְּנֵ֣י רְעוּאֵ֔ל נַ֥חַת וָזֶ֖רַח שַׁמָּ֣ה וּמִזָּ֑ה אֵ֣לֶּה הָי֔וּ בְּנֵ֥י בָשְׂמַ֖ת אֵ֥שֶׁת עֵשָֽׂו׃ וְאֵלֶּה בְּנֵי רְעוּאֵל: נַחַת וְזֶרַח וְשַׁמָּה וּמִזָּה, אֵלֶּה הָיוּ בְּנֵי בָּשְׂמַת אֵשֶׁת עֵשָׂו.
יד וְאֵ֣לֶּה הָי֗וּ בְּנֵ֨י אׇהֳלִיבָמָ֧ה בַת־עֲנָ֛ה בַּת־צִבְע֖וֹן אֵ֣שֶׁת עֵשָׂ֑ו וַתֵּ֣לֶד לְעֵשָׂ֔ו אֶת־יְע֥וּשׁ יעיש וְאֶת־יַעְלָ֖ם וְאֶת־קֹֽרַח׃ וְאֵלֶּה הָיוּ בְּנֵי אָהֳלִיבָמָה בַּת עֲנָה בַּת צִבְעוֹן אֵשֶׁת עֵשָׂו: הִיא יָלְדָה לְעֵשָׂו אֶת יְעוּשׁ וְאֶת יַעְלָם וְאֶת קֹרַח.
טו אֵ֖לֶּה אַלּוּפֵ֣י בְנֵֽי־עֵשָׂ֑ו בְּנֵ֤י אֱלִיפַז֙ בְּכ֣וֹר עֵשָׂ֔ו אַלּ֤וּף תֵּימָן֙ אַלּ֣וּף אוֹמָ֔ר אַלּ֥וּף צְפ֖וֹ אַלּ֥וּף קְנַֽז׃ אֵלֶּה שַׁלִּיטֵי בְּנֵי עֵשָׂו: מִבְּנֵי אֱלִיפַז בְּכוֹר עֵשָׂו – שַׁלִּיט תֵּימָן, שַׁלִּיט אוֹמָר, שַׁלִּיט צְפוֹ, שַׁלִּיט קְנַז.
טז אַלּֽוּף־קֹ֛רַח אַלּ֥וּף גַּעְתָּ֖ם אַלּ֣וּף עֲמָלֵ֑ק אֵ֣לֶּה אַלּוּפֵ֤י אֱלִיפַז֙ בְּאֶ֣רֶץ אֱד֔וֹם אֵ֖לֶּה בְּנֵ֥י עָדָֽה׃ שַׁלִּיט קוֹרַח, שַׁלִּיט גַּעְתָּם, שַׁלִּיט עֲמָלֵק, אֵלֶּה שַׁלִּיטֵי אֱלִיפַז בְּאֶרֶץ אֱדוֹם, אֵלֶּה מִבְּנֵי עָדָה.
יז וְאֵ֗לֶּה בְּנֵ֤י רְעוּאֵל֙ בֶּן־עֵשָׂ֔ו אַלּ֥וּף נַ֙חַת֙ אַלּ֣וּף זֶ֔רַח אַלּ֥וּף שַׁמָּ֖ה אַלּ֣וּף מִזָּ֑ה אֵ֣לֶּה אַלּוּפֵ֤י רְעוּאֵל֙ בְּאֶ֣רֶץ אֱד֔וֹם אֵ֕לֶּה בְּנֵ֥י בָשְׂמַ֖ת אֵ֥שֶׁת עֵשָֽׂו׃ וְאֵלֶּה מִבְּנֵי רְעוּאֵל בֶּן עֵשָׂו: שַׁלִּיט נַחַת, שַׁלִּיט זֶרַח, שַׁלִּיט שַׁמָּה, שַׁלִּיט מִזָּה, אֵלֶּה שַׁלִּיטֵי רְעוּאֵל בְּאֶרֶץ אֱדוֹם, אֵלֶּה מִבְּנֵי בָּשְׂמַת אֵשֶׁת עֵשָׂו.
יח וְאֵ֗לֶּה בְּנֵ֤י אׇהֳלִֽיבָמָה֙ אֵ֣שֶׁת עֵשָׂ֔ו אַלּ֥וּף יְע֛וּשׁ אַלּ֥וּף יַעְלָ֖ם אַלּ֣וּף קֹ֑רַח אֵ֣לֶּה אַלּוּפֵ֞י אׇהֳלִֽיבָמָ֛ה בַּת־עֲנָ֖ה אֵ֥שֶׁת עֵשָֽׂו׃ וְאֵלֶּה מִבְּנֵי אָהֳלִיבָמָה אֵשֶׁת עֵשָׂו: שַׁלִּיט יְעוּשׁ, שַׁלִּיט יַעְלָם, שַׁלִּיט קֹרַח, אֵלֶּה שַׁלִּיטֵי אָהֳלִיבָמָה בַּת עֲנָה אֵשֶׁת עֵשָׂו.
יט אֵ֧לֶּה בְנֵי־עֵשָׂ֛ו וְאֵ֥לֶּה אַלּוּפֵיהֶ֖ם ה֥וּא אֱדֽוֹם׃ אֵלֶּה בְּנֵי עֵשָׂו וְאֵלֶּה שַׁלִּיטֵיהֶם, עֵשָׂו הוּא אֱדוֹם.

שביעי[עריכה]

ס פָּרָשַׁת תֹּלְדוֹת שֵׂעִיר
כ אֵ֤לֶּה בְנֵֽי־שֵׂעִיר֙ הַחֹרִ֔י יֹשְׁבֵ֖י הָאָ֑רֶץ לוֹטָ֥ן וְשׁוֹבָ֖ל וְצִבְע֥וֹן וַעֲנָֽה׃ אֵלֶּה בְּנֵי שֵׂעִיר מֵהָעָם הַחוֹרִי יוֹשְׁבֵי הָאָרֶץ הַקּוֹדְמִים: לוֹטָן וְשׁוֹבָל וְצִבְעוֹן וַעֲנָה.
כא וְדִשׁ֥וֹן וְאֵ֖צֶר וְדִישָׁ֑ן אֵ֣לֶּה אַלּוּפֵ֧י הַחֹרִ֛י בְּנֵ֥י שֵׂעִ֖יר בְּאֶ֥רֶץ אֱדֽוֹם׃ וְדִשׁוֹן וְאֵצֶר וְדִישָׁן, אֵלֶּה שַׁלִּיטֵי הָעָם הַחוֹרִי – בְּנֵי שֵׂעִיר בְּאֶרֶץ אֱדוֹם.
כב וַיִּהְי֥וּ בְנֵי־לוֹטָ֖ן חֹרִ֣י וְהֵימָ֑ם וַאֲח֥וֹת לוֹטָ֖ן תִּמְנָֽע׃ בְּנֵי לוֹטָן הָיוּ: חוֹרִי וְהֵימָם, וַאֲחוֹת לוֹטָן תִּמְנָע.
כג וְאֵ֙לֶּה֙ בְּנֵ֣י שׁוֹבָ֔ל עַלְוָ֥ן וּמָנַ֖חַת וְעֵיבָ֑ל שְׁפ֖וֹ וְאוֹנָֽם׃ וְאֵלֶּה בְּנֵי שׁוֹבָל: עַלְוָן וּמָנַחַת וְעֵיבָל שְׁפוֹ וְאוֹנָם.
כד וְאֵ֥לֶּה בְנֵֽי־צִבְע֖וֹן וְאַיָּ֣ה וַעֲנָ֑ה ה֣וּא עֲנָ֗ה אֲשֶׁ֨ר מָצָ֤א אֶת־הַיֵּמִם֙ בַּמִּדְבָּ֔ר בִּרְעֹת֥וֹ אֶת־הַחֲמֹרִ֖ים לְצִבְע֥וֹן אָבִֽיו׃ וְאֵלֶּה בְּנֵי צִבְעוֹן: אַיָּה וַעֲנָה, הוּא עֲנָה שֶׁגִּלָּה אֶת הַפְּרָדִים בַּמִּדְבָּר כְּשֶׁרָעָה אֶת הַחֲמוֹרִים לְצִבְעוֹן אָבִיו.
כה וְאֵ֥לֶּה בְנֵֽי־עֲנָ֖ה דִּשֹׁ֑ן וְאׇהֳלִיבָמָ֖ה בַּת־עֲנָֽה׃ וְאֵלֶּה בְּנֵי עֲנָה: דִּשׁוֹן וְאָהֳלִיבָמָה בַּת עֲנָה.
כו וְאֵ֖לֶּה בְּנֵ֣י דִישָׁ֑ן חֶמְדָּ֥ן וְאֶשְׁבָּ֖ן וְיִתְרָ֥ן וּכְרָֽן׃ וְאֵלֶּה בְּנֵי דִּישָׁן: חֶמְדָּן וְאֶשְׁבָּן וְיִתְרָן וּכְרָן.
כז אֵ֖לֶּה בְּנֵי־אֵ֑צֶר בִּלְהָ֥ן וְזַעֲוָ֖ן וַעֲקָֽן׃ אֵלֶּה בְּנֵי אֵצֶר: בִּלְהָן וְזַעֲוָן וַעֲקָן.
כח אֵ֥לֶּה בְנֵֽי־דִישָׁ֖ן ע֥וּץ וַאֲרָֽן׃ אֵלֶּה בְּנֵי דִּישָׁן: עוּץ וַאֲרָן.
כט אֵ֖לֶּה אַלּוּפֵ֣י הַחֹרִ֑י אַלּ֤וּף לוֹטָן֙ אַלּ֣וּף שׁוֹבָ֔ל אַלּ֥וּף צִבְע֖וֹן אַלּ֥וּף עֲנָֽה׃ אֵלֶּה שַׁלִּיטֵי הָעָם הַחוֹרִי: שַׁלִּיט לוֹטָן, שַׁלִּיט שׁוֹבָל, שַׁלִּיט צִבְעוֹן, שַׁלִּיט עֲנָה.
ל אַלּ֥וּף דִּשֹׁ֛ן אַלּ֥וּף אֵ֖צֶר אַלּ֣וּף דִּישָׁ֑ן אֵ֣לֶּה אַלּוּפֵ֧י הַחֹרִ֛י לְאַלֻּפֵיהֶ֖ם בְּאֶ֥רֶץ שֵׂעִֽיר׃ שַׁלִּיט דִּשׁוֹן, שַׁלִּיט אֵצֶר, שַׁלִּיט דִּישָׁן, אֵלֶּה שַׁלִּיטֵי הַחוֹרִי, שַׁלִּיטֵיהֶם שֶׁבְּאֶרֶץ שֵׂעִיר.
פ פָּרָשַׁת מַלְכֵי אֱדוֹם
לא וְאֵ֙לֶּה֙ הַמְּלָכִ֔ים אֲשֶׁ֥ר מָלְכ֖וּ בְּאֶ֣רֶץ אֱד֑וֹם לִפְנֵ֥י מְלׇךְ־מֶ֖לֶךְ לִבְנֵ֥י יִשְׂרָאֵֽל׃ וְאֵלֶּה הַמְּלָכִים שֶׁמָּלְכוּ בְּאֶרֶץ אֱדוֹם לִפְנֵי שֶׁמָּלַךְ מֶלֶךְ לִבְנֵי יִשְׂרָאֵל.
לב וַיִּמְלֹ֣ךְ בֶּאֱד֔וֹם בֶּ֖לַע בֶּן־בְּע֑וֹר וְשֵׁ֥ם עִיר֖וֹ דִּנְהָֽבָה׃ מָלַךְ בֶּאֱדוֹם בֶּלַע בֶּן בְּעוֹר וְשֵׁם עִירוֹ דִּנְהָבָה.
לג וַיָּ֖מׇת בָּ֑לַע וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו יוֹבָ֥ב בֶּן־זֶ֖רַח מִבׇּצְרָֽה׃ בֶּלַע מֵת, וּמָלַךְ בִּמְקוֹמוֹ יוֹבָב בֶּן זֶרַח מֵהָעִיר בָּצְרָה.
לד וַיָּ֖מׇת יוֹבָ֑ב וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו חֻשָׁ֖ם מֵאֶ֥רֶץ הַתֵּימָנִֽי׃ יוֹבָב מֵת, וּמָלַךְ בִּמְקוֹמוֹ חוּשָׁם מֵאֶרֶץ הַדָּרוֹם.
לה וַיָּ֖מׇת חֻשָׁ֑ם וַיִּמְלֹ֨ךְ תַּחְתָּ֜יו הֲדַ֣ד בֶּן־בְּדַ֗ד הַמַּכֶּ֤ה אֶת־מִדְיָן֙ בִּשְׂדֵ֣ה מוֹאָ֔ב וְשֵׁ֥ם עִיר֖וֹ עֲוִֽית׃ חוּשָׁם מֵת, וּמָלַךְ בִּמְקוֹמוֹ הֲדַד בֶּן בְּדַד שֶׁהָרַג אֶת אַנְשֵׁי מִדְיָן בִּשְׂדֵה מוֹאָב וְשֵׁם עִירוֹ עֲוִית.
לו וַיָּ֖מׇת הֲדָ֑ד וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו שַׂמְלָ֖ה מִמַּשְׂרֵקָֽה׃ הֲדָד מֵת, וּמָלַךְ בִּמְקוֹמוֹ שַׂמְלָה מֵהָעִיר מַשְׂרֵקָה.
לז וַיָּ֖מׇת שַׂמְלָ֑ה וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו שָׁא֖וּל מֵרְחֹב֥וֹת הַנָּהָֽר׃ שַׂמְלָה מֵת, וּמָלַךְ בִּמְקוֹמוֹ שָׁאוּל מֵהָעִיר רְחוֹבוֹת שֶׁעַל שְׂפַת נְהַר פְּרָת.
לח וַיָּ֖מׇת שָׁא֑וּל וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו בַּ֥עַל חָנָ֖ן בֶּן־עַכְבּֽוֹר׃ שָׁאוּל מֵת, וּמָלַךְ בִּמְקוֹמוֹ בַּעַל חָנָן בֶּן עַכְבּוֹר.
לט וַיָּ֘מׇת֮ בַּ֣עַל חָנָ֣ן בֶּן־עַכְבּוֹר֒ וַיִּמְלֹ֤ךְ תַּחְתָּיו֙ הֲדַ֔ר וְשֵׁ֥ם עִיר֖וֹ פָּ֑עוּ וְשֵׁ֨ם אִשְׁתּ֤וֹ מְהֵֽיטַבְאֵל֙ בַּת־מַטְרֵ֔ד בַּ֖ת מֵ֥י זָהָֽב׃ בַּעַל חָנָן בֶּן עַכְבּוֹר מֵת, וּמָלַךְ בִּמְקוֹמוֹ הֲדַר וְשֵׁם עִירוֹ פָּעוּ, וְשֵׁם אִשְׁתּוֹ מְהֵיטַבְאֵל בַּת מַטְרֵד בַּת צוֹרֵף הַזָּהָב.

מפטיר[עריכה]

מ וְ֠אֵ֠לֶּה שְׁמ֞וֹת אַלּוּפֵ֤י עֵשָׂו֙ לְמִשְׁפְּחֹתָ֔ם לִמְקֹמֹתָ֖ם בִּשְׁמֹתָ֑ם אַלּ֥וּף תִּמְנָ֛ע אַלּ֥וּף עַֽלְוָ֖ה אַלּ֥וּף יְתֵֽת׃ וְאֵלֶּה שְׁמוֹת שַׁלִּיטֵי עֵשָׂו לְמִשְׁפְּחוֹתֵיהֶם שֶׁנִּקְרְאוּ לְפִי מְקוֹמוֹת מְגוּרֵיהֶם: שַׁלִּיט תִּמְנָע, שַׁלִּיט עַלְוָה, שַׁלִּיט יְתֵת.
מא אַלּ֧וּף אׇהֳלִיבָמָ֛ה אַלּ֥וּף אֵלָ֖ה אַלּ֥וּף פִּינֹֽן׃ שַׁלִּיט אָהֳלִיבָמָה, שַׁלִּיט אֵלָה, שַׁלִּיט פִּינוֹן.
מב אַלּ֥וּף קְנַ֛ז אַלּ֥וּף תֵּימָ֖ן אַלּ֥וּף מִבְצָֽר׃ שַׁלִּיט קְנַז, שַׁלִּיט תֵּימָן, שַׁלִּיט מִבְצָר.
מג אַלּ֥וּף מַגְדִּיאֵ֖ל אַלּ֣וּף עִירָ֑ם אֵ֣לֶּה ׀ אַלּוּפֵ֣י אֱד֗וֹם לְמֹֽשְׁבֹתָם֙ בְּאֶ֣רֶץ אֲחֻזָּתָ֔ם ה֥וּא עֵשָׂ֖ו אֲבִ֥י אֱדֽוֹם׃ שַׁלִּיט מַגְדִּיאֵל, שַׁלִּיט עִירָם, אֵלֶּה שַׁלִּיטֵי אֱדוֹם בְּמוֹשְׁבוֹתֵיהֶם בְּאֶרֶץ אֲחֻזָּתָם, הוּא עֵשָׂו אֲבִי הָעָם אֱדוֹם.
