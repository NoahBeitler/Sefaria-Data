קפיצה לניווט קפיצה לחיפוש
א א שִׁ֥יר הַשִּׁירִ֖ים אֲשֶׁ֥ר לִשְׁלֹמֹֽה׃ זֶה הַשִּׁיר הַמֻּבְחָר שֶׁל שְׁלֹמֹה.
ב יִשָּׁקֵ֙נִי֙ מִנְּשִׁיק֣וֹת פִּ֔יהוּ כִּֽי־טוֹבִ֥ים דֹּדֶ֖יךָ מִיָּֽיִן׃ נַשֵּׁק אוֹתִי מִנְּשִׁיקוֹת פִּיךָ, כִּי אַהֲבָתְךָ טוֹבָה יוֹתֵר מִיַּיִן.
ג לְרֵ֙יחַ֙ שְׁמָנֶ֣יךָ טוֹבִ֔ים שֶׁ֖מֶן תּוּרַ֣ק שְׁמֶ֑ךָ עַל־כֵּ֖ן עֲלָמ֥וֹת אֲהֵבֽוּךָ׃ שִׁמְךָ הַטּוֹב יוֹצֵא לְפָנֶיךְ כְּרֵיחַ שֶׁל שְׁמָנִים מְבֻשָּׂמִים וּבְסִבָּתָם נְעָרוֹת מִתְאַהֲבוֹת בְּךָ.
ד מׇשְׁכֵ֖נִי אַחֲרֶ֣יךָ נָּר֑וּצָה הֱבִיאַ֨נִי הַמֶּ֜לֶךְ חֲדָרָ֗יו נָגִ֤ילָה וְנִשְׂמְחָה֙ בָּ֔ךְ נַזְכִּ֤ירָה דֹדֶ֙יךָ֙ מִיַּ֔יִן מֵישָׁרִ֖ים אֲהֵבֽוּךָ׃ מְשׁוֹךְ אוֹתִי אֵלֶיךָ וְנָרוּץ יַחַד אַחֲרֶיךָ כְּקֶדֶם, כְּשֶׁהַמֶּלֶךְ הִכְנִיס אוֹתִי אֶל חֲדָרָיו, נָגִיל וְנִשְׂמַח בְּךָ, נִשְׁאַף מֵרֵיחַ אַהֲבָתְךָ הַטּוֹבָה יוֹתֵר מִיַּיִן, אוֹהֲבִים אוֹתְךָ בְּנֶאֱמָנוּת.
פ
ה שְׁחוֹרָ֤ה אֲנִי֙ וְֽנָאוָ֔ה בְּנ֖וֹת יְרוּשָׁלָ֑͏ִם כְּאׇהֳלֵ֣י קֵדָ֔ר כִּירִיע֖וֹת שְׁלֹמֹֽה׃ אֲנִי אֻמַּת יְרוּשָׁלַיִם, גַּם כְּשֶׁאֵינֶנִּי מְאִירַת פָּנִים עֲדַיִן בְּתוֹכִי אֲנִי נָאָה, כְּמוֹ אוֹהָלִים מֻכֵּי שֶׁמֶשׁ כְּמוֹ יְרִיעוֹת בְּמִקְדַּשׁ שְׁלֹמֹה.
ו אַל־תִּרְא֙וּנִי֙ שֶׁאֲנִ֣י שְׁחַרְחֹ֔רֶת שֶׁשְּׁזָפַ֖תְנִי הַשָּׁ֑מֶשׁ בְּנֵ֧י אִמִּ֣י נִֽחֲרוּ־בִ֗י שָׂמֻ֙נִי֙ נֹטֵרָ֣ה אֶת־הַכְּרָמִ֔ים כַּרְמִ֥י שֶׁלִּ֖י לֹ֥א נָטָֽרְתִּי׃ אַל תִּסְתַּכְּלוּ עָלַי כָּכָה עַל שֶׁאֲנִי כֵּהָה, זֶה רַק שִׁזּוּף בַּשֶּׁמֶשׁ, בְּנֵי עַמִּי זִלְזְלוּ בִּי וְשָׁלְחוּ אוֹתִי לִשְׁמוֹר בְּכַרְמֵי זָרִים, לֹא בַּכֶּרֶם שֶׁלִּי.
ז הַגִּ֣ידָה לִּ֗י שֶׁ֤אָֽהֲבָה֙ נַפְשִׁ֔י אֵיכָ֣ה תִרְעֶ֔ה אֵיכָ֖ה תַּרְבִּ֣יץ בַּֽצׇּהֳרָ֑יִם שַׁלָּמָ֤ה אֶֽהְיֶה֙ כְּעֹ֣טְיָ֔ה עַ֖ל עֶדְרֵ֥י חֲבֵרֶֽיךָ׃ סַפֵּר לִי אָהוּב נַפְשִׁי, אֵיךְ אַתָּה רוֹעֶה אֶת צֹאנְךָ? כֵּיצַד אַתָּה גּוֹרֵם לוֹ לִרְבּוֹץ כְּחוֹם הַיּוֹם? לָמָּה שֶׁאֶהְיֶה כְּמוֹ אֲבֵלָה עַל עֶדְרֵי אֲהוּבֶיךָ שֶׁעָלַי לִרְעוֹת!
ח אִם־לֹ֤א תֵֽדְעִי֙ לָ֔ךְ הַיָּפָ֖ה בַּנָּשִׁ֑ים צְֽאִי־לָ֞ךְ בְּעִקְבֵ֣י הַצֹּ֗אן וּרְעִי֙ אֶת־גְּדִיֹּתַ֔יִךְ עַ֖ל מִשְׁכְּנ֥וֹת הָרֹעִֽים׃ – הַיָּפָה שֶׁבַּנָּשִׁים, אִם אֵינֵךְ יוֹדַעַת – צְאִי בְּעִקְבוֹת רוֹעֵי הַצֹּאן וּרְעִי אֶת הַגְּדִיּוֹת שֶׁלָּךְ עַל יַד מְקוֹם הָרוֹעִים הַבְּקִיאִים.
פ
ט לְסֻֽסָתִי֙ בְּרִכְבֵ֣י פַרְעֹ֔ה דִּמִּיתִ֖יךְ רַעְיָתִֽי׃ דּוֹמָה אַתְּ רַעֲיָתִי לְסוּסַת רְכִיבָה מִצְרִית זְרִיזָה.
י נָאו֤וּ לְחָיַ֙יִךְ֙ בַּתֹּרִ֔ים צַוָּארֵ֖ךְ בַּחֲרוּזִֽים׃ לְחָיַיִךְ כֹּה נָאִים בַּתַּכְשִׁיטִים הַמְּשֻׁלָּשִׁים, וְצַוָּארֵךְ כֹּה יָפֶה בַּחֲרוּזִים.
יא תּוֹרֵ֤י זָהָב֙ נַֽעֲשֶׂה־לָּ֔ךְ עִ֖ם נְקֻדּ֥וֹת הַכָּֽסֶף׃ נַעֲשֶׂה לָךְ מְשֻׁלְּשֵׁי זָהָב עִם נְקֻדּוֹת כֶּסֶף.
יב עַד־שֶׁ֤הַמֶּ֙לֶךְ֙ בִּמְסִבּ֔וֹ נִרְדִּ֖י נָתַ֥ן רֵיחֽוֹ׃ – בְּעוֹד שֶׁהַמֶּלֶךְ יוֹשֵׁב לִשְׁתּוֹת גַּם הַבּוֹשֶׂם שֶׁלִּי מַדִּיף רֵיחַ.
יג צְר֨וֹר הַמֹּ֤ר ׀ דּוֹדִי֙ לִ֔י בֵּ֥ין שָׁדַ֖י יָלִֽין׃ אֲהוּבִי הוּא עֲבוּרִי כְּמוֹ חֲבִילַת בּוֹשֶׂם הַמּוֹר, הוּא יָבוֹא לִישׁוֹן בְּחֵקִי.
יד אֶשְׁכֹּ֨ל הַכֹּ֤פֶר ׀ דּוֹדִי֙ לִ֔י בְּכַרְמֵ֖י עֵ֥ין גֶּֽדִי׃ אֲהוּבִי הוּא עֲבוּרִי כְּמוֹ אֶשְׁכּוֹל בּוֹשֶׂם הַכּוֹפֶר שֶׁגָּדֵל בְּעֵין גֶּדִי.
ס
טו הִנָּ֤ךְ יָפָה֙ רַעְיָתִ֔י הִנָּ֥ךְ יָפָ֖ה עֵינַ֥יִךְ יוֹנִֽים׃ אַתְּ יָפָה רַעֲיָתִי, עֵינַיִךְ יָפוֹת כְּמוֹ יוֹנִים.
טז הִנְּךָ֨ יָפֶ֤ה דוֹדִי֙ אַ֣ף נָעִ֔ים אַף־עַרְשֵׂ֖נוּ רַעֲנָנָֽה׃ – אַתָּה יָפֶה אֲהוּבִי וְגַם נָעִים, וְגַם מִטָּתֵנוּ רַעֲנָנָה.
יז קֹר֤וֹת בָּתֵּ֙ינוּ֙ אֲרָזִ֔ים רחיטנו רַהִיטֵ֖נוּ בְּרוֹתִֽים׃ תִּקְרַת בֵּיתֵנוּ מֵאֲרָזִים, וְהַנְּסָרִים הַמֻּחְלָקִים עֲשׂוּיִים מִבְּרוֹשִׁים.
ב א אֲנִי֙ חֲבַצֶּ֣לֶת הַשָּׁר֔וֹן שֽׁוֹשַׁנַּ֖ת הָעֲמָקִֽים׃ אֲנִי כְּמוֹ פֶּרַח חֲבַצֶּלֶת הַשָּׁרוֹן, כְּמוֹ שׁוֹשַׁנַּת הָעֵמֶק.
ב כְּשֽׁוֹשַׁנָּה֙ בֵּ֣ין הַחוֹחִ֔ים כֵּ֥ן רַעְיָתִ֖י בֵּ֥ין הַבָּנֽוֹת׃ – כְּמוֹ שׁוֹשַׁנָּה בֵּין קוֹצִים כָּךְ רַעְיָתִי בֵּין הַבָּנוֹת.
ג כְּתַפּ֙וּחַ֙ בַּעֲצֵ֣י הַיַּ֔עַר כֵּ֥ן דּוֹדִ֖י בֵּ֣ין הַבָּנִ֑ים בְּצִלּוֹ֙ חִמַּ֣דְתִּי וְיָשַׁ֔בְתִּי וּפִרְי֖וֹ מָת֥וֹק לְחִכִּֽי׃ – כְּמוֹ עֵץ תַּפּוּחַ בֵּין עֲצֵי יַעַר כָּךְ אֲהוּבִי בֵּין הַבָּנִים, בְּצִלּוֹ חָמַדְתִּי לָשֶׁבֶת וּפִרְיוֹ מָתוֹק לְפִי.
ד הֱבִיאַ֙נִי֙ אֶל־בֵּ֣ית הַיָּ֔יִן וְדִגְל֥וֹ עָלַ֖י אַהֲבָֽה׃ הוּא הֱבִיא אוֹתִי לְחֵדֶר הַיַּיִן וְהֵנִיף עָלַי אַהֲבָה.
ה סַמְּכ֙וּנִי֙ בָּֽאֲשִׁישׁ֔וֹת רַפְּד֖וּנִי בַּתַּפּוּחִ֑ים כִּי־חוֹלַ֥ת אַהֲבָ֖ה אָֽנִי׃ שִׂימוּ בְּצִידִי גְּבִיעֵי יַיִן, רַפְּדוּ אֶת מְקוֹמִי בְּתַפּוּחִים, כִּי אֲנִי חוֹלַת אַהֲבָה.
ו שְׂמֹאלוֹ֙ תַּ֣חַת לְרֹאשִׁ֔י וִימִינ֖וֹ תְּחַבְּקֵֽנִי׃ יַד שְׂמֹאלוֹ תּוֹמֶכֶת בְּרֹאשִׁי בְּחִבָּה, וְיַד יְמִינוֹ מְחַבֶּקֶת אוֹתִי.
ז הִשְׁבַּ֨עְתִּי אֶתְכֶ֜ם בְּנ֤וֹת יְרוּשָׁלַ֙͏ִם֙ בִּצְבָא֔וֹת א֖וֹ בְּאַיְל֣וֹת הַשָּׂדֶ֑ה אִם־תָּעִ֧ירוּ ׀ וְֽאִם־תְּע֥וֹרְﬞר֛וּ אֶת־הָאַהֲבָ֖ה עַ֥ד שֶׁתֶּחְפָּֽץ׃ מַשְׁבִּיעָה אֲנִי אֶתְכֶם בְּנֵי יִשְׂרָאֵל בַּצְּבָיִים אוֹ בָּאַיָּלוֹת שֶׁבַּשָּׂדֶה, אַל תָּעִירוּ וְאַל תְּעוֹרְרוּ אֶת הָאַהֲבָה עַד שֶׁתַּחְפּוֹץ לְהִתְעוֹרֵר מֵעַצְמָהּ.
ס
ח ק֣וֹל דּוֹדִ֔י הִנֵּה־זֶ֖ה בָּ֑א מְדַלֵּג֙ עַל־הֶ֣הָרִ֔ים מְקַפֵּ֖ץ עַל־הַגְּבָעֽוֹת׃ אֲנִי שׁוֹמַעַת אֶת קוֹל בּוֹאוֹ שֶׁל אֲהוּבִי, הוּא מְדַלֵּג עַל הָרִים, מְקַפֵּץ מֵעַל גְּבָעוֹת.
ט דּוֹמֶ֤ה דוֹדִי֙ לִצְבִ֔י א֖וֹ לְעֹ֣פֶר הָֽאַיָּלִ֑ים הִנֵּה־זֶ֤ה עוֹמֵד֙ אַחַ֣ר כׇּתְלֵ֔נוּ מַשְׁגִּ֙יחַ֙ מִן־הַֽחַלֹּנ֔וֹת מֵצִ֖יץ מִן־הַחֲרַכִּֽים׃ אֲהוּבִי דּוֹמֶה לִצְבִי אוֹ לְעוֹפֶר בֶּן אַיָּלָה, הִנֵּה הוּא כְּבָר עוֹמֵד מֵאֲחוֹרֵי הַקִּיר, מַשְׁגִּיחַ עָלַי מִן הַחַלּוֹנוֹת, מֵצִיץ מִן הַחֲרַכִּים.
י עָנָ֥ה דוֹדִ֖י וְאָ֣מַר לִ֑י ק֥וּמִי לָ֛ךְ רַעְיָתִ֥י יָפָתִ֖י וּלְכִי־לָֽךְ׃ עָנָה לִי אֲהוּבִי וְאָמַר: קוּמִי לָךְ רַעְיָתִי יָפָתִי וּבוֹאִי!
יא כִּֽי־הִנֵּ֥ה הַסְּתָ֖ו עָבָ֑ר הַגֶּ֕שֶׁם חָלַ֖ף הָלַ֥ךְ לֽוֹ׃ כִּי הִנֵּה הַסְּתָיו עָבָר, הַגֶּשֶׁם חָלַף וְהָלַךְ לוֹ.
יב הַנִּצָּנִים֙ נִרְא֣וּ בָאָ֔רֶץ עֵ֥ת הַזָּמִ֖יר הִגִּ֑יעַ וְק֥וֹל הַתּ֖וֹר נִשְׁמַ֥ע בְּאַרְצֵֽנוּ׃ בָּאָרֶץ כְּבָר נִרְאִים נִצָּנִים, עוֹנַת הַזָּמִיר הִגִּיעָה וְקוֹל הַיּוֹנִים נִשְׁמַע בְּאַרְצֵנוּ.

ד כְּמִגְדַּ֤ל דָּוִיד֙ צַוָּארֵ֔ךְ בָּנ֖וּי לְתַלְפִּיּ֑וֹת אֶ֤לֶף הַמָּגֵן֙ תָּל֣וּי עָלָ֔יו כֹּ֖ל שִׁלְטֵ֥י הַגִּבֹּרִֽים׃ צַוָּארֵךְ מְקֻשָּׁט כְּמִגְדַּל דָּוִד שֶׁבָּנוּי לְמוֹפֵת, תְּלוּיִים עָלָיו אֶלֶף מְגִנִּים וְכָל נַרְתִּיקֵי חִצֵּי הַגִּבּוֹרִים.
ה שְׁנֵ֥י שָׁדַ֛יִךְ כִּשְׁנֵ֥י עֳפָרִ֖ים תְּאוֹמֵ֣י צְבִיָּ֑ה הָרוֹעִ֖ים בַּשּׁוֹשַׁנִּֽים׃ שְׁנֵי שָׁדַיִךְ כִּשְׁנֵי עוֹפָרִים תְּאוֹמִים בְּנֵי צְבִיָּה שֶׁרוֹעִים בֵּין הַשּׁוֹשַׁנִּים.
ו עַ֤ד שֶׁיָּפ֙וּחַ֙ הַיּ֔וֹם וְנָ֖סוּ הַצְּלָלִ֑ים אֵ֤לֶךְ לִי֙ אֶל־הַ֣ר הַמּ֔וֹר וְאֶל־גִּבְעַ֖ת הַלְּבוֹנָֽה׃ עַד שֶׁתִּשְׁקַע הַשֶּׁמֶשׁ וְיִסְתַּלְּקוּ הַצְּלָלִים הָאֲרֻכִּים, אֲטַיֵּל לִי אִתָּךְ אֶל הַר בּוֹשֶׂם הַמּוֹר – אֶל גִּבְעַת הַלְּבוֹנָה.
ז כֻּלָּ֤ךְ יָפָה֙ רַעְיָתִ֔י וּמ֖וּם אֵ֥ין בָּֽךְ׃ כֻּלָּךְ יָפָה רַעְיָתִי, אֵין בָּךְ שׁוּם פְּגָם.
ס
ח אִתִּ֤י מִלְּבָנוֹן֙ כַּלָּ֔ה אִתִּ֖י מִלְּבָנ֣וֹן תָּב֑וֹאִי תָּשׁ֣וּרִי ׀ מֵרֹ֣אשׁ אֲמָנָ֗ה מֵרֹ֤אשׁ שְׂנִיר֙ וְחֶרְמ֔וֹן מִמְּעֹנ֣וֹת אֲרָי֔וֹת מֵֽהַרְרֵ֖י נְמֵרִֽים׃ יַחַד אִתִּי מֵהַלְּבָנוֹן כַּלָּתִי, בּוֹאִי יַחַד אִתִּי מֵהַלְּבָנוֹן, הַבִּיטִי מֵרֹאשׁ הָרֵי אֲמָנָה, מֵרֹאשׁ פִּסְגַת הַחֶרְמוֹן הַמֻּשְׁלֶגֶת, מִמְּעוֹנוֹת הָאֲרָיוֹת מֵהַָרֵי הַנְּמֵרִים.
ט לִבַּבְתִּ֖נִי אֲחֹתִ֣י כַלָּ֑ה לִבַּבְתִּ֙נִי֙ באחד בְּאַחַ֣ת מֵעֵינַ֔יִךְ בְּאַחַ֥ד עֲנָ֖ק מִצַּוְּרֹנָֽיִךְ׃ כָּבַשְׁתְּ אֶת לִבִּי אֲחוֹתִי כַּלָּתִי, כָּבַשְׁתְּ אֶת לִבִּי כְּבָר בְּאַחַת מֵעֵינַיִךְ, בְּאַחַת מִשַּׁרְשְׁרוֹת צַוָּארֵךְ.
י מַה־יָּפ֥וּ דֹדַ֖יִךְ אֲחֹתִ֣י כַלָּ֑ה מַה־טֹּ֤בוּ דֹדַ֙יִךְ֙ מִיַּ֔יִן וְרֵ֥יחַ שְׁמָנַ֖יִךְ מִכׇּל־בְּשָׂמִֽים׃ מַה יָּפָה אַהֲבָתֵךְ אֲחוֹתִי כַּלָּתִי, כֹּה טוֹבָה אַהֲבָתֵךְ יוֹתֵר מִיַּיִן וְרֵיחַ הַשֶּׁמֶן שֶׁלָּךְ מְבֻשָּׂם יוֹתֵר מִכָּל הַבְּשָׂמִים.
יא נֹ֛פֶת תִּטֹּ֥פְנָה שִׂפְתוֹתַ֖יִךְ כַּלָּ֑ה דְּבַ֤שׁ וְחָלָב֙ תַּ֣חַת לְשׁוֹנֵ֔ךְ וְרֵ֥יחַ שַׂלְמֹתַ֖יִךְ כְּרֵ֥יחַ לְבָנֽוֹן׃ שִׂפְתוֹתַיִךְ נוֹטְפוֹת מְתִיקוּת כַּלָּתִי, דְּבַשׁ וְחָלָב שׁוֹפְעִים מִלְּשׁוֹנֵךְ, וְרֵיחַ שִׂמְלוֹתַיִךְ כְּרֵיחַ הַלְּבָנוֹן.
יב גַּ֥ן ׀ נָע֖וּל אֲחֹתִ֣י כַלָּ֑ה גַּ֥ל נָע֖וּל מַעְיָ֥ן חָתֽוּם׃ אֲחוֹתִי כַּלָּתִי הִיא כְּמוֹ גַּן נָעוּל, מְקוֹר מַיִם נָעוּל – מַעְיָן שָׁמוּר.
יג שְׁלָחַ֙יִךְ֙ פַּרְדֵּ֣ס רִמּוֹנִ֔ים עִ֖ם פְּרִ֣י מְגָדִ֑ים כְּפָרִ֖ים עִם־נְרָדִֽים׃ נְטִיעוֹתַיִךְ פַּרְדֵּס רִמּוֹנִים עִם פֵּירוֹת מְתוּקִים, וּבְשָׂמִים שֶׁל כּוֹפֶר וְנֵרְדְּ.
יד נֵ֣רְדְּ ׀ וְכַרְכֹּ֗ם קָנֶה֙ וְקִנָּמ֔וֹן עִ֖ם כׇּל־עֲצֵ֣י לְבוֹנָ֑ה מֹ֚ר וַאֲהָל֔וֹת עִ֖ם כׇּל־רָאשֵׁ֥י בְשָׂמִֽים׃ נֵרְדְּ וְכַרְכּוֹם קָנֶה וְקִנָּמוֹן וְכָל עֲצֵי לְבוֹנָה, מוֹר וַאֲהָל וְכָל שׁוֹרְשֵׁי בְּשָׂמִים.
טו מַעְיַ֣ן גַּנִּ֔ים בְּאֵ֖ר מַ֣יִם חַיִּ֑ים וְנֹזְלִ֖ים מִן־לְבָנֽוֹן׃ מֻשְׁקִים בְּמַעְיַן שֶׁל גַּנִּים, בְּאֵר מַיִם חַיִּים, וְנוֹזְלִים מֵהַלְּבָנוֹן.
טז ע֤וּרִי צָפוֹן֙ וּב֣וֹאִי תֵימָ֔ן הָפִ֥יחִי גַנִּ֖י יִזְּל֣וּ בְשָׂמָ֑יו יָבֹ֤א דוֹדִי֙ לְגַנּ֔וֹ וְיֹאכַ֖ל פְּרִ֥י מְגָדָֽיו׃ הִתְעוֹרְרִי רוּחַ צָפוֹן וּבוֹאִי רוּחַ דָּרוֹם, נַשְּׁבִי בְּגַנִּי וְיַשְׁקֶה הַגֶּשֶׁם אֶת בְּשָׂמָיו, וְאָז יָבוֹא אֲהוּבִי לְגַנּוֹ וְיֹאכַל מִפֵּרוֹתָיו הַמְּתוּקִים.
ה א בָּ֣אתִי לְגַנִּי֮ אֲחֹתִ֣י כַלָּה֒ אָרִ֤יתִי מוֹרִי֙ עִם־בְּשָׂמִ֔י אָכַ֤לְתִּי יַעְרִי֙ עִם־דִּבְשִׁ֔י שָׁתִ֥יתִי יֵינִ֖י עִם־חֲלָבִ֑י אִכְל֣וּ רֵעִ֔ים שְׁת֥וּ וְשִׁכְר֖וּ דּוֹדִֽים׃ בָּאתִי לְגַנִּי אֲחוֹתִי הַכַּלָּה, לִקַּטְתִּי מוֹר וּשְׁאָר בּוֹשֶׂם, אָכַלְתִּי מִקְּנֵה הַסֻּכָּר וּמְתִיקוּתוֹ הַנִּשְׁפֶּכֶת, שָׁתִיתִי מֵהַיַּיִן הָאָדוֹם וְהַלָּבָן, אִכְלוּ יְדִידִים שְׁתוּ וְשִׁכְרוּ אֲהוּבִים.
ס
ב אֲנִ֥י יְשֵׁנָ֖ה וְלִבִּ֣י עֵ֑ר ק֣וֹל ׀ דּוֹדִ֣י דוֹפֵ֗ק פִּתְחִי־לִ֞י אֲחֹתִ֤י רַעְיָתִי֙ יוֹנָתִ֣י תַמָּתִ֔י שֶׁרֹּאשִׁי֙ נִמְלָא־טָ֔ל קְוֻצּוֹתַ֖י רְסִ֥יסֵי לָֽיְלָה׃ – אֲנִי יְשֵׁנָה אַךְ לִבִּי עֵר, וְהִנֵּה קוֹל אֲהוּבִי דּוֹפֵק בַּפֶּתַח: פִּתְחִי לִי אֲחוֹתִי רַעְיָתִי יוֹנָתִי תַּמָּתִי! רֹאשִׁי כְּבָר מָלֵא בְּטַל, תַּלְתַּלַּי נוֹטְפוֹת טִפּוֹת לַיְלָה!
ג פָּשַׁ֙טְתִּי֙ אֶת־כֻּתׇּנְתִּ֔י אֵיכָ֖כָה אֶלְבָּשֶׁ֑נָּה רָחַ֥צְתִּי אֶת־רַגְלַ֖י אֵיכָ֥כָה אֲטַנְּפֵֽם׃ – כְּבָר פָּשַׁטְתִּי אֶת הַכֻּתּוֹנֶת אֵיךְ אֶלְבַּשׁ אוֹתָהּ? כְּבָר רָחַצְתִּי אֶת רַגְלַי אֵיךְ אֲטַנֵּף אוֹתָם?
ד דּוֹדִ֗י שָׁלַ֤ח יָדוֹ֙ מִן־הַחֹ֔ר וּמֵעַ֖י הָמ֥וּ עָלָֽיו׃ אֲהוּבִי סִלֵּק אֶת יָדוֹ מֵחוֹר הַמַּנְעוּל לְהִסְתַּלֵּק, וּמִיַּד הִתְגַּעְגַּעְתִּי אֵלָיו.
ה קַ֥מְתִּֽי אֲנִ֖י לִפְתֹּ֣חַ לְדוֹדִ֑י וְיָדַ֣י נָֽטְפוּ־מ֗וֹר וְאֶצְבְּעֹתַי֙ מ֣וֹר עֹבֵ֔ר עַ֖ל כַּפּ֥וֹת הַמַּנְעֽוּל׃ מִיַּד קַמְתִּי לִפְתּוֹחַ לַאֲהוּבִי וְעַל יָדַי וְאֶצְבְּעוֹתַי לִקְרָאתוֹ בּוֹשֶׂם מוֹר שֶׁנָּטַף עַל יָדִית הַמַּנְעוּל.
ו פָּתַ֤חְתִּֽי אֲנִי֙ לְדוֹדִ֔י וְדוֹדִ֖י חָמַ֣ק עָבָ֑ר נַפְשִׁי֙ יָֽצְאָ֣ה בְדַבְּר֔וֹ בִּקַּשְׁתִּ֙יהוּ֙ וְלֹ֣א מְצָאתִ֔יהוּ קְרָאתִ֖יו וְלֹ֥א עָנָֽנִי׃ פָּתַחְתִּי לַאֲהוּבִי אַךְ אֲהוּבִי כְּבָר חָמַק וְעָבַר, נַפְשִׁי יָצְאָה כְּשֶׁאָמַר שֶׁהוּא עוֹזֵב, חִפַּשְׂתִּי אוֹתוֹ וְלֹא מָצָאתִי, קָרָאתִי וְהוּא לֹא עָנָה לִי.
ז מְצָאֻ֧נִי הַשֹּׁמְרִ֛ים הַסֹּבְבִ֥ים בָּעִ֖יר הִכּ֣וּנִי פְצָע֑וּנִי נָשְׂא֤וּ אֶת־רְדִידִי֙ מֵֽעָלַ֔י שֹׁמְרֵ֖י הַחֹמֽוֹת׃ מָצְאוּ אוֹתִי הַשּׁוֹמְרִים שֶׁמִּסְתּוֹבְבִים בָּעִיר הִכּוּ וּפָצְעוּ אוֹתִי, שׁוֹמְרֵי הַחוֹמוֹת לָקְחוּ אֶת צְעִיפִי מֵעָלַי.
ח הִשְׁבַּ֥עְתִּי אֶתְכֶ֖ם בְּנ֣וֹת יְרוּשָׁלָ֑͏ִם אִֽם־תִּמְצְאוּ֙ אֶת־דּוֹדִ֔י מַה־תַּגִּ֣ידוּ ל֔וֹ שֶׁחוֹלַ֥ת אַהֲבָ֖ה אָֽנִי׃ מַשְׁבִּיעָה אֲנִי אֶתְכֶם בְּנֵי יִשְׂרָאֵל, אִם תִּמְצְאוּ אֶת אֲהוּבִי מַה תַּגִּידוּ לוֹ? – שֶׁאֲנִי חוֹלַת אַהֲבָה!
ט מַה־דּוֹדֵ֣ךְ מִדּ֔וֹד הַיָּפָ֖ה בַּנָּשִׁ֑ים מַה־דּוֹדֵ֣ךְ מִדּ֔וֹד שֶׁכָּ֖כָה הִשְׁבַּעְתָּֽנוּ׃ – מַה מְּיוּחָד בַּאֲהוּבֵךְ מִשְּׁאָר הָאֲהוּבִים, הַיָּפָה בַּנָּשִׁים? מַה מְּיוּחָד בַּאֲהוּבֵךְ מִשְּׁאָר הָאֲהוּבִים שֶׁכָּכָה אַתְּ מַשְׁבִּיעָה אוֹתָנוּ?
י דּוֹדִ֥י צַח֙ וְאָד֔וֹם דָּג֖וּל מֵרְבָבָֽה׃ – אֲהוּבִי בָּהִיר וּפָנָיו וְרֻדּוֹת, אֵין אֶחָד מֵרְבָבָה כָּמוֹהוּ.
יא רֹאשׁ֖וֹ כֶּ֣תֶם פָּ֑ז קְוֻצּוֹתָיו֙ תַּלְתַּלִּ֔ים שְׁחֹר֖וֹת כָּעוֹרֵֽב׃ רֹאשׁוֹ אוֹצַר זָהָב, שְׂעָרוֹתָיו תַּלְתַּלִּים שְׁחוֹרוֹת כְּעוֹרֵב.
יב עֵינָ֕יו כְּיוֹנִ֖ים עַל־אֲפִ֣יקֵי מָ֑יִם רֹֽחֲצוֹת֙ בֶּֽחָלָ֔ב יֹשְׁב֖וֹת עַל־מִלֵּֽאת׃ עֵינָיו כְּמוֹ יוֹנִים עַל מַעְיְנוֹת מַיִם, רוֹחֲצוֹת בְּחָלָב, מְמֻקָּמוֹת בִּשְׁלֵמוּת.
יג לְחָיָו֙ כַּעֲרוּגַ֣ת הַבֹּ֔שֶׂם מִגְדְּל֖וֹת מֶרְקָחִ֑ים שִׂפְתוֹתָיו֙ שֽׁוֹשַׁנִּ֔ים נֹטְפ֖וֹת מ֥וֹר עֹבֵֽר׃ לְחָיָיו כַּעֲרוּגַת בּוֹשֶׂם שֶׁגְּדֵלִים בָּהּ תַּמְרוּקִים, שִׂפְתוֹתָיו כְּמוֹ שׁוֹשַׁנִּים שֶׁנּוֹטְפוֹת מִבּוֹשֶׂם הַמּוֹר.
יד יָדָיו֙ גְּלִילֵ֣י זָהָ֔ב מְמֻלָּאִ֖ים בַּתַּרְשִׁ֑ישׁ מֵעָיו֙ עֶ֣שֶׁת שֵׁ֔ן מְעֻלֶּ֖פֶת סַפִּירִֽים׃ יָדָיו כְּמוֹ מְגִלּוֹת שֶׁל זָהָב מְמֻלָּאוֹת בְּאַבְנֵי תַּרְשִׁישׁ, מֵעָיו כְּמוֹ שֶׁנְהָב מְכֻסֶּה בְּאַבְנֵי סַפִּיר.
טו שׁוֹקָיו֙ עַמּ֣וּדֵי שֵׁ֔שׁ מְיֻסָּדִ֖ים עַל־אַדְנֵי־פָ֑ז מַרְאֵ֙הוּ֙ כַּלְּבָנ֔וֹן בָּח֖וּר כָּאֲרָזִֽים׃ שׁוֹקָיו כְּמוֹ עַמּוּדֵי שַׁיִשׁ עַל יְסוֹדוֹת אַדְנֵי זָהָב, מַרְאֵהוּ כְּאֶרֶז מֻבְחָר בַּלְּבָנוֹן.
טז חִכּוֹ֙ מַֽמְתַקִּ֔ים וְכֻלּ֖וֹ מַחֲמַדִּ֑ים זֶ֤ה דוֹדִי֙ וְזֶ֣ה רֵעִ֔י בְּנ֖וֹת יְרוּשָׁלָֽ͏ִם׃ דִּבְרֵי פִּיו מַמְתַּקִּים, וְכֻלּוֹ מַחֲמַדִּים, זֶה אֲהוּבִי וְזֶה יְדִידִי, בְּנֵי יִשְׂרָאֵל.
ו א אָ֚נָה הָלַ֣ךְ דּוֹדֵ֔ךְ הַיָּפָ֖ה בַּנָּשִׁ֑ים אָ֚נָה פָּנָ֣ה דוֹדֵ֔ךְ וּנְבַקְשֶׁ֖נּוּ עִמָּֽךְ׃ – לְאָן הָלַךְ אֲהוּבֵךְ, הַיָּפָה בַּנָּשִׁים? לְאָן פָּנָה אֲהוּבֵךְ וּנְחַפֵּשׂ אוֹתוֹ אִתָּךְ?
ב דּוֹדִי֙ יָרַ֣ד לְגַנּ֔וֹ לַעֲרֻג֖וֹת הַבֹּ֑שֶׂם לִרְעוֹת֙ בַּגַּנִּ֔ים וְלִלְקֹ֖ט שֽׁוֹשַׁנִּֽים׃ – אֲהוּבִי יָרַד לְגַנּוֹ לַעֲרוּגוֹת הַבְּשָׂמִים, לְטַיֵּיל בַּגַּנִּים וּלְלַקֵּט שׁוֹשַׁנִּים.
ג אֲנִ֤י לְדוֹדִי֙ וְדוֹדִ֣י לִ֔י הָרֹעֶ֖ה בַּשּׁוֹשַׁנִּֽים׃ אֲנִי מְיוֹעֶדֶת לַאֲהוּבִי וַאֲהוּבִי מְיוֹעָד לִי, הוּא הַמְּטַיֵּל בֵּין הַשּׁוֹשַׁנִּים.
ס
ד יָפָ֨ה אַ֤תְּ רַעְיָתִי֙ כְּתִרְצָ֔ה נָאוָ֖ה כִּירוּשָׁלָ֑͏ִם אֲיֻמָּ֖ה כַּנִּדְגָּלֽוֹת׃ רַעְיָתִי, אַתְּ יָפָה כְּעִיר הַבִּירָה, נָאָה כִּירוּשָׁלַיִם, חֲזָקָה כִּצְבָא גְּדוּדִים.
ה הָסֵ֤בִּי עֵינַ֙יִךְ֙ מִנֶּגְדִּ֔י שֶׁ֥הֵ֖ם הִרְהִיבֻ֑נִי שַׂעְרֵךְ֙ כְּעֵ֣דֶר הָֽעִזִּ֔ים שֶׁגָּלְשׁ֖וּ מִן־הַגִּלְעָֽד׃ הַפְנִי אֵלַי אֶת מַבָּטֵךְ הוּא מְרוֹמֵם אוֹתִי, שַׂעְרֵךְ דּוֹמֶה לְעֵדֶר עִזִּים שֶׁגּוֹלְשׁוֹת מֵהַגִּלְעָד.
ו שִׁנַּ֙יִךְ֙ כְּעֵ֣דֶר הָֽרְחֵלִ֔ים שֶׁעָל֖וּ מִן־הָרַחְצָ֑ה שֶׁכֻּלָּם֙ מַתְאִימ֔וֹת וְשַׁכֻּלָ֖ה אֵ֥ין בָּהֶֽם׃ שִׁנַּיִךְ דּוֹמוֹת לְעֵדֶר כְּבָשִׂים שֶׁעָלוּ מִן הָרַחְצָה, כֻּלָּם תּוֹאֲמוֹת וְאֵין בָּהֶם פְּגָם.
ז כְּפֶ֤לַח הָרִמּוֹן֙ רַקָּתֵ֔ךְ מִבַּ֖עַד לְצַמָּתֵֽךְ׃ רַקָּתֵךְ מִבַּעַד לְצַמָּתֵךְ דּוֹמָה לְפֶלַח רִמּוֹן.
ח שִׁשִּׁ֥ים הֵ֙מָּה֙ מְלָכ֔וֹת וּשְׁמֹנִ֖ים פִּֽילַגְשִׁ֑ים וַעֲלָמ֖וֹת אֵ֥ין מִסְפָּֽר׃ שִׁשִּׁים מַלְכּוֹת יֵשׁ לִי וּשְׁמוֹנִים פִּילַגְשִׁים וַעֲלָמוֹת לְלֹא מִסְפָּר.
ט אַחַ֥ת הִיא֙ יוֹנָתִ֣י תַמָּתִ֔י אַחַ֥ת הִיא֙ לְאִמָּ֔הּ בָּרָ֥ה הִ֖יא לְיֽוֹלַדְתָּ֑הּ רָא֤וּהָ בָנוֹת֙ וַֽיְאַשְּׁר֔וּהָ מְלָכ֥וֹת וּפִֽילַגְשִׁ֖ים וַֽיְהַלְלֽוּהָ׃ אַךְ יוֹנָתִי הַמֻּשְׁלֶמֶת הִיא יְחִידָה, הִיא יְחִידָה וּמְיוּחֶדֶת לָאֵם שֶׁיָּלְדָה אוֹתָהּ, כָּל הַבָּנוֹת רָאוּ אוֹתָהּ וְאִשְּׁרוּ, וְהַמְּלָכוֹת וְהַפִּילַגְשִׁים הִלְלוּ אוֹתָהּ.
ס
י מִי־זֹ֥את הַנִּשְׁקָפָ֖ה כְּמוֹ־שָׁ֑חַר יָפָ֣ה כַלְּבָנָ֗ה בָּרָה֙ כַּֽחַמָּ֔ה אֲיֻמָּ֖ה כַּנִּדְגָּלֽוֹת׃ מִי זֹאת שֶׁנִּרְאֵית כְּמוֹ הַשַּׁחַר, יָפָה כַּלְּבָנָה מְיוּחֶדֶת כַּחַמָּה, חֲזָקָה כִּצְבָא גְּדוּדִים!
ס
יא אֶל־גִּנַּ֤ת אֱגוֹז֙ יָרַ֔דְתִּי לִרְא֖וֹת בְּאִבֵּ֣י הַנָּ֑חַל לִרְאוֹת֙ הֲפָֽרְחָ֣ה הַגֶּ֔פֶן הֵנֵ֖צוּ הָרִמֹּנִֽים׃ יָרַדְתִּי אֶל גִּנַּת אֱגוֹז לִרְאוֹת אֶת אֲבִיב פֵּרוֹת הַנַּחַל, לִרְאוֹת הַאִם הַגֶּפֶן פָּרְחָה, אִם צָמְחוּ נִצָּנֵי הָרִמּוֹנִים.
יב לֹ֣א יָדַ֔עְתִּי נַפְשִׁ֣י שָׂמַ֔תְנִי מַרְכְּב֖וֹת עַמִּ֥י נָדִֽיב׃ – לֹא יָדַעְתִּי מִכָּךְ, הָיִיתִי חָפֵצָה לִהְיוֹת בְּמֶרְכְּבוֹת עַמִּי הָאוֹהֵב!
ז א שׁ֤וּבִי שׁ֙וּבִי֙ הַשּׁ֣וּלַמִּ֔ית שׁ֥וּבִי שׁ֖וּבִי וְנֶחֱזֶה־בָּ֑ךְ מַֽה־תֶּחֱזוּ֙ בַּשּׁ֣וּלַמִּ֔ית כִּמְחֹלַ֖ת הַֽמַּחֲנָֽיִם׃ – שׁוּבִי שׁוּבִי יְרוּשָׁלַיִם הַבְּנוּיָה, שׁוּבִי שׁוּבִי וְנַבִּיט בָּךְ, מַה תַּבִּיטוּ בִּירוּשָׁלַיִם הַבְּנוּיָה בְּהִתְכַּנְּסוּת הַמַּחֲנוֹת!
ב מַה־יָּפ֧וּ פְעָמַ֛יִךְ בַּנְּעָלִ֖ים בַּת־נָדִ֑יב חַמּוּקֵ֣י יְרֵכַ֔יִךְ כְּמ֣וֹ חֲלָאִ֔ים מַעֲשֵׂ֖ה יְדֵ֥י אׇמָּֽן׃ מַה יָּפוּ צְעָדַיִךְ בָּעֲלִיָּה לָרֶגֶל, בִּתּוֹ שֶׁל הָאֵל הַמֵּטִיב, סִתְרֵי יְרֵכַיִךְ דּוֹמִים לְתַכְשִׁיטִים מַעֲשֵׂה יְדֵי אָמָּן.
ג שׇׁרְרֵךְ֙ אַגַּ֣ן הַסַּ֔הַר אַל־יֶחְסַ֖ר הַמָּ֑זֶג בִּטְנֵךְ֙ עֲרֵמַ֣ת חִטִּ֔ים סוּגָ֖ה בַּשּׁוֹשַׁנִּֽים׃ טַבּוּרֵךְ כַּאֲגַרְטָל בְּצוּרַת סַהַר שֶׁאֵינוֹ חָסֵר יַיִן לִמְזִיגָה, בִּטְנֵךְ כַּעֲרֵמַת חִטִּים מֻקֶּפֶת שׁוֹשַׁנִּים.
ד שְׁנֵ֥י שָׁדַ֛יִךְ כִּשְׁנֵ֥י עֳפָרִ֖ים תׇּאֳמֵ֥י צְבִיָּֽה׃ שְׁנֵי שָׁדַיִךְ כִּשְׁנֵי עוֹפָרִים תְּאוֹמִים בְּנֵי צְבִיָּה.
ה צַוָּארֵ֖ךְ כְּמִגְדַּ֣ל הַשֵּׁ֑ן עֵינַ֜יִךְ בְּרֵכ֣וֹת בְּחֶשְׁבּ֗וֹן עַל־שַׁ֙עַר֙ בַּת־רַבִּ֔ים אַפֵּךְ֙ כְּמִגְדַּ֣ל הַלְּבָנ֔וֹן צוֹפֶ֖ה פְּנֵ֥י דַמָּֽשֶׂק׃ צַוָּארֵךְ כְּמִגְדַּל שֶׁנְהָב, עֵינַיִךְ צְלוּלוֹת כִּבְרֵכוֹת הַמַּיִם שֶׁעַל יַד הַשַּׁעַר הָרָאשִׁי שֶׁל הָעִיר חֶשְׁבּוֹן, אַפֵּךְ כַּמִּגְדַּל בִּלְבָנוֹן שֶׁצּוֹפֶה לְעֵבֶר דַּמֶּשֶׂק.
ו רֹאשֵׁ֤ךְ עָלַ֙יִךְ֙ כַּכַּרְמֶ֔ל וְדַלַּ֥ת רֹאשֵׁ֖ךְ כָּאַרְגָּמָ֑ן מֶ֖לֶךְ אָס֥וּר בָּרְהָטִֽים׃ רֹאשֵׁךְ שֶׁעָלַיִךְ דּוֹמֶה לְהַר הַכַּרְמֶל, וְצַמּוֹתַיִךְ הַתְּלוּיוֹת כְּחוּטֵי אַרְגָּמָן, הַמֶּלֶךְ קָשׁוּר בְּאַהֲבָה בְּשַׂעֲרוֹתַיִךְ הַגּוֹלְשׁוֹת.
ז מַה־יָּפִית֙ וּמַה־נָּעַ֔מְתְּ אַהֲבָ֖ה בַּתַּֽעֲנוּגִֽים׃ כֹּה יָפָה וְכֹה נְעִימָה אַתְּ, אַהֲבָה עֲנֻגָּה.
ח זֹ֤את קֽוֹמָתֵךְ֙ דָּֽמְתָ֣ה לְתָמָ֔ר וְשָׁדַ֖יִךְ לְאַשְׁכֹּלֽוֹת׃ קוֹמָתֵךְ הַזְּקוּפָה דּוֹמָה לְעֵץ תָּמָר וְשָׁדַיִךְ לְאֶשְׁכּוֹלוֹת הַתְּמָרִים.
ט אָמַ֙רְתִּי֙ אֶעֱלֶ֣ה בְתָמָ֔ר אֹֽחֲזָ֖ה בְּסַנְסִנָּ֑יו וְיִֽהְיוּ־נָ֤א שָׁדַ֙יִךְ֙ כְּאֶשְׁכְּל֣וֹת הַגֶּ֔פֶן וְרֵ֥יחַ אַפֵּ֖ךְ כַּתַּפּוּחִֽים׃ אָמַרְתִּי, אֶעֱלֶה עַל עֵץ הַתָּמָר, אוֹחַז בַּעֲנָפָיו, וְיִהְיוּ עַכְשָׁיו שָׁדַיִךְ כְּאֶשְׁכְּלוֹת עֲנָבִים, וְרֵיחַ נְשִׁימָתֵךְ כְּרֵיחַ תַּפּוּחִים.
י וְחִכֵּ֕ךְ כְּיֵ֥ין הַטּ֛וֹב הוֹלֵ֥ךְ לְדוֹדִ֖י לְמֵישָׁרִ֑ים דּוֹבֵ֖ב שִׂפְתֵ֥י יְשֵׁנִֽים׃ וְטַעֲמֵךְ כְּיַיִן טוֹב עֲבוּר אַהֲבָה אֲמִתִּית, שֶׁגּוֹרֵם גַּם לִישֵׁנִים לְשׂוֹחֵחַ.
יא אֲנִ֣י לְדוֹדִ֔י וְעָלַ֖י תְּשׁוּקָתֽוֹ׃ – אֲנִי מְיוֹעֶדֶת לַאֲהוּבִי וּתְשׁוּקָתוֹ הִיא בִּי.
ס
יב לְכָ֤ה דוֹדִי֙ נֵצֵ֣א הַשָּׂדֶ֔ה נָלִ֖ינָה בַּכְּפָרִֽים׃ בֹּא אֲהוּבִי נֵצֵא לַשָּׂדֶה, נָלוּן בֵּין בִּשְׂמֵי הַכֹּפֶר.
יג נַשְׁכִּ֙ימָה֙ לַכְּרָמִ֔ים נִרְאֶ֞ה אִם־פָּֽרְחָ֤ה הַגֶּ֙פֶן֙ פִּתַּ֣ח הַסְּמָדַ֔ר הֵנֵ֖צוּ הָרִמּוֹנִ֑ים שָׁ֛ם אֶתֵּ֥ן אֶת־דֹּדַ֖י לָֽךְ׃ נָקוּם הַשְׁכֵּם בַּבּוֹקֶר וְנֵצֵא לַכְּרָמִים, נִרְאֶה אִם פָּרְחוּ פִּרְחֵי הַגֶּפֶן, אִם נִפְתְּחוּ עִנְבֵי הַסְּמָדַר הַזְּעִירִים, אִם צָמְחוּ נִצָּנֵי הָרִמּוֹנִים, שָׁם אַעֲנִיק לְךָ אֶת אַהֲבָתִי.
יד הַֽדּוּדָאִ֣ים נָֽתְנוּ־רֵ֗יחַ וְעַל־פְּתָחֵ֙ינוּ֙ כׇּל־מְגָדִ֔ים חֲדָשִׁ֖ים גַּם־יְשָׁנִ֑ים דּוֹדִ֖י צָפַ֥נְתִּי לָֽךְ׃ פִּרְחֵי הַיַּסְמִין הִדִּיפוּ רֵיחַ, לְפִתְחֵנוּ מְצוּיִים כָּל מִינֵי פֵּרוֹת מְתוּקִים טְרִיִּים וִיבֵשִׁים, שָׁמַרְתִּי לְךָ מֵהֶם, אֲהוּבִי!
ח א מִ֤י יִתֶּנְךָ֙ כְּאָ֣ח לִ֔י יוֹנֵ֖ק שְׁדֵ֣י אִמִּ֑י אֶֽמְצָאֲךָ֤ בַחוּץ֙ אֶשָּׁ֣קְךָ֔ גַּ֖ם לֹא־יָבֻ֥זוּ לִֽי׃ מִי יִתֵּן שֶׁתִּהְיֶה כְּאָחִי שֶׁיָּנַק מֵהַשָּׁדַיִם שֶׁל אִמִּי, וּכְשֶׁאֶפְגּוֹשׁ אוֹתְךָ בַּחוּץ אֲנַשֵּׁק אוֹתְךָ וְלֹא יָבוּזוּ לִי.
ב אֶנְהָֽגְﬞךָ֗ אֲבִֽיאֲךָ֛ אֶל־בֵּ֥ית אִמִּ֖י תְּלַמְּדֵ֑נִי אַשְׁקְךָ֙ מִיַּ֣יִן הָרֶ֔קַח מֵעֲסִ֖יס רִמֹּנִֽי׃ וְאוֹבִיל וַאֲבִיא אוֹתְךָ לְבֵית אִמִּי כְּהֶרְגְּלֵנוּ, וְאֶמְזוֹג לְךָ מֵהַיַּיִן הַמְּבֻשָּׂם וּמִמִּיץ הָרִמּוֹן.
ג שְׂמֹאלוֹ֙ תַּ֣חַת רֹאשִׁ֔י וִֽימִינ֖וֹ תְּחַבְּקֵֽנִי׃ יַד שְׂמֹאלוֹ שֶׁל אֲהוּבִי תִּתְמוֹךְ בְּרֹאשִׁי וְיַד יְמִינוֹ תְּחַבֵּק אוֹתִי.
ד הִשְׁבַּ֥עְתִּי אֶתְכֶ֖ם בְּנ֣וֹת יְרוּשָׁלָ֑͏ִם מַה־תָּעִ֧ירוּ ׀ וּֽמַה־תְּעֹ֥רְר֛וּ אֶת־הָאַהֲבָ֖ה עַ֥ד שֶׁתֶּחְפָּֽץ׃ מַשְׁבִּיעָה אֲנִי אֶתְכֶם בְּנֵי יִשְׂרָאֵל, אַל תָּעִירוּ וְאַל תְּעוֹרְרוּ אֶת הָאַהֲבָה עַד שֶׁתַּחְפּוֹץ לְהִתְעוֹרֵר מֵעַצְמָה.
ס
ה מִ֣י זֹ֗את עֹלָה֙ מִן־הַמִּדְבָּ֔ר מִתְרַפֶּ֖קֶת עַל־דּוֹדָ֑הּ תַּ֤חַת הַתַּפּ֙וּחַ֙ עֽוֹרַרְתִּ֔יךָ שָׁ֚מָּה חִבְּלַ֣תְךָ אִמֶּ֔ךָ שָׁ֖מָּה חִבְּלָ֥ה יְלָדַֽתְךָ׃ רְאוּ אֶת אֲהוּבָתִי – מִי זֹאת שֶׁעָלְתָה מִמִּדְבַּר סִינַי מִתְרַפֶּקֶת עַל אֲהוּבָהּ! תַּחַת עֵץ הַתַּפּוּחַ עוֹרַרְתִּי אֶת אַהֲבָתְךָ עַם יִשְׂרָאֵל, שָׁמָּה הָיוּ חֶבְלֵי הַלֵּדָה שֶׁל אִמְּךָ, שָׁמָּה יָלְדָה אוֹתְךָ.
ו שִׂימֵ֨נִי כַֽחוֹתָ֜ם עַל־לִבֶּ֗ךָ כַּֽחוֹתָם֙ עַל־זְרוֹעֶ֔ךָ כִּֽי־עַזָּ֤ה כַמָּ֙וֶת֙ אַהֲבָ֔ה קָשָׁ֥ה כִשְׁא֖וֹל קִנְאָ֑ה רְשָׁפֶ֕יהָ רִשְׁפֵּ֕י אֵ֖שׁ שַׁלְהֶ֥בֶתְיָֽה׃ שִׂים אוֹתִי כְּטַבַּעַת עַל לִבְּךָ – כְּטַבַּעַת עַל זְרוֹעֲךָ, כִּי הָאַהֲבָה עַזָּה כַּמָּוֶת – קִנְאָתְךָ לִי קָשָׁה כְּמוֹ הַשְּׁאוֹל, לוֹהֶטֶת כְּשַׁלְהֶבֶת אֵשׁ.
ז מַ֣יִם רַבִּ֗ים לֹ֤א יֽוּכְלוּ֙ לְכַבּ֣וֹת אֶת־הָֽאַהֲבָ֔ה וּנְהָר֖וֹת לֹ֣א יִשְׁטְפ֑וּהָ אִם־יִתֵּ֨ן אִ֜ישׁ אֶת־כׇּל־ה֤וֹן בֵּיתוֹ֙ בָּאַהֲבָ֔ה בּ֖וֹז יָב֥וּזוּ לֽוֹ׃ גַּם מַיִם רַבִּים לֹא יוּכְלוּ לְכַבּוֹת אֶת הָאַהֲבָה – נְהָרוֹת לֹא יִשְׁטְפוּ אוֹתָהּ, אִם יִתֵּן אִישׁ אֶת כָּל הוֹן בֵּיתוֹ תְּמוּרַת אַהֲבָה זוֹ, יִלְעֲגוּ לוֹ.
ס
ח אָח֥וֹת לָ֙נוּ֙ קְטַנָּ֔ה וְשָׁדַ֖יִם אֵ֣ין לָ֑הּ מַֽה־נַּעֲשֶׂה֙ לַאֲחֹתֵ֔נוּ בַּיּ֖וֹם שֶׁיְּדֻבַּר־בָּֽהּ׃ אָמְרוּ עָלַי: יֵשׁ לָנוּ אָחוֹת צְעִירָה שֶׁטֶּרֶם צָמְחוּ שָׁדֶיהָ, מַה נַּעֲשֶׂה לַאֲחוֹתֵנוּ בַּיּוֹם שֶׁיְּבַקְּשׁוּ לָשֵׂאת אוֹתָהּ?
ט אִם־חוֹמָ֣ה הִ֔יא נִבְנֶ֥ה עָלֶ֖יהָ טִ֣ירַת כָּ֑סֶף וְאִם־דֶּ֣לֶת הִ֔יא נָצ֥וּר עָלֶ֖יהָ ל֥וּחַ אָֽרֶז׃ אִם הִיא חוֹמָה נִבְנֶה עָלֶיהָ טִירָה שֶׁל כֶּסֶף, וְאִם הִיא דֶּלֶת נַקִּיף אוֹתָהּ בְּלוּחַ אֶרֶז.
י אֲנִ֣י חוֹמָ֔ה וְשָׁדַ֖י כַּמִּגְדָּל֑וֹת אָ֛ז הָיִ֥יתִי בְעֵינָ֖יו כְּמוֹצְאֵ֥ת שָׁלֽוֹם׃ אֲנִי שְׁמוּרָה כְּמוֹ חוֹמָה וְשָׁדַי כְּבָר כְּמִגְדָּלִים, כָּךְ אֲנִי מֻשְׁלֶמֶת וּמוֹצֵאת חֵן בְּעֵינֵי אֲהוּבִי.
פ
יא כֶּ֣רֶם הָיָ֤ה לִשְׁלֹמֹה֙ בְּבַ֣עַל הָמ֔וֹן נָתַ֥ן אֶת־הַכֶּ֖רֶם לַנֹּטְרִ֑ים אִ֛ישׁ יָבִ֥א בְּפִרְי֖וֹ אֶ֥לֶף כָּֽסֶף׃ לַמֶּלֶךְ שְׁלֹמֹה הָיָה כֶּרֶם בְּבִקְעַת הַכְּרָמִים, הוּא נָתַן אֶת הַכֶּרֶם לַשּׁוֹמְרִים שֶׁכָּל אִישׁ יָבִיא תְּמוּרַת פֵּרוֹתָיו אֶלֶף שִׁקְלֵי כֶּסֶף.
יב כַּרְמִ֥י שֶׁלִּ֖י לְפָנָ֑י הָאֶ֤לֶף לְךָ֙ שְׁלֹמֹ֔ה וּמָאתַ֖יִם לְנֹטְרִ֥ים אֶת־פִּרְיֽוֹ׃ הֲשִׁיבוּ אֶת כַּרְמִי שֶׁלִּי לְפָנַי, קַח לְךָ שְׁלֹמֹה אֶת הָאֶלֶף, וְעוֹד מָאתַיִם לַשּׁוֹמְרִים אֶת פֵּרוֹתָיו!
יג הַיּוֹשֶׁ֣בֶת בַּגַּנִּ֗ים חֲבֵרִ֛ים מַקְשִׁיבִ֥ים לְקוֹלֵ֖ךְ הַשְׁמִיעִֽנִי׃ – אֲהוּבָתִי הַיּוֹשֶׁבֶת בַּגַּנִּים, חֲבֵרַי מַקְשִׁיבִים לִשְׁמוֹעַ אֶת קוֹלֵךְ, הַשְׁמִיעִי לִי אוֹתוֹ!
יד בְּרַ֣ח ׀ דּוֹדִ֗י וּֽדְמֵה־לְךָ֤ לִצְבִי֙ א֚וֹ לְעֹ֣פֶר הָֽאַיָּלִ֔ים עַ֖ל הָרֵ֥י בְשָׂמִֽים׃ – בְּרַח מֵהֶם אֲהוּבִי, הֱיֵה דּוֹמֶה לִצְבִי אוֹ לְעוֹפֶר בֶּן אַיָּלָה עַל הָרֵי בְּשָׂמִים.
