פ פָּרָשַׁת שְׁקָלִים
ל יא וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יב כִּ֣י תִשָּׂ֞א אֶת־רֹ֥אשׁ בְּנֵֽי־יִשְׂרָאֵל֮ לִפְקֻדֵיהֶם֒ וְנָ֨תְנ֜וּ אִ֣ישׁ כֹּ֧פֶר נַפְשׁ֛וֹ לַיהֹוָ֖ה בִּפְקֹ֣ד אֹתָ֑ם וְלֹא־יִהְיֶ֥ה בָהֶ֛ם נֶ֖גֶף בִּפְקֹ֥ד אֹתָֽם׃ כְּשֶׁתִּמְנֶה אֶת מִסְפַּר אַנְשֵׁי בְּנֵי יִשְׂרָאֵל, כָּל אִישׁ יִתֵּן כּוֹפֶר נֶפֶשׁ לַה' כְּשֶׁיִּסְפְּרוּ אוֹתָם, וְכָךְ לֹא תִּהְיֶה בָּהֶם מַגֵּפָה כְּשֶׁסּוֹפְרִים אוֹתָם.
יג זֶ֣ה ׀ יִתְּנ֗וּ כׇּל־הָעֹבֵר֙ עַל־הַפְּקֻדִ֔ים מַחֲצִ֥ית הַשֶּׁ֖קֶל בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ עֶשְׂרִ֤ים גֵּרָה֙ הַשֶּׁ֔קֶל מַחֲצִ֣ית הַשֶּׁ֔קֶל תְּרוּמָ֖ה לַֽיהֹוָֽה׃ אֶת זֶה יִתֵּן כָּל מִי שֶׁעוֹבֵר לְהִסָּפֵר – כֶּסֶף בְּמִשְׁקַל מַחֲצִית הַשֶּׁקֶל (כְּ7 גְּרַם) בְּשִׁקְלֵי הַקּוֹדֶשׁ, מִשְׁקַל הַשֶּׁקֶל הוּא עֶשְׂרִים גֵּרָה (14.1 גְּרַם) – יִתְּנוּ מַחֲצִית הַשֶּׁקֶל תְּרוּמָה לַה'.
יד כֹּ֗ל הָעֹבֵר֙ עַל־הַפְּקֻדִ֔ים מִבֶּ֛ן עֶשְׂרִ֥ים שָׁנָ֖ה וָמָ֑עְלָה יִתֵּ֖ן תְּרוּמַ֥ת יְהֹוָֽה׃ כָּל מִי שֶׁעוֹבֵר לְהִסָּפֵר – מִגִּיל עֶשְׂרִים שָׁנָה וָמַעְלָה – יִתֵּן אֶת תְּרוּמַת ה'.
טו הֶֽעָשִׁ֣יר לֹֽא־יַרְבֶּ֗ה וְהַדַּל֙ לֹ֣א יַמְעִ֔יט מִֽמַּחֲצִ֖ית הַשָּׁ֑קֶל לָתֵת֙ אֶת־תְּרוּמַ֣ת יְהֹוָ֔ה לְכַפֵּ֖ר עַל־נַפְשֹׁתֵיכֶֽם׃ הֶעָשִׁיר לֹא יַרְבֶּה וְהֶעָנִי לֹא יַמְעִיט מֵחֲצִי שֶׁקֶל – לָתֵת תְּרוּמָה לַה' לְכַפֵּר עַל נַפְשׁוֹתֵיכֶם.
טז וְלָקַחְתָּ֞ אֶת־כֶּ֣סֶף הַכִּפֻּרִ֗ים מֵאֵת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וְנָתַתָּ֣ אֹת֔וֹ עַל־עֲבֹדַ֖ת אֹ֣הֶל מוֹעֵ֑ד וְהָיָה֩ לִבְנֵ֨י יִשְׂרָאֵ֤ל לְזִכָּרוֹן֙ לִפְנֵ֣י יְהֹוָ֔ה לְכַפֵּ֖ר עַל־נַפְשֹׁתֵיכֶֽם׃ קַח מִבְּנֵי יִשְׂרָאֵל אֶת הַכֶּסֶף שֶׁמְּכַפֵּר וְתֵן אוֹתוֹ לְצוֹרֶךְ עֲבוֹדַת אוֹהֶל הַהִתְכַּנְּסוּת, וְהוּא יַזְכִּיר אֶת בְּנֵי יִשְׂרָאֵל לִפְנֵי ה' כְּדֵי לְכַפֵּר עַל נַפְשׁוֹתֵיכֶם.
פ פָּרָשַׁת הַכִּיּוֹר
יז וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יח וְעָשִׂ֜יתָ כִּיּ֥וֹר נְחֹ֛שֶׁת וְכַנּ֥וֹ נְחֹ֖שֶׁת לְרׇחְצָ֑ה וְנָתַתָּ֣ אֹת֗וֹ בֵּֽין־אֹ֤הֶל מוֹעֵד֙ וּבֵ֣ין הַמִּזְבֵּ֔חַ וְנָתַתָּ֥ שָׁ֖מָּה מָֽיִם׃ תַּעֲשֶׂה כִּיּוֹר מִנְּחוֹשֶׁת וּבְסִיסוֹ מִנְּחוֹשֶׁת לִרְחִיצָה, וְתָשִׂים אוֹתוֹ בֵּין אוֹהֶל הַהִתְכַּנְּסוּת וּבֵין הַמִּזְבֵּחַ, וְתָשִׂים שָׁם מַיִם.
יט וְרָחֲצ֛וּ אַהֲרֹ֥ן וּבָנָ֖יו מִמֶּ֑נּוּ אֶת־יְדֵיהֶ֖ם וְאֶת־רַגְלֵיהֶֽם׃ וְאַהֲרֹן וּבָנָיו יִרְחֲצוּ מִמֶּנּוּ אֶת יְדֵיהֶם וְאֶת רַגְלֵיהֶם.
כ בְּבֹאָ֞ם אֶל־אֹ֧הֶל מוֹעֵ֛ד יִרְחֲצוּ־מַ֖יִם וְלֹ֣א יָמֻ֑תוּ א֣וֹ בְגִשְׁתָּ֤ם אֶל־הַמִּזְבֵּ֙חַ֙ לְשָׁרֵ֔ת לְהַקְטִ֥יר אִשֶּׁ֖ה לַֽיהֹוָֽה׃ כְּשֶׁיִּכָּנְסוּ לְאוֹהֶל הַהִתְכַּנְּסוּת יִרְחֲצוּ בְּמַיִם וְלֹא יָמוּתוּ, אוֹ כְּשֶׁיִּגְּשׁוּ לַמִּזְבֵּחַ לְשָׁרֵת לְהַקְטִיר קָרְבָּן עוֹלֶה בָּאֵשׁ לַה'.
כא וְרָחֲצ֛וּ יְדֵיהֶ֥ם וְרַגְלֵיהֶ֖ם וְלֹ֣א יָמֻ֑תוּ וְהָיְתָ֨ה לָהֶ֧ם חׇק־עוֹלָ֛ם ל֥וֹ וּלְזַרְע֖וֹ לְדֹרֹתָֽם׃ הֵם יִרְחֲצוּ אֶת יְדֵיהֶם וְרַגְלֵיהֶם וְלֹא יָמוּתוּ וְיִהְיֶה לָהֶם חוֹק לְעוֹלָם לוֹ וּלְצֶאֱצָאָיו לְדוֹרוֹת.
פ פָּרָשַׁת שֶׁמֶן הַמִּשְׁחָה
כב וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
כג וְאַתָּ֣ה קַח־לְךָ֮ בְּשָׂמִ֣ים רֹאשׁ֒ מׇר־דְּרוֹר֙ חֲמֵ֣שׁ מֵא֔וֹת וְקִנְּמׇן־בֶּ֥שֶׂם מַחֲצִית֖וֹ חֲמִשִּׁ֣ים וּמָאתָ֑יִם וּקְנֵה־בֹ֖שֶׂם חֲמִשִּׁ֥ים וּמָאתָֽיִם׃ אַתָּה קַח לְךָ בְּשָׂמִים חֲשׁוּבִים: מוֹר-דְּרוֹר – בְּמִשְׁקַל חֲמֵשׁ מֵאוֹת שְׁקָלִים (כְּ7 קִילוֹ), וּבוֹשֶׂם קִנָּמוֹן – מַחֲצִיתוֹ בְּמִשְׁקַל מָאתַיִם וַחֲמִשִּׁים שְׁקָלִים (כְּ3.5 קִילוֹ), וְקָנִים שֶׁל בּוֹשֶׂם בְּמִשְׁקַל מָאתַיִם וַחֲמִשִּׁים שְׁקָלִים.
כד וְקִדָּ֕ה חֲמֵ֥שׁ מֵא֖וֹת בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ וְשֶׁ֥מֶן זַ֖יִת הִֽין׃ וְקִדָּה – בְּמִשְׁקַל חֲמֵשׁ מֵאוֹת שִׁקְלֵי קוֹדֶשׁ, וְשֶׁמֶן זַיִת – הִין (כְּ3.6 לִיטֶר).
כה וְעָשִׂ֣יתָ אֹת֗וֹ שֶׁ֚מֶן מִשְׁחַת־קֹ֔דֶשׁ רֹ֥קַח מִרְקַ֖חַת מַעֲשֵׂ֣ה רֹקֵ֑חַ שֶׁ֥מֶן מִשְׁחַת־קֹ֖דֶשׁ יִהְיֶֽה׃ וְתַעֲשֶׂה אִתָּם שֶׁמֶן לִמְרִיחָה כְּדֵי לְקַדֵּשׁ, תַּעֲרוֹבֶת שֶׁל בְּשָׂמִים טְחוּנִים כְּדֶרֶךְ הָרוֹקְחִים, וְיִהְיֶה שֶׁמֶן לִמְרִיחָה כְּדֵי לְקַדֵּשׁ.
כו וּמָשַׁחְתָּ֥ ב֖וֹ אֶת־אֹ֣הֶל מוֹעֵ֑ד וְאֵ֖ת אֲר֥וֹן הָעֵדֻֽת׃ וְתִמְרַח בּוֹ אֶת אוֹהֶל הַהִתְכַּנְּסוּת וְאֶת אֲרוֹן לוּחוֹת הַבְּרִית.
כז וְאֶת־הַשֻּׁלְחָן֙ וְאֶת־כׇּל־כֵּלָ֔יו וְאֶת־הַמְּנֹרָ֖ה וְאֶת־כֵּלֶ֑יהָ וְאֵ֖ת מִזְבַּ֥ח הַקְּטֹֽרֶת׃ וְאֶת הַשֻּׁלְחָן וְאֶת כָּל כֵּלָיו וְאֶת הַמְּנוֹרָה וְאֶת כֵּלֶיהָ וְאֶת מִזְבַּח הַקְּטוֹרֶת.
כח וְאֶת־מִזְבַּ֥ח הָעֹלָ֖ה וְאֶת־כׇּל־כֵּלָ֑יו וְאֶת־הַכִּיֹּ֖ר וְאֶת־כַּנּֽוֹ׃ וְאֶת מִזְבַּח הָעוֹלָה וְאֶת כָּל כֵּלָיו וְאֶת הַכִּיּוֹר וְאֶת בְּסִיסוֹ.
כט וְקִדַּשְׁתָּ֣ אֹתָ֔ם וְהָי֖וּ קֹ֣דֶשׁ קׇֽדָשִׁ֑ים כׇּל־הַנֹּגֵ֥עַ בָּהֶ֖ם יִקְדָּֽשׁ׃ תְּקַדֵּשׁ אוֹתָם וְהֵם יִהְיוּ קוֹדֶשׁ קָדָשִׁים, כָּל מִי שֶׁנּוֹגֵעַ בָּהֶם יְקַדֵּשׁ וִיטַהֵר עַצְמוֹ לִפְנֵי כֵן.
ל וְאֶת־אַהֲרֹ֥ן וְאֶת־בָּנָ֖יו תִּמְשָׁ֑ח וְקִדַּשְׁתָּ֥ אֹתָ֖ם לְכַהֵ֥ן לִֽי׃ וְאֶת אַהֲרֹן וְאֶת בָּנָיו תִּמְרַח וּתְקַדֵּשׁ אוֹתָם לִהְיוֹת לִי כֹּהֲנִים.
לא וְאֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל תְּדַבֵּ֣ר לֵאמֹ֑ר שֶׁ֠מֶן מִשְׁחַת־קֹ֨דֶשׁ יִהְיֶ֥ה זֶ֛ה לִ֖י לְדֹרֹתֵיכֶֽם׃ וְאֶל בְּנֵי יִשְׂרָאֵל תְּדַבֵּר וְתֹאמַר: זֶה יִהְיֶה שֶׁמֶן לִמְרִיחָה כְּדֵי לְקַדֵּשׁ בִּשְׁבִילִי, לְדוֹרוֹת.
לב עַל־בְּשַׂ֤ר אָדָם֙ לֹ֣א יִיסָ֔ךְ וּ֨בְמַתְכֻּנְתּ֔וֹ לֹ֥א תַעֲשׂ֖וּ כָּמֹ֑הוּ קֹ֣דֶשׁ ה֔וּא קֹ֖דֶשׁ יִהְיֶ֥ה לָכֶֽם׃ לֹא יָסוּכוּ מִמֶּנּוּ עַל בְּשַׂר אָדָם זָר, וְאַל תַּעֲשׂוּ כָּמוֹהוּ – הוּא קוֹדֶשׁ, קוֹדֶשׁ יִהְיֶה לָכֶם.
לג אִ֚ישׁ אֲשֶׁ֣ר יִרְקַ֣ח כָּמֹ֔הוּ וַאֲשֶׁ֥ר יִתֵּ֛ן מִמֶּ֖נּוּ עַל־זָ֑ר וְנִכְרַ֖ת מֵעַמָּֽיו׃ אִישׁ שֶׁיִּרְקַח כָּמוֹהוּ אוֹ שֶׁיִּתֵּן מִמֶּנּוּ עַל אָדָם זָר – יֻשְׁמַד מֵעַמּוֹ.
ס פָּרָשַׁת הַקְּטוֹרֶת
לד וַיֹּ֩אמֶר֩ יְהֹוָ֨ה אֶל־מֹשֶׁ֜ה קַח־לְךָ֣ סַמִּ֗ים נָטָ֤ף ׀ וּשְׁחֵ֙לֶת֙ וְחֶלְבְּנָ֔ה סַמִּ֖ים וּלְבֹנָ֣ה זַכָּ֑ה בַּ֥ד בְּבַ֖ד יִהְיֶֽה׃ ה' אָמַר לְמֹשֶׁה: קַח בְּשָׂמִים – נָטָף וּשְׁחֵלֶת וְחֶלְבְּנָה – בְּשָׂמִים, וּלְבוֹנָה זַכָּה, כֻּלָּם יִהְיוּ בְּאוֹתוֹ מִשְׁקָל.
לה וְעָשִׂ֤יתָ אֹתָהּ֙ קְטֹ֔רֶת רֹ֖קַח מַעֲשֵׂ֣ה רוֹקֵ֑חַ מְמֻלָּ֖ח טָה֥וֹר קֹֽדֶשׁ׃ וְתַעֲשֶׂה מֵהֶם קְטוֹרֶת, טְחוּנִים כְּדֶרֶךְ הָרוֹקְחִים, מְעֻרְבָּב, טָהוֹר, קוֹדֶשׁ.
לו וְשָֽׁחַקְתָּ֣ מִמֶּ֘נָּה֮ הָדֵק֒ וְנָתַתָּ֨ה מִמֶּ֜נָּה לִפְנֵ֤י הָעֵדֻת֙ בְּאֹ֣הֶל מוֹעֵ֔ד אֲשֶׁ֛ר אִוָּעֵ֥ד לְךָ֖ שָׁ֑מָּה קֹ֥דֶשׁ קׇֽדָשִׁ֖ים תִּהְיֶ֥ה לָכֶֽם׃ תִּטְחוֹן אוֹתָהּ דַּק דַּק, וְתָשִׂים מִמֶּנָּה לִפְנֵי אֲרוֹן לוּחוֹת הַבְּרִית בְּאוֹהֶל הַהִתְכַּנְּסוּת שֶׁאֶפָּגֵשׁ אִתְּךָ שָׁם, הִיא תִּהְיֶה לָכֶם קוֹדֶשׁ קָדָשִׁים.
לז וְהַקְּטֹ֙רֶת֙ אֲשֶׁ֣ר תַּעֲשֶׂ֔ה בְּמַ֨תְכֻּנְתָּ֔הּ לֹ֥א תַעֲשׂ֖וּ לָכֶ֑ם קֹ֛דֶשׁ תִּהְיֶ֥ה לְךָ֖ לַיהֹוָֽה׃ וְהַקְּטוֹרֶת שֶׁתַּעֲשֶׂה, אַל תַּעֲשׂוּ לָכֶם כָּמוֹהָ, הִיא תִּהְיֶה מְיוּחֶדֶת לְךָ רַק לַעֲבוֹדַת ה'.
לח אִ֛ישׁ אֲשֶׁר־יַעֲשֶׂ֥ה כָמ֖וֹהָ לְהָרִ֣יחַ בָּ֑הּ וְנִכְרַ֖ת מֵעַמָּֽיו׃ אִישׁ שֶׁיַּעֲשֶׂה כָּמוֹהָ לְהָרִיחַ בָּהּ – יֻשְׁמַד מֵעַמּוֹ.
ס פָּרָשַׁת חַכְמֵי הַלֵּב
לא א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב רְאֵ֖ה קָרָ֣אתִֽי בְשֵׁ֑ם בְּצַלְאֵ֛ל בֶּן־אוּרִ֥י בֶן־ח֖וּר לְמַטֵּ֥ה יְהוּדָֽה׃ רְאֵה, הִכְרַזְתִּי עַל בְּצַלְאֵל בֶּן אוּרִי בֶּן חוּר מִשֵּׁבֶט יְהוּדָה.
ג וָאֲמַלֵּ֥א אֹת֖וֹ ר֣וּחַ אֱלֹהִ֑ים בְּחׇכְמָ֛ה וּבִתְבוּנָ֥ה וּבְדַ֖עַת וּבְכׇל־מְלָאכָֽה׃ וּמִלֵּאתִי אוֹתוֹ בְּרוּחַ אֱלֹהִים, בְּחָכְמָה וּבִתְבוּנָה וּבְדַעַת וּבְכִשְׁרוֹן לְכָל מְלָאכָה.
ד לַחְשֹׁ֖ב מַחֲשָׁבֹ֑ת לַעֲשׂ֛וֹת בַּזָּהָ֥ב וּבַכֶּ֖סֶף וּבַנְּחֹֽשֶׁת׃ לְתַכְנֵן תָּכְנִיּוֹת, לַעֲשׂוֹת בְּזָהָב וּבְכֶסֶף וּבִנְחוֹשֶׁת.
ה וּבַחֲרֹ֥שֶׁת אֶ֛בֶן לְמַלֹּ֖את וּבַחֲרֹ֣שֶׁת עֵ֑ץ לַעֲשׂ֖וֹת בְּכׇל־מְלָאכָֽה׃ וּבַחֲרִיטַת אַבְנֵי חֵן לְשִׁבּוּץ, וּבְנַגָּרוּת עֵץ – לַעֲשׂוֹת בְּכָל מְלָאכָה.
ו וַאֲנִ֞י הִנֵּ֧ה נָתַ֣תִּי אִתּ֗וֹ אֵ֣ת אׇהֳלִיאָ֞ב בֶּן־אֲחִֽיסָמָךְ֙ לְמַטֵּה־דָ֔ן וּבְלֵ֥ב כׇּל־חֲכַם־לֵ֖ב נָתַ֣תִּי חׇכְמָ֑ה וְעָשׂ֕וּ אֵ֖ת כׇּל־אֲשֶׁ֥ר צִוִּיתִֽךָ׃ וְהִנֵּה שַׂמְתִּי אִתּוֹ אֶת אָהֳלִיאָב בֶּן אֲחִיסָמָךְ מִשֵּׁבֶט דָּן, וּבְלֵב כָּל חֲכַם לֵב נָתַתִּי חָכְמָה וְהֵם יַעֲשׂוּ אֶת כָּל מַה שֶּׁצִּוִּיתִי אוֹתְךָ:
ז אֵ֣ת ׀ אֹ֣הֶל מוֹעֵ֗ד וְאֶת־הָֽאָרֹן֙ לָֽעֵדֻ֔ת וְאֶת־הַכַּפֹּ֖רֶת אֲשֶׁ֣ר עָלָ֑יו וְאֵ֖ת כׇּל־כְּלֵ֥י הָאֹֽהֶל׃ אֶת אוֹהֶל הַהִתְכַּנְּסוּת, וְאֶת הָאָרוֹן לְלוּחוֹת הַבְּרִית וְאֶת הַכִּסּוּי שֶׁעָלָיו, וְאֶת כָּל כְּלֵי הָאוֹהֶל.
ח וְאֶת־הַשֻּׁלְחָן֙ וְאֶת־כֵּלָ֔יו וְאֶת־הַמְּנֹרָ֥ה הַטְּהֹרָ֖ה וְאֶת־כׇּל־כֵּלֶ֑יהָ וְאֵ֖ת מִזְבַּ֥ח הַקְּטֹֽרֶת׃ וְאֶת הַשֻּׁלְחָן וְאֶת כֵּלָיו וְאֶת הַמְּנוֹרָה הַטְּהוֹרָה וְאֶת כָּל כֵּלֶיהָ וְאֶת מִזְבַּח הַקְּטוֹרֶת.
ט וְאֶת־מִזְבַּ֥ח הָעֹלָ֖ה וְאֶת־כׇּל־כֵּלָ֑יו וְאֶת־הַכִּיּ֖וֹר וְאֶת־כַּנּֽוֹ׃ וְאֶת מִזְבַּח הָעוֹלָה וְאֶת כָּל כֵּלָיו וְאֶת הַכִּיּוֹר וְאֶת בְּסִיסוֹ.
י וְאֵ֖ת בִּגְדֵ֣י הַשְּׂרָ֑ד וְאֶת־בִּגְדֵ֤י הַקֹּ֙דֶשׁ֙ לְאַהֲרֹ֣ן הַכֹּהֵ֔ן וְאֶת־בִּגְדֵ֥י בָנָ֖יו לְכַהֵֽן׃ וְאֶת כִּסּוּיֵי כְּלֵי הַמִּשְׁכָּן, וְאֶת בִּגְדֵי הַקּוֹדֶשׁ לְאַהֲרֹן הַכֹּהֵן וְאֶת בִּגְדֵי בָּנָיו לְכַהֵן.
יא וְאֵ֨ת שֶׁ֧מֶן הַמִּשְׁחָ֛ה וְאֶת־קְטֹ֥רֶת הַסַּמִּ֖ים לַקֹּ֑דֶשׁ כְּכֹ֥ל אֲשֶׁר־צִוִּיתִ֖ךָ יַעֲשֽׂוּ׃ וְאֶת הַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ, וְאֶת קְטוֹרֶת הַבְּשָׂמִים לְהַקְטִיר בַּקּוֹדֶשׁ, כְּכָל מַה שֶּׁצִּוִּיתִי אוֹתְךָ – יַעֲשׂוּ.
פ פָּרָשַׁת הַשַּׁבָּת
יב וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' אָמַר לְמֹשֶׁה:
יג וְאַתָּ֞ה דַּבֵּ֨ר אֶל־בְּנֵ֤י יִשְׂרָאֵל֙ לֵאמֹ֔ר אַ֥ךְ אֶת־שַׁבְּתֹתַ֖י תִּשְׁמֹ֑רוּ כִּי֩ א֨וֹת הִ֜וא בֵּינִ֤י וּבֵֽינֵיכֶם֙ לְדֹרֹ֣תֵיכֶ֔ם לָדַ֕עַת כִּ֛י אֲנִ֥י יְהֹוָ֖ה מְקַדִּשְׁכֶֽם׃ אַתָּה, דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְתֹאמַר: אֲבָל תִּשְׁמְרוּ אֶת שַׁבְּתוֹתַי וְאַל תַּעֲשׂוּ בְּשַׁבָּת אֶת הַמִּשְׁכָּן, כִּי הִיא סִימָן בֵּינִי וּבֵינֵיכֶם לְדוֹרוֹת, לָדַעַת שֶׁאֲנִי ה' מְקַדֵּשׁ אֶתְכֶם.
יד וּשְׁמַרְתֶּם֙ אֶת־הַשַּׁבָּ֔ת כִּ֛י קֹ֥דֶשׁ הִ֖וא לָכֶ֑ם מְחַֽלְלֶ֙יהָ֙ מ֣וֹת יוּמָ֔ת כִּ֗י כׇּל־הָעֹשֶׂ֥ה בָהּ֙ מְלָאכָ֔ה וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מִקֶּ֥רֶב עַמֶּֽיהָ׃ שִׁמְרוּ אֶת הַשַּׁבָּת כִּי הִיא קְדוֹשָׁה לָכֶם, הַמְּחַלֵּל אוֹתָהּ יוּמַת, כִּי כָּל מִי שֶׁעוֹשֶׂה בָּהּ מְלָאכָה, אוֹתָהּ נֶפֶשׁ תֹּאבַד מִתּוֹךְ עַמָּהּ.
טו שֵׁ֣שֶׁת יָמִים֮ יֵעָשֶׂ֣ה מְלָאכָה֒ וּבַיּ֣וֹם הַשְּׁבִיעִ֗י שַׁבַּ֧ת שַׁבָּת֛וֹן קֹ֖דֶשׁ לַיהֹוָ֑ה כׇּל־הָעֹשֶׂ֧ה מְלָאכָ֛ה בְּי֥וֹם הַשַּׁבָּ֖ת מ֥וֹת יוּמָֽת׃ שִׁשָּׁה יָמִים יַעֲשׂוּ מְלָאכָה, וּבַיּוֹם הַשְּׁבִיעִי שְׁבִיתָה שֶׁל חֻפְשַׁת קוֹדֶשׁ לַה', כָּל מִי שֶׁעוֹשֶׂה מְלָאכָה בְּיוֹם הַשַּׁבָּת יוּמַת.
טז וְשָׁמְר֥וּ בְנֵֽי־יִשְׂרָאֵ֖ל אֶת־הַשַּׁבָּ֑ת לַעֲשׂ֧וֹת אֶת־הַשַּׁבָּ֛ת לְדֹרֹתָ֖ם בְּרִ֥ית עוֹלָֽם׃ בְּנֵי יִשְׂרָאֵל יִשְׁמְרוּ אֶת הַשַּׁבָּת, לַעֲשׂוֹת אֶת מִצְוַת הַשַּׁבָּת לְדוֹרוֹת, הֶסְכֵּם לְעוֹלָם.
יז בֵּינִ֗י וּבֵין֙ בְּנֵ֣י יִשְׂרָאֵ֔ל א֥וֹת הִ֖וא לְעֹלָ֑ם כִּי־שֵׁ֣שֶׁת יָמִ֗ים עָשָׂ֤ה יְהֹוָה֙ אֶת־הַשָּׁמַ֣יִם וְאֶת־הָאָ֔רֶץ וּבַיּוֹם֙ הַשְּׁבִיעִ֔י שָׁבַ֖ת וַיִּנָּפַֽשׁ׃ הִיא סִימָן בֵּינִי וּבֵין בְּנֵי יִשְׂרָאֵל לְעוֹלָמִים שֶׁה' עָשָׂה אֶת הַשָּׁמַיִם וְאֶת הָאָרֶץ בְּשִׁשָּׁה יָמִים, וּבַיּוֹם הַשְּׁבִיעִי הוּא שָׁבַת מִמְּלַאכְתּוֹ וְנָח.

שני[עריכה]

ס פָּרָשַׁת הָעֵגֶל
יח וַיִּתֵּ֣ן אֶל־מֹשֶׁ֗ה כְּכַלֹּתוֹ֙ לְדַבֵּ֤ר אִתּוֹ֙ בְּהַ֣ר סִינַ֔י שְׁנֵ֖י לֻחֹ֣ת הָעֵדֻ֑ת לֻחֹ֣ת אֶ֔בֶן כְּתֻבִ֖ים בְּאֶצְבַּ֥ע אֱלֹהִֽים׃ ה' נָתַן לְמֹשֶׁה, כְּשֶׁסִיֵּם לְדַבֵּר אִתּוֹ בְּהַר סִינַי, אֶת שְׁנֵי לוּחוֹת הַבְּרִית – לוּחוֹת אֶבֶן כְּתוּבִים בְּאֶצְבַּע אֱלֹהִים.
לב א וַיַּ֣רְא הָעָ֔ם כִּֽי־בֹשֵׁ֥שׁ מֹשֶׁ֖ה לָרֶ֣דֶת מִן־הָהָ֑ר וַיִּקָּהֵ֨ל הָעָ֜ם עַֽל־אַהֲרֹ֗ן וַיֹּאמְר֤וּ אֵלָיו֙ ק֣וּם ׀ עֲשֵׂה־לָ֣נוּ אֱלֹהִ֗ים אֲשֶׁ֤ר יֵֽלְכוּ֙ לְפָנֵ֔ינוּ כִּי־זֶ֣ה ׀ מֹשֶׁ֣ה הָאִ֗ישׁ אֲשֶׁ֤ר הֶֽעֱלָ֙נוּ֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לֹ֥א יָדַ֖עְנוּ מֶה־הָ֥יָה לֽוֹ׃ הָעָם רָאָה שֶׁמֹּשֶׁה מְאַחֵר לָרֶדֶת מֵהָהָר, הָעָם הִתְקָהֵל לִפְנֵי אַהֲרֹן, וְאָמְרוּ לוֹ: קוּם עֲשֵׂה לָנוּ אֱלֹהִים שֶׁיַּנְהִיגוּ אוֹתָנוּ, כִּי מֹשֶׁה הָאִישׁ הַזֶּה שֶׁהֶעֱלָה אוֹתָנוּ מֵאֶרֶץ מִצְרַיִם – אֵינֶנּוּ יוֹדְעִים מַה קָּרָה לוֹ.
ב וַיֹּ֤אמֶר אֲלֵהֶם֙ אַהֲרֹ֔ן פָּֽרְקוּ֙ נִזְמֵ֣י הַזָּהָ֔ב אֲשֶׁר֙ בְּאׇזְנֵ֣י נְשֵׁיכֶ֔ם בְּנֵיכֶ֖ם וּבְנֹתֵיכֶ֑ם וְהָבִ֖יאוּ אֵלָֽי׃ אָמַר לָהֶם אַהֲרֹן: הֲסִירוּ אֶת נִזְמֵי הַזָּהָב שֶׁבְּאָזְנֵי נְשֵׁיכֶם בְּנֵיכֶם וּבְנוֹתֵיכֶם וְהָבִיאוּ אֵלַי.
ג וַיִּתְפָּֽרְקוּ֙ כׇּל־הָעָ֔ם אֶת־נִזְמֵ֥י הַזָּהָ֖ב אֲשֶׁ֣ר בְּאׇזְנֵיהֶ֑ם וַיָּבִ֖יאוּ אֶֽל־אַהֲרֹֽן׃ כָּל הָעָם הֵסִירוּ אֶת נִזְמֵי הַזָּהָב שֶׁבְּאָזְנֵיהֶם וְהֵבִיאוּ לְאַהֲרֹן.
ד וַיִּקַּ֣ח מִיָּדָ֗ם וַיָּ֤צַר אֹתוֹ֙ בַּחֶ֔רֶט וַֽיַּעֲשֵׂ֖הוּ עֵ֣גֶל מַסֵּכָ֑ה וַיֹּ֣אמְר֔וּ אֵ֤לֶּה אֱלֹהֶ֙יךָ֙ יִשְׂרָאֵ֔ל אֲשֶׁ֥ר הֶעֱל֖וּךָ מֵאֶ֥רֶץ מִצְרָֽיִם׃ הוּא לָקַח מִיָּדָם וְיָצַר מֵהַזָּהָב צוּרָה בִּכְלִי חֲרִיטָה, וְעָשָׂה אוֹתוֹ בִּיצִיקָה בִּדְמוּת שֶׁל עֵגֶל, וְהֵם אָמְרוּ: אֵלֶּה אֱלֹהֶיךָ יִשְׂרָאֵל שֶׁהֶעֱלוּ אוֹתְךָ מֵאֶרֶץ מִצְרַיִם!
ה וַיַּ֣רְא אַהֲרֹ֔ן וַיִּ֥בֶן מִזְבֵּ֖חַ לְפָנָ֑יו וַיִּקְרָ֤א אַֽהֲרֹן֙ וַיֹּאמַ֔ר חַ֥ג לַיהֹוָ֖ה מָחָֽר׃ אַהֲרֹן רָאָה זֹאת, וּבָנָה לְפָנָיו מִזְבֵּחַ, אַהֲרֹן קָרָא וְאָמַר: מָחָר חַג לַה'!
ו וַיַּשְׁכִּ֙ימוּ֙ מִֽמׇּחֳרָ֔ת וַיַּעֲל֣וּ עֹלֹ֔ת וַיַּגִּ֖שׁוּ שְׁלָמִ֑ים וַיֵּ֤שֶׁב הָעָם֙ לֶֽאֱכֹ֣ל וְשָׁת֔וֹ וַיָּקֻ֖מוּ לְצַחֵֽק׃ הֵם הִשְׁכִּימוּ לְמָחֳרָת, הֶעֱלוּ עוֹלוֹת וְהִגִּישׁוּ שְׁלָמִים, הָעָם יָשַׁב לֶאֱכוֹל וְלִשְׁתּוֹת, וְקָמוּ לְהִתְלוֹצֵץ.
פ פָּרָשַׁת תְּפִלַּת מֹשֶׁה
ז וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֑ה לֶךְ־רֵ֕ד כִּ֚י שִׁחֵ֣ת עַמְּךָ֔ אֲשֶׁ֥ר הֶעֱלֵ֖יתָ מֵאֶ֥רֶץ מִצְרָֽיִם׃ ה' דִּבֵּר אֶל מֹשֶׁה: לֵךְ רֵד כִּי עַמְּךָ שֶׁהֶעֱלֵיתָ מֵאֶרֶץ מִצְרַיִם הִשְׁחִית.
ח סָ֣רוּ מַהֵ֗ר מִן־הַדֶּ֙רֶךְ֙ אֲשֶׁ֣ר צִוִּיתִ֔ם עָשׂ֣וּ לָהֶ֔ם עֵ֖גֶל מַסֵּכָ֑ה וַיִּשְׁתַּֽחֲווּ־לוֹ֙ וַיִּזְבְּחוּ־ל֔וֹ וַיֹּ֣אמְר֔וּ אֵ֤לֶּה אֱלֹהֶ֙יךָ֙ יִשְׂרָאֵ֔ל אֲשֶׁ֥ר הֶֽעֱל֖וּךָ מֵאֶ֥רֶץ מִצְרָֽיִם׃ הֵם סָטוּ מַהֵר מֵהַדֶּרֶךְ שֶׁצִּוִּיתִי אוֹתָם – עָשׂוּ לָהֶם יְצִיקָה בִּדְמוּת שֶׁל עֵגֶל, הִשְׁתַּחֲווּ לוֹ וְהִקְרִיבוּ לְפָנָיו, וְאָמְרוּ: אֵלֶּה אֱלֹהֶיךָ יִשְׂרָאֵל שֶׁהֶעֱלוּ אוֹתְךָ מֵאֶרֶץ מִצְרַיִם!
ט וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל־מֹשֶׁ֑ה רָאִ֙יתִי֙ אֶת־הָעָ֣ם הַזֶּ֔ה וְהִנֵּ֥ה עַם־קְשֵׁה־עֹ֖רֶף הֽוּא׃ ה' אָמַר לְמֹשֶׁה: רָאִיתִי אֶת הָעָם הַזֶּה וְהִנֵּה הוּא עַקְשָׁן וְסַרְבָן.
י וְעַתָּה֙ הַנִּ֣יחָה לִּ֔י וְיִֽחַר־אַפִּ֥י בָהֶ֖ם וַאֲכַלֵּ֑ם וְאֶֽעֱשֶׂ֥ה אוֹתְךָ֖ לְג֥וֹי גָּדֽוֹל׃ וְעַכְשָׁיו עֲזוֹב אוֹתִי וְאֶכְעַס עֲלֵיהֶם וְאַשְׁמִיד אוֹתָם, וְאֶעֱשֶׂה מִמְּךָ עַם גָּדוֹל.
יא וַיְחַ֣ל מֹשֶׁ֔ה אֶת־פְּנֵ֖י יְהֹוָ֣ה אֱלֹהָ֑יו וַיֹּ֗אמֶר לָמָ֤ה יְהֹוָה֙ יֶחֱרֶ֤ה אַפְּךָ֙ בְּעַמֶּ֔ךָ אֲשֶׁ֤ר הוֹצֵ֙אתָ֙ מֵאֶ֣רֶץ מִצְרַ֔יִם בְּכֹ֥חַ גָּד֖וֹל וּבְיָ֥ד חֲזָקָֽה׃ מֹשֶׁה הִתְפַּלֵּל לִפְנֵי ה' אֱלֹהָיו, וְאָמַר: ה' לָמָּה תִּכְעַס עַל עַמְּךָ שֶׁהוֹצֵאתָ מֵאֶרֶץ מִצְרַיִם בְּכֹחַ גָּדוֹל וּבְיָד חֲזָקָה?!
יב לָ֩מָּה֩ יֹאמְר֨וּ מִצְרַ֜יִם לֵאמֹ֗ר בְּרָעָ֤ה הֽוֹצִיאָם֙ לַהֲרֹ֤ג אֹתָם֙ בֶּֽהָרִ֔ים וּ֨לְכַלֹּתָ֔ם מֵעַ֖ל פְּנֵ֣י הָֽאֲדָמָ֑ה שׁ֚וּב מֵחֲר֣וֹן אַפֶּ֔ךָ וְהִנָּחֵ֥ם עַל־הָרָעָ֖ה לְעַמֶּֽךָ׃ לָמָּה יֹאמְרוּ מִצְרַיִם שֶׁהוּא הוֹצִיא אוֹתָם בְּרַע – כְּדֵי לַהֲרוֹג אוֹתָם בֶּהָרִים וּלְכַלּוֹת אוֹתָם מֵעַל פְּנֵי הָאֲדָמָה?! חֲזוֹר בְּךָ מִכַּעַסְךָ וְתִתְחָרֵט עַל הָרָעָה שֶׁחָשַׁבְתָּ לַעֲשׂוֹת לְעַמְּךָ!
יג זְכֹ֡ר לְאַבְרָהָם֩ לְיִצְחָ֨ק וּלְיִשְׂרָאֵ֜ל עֲבָדֶ֗יךָ אֲשֶׁ֨ר נִשְׁבַּ֣עְתָּ לָהֶם֮ בָּךְ֒ וַתְּדַבֵּ֣ר אֲלֵהֶ֔ם אַרְבֶּה֙ אֶֽת־זַרְעֲכֶ֔ם כְּכוֹכְבֵ֖י הַשָּׁמָ֑יִם וְכׇל־הָאָ֨רֶץ הַזֹּ֜את אֲשֶׁ֣ר אָמַ֗רְתִּי אֶתֵּן֙ לְזַרְעֲכֶ֔ם וְנָחֲל֖וּ לְעֹלָֽם׃ זְכוֹר לְאַבְרָהָם לְיִצְחָק וּלְיִשְׂרָאֵל עֲבָדֶיךָ שֶׁנִּשְׁבַּעְתָּ לָהֶם בְּשִׁמְךָ, וְאָמַרְתָּ לָהֶם: אַרְבֶּה אֶת צֶאֱצָאֵיכֶם כְּכוֹכְבֵי הַשָּׁמַיִם, וְכָל הָאָרֶץ הַזֹּאת שֶׁאָמַרְתִּי שֶׁאֶתֵּן לְצֶאֱצָאֵיכֶם – וְהֵם יִנְחֲלוּ אוֹתָהּ לְעוֹלָם.
יד וַיִּנָּ֖חֶם יְהֹוָ֑ה עַל־הָ֣רָעָ֔ה אֲשֶׁ֥ר דִּבֶּ֖ר לַעֲשׂ֥וֹת לְעַמּֽוֹ׃ ה' הִתְחָרֵט עַל הָרָעָה שֶׁחָשַׁב לַעֲשׂוֹת לְעַמּוֹ.
פ פָּרָשַׁת הַיְּרִידָה מֵהָהָר
טו וַיִּ֜פֶן וַיֵּ֤רֶד מֹשֶׁה֙ מִן־הָהָ֔ר וּשְׁנֵ֛י לֻחֹ֥ת הָעֵדֻ֖ת בְּיָד֑וֹ לֻחֹ֗ת כְּתֻבִים֙ מִשְּׁנֵ֣י עֶבְרֵיהֶ֔ם מִזֶּ֥ה וּמִזֶּ֖ה הֵ֥ם כְּתֻבִֽים׃ מֹשֶׁה פָּנָה וְיָרַד מֵהָהָר וּשְׁנֵי לוּחוֹת הַבְּרִית בְּיָדוֹ – לוּחוֹת כְּתוּבִים מִשְּׁנֵי עֶבְרֵיהֶם – הֵם כְּתוּבִים מִצַּד זֶה וּמִצַּד זֶה.
טז וְהַ֨לֻּחֹ֔ת מַעֲשֵׂ֥ה אֱלֹהִ֖ים הֵ֑מָּה וְהַמִּכְתָּ֗ב מִכְתַּ֤ב אֱלֹהִים֙ ה֔וּא חָר֖וּת עַל־הַלֻּחֹֽת׃ וְהַלּוּחוֹת – הֵם מַעֲשֵׂה אֱלֹהִים, וְהַכְּתָב – הוּא כְּתִיבַת אֱלֹהִים, חָרוּת עַל הַלּוּחוֹת.
יז וַיִּשְׁמַ֧ע יְהוֹשֻׁ֛עַ אֶת־ק֥וֹל הָעָ֖ם בְּרֵעֹ֑ה וַיֹּ֙אמֶר֙ אֶל־מֹשֶׁ֔ה ק֥וֹל מִלְחָמָ֖ה בַּֽמַּחֲנֶֽה׃ יְהוֹשֻׁעַ שָׁמַע אֶת קוֹל הָעָם שֶׁהֵרִיעוּ, וְאָמַר לְמֹשֶׁה: יֵשׁ קוֹל שֶׁל מִלְחָמָה בַּמַּחֲנֶה!
יח וַיֹּ֗אמֶר אֵ֥ין קוֹל֙ עֲנ֣וֹת גְּבוּרָ֔ה וְאֵ֥ין ק֖וֹל עֲנ֣וֹת חֲלוּשָׁ֑ה ק֣וֹל עַנּ֔וֹת אָנֹכִ֖י שֹׁמֵֽעַ׃ אָמַר לוֹ: אֵין קוֹל צְעָקוֹת שֶׁל גְּבוּרָה וְאֵין קוֹל צְעָקוֹת שֶׁל תְּבוּסָה, אֲנִי שׁוֹמֵעַ קוֹל צְעָקוֹת שֶׁל הִתְלוֹצְצוּת.
יט וַֽיְהִ֗י כַּאֲשֶׁ֤ר קָרַב֙ אֶל־הַֽמַּחֲנֶ֔ה וַיַּ֥רְא אֶת־הָעֵ֖גֶל וּמְחֹלֹ֑ת וַיִּֽחַר־אַ֣ף מֹשֶׁ֗ה וַיַּשְׁלֵ֤ךְ מִיָּדָו֙ אֶת־הַלֻּחֹ֔ת וַיְשַׁבֵּ֥ר אֹתָ֖ם תַּ֥חַת הָהָֽר׃ כְּשֶׁהִתְקָרֵב לַמַּחֲנֶה הוּא רָאָה אֶת הָעֵגֶל וְאֶת הָרִקּוּדִים לִצְלִילֵי הַחֲלִילִים, מֹשֶׁה כָּעַס וְהִשְׁלִיךְ מִיָּדָיו אֶת הַלּוּחוֹת וְשִׁבֵּר אוֹתָם לְמַרְגְּלוֹת הָהָר.
כ וַיִּקַּ֞ח אֶת־הָעֵ֨גֶל אֲשֶׁ֤ר עָשׂוּ֙ וַיִּשְׂרֹ֣ף בָּאֵ֔שׁ וַיִּטְחַ֖ן עַ֣ד אֲשֶׁר־דָּ֑ק וַיִּ֙זֶר֙ עַל־פְּנֵ֣י הַמַּ֔יִם וַיַּ֖שְׁקְ אֶת־בְּנֵ֥י יִשְׂרָאֵֽל׃ הוּא לָקַח אֶת הָעֵגֶל שֶׁעָשׂוּ וְשָׂרַף אוֹתוֹ בָּאֵשׁ, וְטָחַן עַד שֶׁנַּעֲשָׂה דַּק, וְזָרַק עַל פְּנֵי הַמַּיִם, וְנָתַן לִבְנֵי יִשְׂרָאֵל לִשְׁתּוֹת.
כא וַיֹּ֤אמֶר מֹשֶׁה֙ אֶֽל־אַהֲרֹ֔ן מֶֽה־עָשָׂ֥ה לְךָ֖ הָעָ֣ם הַזֶּ֑ה כִּֽי־הֵבֵ֥אתָ עָלָ֖יו חֲטָאָ֥ה גְדֹלָֽה׃ מֹשֶׁה אָמַר לְאַהֲרֹן: מֶה עָשָׂה לְךָ הָעָם הַזֶּה שֶׁהֵבֵאתָ עָלָיו חֵטְא גָּדוֹל?!
כב וַיֹּ֣אמֶר אַהֲרֹ֔ן אַל־יִ֥חַר אַ֖ף אֲדֹנִ֑י אַתָּה֙ יָדַ֣עְתָּ אֶת־הָעָ֔ם כִּ֥י בְרָ֖ע הֽוּא׃ אָמַר אַהֲרֹן: שֶׁאֲדוֹנִי לֹא יִכְעַס! אַתָּה מַכִּיר אֶת הָעָם שֶׁהוּא שָׁקוּעַ בְּיֵצֶר רַע.
כג וַיֹּ֣אמְרוּ לִ֔י עֲשֵׂה־לָ֣נוּ אֱלֹהִ֔ים אֲשֶׁ֥ר יֵלְכ֖וּ לְפָנֵ֑ינוּ כִּי־זֶ֣ה ׀ מֹשֶׁ֣ה הָאִ֗ישׁ אֲשֶׁ֤ר הֶֽעֱלָ֙נוּ֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לֹ֥א יָדַ֖עְנוּ מֶה־הָ֥יָה לֽוֹ׃ אָמְרוּ לִי: עֲשֵׂה לָנוּ אֱלֹהִים שֶׁיַּנְהִיגוּ אוֹתָנוּ, כִּי מֹשֶׁה הָאִישׁ הַזֶּה שֶׁהֶעֱלָה אוֹתָנוּ מֵאֶרֶץ מִצְרַיִם, אֵינֶנּוּ יוֹדְעִים מַה קָּרָה לוֹ.
כד וָאֹמַ֤ר לָהֶם֙ לְמִ֣י זָהָ֔ב הִתְפָּרָ֖קוּ וַיִּתְּנוּ־לִ֑י וָאַשְׁלִכֵ֣הוּ בָאֵ֔שׁ וַיֵּצֵ֖א הָעֵ֥גֶל הַזֶּֽה׃ אָמַרְתִּי לָהֶם: לְמִי יֵשׁ זָהָב? וְהֵם הֵסִירוּ מֵעַצְמָם וְנָתְנוּ לִי, הִשְׁלַכְתִּי אוֹתוֹ לָאֵשׁ וְיָצָא הָעֵגֶל הַזֶּה.
כה וַיַּ֤רְא מֹשֶׁה֙ אֶת־הָעָ֔ם כִּ֥י פָרֻ֖עַ ה֑וּא כִּֽי־פְרָעֹ֣ה אַהֲרֹ֔ן לְשִׁמְצָ֖ה בְּקָמֵיהֶֽם׃ מֹשֶׁה רָאָה אֶת הָעָם שֶׁהוּא חֲסַר עוֹל, כִּי אַהֲרֹן הֵסִיר אֶת עֻלּוֹ, וַהֲרֵיהוּ מֻשְׁמָץ בְּפִי כָּל מִי שֶׁקָּם נֶגְדּוֹ.
כו וַיַּעֲמֹ֤ד מֹשֶׁה֙ בְּשַׁ֣עַר הַֽמַּחֲנֶ֔ה וַיֹּ֕אמֶר מִ֥י לַיהֹוָ֖ה אֵלָ֑י וַיֵּאָסְפ֥וּ אֵלָ֖יו כׇּל־בְּנֵ֥י לֵוִֽי׃ מֹשֶׁה עָמַד בְּשַׁעַר הַמַּחֲנֶה, וְאָמַר: מִי שֶׁנֶּאֱמָן לַה' שֶׁיָּבוֹא אֵלַי! וְהִתְאַסְּפוּ אֵלָיו כָּל בְּנֵי לֵוִי.
כז וַיֹּ֣אמֶר לָהֶ֗ם כֹּֽה־אָמַ֤ר יְהֹוָה֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל שִׂ֥ימוּ אִישׁ־חַרְבּ֖וֹ עַל־יְרֵכ֑וֹ עִבְר֨וּ וָשׁ֜וּבוּ מִשַּׁ֤עַר לָשַׁ֙עַר֙ בַּֽמַּחֲנֶ֔ה וְהִרְג֧וּ אִֽישׁ־אֶת־אָחִ֛יו וְאִ֥ישׁ אֶת־רֵעֵ֖הוּ וְאִ֥ישׁ אֶת־קְרֹבֽוֹ׃ אָמַר לָהֶם: כָּךְ אָמַר ה' אֱלֹהֵי יִשְׂרָאֵל – שִׂימוּ כָּל אֶחָד אֶת חַרְבּוֹ עַל יְרֵכוֹ, עִבְרוּ וְשׁוּבוּ מִבֵּית דִּין לְבֵית דִּין בַּמַּחֲנֶה וְהִרְגוּ אֶת הַחוֹטְאִים, אֲפִלּוּ אִישׁ אֶת אָחִיו וְאִישׁ אֶת חֲבֵרוֹ וְאִישׁ אֶת קְרוֹבוֹ!
כח וַיַּֽעֲשׂ֥וּ בְנֵֽי־לֵוִ֖י כִּדְבַ֣ר מֹשֶׁ֑ה וַיִּפֹּ֤ל מִן־הָעָם֙ בַּיּ֣וֹם הַה֔וּא כִּשְׁלֹ֥שֶׁת אַלְפֵ֖י אִֽישׁ׃ בְּנֵי לֵוִי עָשׂוּ כְּדִבְרֵי מֹשֶׁה, וְנָפְלוּ מֵהָעָם בַּיּוֹם הַהוּא כִּשְׁלוֹשֶׁת אֲלָפִים אֲנָשִׁים.
כט וַיֹּ֣אמֶר מֹשֶׁ֗ה מִלְא֨וּ יֶדְכֶ֤ם הַיּוֹם֙ לַֽיהֹוָ֔ה כִּ֛י אִ֥ישׁ בִּבְנ֖וֹ וּבְאָחִ֑יו וְלָתֵ֧ת עֲלֵיכֶ֛ם הַיּ֖וֹם בְּרָכָֽה׃ מֹשֶׁה אָמַר: הִכָּנְסוּ הַיּוֹם לְתַפְקִידְכֶם לַעֲבוֹדַת ה', כִּי כָּל אִישׁ הָרַג אֲפִלּוּ אֶת בְּנוֹ וְאֶת אָחִיו, וְתִנָּתֵן עֲלֵיכֶם הַיּוֹם בְּרָכָה.
ל וַיְהִי֙ מִֽמׇּחֳרָ֔ת וַיֹּ֤אמֶר מֹשֶׁה֙ אֶל־הָעָ֔ם אַתֶּ֥ם חֲטָאתֶ֖ם חֲטָאָ֣ה גְדֹלָ֑ה וְעַתָּה֙ אֶֽעֱלֶ֣ה אֶל־יְהֹוָ֔ה אוּלַ֥י אֲכַפְּרָ֖ה בְּעַ֥ד חַטַּאתְכֶֽם׃ לְמָחֳרָת מֹשֶׁה אָמַר לָעָם: אַתֶּם חֲטָאתֶם חֵטְא גָּדוֹל, וְעַכְשָׁיו אֶעֱלֶה אֶל ה' אוּלַי אֲכַפֵּר עַל חַטַּאתְכֶם.
לא וַיָּ֧שׇׁב מֹשֶׁ֛ה אֶל־יְהֹוָ֖ה וַיֹּאמַ֑ר אָ֣נָּ֗א חָטָ֞א הָעָ֤ם הַזֶּה֙ חֲטָאָ֣ה גְדֹלָ֔ה וַיַּֽעֲשׂ֥וּ לָהֶ֖ם אֱלֹהֵ֥י זָהָֽב׃ מֹשֶׁה חָזַר אֶל ה' וְאָמַר: אָנָּא, הָעָם הַזֶּה חָטָא חֵטְא גָּדוֹל וְעָשׂוּ לָהֶם אֱלֹהֵי זָהָב.
לב וְעַתָּ֖ה אִם־תִּשָּׂ֣א חַטָּאתָ֑ם וְאִם־אַ֕יִן מְחֵ֣נִי נָ֔א מִֽסִּפְרְךָ֖ אֲשֶׁ֥ר כָּתָֽבְתָּ׃ וְעַכְשָׁיו, אִם תּוּכַל לָשֵׂאת אֶת כּוֹבֶד חַטָּאתָם וְלִסְלוֹחַ – שָׂא, וְאִם לֹא – מְחוֹק אוֹתִי בְּבַקָּשָׁה מִסֵּפֶר הַצַּדִּיקִים שֶׁכָּתַבְתָּ!
לג וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל־מֹשֶׁ֑ה מִ֚י אֲשֶׁ֣ר חָֽטָא־לִ֔י אֶמְחֶ֖נּוּ מִסִּפְרִֽי׃ ה' אָמַר לְמֹשֶׁה: מִי שֶׁחָטָא לִי – אוֹתוֹ אֶמְחוֹק מִסִּפְרִי.
לד וְעַתָּ֞ה לֵ֣ךְ ׀ נְחֵ֣ה אֶת־הָעָ֗ם אֶ֤ל אֲשֶׁר־דִּבַּ֙רְתִּי֙ לָ֔ךְ הִנֵּ֥ה מַלְאָכִ֖י יֵלֵ֣ךְ לְפָנֶ֑יךָ וּבְי֣וֹם פׇּקְדִ֔י וּפָקַדְתִּ֥י עֲלֵהֶ֖ם חַטָּאתָֽם׃ וְעַכְשָׁיו לֵךְ הַנְהֵג אֶת הָעָם אֶל הַמָּקוֹם שֶׁאָמַרְתִּי לְךָ, הִנֵּה מַלְאָכִי בִּלְבָד יֵלֵךְ לְפָנֶיךָ, וּבְבוֹא הַיּוֹם אֶזְכּוֹר לְהַעֲנִישׁ אוֹתָם עַל חַטָּאתָם.
לה וַיִּגֹּ֥ף יְהֹוָ֖ה אֶת־הָעָ֑ם עַ֚ל אֲשֶׁ֣ר עָשׂ֣וּ אֶת־הָעֵ֔גֶל אֲשֶׁ֥ר עָשָׂ֖ה אַהֲרֹֽן׃ ה' הִכָּה אֶת הָעָם בְּמַגֵּפָה, עַל שֶׁעָשׂוּ אֶת הָעֵגֶל שֶׁאַהֲרֹן עָשָׂה.
ס פָּרָשַׁת הֶסְתֵּר הַפָּנִים
לג א וַיְדַבֵּ֨ר יְהֹוָ֤ה אֶל־מֹשֶׁה֙ לֵ֣ךְ עֲלֵ֣ה מִזֶּ֔ה אַתָּ֣ה וְהָעָ֔ם אֲשֶׁ֥ר הֶֽעֱלִ֖יתָ מֵאֶ֣רֶץ מִצְרָ֑יִם אֶל־הָאָ֗רֶץ אֲשֶׁ֣ר נִ֠שְׁבַּ֠עְתִּי לְאַבְרָהָ֨ם לְיִצְחָ֤ק וּֽלְיַעֲקֹב֙ לֵאמֹ֔ר לְזַרְעֲךָ֖ אֶתְּנֶֽנָּה׃ ה' דִּבֵּר לְמֹשֶׁה: לֵךְ עֲלֵה מִכָּאן, אַתָּה וְהָעָם שֶׁהֶעֱלִיתָ מֵאֶרֶץ מִצְרַיִם, לָאָרֶץ שֶׁנִּשְׁבַּעְתִּי לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב שֶׁאֶתֵּן אוֹתָהּ לַצֶּאֱצָאִים שֶׁלָּהֶם.
ב וְשָׁלַחְתִּ֥י לְפָנֶ֖יךָ מַלְאָ֑ךְ וְגֵֽרַשְׁתִּ֗י אֶת־הַֽכְּנַעֲנִי֙ הָֽאֱמֹרִ֔י וְהַֽחִתִּי֙ וְהַפְּרִזִּ֔י הַחִוִּ֖י וְהַיְבוּסִֽי׃ וְאֶשְׁלַח לְפָנֶיךָ מַלְאָךְ, וַאֲגָרֵשׁ אֶת הַכְּנַעֲנִי הָאֱמוֹרִי, וְהַחִתִּי וְהַפְּרִזִּי, הַחִוִּי וְהַיְּבוּסִי.
ג אֶל־אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָ֑שׁ כִּי֩ לֹ֨א אֶֽעֱלֶ֜ה בְּקִרְבְּךָ֗ כִּ֤י עַם־קְשֵׁה־עֹ֙רֶף֙ אַ֔תָּה פֶּן־אֲכֶלְךָ֖ בַּדָּֽרֶךְ׃ עֲלֵה אִתּוֹ לְאֶרֶץ זָבַת חָלָב וּדְבָשׁ, כִּי לֹא אֶעֱלֶה בְּעַצְמִי בְּתוֹכְךָ כִּי אַתָּה עַם עַקְשָׁן וְסַרְבָן – שֶׁלֹּא אַשְׁמִיד אוֹתְךָ בַּדֶּרֶךְ.
ד וַיִּשְׁמַ֣ע הָעָ֗ם אֶת־הַדָּבָ֥ר הָרָ֛ע הַזֶּ֖ה וַיִּתְאַבָּ֑לוּ וְלֹא־שָׁ֛תוּ אִ֥ישׁ עֶדְי֖וֹ עָלָֽיו׃ הָעָם שָׁמַע אֶת הַדָּבָר הָרָע הַזֶּה וְהִתְאַבְּלוּ, וְאַף אֶחָד לֹא שָׂם אֶת תַּכְשִׁיטוֹ עָלָיו.
ה וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה אֱמֹ֤ר אֶל־בְּנֵֽי־יִשְׂרָאֵל֙ אַתֶּ֣ם עַם־קְשֵׁה־עֹ֔רֶף רֶ֧גַע אֶחָ֛ד אֶֽעֱלֶ֥ה בְקִרְבְּךָ֖ וְכִלִּיתִ֑יךָ וְעַתָּ֗ה הוֹרֵ֤ד עֶדְיְךָ֙ מֵֽעָלֶ֔יךָ וְאֵדְעָ֖ה מָ֥ה אֶֽעֱשֶׂה־לָּֽךְ׃ ה' אָמַר לְמֹשֶׁה: אֱמוֹר לִבְנֵי יִשְׂרָאֵל – אַתֶּם עַם עַקְשָׁן וְסַרְבָן! אִם אֶעֱלֶה בְּתוֹכְךָ בְּכַעֲסִי רֶגַע אֶחָד – אַשְׁמִיד אוֹתְךָ! וְעַכְשָׁיו הוֹרֵד אֶת תַּכְשִׁיטְךָ מֵעָלֶיךָ, וְיָדוּעַ לְפָנַי אֵיךְ לְהַעֲנִישׁ אוֹתְךָ.
ו וַיִּֽתְנַצְּל֧וּ בְנֵֽי־יִשְׂרָאֵ֛ל אֶת־עֶדְיָ֖ם מֵהַ֥ר חוֹרֵֽב׃ בְּנֵי יִשְׂרָאֵל הִסִּירוּ אֶת תַּכְשִׁיטֵיהֶם שֶׁהִתְקַשְּׁטוּ בָּהֶם מֵאָז מַעֲמַד הַר סִינַי.
ז וּמֹשֶׁה֩ יִקַּ֨ח אֶת־הָאֹ֜הֶל וְנָֽטָה־ל֣וֹ ׀ מִח֣וּץ לַֽמַּחֲנֶ֗ה הַרְחֵק֙ מִן־הַֽמַּחֲנֶ֔ה וְקָ֥רָא ל֖וֹ אֹ֣הֶל מוֹעֵ֑ד וְהָיָה֙ כׇּל־מְבַקֵּ֣שׁ יְהֹוָ֔ה יֵצֵא֙ אֶל־אֹ֣הֶל מוֹעֵ֔ד אֲשֶׁ֖ר מִח֥וּץ לַֽמַּחֲנֶֽה׃ וּמֹשֶׁה לָקַח אֶת הָאוֹהֶל וְהִקִּים אוֹתוֹ לְעַצְמוֹ מִחוּץ לַמַּחֲנֶה הַרְחֵק מֵהַמַּחֲנֶה, וְקָרָא לוֹ אוֹהֶל הַהִתְכַּנְּסוּת, כָּל מִי שֶׁרוֹצֶה אֶת ה' יוֹצֵא אֶל אוֹתוֹ אוֹהֶל הִתְכַּנְּסוּת שֶׁמִּחוּץ לַמַּחֲנֶה.
ח וְהָיָ֗ה כְּצֵ֤את מֹשֶׁה֙ אֶל־הָאֹ֔הֶל יָק֙וּמוּ֙ כׇּל־הָעָ֔ם וְנִ֨צְּב֔וּ אִ֖ישׁ פֶּ֣תַח אׇהֳל֑וֹ וְהִבִּ֙יטוּ֙ אַחֲרֵ֣י מֹשֶׁ֔ה עַד־בֹּא֖וֹ הָאֹֽהֱלָה׃ כְּשֶׁמֹּשֶׁה הָיָה יוֹצֵא לָאוֹהֶל כָּל הָעָם הָיוּ קָמִים וְעוֹמְדִים, כָּל אֶחָד בְּפֶתַח אָהֳלוֹ, וְהִבִּיטוּ אַחֲרֵי מֹשֶׁה עַד שֶׁנִּכְנַס לָאוֹהֶל.
ט וְהָיָ֗ה כְּבֹ֤א מֹשֶׁה֙ הָאֹ֔הֱלָה יֵרֵד֙ עַמּ֣וּד הֶֽעָנָ֔ן וְעָמַ֖ד פֶּ֣תַח הָאֹ֑הֶל וְדִבֶּ֖ר עִם־מֹשֶֽׁה׃ וּכְשֶׁמֹּשֶׁה הָיָה בָּא לָאוֹהֶל, עַמּוּד הֶעָנָן יָרַד וְעָמַד בְּפֶתַח הָאוֹהֶל, וְדִבֵּר עִם מֹשֶׁה.
י וְרָאָ֤ה כׇל־הָעָם֙ אֶת־עַמּ֣וּד הֶֽעָנָ֔ן עֹמֵ֖ד פֶּ֣תַח הָאֹ֑הֶל וְקָ֤ם כׇּל־הָעָם֙ וְהִֽשְׁתַּחֲו֔וּ אִ֖ישׁ פֶּ֥תַח אׇהֳלֽוֹ׃ כָּל הָעָם רָאָה אֶת עַמּוּד הֶעָנָן עוֹמֵד בְּפֶתַח הָאוֹהֶל, וְכָל הָעָם נֶעֶמְדוּ וְהִשְׁתַּחֲווּ, כָּל אֶחָד בְּפֶתַח אָהֳלוֹ.
יא וְדִבֶּ֨ר יְהֹוָ֤ה אֶל־מֹשֶׁה֙ פָּנִ֣ים אֶל־פָּנִ֔ים כַּאֲשֶׁ֛ר יְדַבֵּ֥ר אִ֖ישׁ אֶל־רֵעֵ֑הוּ וְשָׁב֙ אֶל־הַֽמַּחֲנֶ֔ה וּמְשָׁ֨רְת֜וֹ יְהוֹשֻׁ֤עַ בִּן־נוּן֙ נַ֔עַר לֹ֥א יָמִ֖ישׁ מִתּ֥וֹךְ הָאֹֽהֶל׃ ה' דִּבֵּר אֶל מֹשֶׁה פָּנִים אֶל פָּנִים כְּמוֹ שֶׁאִישׁ מְדַבֵּר אֶל חֲבֵרוֹ, וּמֹשֶׁה חָזַר לַמַּחֲנֶה, וּמְשָׁרְתוֹ יְהוֹשֻׁעַ בִּן נוּן – מְשָׁרֵת, לֹא זָז מִתּוֹךְ הָאוֹהֶל.

שלישי[עריכה]

פ פָּרָשַׁת בַּקָּשַׁת גִּלּוּי הַפָּנִים
יב וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶל־יְהֹוָ֗ה רְ֠אֵ֠ה אַתָּ֞ה אֹמֵ֤ר אֵלַי֙ הַ֚עַל אֶת־הָעָ֣ם הַזֶּ֔ה וְאַתָּה֙ לֹ֣א הֽוֹדַעְתַּ֔נִי אֵ֥ת אֲשֶׁר־תִּשְׁלַ֖ח עִמִּ֑י וְאַתָּ֤ה אָמַ֙רְתָּ֙ יְדַעְתִּ֣יךָֽ בְשֵׁ֔ם וְגַם־מָצָ֥אתָ חֵ֖ן בְּעֵינָֽי׃ מֹשֶׁה אָמַר לַה': רְאֵה, אַתָּה אוֹמֵר לִי "הַעֲלֶה אֶת הָעָם הַזֶּה לָאָרֶץ", וְלֹא הוֹדַעְתָּ לִי שֶׁתִּשְׁלַח אִתִּי רַק מַלְאָךְ, לְהֵפֶךְ, אָמַרְתָּ: "אֶתְגַּלֶּה אֵלֶיךָ בִּשְׁמִי הַמְּיוּחָד, וְגַם מָצָאתָ חֵן בְּעֵינַי".
יג וְעַתָּ֡ה אִם־נָא֩ מָצָ֨אתִי חֵ֜ן בְּעֵינֶ֗יךָ הוֹדִעֵ֤נִי נָא֙ אֶת־דְּרָכֶ֔ךָ וְאֵדָ֣עֲךָ֔ לְמַ֥עַן אֶמְצָא־חֵ֖ן בְּעֵינֶ֑יךָ וּרְאֵ֕ה כִּ֥י עַמְּךָ֖ הַגּ֥וֹי הַזֶּֽה׃ וְעַכְשָׁיו בְּבַקָּשָׁה אִם מָצָאתִי חֵן בְּעֵינֶיךָ לַמֵּד אוֹתִי בְּבַקָּשָׁה אֶת דַּרְכֵי הַהַנְהָגָה שֶׁלְּךָ, וְאַכִּיר אוֹתְךָ כְּדֵי שֶׁאֶמְצָא חֵן בְּעֵינֶיךָ, וַהֲרֵי הָעָם הַזֶּה הוּא עַמְּךָ, לָכֵן אַל תַּעֲזוֹב אוֹתָם.
יד וַיֹּאמַ֑ר פָּנַ֥י יֵלֵ֖כוּ וַהֲנִחֹ֥תִי לָֽךְ׃ ה' אָמַר: אֵלֶךְ בְּעַצְמִי עַד שֶׁאֶתֵּן לָכֶם מְנוּחָה בָּאָרֶץ.
טו וַיֹּ֖אמֶר אֵלָ֑יו אִם־אֵ֤ין פָּנֶ֙יךָ֙ הֹלְכִ֔ים אַֽל־תַּעֲלֵ֖נוּ מִזֶּֽה׃ אָמַר לוֹ: אִם אֵינְךָ הוֹלֵךְ בְּעַצְמְךָ אַל תַּעֲלֶה אוֹתָנוּ מִכָּאן.
טז וּבַמֶּ֣ה ׀ יִוָּדַ֣ע אֵפ֗וֹא כִּֽי־מָצָ֨אתִי חֵ֤ן בְּעֵינֶ֙יךָ֙ אֲנִ֣י וְעַמֶּ֔ךָ הֲל֖וֹא בְּלֶכְתְּךָ֣ עִמָּ֑נוּ וְנִפְלִ֙ינוּ֙ אֲנִ֣י וְעַמְּךָ֔ מִכׇּ֨ל־הָעָ֔ם אֲשֶׁ֖ר עַל־פְּנֵ֥י הָאֲדָמָֽה׃ שֶׁאִם לֹא כֵּן – בְּמָה אִם כֵּן יִוָּדַע שֶׁמָּצָאתִי חֵן בְּעֵינֶיךָ אֲנִי וְעַמְּךָ? הֲרֵי רַק בַּהֲלִיכַת שְׁכִינָתְךָ אִתָּנוּ! וְכָךְ אֲנִי וְעַמְּךָ נִהְיֶה מֻבְדָּלִים מִכָּל הָעַמִּים שֶׁעַל פְּנֵי הָאֲדָמָה.

רביעי[עריכה]

פ פָּרָשַׁת י"ג הַמִּדּוֹת
יז וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה גַּ֣ם אֶת־הַדָּבָ֥ר הַזֶּ֛ה אֲשֶׁ֥ר דִּבַּ֖רְתָּ אֶֽעֱשֶׂ֑ה כִּֽי־מָצָ֤אתָ חֵן֙ בְּעֵינַ֔י וָאֵדָעֲךָ֖ בְּשֵֽׁם׃ ה' אָמַר לְמֹשֶׁה: גַּם אֶת הַדָּבָר הַזֶּה שֶׁבִּקַּשְׁתָּ אֶעֱשֶׂה כִּי מָצָאתָ חֵן בְּעֵינַי וְהִתְגַּלֵּיתִי אֵלֶיךָ בִּשְׁמִי הַמְּפוֹרָשׁ.
יח וַיֹּאמַ֑ר הַרְאֵ֥נִי נָ֖א אֶת־כְּבֹדֶֽךָ׃ אָמַר: הַרְאֵנִי בְּבַקָּשָׁה אֶת כְּבוֹדְךָ!
יט וַיֹּ֗אמֶר אֲנִ֨י אַעֲבִ֤יר כׇּל־טוּבִי֙ עַל־פָּנֶ֔יךָ וְקָרָ֧אתִֽי בְשֵׁ֛ם יְהֹוָ֖ה לְפָנֶ֑יךָ וְחַנֹּתִי֙ אֶת־אֲשֶׁ֣ר אָחֹ֔ן וְרִחַמְתִּ֖י אֶת־אֲשֶׁ֥ר אֲרַחֵֽם׃ אָמַר: אֲנִי אַעֲבִיר אֶת כָּל מִדּוֹתַי הַטּוֹבוֹת לְפָנֶיךָ, וְאַסְבִּיר אֶת הִתְגַּלּוּת ה' לְפָנֶיךָ, וּבְהִתְגַּלּוֹתוֹ אֶתֵּן חֲנִינָה לְמִי שֶׁאֶתֵּן, וַאֲרַחֵם עַל מִי שֶׁאֲרַחֵם.
כ וַיֹּ֕אמֶר לֹ֥א תוּכַ֖ל לִרְאֹ֣ת אֶת־פָּנָ֑י כִּ֛י לֹֽא־יִרְאַ֥נִי הָאָדָ֖ם וָחָֽי׃ אָמַר: לֹא תּוּכַל לִרְאוֹת אֶת גִּלּוּי שְׁכִינָתִי, כִּי אָדָם לֹא יָכוֹל לִרְאוֹת אוֹתִי וּלְהַמְשִׁיךְ לִחְיוֹת.
כא וַיֹּ֣אמֶר יְהֹוָ֔ה הִנֵּ֥ה מָק֖וֹם אִתִּ֑י וְנִצַּבְתָּ֖ עַל־הַצּֽוּר׃ ה' אָמַר: הִנֵּה יֵשׁ לִי מָקוֹם לְכָךְ, תַּעֲמוֹד עַל הַסֶּלַע.
כב וְהָיָה֙ בַּעֲבֹ֣ר כְּבֹדִ֔י וְשַׂמְתִּ֖יךָ בְּנִקְרַ֣ת הַצּ֑וּר וְשַׂכֹּתִ֥י כַפִּ֛י עָלֶ֖יךָ עַד־עׇבְרִֽי׃ כְּשֶׁכְּבוֹדִי יַעֲבוֹר אָשִׂים אוֹתְךָ בִּמְעָרַת הַסֶּלַע, וְאָגֵן בְּיָדִי עָלֶיךָ עַד שֶׁאֶעֱבוֹר.
כג וַהֲסִרֹתִי֙ אֶת־כַּפִּ֔י וְרָאִ֖יתָ אֶת־אֲחֹרָ֑י וּפָנַ֖י לֹ֥א יֵרָאֽוּ׃ וְאָסִיר אֶת יָדִי וְתִרְאֶה אֶת גִּלּוּי מַעֲשַׂי, אֲבָל גִּלּוּי שְׁכִינָתִי עַצְמָהּ לֹא יֵרָאֶה.

חמישי[עריכה]

פ
לד א וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה פְּסׇל־לְךָ֛ שְׁנֵֽי־לֻחֹ֥ת אֲבָנִ֖ים כָּרִאשֹׁנִ֑ים וְכָתַבְתִּי֙ עַל־הַלֻּחֹ֔ת אֶ֨ת־הַדְּבָרִ֔ים אֲשֶׁ֥ר הָי֛וּ עַל־הַלֻּחֹ֥ת הָרִאשֹׁנִ֖ים אֲשֶׁ֥ר שִׁבַּֽרְתָּ׃ ה' אָמַר לְמֹשֶׁה: פְּסוֹל לְךָ שְׁנֵי לוּחוֹת אֲבָנִים כְּמוֹ הָרִאשׁוֹנִים וְאֶכְתּוֹב עַל הַלּוּחוֹת אֶת הַדְּבָרִים שֶׁהָיוּ עַל הַלּוּחוֹת הָרִאשׁוֹנִים שֶׁשָּׁבַרְתָּ.
ב וֶהְיֵ֥ה נָכ֖וֹן לַבֹּ֑קֶר וְעָלִ֤יתָ בַבֹּ֙קֶר֙ אֶל־הַ֣ר סִינַ֔י וְנִצַּבְתָּ֥ לִ֛י שָׁ֖ם עַל־רֹ֥אשׁ הָהָֽר׃ וְתִהְיֶה מוּכָן לַבּוֹקֶר וְתַעֲלֶה בַּבּוֹקֶר לְהַר סִינַי וְתַעֲמוֹד אֶצְלִי שָׁם עַל רֹאשׁ הָהָר.
ג וְאִישׁ֙ לֹֽא־יַעֲלֶ֣ה עִמָּ֔ךְ וְגַם־אִ֥ישׁ אַל־יֵרָ֖א בְּכׇל־הָהָ֑ר גַּם־הַצֹּ֤אן וְהַבָּקָר֙ אַל־יִרְע֔וּ אֶל־מ֖וּל הָהָ֥ר הַהֽוּא׃ וְאִישׁ לֹא יַעֲלֶה אִתְּךָ וְגַם אִישׁ אַל יֵרָאֶה בְּכָל הָהָר, גַּם הַצֹּאן וְהַבָּקָר לֹא יִרְעוּ מוּל אוֹתוֹ הָהָר.
ד וַיִּפְסֹ֡ל שְׁנֵֽי־לֻחֹ֨ת אֲבָנִ֜ים כָּרִאשֹׁנִ֗ים וַיַּשְׁכֵּ֨ם מֹשֶׁ֤ה בַבֹּ֙קֶר֙ וַיַּ֙עַל֙ אֶל־הַ֣ר סִינַ֔י כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֹת֑וֹ וַיִּקַּ֣ח בְּיָד֔וֹ שְׁנֵ֖י לֻחֹ֥ת אֲבָנִֽים׃ הוּא פָּסַל שְׁנֵי לוּחוֹת אֲבָנִים כְּמוֹ הָרִאשׁוֹנִים, מֹשֶׁה הִשְׁכִּים בַּבּוֹקֶר וְעָלָה לְהַר סִינַי כְּמוֹ שֶׁה' צִוָּה אוֹתוֹ, וְהוּא לָקַח בְּיָדוֹ שְׁנֵי לוּחוֹת אֲבָנִים.
ה וַיֵּ֤רֶד יְהֹוָה֙ בֶּֽעָנָ֔ן וַיִּתְיַצֵּ֥ב עִמּ֖וֹ שָׁ֑ם וַיִּקְרָ֥א בְשֵׁ֖ם יְהֹוָֽה׃ ה' יָרַד בֶּעָנָן וְעָמַד אִתּוֹ שָׁם וְהִסְבִּיר אֶת הִתְגַּלּוּת ה' –
ו וַיַּעֲבֹ֨ר יְהֹוָ֥ה ׀ עַל־פָּנָיו֮ וַיִּקְרָא֒ יְהֹוָ֣ה ׀ יְהֹוָ֔ה אֵ֥ל רַח֖וּם וְחַנּ֑וּן אֶ֥רֶךְ אַפַּ֖יִם וְרַב־חֶ֥סֶד וֶאֱמֶֽת׃ ה' עָבַר עַל פָּנָיו שֶׁל מֹשֶׁה וְקָרָא: ה' ה' אֵל רַחֲמָן וּמַעֲנִיק חֲנִינָה, סַבְלָן מִלִּכְעוֹס, וּמַרְבֶּה בְּחֶסֶד וּמִתְנַהֵג בֶּאֱמֶת.
ז נֹצֵ֥ר‏[1] חֶ֙סֶד֙ לָאֲלָפִ֔ים נֹשֵׂ֥א עָוֺ֛ן וָפֶ֖שַׁע וְחַטָּאָ֑ה וְנַקֵּה֙ לֹ֣א יְנַקֶּ֔ה פֹּקֵ֣ד ׀ עֲוֺ֣ן אָב֗וֹת עַל־בָּנִים֙ וְעַל־בְּנֵ֣י בָנִ֔ים עַל־שִׁלֵּשִׁ֖ים וְעַל־רִבֵּעִֽים׃ זוֹכֵר לְהֵטִיב לַטּוֹבִים – עַד אַלְפֵי דּוֹרוֹת, מְוַתֵּר בֵּינְתַיִם מִלְּהַעֲנִישׁ עַל עָווֹן עַל פֶּשַׁע וְעַל חֵטְא, אֲבָל לֹא יִמְחוֹק אוֹתָם, זוֹכֵר עֲווֹן אָבוֹת לְהַעֲנִישׁ בָּנִים וּבְנֵי בָּנִים – נְכָדִים וְנִינִים.
ח וַיְמַהֵ֖ר מֹשֶׁ֑ה וַיִּקֹּ֥ד אַ֖רְצָה וַיִּשְׁתָּֽחוּ׃ מֹשֶׁה מִהֵר וְקָד אַרְצָה וְהִשְׁתַּחֲוָה.
ט וַיֹּ֡אמֶר אִם־נָא֩ מָצָ֨אתִי חֵ֤ן בְּעֵינֶ֙יךָ֙ אֲדֹנָ֔י יֵֽלֶךְ־נָ֥א אֲדֹנָ֖י בְּקִרְבֵּ֑נוּ כִּ֤י עַם־קְשֵׁה־עֹ֙רֶף֙ ה֔וּא וְסָלַחְתָּ֛ לַעֲוֺנֵ֥נוּ וּלְחַטָּאתֵ֖נוּ וּנְחַלְתָּֽנוּ׃ וְאָמַר: ה', בְּבַקָּשָׁה אִם מָצָאתִי חֵן בְּעֵינֶיךָ, שֶׁה' יֵלֶךְ בְּתוֹכֵנוּ בְּבַקָּשָׁה, כִּי הוּא עַם עַקְשָׁן וְסַרְבָן, וְתִסְלַח לַעֲוֹנֵנוּ וּלְחַטָּאתֵנוּ וְתִּתֵּן לָנוּ נַחֲלָה.

ששי[עריכה]

י וַיֹּ֗אמֶר הִנֵּ֣ה אָנֹכִי֮ כֹּרֵ֣ת בְּרִית֒ נֶ֤גֶד כׇּֽל־עַמְּךָ֙ אֶעֱשֶׂ֣ה נִפְלָאֹ֔ת אֲשֶׁ֛ר לֹֽא־נִבְרְא֥וּ בְכׇל־הָאָ֖רֶץ וּבְכׇל־הַגּוֹיִ֑ם וְרָאָ֣ה כׇל־הָ֠עָ֠ם אֲשֶׁר־אַתָּ֨ה בְקִרְבּ֜וֹ אֶת־מַעֲשֵׂ֤ה יְהֹוָה֙ כִּֽי־נוֹרָ֣א ה֔וּא אֲשֶׁ֥ר אֲנִ֖י עֹשֶׂ֥ה עִמָּֽךְ׃ אָמַר: הִנֵּה אֲנִי עוֹשֶׂה הֶסְכֵּם, מוּל כָּל עַמְּךָ אֶעֱשֶׂה נִפְלָאוֹת שֶׁלֹּא נִבְרְאוּ בְּכָל הָאָרֶץ וּבְכָל הָעַמִּים, וְכָל הָעָם שֶׁאַתָּה בְּתוֹכוֹ יִרְאֶה אֶת מַעֲשֵׂה ה' שֶׁאֲנִי עוֹשֶׂה אִתְּךָ – שֶׁהוּא נוֹרָא.
פ פָּרָשַׁת הַמִּצְווֹת
יא שְׁמׇ֨ר־לְךָ֔ אֵ֛ת אֲשֶׁ֥ר אָנֹכִ֖י מְצַוְּךָ֣ הַיּ֑וֹם הִנְנִ֧י גֹרֵ֣שׁ מִפָּנֶ֗יךָ אֶת־הָאֱמֹרִי֙ וְהַֽכְּנַעֲנִ֔י וְהַחִתִּי֙ וְהַפְּרִזִּ֔י וְהַחִוִּ֖י וְהַיְבוּסִֽי׃ זְכוֹר לְךָ אֶת מַה שֶּׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם, אֲגָרֵשׁ מִלְּפָנֶיךָ אֶת הָאֱמוֹרִי וְהַכְּנַעֲנִי וְהַחִתִּי וְהַפְּרִזִּי וְהַחִוִּי וְהַיְּבוּסִי.
יב הִשָּׁ֣מֶר לְךָ֗ פֶּן־תִּכְרֹ֤ת בְּרִית֙ לְיוֹשֵׁ֣ב הָאָ֔רֶץ אֲשֶׁ֥ר אַתָּ֖ה בָּ֣א עָלֶ֑יהָ פֶּן־יִהְיֶ֥ה לְמוֹקֵ֖שׁ בְּקִרְבֶּֽךָ׃ הִזָּהֵר לְךָ שֶׁלֹּא תַּעֲשֶׂה הֶסְכֵּם עִם יוֹשְׁבֵי הָאָרֶץ שֶׁאַתָּה בָּא אֵלֶיהָ, כְּדֵי שֶׁלֹּא יִהְיֶה מִכְשׁוֹל בְּתוֹכְךָ.
יג כִּ֤י אֶת־מִזְבְּחֹתָם֙ תִּתֹּצ֔וּן וְאֶת־מַצֵּבֹתָ֖ם תְּשַׁבֵּר֑וּן וְאֶת־אֲשֵׁרָ֖יו תִּכְרֹתֽוּן׃ אֶלָּא, תְּנַפְּצוּ אֶת מִזְבְּחוֹתֵיהֶם וְתִשְׁבְּרוּ אֶת מַצֵּבוֹתֵיהֶם וְתִכְרְתוּ אֶת אֲשֵׁרוֹתֵיהֶם.
יד כִּ֛י לֹ֥א תִֽשְׁתַּחֲוֶ֖ה לְאֵ֣ל אַחֵ֑ר כִּ֤י יְהֹוָה֙ קַנָּ֣א שְׁמ֔וֹ אֵ֥ל קַנָּ֖א הֽוּא׃ כִּי לֹא תִּשְׁתַּחֲוֶה לְאֵל אַחֵר כִּי ה' – שְׁמוֹ "נַקְמָן" – הוּא אֵל נוֹקֵם.
טו פֶּן־תִּכְרֹ֥ת בְּרִ֖ית לְיוֹשֵׁ֣ב הָאָ֑רֶץ וְזָנ֣וּ ׀ אַחֲרֵ֣י אֱלֹֽהֵיהֶ֗ם וְזָבְחוּ֙ לֵאלֹ֣הֵיהֶ֔ם וְקָרָ֣א לְךָ֔ וְאָכַלְתָּ֖ מִזִּבְחֽוֹ׃ הִזָּהֵר שֶׁלֹּא תַּעֲשֶׂה הֶסְכֵּם עִם יוֹשְׁבֵי הָאָרֶץ, וְהֵם יִבְגְּדוּ בִּי וְיֵלְכוּ אַחֲרֵי אֱלִילֵיהֶם וְיִזְבְּחוּ לֶאֱלִילֵיהֶם וְיִקְרְאוּ לְךָ וְתֹאכַל מִזִּבְחֵיהֶם.
טז וְלָקַחְתָּ֥ מִבְּנֹתָ֖יו לְבָנֶ֑יךָ וְזָנ֣וּ בְנֹתָ֗יו אַחֲרֵי֙ אֱלֹ֣הֵיהֶ֔ן וְהִזְנוּ֙ אֶת־בָּנֶ֔יךָ אַחֲרֵ֖י אֱלֹהֵיהֶֽן׃ וְתִקַּח מִבְּנוֹתָיו לְבָנֶיךָ וּבְנוֹתָיו יִבְגְּדוּ בִּי וְיֵלְכוּ אַחֲרֵי אֱלִילֵיהֶן וְיִמְשְׁכוּ אֶת בָּנֶיךָ אַחֲרֵי אֱלִילֵיהֶן.
יז אֱלֹהֵ֥י מַסֵּכָ֖ה לֹ֥א תַעֲשֶׂה־לָּֽךְ׃ אַל תַּעֲשֶׂה לְךָ פִּסְלֵי מַתֶּכֶת.
יח אֶת־חַ֣ג הַמַּצּוֹת֮ תִּשְׁמֹר֒ שִׁבְעַ֨ת יָמִ֜ים תֹּאכַ֤ל מַצּוֹת֙ אֲשֶׁ֣ר צִוִּיתִ֔ךָ לְמוֹעֵ֖ד חֹ֣דֶשׁ הָאָבִ֑יב כִּ֚י בְּחֹ֣דֶשׁ הָֽאָבִ֔יב יָצָ֖אתָ מִמִּצְרָֽיִם׃ זְכוֹר לְקַיֵּם אֶת חַג הַמַּצּוֹת, שִׁבְעָה יָמִים תֹּאכַל מַצּוֹת כְּמוֹ שֶׁצִּוִּיתִי אוֹתְךָ – בִּזְמַן חוֹדֶשׁ הָאָבִיב, כִּי בְּחוֹדֶשׁ הָאָבִיב יָצָאתָ מִמִּצְרַיִם.
יט כׇּל־פֶּ֥טֶר רֶ֖חֶם לִ֑י וְכׇֽל־מִקְנְךָ֙ תִּזָּכָ֔ר פֶּ֖טֶר שׁ֥וֹר וָשֶֽׂה׃ כָּל בְּכוֹר הוּא שֶׁלִּי, וְכֵן בְּכָל מִקְנְךָ – כְּשֶׁבְּכוֹר שׁוֹר וְשֶׂה יִהְיֶה זָכָר הוּא יִהְיֶה שֶׁלִּי.
כ וּפֶ֤טֶר חֲמוֹר֙ תִּפְדֶּ֣ה בְשֶׂ֔ה וְאִם־לֹ֥א תִפְדֶּ֖ה וַעֲרַפְתּ֑וֹ כֹּ֣ל בְּכ֤וֹר בָּנֶ֙יךָ֙ תִּפְדֶּ֔ה וְלֹֽא־יֵרָא֥וּ פָנַ֖י רֵיקָֽם׃ וּבְכוֹר חֲמוֹר תִּפְדֶּה תְּמוּרַת טָלֶה אוֹ שֶׂה, וְאִם לֹא תַּחֲלִיף – תַּעֲרוֹף אֶת רֹאשׁוֹ, כָּל בְּכוֹר מִבָּנֶיךָ תִּפְדֶּה, וְאַל תִּרְאֶה אֶת פָּנַי בָּעֲלִיָּה לָרֶגֶל לְלֹא קָרְבָּן.
כא שֵׁ֤שֶׁת יָמִים֙ תַּעֲבֹ֔ד וּבַיּ֥וֹם הַשְּׁבִיעִ֖י תִּשְׁבֹּ֑ת בֶּחָרִ֥ישׁ וּבַקָּצִ֖יר תִּשְׁבֹּֽת׃ שִׁשָּׁה יָמִים תַּעֲבוֹד וּבַיּוֹם הַשְּׁבִיעִי לֹא תַּעֲבוֹד, גַּם בִּזְמַן הֶחָרִישׁ וְהַקָּצִיר הַחֲשׁוּבִים תִּשְׁבּוֹת.
כב וְחַ֤ג שָׁבֻעֹת֙ תַּעֲשֶׂ֣ה לְךָ֔ בִּכּוּרֵ֖י קְצִ֣יר חִטִּ֑ים וְחַג֙ הָֽאָסִ֔יף תְּקוּפַ֖ת הַשָּׁנָֽה׃ וְתַעֲשֶׂה לְךָ חַג שָׁבוּעוֹת בִּזְמַן תְּחִלַּת קְצִיר חִטִּים, וְחַג הָאָסִיף בְּסוֹף שְׁנַת אִסּוּף הַפֵּרוֹת.
כג שָׁלֹ֥שׁ פְּעָמִ֖ים בַּשָּׁנָ֑ה יֵרָאֶה֙ כׇּל־זְכ֣וּרְךָ֔ אֶת־פְּנֵ֛י הָֽאָדֹ֥ן ׀ יְהֹוָ֖ה אֱלֹהֵ֥י יִשְׂרָאֵֽל׃ שָׁלוֹשׁ פְּעָמִים בַּשָּׁנָה יֵרָאוּ כָּל זְכָרֶיךָ לִפְנֵי הָאָדוֹן ה' אֱלֹהֵי יִשְׂרָאֵל.
כד כִּֽי־אוֹרִ֤ישׁ גּוֹיִם֙ מִפָּנֶ֔יךָ וְהִרְחַבְתִּ֖י אֶת־גְּבֻלֶ֑ךָ וְלֹא־יַחְמֹ֥ד אִישׁ֙ אֶֽת־אַרְצְךָ֔ בַּעֲלֹֽתְךָ֗ לֵרָאוֹת֙ אֶת־פְּנֵי֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ שָׁלֹ֥שׁ פְּעָמִ֖ים בַּשָּׁנָֽה׃ כְּשֶׁאֶכְבּוֹשׁ גּוֹיִם לְפָנֶיךָ וְאַרְחִיב אֶת גְּבוּלְךָ, אַף אֶחָד לֹא יִרְצֶה לָקַחַת מִמְּךָ אֶת אַרְצְךָ כְּשֶׁתַּעֲלֶה לְהֵרָאוֹת לִפְנֵי ה' אֱלֹהֶיךָ שָׁלוֹשׁ פְּעָמִים בַּשָּׁנָה.
כה לֹֽא־תִשְׁחַ֥ט עַל־חָמֵ֖ץ דַּם־זִבְחִ֑י וְלֹא־יָלִ֣ין לַבֹּ֔קֶר זֶ֖בַח חַ֥ג הַפָּֽסַח׃ אַל תִּשְׁחַט אֶת קָרְבַּן הַפֶּסַח שֶׁדָּמוֹ נִזְרַק לַמִּזְבֵּחַ כְּשֶׁיֵּשׁ לְךָ חָמֵץ, וְאַל תַּשְׁאִיר עַד הַבּוֹקֶר אֶת זֶבַח חַג הַפֶּסַח.
כו רֵאשִׁ֗ית בִּכּוּרֵי֙ אַדְמָ֣תְךָ֔ תָּבִ֕יא בֵּ֖ית יְהֹוָ֣ה אֱלֹהֶ֑יךָ לֹא־תְבַשֵּׁ֥ל גְּדִ֖י בַּחֲלֵ֥ב אִמּֽוֹ׃ אֶת תְּחִלַּת בִּכּוּרֵי אַדְמָתְךָ תָּבִיא לְבֵית ה' אֱלֹהֶיךָ, אַל תְּבַשֵּׁל וְאַל תֹּאכַל בְּשַׂר בְּהֵמָה כְּשֵׁרָה בְּחָלָב בְּהֵמָה כְּשֵׁרָה.

שביעי[עריכה]

פ פָּרָשַׁת הַמַּסְוֶה
כז וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה כְּתׇב־לְךָ֖ אֶת־הַדְּבָרִ֣ים הָאֵ֑לֶּה כִּ֞י עַל־פִּ֣י ׀ הַדְּבָרִ֣ים הָאֵ֗לֶּה כָּרַ֧תִּי אִתְּךָ֛ בְּרִ֖ית וְאֶת־יִשְׂרָאֵֽל׃ ה' אָמַר לְמֹשֶׁה: כְּתוֹב לְךָ אֶת הַדְּבָרִים הָאֵלֶּה, כִּי עַל הַדְּבָרִים הָאֵלֶּה עָשִׂיתִי אִתְּךָ וְעִם יִשְׂרָאֵל הֶסְכֵּם בְּרִית.
כח וַֽיְהִי־שָׁ֣ם עִם־יְהֹוָ֗ה אַרְבָּעִ֥ים יוֹם֙ וְאַרְבָּעִ֣ים לַ֔יְלָה לֶ֚חֶם לֹ֣א אָכַ֔ל וּמַ֖יִם לֹ֣א שָׁתָ֑ה וַיִּכְתֹּ֣ב עַל־הַלֻּחֹ֗ת אֵ֚ת דִּבְרֵ֣י הַבְּרִ֔ית עֲשֶׂ֖רֶת הַדְּבָרִֽים׃ מֹשֶׁה הָיָה שָׁם עִם ה' אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה, לֹא אָכַל לֶחֶם וְלֹא שָׁתָה מַיִם, הוּא כָּתַב עַל הַלּוּחוֹת אֶת דִּבְרֵי הַהֶסְכֵּם – אֶת עֲשֶׂרֶת הַדִּבְּרוֹת.
כט וַיְהִ֗י בְּרֶ֤דֶת מֹשֶׁה֙ מֵהַ֣ר סִינַ֔י וּשְׁנֵ֨י לֻחֹ֤ת הָֽעֵדֻת֙ בְּיַד־מֹשֶׁ֔ה בְּרִדְתּ֖וֹ מִן־הָהָ֑ר וּמֹשֶׁ֣ה לֹֽא־יָדַ֗ע כִּ֥י קָרַ֛ן ע֥וֹר פָּנָ֖יו בְּדַבְּר֥וֹ אִתּֽוֹ׃ כְּשֶׁמֹּשֶׁה יָרַד מֵהַר סִינַי – שְׁנֵי לוּחוֹת הַבְּרִית הָיוּ בְּיַד מֹשֶׁה כְּשֶׁיָּרַד מֵהָהָר, וּמֹשֶׁה לֹא יָדַע שֶׁעוֹר פָּנָיו זָרַח כְּשֶׁה' דִּבֵּר אִתּוֹ.
ל וַיַּ֨רְא אַהֲרֹ֜ן וְכׇל־בְּנֵ֤י יִשְׂרָאֵל֙ אֶת־מֹשֶׁ֔ה וְהִנֵּ֥ה קָרַ֖ן ע֣וֹר פָּנָ֑יו וַיִּֽירְא֖וּ מִגֶּ֥שֶׁת אֵלָֽיו׃ אַהֲרֹן וְכָל בְּנֵי יִשְׂרָאֵל רָאוּ אֶת מֹשֶׁה וְהִנֵּה זָרַח עוֹר פָּנָיו, וּפָחֲדוּ לָגֶשֶׁת אֵלָיו.
לא וַיִּקְרָ֤א אֲלֵהֶם֙ מֹשֶׁ֔ה וַיָּשֻׁ֧בוּ אֵלָ֛יו אַהֲרֹ֥ן וְכׇל־הַנְּשִׂאִ֖ים בָּעֵדָ֑ה וַיְדַבֵּ֥ר מֹשֶׁ֖ה אֲלֵהֶֽם׃ מֹשֶׁה קָרָא לָהֶם, וְאַהֲרֹן וְכָל הַנְּשִׂיאִים שֶׁבַּצִּבּוּר חָזְרוּ אֵלָיו, וּמֹשֶׁה דִּבֵּר אֲלֵיהֶם.
לב וְאַחֲרֵי־כֵ֥ן נִגְּשׁ֖וּ כׇּל־בְּנֵ֣י יִשְׂרָאֵ֑ל וַיְצַוֵּ֕ם אֵת֩ כׇּל־אֲשֶׁ֨ר דִּבֶּ֧ר יְהֹוָ֛ה אִתּ֖וֹ בְּהַ֥ר סִינָֽי׃ וְאַחֲרֵי כֵן נִגְּשׁוּ כָּל בְּנֵי יִשְׂרָאֵל וְהוּא צִוָּה אוֹתָם אֶת כָּל מַה שֶּׁה' דִּבֵּר אִתּוֹ בְּהַר סִינַי.

מפטיר[עריכה]

לג וַיְכַ֣ל מֹשֶׁ֔ה מִדַּבֵּ֖ר אִתָּ֑ם וַיִּתֵּ֥ן עַל־פָּנָ֖יו מַסְוֶֽה׃ מֹשֶׁה סִיֵּם לְדַבֵּר אִתָּם וְשָׂם מַסֵּכָה עַל פָּנָיו.
לד וּבְבֹ֨א מֹשֶׁ֜ה לִפְנֵ֤י יְהֹוָה֙ לְדַבֵּ֣ר אִתּ֔וֹ יָסִ֥יר אֶת־הַמַּסְוֶ֖ה עַד־צֵאת֑וֹ וְיָצָ֗א וְדִבֶּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל אֵ֖ת אֲשֶׁ֥ר יְצֻוֶּֽה׃ כְּשֶׁמֹּשֶׁה נִכְנָס לִפְנֵי ה' לְדַבֵּר אִתּוֹ, הוּא מֵסִיר אֶת הַמַּסֵּכָה עַד צֵאתוֹ, וְהוּא יוֹצֵא וּמְדַבֵּר אֶל בְּנֵי יִשְׂרָאֵל אֶת מַה שֶּׁהוּא מְצֻוֶּה.
לה וְרָא֤וּ בְנֵֽי־יִשְׂרָאֵל֙ אֶת־פְּנֵ֣י מֹשֶׁ֔ה כִּ֣י קָרַ֔ן ע֖וֹר פְּנֵ֣י מֹשֶׁ֑ה וְהֵשִׁ֨יב מֹשֶׁ֤ה אֶת־הַמַּסְוֶה֙ עַל־פָּנָ֔יו עַד־בֹּא֖וֹ לְדַבֵּ֥ר אִתּֽוֹ׃ בְּנֵי יִשְׂרָאֵל רוֹאִים אֶת פְּנֵי מֹשֶׁה שֶׁעוֹר פָּנָיו שֶׁל מֹשֶׁה זוֹרֵחַ, וּמֹשֶׁה מַחֲזִיר אֶת הַמַּסֵּכָה אֶל פָּנָיו עַד שֶׁהוּא נִכְנַָס לְדַבֵּר עִם ה'.
