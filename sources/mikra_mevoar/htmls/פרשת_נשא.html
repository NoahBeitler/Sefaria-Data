ראשון[עריכה]

פ פָּרָשַׁת בְּנֵי גֵרְשׁוֹן
ד כא וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
כב נָשֹׂ֗א אֶת־רֹ֛אשׁ בְּנֵ֥י גֵרְשׁ֖וֹן גַּם־הֵ֑ם לְבֵ֥ית אֲבֹתָ֖ם לְמִשְׁפְּחֹתָֽם׃ סִפְרוּ גַּם אֶת בְּנֵי גֵּרְשׁוֹן בַּחֲלוּקָה לְפִי בָּתֵּי אֲבוֹתֵיהֶם וּלְפִי מִשְׁפְּחוֹתֵיהֶם.
כג מִבֶּן֩ שְׁלֹשִׁ֨ים שָׁנָ֜ה וָמַ֗עְלָה עַ֛ד בֶּן־חֲמִשִּׁ֥ים שָׁנָ֖ה תִּפְקֹ֣ד אוֹתָ֑ם כׇּל־הַבָּא֙ לִצְבֹ֣א צָבָ֔א לַעֲבֹ֥ד עֲבֹדָ֖ה בְּאֹ֥הֶל מוֹעֵֽד׃ תִּסְפּוֹר אוֹתָם מִגִּיל שְׁלוֹשִׁים שָׁנָה וָמַעְלָה עַד גִּיל חֲמִשִּׁים שָׁנָה, כָּל הַבָּאִים לְהִצְטָרֵף לַקָּהָל שֶׁתַּפְקִידָם לַעֲבוֹד עֲבוֹדָה בְּאוֹהֶל הַהִתְכַּנְּסוּת.
כד זֹ֣את עֲבֹדַ֔ת מִשְׁפְּחֹ֖ת הַגֵּרְשֻׁנִּ֑י לַעֲבֹ֖ד וּלְמַשָּֽׂא׃ זֹאת עֲבוֹדַת מִשְׁפְּחוֹת הַגֵּרְשֻׁנִּי – לַעֲבוֹד וְלִסְחוֹב מַשָּׂאוֹת.
כה וְנָ֨שְׂא֜וּ אֶת־יְרִיעֹ֤ת הַמִּשְׁכָּן֙ וְאֶת־אֹ֣הֶל מוֹעֵ֔ד מִכְסֵ֕הוּ וּמִכְסֵ֛ה הַתַּ֥חַשׁ אֲשֶׁר־עָלָ֖יו מִלְמָ֑עְלָה וְאֶ֨ת־מָסַ֔ךְ פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ הֵם יִשְׂאוּ אֶת יְרִיעוֹת הַמִּשְׁכָּן, וְאֶת אוֹהֶל הַהִתְכַּנְּסוּת, אֶת כִּסּוּיוֹ וְאֶת הַכִּסּוּי שֶׁעָשׂוּי מֵעוֹר הַחַיָּה תַּחַשׁ שֶׁעָלָיו מִלְּמַעְלָה, וְאֶת מָסַךְ פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.
כו וְאֵת֩ קַלְעֵ֨י הֶֽחָצֵ֜ר וְאֶת־מָסַ֣ךְ ׀ פֶּ֣תַח ׀ שַׁ֣עַר הֶחָצֵ֗ר אֲשֶׁ֨ר עַל־הַמִּשְׁכָּ֤ן וְעַל־הַמִּזְבֵּ֙חַ֙ סָבִ֔יב וְאֵת֙ מֵֽיתְרֵיהֶ֔ם וְאֶֽת־כׇּל־כְּלֵ֖י עֲבֹדָתָ֑ם וְאֵ֨ת כׇּל־אֲשֶׁ֧ר יֵעָשֶׂ֛ה לָהֶ֖ם וְעָבָֽדוּ׃ וְאֶת הַגְּדֵרוֹת הַקְּלוּעוֹת שֶׁל הֶחָצֵר, וְאֶת מָסַךְ פֶּתַח שַׁעַר הֶחָצֵר שֶׁסְּבִיב הַמִּשְׁכָּן וְהַמִּזְבֵּחַ, וְאֶת מֵיתְרֵיהֶם, וְאֶת כָּל כְּלֵי עֲבוֹדָתָם, וְאֶת כָּל הַחֲפָצִים הָעֲשׂוּיִים לַדְּבָרִים הָאֵלֶּה לְצוֹרֶךְ הָעֲבוֹדָה.
כז עַל־פִּי֩ אַהֲרֹ֨ן וּבָנָ֜יו תִּהְיֶ֗ה כׇּל־עֲבֹדַת֙ בְּנֵ֣י הַגֵּרְשֻׁנִּ֔י לְכׇ֨ל־מַשָּׂאָ֔ם וּלְכֹ֖ל עֲבֹדָתָ֑ם וּפְקַדְתֶּ֤ם עֲלֵהֶם֙ בְּמִשְׁמֶ֔רֶת אֵ֖ת כׇּל־מַשָּׂאָֽם׃ כָּל עֲבוֹדַת בְּנֵי גֵּרְשׁוֹן – כָּל מַשָּׂאָם וְכָל עֲבוֹדָתָם יִהְיוּ עַל פִּי פְּקֻדּוֹת אַהֲרֹן וּבָנָיו, תְּמַנּוּ אוֹתָם עַל תַּפְקִידֵיהֶם – עַל כָּל מַשָּׂאָם.
כח זֹ֣את עֲבֹדַ֗ת מִשְׁפְּחֹ֛ת בְּנֵ֥י הַגֵּרְשֻׁנִּ֖י בְּאֹ֣הֶל מוֹעֵ֑ד וּמִ֨שְׁמַרְתָּ֔ם בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן־אַהֲרֹ֖ן הַכֹּהֵֽן׃ זֹאת עֲבוֹדַת מִשְׁפְּחוֹת בְּנֵי הַגֵּרְשֻׁנִּי בְּאוֹהֶל הַהִתְכַּנְּסוּת, וְהֵם מְמֻנִּים תַּחַת פִּקּוּד אִיתָמָר בֶּן אַהֲרֹן הַכֹּהֵן.
ס פָּרָשַׁת בְּנֵי מְרָרִי
כט בְּנֵ֖י מְרָרִ֑י לְמִשְׁפְּחֹתָ֥ם לְבֵית־אֲבֹתָ֖ם תִּפְקֹ֥ד אֹתָֽם׃ סְפוֹר אֶת בְּנֵי מְרָרִי בַּחֲלוּקָה לְפִי מִשְׁפְּחוֹתֵיהֶם וּבָתֵּי אֲבוֹתֵיהֶם.
ל מִבֶּן֩ שְׁלֹשִׁ֨ים שָׁנָ֜ה וָמַ֗עְלָה וְעַ֛ד בֶּן־חֲמִשִּׁ֥ים שָׁנָ֖ה תִּפְקְדֵ֑ם כׇּל־הַבָּא֙ לַצָּבָ֔א לַעֲבֹ֕ד אֶת־עֲבֹדַ֖ת אֹ֥הֶל מוֹעֵֽד׃ תִּסְפּוֹר אוֹתָם מִגִּיל שְׁלוֹשִׁים שָׁנָה וָמַעְלָה וְעַד גִּיל חֲמִשִּׁים שָׁנָה – כָּל הַבָּאִים לַקָּהָל שֶׁתַּפְקִידוֹ לַעֲבוֹד אֶת עֲבוֹדַת אוֹהֶל הַהִתְכַּנְּסוּת.
לא וְזֹאת֙ מִשְׁמֶ֣רֶת מַשָּׂאָ֔ם לְכׇל־עֲבֹדָתָ֖ם בְּאֹ֣הֶל מוֹעֵ֑ד קַרְשֵׁי֙ הַמִּשְׁכָּ֔ן וּבְרִיחָ֖יו וְעַמּוּדָ֥יו וַאֲדָנָֽיו׃ בְּכָל עֲבוֹדָתָם בְּאוֹהֶל הַהִתְכַּנְּסוּת, תַּפְקִידָם לִסְחוֹב אֶת קַרְשֵׁי הַמִּשְׁכָּן, וּבְרִיחָיו, וְעַמּוּדָיו, וּבְסִיסָיו.
לב וְעַמּוּדֵי֩ הֶחָצֵ֨ר סָבִ֜יב וְאַדְנֵיהֶ֗ם וִיתֵֽדֹתָם֙ וּמֵ֣יתְרֵיהֶ֔ם לְכׇ֨ל־כְּלֵיהֶ֔ם וּלְכֹ֖ל עֲבֹדָתָ֑ם וּבְשֵׁמֹ֣ת תִּפְקְד֔וּ אֶת־כְּלֵ֖י מִשְׁמֶ֥רֶת מַשָּׂאָֽם׃ וְעַמּוּדֵי הֶחָצֵר מִסָּבִיב, וּבְסִיסֵיהֶם, וִיתֵדוֹתֵיהֶם, וּמֵיתְרֵיהֶם, כָּל כְּלֵיהֶם וְכָל עֲבוֹדָתָם, וּתְמַנּוּ כָּל אֶחָד בִּשְׁמוֹ עַל הַכֵּלִים שֶׁתַּפְקִידוֹ לִסְחוֹב.
לג זֹ֣את עֲבֹדַ֗ת מִשְׁפְּחֹת֙ בְּנֵ֣י מְרָרִ֔י לְכׇל־עֲבֹדָתָ֖ם בְּאֹ֣הֶל מוֹעֵ֑ד בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן־אַהֲרֹ֖ן הַכֹּהֵֽן׃ זֹאת עֲבוֹדַת מִשְׁפְּחוֹת בְּנֵי מְרָרִי – כָּל עֲבוֹדָתָם שֶׁבְּאוֹהֶל הַהִתְכַּנְּסוּת, תַּחַת פִּקּוּד אִיתָמָר בֶּן אַהֲרֹן הַכֹּהֵן.
ס פָּרָשַׁת פְּקוּדֵי הַלְוִיִּים
לד וַיִּפְקֹ֨ד מֹשֶׁ֧ה וְאַהֲרֹ֛ן וּנְשִׂיאֵ֥י הָעֵדָ֖ה אֶת־בְּנֵ֣י הַקְּהָתִ֑י לְמִשְׁפְּחֹתָ֖ם וּלְבֵ֥ית אֲבֹתָֽם׃ מֹשֶׁה וְאַהֲרֹן וּנְשִׂיאֵי הַצִּבּוּר סָפְרוּ אֶת בְּנֵי קְהָת בַּחֲלוּקָה לְפִי מִשְׁפְּחוֹתֵיהֶם וּבָתֵּי אֲבוֹתֵיהֶם.
לה מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן־חֲמִשִּׁ֣ים שָׁנָ֑ה כׇּל־הַבָּא֙ לַצָּבָ֔א לַעֲבֹדָ֖ה בְּאֹ֥הֶל מוֹעֵֽד׃ מִגִּיל שְׁלוֹשִׁים שָׁנָה וָמַעְלָה וְעַד גִּיל חֲמִשִּׁים שָׁנָה כָּל הַבָּאִים לַקָּהָל שֶׁתַּפְקִידוֹ עֲבוֹדָה בְּאוֹהֶל הַהִתְכַּנְּסוּת.
לו וַיִּהְי֥וּ פְקֻדֵיהֶ֖ם לְמִשְׁפְּחֹתָ֑ם אַלְפַּ֕יִם שְׁבַ֥ע מֵא֖וֹת וַחֲמִשִּֽׁים׃ מִסְפַּר הָאֲנָשִׁים שֶׁנִּסְפְּרוּ לְפִי מִשְׁפְּחוֹתֵיהֶם הָיָה אַלְפַּיִם שְׁבַע מֵאוֹת וַחֲמִשִּׁים.
לז אֵ֤לֶּה פְקוּדֵי֙ מִשְׁפְּחֹ֣ת הַקְּהָתִ֔י כׇּל־הָעֹבֵ֖ד בְּאֹ֣הֶל מוֹעֵ֑ד אֲשֶׁ֨ר פָּקַ֤ד מֹשֶׁה֙ וְאַהֲרֹ֔ן עַל־פִּ֥י יְהֹוָ֖ה בְּיַד־מֹשֶֽׁה׃ זֶהוּ מִסְפַּר כָּל אַנְשֵׁי מִשְׁפְּחוֹת הַקְּהָתִי הָעוֹבְדִים בְּאוֹהֶל הַהִתְכַּנְּסוּת שֶׁמֹּשֶׁה וְאַהֲרֹן סָפְרוּ עַל פִּי דִּבְרֵי ה' שֶׁנֶּאֶמְרוּ לְמֹשֶׁה.

שני[עריכה]

ס
לח וּפְקוּדֵ֖י בְּנֵ֣י גֵרְשׁ֑וֹן לְמִשְׁפְּחוֹתָ֖ם וּלְבֵ֥ית אֲבֹתָֽם׃ וּמִסְפַּר בְּנֵי גֵּרְשׁוֹן לְמִשְׁפְּחוֹתֵיהֶם וּלְבֵית אֲבוֹתֵיהֶם.
לט מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן־חֲמִשִּׁ֣ים שָׁנָ֑ה כׇּל־הַבָּא֙ לַצָּבָ֔א לַעֲבֹדָ֖ה בְּאֹ֥הֶל מוֹעֵֽד׃ מִגִּיל שְׁלוֹשִׁים שָׁנָה וָמַעְלָה וְעַד גִּיל חֲמִשִּׁים שָׁנָה כָּל הַבָּאִים לַקָּהָל שֶׁתַּפְקִידוֹ עֲבוֹדָה בְּאוֹהֶל הַהִתְכַּנְּסוּת.
מ וַיִּֽהְיוּ֙ פְּקֻ֣דֵיהֶ֔ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם אַלְפַּ֕יִם וְשֵׁ֥שׁ מֵא֖וֹת וּשְׁלֹשִֽׁים׃ מִסְפַּר הָאֲנָשִׁים שֶׁנִּסְפְּרוּ לְפִי מִשְׁפְּחוֹתֵיהֶם וּבָתֵּי אֲבוֹתֵיהֶם הָיָה אַלְפַּיִם וְשֵׁשׁ מֵאוֹת וּשְׁלוֹשִׁים.
מא אֵ֣לֶּה פְקוּדֵ֗י מִשְׁפְּחֹת֙ בְּנֵ֣י גֵרְשׁ֔וֹן כׇּל־הָעֹבֵ֖ד בְּאֹ֣הֶל מוֹעֵ֑ד אֲשֶׁ֨ר פָּקַ֥ד מֹשֶׁ֛ה וְאַהֲרֹ֖ן עַל־פִּ֥י יְהֹוָֽה׃ זֶהוּ מִסְפַּר כָּל אַנְשֵׁי מִשְׁפְּחוֹת בְּנֵי גֵּרְשׁוֹן הָעוֹבְדִים בְּאוֹהֶל הַהִתְכַּנְּסוּת שֶׁסָּפְרוּ מֹשֶׁה וְאַהֲרֹן עַל פִּי דִּבְרֵי ה'.
מב וּפְקוּדֵ֕י מִשְׁפְּחֹ֖ת בְּנֵ֣י מְרָרִ֑י לְמִשְׁפְּחֹתָ֖ם לְבֵ֥ית אֲבֹתָֽם׃ וּמִסְפַּר מִשְׁפְּחוֹת בְּנֵי מְרָרִי שֶׁנִּסְפְּרוּ לְפִי מִשְׁפְּחוֹתֵיהֶם וּבָתֵּי אֲבוֹתֵיהֶם.
מג מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן־חֲמִשִּׁ֣ים שָׁנָ֑ה כׇּל־הַבָּא֙ לַצָּבָ֔א לַעֲבֹדָ֖ה בְּאֹ֥הֶל מוֹעֵֽד׃ מִגִּיל שְׁלוֹשִׁים שָׁנָה וָמַעְלָה וְעַד גִּיל חֲמִשִּׁים שָׁנָה כָּל הַבָּאִים לַקָּהָל שֶׁתַּפְקִידוֹ עֲבוֹדָה בְּאוֹהֶל הַהִתְכַּנְּסוּת.
מד וַיִּהְי֥וּ פְקֻדֵיהֶ֖ם לְמִשְׁפְּחֹתָ֑ם שְׁלֹ֥שֶׁת אֲלָפִ֖ים וּמָאתָֽיִם׃ מִסְפַּר אַנְשֵׁיהֶם שֶׁנִּסְפְּרוּ לְפִי מִשְׁפְּחוֹתֵיהֶם הָיָה שְׁלוֹשֶׁת אֲלָפִים וּמָאתַיִם.
מה אֵ֣לֶּה פְקוּדֵ֔י מִשְׁפְּחֹ֖ת בְּנֵ֣י מְרָרִ֑י אֲשֶׁ֨ר פָּקַ֤ד מֹשֶׁה֙ וְאַהֲרֹ֔ן עַל־פִּ֥י יְהֹוָ֖ה בְּיַד־מֹשֶֽׁה׃ זֶה מִסְפַּר מִשְׁפְּחוֹת בְּנֵי מְרָרִי שֶׁמֹּשֶׁה וְאַהֲרֹן סָפְרוּ עַל פִּי דִּבְרֵי ה' שֶׁנֶּאֶמְרוּ לְמֹשֶׁה.
מו כׇּֽל־הַפְּקֻדִ֡ים אֲשֶׁר֩ פָּקַ֨ד מֹשֶׁ֧ה וְאַהֲרֹ֛ן וּנְשִׂיאֵ֥י יִשְׂרָאֵ֖ל אֶת־הַלְוִיִּ֑ם לְמִשְׁפְּחֹתָ֖ם וּלְבֵ֥ית אֲבֹתָֽם׃ כָּל הַלְוִיִּים הַנִּמְנִים שֶׁמֹּשֶׁה וְאַהֲרֹן וּנְשִׂיאֵי יִשְׂרָאֵל סָפְרוּ לְפִי מִשְׁפְּחוֹתֵיהֶם וּבָתֵּי אֲבוֹתֵיהֶם,
מז מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן־חֲמִשִּׁ֣ים שָׁנָ֑ה כׇּל־הַבָּ֗א לַעֲבֹ֨ד עֲבֹדַ֧ת עֲבֹדָ֛ה וַעֲבֹדַ֥ת מַשָּׂ֖א בְּאֹ֥הֶל מוֹעֵֽד׃ מִגִּיל שְׁלוֹשִׁים שָׁנָה וָמַעְלָה וְעַד גִּיל חֲמִשִּׁים שָׁנָה, כָּל מְגֻיָּס לַעֲבוֹד עֲבוֹדָה לִפְנֵי ה' וַעֲבוֹדַת מַשָּׂא בְּאוֹהֶל הַהִתְכַּנְּסוּת,
מח וַיִּהְי֖וּ פְּקֻדֵיהֶ֑ם שְׁמֹנַ֣ת אֲלָפִ֔ים וַחֲמֵ֥שׁ מֵא֖וֹת וּשְׁמֹנִֽים׃ מִסְפַּר אַנְשֵׁיהֶם הָיָה שְׁמוֹנַת אֲלָפִים וַחֲמֵשׁ מֵאוֹת וּשְׁמוֹנִים.
מט עַל־פִּ֨י יְהֹוָ֜ה פָּקַ֤ד אוֹתָם֙ בְּיַד־מֹשֶׁ֔ה אִ֥ישׁ אִ֛ישׁ עַל־עֲבֹדָת֖וֹ וְעַל־מַשָּׂא֑וֹ וּפְקֻדָ֕יו אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ סָפְרוּ אוֹתָם כְּדִבְרֵי ה' שֶׁנֶּאֶמְרוּ לְמֹשֶׁה – כָּל אֶחָד וְאֶחָד הִתְמַנָּה עַל עֲבוֹדָתוֹ וְעַל מַשָּׂאוֹ, וְהַנִּמְנִים הָיוּ בַּגִּיל שֶׁה' צִוָּה אֶת מֹשֶׁה.

שלישי[עריכה]

פ פָּרָשַׁת פְּקוּדֵי יִשְׂרָאֵל
ה א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב צַ֚ו אֶת־בְּנֵ֣י יִשְׂרָאֵ֔ל וִֽישַׁלְּחוּ֙ מִן־הַֽמַּחֲנֶ֔ה כׇּל־צָר֖וּעַ וְכׇל־זָ֑ב וְכֹ֖ל טָמֵ֥א לָנָֽפֶשׁ׃ צַוֵּה אֶת בְּנֵי יִשְׂרָאֵל שֶׁיְּגָרְשׁוּ מֵהַמַּחֲנֶה כָּל צָרוּעַ וְכָל זָב וְכָל מִי שֶׁנִּטְמָא מִמֵּת.
ג מִזָּכָ֤ר עַד־נְקֵבָה֙ תְּשַׁלֵּ֔חוּ אֶל־מִח֥וּץ לַֽמַּחֲנֶ֖ה תְּשַׁלְּח֑וּם וְלֹ֤א יְטַמְּאוּ֙ אֶת־מַ֣חֲנֵיהֶ֔ם אֲשֶׁ֥ר אֲנִ֖י שֹׁכֵ֥ן בְּתוֹכָֽם׃ תְּגָרְשׁוּ בֵּין זָכָר וּבֵין נְקֵבָה, תְּגָרְשׁוּ אוֹתָם אֶל מִחוּץ לַמַּחֲנֶה, וְלֹא יְטַמְּאוּ אֶת הַמַּחֲנֶה שֶׁלָּהֶם שֶׁאֲנִי שׁוֹכֵן בְּתוֹכָם.
ד וַיַּֽעֲשׂוּ־כֵן֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַיְשַׁלְּח֣וּ אוֹתָ֔ם אֶל־מִח֖וּץ לַֽמַּחֲנֶ֑ה כַּאֲשֶׁ֨ר דִּבֶּ֤ר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה כֵּ֥ן עָשׂ֖וּ בְּנֵ֥י יִשְׂרָאֵֽל׃ בְּנֵי יִשְׂרָאֵל עָשׂוּ כָּךְ וְגֵרְשׁוּ אוֹתָם אֶל מִחוּץ לַמַּחֲנֶה, כְּמוֹ שֶׁה' דִּבֵּר אֶל מֹשֶׁה כָּךְ בְּנֵי יִשְׂרָאֵל עָשׂוּ.
פ פָּרָשַׁת גֵּזֶל הַגֵּר
ה וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ו דַּבֵּר֮ אֶל־בְּנֵ֣י יִשְׂרָאֵל֒ אִ֣ישׁ אֽוֹ־אִשָּׁ֗ה כִּ֤י יַעֲשׂוּ֙ מִכׇּל־חַטֹּ֣את הָֽאָדָ֔ם לִמְעֹ֥ל מַ֖עַל בַּיהֹוָ֑ה וְאָֽשְׁמָ֖ה הַנֶּ֥פֶשׁ הַהִֽוא׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל: אִישׁ אוֹ אִשָּׁה שֶׁיַּעֲשׂוּ מֵעֲבֵרוֹת שֶׁל פְּגִיעָה בְּאָדָם וְיִשָּׁבְעוּ לַשֶּׁקֶר בְּשֵׁם ה' שֶׁלֹּא גָּזְלוּ – הָאָדָם הַהוּא יִהְיֶה אָשֵׁם.
ז וְהִתְוַדּ֗וּ אֶֽת־חַטָּאתָם֮ אֲשֶׁ֣ר עָשׂוּ֒ וְהֵשִׁ֤יב אֶת־אֲשָׁמוֹ֙ בְּרֹאשׁ֔וֹ וַחֲמִישִׁת֖וֹ יֹסֵ֣ף עָלָ֑יו וְנָתַ֕ן לַאֲשֶׁ֖ר אָשַׁ֥ם לֽוֹ׃ הוּא יִתְוַדֶּה עַל חֶטְאוֹ שֶׁעָשָׂה, וְיַחֲזִיר אֶת חוֹבוֹ בְּשָׁוְיוֹ וְיוֹסִיף עָלָיו חֲמִישִׁית, וְיִתֵּן לְזֶה שֶׁהִתְחַיֵּב לוֹ אוֹ לְיוֹרְשׁוֹ.
ח וְאִם־אֵ֨ין לָאִ֜ישׁ גֹּאֵ֗ל לְהָשִׁ֤יב הָאָשָׁם֙ אֵלָ֔יו הָאָשָׁ֛ם הַמּוּשָׁ֥ב לַיהֹוָ֖ה לַכֹּהֵ֑ן מִלְּבַ֗ד אֵ֚יל הַכִּפֻּרִ֔ים אֲשֶׁ֥ר יְכַפֶּר־בּ֖וֹ עָלָֽיו׃ וְאִם אֵין לָאִישׁ קְרוֹב מִשְׁפָּחָה יוֹרֵשׁ לְהַחֲזִיר לוֹ אֶת הַחוֹב, הַחוֹב מֻחְזַר לַה' וְיִנָּתֵן לַכֹּהֵן, בְּנוֹסָף לָאַיִל לְכַפָּרָה שֶׁבּוֹ הוּא מְכַפֵּר עַל עַצְמוֹ.
ט וְכׇל־תְּרוּמָ֞ה לְכׇל־קׇדְשֵׁ֧י בְנֵי־יִשְׂרָאֵ֛ל אֲשֶׁר־יַקְרִ֥יבוּ לַכֹּהֵ֖ן ל֥וֹ יִהְיֶֽה׃ וְכָל תְּרוּמָה – כָּל קָדְשֵׁי בְּנֵי יִשְׂרָאֵל שֶׁמַּגִּישִׁים לַכֹּהֵן, הִיא תִּהְיֶה שֶׁלּוֹ.
י וְאִ֥ישׁ אֶת־קֳדָשָׁ֖יו ל֣וֹ יִהְי֑וּ אִ֛ישׁ אֲשֶׁר־יִתֵּ֥ן לַכֹּהֵ֖ן ל֥וֹ יִהְיֶֽה׃ וְכָל אִישׁ, קָדָשָׁיו יִהְיוּ שֶׁלּוֹ לִפְנֵי שֶׁנָּתַן אוֹתָם, וְאַחֲרֵי שֶׁהָאִישׁ יִתֵּן לַכֹּהֵן הֵם יִהְיוּ שֶׁל הַכֹּהֵן.

רביעי[עריכה]

פ פָּרָשַׁת הַסּוֹטָה
יא וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
יב דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָמַרְתָּ֖ אֲלֵהֶ֑ם אִ֥ישׁ אִישׁ֙ כִּֽי־תִשְׂטֶ֣ה אִשְׁתּ֔וֹ וּמָעֲלָ֥ה ב֖וֹ מָֽעַל׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְתֹאמַר לָהֶם: כָּל אִישׁ שֶׁאִשְׁתּוֹ תִּסְטֶה וְתִבְגוֹד בּוֹ.
יג וְשָׁכַ֨ב אִ֣ישׁ אֹתָהּ֮ שִׁכְבַת־זֶ֒רַע֒ וְנֶעְלַם֙ מֵעֵינֵ֣י אִישָׁ֔הּ וְנִסְתְּרָ֖ה וְהִ֣יא נִטְמָ֑אָה וְעֵד֙ אֵ֣ין בָּ֔הּ וְהִ֖וא לֹ֥א נִתְפָּֽשָׂה׃ וְאִישׁ זָר יִשְׁכַּב אִתָּהּ וְיִתֵּן בָּהּ שִׁכְבַת זֶרַע וְלֹא נוֹדַע לְבַעֲלָהּ, וְהִיא הִתְחַבְּאָה עִם זָר וְנִטְמְאָה מִמֶּנּוּ, וְאֵין מִי שֶׁיָּעִיד עָלֶיהָ וְהִיא לֹא נִתְפְּשָׂה.
יד וְעָבַ֨ר עָלָ֧יו רֽוּחַ־קִנְאָ֛ה וְקִנֵּ֥א אֶת־אִשְׁתּ֖וֹ וְהִ֣וא נִטְמָ֑אָה אוֹ־עָבַ֨ר עָלָ֤יו רֽוּחַ־קִנְאָה֙ וְקִנֵּ֣א אֶת־אִשְׁתּ֔וֹ וְהִ֖יא לֹ֥א נִטְמָֽאָה׃ וְהִתְעוֹרֵר בּוֹ חֲשָׁד בְּנֶאֱמָנוּת אִשְׁתּוֹ וְהִיא אָכֵן נִטְמְאָה, אוֹ שֶׁהִתְעוֹרֵר בּוֹ חֲשָׁד בְּנֶאֱמָנוּת אִשְׁתּוֹ וְהִיא לֹא נִטְמְאָה.
טו וְהֵבִ֨יא הָאִ֣ישׁ אֶת־אִשְׁתּוֹ֮ אֶל־הַכֹּהֵן֒ וְהֵבִ֤יא אֶת־קׇרְבָּנָהּ֙ עָלֶ֔יהָ עֲשִׂירִ֥ת הָאֵיפָ֖ה קֶ֣מַח שְׂעֹרִ֑ים לֹֽא־יִצֹ֨ק עָלָ֜יו שֶׁ֗מֶן וְלֹֽא־יִתֵּ֤ן עָלָיו֙ לְבֹנָ֔ה כִּֽי־מִנְחַ֤ת קְנָאֹת֙ ה֔וּא מִנְחַ֥ת זִכָּר֖וֹן מַזְכֶּ֥רֶת עָוֺֽן׃ הָאִישׁ יָבִיא אֶת אִשְׁתּוֹ לַכֹּהֵן, וְיָבִיא אֶת קָרְבָּנָהּ עֲבוּרָהּ: עֲשִׂירִית הָאֵיפָה (כִּ2.16 לִיטֶר) שֶׁל קֶמַח שְׂעוֹרִים, לֹא יִשְׁפּוֹךְ עָלֶיהָ שֶׁמֶן וְלֹא יִתֵּן עָלֶיהָ לְבוֹנָה, כִּי הִיא מִנְחָה שֶׁל חֲשָׁד, מִנְחַת זִכָּרוֹן – לְהַזְכִּיר עָווֹן.
טז וְהִקְרִ֥יב אֹתָ֖הּ הַכֹּהֵ֑ן וְהֶֽעֱמִדָ֖הּ לִפְנֵ֥י יְהֹוָֽה׃ הַכֹּהֵן יַגִּישׁ אֶת הַמִּנְחָה וְיַעֲמִיד אוֹתָהּ לִפְנֵי ה'.
יז וְלָקַ֧ח הַכֹּהֵ֛ן מַ֥יִם קְדֹשִׁ֖ים בִּכְלִי־חָ֑רֶשׂ וּמִן־הֶֽעָפָ֗ר אֲשֶׁ֤ר יִהְיֶה֙ בְּקַרְקַ֣ע הַמִּשְׁכָּ֔ן יִקַּ֥ח הַכֹּהֵ֖ן וְנָתַ֥ן אֶל־הַמָּֽיִם׃ הַכֹּהֵן יִקַּח בִּכְלִי חֶרֶס מַיִם מְקֻדָּשִׁים מֵהַכִּיּוֹר, וְיִקַּח מֵהֶעָפָר שֶׁיִּהְיֶה בְּקַרְקַע הַמִּשְׁכָּן וְיָשִׂים בַּמַּיִם.
יח וְהֶעֱמִ֨יד הַכֹּהֵ֥ן אֶֽת־הָאִשָּׁה֮ לִפְנֵ֣י יְהֹוָה֒ וּפָרַע֙ אֶת־רֹ֣אשׁ הָֽאִשָּׁ֔ה וְנָתַ֣ן עַל־כַּפֶּ֗יהָ אֵ֚ת מִנְחַ֣ת הַזִּכָּר֔וֹן מִנְחַ֥ת קְנָאֹ֖ת הִ֑וא וּבְיַ֤ד הַכֹּהֵן֙ יִהְי֔וּ מֵ֥י הַמָּרִ֖ים הַמְאָֽרְרִֽים׃ הַכֹּהֵן יַעֲמִיד אֶת הָאִשָּׁה לִפְנֵי ה', יָסִיר אֶת כִּסּוּי הָרֹאשׁ שֶׁל הָאִשָּׁה, וְיָשִׂים עַל יָדֶיהָ אֶת מִנְחַת הַזְכָּרַת הֶעָווֹן – הִיא מִנְחַת חֲשָׁד, וּבְיַד הַכֹּהֵן יִהְיוּ הַמַּיִם הַמָּרִים הַמְּקַלְּלִים.
יט וְהִשְׁבִּ֨יעַ אֹתָ֜הּ הַכֹּהֵ֗ן וְאָמַ֤ר אֶל־הָֽאִשָּׁה֙ אִם־לֹ֨א שָׁכַ֥ב אִישׁ֙ אֹתָ֔ךְ וְאִם־לֹ֥א שָׂטִ֛ית טֻמְאָ֖ה תַּ֣חַת אִישֵׁ֑ךְ הִנָּקִ֕י מִמֵּ֛י הַמָּרִ֥ים הַֽמְאָרְרִ֖ים הָאֵֽלֶּה׃ הַכֹּהֵן יַשְׁבִּיעַ אוֹתָהּ, וְיֹאמַר לָאִשָּׁה: אִם אַף אִישׁ זָר לֹא שָׁכַב אִתָּךְ וְאִם לֹא סָטִית וְלֹא נִטְמֵאת בְּעוֹדֵךְ נְשׂוּאָה לְבַעֲלֵךְ – לֹא תִּנָּזְקִי מֵהַמַּיִם הַמָּרִים הַמְּקַלְּלִים הָאֵלֶּה.
כ וְאַ֗תְּ כִּ֥י שָׂטִ֛ית תַּ֥חַת אִישֵׁ֖ךְ וְכִ֣י נִטְמֵ֑את וַיִּתֵּ֨ן אִ֥ישׁ בָּךְ֙ אֶת־שְׁכׇבְתּ֔וֹ מִֽבַּלְעֲדֵ֖י אִישֵֽׁךְ׃ וְאַתְּ, אִם סָטִית בְּעוֹדֵךְ נְשׂוּאָה לְבַעֲלֵךְ וְנִטְמֵאת, וְשָׁכַב אִתָּךְ אִישׁ אַחֵר חוּץ מֵבַּעֲלֵךְ –
כא וְהִשְׁבִּ֨יעַ הַכֹּהֵ֥ן אֶֽת־הָאִשָּׁה֮ בִּשְׁבֻעַ֣ת הָאָלָה֒ וְאָמַ֤ר הַכֹּהֵן֙ לָֽאִשָּׁ֔ה יִתֵּ֨ן יְהֹוָ֥ה אוֹתָ֛ךְ לְאָלָ֥ה וְלִשְׁבֻעָ֖ה בְּת֣וֹךְ עַמֵּ֑ךְ בְּתֵ֨ת יְהֹוָ֤ה אֶת־יְרֵכֵךְ֙ נֹפֶ֔לֶת וְאֶת־בִּטְנֵ֖ךְ צָבָֽה׃ הַכֹּהֵן יַשְׁבִּיעַ אֶת הָאִשָּׁה בִּשְׁבוּעָה עִם קְלָלָה, הַכֹּהֵן יֹאמַר לָאִשָּׁה: ה' יַעֲשֶׂה שֶׁאַתְּ תִּהְיִי בְּעַמֵּךְ דֻּגְמָה לִקְלָלוֹת בִּשְׁבוּעָה, בְּכָךְ שֶׁה' יַעֲשֶׂה שֶׁיְּרֵכֵךְ תִּפּוֹל וּבִטְנֵךְ תִּתְנַפֵּחַ.
כב וּ֠בָ֠אוּ הַמַּ֨יִם הַמְאָרְרִ֤ים הָאֵ֙לֶּה֙ בְּֽמֵעַ֔יִךְ לַצְבּ֥וֹת בֶּ֖טֶן וְלַנְפִּ֣ל יָרֵ֑ךְ וְאָמְרָ֥ה הָאִשָּׁ֖ה אָמֵ֥ן ׀ אָמֵֽן׃ הַמַּיִם הַמְּקַלְּלִים הָאֵלֶּה יָבוֹאוּ בְּבִטְנֵךְ לְנַפֵּחַ בֶּטֶן וּלְהַפִּיל יָרֵךְ. וְהָאִשָּׁה תֹּאמַר: אָמֵן אָמֵן.
כג וְ֠כָתַ֠ב אֶת־הָאָלֹ֥ת הָאֵ֛לֶּה הַכֹּהֵ֖ן בַּסֵּ֑פֶר וּמָחָ֖ה אֶל־מֵ֥י הַמָּרִֽים׃ הַכֹּהֵן יִכְתּוֹב אֶת הַקְּלָלוֹת הָאֵלֶּה עַל עוֹר מְעֻבָּד, וְיִמְחוֹק אוֹתָם בַּמַּיִם הַמָּרִים.
כד וְהִשְׁקָה֙ אֶת־הָ֣אִשָּׁ֔ה אֶת־מֵ֥י הַמָּרִ֖ים הַמְאָֽרְרִ֑ים וּבָ֥אוּ בָ֛הּ הַמַּ֥יִם הַֽמְאָרְרִ֖ים לְמָרִֽים׃ הוּא יַשְׁקֶה אֶת הָאִשָּׁה בַּמַּיִם הַמָּרִים הַמְּקַלְּלִים, וְהַמַּיִם הַמְּקַלְּלִים יִכָּנְסוּ בָּהּ וִימָרְרוּ לָהּ.
כה וְלָקַ֤ח הַכֹּהֵן֙ מִיַּ֣ד הָֽאִשָּׁ֔ה אֵ֖ת מִנְחַ֣ת הַקְּנָאֹ֑ת וְהֵנִ֤יף אֶת־הַמִּנְחָה֙ לִפְנֵ֣י יְהֹוָ֔ה וְהִקְרִ֥יב אֹתָ֖הּ אֶל־הַמִּזְבֵּֽחַ׃ הַכֹּהֵן יִקַּח מִיַּד הָאִשָּׁה אֶת מִנְחַת הַחֲשָׁד וְיָנִיף אֶת הַמִּנְחָה לִפְנֵי ה' וְיַגִּישׁ אוֹתָהּ לַמִּזְבֵּחַ.
כו וְקָמַ֨ץ הַכֹּהֵ֤ן מִן־הַמִּנְחָה֙ אֶת־אַזְכָּ֣רָתָ֔הּ וְהִקְטִ֖יר הַמִּזְבֵּ֑חָה וְאַחַ֛ר יַשְׁקֶ֥ה אֶת־הָאִשָּׁ֖ה אֶת־הַמָּֽיִם׃ הַכֹּהֵן יִקְמוֹץ מֵהַמִּנְחָה אֶת הַקּוֹמֶץ שֶׁמַּזְכִּיר אוֹתָהּ לַה', וְיַקְטִיר עַל הַמִּזְבֵּחַ, וְאַחַר כָּךְ יַשְׁקֶה אֶת הָאִשָּׁה בַּמַּיִם הָאֵלֶּה.
כז וְהִשְׁקָ֣הּ אֶת־הַמַּ֗יִם וְהָיְתָ֣ה אִֽם־נִטְמְאָה֮ וַתִּמְעֹ֣ל מַ֣עַל בְּאִישָׁהּ֒ וּבָ֨אוּ בָ֜הּ הַמַּ֤יִם הַמְאָֽרְרִים֙ לְמָרִ֔ים וְצָבְתָ֣ה בִטְנָ֔הּ וְנָפְלָ֖ה יְרֵכָ֑הּ וְהָיְתָ֧ה הָאִשָּׁ֛ה לְאָלָ֖ה בְּקֶ֥רֶב עַמָּֽהּ׃ הוּא יַשְׁקֶה אוֹתָהּ בַּמַּיִם, אִם הָיָה שֶׁהִיא נִטְמְאָה וּבָגְדָה בְּבַעֲלָהּ, הַמַּיִם הַמְּקַלְּלִים יִכָּנְסוּ בָּהּ וִימָרְרוּ לָהּ, וּבִטְנָהּ תִּתְנַפֵּחַ וִירֵכָהּ תִּפּוֹל, וְהָאִשָּׁה תִּהְיֶה דֻּגְמָה לִקְלָלָה בִּשְׁבוּעָה בְּתוֹךְ עַמָּהּ.
כח וְאִם־לֹ֤א נִטְמְאָה֙ הָֽאִשָּׁ֔ה וּטְהֹרָ֖ה הִ֑וא וְנִקְּתָ֖ה וְנִזְרְעָ֥ה זָֽרַע׃ וְאִם הָאִשָּׁה לֹא נִטְמְאָה וְהִיא טְהוֹרָה, הִיא תְּזֻכֶּה, וְתִזְכֶּה לְיֶלֶד.
כט זֹ֥את תּוֹרַ֖ת הַקְּנָאֹ֑ת אֲשֶׁ֨ר תִּשְׂטֶ֥ה אִשָּׁ֛ה תַּ֥חַת אִישָׁ֖הּ וְנִטְמָֽאָה׃ אֵלֶּה דִּינֵי הַחֲשָׁד כְּשֶׁאִשָּׁה תִּסְטֶה בְּעוֹדָהּ נְשׂוּאָה לְבַעֲלָהּ וְתִטָּמֵא.
ל א֣וֹ אִ֗ישׁ אֲשֶׁ֨ר תַּעֲבֹ֥ר עָלָ֛יו ר֥וּחַ קִנְאָ֖ה וְקִנֵּ֣א אֶת־אִשְׁתּ֑וֹ וְהֶעֱמִ֤יד אֶת־הָֽאִשָּׁה֙ לִפְנֵ֣י יְהֹוָ֔ה וְעָ֤שָׂה לָהּ֙ הַכֹּהֵ֔ן אֵ֥ת כׇּל־הַתּוֹרָ֖ה הַזֹּֽאת׃ אוֹ אִישׁ שֶׁיַּעֲבוֹר בְּמַחְשַׁבְתּוֹ חֲשָׁד, וְיַחְשׁוֹד בְּאִשְׁתּוֹ, הוּא יַעֲמִיד אֶת הָאִשָּׁה לִפְנֵי ה', וְהַכֹּהֵן יַעֲשֶׂה לָהּ אֶת כָּל הַהוֹרָאוֹת הָאֵלֶּה.
לא וְנִקָּ֥ה הָאִ֖ישׁ מֵעָוֺ֑ן וְהָאִשָּׁ֣ה הַהִ֔וא תִּשָּׂ֖א אֶת־עֲוֺנָֽהּ׃ הָאִישׁ יִהְיֶה נָקִי מֵעָווֹן, וְהָאִשָּׁה הַהִיא תָּבוֹא עַל עוֹנְשָׁהּ עַל חֶטְאָהּ.
פ פָּרָשַׁת הַנָּזִיר
ו א וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
ב דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָמַרְתָּ֖ אֲלֵהֶ֑ם אִ֣ישׁ אֽוֹ־אִשָּׁ֗ה כִּ֤י יַפְלִא֙ לִנְדֹּר֙ נֶ֣דֶר נָזִ֔יר לְהַזִּ֖יר לַֽיהֹוָֽה׃ דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְתֹאמַר לָהֶם: אִישׁ אוֹ אִשָּׁה שֶׁיִּתֵּן מֵעַצְמוֹ לִנְדּוֹר נֶדֶר נְזִירוּת – לִהְיוֹת נָזִיר לַה'.
ג מִיַּ֤יִן וְשֵׁכָר֙ יַזִּ֔יר חֹ֥מֶץ יַ֛יִן וְחֹ֥מֶץ שֵׁכָ֖ר לֹ֣א יִשְׁתֶּ֑ה וְכׇל־מִשְׁרַ֤ת עֲנָבִים֙ לֹ֣א יִשְׁתֶּ֔ה וַעֲנָבִ֛ים לַחִ֥ים וִיבֵשִׁ֖ים לֹ֥א יֹאכֵֽל׃ הוּא יִתְנַזֵּר מִיַּיִן וּמַשְׁקֶה מְשַׁכֵּר – לֹא יִשְׁתֶּה חוֹמֶץ יַיִן וְחוֹמֶץ שֶׁל מַשְׁקֶה מְשַׁכֵּר, וְלֹא יִשְׁתֶּה שׁוּם מַשְׁקֶה שֶׁשָּׁרוּ בּוֹ עֲנָבִים, וְלֹא יֹאכַל עֲנָבִים לַחִים וִיבֵשִׁים.
ד כֹּ֖ל יְמֵ֣י נִזְר֑וֹ מִכֹּל֩ אֲשֶׁ֨ר יֵעָשֶׂ֜ה מִגֶּ֣פֶן הַיַּ֗יִן מֵחַרְצַנִּ֛ים וְעַד־זָ֖ג לֹ֥א יֹאכֵֽל׃ בְּכָל יְמֵי נְזִירוּתוֹ, לֹא יֹאכַל מִכָּל מַה שֶּׁיּוּצַר מֵעֵץ הַגֶּפֶן שֶׁמֵּנִיב יַיִן – מִגַּרְעִינִים וְעַד קְלִפָּה.
ה כׇּל־יְמֵי֙ נֶ֣דֶר נִזְר֔וֹ תַּ֖עַר לֹא־יַעֲבֹ֣ר עַל־רֹאשׁ֑וֹ עַד־מְלֹ֨את הַיָּמִ֜ם אֲשֶׁר־יַזִּ֤יר לַיהֹוָה֙ קָדֹ֣שׁ יִהְיֶ֔ה גַּדֵּ֥ל פֶּ֖רַע שְׂעַ֥ר רֹאשֽׁוֹ׃ בְּכָל יְמֵי נֶדֶר נְזִירוּתוֹ לֹא יַעֲבִיר עַל רֹאשׁוֹ סַכִּין גִלּוּחַ, הוּא יִהְיֶה קָדוֹשׁ וִיגַדֵּל אֶת שְׂעַר רֹאשׁוֹ עַד שֶׁיֻּשְׁלְמוּ הַיָּמִים שֶׁהִתְנַזֵּר לַה'.
ו כׇּל־יְמֵ֥י הַזִּיר֖וֹ לַיהֹוָ֑ה עַל־נֶ֥פֶשׁ מֵ֖ת לֹ֥א יָבֹֽא׃ בְּכָל יְמֵי נְזִירוּתוֹ לַה' לֹא יָבוֹא אֶל מֵת.
ז לְאָבִ֣יו וּלְאִמּ֗וֹ לְאָחִיו֙ וּלְאַ֣חֹת֔וֹ לֹא־יִטַּמָּ֥א לָהֶ֖ם בְּמֹתָ֑ם כִּ֛י נֵ֥זֶר אֱלֹהָ֖יו עַל־רֹאשֽׁוֹ׃ אֲפִלּוּ לְאָבִיו וּלְאִמּוֹ לְאָחִיו וּלְאֲחוֹתוֹ לֹא יִטַּמָּא לָהֶם בְּמוֹתָם, כִּי עַל רֹאשׁוֹ נִמְצָא כֶּתֶר אֱלֹהָיו.
ח כֹּ֖ל יְמֵ֣י נִזְר֑וֹ קָדֹ֥שׁ ה֖וּא לַֽיהֹוָֽה׃ בְּכָל יְמֵי נְזִירוּתוֹ הוּא קָדוֹשׁ לַה'.
ט וְכִֽי־יָמ֨וּת מֵ֤ת עָלָיו֙ בְּפֶ֣תַע פִּתְאֹ֔ם וְטִמֵּ֖א רֹ֣אשׁ נִזְר֑וֹ וְגִלַּ֤ח רֹאשׁוֹ֙ בְּי֣וֹם טׇהֳרָת֔וֹ בַּיּ֥וֹם הַשְּׁבִיעִ֖י יְגַלְּחֶֽנּוּ׃ וְאִם לְפֶתַע פִּתְאוֹם יָמוּת אֶצְלוֹ אָדָם וִיטַמֵּא אֶת נְזִירוּתוֹ – הוּא יְגַלֵּחַ אֶת רֹאשׁוֹ בַּיּוֹם שֶׁיִּטַּהֵר – בַּיּוֹם הַשְּׁבִיעִי יְגַלֵּחַ אוֹתוֹ.
י וּבַיּ֣וֹם הַשְּׁמִינִ֗י יָבִא֙ שְׁתֵּ֣י תֹרִ֔ים א֥וֹ שְׁנֵ֖י בְּנֵ֣י יוֹנָ֑ה אֶ֨ל־הַכֹּהֵ֔ן אֶל־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וּבַיּוֹם הַשְּׁמִינִי יָבִיא שְׁנֵי תּוֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה לַכֹּהֵן אֶל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת.
יא וְעָשָׂ֣ה הַכֹּהֵ֗ן אֶחָ֤ד לְחַטָּאת֙ וְאֶחָ֣ד לְעֹלָ֔ה וְכִפֶּ֣ר עָלָ֔יו מֵאֲשֶׁ֥ר חָטָ֖א עַל־הַנָּ֑פֶשׁ וְקִדַּ֥שׁ אֶת־רֹאשׁ֖וֹ בַּיּ֥וֹם הַהֽוּא׃ הַכֹּהֵן יַעֲשֶׂה אֶחָד מֵהֶם לְחַטָּאת וְאֶחָד לְעוֹלָה וִיכַפֵּר עָלָיו עַל שֶׁחָטָא שֶׁנִּטְמָא לְנֶפֶשׁ אָדָם, וּבְאוֹתוֹ יוֹם הוּא יְקַדֵּשׁ אֶת עַצְמוֹ שׁוּב לְהִתְנַזֵּר.
יב וְהִזִּ֤יר לַֽיהֹוָה֙ אֶת־יְמֵ֣י נִזְר֔וֹ וְהֵבִ֛יא כֶּ֥בֶשׂ בֶּן־שְׁנָת֖וֹ לְאָשָׁ֑ם וְהַיָּמִ֤ים הָרִאשֹׁנִים֙ יִפְּל֔וּ כִּ֥י טָמֵ֖א נִזְרֽוֹ׃ הוּא יִתְנַזֵּר לַה' שׁוּב אֶת יְמֵי נְזִירוּתוֹ, וְיָבִיא כֶּבֶשׂ שֶׁגִּילוֹ עַד שָׁנָה לְקָרְבַּן אָשָׁם, הַיָּמִים הָרִאשׁוֹנִים יִתְבַּטְּלוּ כִּי הוּא נִטְמָא וּנְזִירוּתוֹ נִפְסְלָה.
יג וְזֹ֥את תּוֹרַ֖ת הַנָּזִ֑יר בְּי֗וֹם מְלֹאת֙ יְמֵ֣י נִזְר֔וֹ יָבִ֣יא אֹת֔וֹ אֶל־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וְאֵלֶּה דִּינֵי קָרְבַּן הַנָּזִיר, שֶׁיָּבִיא אֶל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת בַּיּוֹם שֶׁיֻּשְׁלְמוּ יְמֵי נְזִירוּתוֹ.
יד וְהִקְרִ֣יב אֶת־קׇרְבָּנ֣וֹ לַיהֹוָ֡ה כֶּ֩בֶשׂ֩ בֶּן־שְׁנָת֨וֹ תָמִ֤ים אֶחָד֙ לְעֹלָ֔ה וְכַבְשָׂ֨ה אַחַ֧ת בַּת־שְׁנָתָ֛הּ תְּמִימָ֖ה לְחַטָּ֑את וְאַֽיִל־אֶחָ֥ד תָּמִ֖ים לִשְׁלָמִֽים׃ הוּא יַקְרִיב אֶת קָרְבָּנוֹ לַה': כֶּבֶשׂ אֶחָד לְלֹא מוּם שֶׁגִּילוֹ עַד שָׁנָה לְעוֹלָה, וְכִבְשָׂה אַחַת לְלֹא מוּם שֶׁגִּילָהּ עַד שָׁנָה לְחַטָּאת, וְאַיִל אֶחָד לְלֹא מוּם לִשְׁלָמִים.
טו וְסַ֣ל מַצּ֗וֹת סֹ֤לֶת חַלֹּת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֣ים בַּשָּׁ֑מֶן וּמִנְחָתָ֖ם וְנִסְכֵּיהֶֽם׃ וְסַל לֶחֶם שֶׁלֹּא הֶחְמִיץ – פִּתּוֹת עָבוֹת מִסּוֹלֶת מְעֻרְבָּבוֹת בְּשֶׁמֶן, וּפִתּוֹת דַּקּוֹת שֶׁלֹּא הֶחְמִיצוּ מְרוּחִים בְּשֶׁמֶן, וּמִנְחוֹתֵיהֶם וּשְׁפִיכַת הַיַּיִן שֶׁלָּהֶם.
טז וְהִקְרִ֥יב הַכֹּהֵ֖ן לִפְנֵ֣י יְהֹוָ֑ה וְעָשָׂ֥ה אֶת־חַטָּאת֖וֹ וְאֶת־עֹלָתֽוֹ׃ הַכֹּהֵן יַגִּישׁ אוֹתָם לִפְנֵי ה' וְיַעֲשֶׂה אֶת חַטָּאתוֹ וְאֶת עוֹלָתוֹ.
יז וְאֶת־הָאַ֜יִל יַעֲשֶׂ֨ה זֶ֤בַח שְׁלָמִים֙ לַֽיהֹוָ֔ה עַ֖ל סַ֣ל הַמַּצּ֑וֹת וְעָשָׂה֙ הַכֹּהֵ֔ן אֶת־מִנְחָת֖וֹ וְאֶת־נִסְכּֽוֹ׃ וְאֶת הָאַיִל יַעֲשֶׂה לְזֶבַח שְׁלָמִים לַה', בְּנוֹסָף לְסַל הַלֶּחֶם שֶׁלֹּא הֶחְמִיץ, וְהַכֹּהֵן יַעֲשֶׂה אֶת מִנְחָתוֹ וְאֶת שְׁפִיכַת הַיַּיִן שֶׁלּוֹ.
יח וְגִלַּ֣ח הַנָּזִ֗יר פֶּ֛תַח אֹ֥הֶל מוֹעֵ֖ד אֶת־רֹ֣אשׁ נִזְר֑וֹ וְלָקַ֗ח אֶת־שְׂעַר֙ רֹ֣אשׁ נִזְר֔וֹ וְנָתַן֙ עַל־הָאֵ֔שׁ אֲשֶׁר־תַּ֖חַת זֶ֥בַח הַשְּׁלָמִֽים׃ הַנָּזִיר יְגַלֵּחַ אֶת שְׂעָרוֹת נְזִירוּתוֹ בְּפֶתַח אוֹהֶל הַהִתְכַּנְּסוּת, וְיִקַּח אֶת שְׂעַר נְזִירוּתוֹ וְיַשְׁלִיךְ לָאֵשׁ שֶׁמִתַּחַת לְזֶבַח הַשְּׁלָמִים.
יט וְלָקַ֨ח הַכֹּהֵ֜ן אֶת־הַזְּרֹ֣עַ בְּשֵׁלָה֮ מִן־הָאַ֒יִל֒ וְֽחַלַּ֨ת מַצָּ֤ה אַחַת֙ מִן־הַסַּ֔ל וּרְקִ֥יק מַצָּ֖ה אֶחָ֑ד וְנָתַן֙ עַל־כַּפֵּ֣י הַנָּזִ֔יר אַחַ֖ר הִֽתְגַּלְּח֥וֹ אֶת־נִזְרֽוֹ׃ הַכֹּהֵן יִקַּח אֶת הַזְּרוֹעַ הַמְּבֻשֶּׁלֶת מֵהָאַיִל, וּמֵהַסַּל יִקַּח פִּתָּה עָבָה אַחַת שֶׁלֹּא הֶחְמִיצָה וּפִתָּה דַּקָּה אַחַת שֶׁלֹּא הֶחְמִיצָה, וְיָשִׂים אוֹתָם עַל יְדֵי הַנָּזִיר אַחֲרֵי שֶׁגִּלַּח אֶת שְׂעָרוֹת נְזִירוּתוֹ.
כ וְהֵנִיף֩ אוֹתָ֨ם הַכֹּהֵ֥ן ׀ תְּנוּפָה֮ לִפְנֵ֣י יְהֹוָה֒ קֹ֤דֶשׁ הוּא֙ לַכֹּהֵ֔ן עַ֚ל חֲזֵ֣ה הַתְּנוּפָ֔ה וְעַ֖ל שׁ֣וֹק הַתְּרוּמָ֑ה וְאַחַ֛ר יִשְׁתֶּ֥ה הַנָּזִ֖יר יָֽיִן׃ הַכֹּהֵן יָנִיף אוֹתָם לִפְנֵי ה' – וְהֵם יִהְיוּ קוֹדֶשׁ לַכֹּהֵן בְּנוֹסָף לַחֲזֵה שֶׁהוּנַף וְלַשּׁוֹק שֶׁהוּרַם, וְאַחַר כָּךְ מֻתָּר לַנָּזִיר לִשְׁתּוֹת יַיִן.
כא זֹ֣את תּוֹרַ֣ת הַנָּזִיר֮ אֲשֶׁ֣ר יִדֹּר֒ קׇרְבָּנ֤וֹ לַֽיהֹוָה֙ עַל־נִזְר֔וֹ מִלְּבַ֖ד אֲשֶׁר־תַּשִּׂ֣יג יָד֑וֹ כְּפִ֤י נִדְרוֹ֙ אֲשֶׁ֣ר יִדֹּ֔ר כֵּ֣ן יַעֲשֶׂ֔ה עַ֖ל תּוֹרַ֥ת נִזְרֽוֹ׃ וְזֶה דִּין מִי שֶׁיִּנְדּוֹר לִהְיוֹת נָזִיר, זֶה קָרְבָּנוֹ לַה' עַל נְזִירוּתוֹ חוּץ מִקָּרְבָּנוֹת נוֹסָפִים כְּפִי יְכָלְתּוֹ, כְּמוֹ הַנֶּדֶר שֶׁנָּדַר כָּךְ יַעֲשֶׂה, בְּנוֹסָף לְדִינֵי נְזִירוּתוֹ.
פ פָּרָשַׁת בִּרְכַּת כֹּהֲנִים
כב וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:
כג דַּבֵּ֤ר אֶֽל־אַהֲרֹן֙ וְאֶל־בָּנָ֣יו לֵאמֹ֔ר כֹּ֥ה תְבָרְכ֖וּ אֶת־בְּנֵ֣י יִשְׂרָאֵ֑ל אָמ֖וֹר לָהֶֽם׃ דַּבֵּר אֶל אַהֲרֹן וְאֶל בָּנָיו וֶאֱמוֹר: כָּךְ תְּבָרְכוּ אֶת בְּנֵי יִשְׂרָאֵל – אִמְרוּ לָהֶם:

חמישי[עריכה]

ס פָּרָשַׁת חֲנֻכַּת הַמִּשְׁכָּן
ז א וַיְהִ֡י בְּיוֹם֩ כַּלּ֨וֹת מֹשֶׁ֜ה לְהָקִ֣ים אֶת־הַמִּשְׁכָּ֗ן וַיִּמְשַׁ֨ח אֹת֜וֹ וַיְקַדֵּ֤שׁ אֹתוֹ֙ וְאֶת־כׇּל־כֵּלָ֔יו וְאֶת־הַמִּזְבֵּ֖חַ וְאֶת־כׇּל־כֵּלָ֑יו וַיִּמְשָׁחֵ֖ם וַיְקַדֵּ֥שׁ אֹתָֽם׃ בַּיּוֹם שֶׁמֹּשֶׁה סִיֵּם לְהָקִים אֶת הַמִּשְׁכָּן, הוּא מָרַח עָלָיו מֵהַשֶּׁמֶן וְקִדֵּשׁ אוֹתוֹ וְאֶת כָּל כֵּלָיו וְאֶת הַמִּזְבֵּחַ וְאֶת כָּל כֵּלָיו – הוּא מָרַח עֲלֵיהֶם מֵהַשֶּׁמֶן וּבְכָךְ קִדֵּשׁ אוֹתָם.
ב וַיַּקְרִ֙יבוּ֙ נְשִׂיאֵ֣י יִשְׂרָאֵ֔ל רָאשֵׁ֖י בֵּ֣ית אֲבֹתָ֑ם הֵ֚ם נְשִׂיאֵ֣י הַמַּטֹּ֔ת הֵ֥ם הָעֹמְדִ֖ים עַל־הַפְּקֻדִֽים׃ וְאָז נְשִׂיאֵי יִשְׂרָאֵל – רָאשֵׁי מִשְׁפְּחוֹתֵיהֶם, הִגִּישׁוּ אֶת תְּרוּמָתָם, הֵם נְשִׂיאֵי הַשְּׁבָטִים – הֵם הָאַחְרָאִים עַל סְפִירַת הָעָם.
ג וַיָּבִ֨יאוּ אֶת־קׇרְבָּנָ֜ם לִפְנֵ֣י יְהֹוָ֗ה שֵׁשׁ־עֶגְלֹ֥ת צָב֙ וּשְׁנֵ֣י עָשָׂ֣ר בָּקָ֔ר עֲגָלָ֛ה עַל־שְׁנֵ֥י הַנְּשִׂאִ֖ים וְשׁ֣וֹר לְאֶחָ֑ד וַיַּקְרִ֥יבוּ אוֹתָ֖ם לִפְנֵ֥י הַמִּשְׁכָּֽן׃ הֵם הֵבִיאוּ אֶת תְּרוּמָתָם לִפְנֵי ה' – שֵׁשׁ עֲגָלוֹת מְכֻסּוֹת בְּגָג וּשְׁנֵים עָשָׂר שְׁוָרִים, כָּל שְׁנֵי נְשִׂיאִים נָתְנוּ עֲגָלָה וְכָל נָשִׂיא נָתַן שׁוֹר, וְהֵם הִגִּישׁוּ אוֹתָם לִפְנֵי הַמִּשְׁכָּן.
ד וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' אָמַר לְמֹשֶׁה:
ה קַ֚ח מֵֽאִתָּ֔ם וְהָי֕וּ לַעֲבֹ֕ד אֶת־עֲבֹדַ֖ת אֹ֣הֶל מוֹעֵ֑ד וְנָתַתָּ֤ה אוֹתָם֙ אֶל־הַלְוִיִּ֔ם אִ֖ישׁ כְּפִ֥י עֲבֹדָתֽוֹ׃ קַח מֵהֶם, וְהֵם יְשַׁמְּשׁוּ לַעֲבוֹד אֶת עֲבוֹדַת אוֹהֶל הַהִתְכַּנְּסוּת, תֵּן אוֹתָם לַלְוִיִּים – לְכָל מִשְׁפָּחָה כְּפִי צוֹרֶךְ עֲבוֹדָתָהּ.
ו וַיִּקַּ֣ח מֹשֶׁ֔ה אֶת־הָעֲגָלֹ֖ת וְאֶת־הַבָּקָ֑ר וַיִּתֵּ֥ן אוֹתָ֖ם אֶל־הַלְוִיִּֽם׃ מֹשֶׁה לָקַח אֶת הָעֲגָלוֹת וְאֶת הַשְּׁוָרִים וְנָתַן אוֹתָם לַלְוִיִּים.
ז אֵ֣ת ׀ שְׁתֵּ֣י הָעֲגָל֗וֹת וְאֵת֙ אַרְבַּ֣עַת הַבָּקָ֔ר נָתַ֖ן לִבְנֵ֣י גֵרְשׁ֑וֹן כְּפִ֖י עֲבֹדָתָֽם׃ שְׁתֵּי עֲגָלוֹת וְאַרְבָּעָה שְׁוָרִים נָתַן לִבְנֵי גֵּרְשׁוֹן – כְּצוֹרֶךְ עֲבוֹדָתָם.
ח וְאֵ֣ת ׀ אַרְבַּ֣ע הָעֲגָלֹ֗ת וְאֵת֙ שְׁמֹנַ֣ת הַבָּקָ֔ר נָתַ֖ן לִבְנֵ֣י מְרָרִ֑י כְּפִי֙ עֲבֹ֣דָתָ֔ם בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן־אַהֲרֹ֖ן הַכֹּהֵֽן׃ וְאַרְבַּע עֲגָלוֹת וּשְׁמוֹנָה שְׁוָרִים נָתַן לִבְנֵי מְרָרִי – כְּצוֹרֶךְ עֲבוֹדָתָם שֶׁתַּחַת פִּקּוּד אִיתָמָר בֶּן אַהֲרֹן הַכֹּהֵן.
ט וְלִבְנֵ֥י קְהָ֖ת לֹ֣א נָתָ֑ן כִּֽי־עֲבֹדַ֤ת הַקֹּ֙דֶשׁ֙ עֲלֵהֶ֔ם בַּכָּתֵ֖ף יִשָּֽׂאוּ׃ וְלִבְנֵי קְהָת לֹא נָתַן, כִּי עֲלֵיהֶם מֻטֶּלֶת עֲבוֹדַת הַקּוֹדֶשׁ וְלָכֵן הֵם יִסְחֲבוּ בַּכָּתֵף.
י וַיַּקְרִ֣יבוּ הַנְּשִׂאִ֗ים אֵ֚ת חֲנֻכַּ֣ת הַמִּזְבֵּ֔חַ בְּי֖וֹם הִמָּשַׁ֣ח אֹת֑וֹ וַיַּקְרִ֧יבוּ הַנְּשִׂיאִ֛ם אֶת־קׇרְבָּנָ֖ם לִפְנֵ֥י הַמִּזְבֵּֽחַ׃ הַנְּשִׂיאִים הִקְרִיבוּ אֶת קָרְבְּנוֹת חֲנֻכַּת הַמִּזְבֵּחַ בַּיּוֹם שֶׁמָּרְחוּ אוֹתוֹ בְּשֶׁמֶן, הֵם הִגִּישׁוּ אֶת קָרְבָּנָם לִפְנֵי הַמִּזְבֵּחַ.
יא וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל־מֹשֶׁ֑ה נָשִׂ֨יא אֶחָ֜ד לַיּ֗וֹם נָשִׂ֤יא אֶחָד֙ לַיּ֔וֹם יַקְרִ֙יבוּ֙ אֶת־קׇרְבָּנָ֔ם לַחֲנֻכַּ֖ת הַמִּזְבֵּֽחַ׃ ה' אָמַר לְמֹשֶׁה: נָשִׂיא אֶחָד בְּכָל יוֹם, יַקְרִיבוּ אֶת קָרְבָּנָם לַחֲנֻכַּת הַמִּזְבֵּחַ.
ס
יב וַיְהִ֗י הַמַּקְרִ֛יב בַּיּ֥וֹם הָרִאשׁ֖וֹן אֶת־קׇרְבָּנ֑וֹ נַחְשׁ֥וֹן בֶּן־עַמִּינָדָ֖ב לְמַטֵּ֥ה יְהוּדָֽה׃ הַמַּקְרִיב אֶת קָרְבָּנוֹ בַּיּוֹם הָרִאשׁוֹן הָיָה נַחְשׁוֹן בֶּן עַמִּינָדָב מִשֵּׁבֶט יְהוּדָה.
יג וְקׇרְבָּנ֞וֹ קַֽעֲרַת־כֶּ֣סֶף אַחַ֗ת שְׁלֹשִׁ֣ים וּמֵאָה֮ מִשְׁקָלָהּ֒ מִזְרָ֤ק אֶחָד֙ כֶּ֔סֶף שִׁבְעִ֥ים שֶׁ֖קֶל בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ שְׁנֵיהֶ֣ם ׀ מְלֵאִ֗ים סֹ֛לֶת בְּלוּלָ֥ה בַשֶּׁ֖מֶן לְמִנְחָֽה׃ וְקָרְבָּנוֹ: קַעֲרַת כֶּסֶף אַחַת שֶׁמִּשְׁקָלָהּ מֵאָה וּשְׁלוֹשִׁים שֶׁקֶל (קִילוֹ וּ833 גְּרַם), וְכוֹס אַחַת לְקַבָּלַת הַדָּם וּזְרִיקָתוֹ עֲשׂוּיָה מִכֶּסֶף, בְּמִשְׁקָל שִׁבְעִים שֶׁקֶל בְּשֶׁקֶל הַקּוֹדֶשׁ (987 גְּרַם), שְׁנֵיהֶם מְלֵאִים סוֹלֶת מְעֻרְבֶּבֶת בְּשֶׁמֶן לְמִנְחָה.

פד זֹ֣את ׀ חֲנֻכַּ֣ת הַמִּזְבֵּ֗חַ בְּיוֹם֙ הִמָּשַׁ֣ח אֹת֔וֹ מֵאֵ֖ת נְשִׂיאֵ֣י יִשְׂרָאֵ֑ל קַעֲרֹ֨ת כֶּ֜סֶף שְׁתֵּ֣ים עֶשְׂרֵ֗ה מִֽזְרְקֵי־כֶ֙סֶף֙ שְׁנֵ֣ים עָשָׂ֔ר כַּפּ֥וֹת זָהָ֖ב שְׁתֵּ֥ים עֶשְׂרֵֽה׃ אֵלֶּה קָרְבְּנוֹת חֲנֻכַּת הַמִּזְבֵּחַ שֶׁנְּשִׂיאֵי יִשְׂרָאֵל הֵבִיאוּ בַּיּוֹם שֶׁמָּרְחוּ אוֹתוֹ בְּשֶׁמֶן: שְׁתֵּים עֶשְׂרֵה קַעֲרוֹת מִכֶּסֶף, שְׁנֵים עָשָׂר כּוֹסוֹת מִכֶּסֶף לְקַבָּלַת וּזְרִיקַת הַדָּם, וּשְׁתֵּים עֶשְׂרֵה כַּפּוֹת מִזָּהָב.
פה שְׁלֹשִׁ֣ים וּמֵאָ֗ה הַקְּעָרָ֤ה הָֽאַחַת֙ כֶּ֔סֶף וְשִׁבְעִ֖ים הַמִּזְרָ֣ק הָאֶחָ֑ד כֹּ֚ל כֶּ֣סֶף הַכֵּלִ֔ים אַלְפַּ֥יִם וְאַרְבַּע־מֵא֖וֹת בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ׃ מִשְׁקַל כָּל קְעָרָה מֵאָה וּשְׁלוֹשִׁים שֶׁקֶל כֶּסֶף (קִילוֹ וּ833 גְּרַם), מִשְׁקַל כָּל כּוֹס לְקַבָּלַת הַדָּם וּזְרִיקָתוֹ שִׁבְעִים שֶׁקֶל כֶּסֶף (987 גְּרַם), כָּל כֶּסֶף הַכֵּלִים בְּמִשְׁקַל אַלְפַּיִם וְאַרְבַּע מֵאוֹת בְּשֶׁקֶל הַקּוֹדֶשׁ (33,840 קִילוֹ).
פו כַּפּ֨וֹת זָהָ֤ב שְׁתֵּים־עֶשְׂרֵה֙ מְלֵאֹ֣ת קְטֹ֔רֶת עֲשָׂרָ֧ה עֲשָׂרָ֛ה הַכַּ֖ף בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ כׇּל־זְהַ֥ב הַכַּפּ֖וֹת עֶשְׂרִ֥ים וּמֵאָֽה׃ שְׁתֵּים עֶשְׂרֵה כַּפּוֹת זָהָב מְלֵאוֹת בִּקְטוֹרֶת – עֲשָׂרָה שְׁקָלִים (141 גְּרַם) מִשְׁקַל כָּל כַּף בְּשֶׁקֶל הַקּוֹדֶשׁ, כָּל זְהַב הַכַּפּוֹת מֵאָה וְעֶשְׂרִים שֶׁקֶל (1.692 קִילוֹ).

מפטיר[עריכה]

פז כׇּל־הַבָּקָ֨ר לָעֹלָ֜ה שְׁנֵ֧ים עָשָׂ֣ר פָּרִ֗ים אֵילִ֤ם שְׁנֵים־עָשָׂר֙ כְּבָשִׂ֧ים בְּנֵֽי־שָׁנָ֛ה שְׁנֵ֥ים עָשָׂ֖ר וּמִנְחָתָ֑ם וּשְׂעִירֵ֥י עִזִּ֛ים שְׁנֵ֥ים עָשָׂ֖ר לְחַטָּֽאת׃ כָּל הַבָּקָר לָעוֹלָה שְׁנֵים עָשָׂר פָּרִים, שְׁנֵים עָשָׂר אֵילִים, וּשְׁנֵים עָשָׂר כְּבָשִׂים שֶׁגִּילָם עַד שָׁנָה, וּמִנְחוֹתֵיהֶם כַּדִּין, וּשְׁנֵים עָשָׂר תְּיָשִׁים לְחַטָּאת.
פח וְכֹ֞ל בְּקַ֣ר ׀ זֶ֣בַח הַשְּׁלָמִ֗ים עֶשְׂרִ֣ים וְאַרְבָּעָה֮ פָּרִים֒ אֵילִ֤ם שִׁשִּׁים֙ עַתֻּדִ֣ים שִׁשִּׁ֔ים כְּבָשִׂ֥ים בְּנֵי־שָׁנָ֖ה שִׁשִּׁ֑ים זֹ֚את חֲנֻכַּ֣ת הַמִּזְבֵּ֔חַ אַחֲרֵ֖י הִמָּשַׁ֥ח אֹתֽוֹ׃ וְכָל הַבָּקָר לְזֶבַח הַשְּׁלָמִים – עֶשְׂרִים וְאַרְבָּעָה פָּרִים, שִׁשִּׁים אֵילִים, שִׁשִּׁים גְּדָיִים, שִׁשִּׁים כְּבָשִׂים שֶׁגִּילָם עַד שָׁנָה, אֵלֶּה קָרְבְּנוֹת חֲנֻכַּת הַמִּזְבֵּחַ אַחֲרֵי שֶׁמָּרְחוּ אוֹתוֹ בְּשֶׁמֶן.
פט וּבְבֹ֨א מֹשֶׁ֜ה אֶל־אֹ֣הֶל מוֹעֵד֮ לְדַבֵּ֣ר אִתּוֹ֒ וַיִּשְׁמַ֨ע אֶת־הַקּ֜וֹל מִדַּבֵּ֣ר אֵלָ֗יו מֵעַ֤ל הַכַּפֹּ֙רֶת֙ אֲשֶׁר֙ עַל־אֲרֹ֣ן הָעֵדֻ֔ת מִבֵּ֖ין שְׁנֵ֣י הַכְּרֻבִ֑ים וַיְדַבֵּ֖ר אֵלָֽיו׃ וּכְשֶׁמֹּשֶׁה נִכְנַס לְאוֹהֶל הַהִתְכַּנְּסוּת לְדַבֵּר עִם ה', הוּא שָׁמַע אֶת הַקּוֹל מְדַבֵּר אִתּוֹ וּבוֹקֵעַ מֵעַל הַכִּסּוּי שֶׁעַל אֲרוֹן לוּחוֹת הַבְּרִית – מִבֵּין שְׁנֵי צוּרוֹת הַמַּלְאָכִים, וְהוּא דִּבֵּר אֵלָיו.
