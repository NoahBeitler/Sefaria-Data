ראשון[עריכה]

ס פָּרָשַׁת הַשַּׁבָּת
לה א וַיַּקְהֵ֣ל מֹשֶׁ֗ה אֶֽת־כׇּל־עֲדַ֛ת בְּנֵ֥י יִשְׂרָאֵ֖ל וַיֹּ֣אמֶר אֲלֵהֶ֑ם אֵ֚לֶּה הַדְּבָרִ֔ים אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה לַעֲשֹׂ֥ת אֹתָֽם׃ מֹשֶׁה הִקְהִיל אֶת כָּל קְהַל בְּנֵי יִשְׂרָאֵל וְאָמַר לָהֶם: אֵלֶּה הַדְּבָרִים שֶׁה' צִוָּה לַעֲשׂוֹת אוֹתָם:
ב שֵׁ֣שֶׁת יָמִים֮ תֵּעָשֶׂ֣ה מְלָאכָה֒ וּבַיּ֣וֹם הַשְּׁבִיעִ֗י יִהְיֶ֨ה לָכֶ֥ם קֹ֛דֶשׁ שַׁבַּ֥ת שַׁבָּת֖וֹן לַיהֹוָ֑ה כׇּל־הָעֹשֶׂ֥ה ב֛וֹ מְלָאכָ֖ה יוּמָֽת׃ בְּמֶשֶׁךְ שִׁשָּׁה יָמִים תֵּעָשֶׂה מְלָאכָה, וּבַיּוֹם הַשְּׁבִיעִי תִּהְיֶה לָכֶם קְדֻשַּׁת שַׁבַּת – מְנוּחָה לַה', כָּל מִי שֶׁיַּעֲשֶׂה בּוֹ מְלָאכָה – יוּמַת.
ג לֹא־תְבַעֲר֣וּ אֵ֔שׁ בְּכֹ֖ל מֹשְׁבֹֽתֵיכֶ֑ם בְּי֖וֹם הַשַּׁבָּֽת׃ אַל תַּבְעִירוּ אֵשׁ בְּכָל מְקוֹמוֹת מְגוּרֵיכֶם בְּיוֹם הַשַּׁבָּת.
פ פָּרָשַׁת הַתְּרוּמָה
ד וַיֹּ֣אמֶר מֹשֶׁ֔ה אֶל־כׇּל־עֲדַ֥ת בְּנֵֽי־יִשְׂרָאֵ֖ל לֵאמֹ֑ר זֶ֣ה הַדָּבָ֔ר אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה לֵאמֹֽר׃ מֹשֶׁה אָמַר לְכָל צִבּוּר בְּנֵי יִשְׂרָאֵל: זֶה הַדָּבָר שֶׁה' צִוָּה:
ה קְח֨וּ מֵֽאִתְּכֶ֤ם תְּרוּמָה֙ לַֽיהֹוָ֔ה כֹּ֚ל נְדִ֣יב לִבּ֔וֹ יְבִיאֶ֕הָ אֵ֖ת תְּרוּמַ֣ת יְהֹוָ֑ה זָהָ֥ב וָכֶ֖סֶף וּנְחֹֽשֶׁת׃ קְחוּ מֵעַצְמְכֶם תְּרוּמָה לַה' – כָּל נְדִיב לֵב יָבִיא אוֹתָהּ אֶת תְּרוּמַת ה': זָהָב וְכֶסֶף וּנְחוֹשֶׁת.
ו וּתְכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ וְעִזִּֽים׃ וּתְכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת וּפִשְׁתָּן וּשְׂעַר עִזִּים.
ז וְעֹרֹ֨ת אֵילִ֧ם מְאׇדָּמִ֛ים וְעֹרֹ֥ת תְּחָשִׁ֖ים וַעֲצֵ֥י שִׁטִּֽים׃ וְעוֹרוֹת אַיִל צְבוּעִים בְּאָדוֹם, וְעוֹרוֹת שֶׁל הַחַיָּה תַּחַשׁ וַעֲצֵי שִׁטָּה.
ח וְשֶׁ֖מֶן לַמָּא֑וֹר וּבְשָׂמִים֙ לְשֶׁ֣מֶן הַמִּשְׁחָ֔ה וְלִקְטֹ֖רֶת הַסַּמִּֽים׃ וְשֶׁמֶן לַתְּאוּרָה וּבְשָׂמִים לַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ וְלִקְטוֹרֶת הַבְּשָׂמִים.
ט וְאַ֨בְנֵי־שֹׁ֔הַם וְאַבְנֵ֖י מִלֻּאִ֑ים לָאֵפ֖וֹד וְלַחֹֽשֶׁן׃ וְאַבְנֵי שׁוֹהַם וְאַבְנֵי חֵן לְשִׁבּוּץ, לָאֵפוֹד וְלַחוֹשֶׁן.
י וְכׇל־חֲכַם־לֵ֖ב בָּכֶ֑ם יָבֹ֣אוּ וְיַעֲשׂ֔וּ אֵ֛ת כׇּל־אֲשֶׁ֥ר צִוָּ֖ה יְהֹוָֽה׃ וְכָל הַמּוּכְשָׁרִים שֶׁבָּכֶם יָבוֹאוּ וְיַעֲשׂוּ אֶת כָּל מַה שֶּׁה' צִוָּה:
יא אֶ֨ת־הַמִּשְׁכָּ֔ן אֶֽת־אׇהֳל֖וֹ וְאֶת־מִכְסֵ֑הוּ אֶת־קְרָסָיו֙ וְאֶת־קְרָשָׁ֔יו אֶת־בְּרִיחָ֕ו אֶת־עַמֻּדָ֖יו וְאֶת־אֲדָנָֽיו׃ אֶת הַמִּשְׁכָּן אֶת אָהֳלוֹ וְאֶת הַכִּסּוּי שֶׁלּוֹ, אֶת הַקְּרָסִים הַקְּרָשִׁים הַבְּרִיחִים הָעַמּוּדִים וְהַבְּסִיסִים שֶׁלּוֹ.
יב אֶת־הָאָרֹ֥ן וְאֶת־בַּדָּ֖יו אֶת־הַכַּפֹּ֑רֶת וְאֵ֖ת פָּרֹ֥כֶת הַמָּסָֽךְ׃ אֶת הָאָרוֹן וְאֶת הַמּוֹטוֹת לַהֲרָמָתוֹ, אֶת הַכִּסּוּי וְאֶת הַפָּרוֹכֶת.
יג אֶת־הַשֻּׁלְחָ֥ן וְאֶת־בַּדָּ֖יו וְאֶת־כׇּל־כֵּלָ֑יו וְאֵ֖ת לֶ֥חֶם הַפָּנִֽים׃ אֶת הַשֻּׁלְחָן וְאֶת הַמּוֹטוֹת לַהֲרָמָתוֹ וְאֶת כָּל כֵּלָיו, וְאֶת הַלֶּחֶם הַמְּכֻבָּד.
יד וְאֶת־מְנֹרַ֧ת הַמָּא֛וֹר וְאֶת־כֵּלֶ֖יהָ וְאֶת־נֵרֹתֶ֑יהָ וְאֵ֖ת שֶׁ֥מֶן הַמָּאֽוֹר׃ וְאֶת מְנוֹרַת הַתְּאוּרָה וְאֶת כֵּלֶיהָ וְאֶת נֵרוֹתֶיהָ וְאֶת הַשֶּׁמֶן לַתְּאוּרָה.
טו וְאֶת־מִזְבַּ֤ח הַקְּטֹ֙רֶת֙ וְאֶת־בַּדָּ֔יו וְאֵת֙ שֶׁ֣מֶן הַמִּשְׁחָ֔ה וְאֵ֖ת קְטֹ֣רֶת הַסַּמִּ֑ים וְאֶת־מָסַ֥ךְ הַפֶּ֖תַח לְפֶ֥תַח הַמִּשְׁכָּֽן׃ וְאֶת מִזְבַּח הַקְּטוֹרֶת וְאֶת הַמּוֹטוֹת לַהֲרָמָתוֹ, וְאֶת הַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ וְאֶת קְטוֹרֶת הַבְּשָׂמִים וְאֶת מָסַךְ הַפֶּתַח לְפֶתַח הַמִּשְׁכָּן.
טז אֵ֣ת ׀ מִזְבַּ֣ח הָעֹלָ֗ה וְאֶת־מִכְבַּ֤ר הַנְּחֹ֙שֶׁת֙ אֲשֶׁר־ל֔וֹ אֶת־בַּדָּ֖יו וְאֶת־כׇּל־כֵּלָ֑יו אֶת־הַכִּיֹּ֖ר וְאֶת־כַּנּֽוֹ׃ אֶת מִזְבַּח הָעוֹלָה וְאֶת רֶשֶׁת הַנְּחוֹשֶׁת שֶׁלּוֹ אֶת הַמּוֹטוֹת לַהֲרָמָתוֹ וְאֶת כָּל כֵּלָיו, אֶת הַכִּיּוֹר וְאֶת בְּסִיסוֹ.
יז אֵ֚ת קַלְעֵ֣י הֶחָצֵ֔ר אֶת־עַמֻּדָ֖יו וְאֶת־אֲדָנֶ֑יהָ וְאֵ֕ת מָסַ֖ךְ שַׁ֥עַר הֶחָצֵֽר׃ אֶת הַגְּדֵרוֹת הַקְּלוּעוֹת לֶחָצֵר אֶת עַמּוּדֶיהָ וְאֶת בְּסִיסֶיהָ וְאֶת מָסַךְ שַׁעַר הֶחָצֵר.
יח אֶת־יִתְדֹ֧ת הַמִּשְׁכָּ֛ן וְאֶת־יִתְדֹ֥ת הֶחָצֵ֖ר וְאֶת־מֵיתְרֵיהֶֽם׃ אֶת יִתְדוֹת הַמִּשְׁכָּן וְאֶת יִתְדוֹת הֶחָצֵר וְאֶת הַמֵּיתָרִים שֶׁלָּהֶם.
יט אֶת־בִּגְדֵ֥י הַשְּׂרָ֖ד לְשָׁרֵ֣ת בַּקֹּ֑דֶשׁ אֶת־בִּגְדֵ֤י הַקֹּ֙דֶשׁ֙ לְאַהֲרֹ֣ן הַכֹּהֵ֔ן וְאֶת־בִּגְדֵ֥י בָנָ֖יו לְכַהֵֽן׃ אֶת כִּסּוּיֵי הַכֵּלִים שֶׁמְּשָׁרְתִים בָּהֶם בַּקּוֹדֶשׁ, אֶת בִּגְדֵי הַקּוֹדֶשׁ לְאַהֲרֹן הַכֹּהֵן וְאֶת בִּגְדֵי בָּנָיו לְכַהֵן.
כ וַיֵּ֥צְא֛וּ כׇּל־עֲדַ֥ת בְּנֵֽי־יִשְׂרָאֵ֖ל מִלִּפְנֵ֥י מֹשֶֽׁה׃ כָּל צִבּוּר בְּנֵי יִשְׂרָאֵל יָצְאוּ מִלִּפְנֵי מֹשֶׁה.

שני[עריכה]

כא וַיָּבֹ֕אוּ כׇּל־אִ֖ישׁ אֲשֶׁר־נְשָׂא֣וֹ לִבּ֑וֹ וְכֹ֡ל אֲשֶׁר֩ נָדְבָ֨ה רוּח֜וֹ אֹת֗וֹ הֵ֠בִ֠יאוּ אֶת־תְּרוּמַ֨ת יְהֹוָ֜ה לִמְלֶ֨אכֶת אֹ֤הֶל מוֹעֵד֙ וּלְכׇל־עֲבֹ֣דָת֔וֹ וּלְבִגְדֵ֖י הַקֹּֽדֶשׁ׃ כָּל הָאֲנָשִׁים הַמּוּכְשָׁרִים לַעֲשׂוֹת מְלָאכָה בָּאוּ, וְכָל אֵלֶּה שֶׁהִתְנַדְּבוּ לִתְרוֹם הֵבִיאוּ אֶת תְּרוּמַת ה' לִמְלֶאכֶת אוֹהֶל הַהִתְכַּנְּסוּת וּלְכָל עֲבוֹדָתוֹ וּלְבִגְדֵי הַקּוֹדֶשׁ.
כב וַיָּבֹ֥אוּ הָאֲנָשִׁ֖ים עַל־הַנָּשִׁ֑ים כֹּ֣ל ׀ נְדִ֣יב לֵ֗ב הֵ֠בִ֠יאוּ חָ֣ח וָנֶ֜זֶם וְטַבַּ֤עַת וְכוּמָז֙ כׇּל־כְּלִ֣י זָהָ֔ב וְכׇל־אִ֕ישׁ אֲשֶׁ֥ר הֵנִ֛יף תְּנוּפַ֥ת זָהָ֖ב לַיהֹוָֽה׃ הָאֲנָשִׁים בָּאוּ בְּנוֹסַף לַנָּשִׁים, כָּל נְדִיבֵי הַלֵּב הֵבִיאוּ עָגִיל וְנֶזֶם וְטַבַּעַת וּמָחוֹךְ – כָּל כְּלִי זָהָב, וְכָל אִישׁ שֶׁתָּרַם תְּרוּמַת זָהָב לַה'.
כג וְכׇל־אִ֞ישׁ אֲשֶׁר־נִמְצָ֣א אִתּ֗וֹ תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ וְעִזִּ֑ים וְעֹרֹ֨ת אֵילִ֧ם מְאׇדָּמִ֛ים וְעֹרֹ֥ת תְּחָשִׁ֖ים הֵבִֽיאוּ׃ וְכָל אִישׁ שֶׁנִּמְצָא אֶצְלוֹ תְּכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת וּפִשְׁתָּן וְשֵׂעַר עִזִּים וְעוֹרוֹת אַיִל צְבוּעִים בְּאָדוֹם וְעוֹרוֹת הַחַיָּה תַּחַשׁ – הֵבִיאוּ.
כד כׇּל־מֵרִ֗ים תְּר֤וּמַת כֶּ֙סֶף֙ וּנְחֹ֔שֶׁת הֵבִ֕יאוּ אֵ֖ת תְּרוּמַ֣ת יְהֹוָ֑ה וְכֹ֡ל אֲשֶׁר֩ נִמְצָ֨א אִתּ֜וֹ עֲצֵ֥י שִׁטִּ֛ים לְכׇל־מְלֶ֥אכֶת הָעֲבֹדָ֖ה הֵבִֽיאוּ׃ כָּל תּוֹרֵם תְּרוּמַת כֶּסֶף וּנְחוֹשֶׁת הֵבִיאוּ אֶת תְּרוּמַת ה', וְכָל מִי שֶׁנִּמְצָא אֶצְלוֹ עֲצֵי שִׁטָּה לְכָל מְלֶאכֶת הָעֲבוֹדָה – הֵבִיאוּ.
כה וְכׇל־אִשָּׁ֥ה חַכְמַת־לֵ֖ב בְּיָדֶ֣יהָ טָו֑וּ וַיָּבִ֣יאוּ מַטְוֶ֗ה אֶֽת־הַתְּכֵ֙לֶת֙ וְאֶת־הָֽאַרְגָּמָ֔ן אֶת־תּוֹלַ֥עַת הַשָּׁנִ֖י וְאֶת־הַשֵּֽׁשׁ׃ וְכָל הַנָּשִׁים הַמּוּכְשָׁרוֹת טָווּ בִּידֵיהֶן, וְהֵבִיאוּ חוּטִים טְווּיִים – אֶת הַתְּכֵלֶת וְאֶת הָאַרְגָּמָן אֶת הַצֶּבַע הָאָדוֹם מֵהַתּוֹלַעַת וְאֶת הַפִּשְׁתָּן.
כו וְכׇ֨ל־הַנָּשִׁ֔ים אֲשֶׁ֨ר נָשָׂ֥א לִבָּ֛ן אֹתָ֖נָה בְּחׇכְמָ֑ה טָו֖וּ אֶת־הָעִזִּֽים׃ וְכָל הַנָּשִׁים שֶׁהִתְעַלּוּ בְּכִשָּׁרוֹן טָווּ אֶת שְׂעַר הָעִזִּים.
כז וְהַנְּשִׂאִ֣ם הֵבִ֔יאוּ אֵ֚ת אַבְנֵ֣י הַשֹּׁ֔הַם וְאֵ֖ת אַבְנֵ֣י הַמִּלֻּאִ֑ים לָאֵפ֖וֹד וְלַחֹֽשֶׁן׃ וְהַנְּשִׂיאִים הֵבִיאוּ אֶת אַבְנֵי הַשּׁוֹהַם וְאֶת אַבְנֵי הַחֵן לְשִׁבּוּץ, לָאֵפוֹד וְלַחוֹשֶׁן.
כח וְאֶת־הַבֹּ֖שֶׂם וְאֶת־הַשָּׁ֑מֶן לְמָא֕וֹר וּלְשֶׁ֙מֶן֙ הַמִּשְׁחָ֔ה וְלִקְטֹ֖רֶת הַסַּמִּֽים׃ וְאֶת הַבּוֹשֶׂם וְאֶת הַשֶּׁמֶן, לִתְאוּרָה וְלַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ וְלִקְטוֹרֶת הַבְּשָׂמִים.
כט כׇּל־אִ֣ישׁ וְאִשָּׁ֗ה אֲשֶׁ֨ר נָדַ֣ב לִבָּם֮ אֹתָם֒ לְהָבִיא֙ לְכׇל־הַמְּלָאכָ֔ה אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה לַעֲשׂ֖וֹת בְּיַד־מֹשֶׁ֑ה הֵבִ֧יאוּ בְנֵי־יִשְׂרָאֵ֛ל נְדָבָ֖ה לַיהֹוָֽה׃ כָּל אִישׁ וְאִשָּׁה שֶׁטּוּב לִבָּם נִדֵּב אוֹתָם לְהָבִיא לְכָל הַמְּלָאכָה שֶׁה' צִוָּה בְּאֶמְצָעוּת מֹשֶׁה לַעֲשׂוֹת – בְּנֵי יִשְׂרָאֵל הֵבִיאוּ נְדָבָה לַה'.

שלישי (שני במחוברין)[עריכה]

פ פָּרָשַׁת בְּצַלְאֵל
ל וַיֹּ֤אמֶר מֹשֶׁה֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל רְא֛וּ קָרָ֥א יְהֹוָ֖ה בְּשֵׁ֑ם בְּצַלְאֵ֛ל בֶּן־אוּרִ֥י בֶן־ח֖וּר לְמַטֵּ֥ה יְהוּדָֽה׃ מֹשֶׁה אָמַר לִבְנֵי יִשְׂרָאֵל: רְאוּ, ה' הִכְרִיז עַל בְּצַלְאֵל בֶּן אוּרִי בֶּן חוּר מִשֵּׁבֶט יְהוּדָה.
לא וַיְמַלֵּ֥א אֹת֖וֹ ר֣וּחַ אֱלֹהִ֑ים בְּחׇכְמָ֛ה בִּתְבוּנָ֥ה וּבְדַ֖עַת וּבְכׇל־מְלָאכָֽה׃ וּמִלֵּא אוֹתוֹ בְּרוּחַ אֱלֹהִים, בְּחָכְמָה בִּתְבוּנָה וּבְדַעַת לַעֲסוֹק בְּכָל מְלָאכָה.
לב וְלַחְשֹׁ֖ב מַֽחֲשָׁבֹ֑ת לַעֲשֹׂ֛ת בַּזָּהָ֥ב וּבַכֶּ֖סֶף וּבַנְּחֹֽשֶׁת׃ וּלְתַכְנֵן תָּכְנִיּוֹת לַעֲשׂוֹת בַּזָּהָב וּבַכֶּסֶף וּבַנְּחוֹשֶׁת.
לג וּבַחֲרֹ֥שֶׁת אֶ֛בֶן לְמַלֹּ֖את וּבַחֲרֹ֣שֶׁת עֵ֑ץ לַעֲשׂ֖וֹת בְּכׇל־מְלֶ֥אכֶת מַחֲשָֽׁבֶת׃ וּבְאֻמָּנוּת חֲרִיטַת אֲבָנִים מְשֻׁבָּצוֹת וּבְנַגָּרוּת עֵץ, לַעֲשׂוֹת בְּכָל מְלֶאכֶת אֳמָנוּת.
לד וּלְהוֹרֹ֖ת נָתַ֣ן בְּלִבּ֑וֹ ה֕וּא וְאׇֽהֳלִיאָ֥ב בֶּן־אֲחִיסָמָ֖ךְ לְמַטֵּה־דָֽן׃ וְנָתַן בְּלִבּוֹ גַּם כִּשָּׁרוֹן לְהוֹרוֹת לַאֲחֵרִים, הוּא וְאָהֳלִיאָב בֶּן אֲחִיסָמָךְ מִשֵּׁבֶט דָּן.
לה מִלֵּ֨א אֹתָ֜ם חׇכְמַת־לֵ֗ב לַעֲשׂוֹת֮ כׇּל־מְלֶ֣אכֶת חָרָ֣שׁ ׀ וְחֹשֵׁב֒ וְרֹקֵ֞ם בַּתְּכֵ֣לֶת וּבָֽאַרְגָּמָ֗ן בְּתוֹלַ֧עַת הַשָּׁנִ֛י וּבַשֵּׁ֖שׁ וְאֹרֵ֑ג עֹשֵׂי֙ כׇּל־מְלָאכָ֔ה וְחֹשְׁבֵ֖י מַחֲשָׁבֹֽת׃ הוּא מִלֵּא אוֹתָם בְּחָכְמָה לַעֲשׂוֹת כָּל מְלֶאכֶת אֳמָנוּת וְתִכְנוּן, וְרִקְמָה בִּתְכֵלֶת וּבְאַרְגָּמָן בַּצֶּבַע הָאָדוֹם מֵהַתּוֹלַעַת וּבְפִשְׁתָּן, וּבִמְלֶאכֶת אֲרִיגָה – הֵם עוֹשִׂים כָּל מְלָאכָה וּמְתַכְנְנִים תָּכְנִיּוֹת.
לו א וְעָשָׂה֩ בְצַלְאֵ֨ל וְאׇהֳלִיאָ֜ב וְכֹ֣ל ׀ אִ֣ישׁ חֲכַם־לֵ֗ב אֲשֶׁר֩ נָתַ֨ן יְהֹוָ֜ה חׇכְמָ֤ה וּתְבוּנָה֙ בָּהֵ֔מָּה לָדַ֣עַת לַעֲשֹׂ֔ת אֶֽת־כׇּל־מְלֶ֖אכֶת עֲבֹדַ֣ת הַקֹּ֑דֶשׁ לְכֹ֥ל אֲשֶׁר־צִוָּ֖ה יְהֹוָֽה׃ בְּצַלְאֵל וְאָהֳלִיאָב וְכָל הָאֲנָשִׁים הַחֲכָמִים שֶׁה' נָתַן לָהֶם חָכְמָה וּתְבוּנָה לָדַעַת לַעֲשׂוֹת אֶת כָּל מְלֶאכֶת עֲבוֹדַת הַקּוֹדֶשׁ יַעֲשׂוּ אֶת כָּל מַה שֶּׁה' צִוָּה.
ב וַיִּקְרָ֣א מֹשֶׁ֗ה אֶל־בְּצַלְאֵל֮ וְאֶל־אׇֽהֳלִיאָב֒ וְאֶל֙ כׇּל־אִ֣ישׁ חֲכַם־לֵ֔ב אֲשֶׁ֨ר נָתַ֧ן יְהֹוָ֛ה חׇכְמָ֖ה בְּלִבּ֑וֹ כֹּ֚ל אֲשֶׁ֣ר נְשָׂא֣וֹ לִבּ֔וֹ לְקׇרְבָ֥ה אֶל־הַמְּלָאכָ֖ה לַעֲשֹׂ֥ת אֹתָֽהּ׃ מֹשֶׁה קָרָא לִבְצַלְאֵל וּלְאָהֳלִיאָב, וּלְכָל אִישׁ חֲכַם לֵב שֶׁה' נָתַן חָכְמָה בְּלִבּוֹ – כָּל מִי שֶׁהִתְנַדֵּב לָגֶשֶׁת לַעֲשִׂיַּת הַמְּלָאכָה לַעֲשׂוֹת אוֹתָהּ.
ג וַיִּקְח֞וּ מִלִּפְנֵ֣י מֹשֶׁ֗ה אֵ֤ת כׇּל־הַתְּרוּמָה֙ אֲשֶׁ֨ר הֵבִ֜יאוּ בְּנֵ֣י יִשְׂרָאֵ֗ל לִמְלֶ֛אכֶת עֲבֹדַ֥ת הַקֹּ֖דֶשׁ לַעֲשֹׂ֣ת אֹתָ֑הּ וְ֠הֵ֠ם הֵבִ֨יאוּ אֵלָ֥יו ע֛וֹד נְדָבָ֖ה בַּבֹּ֥קֶר בַּבֹּֽקֶר׃ הֵם לָקְחוּ מִלִּפְנֵי מֹשֶׁה אֶת כָּל הַתְּרוּמָה שֶׁבְּנֵי יִשְׂרָאֵל הֵבִיאוּ לִמְלֶאכֶת עֲבוֹדַת הַקּוֹדֶשׁ כְּדֵי לַעֲשׂוֹת אוֹתָהּ, וּבְנֵי יִשְׂרָאֵל הֵבִיאוּ לְמֹשֶׁה עוֹד נְדָבָה מֻקְדָּם בַּבּוֹקֶר.
ד וַיָּבֹ֙אוּ֙ כׇּל־הַ֣חֲכָמִ֔ים הָעֹשִׂ֕ים אֵ֖ת כׇּל־מְלֶ֣אכֶת הַקֹּ֑דֶשׁ אִֽישׁ־אִ֥ישׁ מִמְּלַאכְתּ֖וֹ אֲשֶׁר־הֵ֥מָּה עֹשִֽׂים׃ כָּל הַחֲכָמִים שֶׁעוֹשִׂים אֶת כָּל מְלֶאכֶת הַקּוֹדֶשׁ בָּאוּ – כָּל אֶחָד מִמְּלַאכְתּוֹ שֶׁהוּא מֻמְחֶה לַעֲשׂוֹת.
ה וַיֹּאמְרוּ֙ אֶל־מֹשֶׁ֣ה לֵּאמֹ֔ר מַרְבִּ֥ים הָעָ֖ם לְהָבִ֑יא מִדֵּ֤י הָֽעֲבֹדָה֙ לַמְּלָאכָ֔ה אֲשֶׁר־צִוָּ֥ה יְהֹוָ֖ה לַעֲשֹׂ֥ת אֹתָֽהּ׃ וְאָמְרוּ לְמֹשֶׁה: הָעָם מַרְבִּים לְהָבִיא יוֹתֵר מֵהַנִּצְרָךְ לָעֲבוֹדָה עֲבוּר הַמְּלָאכָה שֶׁה' צִוָּה לַעֲשׂוֹת.
ו וַיְצַ֣ו מֹשֶׁ֗ה וַיַּעֲבִ֨ירוּ ק֥וֹל בַּֽמַּחֲנֶה֮ לֵאמֹר֒ אִ֣ישׁ וְאִשָּׁ֗ה אַל־יַעֲשׂוּ־ע֛וֹד מְלָאכָ֖ה לִתְרוּמַ֣ת הַקֹּ֑דֶשׁ וַיִּכָּלֵ֥א הָעָ֖ם מֵהָבִֽיא׃ מֹשֶׁה צִוָּה וְהִכְרִיזוּ בַּמַּחֲנֶה: אִישׁ וְאִשָּׁה לֹא יַעֲשׂוּ עוֹד מְלָאכָה עֲבוּר תְּרוּמַת הַקּוֹדֶשׁ, וְהָעָם הִפְסִיקוּ לְהָבִיא.
ז וְהַמְּלָאכָ֗ה הָיְתָ֥ה דַיָּ֛ם לְכׇל־הַמְּלָאכָ֖ה לַעֲשׂ֣וֹת אֹתָ֑הּ וְהוֹתֵֽר׃ וְהַחוֹמָרִים הַמְּעֻבָּדִים שֶׁכְּבָר הֵבִיאוּ הִסְפִּיקוּ עֲבוּר עֲשִׂיַּת כָּל הַמְּלָאכָה וְיוֹתֵר.

רביעי[עריכה]

ס פָּרָשַׁת הַיְרִיעוֹת
ח וַיַּעֲשׂ֨וּ כׇל־חֲכַם־לֵ֜ב בְּעֹשֵׂ֧י הַמְּלָאכָ֛ה אֶת־הַמִּשְׁכָּ֖ן עֶ֣שֶׂר יְרִיעֹ֑ת שֵׁ֣שׁ מׇשְׁזָ֗ר וּתְכֵ֤לֶת וְאַרְגָּמָן֙ וְתוֹלַ֣עַת שָׁנִ֔י כְּרֻבִ֛ים מַעֲשֵׂ֥ה חֹשֵׁ֖ב עָשָׂ֥ה אֹתָֽם׃ כָּל חַכְמֵי הַלֵּב שֶׁבְּעוֹשֵׂי הַמְּלָאכָה עָשׂוּ אֶת הַמִּשְׁכָּן מֵעֶשֶׂר יְרִיעוֹת, מִפִּשְׁתָּן שָׁזוּר וּתְכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת, עִם צוּרוֹת שֶׁל מַלְאָכִים, הֵם נַעֲשׂוּ בִּמְלֶאכֶת אֳמָנוּת.
ט אֹ֜רֶךְ הַיְרִיעָ֣ה הָֽאַחַ֗ת שְׁמֹנֶ֤ה וְעֶשְׂרִים֙ בָּֽאַמָּ֔ה וְרֹ֙חַב֙ אַרְבַּ֣ע בָּֽאַמָּ֔ה הַיְרִיעָ֖ה הָאֶחָ֑ת מִדָּ֥ה אַחַ֖ת לְכׇל־הַיְרִיעֹֽת׃ אוֹרֶךְ כָּל יְרִיעָה עֶשְׂרִים וּשְׁמוֹנֶה אַמּוֹת, וְרוֹחַב אַרְבַּע אַמּוֹת כָּל יְרִיעָה, מִדָּה אַחַת לְכָל הַיְּרִיעוֹת.
י וַיְחַבֵּר֙ אֶת־חֲמֵ֣שׁ הַיְרִיעֹ֔ת אַחַ֖ת אֶל־אֶחָ֑ת וְחָמֵ֤שׁ יְרִיעֹת֙ חִבַּ֔ר אַחַ֖ת אֶל־אֶחָֽת׃ הוּא חִבֵּר אֶת חֲמֵשׁ הַיְּרִיעוֹת אַחַת לְאַחַת, וְחָמֵשׁ יְרִיעוֹת נוֹסָפוֹת חִבֵּר אַחַת לְאַחַת.
יא וַיַּ֜עַשׂ לֻֽלְאֹ֣ת תְּכֵ֗לֶת עַ֣ל שְׂפַ֤ת הַיְרִיעָה֙ הָֽאֶחָ֔ת מִקָּצָ֖ה בַּמַּחְבָּ֑רֶת כֵּ֤ן עָשָׂה֙ בִּשְׂפַ֣ת הַיְרִיעָ֔ה הַקִּ֣יצוֹנָ֔ה בַּמַּחְבֶּ֖רֶת הַשֵּׁנִֽית׃ הוּא עָשָׂה לוּלָאוֹת מִתְּכֵלֶת עַל שְׂפַת הַיְּרִיעָה הָאַחַת בַּקָּצֶה בַּמָּקוֹם שֶׁל הַחִבּוּר, וְכָךְ עָשָׂה בִּשְׂפַת הַיְּרִיעָה הַקִּיצוֹנָה בַּצַּד הַשֵּׁנִי שֶׁל הַחִבּוּר.
יב חֲמִשִּׁ֣ים לֻלָאֹ֗ת עָשָׂה֮ בַּיְרִיעָ֣ה הָאֶחָת֒ וַחֲמִשִּׁ֣ים לֻלָאֹ֗ת עָשָׂה֙ בִּקְצֵ֣ה הַיְרִיעָ֔ה אֲשֶׁ֖ר בַּמַּחְבֶּ֣רֶת הַשֵּׁנִ֑ית מַקְבִּילֹת֙ הַלֻּ֣לָאֹ֔ת אַחַ֖ת אֶל־אֶחָֽת׃ חֲמִשִּׁים לוּלָאוֹת עָשָׂה בַּיְּרִיעָה הָאַחַת וַחֲמִשִּׁים לוּלָאוֹת עָשָׂה בִּקְצֵה הַיְּרִיעָה שֶׁבַּצַּד הַשֵּׁנִי שֶׁל הַחִבּוּר – הַלּוּלָאוֹת מַקְבִּילוֹת אַחַת לַשְּׁנִיָּה.
יג וַיַּ֕עַשׂ חֲמִשִּׁ֖ים קַרְסֵ֣י זָהָ֑ב וַיְחַבֵּ֨ר אֶת־הַיְרִיעֹ֜ת אַחַ֤ת אֶל־אַחַת֙ בַּקְּרָסִ֔ים וַֽיְהִ֥י הַמִּשְׁכָּ֖ן אֶחָֽד׃ הוּא עָשָׂה חֲמִשִּׁים קַרְסֵי זָהָב, וְחִבֵּר אֶת הַיְּרִיעוֹת אַחַת לְאַחַת בַּקְּרָסִים, וְאוֹהֶל הַמִּשְׁכָּן הָיָה לְרֶצֶף אֶחָד.
פ
יד וַיַּ֙עַשׂ֙ יְרִיעֹ֣ת עִזִּ֔ים לְאֹ֖הֶל עַל־הַמִּשְׁכָּ֑ן עַשְׁתֵּֽי־עֶשְׂרֵ֥ה יְרִיעֹ֖ת עָשָׂ֥ה אֹתָֽם׃ הוּא עָשָׂה יְרִיעוֹת מִשְּׂעַר עִזִּים לְאוֹהֶל מֵעַל הַמִּשְׁכָּן, אַחַת עֶשְׂרֵה יְרִיעוֹת עָשָׂה אוֹתָם.
טו אֹ֜רֶךְ הַיְרִיעָ֣ה הָאַחַ֗ת שְׁלֹשִׁים֙ בָּֽאַמָּ֔ה וְאַרְבַּ֣ע אַמּ֔וֹת רֹ֖חַב הַיְרִיעָ֣ה הָאֶחָ֑ת מִדָּ֣ה אַחַ֔ת לְעַשְׁתֵּ֥י עֶשְׂרֵ֖ה יְרִיעֹֽת׃ אוֹרֶךְ כָּל יְרִיעָה שְׁלוֹשִׁים אַמָּה וְהָרוֹחַב שֶׁל כָּל יְרִיעָה אַרְבַּע אַמּוֹת, מִדָּה אַחַת לְאַחַת עֶשְׂרֵה הַיְּרִיעוֹת.
טז וַיְחַבֵּ֛ר אֶת־חֲמֵ֥שׁ הַיְרִיעֹ֖ת לְבָ֑ד וְאֶת־שֵׁ֥שׁ הַיְרִיעֹ֖ת לְבָֽד׃ הוּא חִבֵּר אֶת חֲמֵשׁ הַיְּרִיעוֹת לְבָד וְאֶת שֵׁשׁ הַיְּרִיעוֹת לְבָד.
יז וַיַּ֜עַשׂ לֻֽלָאֹ֣ת חֲמִשִּׁ֗ים עַ֚ל שְׂפַ֣ת הַיְרִיעָ֔ה הַקִּיצֹנָ֖ה בַּמַּחְבָּ֑רֶת וַחֲמִשִּׁ֣ים לֻלָאֹ֗ת עָשָׂה֙ עַל־שְׂפַ֣ת הַיְרִיעָ֔ה הַחֹבֶ֖רֶת הַשֵּׁנִֽית׃ הוּא עָשָׂה חֲמִשִּׁים לוּלָאוֹת עַל שְׂפַת הַיְּרִיעָה הַקִּיצוֹנָה בַּמָּקוֹם שֶׁל הַחִבּוּר, וַחֲמִשִּׁים לוּלָאוֹת עָשָׂה עַל שְׂפַת הַיְּרִיעָה בַּצַּד הַשֵּׁנִי שֶׁל הַחִבּוּר.
יח וַיַּ֛עַשׂ קַרְסֵ֥י נְחֹ֖שֶׁת חֲמִשִּׁ֑ים לְחַבֵּ֥ר אֶת־הָאֹ֖הֶל לִֽהְיֹ֥ת אֶחָֽד׃ הוּא עָשָׂה חֲמִשִּׁים קַרְסֵי נְחוֹשֶׁת כְּדֵי לְחַבֵּר אֶת הָאוֹהֶל שֶׁיִּהְיֶה מְאוּחָד.
יט וַיַּ֤עַשׂ מִכְסֶה֙ לָאֹ֔הֶל עֹרֹ֥ת אֵילִ֖ם מְאׇדָּמִ֑ים וּמִכְסֵ֛ה עֹרֹ֥ת תְּחָשִׁ֖ים מִלְמָֽעְלָה׃ הוּא עָשָׂה כִּסּוּי לָאוֹהֶל מֵעוֹרוֹת אֵילִים צְבוּעִים בְּאָדוֹם, וְכִסּוּי מֵעוֹרוֹת הַחַיָּה תַּחַשׁ מִלְּמַעְלָה.

חמישי[עריכה]

ס פָּרָשַׁת הַקְּרָשִׁים
כ וַיַּ֥עַשׂ אֶת־הַקְּרָשִׁ֖ים לַמִּשְׁכָּ֑ן עֲצֵ֥י שִׁטִּ֖ים עֹמְדִֽים׃ הוּא עָשָׂה אֶת הַקְּרָשִׁים לַמִּשְׁכָּן מֵעֲצֵי שִׁטָּה עוֹמְדִים.
כא עֶ֥שֶׂר אַמֹּ֖ת אֹ֣רֶךְ הַקָּ֑רֶשׁ וְאַמָּה֙ וַחֲצִ֣י הָֽאַמָּ֔ה רֹ֖חַב הַקֶּ֥רֶשׁ הָאֶחָֽד׃ אוֹרֶךְ כָּל קֶרֶשׁ עֶשֶׂר אַמּוֹת וְרוֹחַב כָּל קֶרֶשׁ אַמָּה וָחֵצִי.
כב שְׁתֵּ֣י יָדֹ֗ת לַקֶּ֙רֶשׁ֙ הָֽאֶחָ֔ד מְשֻׁ֨לָּבֹ֔ת אַחַ֖ת אֶל־אֶחָ֑ת כֵּ֣ן עָשָׂ֔ה לְכֹ֖ל קַרְשֵׁ֥י הַמִּשְׁכָּֽן׃ לְכָל קֶרֶשׁ שְׁתֵּי יָדִיּוֹת מַקְבִּילוֹת אַחַת לַשְּׁנִיָּה – כָּךְ עָשָׂה לְכָל קַרְשֵׁי הַמִּשְׁכָּן.
כג וַיַּ֥עַשׂ אֶת־הַקְּרָשִׁ֖ים לַמִּשְׁכָּ֑ן עֶשְׂרִ֣ים קְרָשִׁ֔ים לִפְאַ֖ת נֶ֥גֶב תֵּימָֽנָה׃ הוּא עָשָׂה אֶת הַקְּרָשִׁים לַמִּשְׁכָּן, עֶשְׂרִים קְרָשִׁים לְצַד דָּרוֹם.
כד וְאַרְבָּעִים֙ אַדְנֵי־כֶ֔סֶף עָשָׂ֕ה תַּ֖חַת עֶשְׂרִ֣ים הַקְּרָשִׁ֑ים שְׁנֵ֨י אֲדָנִ֜ים תַּֽחַת־הַקֶּ֤רֶשׁ הָאֶחָד֙ לִשְׁתֵּ֣י יְדֹתָ֔יו וּשְׁנֵ֧י אֲדָנִ֛ים תַּֽחַת־הַקֶּ֥רֶשׁ הָאֶחָ֖ד לִשְׁתֵּ֥י יְדֹתָֽיו׃ וְאַרְבָּעִים בְּסִיסֵי כֶּסֶף עָשָׂה מִתַּחַת לְעֶשְׂרִים הַקְּרָשִׁים, שְׁנֵי בְּסִיסִים מִתַּחַת לְכָל קֶרֶשׁ לִשְׁתֵּי יָדִיּוֹתָיו.
כה וּלְצֶ֧לַע הַמִּשְׁכָּ֛ן הַשֵּׁנִ֖ית לִפְאַ֣ת צָפ֑וֹן עָשָׂ֖ה עֶשְׂרִ֥ים קְרָשִֽׁים׃ וּבַדּוֹפֶן הַשְּׁנִיָּה שֶׁל הַמִּשְׁכָּן – לְצַד צָפוֹן, עָשָׂה עֶשְׂרִים קְרָשִׁים.
כו וְאַרְבָּעִ֥ים אַדְנֵיהֶ֖ם כָּ֑סֶף שְׁנֵ֣י אֲדָנִ֗ים תַּ֚חַת הַקֶּ֣רֶשׁ הָאֶחָ֔ד וּשְׁנֵ֣י אֲדָנִ֔ים תַּ֖חַת הַקֶּ֥רֶשׁ הָאֶחָֽד׃ וְאַרְבָּעִים בְּסִיסֵיהֶם עֲשׂוּיִים מִכֶּסֶף, שְׁנֵי בְּסִיסִים מִתַּחַת לְכָל קֶרֶשׁ.
כז וּֽלְיַרְכְּתֵ֥י הַמִּשְׁכָּ֖ן יָ֑מָּה עָשָׂ֖ה שִׁשָּׁ֥ה קְרָשִֽׁים׃ וּבְסוֹף הַמִּשְׁכָּן לְצַד מַעֲרָב עָשָׂה שִׁשָּׁה קְרָשִׁים.
כח וּשְׁנֵ֤י קְרָשִׁים֙ עָשָׂ֔ה לִמְקֻצְעֹ֖ת הַמִּשְׁכָּ֑ן בַּיַּרְכָתָֽיִם׃ וּשְׁנֵי קְרָשִׁים עָשָׂה בְּזָוִיּוֹת הַמִּשְׁכָּן בַּקְּצָווֹת.
כט וְהָי֣וּ תוֹאֲמִם֮ מִלְּמַ֒טָּה֒ וְיַחְדָּ֗ו יִהְי֤וּ תַמִּים֙ אֶל־רֹאשׁ֔וֹ אֶל־הַטַּבַּ֖עַת הָאֶחָ֑ת כֵּ֚ן עָשָׂ֣ה לִשְׁנֵיהֶ֔ם לִשְׁנֵ֖י הַמִּקְצֹעֹֽת׃ וְהַקְּרָשִׁים הָיוּ מַקְבִּילִים מִלְּמַטָּה, וְהָיוּ מֻצְמָדִים יַחַד עַל רָאשֵׁיהֶם בְּטַבַּעַת אַחַת, וְכָךְ עָשָׂה גַּם לִשְׁנֵי הַקְּרָשִׁים שֶׁבַּפִּנּוֹת.
ל וְהָיוּ֙ שְׁמֹנָ֣ה קְרָשִׁ֔ים וְאַדְנֵיהֶ֣ם כֶּ֔סֶף שִׁשָּׁ֥ה עָשָׂ֖ר אֲדָנִ֑ים שְׁנֵ֤י אֲדָנִים֙ שְׁנֵ֣י אֲדָנִ֔ים תַּ֖חַת הַקֶּ֥רֶשׁ הָאֶחָֽד׃ יַחַד הֵם הָיוּ שְׁמוֹנָה קְרָשִׁים בַּמַּעֲרָב, וּבְסִיסֵיהֶם מִכֶּסֶף שִׁשָּׁה עָשָׂר בְּסִיסִים, שְׁנֵי בְּסִיסִים תַּחַת כָּל קֶרֶשׁ.
לא וַיַּ֥עַשׂ בְּרִיחֵ֖י עֲצֵ֣י שִׁטִּ֑ים חֲמִשָּׁ֕ה לְקַרְשֵׁ֥י צֶֽלַע־הַמִּשְׁכָּ֖ן הָאֶחָֽת׃ הוּא עָשָׂה חֲמִשָּׁה בְּרִיחִים מֵעֲצֵי שִׁטָּה לְקַרְשֵׁי אַחַת מִדְּפָנוֹת הַמִּשְׁכָּן.
לב וַחֲמִשָּׁ֣ה בְרִיחִ֔ם לְקַרְשֵׁ֥י צֶֽלַע־הַמִּשְׁכָּ֖ן הַשֵּׁנִ֑ית וַחֲמִשָּׁ֤ה בְרִיחִם֙ לְקַרְשֵׁ֣י הַמִּשְׁכָּ֔ן לַיַּרְכָתַ֖יִם יָֽמָּה׃ וַחֲמִשָּׁה בְּרִיחִים לְקַרְשֵׁי הַדּוֹפֶן הַשְּׁנִיָּה שֶׁל הַמִּשְׁכָּן, וַחֲמִשָּׁה בְּרִיחִים לְקַרְשֵׁי הַמִּשְׁכָּן שֶׁבַּקָּצֶה הַמַּעֲרָבִי.
לג וַיַּ֖עַשׂ אֶת־הַבְּרִ֣יחַ הַתִּיכֹ֑ן לִבְרֹ֙חַ֙ בְּת֣וֹךְ הַקְּרָשִׁ֔ים מִן־הַקָּצֶ֖ה אֶל־הַקָּצֶֽה׃ הוּא עָשָׂה אֶת הַבָּרִיחַ הָאֶמְצָעִי שֶׁיַּעֲבוֹר בְּתוֹךְ הַקְּרָשִׁים מִקָּצֶה לְקָצֶה.
לד וְֽאֶת־הַקְּרָשִׁ֞ים צִפָּ֣ה זָהָ֗ב וְאֶת־טַבְּעֹתָם֙ עָשָׂ֣ה זָהָ֔ב בָּתִּ֖ים לַבְּרִיחִ֑ם וַיְצַ֥ף אֶת־הַבְּרִיחִ֖ם זָהָֽב׃ וְאֶת הַקְּרָשִׁים צִפָּה בְּזָהָב וְאֶת טַבְּעוֹתֵיהֶם עָשָׂה מִזָּהָב – מָקוֹם לַבְּרִיחִים, וְצִפָּה אֶת הַבְּרִיחִים בְּזָהָב.
לה וַיַּ֙עַשׂ֙ אֶת־הַפָּרֹ֔כֶת תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מׇשְׁזָ֑ר מַעֲשֵׂ֥ה חֹשֵׁ֛ב עָשָׂ֥ה אֹתָ֖הּ כְּרֻבִֽים׃ הוּא עָשָׂה אֶת הַפָּרוֹכֶת מִתְּכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת וּפִשְׁתָּן שָׁזוּר, בִּמְלֶאכֶת אֳמָנוּת עָשָׂה אוֹתָהּ, בְּצוּרוֹת שֶׁל מַלְאָכִים.
לו וַיַּ֣עַשׂ לָ֗הּ אַרְבָּעָה֙ עַמּוּדֵ֣י שִׁטִּ֔ים וַיְצַפֵּ֣ם זָהָ֔ב וָוֵיהֶ֖ם זָהָ֑ב וַיִּצֹ֣ק לָהֶ֔ם אַרְבָּעָ֖ה אַדְנֵי־כָֽסֶף׃ הוּא עָשָׂה לָהּ אַרְבָּעָה עַמּוּדֵי שִׁטָּה וְצִפָּה אוֹתָם בְּזָהָב וְעָשָׂה אֶת וָוֵיהֶם מִזָּהָב, וְיָצַק לָהֶם אַרְבָּעָה בְּסִיסִים מִכֶּסֶף.
לז וַיַּ֤עַשׂ מָסָךְ֙ לְפֶ֣תַח הָאֹ֔הֶל תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מׇשְׁזָ֑ר מַעֲשֵׂ֖ה רֹקֵֽם׃ הוּא עָשָׂה מָסָךְ לְפֶתַח הָאוֹהֶל מִתְּכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת וּפִשְׁתָּן שָׁזוּר, בִּמְלֶאכֶת רִקְמָה.
לח וְאֶת־עַמּוּדָ֤יו חֲמִשָּׁה֙ וְאֶת־וָ֣וֵיהֶ֔ם וְצִפָּ֧ה רָאשֵׁיהֶ֛ם וַחֲשֻׁקֵיהֶ֖ם זָהָ֑ב וְאַדְנֵיהֶ֥ם חֲמִשָּׁ֖ה נְחֹֽשֶׁת׃ וְעָשָׂה אֶת חֲמֵשֶׁת עַמּוּדָיו וְאֶת וָוֵיהֶם, וְהוּא צִפָּה אֶת רָאשֵׁיהֶם וְאֶת חִשֻּׁקֵיהֶם בְּזָהָב, וְעָשָׂה אֶת בְּסִיסֵיהֶם חֲמִשָּׁה מִנְּחוֹשֶׁת.
פ פָּרָשַׁת הָאָרוֹן
לז א וַיַּ֧עַשׂ בְּצַלְאֵ֛ל אֶת־הָאָרֹ֖ן עֲצֵ֣י שִׁטִּ֑ים אַמָּתַ֨יִם וָחֵ֜צִי אׇרְכּ֗וֹ וְאַמָּ֤ה וָחֵ֙צִי֙ רׇחְבּ֔וֹ וְאַמָּ֥ה וָחֵ֖צִי קֹמָתֽוֹ׃ בְּצַלְאֵל עָשָׂה אֶת הָאָרוֹן מֵעֲצֵי שִׁטָּה, אַמָּתַיִם וָחֵצִי אָרְכּוֹ וְאַמָּה וָחֵצִי רָחְבּוֹ וְאַמָּה וָחֵצִי גָּבְהוֹ.
ב וַיְצַפֵּ֛הוּ זָהָ֥ב טָה֖וֹר מִבַּ֣יִת וּמִח֑וּץ וַיַּ֥עַשׂ ל֛וֹ זֵ֥ר זָהָ֖ב סָבִֽיב׃ וְצִפָּה אוֹתוֹ בְּזָהָב טָהוֹר מִבִּפְנִים וּמִבַּחוּץ, וְעָשָׂה לוֹ זֵר זָהָב מִסָּבִיב.
ג וַיִּצֹ֣ק ל֗וֹ אַרְבַּע֙ טַבְּעֹ֣ת זָהָ֔ב עַ֖ל אַרְבַּ֣ע פַּעֲמֹתָ֑יו וּשְׁתֵּ֣י טַבָּעֹ֗ת עַל־צַלְעוֹ֙ הָֽאֶחָ֔ת וּשְׁתֵּי֙ טַבָּעֹ֔ת עַל־צַלְע֖וֹ הַשֵּׁנִֽית׃ הוּא יָצַק לוֹ אַרְבַּע טַבָּעוֹת מִזָּהָב עַל אַרְבַּע זָוִיּוֹתָיו – שְׁתֵּי טַבָּעוֹת עַל דּוֹפֶן אַחַת וּשְׁתֵּי טַבָּעוֹת עַל הַדּוֹפֶן הַשְּׁנִיָּה.
ד וַיַּ֥עַשׂ בַּדֵּ֖י עֲצֵ֣י שִׁטִּ֑ים וַיְצַ֥ף אֹתָ֖ם זָהָֽב׃ הוּא עָשָׂה מוֹטוֹת מֵעֲצֵי שִׁטָּה וְצִפָּה אוֹתָם בְּזָהָב.
ה וַיָּבֵ֤א אֶת־הַבַּדִּים֙ בַּטַּבָּעֹ֔ת עַ֖ל צַלְעֹ֣ת הָאָרֹ֑ן לָשֵׂ֖את אֶת־הָאָרֹֽן׃ הוּא הִכְנִיס אֶת הַמּוֹטוֹת בַּטַּבָּעוֹת עַל דְּפָנוֹת הָאָרוֹן כְּדֵי לָשֵׂאת אֶת הָאָרוֹן.
ו וַיַּ֥עַשׂ כַּפֹּ֖רֶת זָהָ֣ב טָה֑וֹר אַמָּתַ֤יִם וָחֵ֙צִי֙ אׇרְכָּ֔הּ וְאַמָּ֥ה וָחֵ֖צִי רׇחְבָּֽהּ׃ הוּא עָשָׂה כִּסּוּי מִזָּהָב טָהוֹר, אָרְכּוֹ שְׂתֵּי אַמּוֹת וָחֵצִי וְרָחְבּוֹ אַמָּה וָחֵצִי.
ז וַיַּ֛עַשׂ שְׁנֵ֥י כְרֻבִ֖ים זָהָ֑ב מִקְשָׁה֙ עָשָׂ֣ה אֹתָ֔ם מִשְּׁנֵ֖י קְצ֥וֹת הַכַּפֹּֽרֶת׃ הוּא עָשָׂה שְׁתֵּי צוּרוֹת שֶׁל מַלְאָכִים מִזָּהָב, הוּא עָשָׂה אוֹתָן מִגּוּשׁ אֶחָד מִשְּׁנֵי קְצָווֹת הַכִּסּוּי.
ח כְּרוּב־אֶחָ֤ד מִקָּצָה֙ מִזֶּ֔ה וּכְרוּב־אֶחָ֥ד מִקָּצָ֖ה מִזֶּ֑ה מִן־הַכַּפֹּ֛רֶת עָשָׂ֥ה אֶת־הַכְּרֻבִ֖ים מִשְּׁנֵ֥י קצוותו קְצוֹתָֽיו׃ צוּרָה אַחַת בְּקָצֶה אֶחָד וְצוּרָה אַחַת בַּקָּצֶה הַשֵּׁנִי, הוּא עָשָׂה אֶת צוּרוֹת הַמַּלְאָכִים מִשְּׁנֵי הַקְּצָווֹת כְּחֵלֶק מֵהַכִּסּוּי.
ט וַיִּהְי֣וּ הַכְּרֻבִים֩ פֹּרְשֵׂ֨י כְנָפַ֜יִם לְמַ֗עְלָה סֹֽכְכִ֤ים בְּכַנְפֵיהֶם֙ עַל־הַכַּפֹּ֔רֶת וּפְנֵיהֶ֖ם אִ֣ישׁ אֶל־אָחִ֑יו אֶ֨ל־הַכַּפֹּ֔רֶת הָי֖וּ פְּנֵ֥י הַכְּרֻבִֽים׃ הַמַּלְאָכִים הָיוּ פּוֹרְשֵׂי כְּנָפַיִם לְמַעְלָה סוֹכְכִים בְּכַנְפֵיהֶם עַל הַכִּסּוּי וּפְנֵיהֶם אֶחָד אֶל הַשֵּׁנִי, פְּנֵי צוּרוֹת הַמַּלְאָכִים הָיוּ לְעֵבֶר הַכִּסּוּי.
פ פָּרָשַׁת הַשֻּׁלְחָן
י וַיַּ֥עַשׂ אֶת־הַשֻּׁלְחָ֖ן עֲצֵ֣י שִׁטִּ֑ים אַמָּתַ֤יִם אׇרְכּוֹ֙ וְאַמָּ֣ה רׇחְבּ֔וֹ וְאַמָּ֥ה וָחֵ֖צִי קֹמָתֽוֹ׃ הוּא עָשָׂה אֶת הַשֻּׁלְחָן מֵעֲצֵי שִׁטָּה, שְׁתֵּי אַמּוֹת אָרְכּוֹ וְאַמָּה רָחְבּוֹ וְאַמָּה וָחֵצִי גָּבְהוֹ.
יא וַיְצַ֥ף אֹת֖וֹ זָהָ֣ב טָה֑וֹר וַיַּ֥עַשׂ ל֛וֹ זֵ֥ר זָהָ֖ב סָבִֽיב׃ הוּא צִפָּה אוֹתוֹ בְּזָהָב טָהוֹר וְעָשָׂה לוֹ זֵר זָהָב מִסָּבִיב.
יב וַיַּ֨עַשׂ ל֥וֹ מִסְגֶּ֛רֶת טֹ֖פַח סָבִ֑יב וַיַּ֧עַשׂ זֵר־זָהָ֛ב לְמִסְגַּרְתּ֖וֹ סָבִֽיב׃ הוּא עָשָׂה לוֹ מִסְגֶּרֶת בְּגוֹבַהּ טֶפַח מִסָּבִיב, הוּא עָשָׂה זֵר זָהָב לְמִסְגַּרְתּוֹ מִסָּבִיב.
יג וַיִּצֹ֣ק ל֔וֹ אַרְבַּ֖ע טַבְּעֹ֣ת זָהָ֑ב וַיִּתֵּן֙ אֶת־הַטַּבָּעֹ֔ת עַ֚ל אַרְבַּ֣ע הַפֵּאֹ֔ת אֲשֶׁ֖ר לְאַרְבַּ֥ע רַגְלָֽיו׃ הוּא יָצַק לוֹ אַרְבַּע טַבָּעוֹת מִזָּהָב, וְשָׂם אֶת הַטַּבָּעוֹת עַל אַרְבַּע הַפִּנּוֹת שֶׁלְּיַד אַרְבַּע רַגְלָיו.
יד לְעֻמַּת֙ הַמִּסְגֶּ֔רֶת הָי֖וּ הַטַּבָּעֹ֑ת בָּתִּים֙ לַבַּדִּ֔ים לָשֵׂ֖את אֶת־הַשֻּׁלְחָֽן׃ הַטַּבָּעוֹת הָיוּ מוּל הַמִּסְגֶּרֶת – מָקוֹם לַמּוֹטוֹת לָשֵׂאת אֶת הַשֻּׁלְחָן.
טו וַיַּ֤עַשׂ אֶת־הַבַּדִּים֙ עֲצֵ֣י שִׁטִּ֔ים וַיְצַ֥ף אֹתָ֖ם זָהָ֑ב לָשֵׂ֖את אֶת־הַשֻּׁלְחָֽן׃ הוּא עָשָׂה אֶת הַמּוֹטוֹת מֵעֲצֵי שִׁטָּה וְצִפָּה אוֹתָם בְּזָהָב, לָשֵׂאת בָּהֶם אֶת הַשֻּׁלְחָן.
טז וַיַּ֜עַשׂ אֶֽת־הַכֵּלִ֣ים ׀ אֲשֶׁ֣ר עַל־הַשֻּׁלְחָ֗ן אֶת־קְעָרֹתָ֤יו וְאֶת־כַּפֹּתָיו֙ וְאֵת֙ מְנַקִּיֹּתָ֔יו וְאֶ֨ת־הַקְּשָׂוֺ֔ת אֲשֶׁ֥ר יֻסַּ֖ךְ בָּהֵ֑ן זָהָ֖ב טָהֽוֹר׃ הוּא עָשָׂה מִזָּהָב טָהוֹר אֶת הַכֵּלִים שֶׁעַל הַשֻּׁלְחָן, אֶת קְעָרוֹתָיו וְאֶת כַּפּוֹתָיו וְאֶת הָעַמּוּדִים שֶׁתּוֹמְכִים מֵהַצְּדָדִים, וְאֶת הַמַּקְלוֹת שֶׁסּוֹכְכִים מֵעַל הַשֻּׁלְחָן שֶׁעֲלֵיהֶם מוּנָח הַלֶּחֶם.

ששי (שלישי במחוברין)[עריכה]

פ פָּרָשַׁת הַמְּנוֹרָה
יז וַיַּ֥עַשׂ אֶת־הַמְּנֹרָ֖ה זָהָ֣ב טָה֑וֹר מִקְשָׁ֞ה עָשָׂ֤ה אֶת־הַמְּנֹרָה֙ יְרֵכָ֣הּ וְקָנָ֔הּ גְּבִיעֶ֛יהָ כַּפְתֹּרֶ֥יהָ וּפְרָחֶ֖יהָ מִמֶּ֥נָּה הָיֽוּ׃ הוּא עָשָׂה אֶת הַמְּנוֹרָה מִזָּהָב טָהוֹר, מִגּוּשׁ אֶחָד עָשָׂה אֶת הַמְּנוֹרָה – רַגְלָהּ וְהַקָּנֶה שֶׁלָּהּ וּגְבִיעֶיהָ כַּפְתּוֹרֶיהָ וּפְרָחֶיהָ – הָיוּ חֵלֶק מִמֶּנָּה.
יח וְשִׁשָּׁ֣ה קָנִ֔ים יֹצְאִ֖ים מִצִּדֶּ֑יהָ שְׁלֹשָׁ֣ה ׀ קְנֵ֣י מְנֹרָ֗ה מִצִּדָּהּ֙ הָֽאֶחָ֔ד וּשְׁלֹשָׁה֙ קְנֵ֣י מְנֹרָ֔ה מִצִּדָּ֖הּ הַשֵּׁנִֽי׃ וְשִׁשָּׁה קָנִים יוֹצְאִים מִצִּדֶּיהָ, שְׁלוֹשָׁה קְנֵי מְנוֹרָה מִצַּד אֶחָד וּשְׁלוֹשָׁה קְנֵי מְנוֹרָה מִצַּד שֵׁנִי.
יט שְׁלֹשָׁ֣ה גְ֠בִעִ֠ים מְֽשֻׁקָּדִ֞ים בַּקָּנֶ֣ה הָאֶחָד֮ כַּפְתֹּ֣ר וָפֶ֒רַח֒ וּשְׁלֹשָׁ֣ה גְבִעִ֗ים מְשֻׁקָּדִ֛ים בְּקָנֶ֥ה אֶחָ֖ד כַּפְתֹּ֣ר וָפָ֑רַח כֵּ֚ן לְשֵׁ֣שֶׁת הַקָּנִ֔ים הַיֹּצְאִ֖ים מִן־הַמְּנֹרָֽה׃ בְּקָנֶה אֶחָד יֵשׁ שְׁלוֹשָׁה תַּבְלִיטִים שֶׁל גְּבִיעִים וְכַפְתּוֹר וּפֶרַח, וּבְקָנֶה אַחֵר שְׁלוֹשָׁה תַּבְלִיטִים שֶׁל גְּבִיעִים וְכַפְתּוֹר וּפֶרַח, כָּךְ בְּשֵׁשֶׁת הַקָּנִים שֶׁיּוֹצְאִים מֵהַקָּנֶה הַמֶּרְכָּזִי שֶׁל הַמְּנוֹרָה.
כ וּבַמְּנֹרָ֖ה אַרְבָּעָ֣ה גְבִעִ֑ים מְשֻׁ֨קָּדִ֔ים כַּפְתֹּרֶ֖יהָ וּפְרָחֶֽיהָ׃ וּבַקָּנֶה הַמֶּרְכָּזִי שֶׁל הַמְּנוֹרָה אַרְבָּעָה גְּבִיעִים, וְתַבְלִיטֵי כַּפְתּוֹרִים וּפְרָחִים.
כא וְכַפְתֹּ֡ר תַּ֩חַת֩ שְׁנֵ֨י הַקָּנִ֜ים מִמֶּ֗נָּה וְכַפְתֹּר֙ תַּ֣חַת שְׁנֵ֤י הַקָּנִים֙ מִמֶּ֔נָּה וְכַפְתֹּ֕ר תַּֽחַת־שְׁנֵ֥י הַקָּנִ֖ים מִמֶּ֑נָּה לְשֵׁ֙שֶׁת֙ הַקָּנִ֔ים הַיֹּצְאִ֖ים מִמֶּֽנָּה׃ וְכַפְתּוֹר מִתַּחַת לִשְׁנֵי הַקָּנִים שֶׁיּוֹצְאִים מִמֶּנָּה, וְכַפְתּוֹר מִתַּחַת לִשְׁנֵי הַקָּנִים שֶׁיּוֹצְאִים מִמֶּנָּה, וְכַפְתּוֹר מִתַּחַת לִשְׁנֵי הַקָּנִים שֶׁיּוֹצְאִים מִמֶּנָּה – כָּךְ בְּשֵׁשֶׁת הַקָּנִים שֶׁיּוֹצְאִים מֵהַקָּנֶה הַמֶּרְכָּזִי.
כב כַּפְתֹּרֵיהֶ֥ם וּקְנֹתָ֖ם מִמֶּ֣נָּה הָי֑וּ כֻּלָּ֛הּ מִקְשָׁ֥ה אַחַ֖ת זָהָ֥ב טָהֽוֹר׃ הַכַּפְתּוֹרִים וְהַקָּנִים הָיוּ חֵלֶק מִמֶּנָּה – כֻּלָּהּ עֲשׂוּיָה מִגּוּשׁ אֶחָד שֶׁל זָהָב טָהוֹר.
כג וַיַּ֥עַשׂ אֶת־נֵרֹתֶ֖יהָ שִׁבְעָ֑ה וּמַלְקָחֶ֥יהָ וּמַחְתֹּתֶ֖יהָ זָהָ֥ב טָהֽוֹר׃ הוּא עָשָׂה אֶת נֵרוֹתֶיהָ שִׁבְעָה, וְאֶת הַמֶּלְקָחַיִם וְהַמַּחְתּוֹת שֶׁלָּהּ, מִזָּהָב טָהוֹר.
כד כִּכָּ֛ר זָהָ֥ב טָה֖וֹר עָשָׂ֣ה אֹתָ֑הּ וְאֵ֖ת כׇּל־כֵּלֶֽיהָ׃ הוּא עָשָׂה אוֹתָהּ וְאֶת כָּל כֵּלֶיהָ מִכִּכָּר אַחַת שֶׁל זָהָב טָהוֹר.
פ פָּרָשַׁת מִזְבַּח הַקְּטוֹרֶת
כה וַיַּ֛עַשׂ אֶת־מִזְבַּ֥ח הַקְּטֹ֖רֶת עֲצֵ֣י שִׁטִּ֑ים אַמָּ֣ה אׇרְכּוֹ֩ וְאַמָּ֨ה רׇחְבּ֜וֹ רָב֗וּעַ וְאַמָּתַ֙יִם֙ קֹֽמָת֔וֹ מִמֶּ֖נּוּ הָי֥וּ קַרְנֹתָֽיו׃ הוּא עָשָׂה אֶת מִזְבַּח הַקְּטוֹרֶת מֵעֲצֵי שִׁטָּה, בְּצוּרַת רִבּוּעַ שֶׁאָרְכּוֹ אַמָּה וְרָחְבּוֹ אַמָּה, וְגָבְהוֹ שְׁתֵּי אַמּוֹת, קַרְנוֹתָיו הָיוּ חֵלֶק מִמֶּנּוּ.
כו וַיְצַ֨ף אֹת֜וֹ זָהָ֣ב טָה֗וֹר אֶת־גַּגּ֧וֹ וְאֶת־קִירֹתָ֛יו סָבִ֖יב וְאֶת־קַרְנֹתָ֑יו וַיַּ֥עַשׂ ל֛וֹ זֵ֥ר זָהָ֖ב סָבִֽיב׃ הוּא צִפָּה אוֹתוֹ בְּזָהָב טָהוֹר – אֶת גַּגּוֹ וְאֶת קִירוֹתָיו מִסָּבִיב וְאֶת קַרְנוֹתָיו, הוּא עָשָׂה לוֹ זֵר מִזָּהָב מִסָּבִיב.
כז וּשְׁתֵּי֩ טַבְּעֹ֨ת זָהָ֜ב עָֽשָׂה־ל֣וֹ ׀ מִתַּ֣חַת לְזֵר֗וֹ עַ֚ל שְׁתֵּ֣י צַלְעֹתָ֔יו עַ֖ל שְׁנֵ֣י צִדָּ֑יו לְבָתִּ֣ים לְבַדִּ֔ים לָשֵׂ֥את אֹת֖וֹ בָּהֶֽם׃ וְעָשָׂה לוֹ שְׁתֵּי טַבָּעוֹת מִזָּהָב מִתַּחַת לְזֵרוֹ עַל שְׁתֵּי דְּפָנוֹתָיו עַל שְׁנֵי צִדָּיו, לְמָקוֹם לַמּוֹטוֹת לָשֵׂאת אוֹתוֹ בָּהֶם.
כח וַיַּ֥עַשׂ אֶת־הַבַּדִּ֖ים עֲצֵ֣י שִׁטִּ֑ים וַיְצַ֥ף אֹתָ֖ם זָהָֽב׃ הוּא עָשָׂה אֶת הַמּוֹטוֹת מֵעֲצֵי שִׁטָּה, וְצִפָּה אוֹתָם בְּזָהָב.
כט וַיַּ֜עַשׂ אֶת־שֶׁ֤מֶן הַמִּשְׁחָה֙ קֹ֔דֶשׁ וְאֶת־קְטֹ֥רֶת הַסַּמִּ֖ים טָה֑וֹר מַעֲשֵׂ֖ה רֹקֵֽחַ׃ הוּא עָשָׂה אֶת הַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ – קוֹדֶשׁ, וְאֶת קְטוֹרֶת הַבְּשָׂמִים – טָהוֹר, מַעֲשֵׂה רְקִיחָה.

שביעי (רביעי במחוברין)[עריכה]

ס פָּרָשַׁת מִזְבַּח הָעוֹלָה
לח א וַיַּ֛עַשׂ אֶת־מִזְבַּ֥ח הָעֹלָ֖ה עֲצֵ֣י שִׁטִּ֑ים חָמֵשׁ֩ אַמּ֨וֹת אׇרְכּ֜וֹ וְחָֽמֵשׁ־אַמּ֤וֹת רׇחְבּוֹ֙ רָב֔וּעַ וְשָׁלֹ֥שׁ אַמּ֖וֹת קֹמָתֽוֹ׃ הוּא עָשָׂה אֶת מִזְבַּח הָעוֹלָה מֵעֲצֵי שִׁטָּה, בְּצוּרַת רִבּוּעַ שֶׁאָרְכּוֹ חָמֵשׁ אַמּוֹת וְרָחְבּוֹ חָמֵשׁ אַמּוֹת, וְגָבְהוֹ שָׁלוֹשׁ אַמּוֹת.
ב וַיַּ֣עַשׂ קַרְנֹתָ֗יו עַ֚ל אַרְבַּ֣ע פִּנֹּתָ֔יו מִמֶּ֖נּוּ הָי֣וּ קַרְנֹתָ֑יו וַיְצַ֥ף אֹת֖וֹ נְחֹֽשֶׁת׃ הוּא עָשָׂה אֶת קַרְנוֹתָיו עַל אַרְבַּע פִּנּוֹתָיו קַרְנוֹתָיו הָיוּ חֵלֶק מִמֶּנּוּ, וְצִפָּה אוֹתוֹ בִּנְחוֹשֶׁת.
ג וַיַּ֜עַשׂ אֶֽת־כׇּל־כְּלֵ֣י הַמִּזְבֵּ֗חַ אֶת־הַסִּירֹ֤ת וְאֶת־הַיָּעִים֙ וְאֶת־הַמִּזְרָקֹ֔ת אֶת־הַמִּזְלָגֹ֖ת וְאֶת־הַמַּחְתֹּ֑ת כׇּל־כֵּלָ֖יו עָשָׂ֥ה נְחֹֽשֶׁת׃ הוּא עָשָׂה אֶת כָּל כְּלֵי הַמִּזְבֵּחַ: אֶת הַסִּירִים לְהוֹצִיא אֶת הָאֵפֶר, הַמַּגְרְפוֹת לְנִקּוּי הָאֵפֶר, הַכּוֹסוֹת לְקַבָּלַת הַדָּם וּזְרִיקָתוֹ, הַקִּלְּשׁוֹנִים לַהֲפִיכַת הַבָּשָׂר, וְהַיָּעִים לִלְקִיחַת הַגֶּחָלִים – אֶת כָּל כֵּלָיו עָשָׂה מִנְּחוֹשֶׁת.
ד וַיַּ֤עַשׂ לַמִּזְבֵּ֙חַ֙ מִכְבָּ֔ר מַעֲשֵׂ֖ה רֶ֣שֶׁת נְחֹ֑שֶׁת תַּ֧חַת כַּרְכֻּבּ֛וֹ מִלְּמַ֖טָּה עַד־חֶצְיֽוֹ׃ הוּא עָשָׂה לַמִּזְבֵּחַ קִשּׁוּט בְּצוּרַת רֶשֶׁת מִנְּחוֹשֶׁת, מִתַּחַת לַבְּלִיטָה שֶׁסְּבִיבוֹ – מִלְּמַטָּה עַד אֶמְצָעִיתוֹ.
ה וַיִּצֹ֞ק אַרְבַּ֧ע טַבָּעֹ֛ת בְּאַרְבַּ֥ע הַקְּצָוֺ֖ת לְמִכְבַּ֣ר הַנְּחֹ֑שֶׁת בָּתִּ֖ים לַבַּדִּֽים׃ הוּא יָצַק אַרְבַּע טַבָּעוֹת בְּאַרְבַּע קְצָווֹת רֶשֶׁת הַנְּחוֹשֶׁת – מָקוֹם לַמּוֹטוֹת.
ו וַיַּ֥עַשׂ אֶת־הַבַּדִּ֖ים עֲצֵ֣י שִׁטִּ֑ים וַיְצַ֥ף אֹתָ֖ם נְחֹֽשֶׁת׃ הוּא עָשָׂה אֶת הַמּוֹטוֹת מֵעֲצֵי שִׁטָּה וְצִפָּה אוֹתָם בִּנְחוֹשֶׁת.
ז וַיָּבֵ֨א אֶת־הַבַּדִּ֜ים בַּטַּבָּעֹ֗ת עַ֚ל צַלְעֹ֣ת הַמִּזְבֵּ֔חַ לָשֵׂ֥את אֹת֖וֹ בָּהֶ֑ם נְב֥וּב לֻחֹ֖ת עָשָׂ֥ה אֹתֽוֹ׃ הוּא הִכְנִיס אֶת הַמּוֹטוֹת בַּטַּבָּעוֹת שֶׁעַל צִדֵּי הַמִּזְבֵּחַ כְּדֵי לָשֵׂאת אוֹתוֹ בָּהֶם, הוּא עָשָׂה אוֹתוֹ חָלוּל מֻקָּף בְּלוּחוֹת.
ס פָּרָשַׁת הַכִּיּוֹר
ח וַיַּ֗עַשׂ אֵ֚ת הַכִּיּ֣וֹר נְחֹ֔שֶׁת וְאֵ֖ת כַּנּ֣וֹ נְחֹ֑שֶׁת בְּמַרְאֹת֙ הַצֹּ֣בְאֹ֔ת אֲשֶׁ֣ר צָֽבְא֔וּ פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ הוּא עָשָׂה אֶת הַכִּיּוֹר מִנְּחוֹשֶׁת וְאֶת בְּסִיסוֹ מִנְּחוֹשֶׁת מִמַּרְאוֹת הַנָּשִׁים שֶׁהִתְקָהֲלוּ בְּפֶתַח אוֹהֶל הַהִתְכַּנְּסוּת כְּדֵי לִתְרוֹם.
ס פָּרָשַׁת הֶחָצֵר
ט וַיַּ֖עַשׂ אֶת־הֶחָצֵ֑ר לִפְאַ֣ת ׀ נֶ֣גֶב תֵּימָ֗נָה קַלְעֵ֤י הֶֽחָצֵר֙ שֵׁ֣שׁ מׇשְׁזָ֔ר מֵאָ֖ה בָּאַמָּֽה׃ הוּא עָשָׂה אֶת הֶחָצֵר, לְצַד נֶגֶב לְדָרוֹם – גְּדֵרוֹת קְלוּעוֹת לֶחָצֵר מִפִּשְׁתָּן שָׁזוּר בְּאוֹרֶךְ מֵאָה אַמָּה.
י עַמּוּדֵיהֶ֣ם עֶשְׂרִ֔ים וְאַדְנֵיהֶ֥ם עֶשְׂרִ֖ים נְחֹ֑שֶׁת וָוֵ֧י הָעַמּוּדִ֛ים וַחֲשֻׁקֵיהֶ֖ם כָּֽסֶף׃ עַמּוּדֵיהֶם עֶשְׂרִים וּבְסִיסֵיהֶם עֶשְׂרִים מִנְּחוֹשֶׁת, וָוֵי הָעַמּוּדִים וְחִשּׁוּקֵיהֶם מִכֶּסֶף.
יא וְלִפְאַ֤ת צָפוֹן֙ מֵאָ֣ה בָֽאַמָּ֔ה עַמּוּדֵיהֶ֣ם עֶשְׂרִ֔ים וְאַדְנֵיהֶ֥ם עֶשְׂרִ֖ים נְחֹ֑שֶׁת וָוֵ֧י הָֽעַמּוּדִ֛ים וַחֲשֻׁקֵיהֶ֖ם כָּֽסֶף׃ וּלְצַד צָפוֹן מֵאָה אַמָּה, עַמּוּדֵיהֶם עֶשְׂרִים וּבְסִיסֵיהֶם עֶשְׂרִים מִנְּחוֹשֶׁת, וָוֵי הָעַמּוּדִים וְחִשֻּׁקֵיהֶם מִכֶּסֶף.
יב וְלִפְאַת־יָ֗ם קְלָעִים֙ חֲמִשִּׁ֣ים בָּֽאַמָּ֔ה עַמּוּדֵיהֶ֣ם עֲשָׂרָ֔ה וְאַדְנֵיהֶ֖ם עֲשָׂרָ֑ה וָוֵ֧י הָעַמֻּדִ֛ים וַחֲשׁוּקֵיהֶ֖ם כָּֽסֶף׃ וּלְצַד מַעֲרָב חֲמִשִּׁים אַמָּה גְּדֵרוֹת קְלוּעוֹת, עַמּוּדֵיהֶן עֲשָׂרָה וּבְסִיסֵיהֶן עֲשָׂרָה, וָוֵי הָעַמּוּדִים וְחִשּׁוּקֵיהֶם מִכֶּסֶף.
יג וְלִפְאַ֛ת קֵ֥דְמָה מִזְרָ֖חָה חֲמִשִּׁ֥ים אַמָּֽה׃ וּלְצַד מִזְרָח חֲמִשִּׁים אַמָּה.
יד קְלָעִ֛ים חֲמֵשׁ־עֶשְׂרֵ֥ה אַמָּ֖ה אֶל־הַכָּתֵ֑ף עַמּוּדֵיהֶ֣ם שְׁלֹשָׁ֔ה וְאַדְנֵיהֶ֖ם שְׁלֹשָֽׁה׃ חֲמֵשׁ עֶשְׂרֵה אַמָּה גְּדֵרוֹת קְלוּעוֹת לַדּוֹפֶן שֶׁלְּצַד הַפֶּתַח, עַמּוּדֵיהֶן שְׁלוֹשָׁה וּבְסִיסֵיהֶן שְׁלוֹשָׁה.
טו וְלַכָּתֵ֣ף הַשֵּׁנִ֗ית מִזֶּ֤ה וּמִזֶּה֙ לְשַׁ֣עַר הֶֽחָצֵ֔ר קְלָעִ֕ים חֲמֵ֥שׁ עֶשְׂרֵ֖ה אַמָּ֑ה עַמֻּדֵיהֶ֣ם שְׁלֹשָׁ֔ה וְאַדְנֵיהֶ֖ם שְׁלֹשָֽׁה׃ וְכֵן לַדּוֹפֶן הַשְּׁנִיָּה, מִצַּד זֶה וּמִצַּד זֶה שֶׁל שַׁעַר הֶחָצֵר – חֲמֵשׁ עֶשְׂרֵה אַמָּה גְּדֵרוֹת קְלוּעוֹת, עַמּוּדֵיהֶן שְׁלוֹשָׁה וּבְסִיסֵיהֶן שְׁלוֹשָׁה.
טז כׇּל־קַלְעֵ֧י הֶחָצֵ֛ר סָבִ֖יב שֵׁ֥שׁ מׇשְׁזָֽר׃ כָּל הַגְּדֵרוֹת הַקְּלוּעוֹת שֶׁל הֶחָצֵר סָבִיב עֲשׂוּיוֹת מִפִּשְׁתָּן שָׁזוּר.
יז וְהָאֲדָנִ֣ים לָֽעַמֻּדִים֮ נְחֹ֒שֶׁת֒ וָוֵ֨י הָֽעַמּוּדִ֜ים וַחֲשׁוּקֵיהֶם֙ כֶּ֔סֶף וְצִפּ֥וּי רָאשֵׁיהֶ֖ם כָּ֑סֶף וְהֵם֙ מְחֻשָּׁקִ֣ים כֶּ֔סֶף כֹּ֖ל עַמֻּדֵ֥י הֶחָצֵֽר׃ וְהַבְּסִיסִים לָעַמּוּדִים מִנְּחוֹשֶׁת, וָוֵי הָעַמּוּדִים וְחִשּׁוּקֵיהֶם מִכֶּסֶף וְצִפּוּי רָאשֵׁיהֶם מִכֶּסֶף, וְהֵם – כָּל עַמּוּדֵי הֶחָצֵר, מְחֻשָּׁקִים בְּחִשּׁוּקֵי כֶּסֶף.

מפטיר[עריכה]

יח וּמָסַ֞ךְ שַׁ֤עַר הֶחָצֵר֙ מַעֲשֵׂ֣ה רֹקֵ֔ם תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מׇשְׁזָ֑ר וְעֶשְׂרִ֤ים אַמָּה֙ אֹ֔רֶךְ וְקוֹמָ֤ה בְרֹ֙חַב֙ חָמֵ֣שׁ אַמּ֔וֹת לְעֻמַּ֖ת קַלְעֵ֥י הֶחָצֵֽר׃ וּמָסַךְ שַׁעַר הֶחָצֵר בִּמְלֶאכֶת רִקְמָה מִתְּכֵלֶת וְאַרְגָּמָן וְצֶבַע אָדוֹם מִתּוֹלַעַת וּפִשְׁתָּן שָׁזוּר, הָאוֹרֶךְ עֶשְׂרִים אַמָּה, וְהָרוֹחַב חָמֵשׁ אַמּוֹת – לַגּוֹבַהּ, בְּגוֹבַהּ הַגְּדֵרוֹת הַקְּלוּעוֹת שֶׁל הֶחָצֵר.
יט וְעַמֻּֽדֵיהֶם֙ אַרְבָּעָ֔ה וְאַדְנֵיהֶ֥ם אַרְבָּעָ֖ה נְחֹ֑שֶׁת וָוֵיהֶ֣ם כֶּ֔סֶף וְצִפּ֧וּי רָאשֵׁיהֶ֛ם וַחֲשֻׁקֵיהֶ֖ם כָּֽסֶף׃ וְעַמּוּדֵי הַשַּׁעַר וְהַמָּסָךְ אַרְבָּעָה, וּבְסִיסֵיהֶם אַרְבָּעָה מִנְּחוֹשֶׁת, וָוֵיהֶם מִכֶּסֶף וְצִפּוּי רָאשֵׁיהֶם וְחִשֻּׁקֵיהֶם מִכֶּסֶף.
כ וְֽכׇל־הַיְתֵדֹ֞ת לַמִּשְׁכָּ֧ן וְלֶחָצֵ֛ר סָבִ֖יב נְחֹֽשֶׁת׃ וְכָל הַיְּתֵדוֹת לַמִּשְׁכָּן וְלֶחָצֵר מִסָּבִיב מִנְּחוֹשֶׁת.
