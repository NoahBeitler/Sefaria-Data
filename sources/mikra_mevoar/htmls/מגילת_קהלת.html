קפיצה לניווט קפיצה לחיפוש
א א דִּבְרֵי֙ קֹהֶ֣לֶת בֶּן־דָּוִ֔ד מֶ֖לֶךְ בִּירוּשָׁלָֽ͏ִם׃ אֵלּוּ דִּבְרֵי קוֹהֶלֶת בֶּן דָּוִד – מֶלֶךְ בִּירוּשָׁלַיִם.
ב הֲבֵ֤ל הֲבָלִים֙ אָמַ֣ר קֹהֶ֔לֶת הֲבֵ֥ל הֲבָלִ֖ים הַכֹּ֥ל הָֽבֶל׃ הֶבֶל הֲבָלִים, אוֹמֵר קוֹהֶלֶת, הֶבֶל הֲבָלִים –הַכֹּל הֶבֶל!
ג מַה־יִּתְר֖וֹן לָֽאָדָ֑ם בְּכׇ֨ל־עֲמָל֔וֹ שֶֽׁיַּעֲמֹ֖ל תַּ֥חַת הַשָּֽׁמֶשׁ׃ אֵיזֶה תּוֹעֶלֶת יֵשׁ לָאָדָם בְּכָל עֲמָלוֹ שֶׁהוּא עָמֵל מִתַּחַת לַשֶּׁמֶשׁ?
ד דּ֤וֹר הֹלֵךְ֙ וְד֣וֹר בָּ֔א וְהָאָ֖רֶץ לְעוֹלָ֥ם עֹמָֽדֶת׃ הֲרֵי דּוֹר הוֹלֵךְ וְדוֹר בָּא וְהָאָרֶץ לְעוֹלָם אֵינָהּ מִשְׁתַּנָּה.
ה וְזָרַ֥ח הַשֶּׁ֖מֶשׁ וּבָ֣א הַשָּׁ֑מֶשׁ וְאֶ֨ל־מְקוֹמ֔וֹ שׁוֹאֵ֛ף זוֹרֵ֥חַֽ ה֖וּא שָֽׁם׃ הַשֶּׁמֶשׁ זוֹרַחַת וְהַשֶּׁמֶשׁ שׁוֹקַעַת, וּמִתְקַדֶּמֶת וְחוֹזֶרֶת לִמְקוֹמָהּ הַקּוֹדֶם וְזוֹרַחַת מִשָּׁם.
ו הוֹלֵךְ֙ אֶל־דָּר֔וֹם וְסוֹבֵ֖ב אֶל־צָפ֑וֹן סוֹבֵ֤ב ׀ סֹבֵב֙ הוֹלֵ֣ךְ הָר֔וּחַ וְעַל־סְבִיבֹתָ֖יו שָׁ֥ב הָרֽוּחַ׃ מֵהַזְּרִיחָה הִיא נָעָה אֶל הַדָּרוֹם וְאָז מִסְתּוֹבֶבֶת אֶל הַצָּפוֹן, מִסְתּוֹבֶבֶת וְעוֹבֶרֶת בְּכָל הַצְּדָדִים וּבְסוֹף הַסִּיבוּב חוֹזֶרֶת לַצַּד שֶׁמִּשָּׁם זָרְחָה.
ז כׇּל־הַנְּחָלִים֙ הֹלְכִ֣ים אֶל־הַיָּ֔ם וְהַיָּ֖ם אֵינֶ֣נּוּ מָלֵ֑א אֶל־מְק֗וֹם שֶׁ֤הַנְּחָלִים֙ הֹֽלְכִ֔ים שָׁ֛ם הֵ֥ם שָׁבִ֖ים לָלָֽכֶת׃ כָּל הַנְּחָלִים זוֹרְמִים אֶל הַיָּם אֲבָל הַיָּם אֵינֶנּוּ מָלֵא, לַמָּקוֹם שֶׁהַנְּחָלִים זוֹרְמִים – לְשָׁם הֵם שָׁבִים וּמַמְשִׁיכִים לִזְרוֹם.
ח כׇּל־הַדְּבָרִ֣ים יְגֵעִ֔ים לֹא־יוּכַ֥ל אִ֖ישׁ לְדַבֵּ֑ר לֹא־תִשְׂבַּ֥ע עַ֙יִן֙ לִרְא֔וֹת וְלֹא־תִמָּלֵ֥א אֹ֖זֶן מִשְּׁמֹֽעַ׃ כָּל הַדְּבָרִים בָּעוֹלָם מְעַיְּפִים, אִישׁ לֹא יוּכַל לְדַבֵּר עַל כֻּלָּם, הָעַיִן לֹא שְׂבֵעָה מִלִּרְאוֹת וְהָאוֹזֶן לֹא מְלֵאָה מִלִּשְׁמוֹעַ.
ט מַה־שֶּֽׁהָיָה֙ ה֣וּא שֶׁיִּהְיֶ֔ה וּמַה־שֶּׁנַּֽעֲשָׂ֔ה ה֖וּא שֶׁיֵּעָשֶׂ֑ה וְאֵ֥ין כׇּל־חָדָ֖שׁ תַּ֥חַת הַשָּֽׁמֶשׁ׃ מַה שֶּׁהָיָה הוּא שֶׁיִּהְיֶה וּמַה שֶּׁנַּעֲשָׂה הוּא שֶׁיֵּעָשֶׂה, וְאֵין כָּל חָדָשׁ מִתַּחַת לַשֶּׁמֶשׁ.
י יֵ֥שׁ דָּבָ֛ר שֶׁיֹּאמַ֥ר רְאֵה־זֶ֖ה חָדָ֣שׁ ה֑וּא כְּבָר֙ הָיָ֣ה לְעֹֽלָמִ֔ים אֲשֶׁ֥ר הָיָ֖ה מִלְּפָנֵֽנוּ׃ וְאִם יֵשׁ דָּבָר שֶׁיֹּאמְרוּ הִנֵּה זֶה חָדָשׁ! הוּא כְּבָר הָיָה בִּתְקוּפָה שֶׁהָיְתָה לְפָנֵנוּ.
יא אֵ֥ין זִכְר֖וֹן לָרִאשֹׁנִ֑ים וְגַ֨ם לָאַחֲרֹנִ֜ים שֶׁיִּהְי֗וּ לֹֽא־יִהְיֶ֤ה לָהֶם֙ זִכָּר֔וֹן עִ֥ם שֶׁיִּהְי֖וּ לָאַחֲרֹנָֽה׃ אֶלָּא שֶׁלֹּא זוֹכְרִים אֶת הָאֵרוּעִים הָרִאשׁוֹנִים, וְגַם הָאֵרוּעִים שֶׁיִּהְיוּ בֶּעָתִיד לֹא יִהְיוּ זְכוּרִים לְאֵלּוּ שֶׁיִּהְיוּ אַחֲרֵיהֶם.
פ
יב אֲנִ֣י קֹהֶ֗לֶת הָיִ֥יתִי מֶ֛לֶךְ עַל־יִשְׂרָאֵ֖ל בִּירוּשָׁלָֽ͏ִם׃ אֲנִי קוֹהֶלֶת הָיִיתִי מֶלֶךְ עַל יִשְׂרָאֵל בִּירוּשָׁלַיִם.
יג וְנָתַ֣תִּי אֶת־לִבִּ֗י לִדְר֤וֹשׁ וְלָתוּר֙ בַּֽחׇכְמָ֔ה עַ֛ל כׇּל־אֲשֶׁ֥ר נַעֲשָׂ֖ה תַּ֣חַת הַשָּׁמָ֑יִם ה֣וּא ׀ עִנְיַ֣ן רָ֗ע נָתַ֧ן אֱלֹהִ֛ים לִבְנֵ֥י הָאָדָ֖ם לַעֲנ֥וֹת בּֽוֹ׃ לָקַחְתִּי לִתְשׂוּמֶת לִבִּי לַחֲקוֹר וּלְחַפֵּשׂ בְּחָכְמָה עַל כָּל מַה שֶּׁנַּעֲשָׂה תַּחַת הַשָּׁמַיִם, זֶה עִנְיַן קָשֶׁה שֶׁאֱלֹהִים נָתַן לִבְנֵי הָאָדָם לְהִתְעַנְיֵן בּוֹ.
יד רָאִ֙יתִי֙ אֶת־כׇּל־הַֽמַּעֲשִׂ֔ים שֶֽׁנַּעֲשׂ֖וּ תַּ֣חַת הַשָּׁ֑מֶשׁ וְהִנֵּ֥ה הַכֹּ֛ל הֶ֖בֶל וּרְע֥וּת רֽוּחַ׃ רָאִיתִי אֶת כָּל הַמַּעֲשִׂים שֶׁעוֹשִׂים תַּחַת הַשֶּׁמֶשׁ, וְהִנֵּה הַכֹּל הֶבֶל וְרַעְיוֹן שָׁוְא.
טו מְעֻוָּ֖ת לֹא־יוּכַ֣ל לִתְקֹ֑ן וְחֶסְר֖וֹן לֹא־יוּכַ֥ל לְהִמָּנֽוֹת׃ דָּבָר מְקוּלְקָל לֹא נִתָּן לְתִקּוּן, וְדָבָר חָסֵר לֹא נִתָּן לִסְפִירָה.
טז דִּבַּ֨רְתִּי אֲנִ֤י עִם־לִבִּי֙ לֵאמֹ֔ר אֲנִ֗י הִנֵּ֨ה הִגְדַּ֤לְתִּי וְהוֹסַ֙פְתִּי֙ חׇכְמָ֔ה עַ֛ל כׇּל־אֲשֶׁר־הָיָ֥ה לְפָנַ֖י עַל־יְרוּשָׁלָ֑͏ִם וְלִבִּ֛י רָאָ֥ה הַרְבֵּ֖ה חׇכְמָ֥ה וָדָֽעַת׃ דִּבַּרְתִּי אֶל עַצְמִי וְאָמַרְתִּי: הֲרֵי אֲנִי הֶחְכַּמְתִּי וְנִתּוֹסְפָה בִּי חָכְמָה יוֹתֵר מִכָּל מִי שֶּׁשָּׁלַט לְפָנַי עַל יְרוּשָׁלַיִם וְלָמַדְתִּי הַרְבֵּה חָכְמָה וִידִיעוֹת.
יז וָאֶתְּנָ֤ה לִבִּי֙ לָדַ֣עַת חׇכְמָ֔ה וְדַ֥עַת הוֹלֵלֹ֖ת וְשִׂכְל֑וּת יָדַ֕עְתִּי שֶׁגַּם־זֶ֥ה ה֖וּא רַעְי֥וֹן רֽוּחַ׃ וְלָקַחְתִּי לִתְשׂוּמֶת לִבִּי לָדַעַת חָכְמָה וְדַעַת שֶׁל בִּלּוּיִים וּשְׁטוּיוֹת, הֵבַנְתִּי שֶׁגַּם זֶה רַעְיוֹן שָׁוְא.
יח כִּ֛י בְּרֹ֥ב חׇכְמָ֖ה רׇב־כָּ֑עַס וְיוֹסִ֥יף דַּ֖עַת יוֹסִ֥יף מַכְאֽוֹב׃ כִּי כְּשֶׁמִּתְרַבָּה הַחָכְמָה מִתְרַבֶּה הַצַּעַר, וְהַמּוֹסִיף יְדִיעוֹת מוֹסִיף כְּאֵב.
ב א אָמַ֤רְתִּֽי אֲנִי֙ בְּלִבִּ֔י לְכָה־נָּ֛א אֲנַסְּכָ֥ה בְשִׂמְחָ֖ה וּרְאֵ֣ה בְט֑וֹב וְהִנֵּ֥ה גַם־ה֖וּא הָֽבֶל׃ אָמַרְתִּי לְעַצְמִי אֵלֵךְ וְאֶמְזוֹג לִי יַיִן בְּשִׂמְחָה וְיִהְיֶה לִי טוֹב! הִתְבָּרֵר שֶׁגַּם שִׂמְחָה זֶה הֶבֶל.
ב לִשְׂח֖וֹק אָמַ֣רְתִּי מְהוֹלָ֑ל וּלְשִׂמְחָ֖ה מַה־זֹּ֥ה עֹשָֽׂה׃ עַל הַצְּחוֹק אֲנִי אוֹמֵר שֶׁהוּא שְׁטוּת, וְעַל הַשִּׂמְחָה – מַה הַתּוֹעֶלֶת בָּהּ?
ג תַּ֣רְתִּי בְלִבִּ֔י לִמְשׁ֥וֹךְ בַּיַּ֖יִן אֶת־בְּשָׂרִ֑י וְלִבִּ֞י נֹהֵ֤ג בַּֽחׇכְמָה֙ וְלֶאֱחֹ֣ז בְּסִכְל֔וּת עַ֣ד אֲשֶׁר־אֶרְאֶ֗ה אֵי־זֶ֨ה ט֜וֹב לִבְנֵ֤י הָאָדָם֙ אֲשֶׁ֤ר יַעֲשׂוּ֙ תַּ֣חַת הַשָּׁמַ֔יִם מִסְפַּ֖ר יְמֵ֥י חַיֵּיהֶֽם׃ חִפַּשְׂתִּי לִבְדּוֹק – לְעַדֵּן בְּיַיִן אֶת גּוּפִי, לַהֲגוֹת בְּדִבְרֵי חָכְמָה וּלְהִתְעַסֵּק גַּם בִּדְבָרִים טִפְּשִׁיִּים, עַד שֶׁאֶרְאֶה מָה הַדָּבָר הַטּוֹב לִבְנֵי הָאָדָם שֶׁיַּעֲשׂוּ בָּעוֹלָם בִּימֵי חַיֵּיהֶם הַסְּפוּרִים.
ד הִגְדַּ֖לְתִּי מַעֲשָׂ֑י בָּנִ֤יתִי לִי֙ בָּתִּ֔ים נָטַ֥עְתִּי לִ֖י כְּרָמִֽים׃ בִּצַּעְתִּי תָּכְנִיּוֹת גְּדוֹלוֹת – בָּנִיתִי לִי בָּתִּים נָטַעְתִּי לִי כְּרָמִים.
ה עָשִׂ֣יתִי לִ֔י גַּנּ֖וֹת וּפַרְדֵּסִ֑ים וְנָטַ֥עְתִּי בָהֶ֖ם עֵ֥ץ כׇּל־פֶּֽרִי׃ עָשִׂיתִי לִי גִּנּוֹת וּפַרְדֵּסִים וְנָטַעְתִּי בָּהֶם כָּל עֲצֵי הַפֵּרוֹת.
ו עָשִׂ֥יתִי לִ֖י בְּרֵכ֣וֹת מָ֑יִם לְהַשְׁק֣וֹת מֵהֶ֔ם יַ֖עַר צוֹמֵ֥חַ עֵצִֽים׃ עָשִׂיתִי לִי בְּרֵכוֹת מַיִם לְהַשְׁקוֹת מֵהֶם יַעַר שֶׁצּוֹמְחִים בּוֹ עֵצִים.
ז קָנִ֙יתִי֙ עֲבָדִ֣ים וּשְׁפָח֔וֹת וּבְנֵי־בַ֖יִת הָ֣יָה לִ֑י גַּ֣ם מִקְנֶה֩ בָקָ֨ר וָצֹ֤אן הַרְבֵּה֙ הָ֣יָה לִ֔י מִכֹּ֛ל שֶֽׁהָי֥וּ לְפָנַ֖י בִּירוּשָׁלָֽ͏ִם׃ קָנִיתִי עֲבָדִים וּשְׁפָחוֹת וְהָיוּ לִי בְּנֵי בַּיִת, גַּם הָיָה לִי הַרְבֵּה עֶדְרֵי בָּקָר וְצֹאן יוֹתֵר מִכָּל אֵלּוּ שֶׁהָיוּ לְפָנַי בִּירוּשָׁלַיִם.
ח כָּנַ֤סְתִּי לִי֙ גַּם־כֶּ֣סֶף וְזָהָ֔ב וּסְגֻלַּ֥ת מְלָכִ֖ים וְהַמְּדִינ֑וֹת עָשִׂ֨יתִי לִ֜י שָׁרִ֣ים וְשָׁר֗וֹת וְתַעֲנֻג֛וֹת בְּנֵ֥י הָאָדָ֖ם שִׁדָּ֥ה וְשִׁדּֽוֹת׃ אָסַפְתִּי לִי גַּם כֶּסֶף וְזָהָב וְאוֹצָרוֹת שֶׁל מְלָכִים וּמְדִינוֹת, עָשִׂיתִי לִי מַקְהֵלוֹת זַמָּרִים וְזַמָּרוֹת וְתַעֲנוּגוֹת בְּנֵי הָאָדָם כֵּס מַלְכוּת וּמֶרְכָּבוֹת.
ט וְגָדַ֣לְתִּי וְהוֹסַ֔פְתִּי מִכֹּ֛ל שֶׁהָיָ֥ה לְפָנַ֖י בִּירוּשָׁלָ֑͏ִם אַ֥ף חׇכְמָתִ֖י עָ֥מְדָה לִּֽי׃ וְהִתְעַשַּׁרְתִּי וְאָסַפְתִּי מָמוֹן יוֹתֵר מִכָּל מַה שֶּׁהָיָה לְפָנַי בִּירוּשָׁלַיִם, וְגַם זָכַרְתִּי אֶת כָּל חָכְמָתִי שֶׁלָּמַדְתִּי.
י וְכֹל֙ אֲשֶׁ֣ר שָֽׁאֲל֣וּ עֵינַ֔י לֹ֥א אָצַ֖לְתִּי מֵהֶ֑ם לֹֽא־מָנַ֨עְתִּי אֶת־לִבִּ֜י מִכׇּל־שִׂמְחָ֗ה כִּֽי־לִבִּ֤י שָׂמֵ֙חַ֙ מִכׇּל־עֲמָלִ֔י וְזֶֽה־הָיָ֥ה חֶלְקִ֖י מִכׇּל־עֲמָלִֽי׃ וְכָל הַדְּבָרִים שֶׁרָאִיתִי וְרָצִיתִי לֹא נִמְנַעְתִּי מֵהֶם, לֹא מָנַעְתִּי מֵעַצְמִי שׁוּם הֲנָאָה אֶלָּא נֶהֱנֵיתִי מִכָּל עֲמָלִי וְזֶה מַה שֶּׁהִרְוַחְתִּי מִכָּל עֲמָלִי.
יא וּפָנִ֣יתִֽי אֲנִ֗י בְּכׇל־מַעֲשַׂי֙ שֶֽׁעָשׂ֣וּ יָדַ֔י וּבֶֽעָמָ֖ל שֶׁעָמַ֣לְתִּי לַעֲשׂ֑וֹת וְהִנֵּ֨ה הַכֹּ֥ל הֶ֙בֶל֙ וּרְע֣וּת ר֔וּחַ וְאֵ֥ין יִתְר֖וֹן תַּ֥חַת הַשָּֽׁמֶשׁ׃ וּפָנִיתִי לִבְחוֹן אֶת כָּל מַעֲשַׂי שֶׁעָשִׂיתִי וְאֶת הֶעָמָל שֶׁעָמַלְתִּי לַעֲשׂוֹת, וְהִנֵּה הַכֹּל הֶבֶל וְרַעְיוֹן שָׁוְא לְלֹא תּוֹעֶלֶת בָּעוֹלָם.
יב וּפָנִ֤יתִֽי אֲנִי֙ לִרְא֣וֹת חׇכְמָ֔ה וְהוֹלֵל֖וֹת וְסִכְל֑וּת כִּ֣י ׀ מֶ֣ה הָאָדָ֗ם שֶׁיָּבוֹא֙ אַחֲרֵ֣י הַמֶּ֔לֶךְ אֵ֥ת אֲשֶׁר־כְּבָ֖ר עָשֽׂוּהוּ׃ וְהִתְפַּנֵּיתִי לִבְחוֹן חָכְמָה וּבִלּוּיִים וּשְׁטוּיוֹת, כִּי אֵיזֶה עוֹד אָדָם יוּכַל לָבוֹא אַחֲרֵי הַמֶּלֶךְ לְנַסּוֹת כָּמוֹנִי אֶת מַה שֶּׁכְּבָר עָשִׂיתִי.
יג וְרָאִ֣יתִי אָ֔נִי שֶׁיֵּ֥שׁ יִתְר֛וֹן לַֽחׇכְמָ֖ה מִן־הַסִּכְל֑וּת כִּֽיתְר֥וֹן הָא֖וֹר מִן־הַחֹֽשֶׁךְ׃ וְרָאִיתִי שֶׁיֵּשׁ יִתְרוֹן לַחָכְמָה עַל הַסִּכְלוּת כְּמוֹ יִתְרוֹן הָאוֹר עַל הַחוֹשֶׁךְ.
יד הֶֽחָכָם֙ עֵינָ֣יו בְּרֹאשׁ֔וֹ וְהַכְּסִ֖יל בַּחֹ֣שֶׁךְ הוֹלֵ֑ךְ וְיָדַ֣עְתִּי גַם־אָ֔נִי שֶׁמִּקְרֶ֥ה אֶחָ֖ד יִקְרֶ֥ה אֶת־כֻּלָּֽם׃ הֶחָכָם נוֹהֵג בִּתְבוּנָה כְּמוֹ אָדָם שֶׁיֵּשׁ לוֹ עֵינַיִם בְּרֹאשׁוֹ, וְהַטִּפֵּשׁ כְּאִלּוּ הוֹלֵךְ בַּחוֹשֶׁךְ, אֲבָל גַּם הֵבַנְתִּי שֶׁאוֹתוֹ מִקְרֶה יִקְרֶה בַּסּוֹף לְכֻלָּם.
טו וְאָמַ֨רְתִּֽי אֲנִ֜י בְּלִבִּ֗י כְּמִקְרֵ֤ה הַכְּסִיל֙ גַּם־אֲנִ֣י יִקְרֵ֔נִי וְלָ֧מָּה חָכַ֛מְתִּי אֲנִ֖י אָ֣ז יֹתֵ֑ר וְדִבַּ֣רְתִּי בְלִבִּ֔י שֶׁגַּם־זֶ֖ה הָֽבֶל׃ וְאָמַרְתִּי לְעַצְמִי, כְּמוֹ שֶׁיִּקְרֶה בַּסּוֹף לַטִּפֵּשׁ יִקְרֶה גַּם לִי, אָז בִּשְׁבִיל מָה הֶחְכַּמְתִּי יוֹתֵר?! אָז אָמַרְתִּי לְעַצְמִי שֶׁגַּם חָכְמָה הִיא הֶבֶל.
טז כִּי֩ אֵ֨ין זִכְר֧וֹן לֶחָכָ֛ם עִֽם־הַכְּסִ֖יל לְעוֹלָ֑ם בְּשֶׁכְּבָ֞ר הַיָּמִ֤ים הַבָּאִים֙ הַכֹּ֣ל נִשְׁכָּ֔ח וְאֵ֛יךְ יָמ֥וּת הֶחָכָ֖ם עִֽם־הַכְּסִֽיל׃ כִּי לְחָכָם כְּמוֹ לְטִפֵּשׁ אֵין זִכָּרוֹן נִצְחִי כִּי בְּמֶשֶׁךְ הַיָּמִים הַבָּאִים הַכֹּל יִשָּׁכַח, אֵיךְ יָמוּת הֶחָכָם כְּמוֹ הַכְּסִיל!
יז וְשָׂנֵ֙אתִי֙ אֶת־הַ֣חַיִּ֔ים כִּ֣י רַ֤ע עָלַי֙ הַֽמַּעֲשֶׂ֔ה שֶׁנַּעֲשָׂ֖ה תַּ֣חַת הַשָּׁ֑מֶשׁ כִּֽי־הַכֹּ֥ל הֶ֖בֶל וּרְע֥וּת רֽוּחַ׃ וְשָׂנֵאתִי אֶת הַחַיִּים כִּי רַע בְּעֵינַי כָּל מַה שֶּׁקּוֹרֶה בָּעוֹלָם, כִּי הַכֹּל הֶבֶל וְרַעְיוֹן שָׁוְא.
יח וְשָׂנֵ֤אתִֽי אֲנִי֙ אֶת־כׇּל־עֲמָלִ֔י שֶׁאֲנִ֥י עָמֵ֖ל תַּ֣חַת הַשָּׁ֑מֶשׁ שֶׁ֣אַנִּיחֶ֔נּוּ לָאָדָ֖ם שֶׁיִּהְיֶ֥ה אַחֲרָֽי׃ וְשָׂנֵאתִי אֶת כָּל עֲמָלִי שֶׁאֲנִי עָמֵל בָּעוֹלָם – שֶׁאַשְׁאִיר אוֹתוֹ לָאָדָם שֶׁיִּהְיֶה אַחֲרַי.
יט וּמִ֣י יוֹדֵ֗עַ הֶֽחָכָ֤ם יִהְיֶה֙ א֣וֹ סָכָ֔ל וְיִשְׁלַט֙ בְּכׇל־עֲמָלִ֔י שֶֽׁעָמַ֥לְתִּי וְשֶׁחָכַ֖מְתִּי תַּ֣חַת הַשָּׁ֑מֶשׁ גַּם־זֶ֖ה הָֽבֶל׃ וּמִי יוֹדֵעַ אִם הוּא יִהְיֶה חָכָם אוֹ טִפֵּשׁ וְהוּא יִשְׁלוֹט בְּכָל עֲמָלִי שֶׁעָמַלְתִּי וְשֶׁהֶחְכַּמְתִּי בְּחַיַּי, גַּם זֶה הֶבֶל.
כ וְסַבּ֥וֹתִֽי אֲנִ֖י לְיַאֵ֣שׁ אֶת־לִבִּ֑י עַ֚ל כׇּל־הֶ֣עָמָ֔ל שֶׁעָמַ֖לְתִּי תַּ֥חַת הַשָּֽׁמֶשׁ׃ וְשִׁנִּיתִי גִּישָׁה וְהִתְיָאַשְׁתִּי מִלַּעֲסוֹק בְּכָל הֶעָמָל שֶׁעָמַלְתִּי בְּחַיַּי.
כא כִּי־יֵ֣שׁ אָדָ֗ם שֶׁעֲמָל֛וֹ בְּחׇכְמָ֥ה וּבְדַ֖עַת וּבְכִשְׁר֑וֹן וּלְאָדָ֞ם שֶׁלֹּ֤א עָֽמַל־בּוֹ֙ יִתְּנֶ֣נּוּ חֶלְק֔וֹ גַּם־זֶ֥ה הֶ֖בֶל וְרָעָ֥ה רַבָּֽה׃ כִּי קוֹרֶה שֶׁאָדָם עָמֵל בְּחָכְמָה וּבְדַעַת וּבְכִשָּׁרוֹן, וְהוּא נוֹתֵן אֶת חֶלְקוֹ לְאָדָם שֶׁלֹּא עָמַל בּוֹ, גַּם זֶה הֶבֶל וְרוֹעַ גָּדוֹל.
כב כִּ֠י מֶֽה־הֹוֶ֤ה לָֽאָדָם֙ בְּכׇל־עֲמָל֔וֹ וּבְרַעְי֖וֹן לִבּ֑וֹ שְׁה֥וּא עָמֵ֖ל תַּ֥חַת הַשָּֽׁמֶשׁ׃ כִּי מַה נִּשְׁאָר לָאָדָם מִכָּל עֲמָלוֹ וּמַחְשָׁבוֹתָיו שֶׁהוּא עָמֵל בָּהֶם בְּחַיָּיו.
כג כִּ֧י כׇל־יָמָ֣יו מַכְאֹבִ֗ים וָכַ֙עַס֙ עִנְיָנ֔וֹ גַּם־בַּלַּ֖יְלָה לֹא־שָׁכַ֣ב לִבּ֑וֹ גַּם־זֶ֖ה הֶ֥בֶל הֽוּא׃ הֲרֵי כָּל יָמָיו כְּאֵבִים וְעִנְיָנִים עֲצוּבִים, גַּם בַּלַּיְלָה לִבּוֹ לֹא נָח, גַּם זֶה הֶבֶל.
כד אֵֽין־ט֤וֹב בָּאָדָם֙ שֶׁיֹּאכַ֣ל וְשָׁתָ֔ה וְהֶרְאָ֧ה אֶת־נַפְשׁ֛וֹ ט֖וֹב בַּעֲמָל֑וֹ גַּם־זֹה֙ רָאִ֣יתִי אָ֔נִי כִּ֛י מִיַּ֥ד הָאֱלֹהִ֖ים הִֽיא׃ אֵין טוֹב לָאָדָם אֶלָּא שֶׁיֹּאכַל וְיִשְׁתֶּה וְיַרְגִּישׁ טוֹב בַּעֲמָלוֹ, וְרָאִיתִי שֶׁגַּם זֶה זְכוּת מִיַּד הָאֱלֹהִים הִיא.
כה כִּ֣י מִ֥י יֹאכַ֛ל וּמִ֥י יָח֖וּשׁ ח֥וּץ מִמֶּֽנִּי׃ כִּי לָמָּה שֶׁמִּישֶׁהוּ יְמַהֵר וְיֹאכַל מִמָּמוֹנִי חוּץ מִמֶּנִּי?
כו כִּ֤י לְאָדָם֙ שֶׁטּ֣וֹב לְפָנָ֔יו נָתַ֛ן חׇכְמָ֥ה וְדַ֖עַת וְשִׂמְחָ֑ה וְלַחוֹטֶא֩ נָתַ֨ן עִנְיָ֜ן לֶאֱסֹ֣ף וְלִכְנ֗וֹס לָתֵת֙ לְטוֹב֙ לִפְנֵ֣י הָֽאֱלֹהִ֔ים גַּם־זֶ֥ה הֶ֖בֶל וּרְע֥וּת רֽוּחַ׃ כִּי ה' נָתַן לְאָדָם שֶׁהוּא טוֹב בְּעֵינָיו חָכְמָה וְדַעַת וְשִׂמְחָה, וְלַחוֹטֶא הוּא נָתַן עִסּוּק לֶאֱסוֹף וּלְכַנֵּס מָמוֹן שֶׁהָאֱלֹהִים יִתֵּן לְמִי שֶׁיִּרְצֶה, גַּם זֶה הֶבֶל וְרַעְיוֹן שָׁוְא.
ג א לַכֹּ֖ל זְמָ֑ן וְעֵ֥ת לְכׇל־חֵ֖פֶץ תַּ֥חַת הַשָּׁמָֽיִם׃ לְכָל דָּבָר יֵשׁ זְמָן בָּעוֹלָם הַזֶּה.
ב עֵ֥ת לָלֶ֖דֶת  וְעֵ֣ת לָמ֑וּת  עֵ֣ת לָטַ֔עַת  וְעֵ֖ת לַעֲק֥וֹר נָטֽוּעַ׃ יֵשׁ זְמָן לָלֶדֶת וּזְמָן לָמוּת, זְמָן לִנְטוֹעַ וּזְמָן לַעֲקוֹר נְטִיעוֹת.
ג עֵ֤ת לַהֲרוֹג֙  וְעֵ֣ת לִרְפּ֔וֹא  עֵ֥ת לִפְר֖וֹץ  וְעֵ֥ת לִבְנֽוֹת׃ זְמָן לַהֲרוֹג וּזְמָן לְרַפֵּא, זְמָן לִפְרוֹץ חוֹמוֹת וּזְמָן לִבְנוֹת.
ד עֵ֤ת לִבְכּוֹת֙  וְעֵ֣ת לִשְׂח֔וֹק  עֵ֥ת סְפ֖וֹד  וְעֵ֥ת רְקֽוֹד׃ זְמָן לִבְכּוֹת וּזְמָן לִצְחוֹק, זְמָן לְהַסְפִּיד וּזְמָן לִרְקוֹד.
ה עֵ֚ת לְהַשְׁלִ֣יךְ אֲבָנִ֔ים  וְעֵ֖ת כְּנ֣וֹס אֲבָנִ֑ים  עֵ֣ת לַחֲב֔וֹק  וְעֵ֖ת לִרְחֹ֥ק מֵחַבֵּֽק׃ זְמָן לִזְרוֹק אֲבָנִים וּזְמָן לֶאֱסוֹף אֲבָנִים, זְמָן לְחַבֵּק וּזְמָן לְהִתְרַחֵק מֵחִבּוּקִים.
ו עֵ֤ת לְבַקֵּשׁ֙  וְעֵ֣ת לְאַבֵּ֔ד  עֵ֥ת לִשְׁמ֖וֹר  וְעֵ֥ת לְהַשְׁלִֽיךְ׃ זְמָן לְחַפֵּשׂ וּזְמָן לְאַבֵּד, זְמָן לִשְׁמוֹר וּזְמָן לִזְרוֹק.
ז עֵ֤ת לִקְר֙וֹעַ֙  וְעֵ֣ת לִתְפּ֔וֹר  עֵ֥ת לַחֲשׁ֖וֹת  וְעֵ֥ת לְדַבֵּֽר׃ זְמָן לִקְרוֹעַ וּזְמָן לִתְפּוֹר, זְמָן לִשְׁתּוֹק וּזְמָן לְדַבֵּר.
ח עֵ֤ת לֶֽאֱהֹב֙  וְעֵ֣ת לִשְׂנֹ֔א  עֵ֥ת מִלְחָמָ֖ה  וְעֵ֥ת שָׁלֽוֹם׃ יֵשׁ זְמָן לֶאֱהוֹב וּזְמָן לִשְׂנוֹא, זְמָן מִלְחָמָה וּזְמָן לְשָׁלוֹם.
ט מַה־יִּתְרוֹן֙ הָֽעוֹשֶׂ֔ה בַּאֲשֶׁ֖ר ה֥וּא עָמֵֽל׃ מַה הַתּוֹעֶלֶת לְמִי שֶׁעוֹשֶׂה בְּמַה שֶּׁהוּא עָמֵל?
י רָאִ֣יתִי אֶת־הָֽעִנְיָ֗ן אֲשֶׁ֨ר נָתַ֧ן אֱלֹהִ֛ים לִבְנֵ֥י הָאָדָ֖ם לַעֲנ֥וֹת בּֽוֹ׃ רָאִיתִי אֶת הַדְּבָרִים שֶׁאֱלֹהִים נָתַן לִבְנֵי הָאָדָם לַעֲסוֹק בָּהֶם.
יא אֶת־הַכֹּ֥ל עָשָׂ֖ה יָפֶ֣ה בְעִתּ֑וֹ גַּ֤ם אֶת־הָעֹלָם֙ נָתַ֣ן בְּלִבָּ֔ם מִבְּלִ֞י אֲשֶׁ֧ר לֹא־יִמְצָ֣א הָאָדָ֗ם אֶת־הַֽמַּעֲשֶׂ֛ה אֲשֶׁר־עָשָׂ֥ה הָאֱלֹהִ֖ים מֵרֹ֥אשׁ וְעַד־סֽוֹף׃ ה' עָשָׂה שֶׁכָּל דָּבָר יַצְלִיחַ בִּזְמַנּוֹ הַמַּתְאִים, וְגַם נָתַן אֶת הָעוֹלָם לְהַנְהָגַת הָאָדָם מִבְּלִי שֶׁיָּבִין אֶת הַמַּעֲשִׂים שֶׁעָשָׂה הָאֱלֹהִים מִתְּחִלָּתָם עַד סוֹפָם.
יב יָדַ֕עְתִּי כִּ֛י אֵ֥ין ט֖וֹב בָּ֑ם כִּ֣י אִם־לִשְׂמ֔וֹחַ וְלַעֲשׂ֥וֹת ט֖וֹב בְּחַיָּֽיו׃ יָדַעְתִּי שֶׁאֵין טוֹב בַּמַּעֲשִׂים אֶלָּא לִשְׂמוֹחַ וְלַעֲשׂוֹת טוֹב בְּחַיָּיו.
יג וְגַ֤ם כׇּל־הָאָדָם֙ שֶׁיֹּאכַ֣ל וְשָׁתָ֔ה וְרָאָ֥ה ט֖וֹב בְּכׇל־עֲמָל֑וֹ מַתַּ֥ת אֱלֹהִ֖ים הִֽיא׃ וְגַם כָּל אָדָם שֶׁיֹּאכַל וְיִשְׁתֶּה וְיִרְאֶה טוֹב בְּכָל עֲמָלוֹ, זֶה מַתָּנָה מֵאֱלֹהִים.
יד יָדַ֗עְתִּי כִּ֠י כׇּל־אֲשֶׁ֨ר יַעֲשֶׂ֤ה הָאֱלֹהִים֙ ה֚וּא יִהְיֶ֣ה לְעוֹלָ֔ם עָלָיו֙ אֵ֣ין לְהוֹסִ֔יף וּמִמֶּ֖נּוּ אֵ֣ין לִגְרֹ֑עַ וְהָאֱלֹהִ֣ים עָשָׂ֔ה שֶׁיִּֽרְא֖וּ מִלְּפָנָֽיו׃ אֲנִי יוֹדֵעַ שֶׁכָּל מַה שֶּׁהָאֱלֹהִים בָּרָא הוּא יִהְיֶה לְעוֹלָם, אִי אֶפְשָׁר לְהוֹסִיף עָלָיו וְאִי אֶפְשָׁר לִגְרוֹעַ מִמֶּנּוּ, וְהָאֱלֹהִים עָשָׂה שֶׁיְּכַבְּדוּ אוֹתוֹ וְיִשְׁמְעוּ לוֹ.
טו מַה־שֶּֽׁהָיָה֙ כְּבָ֣ר ה֔וּא וַאֲשֶׁ֥ר לִהְי֖וֹת כְּבָ֣ר הָיָ֑ה וְהָאֱלֹהִ֖ים יְבַקֵּ֥שׁ אֶת־נִרְדָּֽף׃ מַה שֶּׁהָיָה כְּבָר קַיָּם, וּמַה שֶּׁיִּהְיֶה יִהְיֶה כְּמוֹ מַה שֶּׁכְּבָר הָיָה, וְהָאֱלֹהִים תּוֹמֵךְ בַּנִּרְדָּף וְנוֹקֵם בָּרוֹדֵף.
טז וְע֥וֹד רָאִ֖יתִי תַּ֣חַת הַשָּׁ֑מֶשׁ מְק֤וֹם הַמִּשְׁפָּט֙ שָׁ֣מָּה הָרֶ֔שַׁע וּמְק֥וֹם הַצֶּ֖דֶק שָׁ֥מָּה הָרָֽשַׁע׃ וְעוֹד רָאִיתִי תַּחַת הַשֶּׁמֶשׁ – בַּמָּקוֹם שֶׁל הַמִּשְׁפָּט שָׁם תִּתָּכֵן רִשְׁעוּת גְּדוֹלָה, וּבַמָּקוֹם שֶׁל הַצֶּדֶק דַּוְקָא שָׁם יֵשׁ רִשְׁעוּת.
יז אָמַ֤רְתִּֽי אֲנִי֙ בְּלִבִּ֔י אֶת־הַצַּדִּיק֙ וְאֶת־הָ֣רָשָׁ֔ע יִשְׁפֹּ֖ט הָאֱלֹהִ֑ים כִּי־עֵ֣ת לְכׇל־חֵ֔פֶץ וְעַ֥ל כׇּל־הַֽמַּעֲשֶׂ֖ה שָֽׁם׃ הֵבַנְתִּי בְּמַחֲשַׁבְתִּי: הָאֱלֹהִים יִשְׁפּוֹט גַּם אֶת הַצַּדִּיק וְגַם אֶת הָרָשָׁע, כִּי כָּל דָּבָר יָבוֹא זְמַנּוֹ, וְשָׁם יִהְיֶה דִּין עַל כָּל הַמַּעֲשִׂים.
יח אָמַ֤רְתִּֽי אֲנִי֙ בְּלִבִּ֔י עַל־דִּבְרַת֙ בְּנֵ֣י הָאָדָ֔ם לְבָרָ֖ם הָאֱלֹהִ֑ים וְלִרְא֕וֹת שְׁהֶם־בְּהֵמָ֥ה הֵ֖מָּה לָהֶֽם׃ הֵבַנְתִּי בְּמַחֲשַׁבְתִּי: עַל זֶה שֶׁבְּנֵי הָאָדָם מַנְהִיגִים אֶת עַצְמָם בְּהִתְנַשְּׂאוּת – הָאֱלֹהִים יַשְׁלִיךְ אוֹתָם, לְהַרְאוֹת לָהֶם שֶׁהֵם כְּמוֹ בְּהֵמָה.
יט כִּי֩ מִקְרֶ֨ה בְֽנֵי־הָאָדָ֜ם וּמִקְרֶ֣ה הַבְּהֵמָ֗ה וּמִקְרֶ֤ה אֶחָד֙ לָהֶ֔ם כְּמ֥וֹת זֶה֙ כֵּ֣ן מ֣וֹת זֶ֔ה וְר֥וּחַ אֶחָ֖ד לַכֹּ֑ל וּמוֹתַ֨ר הָאָדָ֤ם מִן־הַבְּהֵמָה֙ אָ֔יִן כִּ֥י הַכֹּ֖ל הָֽבֶל׃ כִּי מַה שֶׁקּוֹרֶה לִבְנֵי הָאָדָם וּמַה שֶׁקּוֹרֶה לַבְּהֵמָה מִקְרֶה אֶחָד קוֹרֶה לָהֶם, כְּמוֹ שֶׁאֵלֶּה מֵתִים כָּךְ גַּם אֵלֶּה, רוּחַ חַיִּים אַחַת יֵשׁ לְכֻלָּם, וְאֵין יִתְרוֹן לָאָדָם עַל הַבְּהֵמָה כִּי הַכֹּל הֶבֶל.
כ הַכֹּ֥ל הוֹלֵ֖ךְ אֶל־מָק֣וֹם אֶחָ֑ד הַכֹּל֙ הָיָ֣ה מִן־הֶֽעָפָ֔ר וְהַכֹּ֖ל שָׁ֥ב אֶל־הֶעָפָֽר׃ הַכֹּל הוֹלֵךְ אֶל מָקוֹם אֶחָד – הַכֹּל נִבְרָא מֵעָפָר וְהַכֹּל חוֹזֵר אֶל הֶעָפָר.
כא מִ֣י יוֹדֵ֗עַ ר֚וּחַ בְּנֵ֣י הָאָדָ֔ם הָעֹלָ֥ה הִ֖יא לְמָ֑עְלָה וְר֙וּחַ֙ הַבְּהֵמָ֔ה הַיֹּרֶ֥דֶת הִ֖יא לְמַ֥טָּה לָאָֽרֶץ׃ שֶׁהֲרֵי מִי רָאָה וְיוֹדֵעַ שֶׁרוּחַ בְּנֵי הָאָדָם הִיא הָעוֹלָה לְמַעְלָה וְרוּחַ הַבְּהֵמָה הִיא הַיּוֹרֶדֶת לְמַטָּה לָאָרֶץ.
כב וְרָאִ֗יתִי כִּ֣י אֵ֥ין טוֹב֙ מֵאֲשֶׁ֨ר יִשְׂמַ֤ח הָאָדָם֙ בְּֽמַעֲשָׂ֔יו כִּי־ה֖וּא חֶלְק֑וֹ כִּ֣י מִ֤י יְבִיאֶ֙נּוּ֙ לִרְא֔וֹת בְּמֶ֖ה שֶׁיִּהְיֶ֥ה אַחֲרָֽיו׃ וְהֵבַנְתִּי שֶׁאֵין טוֹב מִכָּךְ שֶׁהָאָדָם יִשְׂמַח מִמַּעֲשָׂיו כִּי זֶה הַחֵלֶק שֶׁלּוֹ בָּעוֹלָם, כִּי מִי יָבִיא אוֹתוֹ לִרְאוֹת מַה יִּהְיֶה עִם יְרֻשָּׁתוֹ אַחֲרֵי מוֹתוֹ.
ד א וְשַׁ֣בְתִּֽי אֲנִ֗י וָאֶרְאֶה֙ אֶת־כׇּל־הָ֣עֲשֻׁקִ֔ים אֲשֶׁ֥ר נַעֲשִׂ֖ים תַּ֣חַת הַשָּׁ֑מֶשׁ וְהִנֵּ֣ה ׀ דִּמְעַ֣ת הָעֲשֻׁקִ֗ים וְאֵ֤ין לָהֶם֙ מְנַחֵ֔ם וּמִיַּ֤ד עֹֽשְׁקֵיהֶם֙ כֹּ֔חַ וְאֵ֥ין לָהֶ֖ם מְנַחֵֽם׃ וְעוֹד חָזַרְתִּי לְהִתְבּוֹנֵן וְרָאִיתִי אֶת כָּל הַגְּזֵלוֹת שֶׁנַּעֲשׂוֹת בָּעוֹלָם, וְהִנֵּה דִּמְעַת הַנִּגְזָלִים זוֹלֶגֶת וְאֵין לָהֶם מְנַחֵם, וְאֵין מִישֶׁהוּ עִם כֹּחַ שֶׁיַּצִּיל אוֹתָם מִגּוֹזְלֵיהֶם וְאֵין לָהֶם מְנַחֵם.
ב וְשַׁבֵּ֧חַ אֲנִ֛י אֶת־הַמֵּתִ֖ים שֶׁכְּבָ֣ר מֵ֑תוּ מִן־הַ֣חַיִּ֔ים אֲשֶׁ֛ר הֵ֥מָּה חַיִּ֖ים עֲדֶֽנָה׃ וַאֲנִי אוֹמֵר שֶׁהַמֵּתִים – אֵלּוּ שֶׁכְּבָר מֵתוּ, עֲדִיפִים מֵהַחַיִּים – אֵלּוּ שֶׁהֵם עֲדַיִן חַיִּים.
ג וְטוֹב֙ מִשְּׁנֵיהֶ֔ם אֵ֥ת אֲשֶׁר־עֲדֶ֖ן לֹ֣א הָיָ֑ה אֲשֶׁ֤ר לֹֽא־רָאָה֙ אֶת־הַמַּעֲשֶׂ֣ה הָרָ֔ע אֲשֶׁ֥ר נַעֲשָׂ֖ה תַּ֥חַת הַשָּֽׁמֶשׁ׃ וְעָדִיף מִשְּׁנֵיהֶם מִי שֶׁעֲדַיִן לֹא הָיָה בָּעוֹלָם – שֶׁלֹּא רָאָה אֶת הַמַּעֲשִׂים הָרָעִים שֶׁנַּעֲשִׂים בָּעוֹלָם.
ד וְרָאִ֨יתִֽי אֲנִ֜י אֶת־כׇּל־עָמָ֗ל וְאֵת֙ כׇּל־כִּשְׁר֣וֹן הַֽמַּעֲשֶׂ֔ה כִּ֛י הִ֥יא קִנְאַת־אִ֖ישׁ מֵרֵעֵ֑הוּ גַּם־זֶ֥ה הֶ֖בֶל וּרְע֥וּת רֽוּחַ׃ וְרָאִיתִי אֶת כָּל הַמַּאֲמָץ וְהַהַצְלָחָה בַּמַּעֲשִׂים שֶׁהִיא נוֹבַעַת מִקִּנְאַת אִישׁ בַּחֲבֵרוֹ, גַּם זֶה הֶבֶל וְרַעְיוֹן שָׁוְא.
ה הַכְּסִיל֙ חֹבֵ֣ק אֶת־יָדָ֔יו וְאֹכֵ֖ל אֶת־בְּשָׂרֽוֹ׃ הַטִּפֵּשׁ מְחַבֵּק אֶת יָדָיו בְּעַצְלוּת, וְגוּפוֹ מִתְנַוֵּן.
ו ט֕וֹב מְלֹ֥א כַ֖ף נָ֑חַת מִמְּלֹ֥א חׇפְנַ֛יִם עָמָ֖ל וּרְע֥וּת רֽוּחַ׃ עָדִיף לְקַבֵּל בְּנַחַת כַּף מְלֵאָה אַחַת, מִלְּקַבֵּל בְּמַאֲמָץ וְרַעְיוֹנוֹת שָׁוְא חָפְנַיִם מְלֵאוֹת.
ז וְשַׁ֧בְתִּי אֲנִ֛י וָאֶרְאֶ֥ה הֶ֖בֶל תַּ֥חַת הַשָּֽׁמֶשׁ׃ וְחָזַרְתִּי לְהִתְבּוֹנֵן וְרָאִיתִי עוֹד הֶבֶל בָּעוֹלָם –
ח יֵ֣שׁ אֶחָד֩ וְאֵ֨ין שֵׁנִ֜י גַּ֣ם בֵּ֧ן וָאָ֣ח אֵֽין־ל֗וֹ וְאֵ֥ין קֵץ֙ לְכׇל־עֲמָל֔וֹ גַּם־עֵינ֖וֹ עיניו לֹא־תִשְׂבַּ֣ע עֹ֑שֶׁר וּלְמִ֣י ׀ אֲנִ֣י עָמֵ֗ל וּמְחַסֵּ֤ר אֶת־נַפְשִׁי֙ מִטּוֹבָ֔ה גַּם־זֶ֥ה הֶ֛בֶל וְעִנְיַ֥ן רָ֖ע הֽוּא׃ יֵשׁ אָדָם יָחִיד וְאֵין אִתּוֹ אָדָם שֵׁנִי לַחֲלוֹק אִתּוֹ, וְגַם אֵין לוֹ בֵּן אוֹ אָח שֶׁיִּירַשׁ אוֹתוֹ. אֲבָל אֵין סוֹף לְכָל מַאֲמָצָיו, וְגַם עֵינוֹ לֹא תִּשְׂבַּע מֵעוֹשֶׁר. אֲבָל עֲבוּר מִי אֲנִי עָמֵל וּמַחְסִיר טוֹבָה מֵעַצְמִי?! גַּם זֶה הֶבֶל וְעִנְיַן רָע.
ט טוֹבִ֥ים הַשְּׁנַ֖יִם מִן־הָאֶחָ֑ד אֲשֶׁ֧ר יֵשׁ־לָהֶ֛ם שָׂכָ֥ר ט֖וֹב בַּעֲמָלָֽם׃ שְׁנַיִם עֲדִיפִים מֵאֶחָד כִּי יֵשׁ לָהֶם רֶוַח טוֹב בַּעֲמָלָם –
י כִּ֣י אִם־יִפֹּ֔לוּ הָאֶחָ֖ד יָקִ֣ים אֶת־חֲבֵר֑וֹ וְאִ֣יל֗וֹ הָֽאֶחָד֙ שֶׁיִּפּ֔וֹל וְאֵ֥ין שֵׁנִ֖י לַהֲקִימֽוֹ׃ כִּי אִם הֵם יִפְּלוּ – אֶחָד יָקִים אֶת חֲבֵרוֹ, וְאִלּוּ יָחִיד שֶׁיִּפּוֹל – אֵין לוֹ שֵׁנִי לַהֲקִימוֹ.
יא גַּ֛ם אִם־יִשְׁכְּב֥וּ שְׁנַ֖יִם וְחַ֣ם לָהֶ֑ם וּלְאֶחָ֖ד אֵ֥יךְ יֵחָֽם׃ גַּם אִם יִשְׁכְּבוּ שְׁנַיִם יִהְיֶה לָהֶם חָם, וּלְאֶחָד אֵיךְ יִהְיֶה חָם?
יב וְאִֽם־יִתְקְפוֹ֙ הָאֶחָ֔ד הַשְּׁנַ֖יִם יַעַמְד֣וּ נֶגְדּ֑וֹ וְהַחוּט֙ הַֽמְשֻׁלָּ֔שׁ לֹ֥א בִמְהֵרָ֖ה יִנָּתֵֽק׃ וְאִם יַתְקִיף אוֹתוֹ אִישׁ אֶחָד אוֹתָם שְׁנַיִם יַעַמְדוּ מוּלוֹ. וְחֶבֶל שֶׁעָשׂוּי מִשְּׁלוֹשָׁה חוּטִים אֵינוֹ מִתְנַתֵּק בִּמְהֵרָה.
יג ט֛וֹב יֶ֥לֶד מִסְכֵּ֖ן וְחָכָ֑ם מִמֶּ֤לֶךְ זָקֵן֙ וּכְסִ֔יל אֲשֶׁ֛ר לֹא־יָדַ֥ע לְהִזָּהֵ֖ר עֽוֹד׃ עָדִיף צָעִיר חֲסַר כֹּל אַךְ חָכָם, מִמֶּלֶךְ בּוֹגֵר אַךְ טִפֵּשׁ שֶׁכְּבָר אֵינוֹ יוֹדֵעַ לְהִזָּהֵר.
יד כִּֽי־מִבֵּ֥ית הָסוּרִ֖ים יָצָ֣א לִמְלֹ֑ךְ כִּ֛י גַּ֥ם בְּמַלְכוּת֖וֹ נוֹלַ֥ד רָֽשׁ׃ כִּי קָרָה שֶׁיָּצְאוּ לִמְלוֹךְ אֲפִלּוּ מִבֵּית כֶּלֶא, אַךְ הַטִּפֵּשׁ גַּם בְּמַלְכוּתוֹ נוֹלַד לִהְיוֹת חֲסַר כֹּל.
טו רָאִ֙יתִי֙ אֶת־כׇּל־הַ֣חַיִּ֔ים הַֽמְהַלְּכִ֖ים תַּ֣חַת הַשָּׁ֑מֶשׁ עִ֚ם הַיֶּ֣לֶד הַשֵּׁנִ֔י אֲשֶׁ֥ר יַעֲמֹ֖ד תַּחְתָּֽיו׃ רָאִיתִי אֶת כָּל הַחַיִּים שֶׁמִּתְנַהֲלִים בָּעוֹלָם – אֶת מַה שֶּׁיִּקְרֶה עִם הַמֶּלֶךְ הַשֵּׁנִי – הַצָּעִיר, שֶׁיִּמְלוֹךְ בִּמְקוֹם הָרִאשׁוֹן.
טז אֵֽין־קֵ֣ץ לְכׇל־הָעָ֗ם לְכֹ֤ל אֲשֶׁר־הָיָה֙ לִפְנֵיהֶ֔ם גַּ֥ם הָאַחֲרוֹנִ֖ים לֹ֣א יִשְׂמְחוּ־ב֑וֹ כִּֽי־גַם־זֶ֥ה הֶ֖בֶל וְרַעְי֥וֹן רֽוּחַ׃ אֵין סוֹף לַצַּעַר שֶׁל כָּל הָעָם שֶׁהָיָה לִפְנֵי חִלּוּפֵיהֶם, אַךְ גַּם הָאַחֲרוֹנִים לֹא יִשְׂמְחוּ בַּמֶּלֶךְ הַשֵּׁנִי חֲסַר הַנִּסָּיוֹן, כִּי גַּם זֶה הֶבֶל וְרַעְיוֹן שָׁוְא.
יז שְׁמֹ֣ר רגליך רַגְלְךָ֗ כַּאֲשֶׁ֤ר תֵּלֵךְ֙ אֶל־בֵּ֣ית הָאֱלֹהִ֔ים וְקָר֣וֹב לִשְׁמֹ֔עַ מִתֵּ֥ת הַכְּסִילִ֖ים זָ֑בַח כִּֽי־אֵינָ֥ם יוֹדְעִ֖ים לַעֲשׂ֥וֹת רָֽע׃ הִזָּהֵר בִּדְרְָכֶיךָ וּבְמַעֲשֶׂיךָ כְּשֶׁתֵּלֵךְ אֶל בֵּית הָאֱלֹהִים, זֶה יְקָרֵב אוֹתְךָ לַה' שֶׁיִּשְׁמַע אוֹתְךָ, יוֹתֵר מִמַּתְּנַת קָרְבָּן שֶׁל טִפְּשִׁים שֶׁאֵינָם יוֹדְעִים לְתַקֵּן אֶת דַּרְכָּם הָרָעָה.
ה א אַל־תְּבַהֵ֨ל עַל־פִּ֜יךָ וְלִבְּךָ֧ אַל־יְמַהֵ֛ר לְהוֹצִ֥יא דָבָ֖ר לִפְנֵ֣י הָאֱלֹהִ֑ים כִּ֣י הָאֱלֹהִ֤ים בַּשָּׁמַ֙יִם֙ וְאַתָּ֣ה עַל־הָאָ֔רֶץ עַל־כֵּ֛ן יִהְי֥וּ דְבָרֶ֖יךָ מְעַטִּֽים׃ אַל תִּהְיֶה פָּזִיז בְּפִיךָ וְאַל תְּמַהֵר בְּמַחְשַׁבְתְּךָ – שֶׁיֵּצֵא דִּבּוּר נֶגֶד הָאֱלֹהִים, כִּי הָאֱלֹהִים שׁוֹלֵט מֵהַשָּׁמַיִם וְאַתָּה רַק עַל הָאָרֶץ, לָכֵן טוֹב שֶׁיִּהְיוּ דִּבּוּרֶיךָ מְעַטִּים.
ב כִּ֛י בָּ֥א הַחֲל֖וֹם בְּרֹ֣ב עִנְיָ֑ן וְק֥וֹל כְּסִ֖יל בְּרֹ֥ב דְּבָרִֽים׃ כְּמוֹ שֶׁחֲלוֹם מַגִּיעַ מָלֵא בְּעִנְיָנִים, כָּךְ קוֹל הַטִּפֵּשׁ מָלֵא בְּפִטְפּוּטִים רֵקִים.
ג כַּאֲשֶׁר֩ תִּדֹּ֨ר נֶ֜דֶר לֵֽאלֹהִ֗ים אַל־תְּאַחֵר֙ לְשַׁלְּמ֔וֹ כִּ֛י אֵ֥ין חֵ֖פֶץ בַּכְּסִילִ֑ים אֵ֥ת אֲשֶׁר־תִּדֹּ֖ר שַׁלֵּֽם׃ כַּאֲשֶׁר תִּנְדֹּור נֶדֶר לֵאלֹהִים אַל תְּאַחֵר לְשַׁלְּמוֹ כִּי ה' אֵינֶנּוּ חָפֵץ בַּדִּבּוּרִים הָרֵקִים, אֶת מַה שֶּׁתִּנְדּוֹר – שַׁלֵּם.
ד ט֖וֹב אֲשֶׁ֣ר לֹֽא־תִדֹּ֑ר מִשֶּׁתִּדּ֖וֹר וְלֹ֥א תְשַׁלֵּֽם׃ עָדִיף שֶׁלֹּא תִּנְדּוֹר כְּלָל מִשֶּׁתִּנְדּוֹר וְלֹא תְּשַׁלֵּם.
ה אַל־תִּתֵּ֤ן אֶת־פִּ֙יךָ֙ לַחֲטִ֣יא אֶת־בְּשָׂרֶ֔ךָ וְאַל־תֹּאמַר֙ לִפְנֵ֣י הַמַּלְאָ֔ךְ כִּ֥י שְׁגָגָ֖ה הִ֑יא לָ֣מָּה יִקְצֹ֤ף הָֽאֱלֹהִים֙ עַל־קוֹלֶ֔ךָ וְחִבֵּ֖ל אֶת־מַעֲשֵׂ֥ה יָדֶֽיךָ׃ אַל תְּאַפְשֵׁר לְפִיךָ לְהַכְשִׁיל אוֹתְךָ וְלֹא תִּצְטָרֵךְ לְהִתְנַצֵּל בִּפְנֵי הַמַּלְאָךְ הַמַּעֲנִישׁ שֶׁעָבַרְתָּ עַל הַנֶּדֶר בְּטָעוּת, לָמָּה שֶׁהָאֱלֹהִים יִכְעַס עַל דִּבּוּרֶיךָ וְיַשְׁחִית אֶת מַעֲשֵׂה יָדֶיךָ.
ו כִּ֣י בְרֹ֤ב חֲלֹמוֹת֙ וַהֲבָלִ֔ים וּדְבָרִ֖ים הַרְבֵּ֑ה כִּ֥י אֶת־הָאֱלֹהִ֖ים יְרָֽא׃ כִּי זֶה בָּא מֵרוֹב חֲלוֹמוֹת וַהֲבָלִים וְהַרְבֵּה דִּבּוּרִים, אֶלָּא אֶת הָאֱלֹהִים יְרָא.
ז אִם־עֹ֣שֶׁק רָ֠שׁ וְגֵ֨זֶל מִשְׁפָּ֤ט וָצֶ֙דֶק֙ תִּרְאֶ֣ה בַמְּדִינָ֔ה אַל־תִּתְמַ֖הּ עַל־הַחֵ֑פֶץ כִּ֣י גָבֹ֜הַּ מֵעַ֤ל גָּבֹ֙הַּ֙ שֹׁמֵ֔ר וּגְבֹהִ֖ים עֲלֵיהֶֽם׃ אִם תִּרְאֶה בַּמְּדִינָה שֶׁעוֹשְׁקִים עָנִי וּמְעַוְּתִים אֶת הַמִּשְׁפָּט וְהַצֶּדֶק, אַל תִּתְמַהּ עַל רְצוֹן ה', כִּי ה' מַשְׁגִּיחַ מִגָּבוֹהַּ מֵעַל הַכֹּל, וְיֵשׁ חֲזָקִים מֵהָעוֹשְׁקִים שֶׁיִּנְקְמוּ בָּהֶם.
ח וְיִתְר֥וֹן אֶ֖רֶץ בַּכֹּ֣ל היא ה֑וּא מֶ֥לֶךְ לְשָׂדֶ֖ה נֶעֱבָֽד׃ וּמַעֲלַת הָאֲדָמָה לְהַעֲנִיק מָזוֹן שָׁוָה בְּכֻלָּם – גַּם מֶלֶךְ הוּא עֶבֶד לְשָׂדֶה.
ט אֹהֵ֥ב כֶּ֙סֶף֙ לֹא־יִשְׂבַּ֣ע כֶּ֔סֶף וּמִֽי־אֹהֵ֥ב בֶּהָמ֖וֹן לֹ֣א תְבוּאָ֑ה גַּם־זֶ֖ה הָֽבֶל׃ מִי שֶׁאוֹהֵב כֶּסֶף לֹא יִשְׂבַּע מִכֶּסֶף, וּמִי שֶׁאוֹהֵב לְקַבֵּל הָמוֹן לֹא יִתְמַלֵּא מְבֻקָּשׁוֹ, גַּם זֶה הֶבֶל.
י בִּרְבוֹת֙ הַטּוֹבָ֔ה רַבּ֖וּ אוֹכְלֶ֑יהָ וּמַה־כִּשְׁרוֹן֙ לִבְעָלֶ֔יהָ כִּ֖י אִם־רְא֥וּת ראית עֵינָֽיו׃ כְּשֶׁיֵּשׁ טוֹבָה רַבָּה רַבִּים הַנֶּהֱנִים מִמֶּנָּה, וּמַה הַהֲנָאָה לִבְעָלֶיהָ? – רַק רְאִיָּתוֹ אֶת עוֹשְׁרוֹ.
יא מְתוּקָה֙ שְׁנַ֣ת הָעֹבֵ֔ד אִם־מְעַ֥ט וְאִם־הַרְבֵּ֖ה יֹאכֵ֑ל וְהַשָּׂבָע֙ לֶֽעָשִׁ֔יר אֵינֶ֛נּוּ מַנִּ֥יחַֽ ל֖וֹ לִישֽׁוֹן׃ מְתוּקָה הִיא הַשֵּׁנָה שֶׁל הָעוֹבֵד בֵּין אִם אָכַל מְעַט בֵּין אִם אָכַל הַרְבֵּה, וְאִלּוּ הַשּׂוֹבַע שֶׁל הֶעָשִׁיר אֵינֶנּוּ מַנִּיחַ לוֹ לִישׁוֹן מֵרוֹב דְּאָגוֹת.
יב יֵ֚שׁ רָעָ֣ה חוֹלָ֔ה רָאִ֖יתִי תַּ֣חַת הַשָּׁ֑מֶשׁ עֹ֛שֶׁר שָׁמ֥וּר לִבְעָלָ֖יו לְרָעָתֽוֹ׃ יֵשׁ רָעָה חוֹלָה שֶׁרָאִיתִי מִתַּחַת לַשֶּׁמֶשׁ, שֶׁאָדָם שׁוֹמֵר אֶת עוֹשְׁרוֹ וְזֶה מַזִּיק לוֹ.
יג וְאָבַ֛ד הָעֹ֥שֶׁר הַה֖וּא בְּעִנְיַ֣ן רָ֑ע וְהוֹלִ֣יד בֵּ֔ן וְאֵ֥ין בְּיָד֖וֹ מְאֽוּמָה׃ כִּי אִם הָעוֹשֶׁר הַהוּא יֹאבַד בְּאֵרוּעַ רַע, וְהוּא יוֹלִיד בֵּן וְאֵין בְּיָדוֹ מְאוּמָה לְהוֹרִישׁ לוֹ.
יד כַּאֲשֶׁ֤ר יָצָא֙ מִבֶּ֣טֶן אִמּ֔וֹ עָר֛וֹם יָשׁ֥וּב לָלֶ֖כֶת כְּשֶׁבָּ֑א וּמְא֙וּמָה֙ לֹא־יִשָּׂ֣א בַעֲמָל֔וֹ שֶׁיֹּלֵ֖ךְ בְּיָדֽוֹ׃ כְּמוֹ שֶׁיָּצָא מִבֶּטֶן אִמּוֹ כָּךְ יָשׁוּב לָלֶכֶת לְעוֹלָמוֹ – עָרוֹם כְּמוֹ שֶׁבָּא, וְלֹא יִקַּח דָּבָר מִכָּל מַה שֶּׁעָמַל בּוֹ לְהָבִיא אִתּוֹ.
טו וְגַם־זֹה֙ רָעָ֣ה חוֹלָ֔ה כׇּל־עֻמַּ֥ת שֶׁבָּ֖א כֵּ֣ן יֵלֵ֑ךְ וּמַה־יִּתְר֣וֹן ל֔וֹ שֶֽׁיַּעֲמֹ֖ל לָרֽוּחַ׃ וְגַם זוֹ רָעָה חוֹלָה – כְּמוֹ שֶׁבָּא לָעוֹלָם כָּךְ יֵלֵךְ, וּמַה הוּא מַרְוִיחַ בָּזֶה שֶׁיַּעֲמוֹל לִכְלוּם?
טז גַּ֥ם כׇּל־יָמָ֖יו בַּחֹ֣שֶׁךְ יֹאכֵ֑ל וְכָעַ֥ס הַרְבֵּ֖ה וְחׇלְי֥וֹ וָקָֽצֶף׃ גַּם כָּל חַיָּיו הוּא אוֹכֵל וְנֶהֱנֶה בַּחוֹשֶׁךְ בְּצִמְצוּם, וְהוּא כּוֹעֵס הַרְבֵּה וְחוֹלֶה וְעַצְבָּנִי.
יז הִנֵּ֞ה אֲשֶׁר־רָאִ֣יתִי אָ֗נִי ט֣וֹב אֲשֶׁר־יָפֶ֣ה לֶֽאֱכוֹל־וְ֠לִשְׁתּ֠וֹת וְלִרְא֨וֹת טוֹבָ֜ה בְּכׇל־עֲמָל֣וֹ ׀ שֶׁיַּעֲמֹ֣ל תַּֽחַת־הַשֶּׁ֗מֶשׁ מִסְפַּ֧ר יְמֵי־חַיָּ֛ו אֲשֶׁר־נָֽתַן־ל֥וֹ הָאֱלֹהִ֖ים כִּי־ה֥וּא חֶלְקֽוֹ׃ הִנֵּה מַה שֶּׁרָאִיתִי אֲנִי, שֶׁעָדִיף לוֹ וְטוֹב לֶאֱכוֹל וְלִשְׁתּוֹת וְלֵהָנוֹת מִכָּל עֲמָלוֹ שֶׁיַּעֲמוֹל מִתַּחַת לַשֶּׁמֶשׁ בְּמִסְפַּר שְׁנוֹת חַיָּיו שֶׁנָּתַן לוֹ הָאֱלֹהִים, כִּי זֶה חֶלְקוֹ.
יח גַּ֣ם כׇּֽל־הָאָדָ֡ם אֲשֶׁ֣ר נָֽתַן־ל֣וֹ הָאֱלֹהִים֩ עֹ֨שֶׁר וּנְכָסִ֜ים וְהִשְׁלִיט֨וֹ לֶאֱכֹ֤ל מִמֶּ֙נּוּ֙ וְלָשֵׂ֣את אֶת־חֶלְק֔וֹ וְלִשְׂמֹ֖חַ בַּעֲמָל֑וֹ זֹ֕ה מַתַּ֥ת אֱלֹהִ֖ים הִֽיא׃ גַּם כָּל אָדָם שֶׁנָּתַן לוֹ הָאֱלֹהִים עוֹשֶׁר וּנְכָסִים וְהִשְׁלִיטוֹ לֵהָנוֹת מִמֶּנּוּ וּלְקַבֵּל אֶת חֶלְקוֹ וְלִשְׂמוֹחַ בַּעֲמָלוֹ – זוֹ הִיא מַתְּנַת אֱלֹהִים.
יט כִּ֚י לֹ֣א הַרְבֵּ֔ה יִזְכֹּ֖ר אֶת־יְמֵ֣י חַיָּ֑יו כִּ֧י הָאֱלֹהִ֛ים מַעֲנֶ֖ה בְּשִׂמְחַ֥ת לִבּֽוֹ׃ כִּי יִזְכּוֹר שֶׁלֹּא רַבִּים הֵם יְמֵי חַיָּיו שֶׁהָאֱלֹהִים מְשַׂמֵּחַ אוֹתוֹ בְּשִׂמְחַת לִבּוֹ.
ו א יֵ֣שׁ רָעָ֔ה אֲשֶׁ֥ר רָאִ֖יתִי תַּ֣חַת הַשָּׁ֑מֶשׁ וְרַבָּ֥ה הִ֖יא עַל־הָאָדָֽם׃ רָאִיתִי שֶׁיֵּשׁ רָעָה תַּחַת הַשֶּׁמֶשׁ, וְהִיא רָעָה גְּדוֹלָה לָאָדָם.
ב אִ֣ישׁ אֲשֶׁ֣ר יִתֶּן־ל֣וֹ הָאֱלֹהִ֡ים עֹ֩שֶׁר֩ וּנְכָסִ֨ים וְכָב֜וֹד וְֽאֵינֶ֨נּוּ חָסֵ֥ר לְנַפְשׁ֣וֹ ׀ מִכֹּ֣ל אֲשֶׁר־יִתְאַוֶּ֗ה וְלֹֽא־יַשְׁלִיטֶ֤נּוּ הָֽאֱלֹהִים֙ לֶאֱכֹ֣ל מִמֶּ֔נּוּ כִּ֛י אִ֥ישׁ נׇכְרִ֖י יֹֽאכְﬞלֶ֑נּוּ זֶ֥ה הֶ֛בֶל וׇחֳלִ֥י רָ֖ע הֽוּא׃ אִישׁ שֶׁהָאֱלֹהִים יִתֵּן לוֹ עוֹשֶׁר וּנְכָסִים וְכָבוֹד וְלֹא חָסֵר לוֹ כְּלוּם מִכָּל דָּבָר שֶׁיִּרְצֶה, אֲבָל הָאֱלֹהִים לֹא יְאַפְשֵׁר לוֹ לֵהָנוֹת מִמֶּנּוּ, אֶלָּא אִישׁ אַחֵר יְקַבְּלוֹ, זֶהוּ הֶבֶל וָחֳלִי רָע.
ג אִם־יוֹלִ֣יד אִ֣ישׁ מֵאָ֡ה וְשָׁנִים֩ רַבּ֨וֹת יִֽחְיֶ֜ה וְרַ֣ב ׀ שֶׁיִּהְי֣וּ יְמֵֽי־שָׁנָ֗יו וְנַפְשׁוֹ֙ לֹא־תִשְׂבַּ֣ע מִן־הַטּוֹבָ֔ה וְגַם־קְבוּרָ֖ה לֹא־הָ֣יְתָה לּ֑וֹ אָמַ֕רְתִּי ט֥וֹב מִמֶּ֖נּוּ הַנָּֽפֶל׃ אִם יוֹלִיד אוֹתוֹ אִישׁ מֵאָה בָּנִים, וְיִחְיֶה שָׁנִים רַבּוֹת כְּכָל שֶׁיְּמֵי חַיָּיו יִהְיוּ רַבִּים, וְנַפְשׁוֹ לֹא תִּשְׂבַּע מִן הַטּוֹבָה, וְגַם קְבוּרָה לֹא תִּהְיֶה לוֹ, אָמַרְתִּי שֶׁנֶּפֶל עָדִיף מִמֶּנּוּ.
ד כִּֽי־בַהֶ֥בֶל בָּ֖א וּבַחֹ֣שֶׁךְ יֵלֵ֑ךְ וּבַחֹ֖שֶׁךְ שְׁמ֥וֹ יְכֻסֶּֽה׃ כִּי הַנֶּפֶל בָּא בְּלִי כְּלוּם וְהוּא אוֹבֵד בַּחוֹשֶׁךְ, וּבַחוֹשֶׁךְ יֵעָלֵם זִכְרוֹ.
ה גַּם־שֶׁ֥מֶשׁ לֹא־רָאָ֖ה וְלֹ֣א יָדָ֑ע נַ֥חַת לָזֶ֖ה מִזֶּֽה׃ וְגַם לֹא רָאָה שֶׁמֶשׁ וְלֹא יָדַע כְּלוּם, לְנֶפֶל זֶה יֵשׁ נַחַת יוֹתֵר מִלְּאִישׁ זֶה.
ו וְאִלּ֣וּ חָיָ֗ה אֶ֤לֶף שָׁנִים֙ פַּעֲמַ֔יִם וְטוֹבָ֖ה לֹ֣א רָאָ֑ה הֲלֹ֛א אֶל־מָק֥וֹם אֶחָ֖ד הַכֹּ֥ל הוֹלֵֽךְ׃ וְאִלּוּ חָיָה פַּעֲמַיִם אֶלֶף שָׁנִים, אִם לֹא רָאָה שׁוּם טוֹבָה, הֲרֵי הוּא וְהַנֶּפֶל הוֹלְכִים אֶל אוֹתוֹ מָקוֹם.
ז כׇּל־עֲמַ֥ל הָאָדָ֖ם לְפִ֑יהוּ וְגַם־הַנֶּ֖פֶשׁ לֹ֥א תִמָּלֵֽא׃ כָּל עֲמַל הָאָדָם זֶה לַאֲכִילָתוֹ, וְגַם הַנֶּפֶשׁ לֹא תִּתְמַלֵּא מִתַּאֲווֹתֶיהָ.
ח כִּ֛י מַה־יּוֹתֵ֥ר לֶחָכָ֖ם מִֽן־הַכְּסִ֑יל מַה־לֶּעָנִ֣י יוֹדֵ֔עַ לַהֲלֹ֖ךְ נֶ֥גֶד הַחַיִּֽים׃ כִּי מַה יֵּשׁ לֶחָכָם יוֹתֵר מִן הַכְּסִיל? מַה צָּרִיךְ הֶעָנִי לָדַעַת יוֹתֵר בְּהִתְמוֹדְדוּיוֹת הַחַיִּים?
ט ט֛וֹב מַרְאֵ֥ה עֵינַ֖יִם מֵֽהֲלׇךְ־נָ֑פֶשׁ גַּם־זֶ֥ה הֶ֖בֶל וּרְע֥וּת רֽוּחַ׃ עָדִיף לֵהָנוֹת מִמַּה שֶּׁרוֹאִים מִלְּחַפֵּשׂ תַּאֲווֹת, גַּם זֶה הֶבֶל וְרַעְיוֹן שָׁוְא.
י מַה־שֶּֽׁהָיָ֗ה כְּבָר֙ נִקְרָ֣א שְׁמ֔וֹ וְנוֹדָ֖ע אֲשֶׁר־ה֣וּא אָדָ֑ם וְלֹא־יוּכַ֣ל לָדִ֔ין עִ֥ם שהתקיף שֶׁתַּקִּ֖יף מִמֶּֽנּוּ׃ מַה שֶּׁהָיָה הָיָה וּמִי שֶׁכְּבָר הִתְפַּרְסֵם וְנוֹדָע שֶׁהוּא אָדָם חָשׁוּב, לֹא יוּכַל לְהִתְמוֹדֵד עִם מִי שֶׁחָזָק מִמֶּנּוּ.
יא כִּ֛י יֵשׁ־דְּבָרִ֥ים הַרְבֵּ֖ה מַרְבִּ֣ים הָ֑בֶל מַה־יֹּתֵ֖ר לָאָדָֽם׃ כִּי יֵשׁ הַרְבֵּה דְּבָרִים שֶׁמַּרְבִּים רַק הֶבֶל, מַה הָרֶוַח בָּהֶם לָאָדָם.
יב כִּ֣י מִֽי־יוֹדֵ֩עַ֩ מַה־טּ֨וֹב לָֽאָדָ֜ם בַּֽחַיִּ֗ים מִסְפַּ֛ר יְמֵי־חַיֵּ֥י הֶבְל֖וֹ וְיַעֲשֵׂ֣ם כַּצֵּ֑ל אֲשֶׁר֙ מִֽי־יַגִּ֣יד לָֽאָדָ֔ם מַה־יִּהְיֶ֥ה אַחֲרָ֖יו תַּ֥חַת הַשָּֽׁמֶשׁ׃ כִּי מִי יוֹדֵעַ מַה טּוֹב לָאָדָם שֶׁיַּעֲשֶׂה בְּחַיָּיו בְּמִסְפַּר יְמֵי חַיֵּי הֶבְלוֹ הַחוֹלְפִים כַּצֵּל, כִּי מִי יַגִּיד לָאָדָם מַה יִּהְיֶה אַחֲרָיו תַּחַת הַשֶּׁמֶשׁ.
ז א ט֥וֹב שֵׁ֖ם מִשֶּׁ֣מֶן ט֑וֹב וְי֣וֹם הַמָּ֔וֶת מִיּ֖וֹם הִוָּלְדֽוֹ׃ עָדִיף שֵׁם טוֹב מִשֶּׁמֶן מְשֻׁבָּח, לָכֵן עָדִיף יוֹם הַמָּוֶת בְּשֵׁם טוֹב מֵהַיּוֹם שֶׁרַק נוֹלָד.
ב ט֞וֹב לָלֶ֣כֶת אֶל־בֵּֽית־אֵ֗בֶל מִלֶּ֙כֶת֙ אֶל־בֵּ֣ית מִשְׁתֶּ֔ה בַּאֲשֶׁ֕ר ה֖וּא ס֣וֹף כׇּל־הָאָדָ֑ם וְהַחַ֖י יִתֵּ֥ן אֶל־לִבּֽוֹ׃ עָדִיף לָלֶכֶת אֶל בֵּית אֵבֶל מִלָּלֶכֶת אֶל בֵּית מִשְׁתֶּה, כִּי זֶה סוֹף כָּל אָדָם, וְהַחַי יַפְנֶה אָז אֶת תְּשׂוּמֶת לִבּוֹ.
ג ט֥וֹב כַּ֖עַס מִשְּׂח֑וֹק כִּֽי־בְרֹ֥עַ פָּנִ֖ים יִ֥יטַב לֵֽב׃ עָדִיף כַּעַס מִצְּחוֹק כִּי בְּפָנִים זְעוּפוֹת הַלֵּב נִרְגָּע.
ד לֵ֤ב חֲכָמִים֙ בְּבֵ֣ית אֵ֔בֶל וְלֵ֥ב כְּסִילִ֖ים בְּבֵ֥ית שִׂמְחָֽה׃ חֲכָמִים חוֹשְׁבִים עַל בֵּית הָאֵבֶל, וְהַטִּפְּשִׁים עַל בֵּית הַשִּׂמְחָה.
ה ט֕וֹב לִשְׁמֹ֖עַ גַּעֲרַ֣ת חָכָ֑ם מֵאִ֕ישׁ שֹׁמֵ֖עַ שִׁ֥יר כְּסִילִֽים׃ עָדִיף לִשְׁמוֹעַ גְּעָרָה מֵחָכָם מִלְּהַאֲזִין לְשִׁיר יָפֶה שֶׁל טִפְּשִׁים.
ו כִּ֣י כְק֤וֹל הַסִּירִים֙ תַּ֣חַת הַסִּ֔יר כֵּ֖ן שְׂחֹ֣ק הַכְּסִ֑יל וְגַם־זֶ֖ה הָֽבֶל׃ כִּי כְּמוֹ קוֹל שְׂרֵיפַת הַקּוֹצִים תַּחַת הַסִּיר כָּךְ צְחוֹק הַטִּפֵּשׁ, וְגַם זֶה הֶבֶל.
ז כִּ֥י הָעֹ֖שֶׁק יְהוֹלֵ֣ל חָכָ֑ם וִֽיאַבֵּ֥ד אֶת־לֵ֖ב מַתָּנָֽה׃ כִּי הַגֶּזֶל מְטַפֵּשׁ אֶת הֶחָכָם, וּמַתָּנָה מְאַבֶּדֶת אֶת חָכְמָתוֹ.
ח ט֛וֹב אַחֲרִ֥ית דָּבָ֖ר מֵֽרֵאשִׁית֑וֹ ט֥וֹב אֶֽרֶךְ־ר֖וּחַ מִגְּבַהּ־רֽוּחַ׃ עֲדִיפָה הַשְׁלָמַת דָּבָר מֵאֲשֶׁר הַתְחָלָתוֹ, עָדִיף סַבְלָן מִגַּאַוְתָן.
ט אַל־תְּבַהֵ֥ל בְּרֽוּחֲךָ֖ לִכְע֑וֹס כִּ֣י כַ֔עַס בְּחֵ֥יק כְּסִילִ֖ים יָנֽוּחַ׃ אַל תִּהְיֶה בַּעַל אוֹפִי מְמַהֵר לִכְעוֹס, כִּי הַכַּעַס נִמְצָא אֵצֶל טִפְּשִׁים.
י אַל־תֹּאמַר֙ מֶ֣ה הָיָ֔ה שֶׁ֤הַיָּמִים֙ הָרִ֣אשֹׁנִ֔ים הָי֥וּ טוֹבִ֖ים מֵאֵ֑לֶּה כִּ֛י לֹ֥א מֵחׇכְמָ֖ה שָׁאַ֥לְתָּ עַל־זֶֽה׃ אַל תֹּאמַר בְּגַעְגּוּעַ: "מֶה הָיָה פַּעַם – שֶׁהַזְּמָנִים הַקּוֹדְמִים הָיוּ טוֹבִים מֵאֵלֶּה", כִּי לֹא מִתּוֹךְ חָכְמָה הִתְלוֹנַנְתָּ עַל זֶה.
יא טוֹבָ֥ה חׇכְמָ֖ה עִֽם־נַחֲלָ֑ה וְיֹתֵ֖ר לְרֹאֵ֥י הַשָּֽׁמֶשׁ׃ טוֹב לְמִי שֶׁיֵּשׁ חָכְמָה עִם יְרֻשָּׁה, זֶה מוֹעִיל לְמִי שֶׁחַי בָּעוֹלָם.
יב כִּ֛י בְּצֵ֥ל הַֽחׇכְמָ֖ה בְּצֵ֣ל הַכָּ֑סֶף וְיִתְר֣וֹן דַּ֔עַת הַֽחׇכְמָ֖ה תְּחַיֶּ֥ה בְעָלֶֽיהָ׃ כִּי הַחָכְמָה מְגִנָּה וְהַכֶּסֶף מֵגֵן, וּמַעֲלַת הַדַּעַת הִיא שֶׁהַחָכְמָה מְחַיָּה אֶת בְּעָלֶיהָ.
יג רְאֵ֖ה אֶת־מַעֲשֵׂ֣ה הָאֱלֹהִ֑ים כִּ֣י מִ֤י יוּכַל֙ לְתַקֵּ֔ן אֵ֖ת אֲשֶׁ֥ר עִוְּתֽוֹ׃ וְקַבֵּל אֶת מַעֲשֵׂה הָאֱלֹהִים כְּפִי שֶׁהוּא, כִּי מִי יוּכַל לְתַקֵּן אֶת מַה שֶּׁה' כִּבְיָכוֹל קִלְקֵל.
יד בְּי֤וֹם טוֹבָה֙ הֱיֵ֣ה בְט֔וֹב וּבְי֥וֹם רָעָ֖ה רְאֵ֑ה גַּ֣ם אֶת־זֶ֤ה לְעֻמַּת־זֶה֙ עָשָׂ֣ה הָֽאֱלֹהִ֔ים עַל־דִּבְרַ֗ת שֶׁלֹּ֨א יִמְצָ֧א הָֽאָדָ֛ם אַחֲרָ֖יו מְאֽוּמָה׃ בְּיוֹם טוֹבָה הַרְגֵּשׁ טוֹב, וּבְיוֹם רָעָה קַבֵּל אֶת הַדִּין, גַּם אֶת הָרָע עָשָׂה הָאֱלֹהִים כְּעוֹנֶשׁ עַל מַעֲשֶׂה רָע, כְּדֵי שֶׁלֹּא יִמְצָא הָאָדָם אַחֲרֵי מוֹתוֹ שׁוּם חוֹבָה.
טו אֶת־הַכֹּ֥ל רָאִ֖יתִי בִּימֵ֣י הֶבְלִ֑י יֵ֤שׁ צַדִּיק֙ אֹבֵ֣ד בְּצִדְק֔וֹ וְיֵ֣שׁ רָשָׁ֔ע מַאֲרִ֖יךְ בְּרָעָתֽוֹ׃ אֶת הַכֹּל רָאִיתִי בִּשְׁנוֹת הַהֶבֶל שֶׁלִּי, יֵשׁ צַדִּיק שֶׁאוֹבֵד בִּגְלַל צִדְקוּתוֹ וְיֵשׁ רָשָׁע שֶׁמַּאֲרִיךְ יָמִים בְּעֶזְרַת רִשְׁעוּתוֹ.
טז אַל־תְּהִ֤י צַדִּיק֙ הַרְבֵּ֔ה וְאַל־תִּתְחַכַּ֖ם יוֹתֵ֑ר לָ֖מָּה תִּשּׁוֹמֵֽם׃ אַל תִּהְיֶה יוֹתֵר מִדַּי צַדִּיק וְאַל תִּתְחַכֵּם יוֹתֵר מִדַּי, לָמָּה שֶׁתִּהְיֶה מְשֻׁגַּע וּמְנֻתָּק.
יז אַל־תִּרְשַׁ֥ע הַרְבֵּ֖ה וְאַל־תְּהִ֣י סָכָ֑ל לָ֥מָּה תָמ֖וּת בְּלֹ֥א עִתֶּֽךָ׃ אַל תַּרְשִׁיעַ יוֹתֵר מֵהַמֻּתָּר וְאַל תִּהְיֶה טִפֵּשׁ, לָמָּה שֶׁתָּמוּת לִפְנֵי זְמַנְּךָ.
יח ט֚וֹב אֲשֶׁ֣ר תֶּאֱחֹ֣ז בָּזֶ֔ה וְגַם־מִזֶּ֖ה אַל־תַּנַּ֣ח אֶת־יָדֶ֑ךָ כִּֽי־יְרֵ֥א אֱלֹהִ֖ים יֵצֵ֥א אֶת־כֻּלָּֽם׃ טוֹב שֶׁתֶּאֱחוֹז בְּצִדְקוּת, וְגַם אַל תַּעֲזוֹב אֶת הַחָכְמָה, כִּי יְרֵא אֱלֹהִים עוֹשֶׂה אֶת שְׁנֵיהֶם.
יט הַֽחׇכְמָ֖ה תָּעֹ֣ז לֶחָכָ֑ם מֵֽעֲשָׂרָה֙ שַׁלִּיטִ֔ים אֲשֶׁ֥ר הָי֖וּ בָּעִֽיר׃ הַחָכְמָה נוֹתֶנֶת עָצְמָה לֶחָכָם, יוֹתֵר מֵעֲשָׂרָה שַׁלִּיטִים חֲזָקִים שֶׁנִּמְצָאִים בָּעִיר.
כ כִּ֣י אָדָ֔ם אֵ֥ין צַדִּ֖יק בָּאָ֑רֶץ אֲשֶׁ֥ר יַעֲשֶׂה־טּ֖וֹב וְלֹ֥א יֶחֱטָֽא׃ הֲרֵי אֵין בָּאָרֶץ אָדָם צַדִּיק שֶׁיַּעֲשֶׂה רַק טוֹב וְלֹא יִטְעֶה.
כא גַּ֤ם לְכׇל־הַדְּבָרִים֙ אֲשֶׁ֣ר יְדַבֵּ֔רוּ אַל־תִּתֵּ֖ן לִבֶּ֑ךָ אֲשֶׁ֛ר לֹֽא־תִשְׁמַ֥ע אֶֽת־עַבְדְּךָ֖ מְקַלְלֶֽךָ׃ גַּם אַל תָּשִׂים לֵב לְכָל הַדְּבָרִים הָרָעִים שֶׁמְּדַבְּרִים, אַל תַּקְשִׁיב אִם מִי שֶׁצָּרִיךְ לְכַבֵּד אוֹתְךָ מְזַלְזֵל בְּךָ.
כב כִּ֛י גַּם־פְּעָמִ֥ים רַבּ֖וֹת יָדַ֣ע לִבֶּ֑ךָ אֲשֶׁ֥ר גַּם־אַתָּ֖ה את קִלַּ֥לְתָּ אֲחֵרִֽים׃ כִּי הֲרֵי אַתָּה יוֹדֵעַ שֶׁגַּם פְּעָמִים רַבּוֹת גַּם אַתָּה זִלְזַלְתָּ בַּאֲחֵרִים.
כג כׇּל־זֹ֖ה נִסִּ֣יתִי בַֽחׇכְמָ֑ה אָמַ֣רְתִּי אֶחְכָּ֔מָה וְהִ֖יא רְחוֹקָ֥ה מִמֶּֽנִּי׃ בְּכָל זֶה הִתְעַלֵּתִי בַּחָכְמָה, רָצִיתִי לְהַחְכִּים עוֹד, וְזֶה הָיָה לְמַעְלָה מִיְּכָלְתִּי.
כד רָח֖וֹק מַה־שֶּׁהָיָ֑ה וְעָמֹ֥ק ׀ עָמֹ֖ק מִ֥י יִמְצָאֶֽנּוּ׃ מַה שֶּׁכְּכָר הָיָה הוּא רָחוֹק, וְיֵשׁ דְּבָרִים עֲמֻקִּים – מִי יִמְצָא אוֹתָם.
כה סַבּ֨וֹתִֽי אֲנִ֤י וְלִבִּי֙ לָדַ֣עַת וְלָת֔וּר וּבַקֵּ֥שׁ חׇכְמָ֖ה וְחֶשְׁבּ֑וֹן וְלָדַ֙עַת֙ רֶ֣שַׁע כֶּ֔סֶל וְהַסִּכְל֖וּת הוֹלֵלֽוֹת׃ פָּנִיתִי בְּמַחְשַׁבְתִּי לָדַעַת וּלְסַיֵּר, וּלְחַפֵּשׂ חָכְמָה וּמַחְשָׁבוֹת, וּלְהַכִּיר רִשְׁעוּת טִפְּשִׁית וְהוֹלְלוּת שֶׁל בּוּרוּת.
כו וּמוֹצֶ֨א אֲנִ֜י מַ֣ר מִמָּ֗וֶת אֶת־הָֽאִשָּׁה֙ אֲשֶׁר־הִ֨יא מְצוֹדִ֧ים וַחֲרָמִ֛ים לִבָּ֖הּ אֲסוּרִ֣ים יָדֶ֑יהָ ט֞וֹב לִפְנֵ֤י הָאֱלֹהִים֙ יִמָּלֵ֣ט מִמֶּ֔נָּה וְחוֹטֵ֖א יִלָּ֥כֶד בָּֽהּ׃ וְגִלִּיתִי דָּבָר מַר מִמָּוֶת – אֶת הָאִשָּׁה הַזָּרָה, שֶׁהִיא – לִבָּהּ כְּמוֹ מַלְכּוֹדוֹת וּרְשָׁתוֹת וְיָדֶיהָ כְּמוֹ אֲזִיקִים, הַמֵּטִיב דְּרָכָיו לִפְנֵי הָאֱלֹהִים יִמָּלֵט מִמֶּנָּה, וְחוֹטֵא יִלָּכֵד בָּהּ.
כז רְאֵה֙ זֶ֣ה מָצָ֔אתִי אָמְרָ֖ה קֹהֶ֑לֶת אַחַ֥ת לְאַחַ֖ת לִמְצֹ֥א חֶשְׁבּֽוֹן׃ רְאֵה, אֶת זֶה גִּלִּיתִי, אָמַר קוֹהֶלֶת, כְּשֶׁבָּדַקְתִּי אַחַת אַחַת לְחַשֵּׁב הַכֹּל.
כח אֲשֶׁ֛ר עוֹד־בִּקְשָׁ֥ה נַפְשִׁ֖י וְלֹ֣א מָצָ֑אתִי אָדָ֞ם אֶחָ֤ד מֵאֶ֙לֶף֙ מָצָ֔אתִי וְאִשָּׁ֥ה בְכׇל־אֵ֖לֶּה לֹ֥א מָצָֽאתִי׃ עוֹד דָּבָר חִפַּשְׂתִּי וְלֹא מָצָאתִי – אָדָם אֶחָד מֵאֶלֶף מָצָאתִי נֶאֱמָן, וְאִשָּׁה גַּם אַחַת מֵאֶלֶף לֹא מָצָאתִי בְּכָל נְשׁוֹתַי.
כט לְבַד֙ רְאֵה־זֶ֣ה מָצָ֔אתִי אֲשֶׁ֨ר עָשָׂ֧ה הָאֱלֹהִ֛ים אֶת־הָאָדָ֖ם יָשָׁ֑ר וְהֵ֥מָּה בִקְשׁ֖וּ חִשְּׁבֹנ֥וֹת רַבִּֽים׃ רְאֵה, רַק אֶת זֶה מָצָאתִי, שֶׁהָאֱלֹהִים עָשָׂה אֶת הָאָדָם יָשָׁר, וְהָאֲנָשִׁים הֵם שֶׁחִפְּשׂוּ מְזִמּוֹת רַבּוֹת.
ח א מִ֚י כְּהֶ֣חָכָ֔ם וּמִ֥י יוֹדֵ֖עַ פֵּ֣שֶׁר דָּבָ֑ר חׇכְמַ֤ת אָדָם֙ תָּאִ֣יר פָּנָ֔יו וְעֹ֥ז פָּנָ֖יו יְשֻׁנֶּֽא׃ מִי חָשׁוּב כְּמוֹ חָכָם! מִי יוֹדֵעַ סִבַּת כָּל דָּבָר! חָכְמַת הָאָדָם מְאִירָה אֶת פָּנָיו וְהֵן מִשְׁתַּנּוֹת וּמְקַבְּלוֹת עָצְמָה.
ב אֲנִי֙ פִּי־מֶ֣לֶךְ שְׁמֹ֔ר וְעַ֕ל דִּבְרַ֖ת שְׁבוּעַ֥ת אֱלֹהִֽים׃ אֲנִי מַקְפִּיד לְקַיֵּם אֶת דִּבְרֵי פִּי הַמֶּלֶךְ וְאֶת דִּבְרֵי שְׁבוּעַת הָאֱלֹהִים.
ג אַל־תִּבָּהֵ֤ל מִפָּנָיו֙ תֵּלֵ֔ךְ אַֽל־תַּעֲמֹ֖ד בְּדָבָ֣ר רָ֑ע כִּ֛י כׇּל־אֲשֶׁ֥ר יַחְפֹּ֖ץ יַעֲשֶֽׂה׃ אַל תִּבָּהֵל לְנַסּוֹת לִבְרוֹחַ מִפָּנָיו, אַל תִּתְעַסֵּק בְּדָבָר רָע, כִּי כָּל מַה שֶּׁיַּחְפּוֹץ הוּא יַעֲשֶׂה.
ד בַּאֲשֶׁ֥ר דְּבַר־מֶ֖לֶךְ שִׁלְט֑וֹן וּמִ֥י יֹֽאמַר־ל֖וֹ מַֽה־תַּעֲשֶֽׂה׃ מִפְּנֵי שֶׁדְּבַר מֶלֶךְ הוּא שֶׁשּׁוֹלֵט, כִּי מִי יֹאמַר לוֹ מַה תַּעֲשֶׂה.
ה שׁוֹמֵ֣ר מִצְוָ֔ה לֹ֥א יֵדַ֖ע דָּבָ֣ר רָ֑ע וְעֵ֣ת וּמִשְׁפָּ֔ט יֵדַ֖ע לֵ֥ב חָכָֽם׃ לְשׁוֹמֵר מִצְוַת ה' לֹא יִקְרֶה דָּבָר רָע, לָכֵן חָכָם חוֹשֵׁב עַל זְמַן הַמִּשְׁפָּט.
ו כִּ֣י לְכׇל־חֵ֔פֶץ יֵ֖שׁ עֵ֣ת וּמִשְׁפָּ֑ט כִּֽי־רָעַ֥ת הָאָדָ֖ם רַבָּ֥ה עָלָֽיו׃ כִּי לְכָל מִלּוּי תַּאֲוָה יֵשׁ זְמָן לְמִשְׁפָּט, כִּי רָעַת הָאָדָם גּוֹרֶמֶת לוֹ רָעָה רַבָּה.
ז כִּֽי־אֵינֶ֥נּוּ יֹדֵ֖עַ מַה־שֶּׁיִּֽהְיֶ֑ה כִּ֚י כַּאֲשֶׁ֣ר יִֽהְיֶ֔ה מִ֖י יַגִּ֥יד לֽוֹ׃ כִּי אֵינֶנּוּ יוֹדֵעַ מַה יִּהְיֶה עוֹנְשׁוֹ, וְכַאֲשֶׁר זֶה יִקְרֶה מִי יַזְהִיר אוֹתוֹ.
ח אֵ֣ין אָדָ֞ם שַׁלִּ֤יט בָּר֙וּחַ֙ לִכְל֣וֹא אֶת־הָר֔וּחַ וְאֵ֤ין שִׁלְטוֹן֙ בְּי֣וֹם הַמָּ֔וֶת וְאֵ֥ין מִשְׁלַ֖חַת בַּמִּלְחָמָ֑ה וְלֹֽא־יְמַלֵּ֥ט רֶ֖שַׁע אֶת־בְּעָלָֽיו׃ אֵין אָדָם שַׁלִּיט לִכְלוֹא אֶת רוּחַ הַחַיִּים מִלְּהִסְתַּלֵּק, וְאֵין שְׁלִיטָה בְּעִתּוּי יוֹם הַמָּוֶת, אִי אֶפְשָׁר לָצֵאת לַמִּלְחָמָה נֶגְדּוֹ, הָרִשְׁעוּת לֹא תְּמַלֵּט אֶת בְּעָלֶיהָ.
ט אֶת־כׇּל־זֶ֤ה רָאִ֙יתִי֙ וְנָת֣וֹן אֶת־לִבִּ֔י לְכׇֽל־מַעֲשֶׂ֔ה אֲשֶׁ֥ר נַעֲשָׂ֖ה תַּ֣חַת הַשָּׁ֑מֶשׁ עֵ֗ת אֲשֶׁ֨ר שָׁלַ֧ט הָאָדָ֛ם בְּאָדָ֖ם לְרַ֥ע לֽוֹ׃ אֶת כָּל זֶה רָאִיתִי, וְשַׂמְתִּי לִבִּי לְכָל הַמַּעֲשִׂים שֶׁנַּעֲשִׂים תַּחַת הַשֶּׁמֶשׁ, בִּזְמַן שֶׁאָדָם שׁוֹלֵט לְרָעָה בְּאָדָם זֶה מִתְנַקֵּם בּוֹ.
י וּבְכֵ֡ן רָאִ֩יתִי֩ רְשָׁעִ֨ים קְבֻרִ֜ים וָבָ֗אוּ וּמִמְּק֤וֹם קָדוֹשׁ֙ יְהַלֵּ֔כוּ וְיִֽשְׁתַּכְּח֥וּ בָעִ֖יר אֲשֶׁ֣ר כֵּן־עָשׂ֑וּ גַּם־זֶ֖ה הָֽבֶל׃ וּבְכֵן, רָאִיתִי רְשָׁעִים שֶׁהִגִּיעוּ לִקְבוּרָה בְּשַׁלְוָה כְּאִלּוּ הִגִּיעוּ מִמָּקוֹם קָדוֹשׁ, וְנִשְׁכַּח בָּעִיר שֶׁעָשׂוּ רַע, גַּם זֶה הֶבֶל.
יא אֲשֶׁר֙ אֵין־נַעֲשָׂ֣ה פִתְגָ֔ם מַעֲשֵׂ֥ה הָרָעָ֖ה מְהֵרָ֑ה עַל־כֵּ֡ן מָלֵ֞א לֵ֧ב בְּֽנֵי־הָאָדָ֛ם בָּהֶ֖ם לַעֲשׂ֥וֹת רָֽע׃ בִּגְלַל שֶׁהַמִּשְׁפַּט עַל הַמַּעֲשִׂים הָרָעִים לֹא נַעֲשָׂה בִּמְהֵרָה, לָכֵן יֵצֶר לֵב בְּנֵי הָאָדָם מִתְמַלֵּא בְּתוֹכָם בְּרָצוֹן לַעֲשׂוֹת רָע.
יב אֲשֶׁ֣ר חֹטֶ֗א עֹשֶׂ֥ה רָ֛ע מְאַ֖ת וּמַאֲרִ֣יךְ ל֑וֹ כִּ֚י גַּם־יוֹדֵ֣עַ אָ֔נִי אֲשֶׁ֤ר יִהְיֶה־טּוֹב֙ לְיִרְאֵ֣י הָאֱלֹהִ֔ים אֲשֶׁ֥ר יִֽירְא֖וּ מִלְּפָנָֽיו׃ כִּי חוֹטֶא עוֹשֶׂה רָע מֵאָה פְּעָמִים וַה' מַמְתִּין מִלְּהַעֲנִישׁוֹ, כִּי אָמְנָם יוֹדֵעַ אֲנִי שֶׁיִּהְיֶה טוֹב לְיִרְאֵי הָאֱלֹהִים שֶׁמְּכַבְּדִים אוֹתוֹ בְּדַרְכָּם.
יג וְטוֹב֙ לֹֽא־יִהְיֶ֣ה לָֽרָשָׁ֔ע וְלֹֽא־יַאֲרִ֥יךְ יָמִ֖ים כַּצֵּ֑ל אֲשֶׁ֛ר אֵינֶ֥נּוּ יָרֵ֖א מִלִּפְנֵ֥י אֱלֹהִֽים׃ וְלָרָשָׁע לֹא יִהְיֶה טוֹב, וְהוּא לֹא יַאֲרִיךְ יָמִים, כְּמוֹ צֵל, כִּי אֵינֶנּוּ מְכַבֵּד אֱלֹהִים בְּדַרְכּוֹ.
יד יֶשׁ־הֶ֘בֶל֮ אֲשֶׁ֣ר נַעֲשָׂ֣ה עַל־הָאָ֒רֶץ֒ אֲשֶׁ֣ר ׀ יֵ֣שׁ צַדִּיקִ֗ים אֲשֶׁ֨ר מַגִּ֤יעַ אֲלֵהֶם֙ כְּמַעֲשֵׂ֣ה הָרְשָׁעִ֔ים וְיֵ֣שׁ רְשָׁעִ֔ים שֶׁמַּגִּ֥יעַ אֲלֵהֶ֖ם כְּמַעֲשֵׂ֣ה הַצַּדִּיקִ֑ים אָמַ֕רְתִּי שֶׁגַּם־זֶ֖ה הָֽבֶל׃ יֶשׁ הֶבֶל נוֹסָף שֶׁנַּעֲשָׂה בָּאָרֶץ, שֶׁיֵּשׁ צַדִּיקִים שֶׁסּוֹפְגִים אֶת מַה שֶּׁמַּגִּיעַ עַל מַעֲשֵׂה הָרְשָׁעִים, וְיֵשׁ רְשָׁעִים שֶׁמְּקַבְּלִים אֶת מַה שֶּׁמַּגִּיעַ עַל מַעֲשֵׂה הַצַּדִּיקִים, אָמַרְתִּי שֶׁגַּם זֶה הֶבֶל.
טו וְשִׁבַּ֤חְתִּֽי אֲנִי֙ אֶת־הַשִּׂמְחָ֔ה אֲשֶׁ֨ר אֵֽין־ט֤וֹב לָֽאָדָם֙ תַּ֣חַת הַשֶּׁ֔מֶשׁ כִּ֛י אִם־לֶאֱכֹ֥ל וְלִשְׁתּ֖וֹת וְלִשְׂמ֑וֹחַ וְה֞וּא יִלְוֶ֣נּוּ בַעֲמָל֗וֹ יְמֵ֥י חַיָּ֛יו אֲשֶׁר־נָֽתַן־ל֥וֹ הָאֱלֹהִ֖ים תַּ֥חַת הַשָּֽׁמֶשׁ׃ וְשִׁבַּחְתִּי אֶת הַשִּׂמְחָה – שֶׁאֵין דָּבָר טוֹב לָאָדָם לַעֲשׂוֹת תַּחַת הַשֶּׁמֶשׁ אֶלָּא לֶאֱכוֹל וְלִשְׁתּוֹת וְלִשְׂמוֹחַ, וְזֶה מַה שֶּׁיְּלַוֶּה אוֹתוֹ בַּעֲמָלוֹ בִּימֵי חַיָּיו שֶׁהָאֱלֹהִים נָתַן לוֹ תַּחַת הַשֶּׁמֶשׁ.
טז כַּאֲשֶׁ֨ר נָתַ֤תִּי אֶת־לִבִּי֙ לָדַ֣עַת חׇכְמָ֔ה וְלִרְאוֹת֙ אֶת־הָ֣עִנְיָ֔ן אֲשֶׁ֥ר נַעֲשָׂ֖ה עַל־הָאָ֑רֶץ כִּ֣י גַ֤ם בַּיּוֹם֙ וּבַלַּ֔יְלָה שֵׁנָ֕ה בְּעֵינָ֖יו אֵינֶ֥נּוּ רֹאֶֽה׃ זֹאת אָמַרְתִּי כְּשֶׁשַּׂמְתִּי אֶת לִבִּי לָדַעַת חָכְמָה וְלִרְאוֹת אֶת הָעִנְיָן שֶׁנַּעֲשָׂה בָּאָרֶץ, הֲרֵי גַּם בַּיּוֹם וְגַם בַּלַּיְלָה הָאָדָם אֵינֶנּוּ רוֹאֶה שֵׁנָה בְּעֵינָיו.
יז וְרָאִ֘יתִי֮ אֶת־כׇּל־מַעֲשֵׂ֣ה הָאֱלֹהִים֒ כִּי֩ לֹ֨א יוּכַ֜ל הָאָדָ֗ם לִמְצוֹא֙ אֶת־הַֽמַּעֲשֶׂה֙ אֲשֶׁ֣ר נַעֲשָׂ֣ה תַֽחַת־הַשֶּׁ֔מֶשׁ בְּ֠שֶׁ֠ל אֲשֶׁ֨ר יַעֲמֹ֧ל הָאָדָ֛ם לְבַקֵּ֖שׁ וְלֹ֣א יִמְצָ֑א וְגַ֨ם אִם־יֹאמַ֤ר הֶֽחָכָם֙ לָדַ֔עַת לֹ֥א יוּכַ֖ל לִמְצֹֽא׃ וְרָאִיתִי אֶת כָּל מַעֲשֵׂה הָאֱלֹהִים, שֶׁאָדָם לֹא יוּכַל לִמְצוֹא אֶת סִבַּת מַה שֶּׁנַּעֲשָׂה תַּחַת הַשֶּׁמֶשׁ, כִּי כְּכָל שֶׁיַּעֲמוֹל הָאָדָם לְחַפֵּשׁ הוּא לֹא יִמְצָא, וְגַם אִם הֶחָכָם הַגָּדוֹל יִרְצֶה לָדַעַת הוּא לֹא יוּכַל לִמְצֹא.
ט א כִּ֣י אֶת־כׇּל־זֶ֞ה נָתַ֤תִּי אֶל־לִבִּי֙ וְלָב֣וּר אֶת־כׇּל־זֶ֔ה אֲשֶׁ֨ר הַצַּדִּיקִ֧ים וְהַחֲכָמִ֛ים וַעֲבָדֵיהֶ֖ם בְּיַ֣ד הָאֱלֹהִ֑ים גַּֽם־אַהֲבָ֣ה גַם־שִׂנְאָ֗ה אֵ֤ין יוֹדֵ֙עַ֙ הָֽאָדָ֔ם הַכֹּ֖ל לִפְנֵיהֶֽם׃ כִּי שַׂמְתִּי לֵב לְכָל זֶה וְהֶחְלַטְתִּי לְבָרֵר אֶת כָּל זֶה, שֶׁהַצַּדִּיקִים וְהַחֲכָמִים וְעַבְדֵיהֶם בְּיַד הָאֱלֹהִים, גַּם גְּמוּל שֶׁל אַהֲבָה וְגַם שֶׁל שִׂנְאָה, וְהָאָדָם אֵינוֹ יוֹדֵעַ וּמֵבִין אֶת כָּל הַקּוֹרֶה לָהֶם.
ב הַכֹּ֞ל כַּאֲשֶׁ֣ר לַכֹּ֗ל מִקְרֶ֨ה אֶחָ֜ד לַצַּדִּ֤יק וְלָרָשָׁע֙ לַטּוֹב֙ וְלַטָּה֣וֹר וְלַטָּמֵ֔א וְלַ֨זֹּבֵ֔חַ וְלַאֲשֶׁ֖ר אֵינֶ֣נּוּ זֹבֵ֑חַ כַּטּוֹב֙ כַּֽחֹטֶ֔א הַנִּשְׁבָּ֕ע כַּאֲשֶׁ֖ר שְׁבוּעָ֥ה יָרֵֽא׃ לְכֻלָּם קוֹרֶה כְּמוֹ לְכֻלָּם, אוֹתוֹ מִקְרֶה קוֹרֶה לַצַּדִּיק – וְלָרָשָׁע, לַטּוֹב וְלַטָּהוֹר – וְלַטָּמֵא, לַמַּקְרִיב לַה' וּלְמִי שֶׁאֵינֶנּוּ מַקְרִיב; הַטּוֹב הוּא כַּחוֹטֶא, הַנִּשְׁבָּע לַשֶּׁקֶר כְּמוֹ הַנִּמְנָע מִשְּׁבוּעָה.
ג זֶ֣ה ׀ רָ֗ע בְּכֹ֤ל אֲשֶֽׁר־נַעֲשָׂה֙ תַּ֣חַת הַשֶּׁ֔מֶשׁ כִּֽי־מִקְרֶ֥ה אֶחָ֖ד לַכֹּ֑ל וְגַ֣ם לֵ֣ב בְּֽנֵי־הָ֠אָדָ֠ם מָלֵא־רָ֨ע וְהוֹלֵל֤וֹת בִּלְבָבָם֙ בְּחַיֵּיהֶ֔ם וְאַחֲרָ֖יו אֶל־הַמֵּתִֽים׃ זֶה דָּבָר רָע בְּכָל מַה שֶּׁקּוֹרֶה תַּחַת הַשֶּׁמֶשׁ, שֶׁמִּקְרֶה אֶחָד קוֹרֶה לְכֻלָּם, וְגַם לֵב בְּנֵי הָאָדָם מָלֵא רָע, בְּחַיֵּיהֶם לְבָבָם מָלֵא בְּהוֹלֵלוּת, וְסוֹפָם מָוֶת.
ד כִּי־מִי֙ אֲשֶׁ֣ר יבחר יְחֻבַּ֔ר אֶ֥ל כׇּל־הַחַיִּ֖ים יֵ֣שׁ בִּטָּח֑וֹן כִּֽי־לְכֶ֤לֶב חַי֙ ה֣וּא ט֔וֹב מִן־הָאַרְיֵ֖ה הַמֵּֽת׃ כִּי לְמִי יֵשׁ בִּטָּחוֹן שֶׁיִּתְחַבֵּר אֶל חַיֵּי הַנֶּצַח, הֲרֵי לְכֶלֶב חַי יוֹתֵר טוֹב מִלָּאַרְיֵה הַמֵּת.
ה כִּ֧י הַֽחַיִּ֛ים יוֹדְעִ֖ים שֶׁיָּמֻ֑תוּ וְהַמֵּתִ֞ים אֵינָ֧ם יוֹדְעִ֣ים מְא֗וּמָה וְאֵֽין־ע֤וֹד לָהֶם֙ שָׂכָ֔ר כִּ֥י נִשְׁכַּ֖ח זִכְרָֽם׃ כִּי אֵלֶּה שֶׁחַיִּים יוֹדְעִים שֶׁיָּמוּתוּ, וְהַמֵּתִים אֵינָם יוֹדְעִים כְּלוּם, וְאֵין לָהֶם עוֹד שָׂכָר, הֲרֵי נִשְׁכַּח זִכְרָם.
ו גַּ֣ם אַהֲבָתָ֧ם גַּם־שִׂנְאָתָ֛ם גַּם־קִנְאָתָ֖ם כְּבָ֣ר אָבָ֑דָה וְחֵ֨לֶק אֵין־לָהֶ֥ם עוֹד֙ לְעוֹלָ֔ם בְּכֹ֥ל אֲשֶֽׁר־נַעֲשָׂ֖ה תַּ֥חַת הַשָּֽׁמֶשׁ׃ גַּם אַהֲבָתָם גַּם שִׂנְאָתָם וְגַם קִנְאָתָם כְּבָר נֶעֶלְמוּ, וּלְעוֹלָם לֹא יִהְיֶה לָהֶם עוֹד חֵלֶק בְּכָל מַה שֶׁקּוֹרֶה תַּחַת הַשֶּׁמֶשׁ.
ז לֵ֣ךְ אֱכֹ֤ל בְּשִׂמְחָה֙ לַחְמֶ֔ךָ וּֽשְׁﬞתֵ֥ה בְלֶב־ט֖וֹב יֵינֶ֑ךָ כִּ֣י כְבָ֔ר רָצָ֥ה הָאֱלֹהִ֖ים אֶֽת־מַעֲשֶֽׂיךָ׃ לֵךְ אֱכוֹל בְּשִׂמְחָה אֶת לַחְמְךָ וּשְׁתֵה בְּהַרְגָּשָׁה טוֹבָה אֶת יֵינְךָ – כַּאֲשֶׁר הָאֱלֹהִים כְּבָר מְרוּצֶה מִמַּעֲשֶׂיךָ.
ח בְּכׇל־עֵ֕ת יִהְי֥וּ בְגָדֶ֖יךָ לְבָנִ֑ים וְשֶׁ֖מֶן עַל־רֹאשְׁךָ֥ אַל־יֶחְסָֽר׃ וּבְכָל זְמָן יִהְיוּ בְּגָדֶיךָ חֲגִיגִיִּים, וְאַל יֶחְסַר שֶׁמֶן מְבֻשָּׂם עַל רֹאשְׁךָ.
ט רְאֵ֨ה חַיִּ֜ים עִם־אִשָּׁ֣ה אֲשֶׁר־אָהַ֗בְתָּ כׇּל־יְמֵי֙ חַיֵּ֣י הֶבְלֶ֔ךָ אֲשֶׁ֤ר נָֽתַן־לְךָ֙ תַּ֣חַת הַשֶּׁ֔מֶשׁ כֹּ֖ל יְמֵ֣י הֶבְלֶ֑ךָ כִּ֣י ה֤וּא חֶלְקְךָ֙ בַּֽחַיִּ֔ים וּבַעֲמָ֣לְךָ֔ אֲשֶׁר־אַתָּ֥ה עָמֵ֖ל תַּ֥חַת הַשָּֽׁמֶשׁ׃ וְתִחְיֶה עִם אִשָּׁה שֶׁאַתָּה אוֹהֵב בְּכָל יְמֵי חַיֵּי הַהֶבֶל שֶׁלְּךָ שֶׁה' נָתַן לְךָ בָּעוֹלָם הַזֶּה – כָּל יְמֵי הַהֶבֶל שֶׁלְּךָ, כִּי זֶה הַחֵלֶק שֶׁלְּךָ בַּחַיִּים וּבַעֲבוֹדָתְךָ שֶׁאַתָּה עָמֵל בָּעוֹלָם הַזֶּה.
י כֹּ֠ל אֲשֶׁ֨ר תִּמְצָ֧א יָֽדְךָ֛ לַעֲשׂ֥וֹת בְּכֹחֲךָ֖ עֲשֵׂ֑ה כִּי֩ אֵ֨ין מַעֲשֶׂ֤ה וְחֶשְׁבּוֹן֙ וְדַ֣עַת וְחׇכְמָ֔ה בִּשְׁא֕וֹל אֲשֶׁ֥ר אַתָּ֖ה הֹלֵ֥ךְ שָֽׁמָּה׃ עֲשֵׂה כָּל מַה שֶׁתַּצְלִיחַ לַעֲשׂוֹת בְּכוֹחֲךָ, כִּי בַּשְּׁאוֹל שֶׁאַתָּה הוֹלֵךְ שָׁמָּה אֵין מַעֲשִׂים וְחֶשְׁבּוֹנוֹת וְלֹא דַּעַת וְחָכְמָה.
יא שַׁ֜בְתִּי וְרָאֹ֣ה תַֽחַת־הַשֶּׁ֗מֶשׁ כִּ֣י לֹא֩ לַקַּלִּ֨ים הַמֵּר֜וֹץ וְלֹ֧א לַגִּבּוֹרִ֣ים הַמִּלְחָמָ֗ה וְ֠גַ֠ם לֹ֣א לַחֲכָמִ֥ים לֶ֙חֶם֙ וְגַ֨ם לֹ֤א לַנְּבֹנִים֙ עֹ֔שֶׁר וְגַ֛ם לֹ֥א לַיֹּדְעִ֖ים חֵ֑ן כִּי־עֵ֥ת וָפֶ֖גַע יִקְרֶ֥ה אֶת־כֻּלָּֽם׃ שַׁבְתִּי וְהִבְחַנְתִּי בָּעוֹלָם הַזֶּה שֶׁלֹּא בְּהֶכְרֵחַ הַמְּהִירִים מְנַצְּחִים בַּמֵּרוֹץ וְלֹא הַגִּבּוֹרִים בַּמִּלְחָמָה וְלֹא לַחֲכָמִים דַּוְקָא יֵשׁ לֶחֶם וְלֹא לַנְּבוֹנִים יֵשׁ עוֹשֶׁר וְגַם לֹא לַיַּדְעָנִים חֵן, אֶלָּא פִּגְעֵי הַזְּמָן קוֹרִים לְכֻלָּם.
יב כִּ֡י גַּם֩ לֹֽא־יֵדַ֨ע הָאָדָ֜ם אֶת־עִתּ֗וֹ כַּדָּגִים֙ שֶׁנֶּֽאֱחָזִים֙ בִּמְצוֹדָ֣ה רָעָ֔ה וְכַ֨צִּפֳּרִ֔ים הָאֲחֻז֖וֹת בַּפָּ֑ח כָּהֵ֗ם יֽוּקָשִׁים֙ בְּנֵ֣י הָֽאָדָ֔ם לְעֵ֣ת רָעָ֔ה כְּשֶׁתִּפּ֥וֹל עֲלֵיהֶ֖ם פִּתְאֹֽם׃ כִּי הָאָדָם גַּם לֹא יֵדַע מָתַי יַגִּיעַ זְמַנּוֹ, כְּמוֹ הַדָּגִים שֶׁנִּתְפָּסִים בְּרֶשֶׁת זְדוֹנִית וּכְמוֹ הַצִּפֳּרִים הַתְּפוּסוֹת בְּמַלְכּוֹדֶת, כְּמוֹתָם נִלְכָּדִים בְּנֵי אָדָם בְּאֵרוּעַ רָע כְּשֶׁהוּא נוֹפֵל עֲלֵיהֶם פִּתְאוֹם.
יג גַּם־זֹ֛ה רָאִ֥יתִי חׇכְמָ֖ה תַּ֣חַת הַשָּׁ֑מֶשׁ וּגְדוֹלָ֥ה הִ֖יא אֵלָֽי׃ גַּם חָכְמָה זוֹ רָאִיתִי תַּחַת הַשֶּׁמֶשׁ וַחֲשׁוּבָה הִיא בְּעֵינַי;
יד עִ֣יר קְטַנָּ֔ה וַאֲנָשִׁ֥ים בָּ֖הּ מְעָ֑ט וּבָֽא־אֵלֶ֜יהָ מֶ֤לֶךְ גָּדוֹל֙ וְסָבַ֣ב אֹתָ֔הּ וּבָנָ֥ה עָלֶ֖יהָ מְצוֹדִ֥ים גְּדֹלִֽים׃ עִיר קְטַנָּה וּבָהּ מְעַט אֲנָשִׁים, וּבָא אֵלֶיהָ מֶלֶךְ גָּדוֹל וְהִקִּיף אוֹתָהּ וּבָנָה סְבִיבָהּ מַלְכּוֹדוֹת רַבּוֹת.
טו וּמָ֣צָא בָ֗הּ אִ֤ישׁ מִסְכֵּן֙ חָכָ֔ם וּמִלַּט־ה֥וּא אֶת־הָעִ֖יר בְּחׇכְמָת֑וֹ וְאָדָם֙ לֹ֣א זָכַ֔ר אֶת־הָאִ֥ישׁ הַמִּסְכֵּ֖ן הַהֽוּא׃ וְנִמְצָא בָּעִיר אִישׁ חֲסַר כֹּל וְחָכָם, וְהוּא הִצִּיל אֶת הָעִיר בְּחָכְמָתוֹ, אַךְ אַף אָדָם לֹא זָכַר אֶת הָאִישׁ חֲסַר הַכֹּל הַהוּא.
טז וְאָמַ֣רְתִּי אָ֔נִי טוֹבָ֥ה חׇכְמָ֖ה מִגְּבוּרָ֑ה וְחׇכְמַ֤ת הַמִּסְכֵּן֙ בְּזוּיָ֔ה וּדְבָרָ֖יו אֵינָ֥ם נִשְׁמָעִֽים׃ וְאָמַרְתִּי אֲנִי: עֲדִיפָה חָכְמָה מִגְּבוּרָה, וְחָכְמַת חֲסַר הַכֹּל בְּזוּיָה וּדְבָרָיו אֵינָם נִשְׁמָעִים.
יז דִּבְרֵ֣י חֲכָמִ֔ים בְּנַ֖חַת נִשְׁמָעִ֑ים מִזַּעֲקַ֥ת מוֹשֵׁ֖ל בַּכְּסִילִֽים׃ דִּבְרֵי חֲכָמִים שֶׁנֶּאֱמָרִים בְּנַחַת מִתְקַבְּלִים יוֹתֵר מִצַּעֲקַת הַשַּׁלִּיט עַל הַטִּפְּשִׁים.
יח טוֹבָ֥ה חׇכְמָ֖ה מִכְּלֵ֣י קְרָ֑ב וְחוֹטֶ֣א אֶחָ֔ד יְאַבֵּ֖ד טוֹבָ֥ה הַרְבֵּֽה׃ עֲדִיפָה חָכְמָה מִכְּלֵי נֶשֶׁק, וְטָעוּת אַחַת מְאַבֶּדֶת הַרְבֵּה טוֹב.
י א זְב֣וּבֵי מָ֔וֶת יַבְאִ֥ישׁ יַבִּ֖יעַ שֶׁ֣מֶן רוֹקֵ֑חַ יָקָ֛ר מֵחׇכְמָ֥ה מִכָּב֖וֹד סִכְל֥וּת מְעָֽט׃ כְּמוֹ זְבוּבִים מֵתִים שֶׁמַּסְרִיחִים וּמַעֲלִים בּוּעוֹת בְּשֶׁמֶן מְבֻשָּׂם כָּךְ מְעַט טִפְּשׁוּת מַפִּילָה חָכְמָה וְכָבוֹד.
ב לֵ֤ב חָכָם֙ לִֽימִינ֔וֹ וְלֵ֥ב כְּסִ֖יל לִשְׂמֹאלֽוֹ׃ לֵב חָכָם מְבִיאוֹ לְדֶרֶךְ נְכוֹנָה וְלֵב טִפֵּשׁ מוֹשְׁכוֹ לְדֶרֶךְ קְלוֹקֶלֶת.
ג וְגַם־בַּדֶּ֛רֶךְ כשהסכל כְּשֶׁסָּכָ֥ל הֹלֵ֖ךְ לִבּ֣וֹ חָסֵ֑ר וְאָמַ֥ר לַכֹּ֖ל סָכָ֥ל הֽוּא׃ כָּל דָּבָר טִפֵּשׁ עוֹשֶׂה לֹא נָכוֹן, וּמַרְאֶה לְכֻלָּם שֶׁהוּא טִפֵּשׁ.
ד אִם־ר֤וּחַ הַמּוֹשֵׁל֙ תַּעֲלֶ֣ה עָלֶ֔יךָ מְקוֹמְךָ֖ אַל־תַּנַּ֑ח כִּ֣י מַרְפֵּ֔א יַנִּ֖יחַ חֲטָאִ֥ים גְּדוֹלִֽים׃ אִם הַמּוֹשֵׁל יִכְעַס עָלֶיךָ אַל תְּנַסֶּה לִבְרוֹחַ, כִּי אֶפְשָׁר לְתַקֵּן חֶסְרוֹנוֹת גְּדוֹלִים.
ה יֵ֣שׁ רָעָ֔ה רָאִ֖יתִי תַּ֣חַת הַשָּׁ֑מֶשׁ כִּשְׁגָגָ֕ה שֶׁיֹּצָ֖א מִלִּפְנֵ֥י הַשַּׁלִּֽיט׃ יֵשׁ רָעָה שֶׁרָאִיתִי בָּעוֹלָם הַזֶּה שֶׁנִּרְאֵית כְּמוֹ טָעוּת שֶׁיּוֹצֵאת מִלִּפְנֵי הַשַּׁלִּיט.
ו נִתַּ֣ן הַסֶּ֔כֶל בַּמְּרוֹמִ֖ים רַבִּ֑ים וַעֲשִׁירִ֖ים בַּשֵּׁ֥פֶל יֵשֵֽׁבוּ׃ טִפְּשִׁים מִתְעַלִּים לִמְרוֹמֵי מְרוֹמִים, וַעֲשִׁירִים מֻשְׁפָּלִים לָשֶׁבֶת עַל הָאָרֶץ.
ז רָאִ֥יתִי עֲבָדִ֖ים עַל־סוּסִ֑ים וְשָׂרִ֛ים הֹלְכִ֥ים כַּעֲבָדִ֖ים עַל־הָאָֽרֶץ׃ רָאִיתִי עֲבָדִים רוֹכְבִים עַל סוּסִים, וְשָׂרִים הוֹלְכִים כְּמוֹ עֲבָדִים עַל הָאָרֶץ.
ח חֹפֵ֥ר גּוּמָּ֖ץ בּ֣וֹ יִפּ֑וֹל וּפֹרֵ֥ץ גָּדֵ֖ר יִשְּׁכֶ֥נּוּ נָחָֽשׁ׃ הַחוֹפֵר בּוֹר נוֹפֵל לְתוֹכוֹ, וּפוֹרֵץ גָּדֵר נוֹשֵׁךְ אוֹתוֹ נָחָשׁ.
ט מַסִּ֣יעַ אֲבָנִ֔ים יֵעָצֵ֖ב בָּהֶ֑ם בּוֹקֵ֥עַ עֵצִ֖ים יִסָּ֥כֶן בָּֽם׃ הַמַּסִּיעַ אֲבָנִים מִצְטַעֵר בָּהֶם, וְהַבּוֹקֵעַ עֵצִים מִסְתַּכֵּן בָּהֶם.
י אִם־קֵהָ֣ה הַבַּרְזֶ֗ל וְהוּא֙ לֹא־פָנִ֣ים קִלְקַ֔ל וַחֲיָלִ֖ים יְגַבֵּ֑ר וְיִתְר֥וֹן הַכְשֵׁ֖יר חׇכְמָֽה׃ אִם בַּרְזֶל הַסַּכִּין קֵהֶה וּפָנָיו לֹא הֻחְלְקוּ צָרִיךְ לְהַגְבִּיר כֹּחַ, וְכָךְ הַכְשָׁרַת אָדָם בְּחָכְמָה.
יא אִם־יִשֹּׁ֥ךְ הַנָּחָ֖שׁ בְּלוֹא־לָ֑חַשׁ וְאֵ֣ין יִתְר֔וֹן לְבַ֖עַל הַלָּשֽׁוֹן׃ וְכִי נָחָשׁ נוֹשֵׁךְ בְּלִי לְחִישָׁה, כָּךְ אֵין תּוֹעֶלֶת לְמֵפִיץ לָשׁוֹן הָרַע.
יב דִּבְרֵ֥י פִי־חָכָ֖ם חֵ֑ן וְשִׂפְת֥וֹת כְּסִ֖יל תְּבַלְּעֶֽנּוּ׃ דִּבְרֵי פִּי חָכָם מְלֵאִים בְּחֵן, וְהַיּוֹצֵא מִשִּׂפְתֵי טִפֵּשׁ יַזִּיקוּ לוֹ.
יג תְּחִלַּ֥ת דִּבְרֵי־פִ֖יהוּ סִכְל֑וּת וְאַחֲרִ֣ית פִּ֔יהוּ הוֹלֵל֖וּת רָעָֽה׃ תְּחִלַּת דִּבְרֵי פִּיו בְּטִפְּשׁוּת, וְדִבּוּרָיו מִתְדַּרְדְּרִים לְהוֹלֵלוּת שֶׁל רוֹעַ.
יד וְהַסָּכָ֖ל יַרְבֶּ֣ה דְבָרִ֑ים לֹא־יֵדַ֤ע הָאָדָם֙ מַה־שֶּׁיִּֽהְיֶ֔ה וַאֲשֶׁ֤ר יִֽהְיֶה֙ מֵֽאַחֲרָ֔יו מִ֖י יַגִּ֥יד לֽוֹ׃ וְהַטִּפֵּשׁ מַרְבֶּה בְּדִבּוּרֵי סְרָק, אֲבָל הָאָדָם אֵינוֹ יוֹדֵעַ מַה שֶּׁיִּהְיֶה, מִי יְסַפֵּר לוֹ מַה יִּקְרֶה אַחֲרָיו.
טו עֲמַ֥ל הַכְּסִילִ֖ים תְּיַגְּעֶ֑נּוּ אֲשֶׁ֥ר לֹֽא־יָדַ֖ע לָלֶ֥כֶת אֶל־עִֽיר׃ הֶעָמָל הַמְּיֻתָּר שֶׁל הַטִּפֵּשׁ מְיַגַּעַת אוֹתוֹ עַד שֶׁאֵינוֹ יוֹדֵעַ לְהַגִּיעַ לְעִיר מִבְטָחִים.
טז אִֽי־לָ֣ךְ אֶ֔רֶץ שֶׁמַּלְכֵּ֖ךְ נָ֑עַר וְשָׂרַ֖יִךְ בַּבֹּ֥קֶר יֹאכֵֽלוּ׃ אוֹי לָךְ אֶרֶץ שֶׁמַּלְכֵּךְ מְשָׁרֵת אֶת יִצְרוֹ, וְשָׂרַיִךְ סוֹעֲדִים בַּבּוֹקֶר בִּמְקוֹם לִדְאוֹג לַמַּמְלָכָה.
יז אַשְׁרֵ֣יךְ אֶ֔רֶץ שֶׁמַּלְכֵּ֖ךְ בֶּן־חוֹרִ֑ים וְשָׂרַ֙יִךְ֙ בָּעֵ֣ת יֹאכֵ֔לוּ בִּגְבוּרָ֖ה וְלֹ֥א בַשְּׁתִֽי׃ אַשְׁרֵיךְ אֶרֶץ שֶׁמַּלְכֵּךְ בֶּן חוֹרִין מִיִּצְרוֹ, וְשָׂרַיִךְ סוֹעֲדִים בַּזְּמַן הַמַּתְאִים, גִּבּוֹרִים וְלֹא שַׁתְיָנִים.
יח בַּעֲצַלְתַּ֖יִם יִמַּ֣ךְ הַמְּקָרֶ֑ה וּבְשִׁפְל֥וּת יָדַ֖יִם יִדְלֹ֥ף הַבָּֽיִת׃ בְּעַצְלוּת – הַתִּקְרָה הוֹלֶכֶת וּמַנְמִיכָה, וּבְעַצְלוּת הַיָּדַיִם מִלְּתַקֵּן – הַבַּיִת דּוֹלֵף.
יט לִשְׂחוֹק֙ עֹשִׂ֣ים לֶ֔חֶם וְיַ֖יִן יְשַׂמַּ֣ח חַיִּ֑ים וְהַכֶּ֖סֶף יַעֲנֶ֥ה אֶת־הַכֹּֽל׃ כְּדֵי לְאַפְשֵׁר צְחוֹק עוֹשִׂים לֶחֶם, וְיַיִן מְשַׂמֵּחַ אֶת הַחַיִּים, וְהַכֶּסֶף מְאַפְשֵׁר לְהַשִּׂיג אֶת הַכֹּל.
כ גַּ֣ם בְּמַדָּֽעֲךָ֗ מֶ֚לֶךְ אַל־תְּקַלֵּ֔ל וּבְחַדְרֵי֙ מִשְׁכָּ֣בְךָ֔ אַל־תְּקַלֵּ֖ל עָשִׁ֑יר כִּ֣י ע֤וֹף הַשָּׁמַ֙יִם֙ יוֹלִ֣יךְ אֶת־הַקּ֔וֹל וּבַ֥עַל הכנפים כְּנָפַ֖יִם יַגֵּ֥יד דָּבָֽר׃ אֲפִלּוּ בְּמַחְשַׁבְתְּךָ אַל תְּבַזֶּה מֶלֶךְ, וּבַחֲדַר הַשֵּׁנָה הַפְּנִימִי אַל תְּבַזֶּה עָשִׁיר, כִּי עוֹף הַשָּׁמַיִם מוֹלִיךְ אֶת הַקּוֹל וּבַעַל כְּנָפַיִם מְסַפֵּר כָּל דָּבָר.
יא א שַׁלַּ֥ח לַחְמְךָ֖ עַל־פְּנֵ֣י הַמָּ֑יִם כִּֽי־בְרֹ֥ב הַיָּמִ֖ים תִּמְצָאֶֽנּוּ׃ תְּרוֹם, שְׁלַח אֶת לַחְמְךָ עַל פְּנֵי הַמַּיִם כִּי אַחַר יָמִים רַבִּים תִּמְצָא אוֹתוֹ.
ב תֶּן־חֵ֥לֶק לְשִׁבְעָ֖ה וְגַ֣ם לִשְׁמוֹנָ֑ה כִּ֚י לֹ֣א תֵדַ֔ע מַה־יִּהְיֶ֥ה רָעָ֖ה עַל־הָאָֽרֶץ׃ תֶּן חֵלֶק לְשִׁבְעָה נִצְרָכִים וַאֲפִלּוּ לִשְׁמוֹנָה, כִּי אֵינְךָ יוֹדֵעַ מָה הָרָעָה שֶׁתָּבוֹא עַל הָאָרֶץ.
ג אִם־יִמָּלְא֨וּ הֶעָבִ֥ים גֶּ֙שֶׁם֙ עַל־הָאָ֣רֶץ יָרִ֔יקוּ וְאִם־יִפּ֥וֹל עֵ֛ץ בַּדָּר֖וֹם וְאִ֣ם בַּצָּפ֑וֹן מְק֛וֹם שֶׁיִּפּ֥וֹל הָעֵ֖ץ שָׁ֥ם יְהֽוּא׃ אוּלַי יִתְמַלְּאוּ הֶעָבִים בְּגֶשֶׁם וְהֵם יָרִיקוּ אוֹתוֹ בְּחָזְקָה עַל הָאָרֶץ, וְאוּלַי יִפּוֹל עֵץ בַּדָּרוֹם אוֹ בַּצָּפוֹן, בַּמָּקוֹם שֶׁיִּפּוֹל הָעֵץ שָׁם יִהְיוּ הַנִּצְרָכִים.
ד שֹׁמֵ֥ר ר֖וּחַ לֹ֣א יִזְרָ֑ע וְרֹאֶ֥ה בֶעָבִ֖ים לֹ֥א יִקְצֽוֹר׃ הַמַּמְתִּין לְרוּחַ מַתְאִימָה לְעוֹלָם לֹא יִזְרַע, וְהַמַּבִּיט בָּעֲנָנִים לְעוֹלָם לֹא יִקְצוֹר.
ה כַּאֲשֶׁ֨ר אֵֽינְךָ֤ יוֹדֵ֙עַ֙ מַה־דֶּ֣רֶךְ הָר֔וּחַ כַּעֲצָמִ֖ים בְּבֶ֣טֶן הַמְּלֵאָ֑ה כָּ֗כָה לֹ֤א תֵדַע֙ אֶת־מַעֲשֵׂ֣ה הָֽאֱלֹהִ֔ים אֲשֶׁ֥ר יַעֲשֶׂ֖ה אֶת־הַכֹּֽל׃ כְּפִי שֶׁאֵינְךָ יוֹדֵעַ מַה תִּהְיֶה דֶּרֶךְ הָרוּחַ, כְּמוֹ הַדְּבָרִים הַסְּגוּרִים בְּתוֹךְ בֶּטֶן מְלֵאָה, כָּכָה לֹא תֵּדַע אֶת מַעֲשֵׂה הָאֱלֹהִים שֶׁעוֹשֶׂה אֶת הַכֹּל.
ו בַּבֹּ֙קֶר֙ זְרַ֣ע אֶת־זַרְעֶ֔ךָ וְלָעֶ֖רֶב אַל־תַּנַּ֣ח יָדֶ֑ךָ כִּי֩ אֵֽינְךָ֨ יוֹדֵ֜עַ אֵ֣י זֶ֤ה יִכְשָׁר֙ הֲזֶ֣ה אוֹ־זֶ֔ה וְאִם־שְׁנֵיהֶ֥ם כְּאֶחָ֖ד טוֹבִֽים׃ לָכֵן, זְרַע אֶת זַרְעֲךָ מִיַּד בַּבּוֹקֶר, וְאַל תָּנִיח אֶת יָדְךָ גַּם בָּעֶרֶב, כִּי אֵינְךָ יוֹדֵעַ אֵיזֶה יַצְלִיחַ, אִם זֶה אוֹ זֶה אוֹ שֶׁשְּׁנֵיהֶם יַחַד טוֹבִים.
ז וּמָת֖וֹק הָא֑וֹר וְט֥וֹב לַֽעֵינַ֖יִם לִרְא֥וֹת אֶת־הַשָּֽׁמֶשׁ׃ הָאוֹר מָתוֹק, וְטוֹב לַעֵינַיִם לִחְיוֹת וְלִרְאוֹת אֶת הַשֶּׁמֶשׁ.
ח כִּ֣י אִם־שָׁנִ֥ים הַרְבֵּ֛ה יִחְיֶ֥ה הָאָדָ֖ם בְּכֻלָּ֣ם יִשְׂמָ֑ח וְיִזְכֹּר֙ אֶת־יְמֵ֣י הַחֹ֔שֶׁךְ כִּֽי־הַרְבֵּ֥ה יִהְי֖וּ כׇּל־שֶׁבָּ֥א הָֽבֶל׃ כִּי אִם הָאָדָם יִחְיֶה שָׁנִים רַבּוֹת וְיִשְׂמַח בְּכֻלָּם, אֲבָל יִזְכּוֹר אֶת יְמֵי הַחוֹשֶׁךְ בַּקֶּבֶר שֶׁיִּהְיוּ הַרְבֵּה יוֹתֵר – כָּל מַה שֶּׁבָּא לוֹ הוּא הֶבֶל.
ט שְׂמַ֧ח בָּח֣וּר בְּיַלְדוּתֶ֗יךָ וִֽיטִֽיבְךָ֤ לִבְּךָ֙ בִּימֵ֣י בְחוּרוֹתֶ֔יךָ וְהַלֵּךְ֙ בְּדַרְכֵ֣י לִבְּךָ֔ וּבְמַרְאֵ֖י עֵינֶ֑יךָ וְדָ֕ע כִּ֧י עַל־כׇּל־אֵ֛לֶּה יְבִֽיאֲךָ֥ הָאֱלֹהִ֖ים בַּמִּשְׁפָּֽט׃ שְׂמַח בָּחוּר בִּצְעִירוּתְךָ, וֶהֱיֵה בְּהַרְגָּשָׁה טוֹבָה כְּשֶׁאַתָּה בָּחוּר, וְלֵךְ בַּדֶּרֶךְ שֶׁלִּבְּךָ חָפֵץ וְאַחַר רְאִיַּת עֵינֶיךָ, אַךְ דַּע שֶׁעַל כָּל אֵלֶּה הָאֱלֹהִים יָבִיא אוֹתְךָ לְמִשְׁפָּט.
י וְהָסֵ֥ר כַּ֙עַס֙ מִלִּבֶּ֔ךָ וְהַעֲבֵ֥ר רָעָ֖ה מִבְּשָׂרֶ֑ךָ כִּֽי־הַיַּלְד֥וּת וְהַֽשַּׁחֲר֖וּת הָֽבֶל׃ וְלָכֵן הָסֵר עַצְבוּת מִלִּבְּךָ וְהַעֲבֵר רָעָה מִגּוּפְךָ, כִּי שְׁנוֹת הַיַּלְדוּת וְהַשֵּׂעַר הַשָּׁחוֹר הֵם הֶבֶל.
יב א וּזְכֹר֙ אֶת־בּ֣וֹרְאֶ֔יךָ בִּימֵ֖י בְּחוּרֹתֶ֑יךָ עַ֣ד אֲשֶׁ֤ר לֹא־יָבֹ֙אוּ֙ יְמֵ֣י הָֽרָעָ֔ה וְהִגִּ֣יעוּ שָׁנִ֔ים אֲשֶׁ֣ר תֹּאמַ֔ר אֵֽין־לִ֥י בָהֶ֖ם חֵֽפֶץ׃ וּזְכוֹר אֶת בּוֹרְאֲךָ בַּשָּׁנִים שֶׁאַתָּה בָּחוּר, לִפְנֵי שֶׁיָּבוֹאוּ הַיָּמִים הָרָעִים וְיַגִּיעוּ שָׁנִים שֶׁתֹּאמַר עֲלֵיהֶן: אֵין לִי בָּהֶם חֵפֶץ.
ב עַ֠ד אֲשֶׁ֨ר לֹֽא־תֶחְשַׁ֤ךְ הַשֶּׁ֙מֶשׁ֙ וְהָא֔וֹר וְהַיָּרֵ֖חַ וְהַכּוֹכָבִ֑ים וְשָׁ֥בוּ הֶעָבִ֖ים אַחַ֥ר הַגָּֽשֶׁם׃ לִפְנֵי שֶׁיֵּחָשְׁכוּ בְּעֵינֶיךָ הַשֶּׁמֶשׁ וְהָאוֹר וְהַיָּרֵחַ וְהַכּוֹכָבִים, וְיַחְזְרוּ הָעֲנָנִים שֶׁאַחֲרֵי הַגֶּשֶׁם.
ג בַּיּ֗וֹם שֶׁיָּזֻ֙עוּ֙ שֹׁמְרֵ֣י הַבַּ֔יִת וְהִֽתְעַוְּת֖וּ אַנְשֵׁ֣י הֶחָ֑יִל וּבָטְל֤וּ הַטֹּֽחֲנוֹת֙ כִּ֣י מִעֵ֔טוּ וְחָשְׁכ֥וּ הָרֹא֖וֹת בָּאֲרֻבּֽוֹת׃ בַּיּוֹם שֶׁיִּרְעֲדוּ הָרַגְלַיִם וְיִתְעַוְּתוּ הַזְּרוֹעוֹת, וְיִתְבַּטֵּל הַשִּׁמּוּשׁ בַּשִּׁנַּיִם שֶׁהִתְמַעֲטוּ, וְיֵחָשְׁכוּ הָעֵינַיִם בְּחוֹרֵיהֶן.
ד וְסֻגְּר֤וּ דְלָתַ֙יִם֙ בַּשּׁ֔וּק בִּשְׁפַ֖ל ק֣וֹל הַֽטַּחֲנָ֑ה וְיָקוּם֙ לְק֣וֹל הַצִּפּ֔וֹר וְיִשַּׁ֖חוּ כׇּל־בְּנ֥וֹת הַשִּֽׁיר׃ וְיִסָּתְמוּ הַנְּקָבִים הַחִצּוֹנִים בְּשׁוּלֵי מַעֲרֶכֶת הָעִכּוּל, וְיִתְעוֹרֵר לִפְנוֹת בּוֹקֶר לְקוֹל צִפּוֹר, וְתֵרֵד בְּאָזְנָיו אֵכוּת צְלִילֵי הַשִּׁירִים.
ה גַּ֣ם מִגָּבֹ֤הַּ יִרָ֙אוּ֙ וְחַתְחַתִּ֣ים בַּדֶּ֔רֶךְ וְיָנֵ֤אץ הַשָּׁקֵד֙ וְיִסְתַּבֵּ֣ל הֶֽחָגָ֔ב וְתָפֵ֖ר הָֽאֲבִיּוֹנָ֑ה כִּֽי־הֹלֵ֤ךְ הָאָדָם֙ אֶל־בֵּ֣ית עוֹלָמ֔וֹ וְסָבְב֥וּ בַשּׁ֖וּק הַסּוֹפְדִֽים׃ גַּם יִפְחֲדוּ מִמָּקוֹם גָּבוֹהַּ וּמִמִּכְשׁוֹלִים בַּדֶּרֶךְ, וְיִבְלְטוּ עֲצָמוֹת וְיִכְבַּד הַדּוֹפֶק, וְתִתְבַּטֵּל הַתַּאֲוָה, כִּי הָאָדָם הוֹלֵךְ לִקְרַאת בֵּית עוֹלָמוֹ וְהַסּוֹפְדִים כְּבָר מִסְתּוֹבְבִים בַּחוּצוֹת.
ו עַ֣ד אֲשֶׁ֤ר לֹֽא־יֵרָתֵק֙ ירחק חֶ֣בֶל הַכֶּ֔סֶף וְתָר֖וּץ גֻּלַּ֣ת הַזָּהָ֑ב וְתִשָּׁ֤בֶר כַּד֙ עַל־הַמַּבּ֔וּעַ וְנָרֹ֥ץ הַגַּלְגַּ֖ל אֶל־הַבּֽוֹר׃ לִפְנֵי שֶׁתִּתְנַתֵּק הַשִּׁדְרָה וְתִתְרַסֵּק הַגֻּלְגּוֹלֶת, הַכֶּרֶס תִּבָּקַע וְהַגּוּף יֻשְׁלַךְ אֶל הַבּוֹר.
ז וְיָשֹׁ֧ב הֶעָפָ֛ר עַל־הָאָ֖רֶץ כְּשֶׁהָיָ֑ה וְהָר֣וּחַ תָּשׁ֔וּב אֶל־הָאֱלֹהִ֖ים אֲשֶׁ֥ר נְתָנָֽהּ׃ וְהֶעָפָר יָשׁוּב לָאָרֶץ כְּמוֹ שֶׁהָיָה, וְרוּחַ הַחַיִּים תָּשׁוּב אֶל הָאֱלֹהִים שֶׁנָּתַן אוֹתָהּ.
ח הֲבֵ֧ל הֲבָלִ֛ים אָמַ֥ר הַקּוֹהֶ֖לֶת הַכֹּ֥ל הָֽבֶל׃ הֲבֵל הֲבָלִים הִסִּיק קוֹהֶלֶת – הַכֹּל הֶבֶל.
ט וְיֹתֵ֕ר שֶׁהָיָ֥ה קֹהֶ֖לֶת חָכָ֑ם ע֗וֹד לִמַּד־דַּ֙עַת֙ אֶת־הָעָ֔ם וְאִזֵּ֣ן וְחִקֵּ֔ר תִּקֵּ֖ן מְשָׁלִ֥ים הַרְבֵּֽה׃ וְנוֹסַף עַל שֶׁהָיָה קוֹהֶלֶת חָכָם בְּעַצְמוֹ הוּא גַּם לִמֵּד אֶת הָעָם יְדִיעוֹת, וְהֵבִין וְחָקַר, וְחִבֵּר הַרְבֵּה מְשָׁלִים.
י בִּקֵּ֣שׁ קֹהֶ֔לֶת לִמְצֹ֖א דִּבְרֵי־חֵ֑פֶץ וְכָת֥וּב יֹ֖שֶׁר דִּבְרֵ֥י אֱמֶֽת׃ קוֹהֶלֶת חִפֵּשׂ לִמְצוֹא דְּבָרִים מְעַנְיְנִים, וְלִכְתּוֹב נָכוֹן דְּבָרִים אֲמִתִּיִּים.
יא דִּבְרֵ֤י חֲכָמִים֙ כַּדָּ֣רְבֹנ֔וֹת וּֽכְמַשְׂמְר֥וֹת נְטוּעִ֖ים בַּעֲלֵ֣י אֲסֻפּ֑וֹת נִתְּנ֖וּ מֵרֹעֶ֥ה אֶחָֽד׃ דִּבְרֵי חֲכָמִים קְבוּעִים כְּקוֹצֵי דָּרְבָּן, וְשֶׁל אוֹסְפֵי חָכְמָה – כְּמַסְמְרִים, נִתְּנוּ מֵהַמַּנְהִיג הַיָּחִיד.
יב וְיֹתֵ֥ר מֵהֵ֖מָּה בְּנִ֣י הִזָּהֵ֑ר עֲשׂ֨וֹת סְפָרִ֤ים הַרְבֵּה֙ אֵ֣ין קֵ֔ץ וְלַ֥הַג הַרְבֵּ֖ה יְגִעַ֥ת בָּשָֽׂר׃ וְהִזָּהֵר בְּנִי גַּם בְּעוֹד דְּבָרִים, אֵין סוֹף לְפָרְטָם בִּסְפָרִים רַבִּים, וִיגִיעָה רַבָּה לְלָמְדָּם.
יג ס֥וֹף דָּבָ֖ר הַכֹּ֣ל נִשְׁמָ֑ע אֶת־הָאֱלֹהִ֤ים יְרָא֙ וְאֶת־מִצְוֺתָ֣יו שְׁמ֔וֹר כִּי־זֶ֖ה כׇּל־הָאָדָֽם׃ וְזוֹ כָּל מַשְׁמָעוּת מַסְקָנַת הַדְּבָרִים: כַּבֵּד אֶת דְּבַר הָאֱלֹהִים וּשְׁמוֹר אֶת מִצְוֹתָיו, כִּי זֶה כָּל מַטְּרַת הָאָדָם.
יד כִּ֚י אֶת־כׇּל־מַֽעֲשֶׂ֔ה הָאֱלֹהִ֛ים יָבִ֥א בְמִשְׁפָּ֖ט עַ֣ל כׇּל־נֶעְלָ֑ם אִם־ט֖וֹב וְאִם־רָֽע׃ כִּי הָאֱלֹהִים יָבִיא לְמִשְׁפָּט אֶת כָּל הַמַּעֲשִׂים, כּוֹלֵל הַבִּלְתִּי יְדוּעִים, הַטּוֹבִים וְהָרָעִים.
